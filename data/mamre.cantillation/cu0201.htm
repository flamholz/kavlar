שמות

א,א וְאֵ֗לֶּה שְׁמוֹת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל הַבָּאִ֖ים מִצְרָ֑יְמָה אֵ֣ת יַֽעֲקֹ֔ב אִ֥ישׁ וּבֵית֖וֹ בָּֽאוּ׃ וְאִלֵּין, שְׁמָהָת בְּנֵי יִשְׂרָאֵל, דְּעָאלוּ, לְמִצְרָיִם:  עִם יַעֲקוֹב, גְּבַר וַאֲנָשׁ בֵּיתֵיהּ עָאלוּ. 
א,ב רְאוּבֵ֣ן שִׁמְע֔וֹן לֵוִ֖י וִֽיהוּדָֽה׃ רְאוּבֵן שִׁמְעוֹן, לֵוִי וִיהוּדָה. 
א,ג יִשָּׂשכָ֥ר זְבוּלֻ֖ן וּבִנְיָמִֽן׃ יִשָּׂשכָר זְבוּלוּן, וּבִנְיָמִין. 
א,ד דָּ֥ן וְנַפְתָּלִ֖י גָּ֥ד וְאָשֵֽׁר׃ דָּן וְנַפְתָּלִי, גָּד וְאָשֵׁר. 
א,ה וַֽיְהִ֗י כָּל־נֶ֛פֶשׁ יֹֽצְאֵ֥י יֶֽרֶךְ־יַעֲקֹ֖ב שִׁבְעִ֣ים נָ֑פֶשׁ וְיוֹסֵ֖ף הָיָ֥ה בְמִצְרָֽיִם׃ וַהֲוַאָה, כָּל נַפְשָׁתָא נָפְקֵי יִרְכָּא דְּיַעֲקוֹב--שִׁבְעִין נַפְשָׁן; וְיוֹסֵף, דַּהֲוָה בְּמִצְרָיִם. 
א,ו וַיָּ֤מָת יוֹסֵף֙ וְכָל־אֶחָ֔יו וְכֹ֖ל הַדּ֥וֹר הַהֽוּא׃ וּמִית יוֹסֵף וְכָל אֲחוֹהִי, וְכֹל דָּרָא הַהוּא. 
א,ז וּבְנֵ֣י יִשְׂרָאֵ֗ל פָּר֧וּ וַֽיִּשְׁרְצ֛וּ וַיִּרְבּ֥וּ וַיַּֽעַצְמ֖וּ בִּמְאֹ֣ד מְאֹ֑ד וַתִּמָּלֵ֥א הָאָ֖רֶץ אֹתָֽם׃ {פ} וּבְנֵי יִשְׂרָאֵל, נְפִישׁוּ וְאִתְיַלַּדוּ וּסְגִיאוּ וּתְקִיפוּ--לַחְדָּא לַחְדָּא; וְאִתְמְלִיאַת אַרְעָא, מִנְּהוֹן. {פ}
א,ח וַיָּ֥קָם מֶֽלֶךְ־חָדָ֖שׁ עַל־מִצְרָ֑יִם אֲשֶׁ֥ר לֹֽא־יָדַ֖ע אֶת־יוֹסֵֽף׃ וְקָם מַלְכָּא חֲדַתָּא, עַל מִצְרָיִם, דְּלָא מְקַיֵּים, גְּזֵירַת יוֹסֵף. 
א,ט וַיֹּ֖אמֶר אֶל־עַמּ֑וֹ הִנֵּ֗ה עַ֚ם בְּנֵ֣י יִשְׂרָאֵ֔ל רַ֥ב וְעָצ֖וּם מִמֶּֽנּוּ׃ וַאֲמַר, לְעַמֵּיהּ:  הָא, עַמָּא בְּנֵי יִשְׂרָאֵל--סָגַן וְתָקְפִין, מִנַּנָא. 
א,י הָ֥בָה נִֽתְחַכְּמָ֖ה ל֑וֹ פֶּן־יִרְבֶּ֗ה וְהָיָ֞ה כִּֽי־תִקְרֶ֤אנָה מִלְחָמָה֙ וְנוֹסַ֤ף גַּם־הוּא֙ עַל־שֹׂ֣נְאֵ֔ינוּ וְנִלְחַם־בָּ֖נוּ וְעָלָ֥ה מִן־הָאָֽרֶץ׃ הַבוּ נִתְחַכַּם, לְהוֹן:  דִּלְמָא יִסְגּוֹן, וִיהֵי אֲרֵי יְעָרְעִנַּנָא קְרָב וְיִתּוֹסְפוּן אַף אִנּוּן עַל סָנְאַנָא, וִיגִיחוּן בַּנָא קְרָב, וְיִסְּקוּן מִן אַרְעָא. 
א,יא וַיָּשִׂ֤ימוּ עָלָיו֙ שָׂרֵ֣י מִסִּ֔ים לְמַ֥עַן עַנֹּת֖וֹ בְּסִבְלֹתָ֑ם וַיִּ֜בֶן עָרֵ֤י מִסְכְּנוֹת֙ לְפַרְעֹ֔ה אֶת־פִּתֹ֖ם וְאֶת־רַֽעַמְסֵֽס׃ וּמַנִּיאוּ עֲלֵיהוֹן שִׁלְטוֹנִין מַבְאֲשִׁין, בְּדִיל לְעַנּוֹאֵיהוֹן בְּפֻלְחָנְהוֹן; וּבְנוֹ קִרְוֵי בֵּית אוֹצְרֵי, לְפַרְעֹה--יָת פִּיתוֹם, וְיָת רַעַמְסֵס. 
א,יב וְכַֽאֲשֶׁר֙ יְעַנּ֣וּ אֹת֔וֹ כֵּ֥ן יִרְבֶּ֖ה וְכֵ֣ן יִפְרֹ֑ץ וַיָּקֻ֕צוּ מִפְּנֵ֖י בְּנֵ֥י יִשְׂרָאֵֽל׃ וּכְמָא דִּמְעַנַּן לְהוֹן, כֵּין סָגַן וְכֵין תָּקְפִין; וְעַקַת לְמִצְרָאֵי, מִן קֳדָם בְּנֵי יִשְׂרָאֵל. 
א,יג וַיַּֽעֲבִ֧דוּ מִצְרַ֛יִם אֶת־בְּנֵ֥י יִשְׂרָאֵ֖ל בְּפָֽרֶךְ׃ וְאַפְלַחוּ מִצְרָאֵי יָת בְּנֵי יִשְׂרָאֵל, בְּקַשְׁיוּ. 
א,יד וַיְמָֽרְר֨וּ אֶת־חַיֵּיהֶ֜ם בַּֽעֲבֹדָ֣ה קָשָׁ֗ה בְּחֹ֨מֶר֙ וּבִלְבֵנִ֔ים וּבְכָל־עֲבֹדָ֖ה בַּשָּׂדֶ֑ה אֵ֚ת כָּל־עֲבֹ֣דָתָ֔ם אֲשֶׁר־עָֽבְד֥וּ בָהֶ֖ם בְּפָֽרֶךְ׃ וְאַמַּרוּ יָת חַיֵּיהוֹן בְּפֻלְחָנָא קַשְׁיָא, בְּטִינָא וּבְלִבְנֵי, וּבְכָל פֻּלְחָנָא, בְּחַקְלָא--יָת, כָּל פֻּלְחָנְהוֹן, דְּאַפְלַחוּ בְּהוֹן, בְּקַשְׁיוּ. 
א,טו וַיֹּ֨אמֶר֙ מֶ֣לֶךְ מִצְרַ֔יִם לַֽמְיַלְּדֹ֖ת הָֽעִבְרִיֹּ֑ת אֲשֶׁ֨ר שֵׁ֤ם הָֽאַחַת֙ שִׁפְרָ֔ה וְשֵׁ֥ם הַשֵּׁנִ֖ית פּוּעָֽה׃ וַאֲמַר מַלְכָּא דְּמִצְרַיִם, לְחָיָתָא יְהוּדַיָתָא, דְּשׁוֹם חֲדָא שִׁפְרָה, וְשׁוֹם תִּנְיֵיתָא פּוּעָה. 
א,טז וַיֹּ֗אמֶר בְּיַלֶּדְכֶן֙ אֶת־הָֽעִבְרִיּ֔וֹת וּרְאִיתֶ֖ן עַל־הָֽאָבְנָ֑יִם אִם־בֵּ֥ן הוּא֙ וַֽהֲמִתֶּ֣ן אֹת֔וֹ וְאִם־בַּ֥ת הִ֖וא וָחָֽיָה׃ וַאֲמַר, כַּד תִּהְוְיָן מְיַלְּדָן יָת יְהוּדַיָתָא, וְתִחְזְיָן, עַל מַתְבְּרָא:  אִם בְּרָא הוּא וְתִקְטְלָן יָתֵיהּ, וְאִם בְּרַתָּא הִיא תְּקַיְּמוּנַּהּ. 
א,יז וַתִּירֶ֤אןָ הַֽמְיַלְּדֹת֙ אֶת־הָ֣אֱלֹהִ֔ים וְלֹ֣א עָשׂ֔וּ כַּֽאֲשֶׁ֛ר דִּבֶּ֥ר אֲלֵיהֶ֖ן מֶ֣לֶךְ מִצְרָ֑יִם וַתְּחַיֶּ֖יןָ אֶת־הַיְלָדִֽים׃ וּדְחִילָא חָיָתָא, מִן קֳדָם יְיָ, וְלָא עֲבַדָא, כְּמָא דְּמַלֵּיל עִמְּהוֹן מַלְכָּא דְּמִצְרָיִם; וְקַיִּימָא, יָת בְּנַיָּא. 
א,יח וַיִּקְרָ֤א מֶֽלֶךְ־מִצְרַ֨יִם֙ לַֽמְיַלְּדֹ֔ת וַיֹּ֣אמֶר לָהֶ֔ן מַדּ֥וּעַ עֲשִׂיתֶ֖ן הַדָּבָ֣ר הַזֶּ֑ה וַתְּחַיֶּ֖יןָ אֶת־הַיְלָדִֽים׃ וּקְרָא מַלְכָּא דְּמִצְרַיִם, לְחָיָתָא, וַאֲמַר לְהוֹן, מָדֵין עֲבַדְתִּין פִּתְגָמָא הָדֵין; וְקַיֵּימְתִּין, יָת בְּנַיָּא. 
א,יט וַתֹּאמַ֤רְןָ הַֽמְיַלְּדֹת֙ אֶל־פַּרְעֹ֔ה כִּ֣י לֹ֧א כַנָּשִׁ֛ים הַמִּצְרִיֹּ֖ת הָֽעִבְרִיֹּ֑ת כִּֽי־חָי֣וֹת הֵ֔נָּה בְּטֶ֨רֶם תָּב֧וֹא אֲלֵהֶ֛ן הַמְיַלֶּ֖דֶת וְיָלָֽדוּ׃ וַאֲמַרָא חָיָתָא לְפַרְעֹה, אֲרֵי לָא כִּנְשַׁיָּא מִצְרַיָתָא יְהוּדַיָתָא:  אֲרֵי חַכִּימָן אִנִּין, עַד לָא עַאלַת לְוָתְהוֹן חָיְתָא יָלְדָן. 
א,כ וַיֵּ֥יטֶב אֱלֹהִ֖ים לַֽמְיַלְּדֹ֑ת וַיִּ֧רֶב הָעָ֛ם וַיַּֽעַצְמ֖וּ מְאֹֽד׃ וְאוֹטֵיב יְיָ, לְחָיָתָא; וּסְגִי עַמָּא וּתְקִיפוּ, לַחְדָּא. 
א,כא וַיְהִ֕י כִּֽי־יָרְא֥וּ הַֽמְיַלְּדֹ֖ת אֶת־הָֽאֱלֹהִ֑ים וַיַּ֥עַשׂ לָהֶ֖ם בָּתִּֽים׃ וַהֲוָה, כַּד דְּחִילָא חָיָתָא מִן קֳדָם יְיָ; וַעֲבַד לְהוֹן, בָּתִּין. 
א,כב וַיְצַ֣ו פַּרְעֹ֔ה לְכָל־עַמּ֖וֹ לֵאמֹ֑ר כָּל־הַבֵּ֣ן הַיִּלּ֗וֹד הַיְאֹ֨רָה֙ תַּשְׁלִיכֻ֔הוּ וְכָל־הַבַּ֖ת תְּחַיּֽוּן׃ {פ} וּפַקֵּיד פַּרְעֹה, לְכָל עַמֵּיהּ לְמֵימַר:  כָּל בְּרָא דְּיִתְיְלֵיד לִיהוּדָאֵי, בְּנַהְרָא תִּרְמוֹנֵיהּ, וְכָל בְּרַתָּא, תְּקַיְּמוּן. {פ}
ב,א וַיֵּ֥לֶךְ אִ֖ישׁ מִבֵּ֣ית לֵוִ֑י וַיִּקַּ֖ח אֶת־בַּת־לֵוִֽי׃ וַאֲזַל גֻּבְרָא, מִדְּבֵית לֵוִי; וּנְסֵיב, יָת בַּת לֵוִי. 
ב,ב וַתַּ֥הַר הָֽאִשָּׁ֖ה וַתֵּ֣לֶד בֵּ֑ן וַתֵּ֤רֶא אֹתוֹ֙ כִּי־ט֣וֹב ה֔וּא וַֽתִּצְפְּנֵ֖הוּ שְׁלֹשָׁ֥ה יְרָחִֽים׃ וְעַדִּיאַת אִתְּתָא, וִילֵידַת בַּר; וַחֲזָת יָתֵיהּ אֲרֵי טָב הוּא, וְאַטְמַרְתֵּיהּ תְּלָתָא יַרְחִין. 
ב,ג וְלֹֽא־יָכְלָ֣ה עוֹד֮ הַצְּפִינוֹ֒ וַתִּֽקַּֽח־לוֹ֙ תֵּ֣בַת גֹּ֔מֶא וַתַּחְמְרָ֥ה בַֽחֵמָ֖ר וּבַזָּ֑פֶת וַתָּ֤שֶׂם בָּהּ֙ אֶת־הַיֶּ֔לֶד וַתָּ֥שֶׂם בַּסּ֖וּף עַל־שְׂפַ֥ת הַיְאֹֽר׃ וְלָא יְכֵילַת עוֹד, לְאַטְמָרוּתֵיהּ, וּנְסֵיבַת לֵיהּ תֵּיבְתָא דְּגֹמֶא, וַחֲפָתַהּ בְּחֵימְרָא וּבְזִפְתָּא; וְשַׁוִּיאַת בַּהּ יָת רָבְיָא, וְשַׁוִּיתַהּ בְּיַעְרָא עַל כֵּיף נַהְרָא. 
ב,ד וַתֵּֽתַצַּ֥ב אֲחֹת֖וֹ מֵֽרָחֹ֑ק לְדֵעָ֕ה מַה־יֵּֽעָשֶׂ֖ה לֽוֹ׃ וְאִתְעַתַּדַת אַחָתֵיהּ, מֵרַחִיק, לְמִדַּע, מָא יִתְעֲבֵיד לֵיהּ. 
ב,ה וַתֵּ֤רֶד בַּת־פַּרְעֹה֙ לִרְחֹ֣ץ עַל־הַיְאֹ֔ר וְנַֽעֲרֹתֶ֥יהָ הֹֽלְכֹ֖ת עַל־יַ֣ד הַיְאֹ֑ר וַתֵּ֤רֶא אֶת־הַתֵּבָה֙ בְּת֣וֹךְ הַסּ֔וּף וַתִּשְׁלַ֥ח אֶת־אֲמָתָ֖הּ וַתִּקָּחֶֽהָ׃ וּנְחַתַת בַּת פַּרְעֹה לְמִסְחֵי עַל נַהְרָא, וְעוּלֵימְתַהָא מְהַלְּכָן עַל כֵּיף נַהְרָא; וַחֲזָת יָת תֵּיבְתָא בְּגוֹ יַעְרָא, וְאוֹשֵׁיטַת יָת אַמְּתַהּ וּנְסֵיבְתַּהּ. 
ב,ו וַתִּפְתַּח֙ וַתִּרְאֵ֣הוּ אֶת־הַיֶּ֔לֶד וְהִנֵּה־נַ֖עַר בֹּכֶ֑ה וַתַּחְמֹ֣ל עָלָ֔יו וַתֹּ֕אמֶר מִיַּלְדֵ֥י הָֽעִבְרִ֖ים זֶֽה׃ וּפְתַחַת וַחֲזָת יָת רָבְיָא, וְהָא עוּלֵימָא בָּכֵי; וְחַסַת עֲלוֹהִי--וַאֲמַרַת, מִבְּנֵי יְהוּדָאֵי הוּא דֵּין. 
ב,ז וַתֹּ֣אמֶר אֲחֹתוֹ֮ אֶל־בַּת־פַּרְעֹה֒ הַֽאֵלֵ֗ךְ וְקָרָ֤אתִי לָךְ֙ אִשָּׁ֣ה מֵינֶ֔קֶת מִ֖ן הָֽעִבְרִיֹּ֑ת וְתֵינִ֥ק לָ֖ךְ אֶת־הַיָּֽלֶד׃ וַאֲמַרַת אֲחָתֵיהּ, לְבַת פַּרְעֹה, הַאֵיזֵיל וְאֶקְרֵי לִיךְ אִתְּתָא מֵינִקְתָּא, מִן יְהוּדַיָתָא; וְתוֹנִיק לִיךְ, יָת רָבְיָא. 
ב,ח וַתֹּֽאמֶר־לָ֥הּ בַּת־פַּרְעֹ֖ה לֵ֑כִי וַתֵּ֨לֶךְ֙ הָֽעַלְמָ֔ה וַתִּקְרָ֖א אֶת־אֵ֥ם הַיָּֽלֶד׃ וַאֲמַרַת לַהּ בַּת פַּרְעֹה, אִיזִילִי; וַאֲזַלַת, עוּלֵימְתָא, וּקְרָת, יָת אִמֵּיהּ דְּרָבְיָא. 
ב,ט וַתֹּ֧אמֶר לָ֣הּ בַּת־פַּרְעֹ֗ה הֵילִ֜יכִי אֶת־הַיֶּ֤לֶד הַזֶּה֙ וְהֵֽינִקִ֣הוּ לִ֔י וַֽאֲנִ֖י אֶתֵּ֣ן אֶת־שְׂכָרֵ֑ךְ וַתִּקַּ֧ח הָֽאִשָּׁ֛ה הַיֶּ֖לֶד וַתְּנִיקֵֽהוּ׃ וַאֲמַרַת לַהּ בַּת פַּרְעֹה, הָלִיכִי יָת רָבְיָא הָדֵין וְאוֹנִיקִיהוּ לִי, וַאֲנָא, אֶתֵּין יָת אַגְרִיךְ; וּנְסֵיבַת אִתְּתָא רָבְיָא, וְאוֹנִיקְתֵּיהּ. 
ב,י וַיִּגְדַּ֣ל הַיֶּ֗לֶד וַתְּבִאֵ֨הוּ֙ לְבַת־פַּרְעֹ֔ה וַֽיְהִי־לָ֖הּ לְבֵ֑ן וַתִּקְרָ֤א שְׁמוֹ֙ מֹשֶׁ֔ה וַתֹּ֕אמֶר כִּ֥י מִן־הַמַּ֖יִם מְשִׁיתִֽהוּ׃ וּרְבָא רָבְיָא, וְאֵיתִיתֵיהּ לְבַת פַּרְעֹה, וַהֲוָה לַהּ, לְבַר; וּקְרָת שְׁמֵיהּ, מֹשֶׁה, וַאֲמַרַת, אֲרֵי מִן מַיָּא שְׁחַלְתֵּיהּ. 
ב,יא וַיְהִ֣י ׀ בַּיָּמִ֣ים הָהֵ֗ם וַיִּגְדַּ֤ל מֹשֶׁה֙ וַיֵּצֵ֣א אֶל־אֶחָ֔יו וַיַּ֖רְא בְּסִבְלֹתָ֑ם וַיַּרְא֙ אִ֣ישׁ מִצְרִ֔י מַכֶּ֥ה אִישׁ־עִבְרִ֖י מֵֽאֶחָֽיו׃ וַהֲוָה בְּיוֹמַיָּא הָאִנּוּן, וּרְבָא מֹשֶׁה וּנְפַק לְוָת אֲחוֹהִי, וַחֲזָא, בְּפֻלְחָנְהוֹן; וַחֲזָא גְּבַר מִצְרַאי, מָחֵי לִגְבַר יְהוּדַאי מֵאֲחוֹהִי. 
ב,יב וַיִּ֤פֶן כֹּה֙ וָכֹ֔ה וַיַּ֖רְא כִּ֣י אֵ֣ין אִ֑ישׁ וַיַּךְ֙ אֶת־הַמִּצְרִ֔י וַֽיִּטְמְנֵ֖הוּ בַּחֽוֹל׃ וְאִתְפְּנִי לְכָא וּלְכָא, וַחֲזָא אֲרֵי לֵית אֲנָשׁ; וּמְחָא, יָת מִצְרָאָה, וְטַמְרֵיהּ, בְּחָלָא. 
ב,יג וַיֵּצֵא֙ בַּיּ֣וֹם הַשֵּׁנִ֔י וְהִנֵּ֛ה שְׁנֵֽי־אֲנָשִׁ֥ים עִבְרִ֖ים נִצִּ֑ים וַיֹּ֨אמֶר֙ לָֽרָשָׁ֔ע לָ֥מָּה תַכֶּ֖ה רֵעֶֽךָ׃ וּנְפַק בְּיוֹמָא תִּנְיָנָא, וְהָא תְּרֵין גֻּבְרִין יְהוּדָאִין נָצַן; וַאֲמַר, לְחַיָּבָא, לְמָא אַתְּ מָחֵי, לְחַבְרָךְ. 
ב,יד וַ֠יֹּאמֶר מִ֣י שָֽׂמְךָ֞ לְאִ֨ישׁ שַׂ֤ר וְשֹׁפֵט֙ עָלֵ֔ינוּ הַלְהָרְגֵ֨נִי֙ אַתָּ֣ה אֹמֵ֔ר כַּֽאֲשֶׁ֥ר הָרַ֖גְתָּ אֶת־הַמִּצְרִ֑י וַיִּירָ֤א מֹשֶׁה֙ וַיֹּאמַ֔ר אָכֵ֖ן נוֹדַ֥ע הַדָּבָֽר׃ וַאֲמַר מַאן שַׁוְיָךְ לִגְבַר רַב וְדַיָּן, עֲלַנָא--הַלְמִקְטְלִי אַתְּ אָמַר, כְּמָא דִּקְטַלְתָּא יָת מִצְרָאָה; וּדְחֵיל מֹשֶׁה וַאֲמַר, בְּקֻשְׁטָא אִתְיְדַע פִּתְגָמָא. 
ב,טו וַיִּשְׁמַ֤ע פַּרְעֹה֙ אֶת־הַדָּבָ֣ר הַזֶּ֔ה וַיְבַקֵּ֖שׁ לַֽהֲרֹ֣ג אֶת־מֹשֶׁ֑ה וַיִּבְרַ֤ח מֹשֶׁה֙ מִפְּנֵ֣י פַרְעֹ֔ה וַיֵּ֥שֶׁב בְּאֶֽרֶץ־מִדְיָ֖ן וַיֵּ֥שֶׁב עַֽל־הַבְּאֵֽר׃ וּשְׁמַע פַּרְעֹה יָת פִּתְגָמָא הָדֵין, וּבְעָא לְמִקְטַל יָת מֹשֶׁה; וַעֲרַק מֹשֶׁה מִן קֳדָם פַּרְעֹה, וִיתֵיב בְּאַרְעָא דְּמִדְיָן וִיתֵיב עַל בֵּירָא. 
ב,טז וּלְכֹהֵ֥ן מִדְיָ֖ן שֶׁ֣בַע בָּנ֑וֹת וַתָּבֹ֣אנָה וַתִּדְלֶ֗נָה וַתְּמַלֶּ֨אנָה֙ אֶת־הָ֣רְהָטִ֔ים לְהַשְׁק֖וֹת צֹ֥אן אֲבִיהֶֽן׃ וּלְרַבָּא דְּמִדְיָן, שְׁבַע בְּנָן; וַאֲתַאָה וּדְלַאָה, וּמְלַאָה יָת רָטַיָּא, לְאַשְׁקָאָה, עָנָא דַּאֲבוּהוֹן. 
ב,יז וַיָּבֹ֥אוּ הָֽרֹעִ֖ים וַיְגָֽרְשׁ֑וּם וַיָּ֤קָם מֹשֶׁה֙ וַיּ֣וֹשִׁעָ֔ן וַיַּ֖שְׁקְ אֶת־צֹאנָֽם׃ וַאֲתוֹ רָעַיָּא, וּטְרַדוּנִין; וְקָם מֹשֶׁה וּפְרַקִנִּין, וְאַשְׁקִי יָת עָנְהוֹן. 
ב,יח וַתָּבֹ֕אנָה אֶל־רְעוּאֵ֖ל אֲבִיהֶ֑ן וַיֹּ֕אמֶר מַדּ֛וּעַ מִֽהַרְתֶּ֥ן בֹּ֖א הַיּֽוֹם׃ וַאֲתַאָה, לְוָת רְעוּאֵל אֲבוּהוֹן; וַאֲמַר, מָדֵין אוֹחִיתִין לְמֵיתֵי יוֹמָא דֵּין. 
ב,יט וַתֹּאמַ֕רְןָ אִ֣ישׁ מִצְרִ֔י הִצִּילָ֖נוּ מִיַּ֣ד הָֽרֹעִ֑ים וְגַם־דָּלֹ֤ה דָלָה֙ לָ֔נוּ וַיַּ֖שְׁקְ אֶת־הַצֹּֽאן׃ וַאֲמַרָא--גֻּבְרָא מִצְרָאָה, שֵׁיזְבַנָא מִיְּדָא דְּרָעַיָּא; וְאַף מִדְלָא דְּלָא לַנָא, וְאַשְׁקִי יָת עָנָא. 
ב,כ וַיֹּ֥אמֶר אֶל־בְּנֹתָ֖יו וְאַיּ֑וֹ לָ֤מָּה זֶּה֙ עֲזַבְתֶּ֣ן אֶת־הָאִ֔ישׁ קִרְאֶ֥ן ל֖וֹ וְיֹ֥אכַל לָֽחֶם׃ וַאֲמַר לִבְנָתֵיהּ, וְאָן הוּא; לְמָא דְּנָן שְׁבַקְתִּין יָת גֻּבְרָא, קְרַן לֵיהּ וְיֵיכוֹל לַחְמָא. 
ב,כא וַיּ֥וֹאֶל מֹשֶׁ֖ה לָשֶׁ֣בֶת אֶת־הָאִ֑ישׁ וַיִּתֵּ֛ן אֶת־צִפֹּרָ֥ה בִתּ֖וֹ לְמֹשֶֽׁה׃ וּצְבִי מֹשֶׁה, לְמִתַּב עִם גֻּבְרָא; וִיהַב יָת צִפּוֹרָה בְּרַתֵּיהּ, לְמֹשֶׁה. 
ב,כב וַתֵּ֣לֶד בֵּ֔ן וַיִּקְרָ֥א אֶת־שְׁמ֖וֹ גֵּֽרְשֹׁ֑ם כִּ֣י אָמַ֔ר גֵּ֣ר הָיִ֔יתִי בְּאֶ֖רֶץ נָכְרִיָּֽה׃ {פ} וִילֵידַת בַּר, וּקְרָא יָת שְׁמֵיהּ גֵּרְשׁוֹם:  אֲרֵי אֲמַר--דַּיָּר הֲוֵיתִי, בַּאֲרַע נֻכְרָאָה. {פ}
ב,כג וַיְהִי֩ בַיָּמִ֨ים הָֽרַבִּ֜ים הָהֵ֗ם וַיָּ֨מָת֙ מֶ֣לֶךְ מִצְרַ֔יִם וַיֵּאָֽנְח֧וּ בְנֵֽי־יִשְׂרָאֵ֛ל מִן־הָֽעֲבֹדָ֖ה וַיִּזְעָ֑קוּ וַתַּ֧עַל שַׁוְעָתָ֛ם אֶל־הָֽאֱלֹהִ֖ים מִן־הָֽעֲבֹדָֽה׃ וַהֲוָה בְּיוֹמַיָּא סַגִּיאַיָּא הָאִנּוּן, וּמִית מַלְכָּא דְּמִצְרַיִם, וְאִתְאָנַחוּ בְּנֵי יִשְׂרָאֵל מִן פֻּלְחָנָא דַּהֲוָה קְשֵׁי עֲלֵיהוֹן, וּזְעִיקוּ; וּסְלֵיקַת קְבִילַתְהוֹן לִקְדָם יְיָ, מִן פֻּלְחָנָא. 
ב,כד וַיִּשְׁמַ֥ע אֱלֹהִ֖ים אֶת־נַֽאֲקָתָ֑ם וַיִּזְכֹּ֤ר אֱלֹהִים֙ אֶת־בְּרִית֔וֹ אֶת־אַבְרָהָ֖ם אֶת־יִצְחָ֥ק וְאֶֽת־יַעֲקֹֽב׃ וּשְׁמִיעַ קֳדָם יְיָ, יָת קְבִילַתְהוֹן; וּדְכִיר יְיָ יָת קְיָמֵיהּ, דְּעִם אַבְרָהָם דְּעִם יִצְחָק וּדְעִם יַעֲקוֹב. 
ב,כה וַיַּ֥רְא אֱלֹהִ֖ים אֶת־בְּנֵ֣י יִשְׂרָאֵ֑ל וַיֵּ֖דַע אֱלֹהִֽים׃ {ס} וּגְלֵי קֳדָם יְיָ, שִׁעְבּוּדָא דִּבְנֵי יִשְׂרָאֵל; וַאֲמַר בְּמֵימְרֵיהּ לְמִפְרַקְהוֹן, יְיָ. {ס}
ג,א וּמֹשֶׁ֗ה הָיָ֥ה רֹעֶ֛ה אֶת־צֹ֛אן יִתְר֥וֹ חֹֽתְנ֖וֹ כֹּהֵ֣ן מִדְיָ֑ן וַיִּנְהַ֤ג אֶת־הַצֹּאן֙ אַחַ֣ר הַמִּדְבָּ֔ר וַיָּבֹ֛א אֶל־הַ֥ר הָֽאֱלֹהִ֖ים חֹרֵֽבָה׃ וּמֹשֶׁה, הֲוָה רָעֵי יָת עָנָא דְּיִתְרוֹ חֲמוּהִי--רַבָּא דְּמִדְיָן; וְדַבַּר יָת עָנָא לַאֲתַר שְׁפַר רִעְיָא לְמַדְבְּרָא, וַאֲתָא לְטוּרָא דְּאִתְגְּלִי עֲלוֹהִי יְקָרָא דַּייָ לְחוֹרֵב. 
ג,ב וַ֠יֵּרָא מַלְאַ֨ךְ יְהוָ֥ה אֵלָ֛יו בְּלַבַּת־אֵ֖שׁ מִתּ֣וֹךְ הַסְּנֶ֑ה וַיַּ֗רְא וְהִנֵּ֤ה הַסְּנֶה֙ בֹּעֵ֣ר בָּאֵ֔שׁ וְהַסְּנֶ֖ה אֵינֶ֥נּוּ אֻכָּֽל׃ וְאִתְגְּלִי מַלְאֲכָא דַּייָ לֵיהּ, בְּשַׁלְהוֹבִית אִישָׁתָא--מִגּוֹ אֲסַנָּא; וַחֲזָא, וְהָא אֲסַנָּא בָּעַר בְּאִישָׁתָא, וַאֲסַנָּא, לָיְתוֹהִי מִתְאֲכִיל. 
ג,ג וַיֹּ֣אמֶר מֹשֶׁ֔ה אָסֻֽרָה־נָּ֣א וְאֶרְאֶ֔ה אֶת־הַמַּרְאֶ֥ה הַגָּדֹ֖ל הַזֶּ֑ה מַדּ֖וּעַ לֹֽא־יִבְעַ֥ר הַסְּנֶֽה׃ וַאֲמַר מֹשֶׁה--אֶתְפְּנֵי כְּעַן וְאֶחְזֵי, יָת חֶזְוָנָא רַבָּא הָדֵין:  מָדֵין, לָא מִתּוֹקַד אֲסַנָּא. 
ג,ד וַיַּ֥רְא יְהוָ֖ה כִּ֣י סָ֣ר לִרְא֑וֹת וַיִּקְרָא֩ אֵלָ֨יו אֱלֹהִ֜ים מִתּ֣וֹךְ הַסְּנֶ֗ה וַיֹּ֛אמֶר מֹשֶׁ֥ה מֹשֶׁ֖ה וַיֹּ֥אמֶר הִנֵּֽנִי׃ וַחֲזָא יְיָ, אֲרֵי אִתְפְּנֵי לְמִחְזֵי; וּקְרָא לֵיהּ יְיָ מִגּוֹ אֲסַנָּא, וַאֲמַר מֹשֶׁה מֹשֶׁה--וַאֲמַר הָאֲנָא. 
ג,ה וַיֹּ֖אמֶר אַל־תִּקְרַ֣ב הֲלֹ֑ם שַׁל־נְעָלֶ֨יךָ֙ מֵעַ֣ל רַגְלֶ֔יךָ כִּ֣י הַמָּק֗וֹם אֲשֶׁ֤ר אַתָּה֙ עוֹמֵ֣ד עָלָ֔יו אַדְמַת־קֹ֖דֶשׁ הֽוּא׃ וַאֲמַר, לָא תִּקְרַב הָלְכָא; שְׁרִי סֵינָךְ, מֵעַל רַגְלָךְ--אֲרֵי אַתְרָא דְּאַתְּ קָאֵים עֲלוֹהִי, אֲתַר קַדִּישׁ הוּא. 
ג,ו וַיֹּ֗אמֶר אָֽנֹכִי֙ אֱלֹהֵ֣י אָבִ֔יךָ אֱלֹהֵ֧י אַבְרָהָ֛ם אֱלֹהֵ֥י יִצְחָ֖ק וֵֽאלֹהֵ֣י יַֽעֲקֹ֑ב וַיַּסְתֵּ֤ר מֹשֶׁה֙ פָּנָ֔יו כִּ֣י יָרֵ֔א מֵֽהַבִּ֖יט אֶל־הָֽאֱלֹהִֽים׃ וַאֲמַר, אֲנָא אֱלָהָא דַּאֲבוּךְ, אֱלָהֵיהּ דְּאַבְרָהָם אֱלָהֵיהּ דְּיִצְחָק, וֵאלָהֵיהּ דְּיַעֲקוֹב; וּכְבַשִׁנּוּן מֹשֶׁה, לְאַפּוֹהִי, אֲרֵי דְּחֵיל, מִלְּאִסְתַּכָּלָא בְּצֵית יְקָרָא דַּייָ. 
ג,ז וַיֹּ֣אמֶר יְהוָ֔ה רָאֹ֥ה רָאִ֛יתִי אֶת־עֳנִ֥י עַמִּ֖י אֲשֶׁ֣ר בְּמִצְרָ֑יִם וְאֶת־צַֽעֲקָתָ֤ם שָׁמַ֨עְתִּי֙ מִפְּנֵ֣י נֹֽגְשָׂ֔יו כִּ֥י יָדַ֖עְתִּי אֶת־מַכְאֹבָֽיו׃ וַאֲמַר יְיָ, מִגְלָא גְּלֵי קֳדָמַי שִׁעְבּוּד עַמִּי דִּבְמִצְרָיִם; וְיָת קְבִילַתְהוֹן שְׁמִיעַ קֳדָמַי מִן קֳדָם מַפְלְחֵיהוֹן, אֲרֵי גְּלֵי קֳדָמַי כֵּיבֵיהוֹן. 
ג,ח וָֽאֵרֵ֞ד לְהַצִּיל֣וֹ ׀ מִיַּ֣ד מִצְרַ֗יִם וּֽלְהַעֲלֹתוֹ֮ מִן־הָאָ֣רֶץ הַהִוא֒ אֶל־אֶ֤רֶץ טוֹבָה֙ וּרְחָבָ֔ה אֶל־אֶ֛רֶץ זָבַ֥ת חָלָ֖ב וּדְבָ֑שׁ אֶל־מְק֤וֹם הַֽכְּנַעֲנִי֙ וְהַ֣חִתִּ֔י וְהָֽאֱמֹרִי֙ וְהַפְּרִזִּ֔י וְהַֽחִוִּ֖י וְהַיְבוּסִֽי׃ וְאִתְגְּלִיתִי לְשֵׁיזָבוּתְהוֹן מִיְּדָא דְּמִצְרָאֵי, וּלְאַסָּקוּתְהוֹן מִן אַרְעָא הַהִיא, לַאֲרַע טָבָא וּפַתְיָא, לַאֲרַע עָבְדָא חֲלָב וּדְבָשׁ--לַאֲתַר כְּנַעֲנָאֵי, וְחִתָּאֵי, וֶאֱמוֹרָאֵי וּפְרִזָּאֵי, וְחִוָּאֵי וִיבוּסָאֵי. 
ג,ט וְעַתָּ֕ה הִנֵּ֛ה צַֽעֲקַ֥ת בְּנֵֽי־יִשְׂרָאֵ֖ל בָּ֣אָה אֵלָ֑י וְגַם־רָאִ֨יתִי֙ אֶת־הַלַּ֔חַץ אֲשֶׁ֥ר מִצְרַ֖יִם לֹֽחֲצִ֥ים אֹתָֽם׃ וּכְעַן, הָא קְבִילַת בְּנֵי יִשְׂרָאֵל עַאלַת לִקְדָמָי; וְאַף גְּלֵי קֳדָמַי, דֻּחְקָא, דְּמִצְרָאֵי, דָּחֲקִין לְהוֹן. 
ג,י וְעַתָּ֣ה לְכָ֔ה וְאֶֽשְׁלָחֲךָ֖ אֶל־פַּרְעֹ֑ה וְהוֹצֵ֛א אֶת־עַמִּ֥י בְנֵֽי־יִשְׂרָאֵ֖ל מִמִּצְרָֽיִם׃ וּכְעַן אֵיתַא, וְאֶשְׁלְחִנָּךְ לְוָת פַּרְעֹה; וְאַפֵּיק יָת עַמִּי בְּנֵי יִשְׂרָאֵל, מִמִּצְרָיִם. 
ג,יא וַיֹּ֤אמֶר מֹשֶׁה֙ אֶל־הָ֣אֱלֹהִ֔ים מִ֣י אָנֹ֔כִי כִּ֥י אֵלֵ֖ךְ אֶל־פַּרְעֹ֑ה וְכִ֥י אוֹצִ֛יא אֶת־בְּנֵ֥י יִשְׂרָאֵ֖ל מִמִּצְרָֽיִם׃ וַאֲמַר מֹשֶׁה, קֳדָם יְיָ, מַאן אֲנָא, אֲרֵי אֵיזֵיל לְוָת פַּרְעֹה; וַאֲרֵי אַפֵּיק יָת בְּנֵי יִשְׂרָאֵל, מִמִּצְרָיִם. 
ג,יב וַיֹּ֨אמֶר֙ כִּֽי־אֶֽהְיֶ֣ה עִמָּ֔ךְ וְזֶה־לְּךָ֣ הָא֔וֹת כִּ֥י אָֽנֹכִ֖י שְׁלַחְתִּ֑יךָ בְּהוֹצִֽיאֲךָ֤ אֶת־הָעָם֙ מִמִּצְרַ֔יִם תַּֽעַבְדוּן֙ אֶת־הָ֣אֱלֹהִ֔ים עַ֖ל הָהָ֥ר הַזֶּֽה׃ וַאֲמַר, אֲרֵי יְהֵי מֵימְרִי בְּסַעֲדָךְ, וְדֵין לָךְ אָתָא, אֲרֵי אֲנָא שְׁלַחְתָּךְ:  בְּאַפָּקוּתָךְ יָת עַמָּא, מִמִּצְרַיִם, תִּפְלְחוּן קֳדָם יְיָ, עַל טוּרָא הָדֵין. 
ג,יג וַיֹּ֨אמֶר מֹשֶׁ֜ה אֶל־הָֽאֱלֹהִ֗ים הִנֵּ֨ה אָֽנֹכִ֣י בָא֮ אֶל־בְּנֵ֣י יִשְׂרָאֵל֒ וְאָֽמַרְתִּ֣י לָהֶ֔ם אֱלֹהֵ֥י אֲבֽוֹתֵיכֶ֖ם שְׁלָחַ֣נִי אֲלֵיכֶ֑ם וְאָֽמְרוּ־לִ֣י מַה־שְּׁמ֔וֹ מָ֥ה אֹמַ֖ר אֲלֵהֶֽם׃ וַאֲמַר מֹשֶׁה קֳדָם יְיָ, הָא אֲנָא אָתֵי לְוָת בְּנֵי יִשְׂרָאֵל, וְאֵימַר לְהוֹן, אֱלָהָא דַּאֲבָהָתְכוֹן שַׁלְחַנִי לְוָתְכוֹן; וְיֵימְרוּן לִי מַאן שְׁמֵיהּ, מָא אֵימַר לְהוֹן. 
ג,יד וַיֹּ֤אמֶר אֱלֹהִים֙ אֶל־מֹשֶׁ֔ה אֶֽהְיֶ֖ה אֲשֶׁ֣ר אֶֽהְיֶ֑ה וַיֹּ֗אמֶר כֹּ֤ה תֹאמַר֙ לִבְנֵ֣י יִשְׂרָאֵ֔ל אֶֽהְיֶ֖ה שְׁלָחַ֥נִי אֲלֵיכֶֽם׃ וַאֲמַר יְיָ לְמֹשֶׁה, אֶהְיֶה אֲשֶׁר אֶהְיֶה; וַאֲמַר, כִּדְנָן תֵּימַר לִבְנֵי יִשְׂרָאֵל, אֶהְיֶה, שַׁלְחַנִי לְוָתְכוֹן. 
ג,טו וַיֹּאמֶר֩ ע֨וֹד אֱלֹהִ֜ים אֶל־מֹשֶׁ֗ה כֹּֽה־תֹאמַר֮ אֶל־בְּנֵ֣י יִשְׂרָאֵל֒ יְהוָ֞ה אֱלֹהֵ֣י אֲבֹֽתֵיכֶ֗ם אֱלֹהֵ֨י אַבְרָהָ֜ם אֱלֹהֵ֥י יִצְחָ֛ק וֵֽאלֹהֵ֥י יַֽעֲקֹ֖ב שְׁלָחַ֣נִי אֲלֵיכֶ֑ם זֶה־שְּׁמִ֣י לְעֹלָ֔ם וְזֶ֥ה זִכְרִ֖י לְדֹ֥ר דֹּֽר׃ וַאֲמַר עוֹד יְיָ לְמֹשֶׁה, כִּדְנָן תֵּימַר לִבְנֵי יִשְׂרָאֵל, יְיָ אֱלָהָא דַּאֲבָהָתְכוֹן אֱלָהֵיהּ דְּאַבְרָהָם אֱלָהֵיהּ דְּיִצְחָק וֵאלָהֵיהּ דְּיַעֲקוֹב, שַׁלְחַנִי לְוָתְכוֹן; דֵּין שְׁמִי לְעָלַם, וְדֵין דֻּכְרָנִי לְכָל דָּר וְדָר. 
ג,טז לֵ֣ךְ וְאָֽסַפְתָּ֞ אֶת־זִקְנֵ֣י יִשְׂרָאֵ֗ל וְאָֽמַרְתָּ֤ אֲלֵהֶם֙ יְהוָ֞ה אֱלֹהֵ֤י אֲבֹֽתֵיכֶם֙ נִרְאָ֣ה אֵלַ֔י אֱלֹהֵ֧י אַבְרָהָ֛ם יִצְחָ֥ק וְיַֽעֲקֹ֖ב לֵאמֹ֑ר פָּקֹ֤ד פָּקַ֨דְתִּי֙ אֶתְכֶ֔ם וְאֶת־הֶֽעָשׂ֥וּי לָכֶ֖ם בְּמִצְרָֽיִם׃ אִיזֵיל וְתִכְנוֹשׁ יָת סָבֵי יִשְׂרָאֵל, וְתֵימַר לְהוֹן יְיָ אֱלָהָא דַּאֲבָהָתְכוֹן אִתְגְּלִי לִי, אֱלָהֵיהּ דְּאַבְרָהָם יִצְחָק וְיַעֲקוֹב, לְמֵימַר:  מִדְכָר דְּכִירְנָא יָתְכוֹן, וְיָת דְּאִתְעֲבֵיד לְכוֹן בְּמִצְרָיִם. 
ג,יז וָֽאֹמַ֗ר אַֽעֲלֶ֣ה אֶתְכֶם֮ מֵֽעֳנִ֣י מִצְרַיִם֒ אֶל־אֶ֤רֶץ הַֽכְּנַעֲנִי֙ וְהַ֣חִתִּ֔י וְהָֽאֱמֹרִי֙ וְהַפְּרִזִּ֔י וְהַֽחִוִּ֖י וְהַיְבוּסִ֑י אֶל־אֶ֛רֶץ זָבַ֥ת חָלָ֖ב וּדְבָֽשׁ׃ וַאֲמַרִית, אַסֵּיק יָתְכוֹן מִשִּׁעְבּוּד מִצְרָאֵי, לַאֲרַע כְּנַעֲנָאֵי וְחִתָּאֵי, וֶאֱמוֹרָאֵי וּפְרִזָּאֵי וְחִוָּאֵי וִיבוּסָאֵי--לַאֲרַע עָבְדָא חֲלָב, וּדְבָשׁ. 
ג,יח וְשָֽׁמְע֖וּ לְקֹלֶ֑ךָ וּבָאתָ֡ אַתָּה֩ וְזִקְנֵ֨י יִשְׂרָאֵ֜ל אֶל־מֶ֣לֶךְ מִצְרַ֗יִם וַֽאֲמַרְתֶּ֤ם אֵלָיו֙ יְהוָ֞ה אֱלֹהֵ֤י הָֽעִבְרִיִּים֙ נִקְרָ֣ה עָלֵ֔ינוּ וְעַתָּ֗ה נֵֽלְכָה־נָּ֞א דֶּ֣רֶךְ שְׁלֹ֤שֶׁת יָמִים֙ בַּמִּדְבָּ֔ר וְנִזְבְּחָ֖ה לַֽיהוָ֥ה אֱלֹהֵֽינוּ׃ וִיקַבְּלוּן, מִנָּךְ; וְתֵיתֵי אַתְּ וְסָבֵי יִשְׂרָאֵל לְוָת מַלְכָּא דְּמִצְרַיִם, וְתֵימְרוּן לֵיהּ יְיָ אֱלָהָא דִּיהוּדָאֵי אִתְקְרִי עֲלַנָא, וּכְעַן נֵיזֵיל כְּעַן מַהְלַךְ תְּלָתָא יוֹמִין בְּמַדְבְּרָא, וּנְדַבַּח קֳדָם יְיָ אֱלָהַנָא. 
ג,יט וַֽאֲנִ֣י יָדַ֔עְתִּי כִּ֠י לֹֽא־יִתֵּ֥ן אֶתְכֶ֛ם מֶ֥לֶךְ מִצְרַ֖יִם לַֽהֲלֹ֑ךְ וְלֹ֖א בְּיָ֥ד חֲזָקָֽה׃ וּקְדָמַי גְּלֵי--אֲרֵי לָא יִשְׁבּוֹק יָתְכוֹן מַלְכָּא דְּמִצְרַיִם, לְמֵיזַל:  וְלָא, מִן קֳדָם דְּחֵילֵיהּ תַּקִּיף. 
ג,כ וְשָֽׁלַחְתִּ֤י אֶת־יָדִי֙ וְהִכֵּיתִ֣י אֶת־מִצְרַ֔יִם בְּכֹל֙ נִפְלְאֹתַ֔י אֲשֶׁ֥ר אֶֽעֱשֶׂ֖ה בְּקִרְבּ֑וֹ וְאַֽחֲרֵי־כֵ֖ן יְשַׁלַּ֥ח אֶתְכֶֽם׃ וְאֶשְׁלַח יָת מַחַת גְּבוּרְתִי, וְאֶמְחֵי יָת מִצְרָאֵי, בְּכֹל פְּרִישָׁתַי, דְּאַעֲבֵיד בֵּינֵיהוֹן; וּבָתַר כֵּין, יְשַׁלַּח יָתְכוֹן. 
ג,כא וְנָֽתַתִּ֛י אֶת־חֵ֥ן הָֽעָם־הַזֶּ֖ה בְּעֵינֵ֣י מִצְרָ֑יִם וְהָיָה֙ כִּ֣י תֵֽלֵכ֔וּן לֹ֥א תֵֽלְכ֖וּ רֵיקָֽם׃ וְאֶתֵּין יָת עַמָּא הָדֵין לְרַחֲמִין, בְּעֵינֵי מִצְרָאֵי; וִיהֵי אֲרֵי תְּהָכוּן, לָא תְּהָכוּן רֵיקָנִין. 
ג,כב וְשָֽׁאֲלָ֨ה אִשָּׁ֤ה מִשְּׁכֶנְתָּהּ֙ וּמִגָּרַ֣ת בֵּיתָ֔הּ כְּלֵי־כֶ֛סֶף וּכְלֵ֥י זָהָ֖ב וּשְׂמָלֹ֑ת וְשַׂמְתֶּ֗ם עַל־בְּנֵיכֶם֙ וְעַל־בְּנֹ֣תֵיכֶ֔ם וְנִצַּלְתֶּ֖ם אֶת־מִצְרָֽיִם׃ וְתִשְׁאַל אִתְּתָא מִשֵּׁיבָבְתַהּ וּמִקָּרִיבַת בֵּיתַהּ, מָנִין דִּכְסַף וּמָנִין דִּדְהַב וּלְבוּשִׁין; וּתְשַׁוּוֹן, עַל בְּנֵיכוֹן וְעַל בְּנָתְכוֹן, וּתְרוֹקְנוּן, יָת מִצְרָיִם. 
ד,א וַיַּ֤עַן מֹשֶׁה֙ וַיֹּ֔אמֶר וְהֵן֙ לֹֽא־יַאֲמִ֣ינוּ לִ֔י וְלֹ֥א יִשְׁמְע֖וּ בְּקֹלִ֑י כִּ֣י יֹֽאמְר֔וּ לֹֽא־נִרְאָ֥ה אֵלֶ֖יךָ יְהוָֽה׃ וַאֲתֵיב מֹשֶׁה, וַאֲמַר, וְהָא לָא יְהֵימְנוּן לִי, וְלָא יְקַבְּלוּן מִנִּי:  אֲרֵי יֵימְרוּן, לָא אִתְגְּלִי לָךְ יְיָ. 
ד,ב וַיֹּ֧אמֶר אֵלָ֛יו יְהוָ֖ה מזה (מַה־זֶּ֣ה) בְיָדֶ֑ךָ וַיֹּ֖אמֶר מַטֶּֽה׃ וַאֲמַר לֵיהּ יְיָ, מָא דֵּין בִּידָךְ; וַאֲמַר, חֻטְרָא. 
ד,ג וַיֹּ֨אמֶר֙ הַשְׁלִיכֵ֣הוּ אַ֔רְצָה וַיַּשְׁלִכֵ֥הוּ אַ֖רְצָה וַיְהִ֣י לְנָחָ֑שׁ וַיָּ֥נָס מֹשֶׁ֖ה מִפָּנָֽיו׃ וַאֲמַר רְמוֹהִי לְאַרְעָא, וּרְמָהִי לְאַרְעָא וַהֲוָה לְחִוְיָא; וַעֲרַק מֹשֶׁה, מִן קֳדָמוֹהִי. 
ד,ד וַיֹּ֤אמֶר יְהוָה֙ אֶל־מֹשֶׁ֔ה שְׁלַח֙ יָֽדְךָ֔ וֶֽאֱחֹ֖ז בִּזְנָב֑וֹ וַיִּשְׁלַ֤ח יָדוֹ֙ וַיַּ֣חֲזֶק בּ֔וֹ וַיְהִ֥י לְמַטֶּ֖ה בְּכַפּֽוֹ׃ וַאֲמַר יְיָ, לְמֹשֶׁה, אוֹשֵׁיט יְדָךְ, וְאֵיחוֹד בְּדַנְבֵּיהּ; וְאוֹשֵׁיט יְדֵיהּ וְאַתְקֵיף בֵּיהּ, וַהֲוָה לְחֻטְרָא בִּידֵיהּ. 
ד,ה לְמַ֣עַן יַֽאֲמִ֔ינוּ כִּֽי־נִרְאָ֥ה אֵלֶ֛יךָ יְהוָ֖ה אֱלֹהֵ֣י אֲבֹתָ֑ם אֱלֹהֵ֧י אַבְרָהָ֛ם אֱלֹהֵ֥י יִצְחָ֖ק וֵֽאלֹהֵ֥י יַֽעֲקֹֽב׃ בְּדִיל דִּיהֵימְנוּן, אֲרֵי אִתְגְּלִי לָךְ יְיָ אֱלָהָא דַּאֲבָהָתְהוֹן:  אֱלָהֵיהּ דְּאַבְרָהָם אֱלָהֵיהּ דְּיִצְחָק, וֵאלָהֵיהּ דְּיַעֲקוֹב. 
ד,ו וַיֹּאמֶר֩ יְהוָ֨ה ל֜וֹ ע֗וֹד הָֽבֵא־נָ֤א יָֽדְךָ֙ בְּחֵיקֶ֔ךָ וַיָּבֵ֥א יָד֖וֹ בְּחֵיק֑וֹ וַיּ֣וֹצִאָ֔הּ וְהִנֵּ֥ה יָד֖וֹ מְצֹרַ֥עַת כַּשָּֽׁלֶג׃ וַאֲמַר יְיָ לֵיהּ עוֹד, אַעֵיל כְּעַן יְדָךְ בְּעִטְפָךְ, וְאַעֵיל יְדֵיהּ, בְּעִטְפֵיהּ; וְאַפְּקַהּ, וְהָא יְדֵיהּ חָוְרָא כְּתַלְגָּא. 
ד,ז וַיֹּ֗אמֶר הָשֵׁ֤ב יָֽדְךָ֙ אֶל־חֵיקֶ֔ךָ וַיָּ֥שֶׁב יָד֖וֹ אֶל־חֵיק֑וֹ וַיּֽוֹצִאָהּ֙ מֵֽחֵיק֔וֹ וְהִנֵּה־שָׁ֖בָה כִּבְשָׂרֽוֹ׃ וַאֲמַר, אֲתֵיב יְדָךְ לְעִטְפָךְ, וַאֲתֵיב יְדֵיהּ, לְעִטְפֵיהּ; וְאַפְּקַהּ, מֵעִטְפֵיהּ, וְהָא תַּבַת, הֲוָת כְּבִסְרֵיהּ. 
ד,ח וְהָיָה֙ אִם־לֹ֣א יַֽאֲמִ֣ינוּ לָ֔ךְ וְלֹ֣א יִשְׁמְע֔וּ לְקֹ֖ל הָאֹ֣ת הָֽרִאשׁ֑וֹן וְהֶֽאֱמִ֔ינוּ לְקֹ֖ל הָאֹ֥ת הָאַֽחֲרֽוֹן׃ וִיהֵי, אִם לָא יְהֵימְנוּן לָךְ, וְלָא יְקַבְּלוּן, לְקָל אָתָא קַדְמָאָה--וִיהֵימְנוּן, לְקָל אָתָא בָּתְרָאָה. 
ד,ט וְהָיָ֡ה אִם־לֹ֣א יַֽאֲמִ֡ינוּ גַּם֩ לִשְׁנֵ֨י הָֽאֹת֜וֹת הָאֵ֗לֶּה וְלֹ֤א יִשְׁמְעוּן֙ לְקֹלֶ֔ךָ וְלָֽקַחְתָּ֙ מִמֵּימֵ֣י הַיְאֹ֔ר וְשָֽׁפַכְתָּ֖ הַיַּבָּשָׁ֑ה וְהָי֤וּ הַמַּ֨יִם֙ אֲשֶׁ֣ר תִּקַּ֣ח מִן־הַיְאֹ֔ר וְהָי֥וּ לְדָ֖ם בַּיַּבָּֽשֶׁת׃ וִיהֵי אִם לָא יְהֵימְנוּן אַף לִתְרֵין אָתַיָּא הָאִלֵּין, וְלָא יְקַבְּלוּן מִנָּךְ--וְתִסַּב מִמַּיָּא דִּבְנַהְרָא, וְתֵישׁוֹד לְיַבֶּשְׁתָּא; וִיהוֹן מַיָּא דְּתִסַּב מִן נַהְרָא, וִיהוֹן לִדְמָא בְּיַבֶּשְׁתָּא. 
ד,י וַיֹּ֨אמֶר מֹשֶׁ֣ה אֶל־יְהוָה֮ בִּ֣י אֲדֹנָי֒ לֹא֩ אִ֨ישׁ דְּבָרִ֜ים אָנֹ֗כִי גַּ֤ם מִתְּמוֹל֙ גַּ֣ם מִשִּׁלְשֹׁ֔ם גַּ֛ם מֵאָ֥ז דַּבֶּרְךָ֖ אֶל־עַבְדֶּ֑ךָ כִּ֧י כְבַד־פֶּ֛ה וּכְבַ֥ד לָשׁ֖וֹן אָנֹֽכִי׃ וַאֲמַר מֹשֶׁה קֳדָם יְיָ, בְּבָעוּ יְיָ, לָא גְּבַר דְּמִלּוּל אֲנָא אַף מֵאֶתְמָלִי אַף מִדְּקַמּוֹהִי, אַף מֵעִדָּן דְּמַלֵּילְתָּא עִם עַבְדָּךְ:  אֲרֵי יַקִּיר מַמְלַל וְעַמִּיק לִישָׁן, אֲנָא. 
ד,יא וַיֹּ֨אמֶר יְהוָ֜ה אֵלָ֗יו מִ֣י שָׂ֣ם פֶּה֮ לָֽאָדָם֒ א֚וֹ מִֽי־יָשׂ֣וּם אִלֵּ֔ם א֣וֹ חֵרֵ֔שׁ א֥וֹ פִקֵּ֖חַ א֣וֹ עִוֵּ֑ר הֲלֹ֥א אָֽנֹכִ֖י יְהוָֽה׃ וַאֲמַר יְיָ לֵיהּ, מַאן שַׁוִּי פֻּמָּא לַאֲנָשָׁא, אוֹ מַאן שַׁוִּי אִלֵּימָא, אוֹ חֶרְשָׁא אוֹ פְּתִיחָא אוֹ עֲוִירָא--הֲלָא אֲנָא, יְיָ. 
ד,יב וְעַתָּ֖ה לֵ֑ךְ וְאָֽנֹכִי֙ אֶֽהְיֶ֣ה עִם־פִּ֔יךָ וְהֽוֹרֵיתִ֖יךָ אֲשֶׁ֥ר תְּדַבֵּֽר׃ וּכְעַן, אִיזֵיל; וּמֵימְרִי יְהֵי עִם פֻּמָּךְ, וְאַלְּפִנָּךְ דִּתְמַלֵּיל. 
ד,יג וַיֹּ֖אמֶר בִּ֣י אֲדֹנָ֑י שְֽׁלַֽח־נָ֖א בְּיַד־תִּשְׁלָֽח׃ וַאֲמַר, בְּבָעוּ יְיָ; שְׁלַח כְּעַן, בְּיַד מַאן דְּכָשַׁר לְמִשְׁלַח. 
ד,יד וַיִּֽחַר־אַ֨ף יְהוָ֜ה בְּמֹשֶׁ֗ה וַיֹּ֨אמֶר֙ הֲלֹ֨א אַֽהֲרֹ֤ן אָחִ֨יךָ֙ הַלֵּוִ֔י יָדַ֕עְתִּי כִּֽי־דַבֵּ֥ר יְדַבֵּ֖ר ה֑וּא וְגַ֤ם הִנֵּה־הוּא֙ יֹצֵ֣א לִקְרָאתֶ֔ךָ וְרָֽאֲךָ֖ וְשָׂמַ֥ח בְּלִבּֽוֹ׃ וּתְקֵיף רֻגְזָא דַּייָ בְּמֹשֶׁה, וַאֲמַר הֲלָא אַהֲרוֹן אֲחוּךְ לֵיוָאָה--גְּלֵי קֳדָמַי, אֲרֵי מַלָּלָא יְמַלֵּיל הוּא; וְאַף הָא הוּא נָפֵיק לְקַדָּמוּתָךְ, וְיִחְזֵינָךְ וְיִחְדֵּי בְּלִבֵּיהּ. 
ד,טו וְדִבַּרְתָּ֣ אֵלָ֔יו וְשַׂמְתָּ֥ אֶת־הַדְּבָרִ֖ים בְּפִ֑יו וְאָֽנֹכִ֗י אֶֽהְיֶ֤ה עִם־פִּ֨יךָ֙ וְעִם־פִּ֔יהוּ וְהֽוֹרֵיתִ֣י אֶתְכֶ֔ם אֵ֖ת אֲשֶׁ֥ר תַּֽעֲשֽׂוּן׃ וּתְמַלֵּיל עִמֵּיהּ, וּתְשַׁוֵּי יָת פִּתְגָמַיָּא בְּפֻמֵּיהּ; וּמֵימְרִי, יְהֵי עִם פֻּמָּךְ וְעִם פֻּמֵּיהּ, וְאַלֵּיף יָתְכוֹן, יָת דְּתַעְבְּדוּן. 
ד,טז וְדִבֶּר־ה֥וּא לְךָ֖ אֶל־הָעָ֑ם וְהָ֤יָה הוּא֙ יִֽהְיֶה־לְּךָ֣ לְפֶ֔ה וְאַתָּ֖ה תִּֽהְיֶה־לּ֥וֹ לֵֽאלֹהִֽים׃ וִימַלֵּיל הוּא לָךְ, עִם עַמָּא; וִיהֵי הוּא יְהֵי לָךְ לִמְתֻרְגְּמָן, וְאַתְּ תְּהֵי לֵיהּ לְרָב. 
ד,יז וְאֶת־הַמַּטֶּ֥ה הַזֶּ֖ה תִּקַּ֣ח בְּיָדֶ֑ךָ אֲשֶׁ֥ר תַּֽעֲשֶׂה־בּ֖וֹ אֶת־הָֽאֹתֹֽת׃ {פ} וְיָת חֻטְרָא הָדֵין, תִּסַּב בִּידָךְ, דְּתַעֲבֵיד בֵּיהּ, יָת אָתַיָּא. {פ}
ד,יח וַיֵּ֨לֶךְ מֹשֶׁ֜ה וַיָּ֣שָׁב ׀ אֶל־יֶ֣תֶר חֹֽתְנ֗וֹ וַיֹּ֤אמֶר לוֹ֙ אֵ֣לְכָה נָּ֗א וְאָשׁ֨וּבָה֙ אֶל־אַחַ֣י אֲשֶׁר־בְּמִצְרַ֔יִם וְאֶרְאֶ֖ה הַֽעוֹדָ֣ם חַיִּ֑ים וַיֹּ֧אמֶר יִתְר֛וֹ לְמֹשֶׁ֖ה לֵ֥ךְ לְשָׁלֽוֹם׃ וַאֲזַל מֹשֶׁה וְתָב לְוָת יֶתֶר חֲמוּהִי, וַאֲמַר לֵיהּ אֵיזֵיל כְּעַן וַאֲתוּב לְוָת אַחַי דִּבְמִצְרַיִם, וְאֶחְזֵי, הַעַד כְּעַן קַיָּמִין; וַאֲמַר יִתְרוֹ לְמֹשֶׁה, אִיזֵיל לִשְׁלָם. 
ד,יט וַיֹּ֨אמֶר יְהוָ֤ה אֶל־מֹשֶׁה֙ בְּמִדְיָ֔ן לֵ֖ךְ שֻׁ֣ב מִצְרָ֑יִם כִּי־מֵ֨תוּ֙ כָּל־הָ֣אֲנָשִׁ֔ים הַֽמְבַקְשִׁ֖ים אֶת־נַפְשֶֽׁךָ׃ וַאֲמַר יְיָ לְמֹשֶׁה בְּמִדְיָן, אִיזֵיל תּוּב לְמִצְרָיִם:  אֲרֵי מִיתוּ, כָּל גֻּבְרַיָּא, דִּבְעוֹ, לְמִקְטְלָךְ. 
ד,כ וַיִּקַּ֨ח מֹשֶׁ֜ה אֶת־אִשְׁתּ֣וֹ וְאֶת־בָּנָ֗יו וַיַּרְכִּבֵם֙ עַֽל־הַחֲמֹ֔ר וַיָּ֖שָׁב אַ֣רְצָה מִצְרָ֑יִם וַיִּקַּ֥ח מֹשֶׁ֛ה אֶת־מַטֵּ֥ה הָֽאֱלֹהִ֖ים בְּיָדֽוֹ׃ וּדְבַר מֹשֶׁה יָת אִתְּתֵיהּ וְיָת בְּנוֹהִי, וְאַרְכֵּיבִנּוּן עַל חֲמָרָא, וְתָב, לְאַרְעָא דְּמִצְרָיִם; וּנְסֵיב מֹשֶׁה יָת חֻטְרָא דְּאִתְעֲבִידוּ בֵּיהּ נִסִּין מִן קֳדָם יְיָ, בִּידֵיהּ. 
ד,כא וַיֹּ֣אמֶר יְהוָה֮ אֶל־מֹשֶׁה֒ בְּלֶכְתְּךָ֙ לָשׁ֣וּב מִצְרַ֔יְמָה רְאֵ֗ה כָּל־הַמֹּֽפְתִים֙ אֲשֶׁר־שַׂ֣מְתִּי בְיָדֶ֔ךָ וַֽעֲשִׂיתָ֖ם לִפְנֵ֣י פַרְעֹ֑ה וַֽאֲנִי֙ אֲחַזֵּ֣ק אֶת־לִבּ֔וֹ וְלֹ֥א יְשַׁלַּ֖ח אֶת־הָעָֽם׃ וַאֲמַר יְיָ, לְמֹשֶׁה, בִּמְהָכָךְ לִמְתָּב לְמִצְרַיִם, חֲזִי כָּל מוֹפְתַיָּא דְּשַׁוִּיתִי בִּידָךְ וְתַעֲבֵידִנּוּן קֳדָם פַּרְעֹה; וַאֲנָא אֲתַקֵּיף יָת לִבֵּיהּ, וְלָא יְשַׁלַּח יָת עַמָּא. 
ד,כב וְאָֽמַרְתָּ֖ אֶל־פַּרְעֹ֑ה כֹּ֚ה אָמַ֣ר יְהוָ֔ה בְּנִ֥י בְכֹרִ֖י יִשְׂרָאֵֽל׃ וְתֵימַר, לְפַרְעֹה:  כִּדְנָן אֲמַר יְיָ, בְּרִי בֻּכְרִי יִשְׂרָאֵל. 
ד,כג וָֽאֹמַ֣ר אֵלֶ֗יךָ שַׁלַּ֤ח אֶת־בְּנִי֙ וְיַֽעַבְדֵ֔נִי וַתְּמָאֵ֖ן לְשַׁלְּח֑וֹ הִנֵּה֙ אָֽנֹכִ֣י הֹרֵ֔ג אֶת־בִּנְךָ֖ בְּכֹרֶֽךָ׃ וַאֲמַרִית לָךְ, שַׁלַּח יָת בְּרִי וְיִפְלַח קֳדָמַי, וּמְסָרֵיב אַתְּ, לְשַׁלָּחוּתֵיהּ--הָא אֲנָא קָטֵיל, יָת בְּרָךְ בֻּכְרָךְ. 
ד,כד וַיְהִ֥י בַדֶּ֖רֶךְ בַּמָּל֑וֹן וַיִּפְגְּשֵׁ֣הוּ יְהוָ֔ה וַיְבַקֵּ֖שׁ הֲמִיתֽוֹ׃ וַהֲוָה בְּאוֹרְחָא, בְּבֵית מְבָתָא; וְעָרַע בֵּיהּ מַלְאֲכָא דַּייָ, וּבְעָא לְמִקְטְלֵיהּ. 
ד,כה וַתִּקַּ֨ח צִפֹּרָ֜ה צֹ֗ר וַתִּכְרֹת֙ אֶת־עָרְלַ֣ת בְּנָ֔הּ וַתַּגַּ֖ע לְרַגְלָ֑יו וַתֹּ֕אמֶר כִּ֧י חֲתַן־דָּמִ֛ים אַתָּ֖ה לִֽי׃ וּנְסֵיבַת צִפּוֹרָה טִנָּרָא, וּגְזַרַת יָת עָרְלַת בְּרַהּ, וְקָרֵיבַת, לִקְדָמוֹהִי; וַאֲמַרַת, בִּדְמָא דִּמְהֻלְתָּא הָדֵין אִתְיְהֵיב חַתְנָא לַנָא. 
ד,כו וַיִּ֖רֶף מִמֶּ֑נּוּ אָ֚ז אָֽמְרָ֔ה חֲתַ֥ן דָּמִ֖ים לַמּוּלֹֽת׃ {פ} וְנָח, מִנֵּיהּ; בְּכֵין, אֲמַרַת, אִלּוּלֵי דְּמָא דִּמְהֻלְתָּא הָדֵין, אִתְחַיַּב חַתְנָא קְטוֹל. {פ}
ד,כז וַיֹּ֤אמֶר יְהוָה֙ אֶֽל־אַהֲרֹ֔ן לֵ֛ךְ לִקְרַ֥את מֹשֶׁ֖ה הַמִּדְבָּ֑רָה וַיֵּ֗לֶךְ וַֽיִּפְגְּשֵׁ֛הוּ בְּהַ֥ר הָֽאֱלֹהִ֖ים וַיִּשַּׁק־לֽוֹ׃ וַאֲמַר יְיָ לְאַהֲרוֹן, אִיזֵיל לְקַדָּמוּת מֹשֶׁה לְמַדְבְּרָא; וַאֲזַל, וְעָרְעֵיהּ בְּטוּרָא דְּאִתְגְּלִי עֲלוֹהִי יְקָרָא דַּייָ--וְנַשֵּׁיק לֵיהּ. 
ד,כח וַיַּגֵּ֤ד מֹשֶׁה֙ לְאַֽהֲרֹ֔ן אֵ֛ת כָּל־דִּבְרֵ֥י יְהוָ֖ה אֲשֶׁ֣ר שְׁלָח֑וֹ וְאֵ֥ת כָּל־הָֽאֹתֹ֖ת אֲשֶׁ֥ר צִוָּֽהוּ׃ וְחַוִּי מֹשֶׁה לְאַהֲרוֹן, יָת כָּל פִּתְגָמַיָּא דַּייָ דְּשַׁלְחֵיהּ, וְיָת כָּל אָתַיָּא, דְּפַקְּדֵיהּ. 
ד,כט וַיֵּ֥לֶךְ מֹשֶׁ֖ה וְאַֽהֲרֹ֑ן וַיַּ֣אַסְפ֔וּ אֶת־כָּל־זִקְנֵ֖י בְּנֵ֥י יִשְׂרָאֵֽל׃ וַאֲזַל מֹשֶׁה, וְאַהֲרוֹן; וּכְנַשׁוּ, יָת כָּל סָבֵי בְּנֵי יִשְׂרָאֵל. 
ד,ל וַיְדַבֵּ֣ר אַֽהֲרֹ֔ן אֵ֚ת כָּל־הַדְּבָרִ֔ים אֲשֶׁר־דִּבֶּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֑ה וַיַּ֥עַשׂ הָֽאֹתֹ֖ת לְעֵינֵ֥י הָעָֽם׃ וּמַלֵּיל אַהֲרוֹן--יָת כָּל פִּתְגָמַיָּא, דְּמַלֵּיל יְיָ עִם מֹשֶׁה; וַעֲבַד אָתַיָּא, לְעֵינֵי עַמָּא. 
ד,לא וַֽיַּאֲמֵ֖ן הָעָ֑ם וַֽיִּשְׁמְע֡וּ כִּֽי־פָקַ֨ד יְהוָ֜ה אֶת־בְּנֵ֣י יִשְׂרָאֵ֗ל וְכִ֤י רָאָה֙ אֶת־עָנְיָ֔ם וַֽיִּקְּד֖וּ וַיִּֽשְׁתַּחֲוֽוּ׃ וְהֵימִין, עַמָּא; וּשְׁמַעוּ אֲרֵי דְּכִיר יְיָ יָת בְּנֵי יִשְׂרָאֵל, וַאֲרֵי גְּלֵי קֳדָמוֹהִי שִׁעְבּוּדְהוֹן, וּכְרַעוּ, וּסְגִידוּ. 
ה,א וְאַחַ֗ר בָּ֚אוּ מֹשֶׁ֣ה וְאַֽהֲרֹ֔ן וַיֹּֽאמְר֖וּ אֶל־פַּרְעֹ֑ה כֹּֽה־אָמַ֤ר יְהוָה֙ אֱלֹהֵ֣י יִשְׂרָאֵ֔ל שַׁלַּח֙ אֶת־עַמִּ֔י וְיָחֹ֥גּוּ לִ֖י בַּמִּדְבָּֽר׃ וּבָתָר כֵּין, עָאלוּ מֹשֶׁה וְאַהֲרוֹן, וַאֲמַרוּ, לְפַרְעֹה:  כִּדְנָן אֲמַר יְיָ, אֱלָהָא דְּיִשְׂרָאֵל, שַׁלַּח יָת עַמִּי, וְיֵיחֲגוּן קֳדָמַי בְּמַדְבְּרָא. 
ה,ב וַיֹּ֣אמֶר פַּרְעֹ֔ה מִ֤י יְהוָה֙ אֲשֶׁ֣ר אֶשְׁמַ֣ע בְּקֹל֔וֹ לְשַׁלַּ֖ח אֶת־יִשְׂרָאֵ֑ל לֹ֤א יָדַ֨עְתִּי֙ אֶת־יְהוָ֔ה וְגַ֥ם אֶת־יִשְׂרָאֵ֖ל לֹ֥א אֲשַׁלֵּֽחַ׃ וַאֲמַר פַּרְעֹה--שְׁמָא דַּייָ לָא אִתְגְּלִי לִי דַּאֲקַבֵּיל לְמֵימְרֵיהּ, לְשַׁלָּחָא יָת יִשְׂרָאֵל:  לָא אִתְגְּלִי לִי שְׁמָא דַּייָ, וְאַף יָת יִשְׂרָאֵל לָא אֲשַׁלַּח. 
ה,ג וַיֹּ֣אמְר֔וּ אֱלֹהֵ֥י הָֽעִבְרִ֖ים נִקְרָ֣א עָלֵ֑ינוּ נֵ֣לְכָה נָּ֡א דֶּרֶךְ֩ שְׁלֹ֨שֶׁת יָמִ֜ים בַּמִּדְבָּ֗ר וְנִזְבְּחָה֙ לַֽיהוָ֣ה אֱלֹהֵ֔ינוּ פֶּ֨ן־יִפְגָּעֵ֔נוּ בַּדֶּ֖בֶר א֥וֹ בֶחָֽרֶב׃ וַאֲמַרוּ, אֱלָהָא דִּיהוּדָאֵי אִתְגְלִי עֲלַנָא; נֵיזֵיל כְּעַן מַהְלַךְ תְּלָתָא יוֹמִין בְּמַדְבְּרָא, וּנְדַבַּח קֳדָם יְיָ אֱלָהַנָא--דִּלְמָא יְעָרְעִנַּנָא, בְּמוֹת אוֹ בִּקְטוֹל. 
ה,ד וַיֹּ֤אמֶר אֲלֵהֶם֙ מֶ֣לֶךְ מִצְרַ֔יִם לָ֚מָּה מֹשֶׁ֣ה וְאַֽהֲרֹ֔ן תַּפְרִ֥יעוּ אֶת־הָעָ֖ם מִמַּֽעֲשָׂ֑יו לְכ֖וּ לְסִבְלֹֽתֵיכֶֽם׃ וַאֲמַר לְהוֹן, מַלְכָּא דְּמִצְרַיִם, לְמָא מֹשֶׁה וְאַהֲרוֹן, תְּבַטְּלוּן יָת עַמָּא מֵעֲבִידַתְהוֹן; אִיזִילוּ, לְפֻלְחָנְכוֹן. 
ה,ה וַיֹּ֣אמֶר פַּרְעֹ֔ה הֵן־רַבִּ֥ים עַתָּ֖ה עַ֣ם הָאָ֑רֶץ וְהִשְׁבַּתֶּ֥ם אֹתָ֖ם מִסִּבְלֹתָֽם׃ וַאֲמַר פַּרְעֹה, הָא מִדְּסַגִּיאִין כְּעַן עַמָּא דְּאַרְעָא; וּתְבַטְּלוּן יָתְהוֹן, מִפֻּלְחָנְהוֹן. 
ה,ו וַיְצַ֥ו פַּרְעֹ֖ה בַּיּ֣וֹם הַה֑וּא אֶת־הַנֹּֽגְשִׂ֣ים בָּעָ֔ם וְאֶת־שֹֽׁטְרָ֖יו לֵאמֹֽר׃ וּפַקֵּיד פַּרְעֹה, בְּיוֹמָא הַהוּא, יָת שִׁלְטוֹנֵי עַמָּא, וְיָת סָרְכוֹהִי לְמֵימַר. 
ה,ז לֹ֣א תֹֽאסִפ֞וּן לָתֵ֨ת תֶּ֧בֶן לָעָ֛ם לִלְבֹּ֥ן הַלְּבֵנִ֖ים כִּתְמ֣וֹל שִׁלְשֹׁ֑ם הֵ֚ם יֵֽלְכ֔וּ וְקֹֽשְׁשׁ֥וּ לָהֶ֖ם תֶּֽבֶן׃ לָא תֵּיסְפוּן לְמִתַּן תִּבְנָא לְעַמָּא, לְמִרְמֵי לִבְנִין--כְּמֵאֶתְמָלִי וּמִדְּקַמּוֹהִי:  אִנּוּן, יֵיזְלוּן, וִיגָבְבוּן לְהוֹן, תִּבְנָא. 
ה,ח וְאֶת־מַתְכֹּ֨נֶת הַלְּבֵנִ֜ים אֲשֶׁ֣ר הֵם֩ עֹשִׂ֨ים תְּמ֤וֹל שִׁלְשֹׁם֙ תָּשִׂ֣ימוּ עֲלֵיהֶ֔ם לֹ֥א תִגְרְע֖וּ מִמֶּ֑נּוּ כִּֽי־נִרְפִּ֣ים הֵ֔ם עַל־כֵּ֗ן הֵ֤ם צֹֽעֲקִים֙ לֵאמֹ֔ר נֵֽלְכָ֖ה נִזְבְּחָ֥ה לֵֽאלֹהֵֽינוּ׃ וְיָת סְכוֹם לִבְנַיָּא דְּאִנּוּן עָבְדִין מֵאֶתְמָלִי וּמִדְּקַמּוֹהִי, תְּמַנּוֹן עֲלֵיהוֹן--לָא תִּמְנְעוּן, מִנֵּיהּ:  אֲרֵי בַּטְלָנִין אִנּוּן--עַל כֵּין אִנּוּן מַצְוְחִין לְמֵימַר, נֵיזֵיל נְדַבַּח קֳדָם אֱלָהַנָא. 
ה,ט תִּכְבַּ֧ד הָֽעֲבֹדָ֛ה עַל־הָֽאֲנָשִׁ֖ים וְיַֽעֲשׂוּ־בָ֑הּ וְאַל־יִשְׁע֖וּ בְּדִבְרֵי־שָֽׁקֶר׃ יִתְקַף פֻּלְחָנָא עַל גֻּבְרַיָּא, וְיִתְעַסְּקוּן בַּהּ; וְלָא יִתְעַסְּקוּן, בְּפִתְגָמִין בְּטֵילִין. 
ה,י וַיֵּ֨צְא֜וּ נֹֽגְשֵׂ֤י הָעָם֙ וְשֹׁ֣טְרָ֔יו וַיֹּֽאמְר֥וּ אֶל־הָעָ֖ם לֵאמֹ֑ר כֹּ֚ה אָמַ֣ר פַּרְעֹ֔ה אֵינֶ֛נִּי נֹתֵ֥ן לָכֶ֖ם תֶּֽבֶן׃ וּנְפַקוּ שִׁלְטוֹנֵי עַמָּא, וְסָרְכוֹהִי, וַאֲמַרוּ לְעַמָּא, לְמֵימַר:  כִּדְנָן אֲמַר פַּרְעֹה, לֵית אֲנָא יָהֵיב לְכוֹן תִּבְנָא. 
ה,יא אַתֶּ֗ם לְכ֨וּ קְח֤וּ לָכֶם֙ תֶּ֔בֶן מֵֽאֲשֶׁ֖ר תִּמְצָ֑אוּ כִּ֣י אֵ֥ין נִגְרָ֛ע מֵעֲבֹֽדַתְכֶ֖ם דָּבָֽר׃ אַתּוּן, אִיזִילוּ סַבוּ לְכוֹן תִּבְנָא, מֵאֲתַר, דְּתַשְׁכְּחוּן:  אֲרֵי לָא יִתְמְנַע מִפֻּלְחָנְכוֹן, מִדָּעַם. 
ה,יב וַיָּ֥פֶץ הָעָ֖ם בְּכָל־אֶ֣רֶץ מִצְרָ֑יִם לְקֹשֵׁ֥שׁ קַ֖שׁ לַתֶּֽבֶן׃ וְאִתְבַּדַּר עַמָּא, בְּכָל אַרְעָא דְּמִצְרָיִם, לְגָבָבָא גִּלֵּי, לְתִבְנָא. 
ה,יג וְהַנֹּֽגְשִׂ֖ים אָצִ֣ים לֵאמֹ֑ר כַּלּ֤וּ מַֽעֲשֵׂיכֶם֙ דְּבַר־י֣וֹם בְּיוֹמ֔וֹ כַּֽאֲשֶׁ֖ר בִּֽהְי֥וֹת הַתֶּֽבֶן׃ וְשִׁלְטוֹנַיָּא, דָּחֲקִין לְמֵימַר:  אַשְׁלִימוּ עֲבִידַתְכוֹן פִּתְגָּם יוֹם בְּיוֹמֵיהּ, כְּמָא דַּהֲוֵיתוֹן עָבְדִין כַּד מִתְיְהֵיב לְכוֹן תִּבְנָא. 
ה,יד וַיֻּכּ֗וּ שֹֽׁטְרֵי֙ בְּנֵ֣י יִשְׂרָאֵ֔ל אֲשֶׁר־שָׂ֣מוּ עֲלֵהֶ֔ם נֹֽגְשֵׂ֥י פַרְעֹ֖ה לֵאמֹ֑ר מַדּ֡וּעַ לֹא֩ כִלִּיתֶ֨ם חָקְכֶ֤ם לִלְבֹּן֙ כִּתְמ֣וֹל שִׁלְשֹׁ֔ם גַּם־תְּמ֖וֹל גַּם־הַיּֽוֹם׃ וּלְקוֹ, סָרְכֵי בְּנֵי יִשְׂרָאֵל, דְּמַנִּיאוּ עֲלֵיהוֹן, שִׁלְטוֹנֵי פַּרְעֹה לְמֵימַר:  מָדֵין לָא אַשְׁלֵימְתּוּן גְּזֵירַתְכוֹן לְמִרְמֵי לִבְנִין, כְּמֵאֶתְמָלִי וּמִדְּקַמּוֹהִי--אַף תְּמָלִי, אַף יוֹמָא דֵּין. 
ה,טו וַיָּבֹ֗אוּ שֹֽׁטְרֵי֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וַיִּצְעֲק֥וּ אֶל־פַּרְעֹ֖ה לֵאמֹ֑ר לָ֧מָּה תַֽעֲשֶׂ֦ה כֹ֖ה לַֽעֲבָדֶֽיךָ׃ וַאֲתוֹ, סָרְכֵי בְּנֵי יִשְׂרָאֵל, וּצְוַחוּ קֳדָם פַּרְעֹה, לְמֵימַר:  לְמָא מִתְעֲבֵיד כְּדֵין, לְעַבְדָךְ. 
ה,טז תֶּ֗בֶן אֵ֤ין נִתָּן֙ לַֽעֲבָדֶ֔יךָ וּלְבֵנִ֛ים אֹֽמְרִ֥ים לָ֖נוּ עֲשׂ֑וּ וְהִנֵּ֧ה עֲבָדֶ֛יךָ מֻכִּ֖ים וְחָטָ֥את עַמֶּֽךָ׃ תִּבְנָא, לָא מִתְיְהֵיב לְעַבְדָךְ, וְלִבְנַיָּא אָמְרִין לַנָא, עֲבִידוּ; וְהָא עַבְדָךְ לָקַן, וְחָטַן עֲלֵיהוֹן עַמָּךְ. 
ה,יז וַיֹּ֛אמֶר נִרְפִּ֥ים אַתֶּ֖ם נִרְפִּ֑ים עַל־כֵּן֙ אַתֶּ֣ם אֹֽמְרִ֔ים נֵֽלְכָ֖ה נִזְבְּחָ֥ה לַֽיהוָֽה׃ וַאֲמַר בַּטְלָנִין אַתּוּן, בַּטְלָנִין; עַל כֵּין אַתּוּן אָמְרִין, נֵיזֵיל נְדַבַּח קֳדָם יְיָ. 
ה,יח וְעַתָּה֙ לְכ֣וּ עִבְד֔וּ וְתֶ֖בֶן לֹֽא־יִנָּתֵ֣ן לָכֶ֑ם וְתֹ֥כֶן לְבֵנִ֖ים תִּתֵּֽנוּ׃ וּכְעַן אִיזִילוּ פְּלַחוּ, וְתִבְנָא לָא יִתְיְהֵיב לְכוֹן; וּסְכוֹם לִבְנַיָּא, תִּתְּנוּן. 
ה,יט וַיִּרְא֞וּ שֹֽׁטְרֵ֧י בְנֵֽי־יִשְׂרָאֵ֛ל אֹתָ֖ם בְּרָ֣ע לֵאמֹ֑ר לֹֽא־תִגְרְע֥וּ מִלִּבְנֵיכֶ֖ם דְּבַר־י֥וֹם בְּיוֹמֽוֹ׃ וַחֲזוֹ סָרְכֵי בְּנֵי יִשְׂרָאֵל, יָתְהוֹן--בְּבִישׁ לְמֵימַר:  לָא תִּמְנְעוּן מִלִּבְנֵיכוֹן, פִּתְגָּם יוֹם בְּיוֹמֵיהּ. 
ה,כ וַֽיִּפְגְּעוּ֙ אֶת־מֹשֶׁ֣ה וְאֶֽת־אַהֲרֹ֔ן נִצָּבִ֖ים לִקְרָאתָ֑ם בְּצֵאתָ֖ם מֵאֵ֥ת פַּרְעֹֽה׃ וְעָרַעוּ יָת מֹשֶׁה וְיָת אַהֲרוֹן, קָיְמִין לְקַדָּמוּתְהוֹן, בְּמִפַּקְהוֹן, מִלְּוָת פַּרְעֹה. 
ה,כא וַיֹּֽאמְר֣וּ אֲלֵהֶ֔ם יֵ֧רֶא יְהוָ֛ה עֲלֵיכֶ֖ם וְיִשְׁפֹּ֑ט אֲשֶׁ֧ר הִבְאַשְׁתֶּ֣ם אֶת־רֵיחֵ֗נוּ בְּעֵינֵ֤י פַרְעֹה֙ וּבְעֵינֵ֣י עֲבָדָ֔יו לָֽתֶת־חֶ֥רֶב בְּיָדָ֖ם לְהָרְגֵֽנוּ׃ וַאֲמַרוּ לְהוֹן, יִתְגְּלֵי יְיָ עֲלֵיכוֹן וְיִתְפְּרַע:  דְּאַבְאֵישְׁתּוּן יָת רֵיחַנָא, בְּעֵינֵי פַּרְעֹה וּבְעֵינֵי עַבְדּוֹהִי, לְמִתַּן חַרְבָּא בְּיַדְהוֹן, לְמִקְטְלַנָא. 
ה,כב וַיָּ֧שָׁב מֹשֶׁ֛ה אֶל־יְהוָ֖ה וַיֹּאמַ֑ר אֲדֹנָ֗י לָמָ֤ה הֲרֵעֹ֨תָה֙ לָעָ֣ם הַזֶּ֔ה לָ֥מָּה זֶּ֖ה שְׁלַחְתָּֽנִי׃ וְתָב מֹשֶׁה לִקְדָם יְיָ, וַאֲמַר:  יְיָ, לְמָא אַבְאֵישְׁתָּא לְעַמָּא הָדֵין--לְמָא דְּנָן, שְׁלַחְתָּנִי. 
ה,כג וּמֵאָ֞ז בָּ֤אתִי אֶל־פַּרְעֹה֙ לְדַבֵּ֣ר בִּשְׁמֶ֔ךָ הֵרַ֖ע לָעָ֣ם הַזֶּ֑ה וְהַצֵּ֥ל לֹֽא־הִצַּ֖לְתָּ אֶת־עַמֶּֽךָ׃ וּמֵעִדָּן דְּעַאלִית לְוָת פַּרְעֹה, לְמַלָּלָא בִּשְׁמָךְ, אַבְאֵישׁ, לְעַמָּא הָדֵין; וְשֵׁיזָבָא לָא שֵׁיזֵיבְתָּא, יָת עַמָּךְ. 
ו,א וַיֹּ֤אמֶר יְהוָה֙ אֶל־מֹשֶׁ֔ה עַתָּ֣ה תִרְאֶ֔ה אֲשֶׁ֥ר אֶֽעֱשֶׂ֖ה לְפַרְעֹ֑ה כִּ֣י בְיָ֤ד חֲזָקָה֙ יְשַׁלְּחֵ֔ם וּבְיָ֣ד חֲזָקָ֔ה יְגָֽרְשֵׁ֖ם מֵֽאַרְצֽוֹ׃ {ס} וַאֲמַר יְיָ, לְמֹשֶׁה, כְּעַן תִּחְזֵי, דְּאַעֲבֵיד לְפַרְעֹה:  אֲרֵי בְּיַד תַּקִּיפָא, יְשַׁלְּחִנּוּן, וּבְיַד תַּקִּיפָא, יְתָרֵיכִנּוּן מֵאַרְעֵיהּ. {ס}
