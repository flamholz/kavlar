במדבר

א,א וַיְדַבֵּ֨ר יְהוָ֧ה אֶל־מֹשֶׁ֛ה בְּמִדְבַּ֥ר סִינַ֖י בְּאֹ֣הֶל מוֹעֵ֑ד בְּאֶחָד֩ לַחֹ֨דֶשׁ הַשֵּׁנִ֜י בַּשָּׁנָ֣ה הַשֵּׁנִ֗ית לְצֵאתָ֛ם מֵאֶ֥רֶץ מִצְרַ֖יִם לֵאמֹֽר׃ וּמַלֵּיל יְיָ עִם מֹשֶׁה בְּמַדְבְּרָא דְּסִינַי, בְּמַשְׁכַּן זִמְנָא:  בְּחַד לְיַרְחָא תִּנְיָנָא בְּשַׁתָּא תִּנְיֵיתָא, לְמִפַּקְהוֹן מֵאַרְעָא דְּמִצְרַיִם--לְמֵימַר. 
א,ב שְׂא֗וּ אֶת־רֹאשׁ֙ כָּל־עֲדַ֣ת בְּנֵֽי־יִשְׂרָאֵ֔ל לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֣ר שֵׁמ֔וֹת כָּל־זָכָ֖ר לְגֻלְגְּלֹתָֽם׃ קַבִּילוּ, יָת חֻשְׁבַּן כָּל כְּנִשְׁתָּא דִּבְנֵי יִשְׂרָאֵל, לְזַרְעֲיָתְהוֹן, לְבֵית אֲבָהָתְהוֹן--בְּמִנְיַן שְׁמָהָן, כָּל דְּכוּרָא לְגֻלְגְּלָתְהוֹן. 
א,ג מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כָּל־יֹצֵ֥א צָבָ֖א בְּיִשְׂרָאֵ֑ל תִּפְקְד֥וּ אֹתָ֛ם לְצִבְאֹתָ֖ם אַתָּ֥ה וְאַֽהֲרֹֽן׃ מִבַּר עַסְרִין שְׁנִין וּלְעֵילָא, כָּל נָפֵיק חֵילָא בְּיִשְׂרָאֵל--תִּמְנוֹן יָתְהוֹן לְחֵילֵיהוֹן, אַתְּ וְאַהֲרוֹן. 
א,ד וְאִתְּכֶ֣ם יִֽהְי֔וּ אִ֥ישׁ אִ֖ישׁ לַמַּטֶּ֑ה אִ֛ישׁ רֹ֥אשׁ לְבֵית־אֲבֹתָ֖יו הֽוּא׃ וְעִמְּכוֹן יְהוֹן, גֻּבְרָא גֻּבְרָא לְשִׁבְטָא--גְּבַר רֵישׁ לְבֵית אֲבָהָתוֹהִי, הוּא. 
א,ה וְאֵ֨לֶּה֙ שְׁמ֣וֹת הָֽאֲנָשִׁ֔ים אֲשֶׁ֥ר יַֽעַמְד֖וּ אִתְּכֶ֑ם לִרְאוּבֵ֕ן אֱלִיצ֖וּר בֶּן־שְׁדֵיאֽוּר׃ וְאִלֵּין שְׁמָהָת גֻּבְרַיָּא, דִּיקוּמוּן עִמְּכוֹן:  לִרְאוּבֵן, אֱלִיצוּר בַּר שְׁדֵיאוּר. 
א,ו לְשִׁמְע֕וֹן שְׁלֻֽמִיאֵ֖ל בֶּן־צוּרִֽישַׁדָּֽי׃ לְשִׁמְעוֹן, שְׁלוּמִיאֵל בַּר צוּרִישַׁדָּי. 
א,ז לִֽיהוּדָ֕ה נַחְשׁ֖וֹן בֶּן־עַמִּֽינָדָֽב׃ לִיהוּדָה, נַחְשׁוֹן בַּר עַמִּינָדָב. 
א,ח לְיִ֨שָּׂשכָ֔ר נְתַנְאֵ֖ל בֶּן־צוּעָֽר׃ לְיִשָּׂשכָר, נְתַנְאֵל בַּר צוּעָר. 
א,ט לִזְבוּלֻ֕ן אֱלִיאָ֖ב בֶּן־חֵלֹֽן׃ לִזְבוּלוּן, אֱלִיאָב בַּר חֵילוֹן. 
א,י לִבְנֵ֣י יוֹסֵ֔ף לְאֶפְרַ֕יִם אֱלִֽישָׁמָ֖ע בֶּן־עַמִּיה֑וּד לִמְנַשֶּׁ֕ה גַּמְלִיאֵ֖ל בֶּן־פְּדָהצֽוּר׃ לִבְנֵי יוֹסֵף--לְאֶפְרַיִם, אֱלִישָׁמָע בַּר עַמִּיהוּד; לִמְנַשֶּׁה, גַּמְלִיאֵל בַּר פְּדָהצוּר. 
א,יא לְבִ֨נְיָמִ֔ן אֲבִידָ֖ן בֶּן־גִּדְעֹנִֽי׃ לְבִנְיָמִין, אֲבִידָן בַּר גִּדְעוֹנִי. 
א,יב לְדָ֕ן אֲחִיעֶ֖זֶר בֶּן־עַמִּֽישַׁדָּֽי׃ לְדָן, אֲחִיעֶזֶר בַּר עַמִּישַׁדָּי. 
א,יג לְאָשֵׁ֕ר פַּגְעִיאֵ֖ל בֶּן־עָכְרָֽן׃ לְאָשֵׁר, פַּגְעִיאֵל בַּר עָכְרָן. 
א,יד לְגָ֕ד אֶלְיָסָ֖ף בֶּן־דְּעוּאֵֽל׃ לְגָד, אֶלְיָסָף בַּר דְּעוּאֵל. 
א,טו לְנַ֨פְתָּלִ֔י אֲחִירַ֖ע בֶּן־עֵינָֽן׃ לְנַפְתָּלִי, אֲחִירַע בַּר עֵינָן. 
א,טז אֵ֚לֶּה קריאי (קְרוּאֵ֣י) הָֽעֵדָ֔ה נְשִׂיאֵ֖י מַטּ֣וֹת אֲבוֹתָ֑ם רָאשֵׁ֛י אַלְפֵ֥י יִשְׂרָאֵ֖ל הֵֽם׃ אִלֵּין מְעָרְעֵי כְּנִשְׁתָּא, רַבְרְבֵי שִׁבְטֵי אֲבָהָתְהוֹן:  רֵישֵׁי אַלְפַיָּא דְּיִשְׂרָאֵל, אִנּוּן. 
א,יז וַיִּקַּ֥ח מֹשֶׁ֖ה וְאַֽהֲרֹ֑ן אֵ֚ת הָֽאֲנָשִׁ֣ים הָאֵ֔לֶּה אֲשֶׁ֥ר נִקְּב֖וּ בְּשֵׁמֹֽת׃ וּדְבַר מֹשֶׁה וְאַהֲרוֹן, יָת גֻּבְרַיָּא הָאִלֵּין, דְּאִתְפָּרַשׁוּ, בִּשְׁמָהָן. 
א,יח וְאֵ֨ת כָּל־הָֽעֵדָ֜ה הִקְהִ֗ילוּ בְּאֶחָד֙ לַחֹ֣דֶשׁ הַשֵּׁנִ֔י וַיִּתְיַֽלְד֥וּ עַל־מִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֣ר שֵׁמ֗וֹת מִבֶּ֨ן עֶשְׂרִ֥ים שָׁנָ֛ה וָמַ֖עְלָה לְגֻלְגְּלֹתָֽם׃ וְיָת כָּל כְּנִשְׁתָּא כְּנַשׁוּ, בְּחַד לְיַרְחָא תִּנְיָנָא, וְאִתְיַחַסוּ עַל זַרְעֲיָתְהוֹן, לְבֵית אֲבָהָתְהוֹן:  בְּמִנְיַן שְׁמָהָן, מִבַּר עַסְרִין שְׁנִין וּלְעֵילָא--לְגֻלְגְּלָתְהוֹן. 
א,יט כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶׁ֑ה וַֽיִּפְקְדֵ֖ם בְּמִדְבַּ֥ר סִינָֽי׃ {ס} כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה; וּמְנָנוּן, בְּמַדְבְּרָא דְּסִינָי. {ס}
א,כ וַיִּֽהְי֤וּ בְנֵֽי־רְאוּבֵן֙ בְּכֹ֣ר יִשְׂרָאֵ֔ל תּֽוֹלְדֹתָ֥ם לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֤ר שֵׁמוֹת֙ לְגֻלְגְּלֹתָ֔ם כָּל־זָכָ֗ר מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כֹּ֖ל יֹצֵ֥א צָבָֽא׃ וַהֲווֹ בְּנֵי רְאוּבֵן בֻּכְרֵיהּ דְּיִשְׂרָאֵל, תּוֹלְדָתְהוֹן לְזַרְעֲיָתְהוֹן לְבֵית אֲבָהָתְהוֹן:  בְּמִנְיַן שְׁמָהָן, לְגֻלְגְּלָתְהוֹן--כָּל דְּכוּרָא מִבַּר עַסְרִין שְׁנִין וּלְעֵילָא, כֹּל נָפֵיק חֵילָא. 
א,כא פְּקֻֽדֵיהֶ֖ם לְמַטֵּ֣ה רְאוּבֵ֑ן שִׁשָּׁ֧ה וְאַרְבָּעִ֛ים אֶ֖לֶף וַֽחֲמֵ֥שׁ מֵאֽוֹת׃ {פ} מִנְיָנֵיהוֹן, לְשִׁבְטָא דִּרְאוּבֵן--אַרְבְּעִין וְשִׁתָּא אַלְפִין, וַחֲמֵישׁ מְאָה. {פ}
א,כב לִבְנֵ֣י שִׁמְע֔וֹן תּֽוֹלְדֹתָ֥ם לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם פְּקֻדָ֗יו בְּמִסְפַּ֤ר שֵׁמוֹת֙ לְגֻלְגְּלֹתָ֔ם כָּל־זָכָ֗ר מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כֹּ֖ל יֹצֵ֥א צָבָֽא׃ לִבְנֵי שִׁמְעוֹן, תּוֹלְדָתְהוֹן לְזַרְעֲיָתְהוֹן לְבֵית אֲבָהָתְהוֹן:  מִנְיָנוֹהִי, בְּמִנְיַן שְׁמָהָן לְגֻלְגְּלָתְהוֹן--כָּל דְּכוּרָא מִבַּר עַסְרִין שְׁנִין וּלְעֵילָא, כֹּל נָפֵיק חֵילָא. 
א,כג פְּקֻֽדֵיהֶ֖ם לְמַטֵּ֣ה שִׁמְע֑וֹן תִּשְׁעָ֧ה וַֽחֲמִשִּׁ֛ים אֶ֖לֶף וּשְׁלֹ֥שׁ מֵאֽוֹת׃ {פ} מִנְיָנֵיהוֹן, לְשִׁבְטָא דְּשִׁמְעוֹן--חַמְשִׁין וְתִשְׁעָא אַלְפִין, וּתְלָת מְאָה. {פ}
א,כד לִבְנֵ֣י גָ֔ד תּֽוֹלְדֹתָ֥ם לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֣ר שֵׁמ֗וֹת מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כֹּ֖ל יֹצֵ֥א צָבָֽא׃ לִבְנֵי גָּד, תּוֹלְדָתְהוֹן לְזַרְעֲיָתְהוֹן לְבֵית אֲבָהָתְהוֹן:  בְּמִנְיַן שְׁמָהָן, מִבַּר עַסְרִין שְׁנִין וּלְעֵילָא--כֹּל, נָפֵיק חֵילָא. 
א,כה פְּקֻֽדֵיהֶ֖ם לְמַטֵּ֣ה גָ֑ד חֲמִשָּׁ֤ה וְאַרְבָּעִים֙ אֶ֔לֶף וְשֵׁ֥שׁ מֵא֖וֹת וַֽחֲמִשִּֽׁים׃ {פ} מִנְיָנֵיהוֹן, לְשִׁבְטָא דְּגָד--אַרְבְּעִין וְחַמְשָׁא אַלְפִין, וְשֵׁית מְאָה וְחַמְשִׁין. {פ}
א,כו לִבְנֵ֣י יְהוּדָ֔ה תּֽוֹלְדֹתָ֥ם לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֣ר שֵׁמֹ֗ת מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כֹּ֖ל יֹצֵ֥א צָבָֽא׃ לִבְנֵי יְהוּדָה, תּוֹלְדָתְהוֹן לְזַרְעֲיָתְהוֹן לְבֵית אֲבָהָתְהוֹן:  בְּמִנְיַן שְׁמָהָן, מִבַּר עַסְרִין שְׁנִין וּלְעֵילָא--כֹּל, נָפֵיק חֵילָא. 
א,כז פְּקֻֽדֵיהֶ֖ם לְמַטֵּ֣ה יְהוּדָ֑ה אַרְבָּעָ֧ה וְשִׁבְעִ֛ים אֶ֖לֶף וְשֵׁ֥שׁ מֵאֽוֹת׃ {פ} מִנְיָנֵיהוֹן, לְשִׁבְטָא דִּיהוּדָה--שִׁבְעִין וְאַרְבְּעָא אַלְפִין, וְשֵׁית מְאָה. {פ}
א,כח לִבְנֵ֣י יִשָּׂשכָ֔ר תּֽוֹלְדֹתָ֥ם לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֣ר שֵׁמֹ֗ת מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כֹּ֖ל יֹצֵ֥א צָבָֽא׃ לִבְנֵי יִשָּׂשכָר, תּוֹלְדָתְהוֹן לְזַרְעֲיָתְהוֹן לְבֵית אֲבָהָתְהוֹן:  בְּמִנְיַן שְׁמָהָן, מִבַּר עַסְרִין שְׁנִין וּלְעֵילָא--כֹּל, נָפֵיק חֵילָא. 
א,כט פְּקֻֽדֵיהֶ֖ם לְמַטֵּ֣ה יִשָּׂשכָ֑ר אַרְבָּעָ֧ה וַֽחֲמִשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת׃ {פ} מִנְיָנֵיהוֹן, לְשִׁבְטָא דְּיִשָּׂשכָר--חַמְשִׁין וְאַרְבְּעָא אַלְפִין, וְאַרְבַּע מְאָה. {פ}
א,ל לִבְנֵ֣י זְבוּלֻ֔ן תּֽוֹלְדֹתָ֥ם לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֣ר שֵׁמֹ֗ת מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כֹּ֖ל יֹצֵ֥א צָבָֽא׃ לִבְנֵי זְבוּלוּן, תּוֹלְדָתְהוֹן לְזַרְעֲיָתְהוֹן לְבֵית אֲבָהָתְהוֹן:  בְּמִנְיַן שְׁמָהָן, מִבַּר עַסְרִין שְׁנִין וּלְעֵילָא--כֹּל, נָפֵיק חֵילָא. 
א,לא פְּקֻֽדֵיהֶ֖ם לְמַטֵּ֣ה זְבוּלֻ֑ן שִׁבְעָ֧ה וַֽחֲמִשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת׃ {פ} מִנְיָנֵיהוֹן, לְשִׁבְטָא דִּזְבוּלוּן--חַמְשִׁין וְשִׁבְעָא אַלְפִין, וְאַרְבַּע מְאָה. {פ}
א,לב לִבְנֵ֤י יוֹסֵף֙ לִבְנֵ֣י אֶפְרַ֔יִם תּֽוֹלְדֹתָ֥ם לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֣ר שֵׁמֹ֗ת מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כֹּ֖ל יֹצֵ֥א צָבָֽא׃ לִבְנֵי יוֹסֵף לִבְנֵי אֶפְרַיִם, תּוֹלְדָתְהוֹן לְזַרְעֲיָתְהוֹן לְבֵית אֲבָהָתְהוֹן:  בְּמִנְיַן שְׁמָהָן, מִבַּר עַסְרִין שְׁנִין וּלְעֵילָא--כֹּל, נָפֵיק חֵילָא. 
א,לג פְּקֻֽדֵיהֶ֖ם לְמַטֵּ֣ה אֶפְרָ֑יִם אַרְבָּעִ֥ים אֶ֖לֶף וַֽחֲמֵ֥שׁ מֵאֽוֹת׃ {פ} מִנְיָנֵיהוֹן, לְשִׁבְטָא דְּאֶפְרָיִם--אַרְבְּעִין אַלְפִין, וַחֲמֵישׁ מְאָה. {פ}
א,לד לִבְנֵ֣י מְנַשֶּׁ֔ה תּֽוֹלְדֹתָ֥ם לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֣ר שֵׁמ֗וֹת מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כֹּ֖ל יֹצֵ֥א צָבָֽא׃ לִבְנֵי מְנַשֶּׁה, תּוֹלְדָתְהוֹן לְזַרְעֲיָתְהוֹן לְבֵית אֲבָהָתְהוֹן:  בְּמִנְיַן שְׁמָהָן, מִבַּר עַסְרִין שְׁנִין וּלְעֵילָא--כֹּל, נָפֵיק חֵילָא. 
א,לה פְּקֻֽדֵיהֶ֖ם לְמַטֵּ֣ה מְנַשֶּׁ֑ה שְׁנַ֧יִם וּשְׁלֹשִׁ֛ים אֶ֖לֶף וּמָאתָֽיִם׃ {פ} מִנְיָנֵיהוֹן, לְשִׁבְטָא דִּמְנַשֶּׁה--תְּלָתִין וּתְרֵין אַלְפִין, וּמָאתַן. {פ}
א,לו לִבְנֵ֣י בִנְיָמִ֔ן תּֽוֹלְדֹתָ֥ם לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֣ר שֵׁמֹ֗ת מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כֹּ֖ל יֹצֵ֥א צָבָֽא׃ לִבְנֵי בִּנְיָמִין, תּוֹלְדָתְהוֹן לְזַרְעֲיָתְהוֹן לְבֵית אֲבָהָתְהוֹן:  בְּמִנְיַן שְׁמָהָן, מִבַּר עַסְרִין שְׁנִין וּלְעֵילָא--כֹּל, נָפֵיק חֵילָא. 
א,לז פְּקֻֽדֵיהֶ֖ם לְמַטֵּ֣ה בִנְיָמִ֑ן חֲמִשָּׁ֧ה וּשְׁלֹשִׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת׃ {פ} מִנְיָנֵיהוֹן, לְשִׁבְטָא דְּבִנְיָמִין--תְּלָתִין וְחַמְשָׁא אַלְפִין, וְאַרְבַּע מְאָה. {פ}
א,לח לִבְנֵ֣י דָ֔ן תּֽוֹלְדֹתָ֥ם לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֣ר שֵׁמֹ֗ת מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כֹּ֖ל יֹצֵ֥א צָבָֽא׃ לִבְנֵי דָּן, תּוֹלְדָתְהוֹן לְזַרְעֲיָתְהוֹן לְבֵית אֲבָהָתְהוֹן:  בְּמִנְיַן שְׁמָהָן, מִבַּר עַסְרִין שְׁנִין וּלְעֵילָא--כֹּל, נָפֵיק חֵילָא. 
א,לט פְּקֻֽדֵיהֶ֖ם לְמַטֵּ֣ה דָ֑ן שְׁנַ֧יִם וְשִׁשִּׁ֛ים אֶ֖לֶף וּשְׁבַ֥ע מֵאֽוֹת׃ {פ} מִנְיָנֵיהוֹן, לְשִׁבְטָא דְּדָן--שִׁתִּין וּתְרֵין אַלְפִין, וּשְׁבַע מְאָה. {פ}
א,מ לִבְנֵ֣י אָשֵׁ֔ר תּֽוֹלְדֹתָ֥ם לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֣ר שֵׁמֹ֗ת מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כֹּ֖ל יֹצֵ֥א צָבָֽא׃ לִבְנֵי אָשֵׁר, תּוֹלְדָתְהוֹן לְזַרְעֲיָתְהוֹן לְבֵית אֲבָהָתְהוֹן:  בְּמִנְיַן שְׁמָהָן, מִבַּר עַסְרִין שְׁנִין וּלְעֵילָא--כֹּל, נָפֵיק חֵילָא. 
א,מא פְּקֻֽדֵיהֶ֖ם לְמַטֵּ֣ה אָשֵׁ֑ר אֶחָ֧ד וְאַרְבָּעִ֛ים אֶ֖לֶף וַֽחֲמֵ֥שׁ מֵאֽוֹת׃ {פ} מִנְיָנֵיהוֹן, לְשִׁבְטָא דְּאָשֵׁר--אַרְבְּעִין וְחַד אַלְפִין, וַחֲמֵישׁ מְאָה. {פ}
א,מב בְּנֵ֣י נַפְתָּלִ֔י תּֽוֹלְדֹתָ֥ם לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם בְּמִסְפַּ֣ר שֵׁמֹ֗ת מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כֹּ֖ל יֹצֵ֥א צָבָֽא׃ בְּנֵי נַפְתָּלִי, תּוֹלְדָתְהוֹן לְזַרְעֲיָתְהוֹן לְבֵית אֲבָהָתְהוֹן:  בְּמִנְיַן שְׁמָהָן, מִבַּר עַסְרִין שְׁנִין וּלְעֵילָא--כֹּל, נָפֵיק חֵילָא. 
א,מג פְּקֻֽדֵיהֶ֖ם לְמַטֵּ֣ה נַפְתָּלִ֑י שְׁלֹשָׁ֧ה וַֽחֲמִשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת׃ {פ} מִנְיָנֵיהוֹן, לְשִׁבְטָא דְּנַפְתָּלִי--חַמְשִׁין וּתְלָתָא אַלְפִין, וְאַרְבַּע מְאָה. {פ}
א,מד אֵ֣לֶּה הַפְּקֻדִ֡ים אֲשֶׁר֩ פָּקַ֨ד מֹשֶׁ֤ה וְאַֽהֲרֹן֙ וּנְשִׂיאֵ֣י יִשְׂרָאֵ֔ל שְׁנֵ֥ים עָשָׂ֖ר אִ֑ישׁ אִישׁ־אֶחָ֥ד לְבֵית־אֲבֹתָ֖יו הָיֽוּ׃ אִלֵּין מִנְיָנַיָּא דִּמְנָא מֹשֶׁה וְאַהֲרוֹן, וְרַבְרְבֵי יִשְׂרָאֵל--תְּרֵי עֲסַר, גֻּבְרִין:  גֻּבְרָא חַד לְבֵית אֲבָהָתוֹהִי, הֲווֹ. 
א,מה וַיִּֽהְי֛וּ כָּל־פְּקוּדֵ֥י בְנֵֽי־יִשְׂרָאֵ֖ל לְבֵ֣ית אֲבֹתָ֑ם מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה כָּל־יֹצֵ֥א צָבָ֖א בְּיִשְׂרָאֵֽל׃ וַהֲווֹ כָּל מִנְיָנֵי בְּנֵי יִשְׂרָאֵל, לְבֵית אֲבָהָתְהוֹן, מִבַּר עַסְרִין שְׁנִין וּלְעֵילָא, כָּל נָפֵיק חֵילָא בְּיִשְׂרָאֵל. 
א,מו וַיִּֽהְיוּ֙ כָּל־הַפְּקֻדִ֔ים שֵׁשׁ־מֵא֥וֹת אֶ֖לֶף וּשְׁלֹ֣שֶׁת אֲלָפִ֑ים וַֽחֲמֵ֥שׁ מֵא֖וֹת וַֽחֲמִשִּֽׁים׃ וַהֲווֹ, כָּל מִנְיָנַיָּא--שֵׁית מְאָה, וּתְלָתָא אַלְפִין; וַחֲמֵישׁ מְאָה, וְחַמְשִׁין. 
א,מז וְהַלְוִיִּ֖ם לְמַטֵּ֣ה אֲבֹתָ֑ם לֹ֥א הָתְפָּֽקְד֖וּ בְּתוֹכָֽם׃ {פ} וְלֵיוָאֵי, לְשִׁבְטָא דַּאֲבָהָתְהוֹן--לָא אִתְמְנִיאוּ, בֵּינֵיהוֹן. {פ}
א,מח וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
א,מט אַ֣ךְ אֶת־מַטֵּ֤ה לֵוִי֙ לֹ֣א תִפְקֹ֔ד וְאֶת־רֹאשָׁ֖ם לֹ֣א תִשָּׂ֑א בְּת֖וֹךְ בְּנֵ֥י יִשְׂרָאֵֽל׃ בְּרַם יָת שִׁבְטָא דְּלֵוִי לָא תִּמְנֵי, וְיָת חֻשְׁבָּנְהוֹן לָא תְּקַבֵּיל, בְּגוֹ, בְּנֵי יִשְׂרָאֵל. 
א,נ וְאַתָּ֡ה הַפְקֵ֣ד אֶת־הַלְוִיִּם֩ עַל־מִשְׁכַּ֨ן הָֽעֵדֻ֜ת וְעַ֣ל כָּל־כֵּלָיו֮ וְעַ֣ל כָּל־אֲשֶׁר־לוֹ֒ הֵ֜מָּה יִשְׂא֤וּ אֶת־הַמִּשְׁכָּן֙ וְאֶת־כָּל־כֵּלָ֔יו וְהֵ֖ם יְשָֽׁרְתֻ֑הוּ וְסָבִ֥יב לַמִּשְׁכָּ֖ן יַֽחֲנֽוּ׃ וְאַתְּ מַנִּי יָת לֵיוָאֵי עַל מַשְׁכְּנָא דְּסָהֲדוּתָא וְעַל כָּל מָנוֹהִי, וְעַל כָּל דְּלֵיהּ--אִנּוּן יִטְּלוּן יָת מַשְׁכְּנָא וְיָת כָּל מָנוֹהִי, וְאִנּוּן יְשַׁמְּשׁוּנֵּיהּ; וּסְחוֹר סְחוֹר לְמַשְׁכְּנָא, יִשְׁרוֹן. 
א,נא וּבִנְסֹ֣עַ הַמִּשְׁכָּ֗ן יוֹרִ֤ידוּ אֹתוֹ֙ הַלְוִיִּ֔ם וּבַֽחֲנֹת֙ הַמִּשְׁכָּ֔ן יָקִ֥ימוּ אֹת֖וֹ הַלְוִיִּ֑ם וְהַזָּ֥ר הַקָּרֵ֖ב יוּמָֽת׃ וּבְמִטַּל מַשְׁכְּנָא, יְפָרְקוּן יָתֵיהּ לֵיוָאֵי, וּבְמִשְׁרֵי מַשְׁכְּנָא, יְקִימוּן יָתֵיהּ לֵיוָאֵי; וְחִילוֹנַי דְּיִקְרַב, יִתְקְטִיל. 
א,נב וְחָנ֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל אִ֧ישׁ עַֽל־מַחֲנֵ֛הוּ וְאִ֥ישׁ עַל־דִּגְל֖וֹ לְצִבְאֹתָֽם׃ וְשָׁרַן, בְּנֵי יִשְׂרָאֵל, גְּבַר עַל מִשְׁרוֹהִי וּגְבַר עַל טִקְסֵיהּ, לְחֵילֵיהוֹן. 
א,נג וְהַלְוִיִּ֞ם יַֽחֲנ֤וּ סָבִיב֙ לְמִשְׁכַּ֣ן הָֽעֵדֻ֔ת וְלֹֽא־יִהְיֶ֣ה קֶ֔צֶף עַל־עֲדַ֖ת בְּנֵ֣י יִשְׂרָאֵ֑ל וְשָֽׁמְרוּ֙ הַלְוִיִּ֔ם אֶת־מִשְׁמֶ֖רֶת מִשְׁכַּ֥ן הָֽעֵדֽוּת׃ וְלֵיוָאֵי יִשְׁרוֹן סְחוֹר סְחוֹר, לְמַשְׁכְּנָא דְּסָהֲדוּתָא, וְלָא יְהֵי רֻגְזָא, עַל כְּנִשְׁתָּא דִּבְנֵי יִשְׂרָאֵל; וְיִטְּרוּן, לֵיוָאֵי, יָת מַטְּרַת, מַשְׁכְּנָא דְּסָהֲדוּתָא. 
א,נד וַֽיַּעֲשׂ֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל כְּ֠כֹל אֲשֶׁ֨ר צִוָּ֧ה יְהוָ֛ה אֶת־מֹשֶׁ֖ה כֵּ֥ן עָשֽׂוּ׃ {פ} וַעֲבַדוּ, בְּנֵי יִשְׂרָאֵל:  כְּכֹל דְּפַקֵּיד יְיָ, יָת מֹשֶׁה--כֵּין עֲבַדוּ. {פ}
ב,א וַיְדַבֵּ֣ר יְהוָ֔ה אֶל־מֹשֶׁ֥ה וְאֶֽל־אַהֲרֹ֖ן לֵאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה וּלְאַהֲרוֹן לְמֵימַר. 
ב,ב אִ֣ישׁ עַל־דִּגְל֤וֹ בְאֹתֹת֙ לְבֵ֣ית אֲבֹתָ֔ם יַֽחֲנ֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל מִנֶּ֕גֶד סָבִ֥יב לְאֹֽהֶל־מוֹעֵ֖ד יַֽחֲנֽוּ׃ גְּבַר עַל טִקְסֵיהּ בְּאָתְוָן לְבֵית אֲבָהָתְהוֹן, יִשְׁרוֹן בְּנֵי יִשְׂרָאֵל:  מִקֳּבֵיל, סְחוֹר סְחוֹר לְמַשְׁכַּן זִמְנָא יִשְׁרוֹן. 
ב,ג וְהַֽחֹנִים֙ קֵ֣דְמָה מִזְרָ֔חָה דֶּ֛גֶל מַֽחֲנֵ֥ה יְהוּדָ֖ה לְצִבְאֹתָ֑ם וְנָשִׂיא֙ לִבְנֵ֣י יְהוּדָ֔ה נַחְשׁ֖וֹן בֶּן־עַמִּֽינָדָֽב׃ וְדִשְׁרַן קִדּוּמָא מַדְנְחָא, טֵיקַס מַשְׁרִית יְהוּדָה לְחֵילֵיהוֹן; וְרַבָּא לִבְנֵי יְהוּדָה, נַחְשׁוֹן בַּר עַמִּינָדָב. 
ב,ד וּצְבָא֖וֹ וּפְקֻֽדֵיהֶ֑ם אַרְבָּעָ֧ה וְשִׁבְעִ֛ים אֶ֖לֶף וְשֵׁ֥שׁ מֵאֽוֹת׃ וְחֵילֵיהּ, וּמִנְיָנֵיהוֹן--שִׁבְעִין וְאַרְבְּעָא אַלְפִין, וְשֵׁית מְאָה. 
ב,ה וְהַֽחֹנִ֥ים עָלָ֖יו מַטֵּ֣ה יִשָּׂשכָ֑ר וְנָשִׂיא֙ לִבְנֵ֣י יִשָּׂשכָ֔ר נְתַנְאֵ֖ל בֶּן־צוּעָֽר׃ וְדִשְׁרַן סְמִיכִין עֲלוֹהִי, שִׁבְטָא דְּיִשָּׂשכָר; וְרַבָּא לִבְנֵי יִשָּׂשכָר, נְתַנְאֵל בַּר צוּעָר. 
ב,ו וּצְבָא֖וֹ וּפְקֻדָ֑יו אַרְבָּעָ֧ה וַֽחֲמִשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת׃ וְחֵילֵיהּ, וּמִנְיָנוֹהִי--חַמְשִׁין וְאַרְבְּעָא אַלְפִין, וְאַרְבַּע מְאָה. 
ב,ז מַטֵּ֖ה זְבוּלֻ֑ן וְנָשִׂיא֙ לִבְנֵ֣י זְבוּלֻ֔ן אֱלִיאָ֖ב בֶּן־חֵלֹֽן׃ שִׁבְטָא, דִּזְבוּלוּן; וְרַבָּא לִבְנֵי זְבוּלוּן, אֱלִיאָב בַּר חֵילוֹן. 
ב,ח וּצְבָא֖וֹ וּפְקֻדָ֑יו שִׁבְעָ֧ה וַֽחֲמִשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת׃ וְחֵילֵיהּ, וּמִנְיָנוֹהִי--חַמְשִׁין וְשִׁבְעָא אַלְפִין, וְאַרְבַּע מְאָה. 
ב,ט כָּֽל־הַפְּקֻדִ֞ים לְמַֽחֲנֵ֣ה יְהוּדָ֗ה מְאַ֨ת אֶ֜לֶף וּשְׁמֹנִ֥ים אֶ֛לֶף וְשֵֽׁשֶׁת־אֲלָפִ֥ים וְאַרְבַּע־מֵא֖וֹת לְצִבְאֹתָ֑ם רִֽאשֹׁנָ֖ה יִסָּֽעוּ׃ {ס} כָּל מִנְיָנַיָּא לְמַשְׁרִית יְהוּדָה, מְאָה וּתְמָנַן וְשִׁתָּא אַלְפִין וְאַרְבַּע מְאָה--לְחֵילֵיהוֹן; בְּקַדְמֵיתָא, נָטְלִין. {ס}
ב,י דֶּ֣גֶל מַֽחֲנֵ֧ה רְאוּבֵ֛ן תֵּימָ֖נָה לְצִבְאֹתָ֑ם וְנָשִׂיא֙ לִבְנֵ֣י רְאוּבֵ֔ן אֱלִיצ֖וּר בֶּן־שְׁדֵיאֽוּר׃ טֵיקַס מַשְׁרִית רְאוּבֵן דָּרוֹמָא, לְחֵילֵיהוֹן; וְרַבָּא לִבְנֵי רְאוּבֵן, אֱלִיצוּר בַּר שְׁדֵיאוּר. 
ב,יא וּצְבָא֖וֹ וּפְקֻדָ֑יו שִׁשָּׁ֧ה וְאַרְבָּעִ֛ים אֶ֖לֶף וַֽחֲמֵ֥שׁ מֵאֽוֹת׃ וְחֵילֵיהּ, וּמִנְיָנוֹהִי--אַרְבְּעִין וְשִׁתָּא אַלְפִין, וַחֲמֵישׁ מְאָה. 
ב,יב וְהַֽחוֹנִ֥ם עָלָ֖יו מַטֵּ֣ה שִׁמְע֑וֹן וְנָשִׂיא֙ לִבְנֵ֣י שִׁמְע֔וֹן שְׁלֻֽמִיאֵ֖ל בֶּן־צוּרִֽישַׁדָּֽי׃ וְדִשְׁרַן סְמִיכִין עֲלוֹהִי, שִׁבְטָא דְּשִׁמְעוֹן; וְרַבָּא לִבְנֵי שִׁמְעוֹן, שְׁלוּמִיאֵל בַּר צוּרִישַׁדָּי. 
ב,יג וּצְבָא֖וֹ וּפְקֻֽדֵיהֶ֑ם תִּשְׁעָ֧ה וַֽחֲמִשִּׁ֛ים אֶ֖לֶף וּשְׁלֹ֥שׁ מֵאֽוֹת׃ וְחֵילֵיהּ, וּמִנְיָנֵיהוֹן--חַמְשִׁין וְתִשְׁעָא אַלְפִין, וּתְלָת מְאָה. 
ב,יד וּמַטֵּ֖ה גָּ֑ד וְנָשִׂיא֙ לִבְנֵ֣י גָ֔ד אֶלְיָסָ֖ף בֶּן־רְעוּאֵֽל׃ וְשִׁבְטָא, דְּגָד; וְרַבָּא לִבְנֵי גָּד, אֶלְיָסָף בַּר רְעוּאֵל. 
ב,טו וּצְבָא֖וֹ וּפְקֻֽדֵיהֶ֑ם חֲמִשָּׁ֤ה וְאַרְבָּעִים֙ אֶ֔לֶף וְשֵׁ֥שׁ מֵא֖וֹת וַֽחֲמִשִּֽׁים׃ וְחֵילֵיהּ, וּמִנְיָנֵיהוֹן--אַרְבְּעִין וְחַמְשָׁא אַלְפִין, וְשֵׁית מְאָה וְחַמְשִׁין. 
ב,טז כָּֽל־הַפְּקֻדִ֞ים לְמַֽחֲנֵ֣ה רְאוּבֵ֗ן מְאַ֨ת אֶ֜לֶף וְאֶחָ֨ד וַֽחֲמִשִּׁ֥ים אֶ֛לֶף וְאַרְבַּע־מֵא֥וֹת וַֽחֲמִשִּׁ֖ים לְצִבְאֹתָ֑ם וּשְׁנִיִּ֖ם יִסָּֽעוּ׃ {ס} כָּל מִנְיָנַיָּא לְמַשְׁרִית רְאוּבֵן, מְאָה וְחַמְשִׁין וְחַד אַלְפִין וְאַרְבַּע מְאָה וְחַמְשִׁין--לְחֵילֵיהוֹן; בְּתִנְיֵיתָא, נָטְלִין. {ס}
ב,יז וְנָסַ֧ע אֹֽהֶל־מוֹעֵ֛ד מַֽחֲנֵ֥ה הַלְוִיִּ֖ם בְּת֣וֹךְ הַֽמַּחֲנֹ֑ת כַּֽאֲשֶׁ֤ר יַֽחֲנוּ֙ כֵּ֣ן יִסָּ֔עוּ אִ֥ישׁ עַל־יָד֖וֹ לְדִגְלֵיהֶֽם׃ {ס} וְנָטֵיל מַשְׁכַּן זִמְנָא מַשְׁרִית לֵיוָאֵי, בְּגוֹ מַשְׁרְיָתָא; כְּמָא דְּשָׁרַן כֵּין נָטְלִין, גְּבַר עַל אַתְרֵיהּ לְטִקְסֵיהוֹן. {ס}
ב,יח דֶּ֣גֶל מַֽחֲנֵ֥ה אֶפְרַ֛יִם לְצִבְאֹתָ֖ם יָ֑מָּה וְנָשִׂיא֙ לִבְנֵ֣י אֶפְרַ֔יִם אֱלִֽישָׁמָ֖ע בֶּן־עַמִּיהֽוּד׃ טֵיקַס מַשְׁרִית אֶפְרַיִם לְחֵילֵיהוֹן, מַעְרְבָא; וְרַבָּא לִבְנֵי אֶפְרַיִם, אֱלִישָׁמָע בַּר עַמִּיהוּד. 
ב,יט וּצְבָא֖וֹ וּפְקֻֽדֵיהֶ֑ם אַרְבָּעִ֥ים אֶ֖לֶף וַֽחֲמֵ֥שׁ מֵאֽוֹת׃ וְחֵילֵיהּ, וּמִנְיָנֵיהוֹן--אַרְבְּעִין אַלְפִין, וַחֲמֵישׁ מְאָה. 
ב,כ וְעָלָ֖יו מַטֵּ֣ה מְנַשֶּׁ֑ה וְנָשִׂיא֙ לִבְנֵ֣י מְנַשֶּׁ֔ה גַּמְלִיאֵ֖ל בֶּן־פְּדָהצֽוּר׃ וְדִסְמִיכִין עֲלוֹהִי, שִׁבְטָא דִּמְנַשֶּׁה; וְרַבָּא לִבְנֵי מְנַשֶּׁה, גַּמְלִיאֵל בַּר פְּדָהצוּר. 
ב,כא וּצְבָא֖וֹ וּפְקֻֽדֵיהֶ֑ם שְׁנַ֧יִם וּשְׁלֹשִׁ֛ים אֶ֖לֶף וּמָאתָֽיִם׃ וְחֵילֵיהּ, וּמִנְיָנֵיהוֹן--תְּלָתִין וּתְרֵין אַלְפִין, וּמָאתַן. 
ב,כב וּמַטֵּ֖ה בִּנְיָמִ֑ן וְנָשִׂיא֙ לִבְנֵ֣י בִנְיָמִ֔ן אֲבִידָ֖ן בֶּן־גִּדְעֹנִֽי׃ וְשִׁבְטָא, דְּבִנְיָמִין; וְרַבָּא לִבְנֵי בִּנְיָמִין, אֲבִידָן בַּר גִּדְעוֹנִי. 
ב,כג וּצְבָא֖וֹ וּפְקֻֽדֵיהֶ֑ם חֲמִשָּׁ֧ה וּשְׁלֹשִׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת׃ וְחֵילֵיהּ, וּמִנְיָנֵיהוֹן--תְּלָתִין וְחַמְשָׁא אַלְפִין, וְאַרְבַּע מְאָה. 
ב,כד כָּֽל־הַפְּקֻדִ֞ים לְמַֽחֲנֵ֣ה אֶפְרַ֗יִם מְאַ֥ת אֶ֛לֶף וּשְׁמֹֽנַת־אֲלָפִ֥ים וּמֵאָ֖ה לְצִבְאֹתָ֑ם וּשְׁלִשִׁ֖ים יִסָּֽעוּ׃ {ס} כָּל מִנְיָנַיָּא לְמַשְׁרִית אֶפְרַיִם, מְאָה וּתְמָנְיָא אַלְפִין וּמְאָה--לְחֵילֵיהוֹן; בִּתְלִיתֵיתָא, נָטְלִין. {ס}
ב,כה דֶּ֣גֶל מַֽחֲנֵ֥ה דָ֛ן צָפֹ֖נָה לְצִבְאֹתָ֑ם וְנָשִׂיא֙ לִבְנֵ֣י דָ֔ן אֲחִיעֶ֖זֶר בֶּן־עַמִּֽישַׁדָּֽי׃ טֵיקַס מַשְׁרִית דָּן צִפּוּנָא, לְחֵילֵיהוֹן; וְרַבָּא לִבְנֵי דָּן, אֲחִיעֶזֶר בַּר עַמִּישַׁדָּי. 
ב,כו וּצְבָא֖וֹ וּפְקֻֽדֵיהֶ֑ם שְׁנַ֧יִם וְשִׁשִּׁ֛ים אֶ֖לֶף וּשְׁבַ֥ע מֵאֽוֹת׃ וְחֵילֵיהּ, וּמִנְיָנֵיהוֹן--שִׁתִּין וּתְרֵין אַלְפִין, וּשְׁבַע מְאָה. 
ב,כז וְהַֽחֹנִ֥ים עָלָ֖יו מַטֵּ֣ה אָשֵׁ֑ר וְנָשִׂיא֙ לִבְנֵ֣י אָשֵׁ֔ר פַּגְעִיאֵ֖ל בֶּן־עָכְרָֽן׃ וְדִשְׁרַן סְמִיכִין עֲלוֹהִי, שִׁבְטָא דְּאָשֵׁר; וְרַבָּא לִבְנֵי אָשֵׁר, פַּגְעִיאֵל בַּר עָכְרָן. 
ב,כח וּצְבָא֖וֹ וּפְקֻֽדֵיהֶ֑ם אֶחָ֧ד וְאַרְבָּעִ֛ים אֶ֖לֶף וַֽחֲמֵ֥שׁ מֵאֽוֹת׃ וְחֵילֵיהּ, וּמִנְיָנֵיהוֹן--אַרְבְּעִין וְחַד אַלְפִין, וַחֲמֵישׁ מְאָה. 
ב,כט וּמַטֵּ֖ה נַפְתָּלִ֑י וְנָשִׂיא֙ לִבְנֵ֣י נַפְתָּלִ֔י אֲחִירַ֖ע בֶּן־עֵינָֽן׃ וְשִׁבְטָא, דְּנַפְתָּלִי; וְרַבָּא לִבְנֵי נַפְתָּלִי, אֲחִירַע בַּר עֵינָן. 
ב,ל וּצְבָא֖וֹ וּפְקֻֽדֵיהֶ֑ם שְׁלֹשָׁ֧ה וַֽחֲמִשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת׃ וְחֵילֵיהּ, וּמִנְיָנֵיהוֹן--חַמְשִׁין וּתְלָתָא אַלְפִין, וְאַרְבַּע מְאָה. 
ב,לא כָּל־הַפְּקֻדִים֙ לְמַ֣חֲנֵה דָ֔ן מְאַ֣ת אֶ֗לֶף וְשִׁבְעָ֧ה וַֽחֲמִשִּׁ֛ים אֶ֖לֶף וְשֵׁ֣שׁ מֵא֑וֹת לָאַֽחֲרֹנָ֥ה יִסְע֖וּ לְדִגְלֵיהֶֽם׃ {פ} כָּל מִנְיָנַיָּא, לְמַשְׁרִית דָּן--מְאָה וְחַמְשִׁין וְשִׁבְעָא אַלְפִין, וְשֵׁית מְאָה; בְּבָתְרֵיתָא נָטְלִין, לְטִקְסֵיהוֹן. {פ}
ב,לב אֵ֛לֶּה פְּקוּדֵ֥י בְנֵֽי־יִשְׂרָאֵ֖ל לְבֵ֣ית אֲבֹתָ֑ם כָּל־פְּקוּדֵ֤י הַֽמַּחֲנֹת֙ לְצִבְאֹתָ֔ם שֵׁשׁ־מֵא֥וֹת אֶ֨לֶף֙ וּשְׁלֹ֣שֶׁת אֲלָפִ֔ים וַֽחֲמֵ֥שׁ מֵא֖וֹת וַֽחֲמִשִּֽׁים׃ אִלֵּין מִנְיָנֵי בְּנֵי יִשְׂרָאֵל, לְבֵית אֲבָהָתְהוֹן:  כָּל מִנְיָנֵי מַשְׁרְיָתָא, לְחֵילֵיהוֹן--שֵׁית מְאָה וּתְלָתָא אַלְפִין, וַחֲמֵישׁ מְאָה וְחַמְשִׁין. 
ב,לג וְהַ֨לְוִיִּ֔ם לֹ֣א הָתְפָּֽקְד֔וּ בְּת֖וֹךְ בְּנֵ֣י יִשְׂרָאֵ֑ל כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ וְלֵיוָאֵי--לָא אִתְמְנִיאוּ, בְּגוֹ בְּנֵי יִשְׂרָאֵל:  כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. 
ב,לד וַֽיַּעֲשׂ֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל כְּ֠כֹל אֲשֶׁר־צִוָּ֨ה יְהוָ֜ה אֶת־מֹשֶׁ֗ה כֵּֽן־חָנ֤וּ לְדִגְלֵיהֶם֙ וְכֵ֣ן נָסָ֔עוּ אִ֥ישׁ לְמִשְׁפְּחֹתָ֖יו עַל־בֵּ֥ית אֲבֹתָֽיו׃ {פ} וַעֲבַדוּ, בְּנֵי יִשְׂרָאֵל:  כְּכֹל דְּפַקֵּיד יְיָ יָת מֹשֶׁה, כֵּין שָׁרַן לְטִקְסֵיהוֹן וְכֵין נָטְלִין--גְּבַר לְזַרְעִיתֵיהּ, עַל בֵּית אֲבָהָתוֹהִי. {פ}
ג,א וְאֵ֛לֶּה תּֽוֹלְדֹ֥ת אַֽהֲרֹ֖ן וּמֹשֶׁ֑ה בְּי֗וֹם דִּבֶּ֧ר יְהוָ֛ה אֶת־מֹשֶׁ֖ה בְּהַ֥ר סִינָֽי׃ וְאִלֵּין תּוֹלְדָת אַהֲרוֹן, וּמֹשֶׁה:  בְּיוֹמָא, דְּמַלֵּיל יְיָ יָת מֹשֶׁה--בְּטוּרָא דְּסִינָי. 
ג,ב וְאֵ֛לֶּה שְׁמ֥וֹת בְּֽנֵי־אַהֲרֹ֖ן הַבְּכֹ֣ר ׀ נָדָ֑ב וַֽאֲבִיה֕וּא אֶלְעָזָ֖ר וְאִֽיתָמָֽר׃ וְאִלֵּין שְׁמָהָת בְּנֵי אַהֲרוֹן, בֻּכְרָא נָדָב, וַאֲבִיהוּא, אֶלְעָזָר וְאִיתָמָר. 
ג,ג אֵ֗לֶּה שְׁמוֹת֙ בְּנֵ֣י אַֽהֲרֹ֔ן הַכֹּֽהֲנִ֖ים הַמְּשֻׁחִ֑ים אֲשֶׁר־מִלֵּ֥א יָדָ֖ם לְכַהֵֽן׃ אִלֵּין, שְׁמָהָת בְּנֵי אַהֲרוֹן, כָּהֲנַיָּא, דְּאִתְרַבִּיאוּ--דְּאִתְקָרַב קֻרְבָּנְהוֹן, לְשַׁמָּשָׁא. 
ג,ד וַיָּ֣מָת נָדָ֣ב וַֽאֲבִיה֣וּא לִפְנֵ֣י יְהוָ֡ה בְּֽהַקְרִבָם֩ אֵ֨שׁ זָרָ֜ה לִפְנֵ֤י יְהוָה֙ בְּמִדְבַּ֣ר סִינַ֔י וּבָנִ֖ים לֹֽא־הָי֣וּ לָהֶ֑ם וַיְכַהֵ֤ן אֶלְעָזָר֙ וְאִ֣יתָמָ֔ר עַל־פְּנֵ֖י אַֽהֲרֹ֥ן אֲבִיהֶֽם׃ {פ} וּמִית נָדָב וַאֲבִיהוּא קֳדָם יְיָ בְּקָרוֹבֵיהוֹן אִישָׁתָא נֻכְרֵיתָא קֳדָם יְיָ, בְּמַדְבְּרָא דְּסִינַי, וּבְנִין, לָא הֲווֹ לְהוֹן; וְשַׁמֵּישׁ אֶלְעָזָר וְאִיתָמָר, עַל אַפֵּי אַהֲרוֹן אֲבוּהוֹן. {פ}
ג,ה וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ג,ו הַקְרֵב֙ אֶת־מַטֵּ֣ה לֵוִ֔י וְהַֽעֲמַדְתָּ֣ אֹת֔וֹ לִפְנֵ֖י אַֽהֲרֹ֣ן הַכֹּהֵ֑ן וְשֵֽׁרְת֖וּ אֹתֽוֹ׃ קָרֵיב, יָת שִׁבְטָא דְּלֵוִי, וּתְקִים יָתֵיהּ, קֳדָם אַהֲרוֹן כָּהֲנָא; וִישַׁמְּשׁוּן, יָתֵיהּ. 
ג,ז וְשָֽׁמְר֣וּ אֶת־מִשְׁמַרְתּ֗וֹ וְאֶת־מִשְׁמֶ֨רֶת֙ כָּל־הָ֣עֵדָ֔ה לִפְנֵ֖י אֹ֣הֶל מוֹעֵ֑ד לַֽעֲבֹ֖ד אֶת־עֲבֹדַ֥ת הַמִּשְׁכָּֽן׃ וְיִטְּרוּן יָת מַטַּרְתֵּיהּ, וְיָת מַטְּרַת כָּל כְּנִשְׁתָּא, קֳדָם, מַשְׁכַּן זִמְנָא--לְמִפְלַח, יָת פֻּלְחַן מַשְׁכְּנָא. 
ג,ח וְשָֽׁמְר֗וּ אֶֽת־כָּל־כְּלֵי֙ אֹ֣הֶל מוֹעֵ֔ד וְאֶת־מִשְׁמֶ֖רֶת בְּנֵ֣י יִשְׂרָאֵ֑ל לַֽעֲבֹ֖ד אֶת־עֲבֹדַ֥ת הַמִּשְׁכָּֽן׃ וְיִטְּרוּן, יָת כָּל מָנֵי מַשְׁכַּן זִמְנָא, וְיָת מַטְּרַת, בְּנֵי יִשְׂרָאֵל--לְמִפְלַח, יָת פֻּלְחַן מַשְׁכְּנָא. 
ג,ט וְנָֽתַתָּה֙ אֶת־הַלְוִיִּ֔ם לְאַֽהֲרֹ֖ן וּלְבָנָ֑יו נְתוּנִ֨ם נְתוּנִ֥ם הֵ֨מָּה֙ ל֔וֹ מֵאֵ֖ת בְּנֵ֥י יִשְׂרָאֵֽל׃ וְתִתֵּין, יָת לֵיוָאֵי, לְאַהֲרוֹן, וְלִבְנוֹהִי:  מְסִירִין יְהִיבִין אִנּוּן לֵיהּ, מִן בְּנֵי יִשְׂרָאֵל. 
ג,י וְאֶת־אַֽהֲרֹ֤ן וְאֶת־בָּנָיו֙ תִּפְקֹ֔ד וְשָֽׁמְר֖וּ אֶת־כְּהֻנָּתָ֑ם וְהַזָּ֥ר הַקָּרֵ֖ב יוּמָֽת׃ {פ} וְיָת אַהֲרוֹן וְיָת בְּנוֹהִי תְּמַנֵּי, וְיִטְּרוּן יָת כְּהֻנַּתְהוֹן; וְחִילוֹנַי דְּיִקְרַב, יִתְקְטִיל. {פ}
ג,יא וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ג,יב וַֽאֲנִ֞י הִנֵּ֧ה לָקַ֣חְתִּי אֶת־הַלְוִיִּ֗ם מִתּוֹךְ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל תַּ֧חַת כָּל־בְּכ֛וֹר פֶּ֥טֶר רֶ֖חֶם מִבְּנֵ֣י יִשְׂרָאֵ֑ל וְהָ֥יוּ לִ֖י הַלְוִיִּֽם׃ וַאֲנָא הָא קָרֵיבִית יָת לֵיוָאֵי, מִגּוֹ בְּנֵי יִשְׂרָאֵל, חֲלָף כָּל בֻּכְרָא פָּתַח וַלְדָּא, מִבְּנֵי יִשְׂרָאֵל; וִיהוֹן מְשַׁמְּשִׁין קֳדָמַי, לֵיוָאֵי. 
ג,יג כִּ֣י לִי֮ כָּל־בְּכוֹר֒ בְּיוֹם֩ הַכֹּתִ֨י כָל־בְּכ֜וֹר בְּאֶ֣רֶץ מִצְרַ֗יִם הִקְדַּ֨שְׁתִּי לִ֤י כָל־בְּכוֹר֙ בְּיִשְׂרָאֵ֔ל מֵֽאָדָ֖ם עַד־בְּהֵמָ֑ה לִ֥י יִֽהְי֖וּ אֲנִ֥י יְהוָֽה׃ {פ} אֲרֵי דִּילִי, כָּל בֻּכְרָא--בְּיוֹמָא דִּקְטַלִית כָּל בֻּכְרָא בְּאַרְעָא דְּמִצְרַיִם אַקְדֵּישִׁית קֳדָמַי כָּל בֻּכְרָא בְּיִשְׂרָאֵל, מֵאֲנָשָׁא עַד בְּעִירָא:  דִּילִי יְהוֹן, אֲנָא יְיָ. {פ}
ג,יד וַיְדַבֵּ֤ר יְהוָה֙ אֶל־מֹשֶׁ֔ה בְּמִדְבַּ֥ר סִינַ֖י לֵאמֹֽר׃ וּמַלֵּיל יְיָ עִם מֹשֶׁה, בְּמַדְבְּרָא דְּסִינַי לְמֵימַר. 
ג,טו פְּקֹד֙ אֶת־בְּנֵ֣י לֵוִ֔י לְבֵ֥ית אֲבֹתָ֖ם לְמִשְׁפְּחֹתָ֑ם כָּל־זָכָ֛ר מִבֶּן־חֹ֥דֶשׁ וָמַ֖עְלָה תִּפְקְדֵֽם׃ מְנִי יָת בְּנֵי לֵוִי, לְבֵית אֲבָהָתְהוֹן לְזַרְעֲיָתְהוֹן:  כָּל דְּכוּרָא מִבַּר יַרְחָא וּלְעֵילָא, תִּמְנֵינוּן. 
ג,טז וַיִּפְקֹ֥ד אֹתָ֛ם מֹשֶׁ֖ה עַל־פִּ֣י יְהוָ֑ה כַּֽאֲשֶׁ֖ר צֻוָּֽה׃ וּמְנָא יָתְהוֹן מֹשֶׁה, עַל מֵימְרָא דַּייָ--כְּמָא דְּאִתְפַּקַּד. 
ג,יז וַיִּֽהְיוּ־אֵ֥לֶּה בְנֵֽי־לֵוִ֖י בִּשְׁמֹתָ֑ם גֵּֽרְשׁ֕וֹן וּקְהָ֖ת וּמְרָרִֽי׃ וַהֲווֹ אִלֵּין בְּנֵי לֵוִי, בִּשְׁמָהָתְהוֹן--גֵּרְשׁוֹן, וּקְהָת וּמְרָרִי. 
ג,יח וְאֵ֛לֶּה שְׁמ֥וֹת בְּֽנֵי־גֵרְשׁ֖וֹן לְמִשְׁפְּחֹתָ֑ם לִבְנִ֖י וְשִׁמְעִֽי׃ וְאִלֵּין שְׁמָהָת בְּנֵי גֵּרְשׁוֹן, לְזַרְעֲיָתְהוֹן--לִבְנִי, וְשִׁמְעִי. 
ג,יט וּבְנֵ֥י קְהָ֖ת לְמִשְׁפְּחֹתָ֑ם עַמְרָ֣ם וְיִצְהָ֔ר חֶבְר֖וֹן וְעֻזִּיאֵֽל׃ וּבְנֵי קְהָת, לְזַרְעֲיָתְהוֹן--עַמְרָם וְיִצְהָר, חֶבְרוֹן וְעֻזִּיאֵל. 
ג,כ וּבְנֵ֧י מְרָרִ֛י לְמִשְׁפְּחֹתָ֖ם מַחְלִ֣י וּמוּשִׁ֑י אֵ֥לֶּה הֵ֛ם מִשְׁפְּחֹ֥ת הַלֵּוִ֖י לְבֵ֥ית אֲבֹתָֽם׃ וּבְנֵי מְרָרִי, לְזַרְעֲיָתְהוֹן--מַחְלִי וּמוּשִׁי; אִלֵּין אִנּוּן זַרְעֲיָת לֵוִי, לְבֵית אֲבָהָתְהוֹן. 
ג,כא לְגֵ֣רְשׁ֔וֹן מִשְׁפַּ֨חַת֙ הַלִּבְנִ֔י וּמִשְׁפַּ֖חַת הַשִּׁמְעִ֑י אֵ֣לֶּה הֵ֔ם מִשְׁפְּחֹ֖ת הַגֵּֽרְשֻׁנִּֽי׃ לְגֵרְשׁוֹן--זַרְעִית לִבְנִי, וְזַרְעִית שִׁמְעִי; אִלֵּין אִנּוּן, זַרְעֲיָת גֵּרְשׁוֹן. 
ג,כב פְּקֻֽדֵיהֶם֙ בְּמִסְפַּ֣ר כָּל־זָכָ֔ר מִבֶּן־חֹ֖דֶשׁ וָמָ֑עְלָה פְּקֻ֣דֵיהֶ֔ם שִׁבְעַ֥ת אֲלָפִ֖ים וַֽחֲמֵ֥שׁ מֵאֽוֹת׃ מִנְיָנֵיהוֹן בְּמִנְיַן כָּל דְּכוּרָא, מִבַּר יַרְחָא וּלְעֵילָא; מִנְיָנֵיהוֹן, שִׁבְעָא אַלְפִין וַחֲמֵישׁ מְאָה. 
ג,כג מִשְׁפְּחֹ֖ת הַגֵּֽרְשֻׁנִּ֑י אַֽחֲרֵ֧י הַמִּשְׁכָּ֛ן יַֽחֲנ֖וּ יָֽמָּה׃ זַרְעֲיָת, גֵּרְשׁוֹן, אֲחוֹרֵי מַשְׁכְּנָא יִשְׁרוֹן, מַעְרְבָא. 
ג,כד וּנְשִׂ֥יא בֵֽית־אָ֖ב לַגֵּֽרְשֻׁנִּ֑י אֶלְיָסָ֖ף בֶּן־לָאֵֽל׃ וְרַב בֵּית בָּא, לְבֵית גֵּרְשׁוֹן, אֶלְיָסָף, בַּר לָאֵל. 
ג,כה וּמִשְׁמֶ֤רֶת בְּנֵֽי־גֵרְשׁוֹן֙ בְּאֹ֣הֶל מוֹעֵ֔ד הַמִּשְׁכָּ֖ן וְהָאֹ֑הֶל מִכְסֵ֕הוּ וּמָסַ֕ךְ פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ וּמַטְּרַת בְּנֵי גֵּרְשׁוֹן בְּמַשְׁכַּן זִמְנָא, מַשְׁכְּנָא וּפְרָסָא; חוּפָאֵיהּ--וּפְרָסָא, דִּתְרַע מַשְׁכַּן זִמְנָא. 
ג,כו וְקַלְעֵ֣י הֶֽחָצֵ֗ר וְאֶת־מָסַךְ֙ פֶּ֣תַח הֶֽחָצֵ֔ר אֲשֶׁ֧ר עַל־הַמִּשְׁכָּ֛ן וְעַל־הַמִּזְבֵּ֖חַ סָבִ֑יב וְאֵת֙ מֵֽיתָרָ֔יו לְכֹ֖ל עֲבֹֽדָתֽוֹ׃ {ס} וּסְרָדֵי דָּרְתָא, וְיָת פְּרָסָא דִּתְרַע דָּרְתָא, דְּעַל מַשְׁכְּנָא וְעַל מַדְבְּחָא, סְחוֹר סְחוֹר; וְיָת, אֲטוּנוֹהִי, לְכֹל, פֻּלְחָנֵיהּ. {ס}
ג,כז וְלִקְהָ֗ת מִשְׁפַּ֤חַת הַֽעַמְרָמִי֙ וּמִשְׁפַּ֣חַת הַיִּצְהָרִ֔י וּמִשְׁפַּ֨חַת֙ הַֽחֶבְרֹנִ֔י וּמִשְׁפַּ֖חַת הָעָזִּֽיאֵלִ֑י אֵ֥לֶּה הֵ֖ם מִשְׁפְּחֹ֥ת הַקְּהָתִֽי׃ וְלִקְהָת, זַרְעִית עַמְרָם וְזַרְעִית יִצְהָר, וְזַרְעִית חֶבְרוֹן, וְזַרְעִית עֻזִּיאֵל; אִלֵּין אִנּוּן, זַרְעֲיָת קְהָת. 
ג,כח בְּמִסְפַּר֙ כָּל־זָכָ֔ר מִבֶּן־חֹ֖דֶשׁ וָמָ֑עְלָה שְׁמֹנַ֤ת אֲלָפִים֙ וְשֵׁ֣שׁ מֵא֔וֹת שֹֽׁמְרֵ֖י מִשְׁמֶ֥רֶת הַקֹּֽדֶשׁ׃ בְּמִנְיַן, כָּל דְּכוּרָא, מִבַּר יַרְחָא, וּלְעֵילָא--תְּמָנְיָא אַלְפִין וְשֵׁית מְאָה, נָטְרֵי מַטַּרְתָּא דְּקֻדְשָׁא. 
ג,כט מִשְׁפְּחֹ֥ת בְּנֵֽי־קְהָ֖ת יַֽחֲנ֑וּ עַ֛ל יֶ֥רֶךְ הַמִּשְׁכָּ֖ן תֵּימָֽנָה׃ זַרְעֲיָת בְּנֵי קְהָת, יִשְׁרוֹן, עַל שִׁדָּא דְּמַשְׁכְּנָא, דָּרוֹמָא. 
ג,ל וּנְשִׂ֥יא בֵֽית־אָ֖ב לְמִשְׁפְּחֹ֣ת הַקְּהָתִ֑י אֱלִֽיצָפָ֖ן בֶּן־עֻזִּיאֵֽל׃ וְרַב בֵּית בָּא, לְזַרְעֲיָת קְהָת, אֱלִיצָפָן, בַּר עֻזִּיאֵל. 
ג,לא וּמִשְׁמַרְתָּ֗ם הָֽאָרֹ֤ן וְהַשֻּׁלְחָן֙ וְהַמְּנֹרָ֣ה וְהַֽמִּזְבְּחֹ֔ת וּכְלֵ֣י הַקֹּ֔דֶשׁ אֲשֶׁ֥ר יְשָֽׁרְת֖וּ בָּהֶ֑ם וְהַ֨מָּסָ֔ךְ וְכֹ֖ל עֲבֹֽדָתֽוֹ׃ וּמַטַּרְתְּהוֹן, אֲרוֹנָא וּפָתוּרָא וּמְנָרְתָא וּמַדְבְּחַיָּא, וּמָנֵי קֻדְשָׁא, דִּישַׁמְּשׁוּן בְּהוֹן; וּפְרָסָא--וְכֹל, פֻּלְחָנֵיהּ. 
ג,לב וּנְשִׂיא֙ נְשִׂיאֵ֣י הַלֵּוִ֔י אֶלְעָזָ֖ר בֶּן־אַֽהֲרֹ֣ן הַכֹּהֵ֑ן פְּקֻדַּ֕ת שֹֽׁמְרֵ֖י מִשְׁמֶ֥רֶת הַקֹּֽדֶשׁ׃ וְאַמַּרְכְּלָא דִּמְמֻנַּא עַל רַבְרְבֵי לֵיוָאֵי, אֶלְעָזָר בַּר אַהֲרוֹן כָּהֲנָא, מִתְּחוֹת יְדוֹהִי מְמֻנַּן, נָטְרֵי מַטַּרְתָּא דְּקֻדְשָׁא. 
ג,לג לִמְרָרִ֕י מִשְׁפַּ֨חַת֙ הַמַּחְלִ֔י וּמִשְׁפַּ֖חַת הַמּוּשִׁ֑י אֵ֥לֶּה הֵ֖ם מִשְׁפְּחֹ֥ת מְרָרִֽי׃ לִמְרָרִי--זַרְעִית מַחְלִי, וְזַרְעִית מוּשִׁי; אִלֵּין אִנּוּן, זַרְעֲיָת מְרָרִי. 
ג,לד וּפְקֻֽדֵיהֶם֙ בְּמִסְפַּ֣ר כָּל־זָכָ֔ר מִבֶּן־חֹ֖דֶשׁ וָמָ֑עְלָה שֵׁ֥שֶׁת אֲלָפִ֖ים וּמָאתָֽיִם׃ וּמִנְיָנֵיהוֹן בְּמִנְיַן כָּל דְּכוּרָא, מִבַּר יַרְחָא וּלְעֵילָא--שִׁתָּא אַלְפִין, וּמָאתַן. 
ג,לה וּנְשִׂ֤יא בֵֽית־אָב֙ לְמִשְׁפְּחֹ֣ת מְרָרִ֔י צֽוּרִיאֵ֖ל בֶּן־אֲבִיחָ֑יִל עַ֣ל יֶ֧רֶךְ הַמִּשְׁכָּ֛ן יַֽחֲנ֖וּ צָפֹֽנָה׃ וְרַב בֵּית בָּא לְזַרְעֲיָת מְרָרִי, צוּרִיאֵל בַּר אֲבִיחָיִל; עַל שִׁדָּא דְּמַשְׁכְּנָא יִשְׁרוֹן, צִפּוּנָא. 
ג,לו וּפְקֻדַּ֣ת מִשְׁמֶרֶת֮ בְּנֵ֣י מְרָרִי֒ קַרְשֵׁי֙ הַמִּשְׁכָּ֔ן וּבְרִיחָ֖יו וְעַמֻּדָ֣יו וַֽאֲדָנָ֑יו וְכָ֨ל־כֵּלָ֔יו וְכֹ֖ל עֲבֹֽדָתֽוֹ׃ וְדִמְסִיר לְמִטַּר, לִבְנֵי מְרָרִי--דַּפֵּי מַשְׁכְּנָא, וְעָבְרוֹהִי וְעַמּוּדוֹהִי וְסָמְכוֹהִי; וְכָל מָנוֹהִי--וְכֹל, פֻּלְחָנֵיהּ. 
ג,לז וְעַמֻּדֵ֧י הֶֽחָצֵ֛ר סָבִ֖יב וְאַדְנֵיהֶ֑ם וִיתֵֽדֹתָ֖ם וּמֵֽיתְרֵיהֶֽם׃ וְעַמּוּדֵי דָּרְתָא סְחוֹר סְחוֹר, וְסָמְכֵיהוֹן; וְסִכֵּיהוֹן, וַאֲטוּנֵיהוֹן. 
ג,לח וְהַֽחֹנִ֣ים לִפְנֵ֣י הַמִּשְׁכָּ֡ן קֵ֣דְמָה לִפְנֵי֩ אֹֽהֶל־מוֹעֵ֨ד ׀ מִזְרָ֜חָה מֹשֶׁ֣ה ׀ וְאַֽהֲרֹ֣ן וּבָנָ֗יו שֹֽׁמְרִים֙ מִשְׁמֶ֣רֶת הַמִּקְדָּ֔שׁ לְמִשְׁמֶ֖רֶת בְּנֵ֣י יִשְׂרָאֵ֑ל וְהַזָּ֥ר הַקָּרֵ֖ב יוּמָֽת׃ וְדִשְׁרַן קֳדָם מַשְׁכְּנָא קִדּוּמָא קֳדָם מַשְׁכַּן זִמְנָא מַדְנְחָא מֹשֶׁה וְאַהֲרוֹן וּבְנוֹהִי, נָטְרֵי מַטְּרַת מַקְדְּשָׁא, לְמַטְּרַת, בְּנֵי יִשְׂרָאֵל; וְחִילוֹנַי דְּיִקְרַב, יִתְקְטִיל. 
ג,לט כָּל־פְּקוּדֵ֨י הַלְוִיִּ֜ם אֲשֶׁר֩ פָּקַ֨ד מֹשֶׁ֧ה וְׄאַֽׄהֲׄרֹ֛ׄןׄ עַל־פִּ֥י יְהוָ֖ה לְמִשְׁפְּחֹתָ֑ם כָּל־זָכָר֙ מִבֶּן־חֹ֣דֶשׁ וָמַ֔עְלָה שְׁנַ֥יִם וְעֶשְׂרִ֖ים אָֽלֶף׃ {ס} כָּל מִנְיָנֵי לֵיוָאֵי דִּמְנָא מֹשֶׁה וְאַהֲרוֹן, עַל מֵימְרָא דַּייָ--לְזַרְעֲיָתְהוֹן:  כָּל דְּכוּרָא מִבַּר יַרְחָא וּלְעֵילָא, עַסְרִין וּתְרֵין אַלְפִין. {ס}
ג,מ וַיֹּ֨אמֶר יְהוָ֜ה אֶל־מֹשֶׁ֗ה פְּקֹ֨ד כָּל־בְּכֹ֤ר זָכָר֙ לִבְנֵ֣י יִשְׂרָאֵ֔ל מִבֶּן־חֹ֖דֶשׁ וָמָ֑עְלָה וְשָׂ֕א אֵ֖ת מִסְפַּ֥ר שְׁמֹתָֽם׃ וַאֲמַר יְיָ לְמֹשֶׁה, מְנִי כָּל בְּכוֹרַיָּא דְּכוּרַיָּא לִבְנֵי יִשְׂרָאֵל, מִבַּר יַרְחָא, וּלְעֵילָא; וְקַבֵּיל, יָת חֻשְׁבַּן שְׁמָהָתְהוֹן. 
ג,מא וְלָֽקַחְתָּ֨ אֶת־הַלְוִיִּ֥ם לִי֙ אֲנִ֣י יְהוָ֔ה תַּ֥חַת כָּל־בְּכֹ֖ר בִּבְנֵ֣י יִשְׂרָאֵ֑ל וְאֵת֙ בֶּֽהֱמַ֣ת הַלְוִיִּ֔ם תַּ֣חַת כָּל־בְּכ֔וֹר בְּבֶֽהֱמַ֖ת בְּנֵ֥י יִשְׂרָאֵֽל׃ וּתְקָרֵיב יָת לֵיוָאֵי קֳדָמַי אֲנָא יְיָ, חֲלָף כָּל בֻּכְרָא בִּבְנֵי יִשְׂרָאֵל; וְיָת, בְּעִירָא דְּלֵיוָאֵי, חֲלָף כָּל בֻּכְרָא, בִּבְעִירָא דִּבְנֵי יִשְׂרָאֵל. 
ג,מב וַיִּפְקֹ֣ד מֹשֶׁ֔ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֹת֑וֹ אֶֽת־כָּל־בְּכ֖וֹר בִּבְנֵ֥י יִשְׂרָאֵֽל׃ וּמְנָא מֹשֶׁה, כְּמָא דְּפַקֵּיד יְיָ יָתֵיהּ, יָת כָּל בֻּכְרָא, בִּבְנֵי יִשְׂרָאֵל. 
ג,מג וַיְהִי֩ כָל־בְּכ֨וֹר זָכָ֜ר בְּמִסְפַּ֥ר שֵׁמֹ֛ת מִבֶּן־חֹ֥דֶשׁ וָמַ֖עְלָה לִפְקֻֽדֵיהֶ֑ם שְׁנַ֤יִם וְעֶשְׂרִים֙ אֶ֔לֶף שְׁלֹשָׁ֥ה וְשִׁבְעִ֖ים וּמָאתָֽיִם׃ {פ} וַהֲווֹ כָּל בְּכוֹרַיָּא דְּכוּרַיָּא בְּמִנְיַן שְׁמָהָן, מִבַּר יַרְחָא וּלְעֵילָא--לְמִנְיָנֵיהוֹן:  עַסְרִין וּתְרֵין אַלְפִין, וּמָאתַן וְשִׁבְעִין וּתְלָתָא. {פ}
ג,מד וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ג,מה קַ֣ח אֶת־הַלְוִיִּ֗ם תַּ֤חַת כָּל־בְּכוֹר֙ בִּבְנֵ֣י יִשְׂרָאֵ֔ל וְאֶת־בֶּֽהֱמַ֥ת הַלְוִיִּ֖ם תַּ֣חַת בְּהֶמְתָּ֑ם וְהָֽיוּ־לִ֥י הַלְוִיִּ֖ם אֲנִ֥י יְהוָֽה׃ קָרֵיב יָת לֵיוָאֵי, חֲלָף כָּל בֻּכְרָא בִּבְנֵי יִשְׂרָאֵל, וְיָת בְּעִירָא דְּלֵיוָאֵי, חֲלָף בְּעִירְהוֹן; וִיהוֹן מְשַׁמְּשִׁין קֳדָמַי לֵיוָאֵי, אֲנָא יְיָ. 
ג,מו וְאֵת֙ פְּדוּיֵ֣י הַשְּׁלֹשָׁ֔ה וְהַשִּׁבְעִ֖ים וְהַמָּאתָ֑יִם הָעֹֽדְפִים֙ עַל־הַלְוִיִּ֔ם מִבְּכ֖וֹר בְּנֵ֥י יִשְׂרָאֵֽל׃ וְיָת פֻּרְקַן מָאתַן וְשִׁבְעִין וּתְלָתָא--דְּיַתִּירִין, עַל לֵיוָאֵי, מִבְּכוֹרַיָּא, דִּבְנֵי יִשְׂרָאֵל. 
ג,מז וְלָֽקַחְתָּ֗ חֲמֵ֧שֶׁת חֲמֵ֛שֶׁת שְׁקָלִ֖ים לַגֻּלְגֹּ֑לֶת בְּשֶׁ֤קֶל הַקֹּ֨דֶשׁ֙ תִּקָּ֔ח עֶשְׂרִ֥ים גֵּרָ֖ה הַשָּֽׁקֶל׃ וְתִסַּב, חֲמֵישׁ חֲמֵישׁ סִלְעִין--לְגֻלְגֻּלְתָּא:  בְּסִלְעֵי קֻדְשָׁא תִּסַּב, עַסְרִין מָעִין סִלְעָא. 
ג,מח וְנָֽתַתָּ֣ה הַכֶּ֔סֶף לְאַֽהֲרֹ֖ן וּלְבָנָ֑יו פְּדוּיֵ֕י הָעֹֽדְפִ֖ים בָּהֶֽם׃ וְתִתֵּין כַּסְפָּא, לְאַהֲרוֹן וְלִבְנוֹהִי--פֻּרְקַן, דְּיַתִּירִין דִּבְהוֹן. 
ג,מט וַיִּקַּ֣ח מֹשֶׁ֔ה אֵ֖ת כֶּ֣סֶף הַפִּדְי֑וֹם מֵאֵת֙ הָעֹ֣דְפִ֔ים עַ֖ל פְּדוּיֵ֥י הַלְוִיִּֽם׃ וּנְסֵיב מֹשֶׁה, יָת כְּסַף פֻּרְקָנְהוֹן--מִן, דְּיַתִּירִין, עַל, פְּרִיקֵי לֵיוָאֵי. 
ג,נ מֵאֵ֗ת בְּכ֛וֹר בְּנֵ֥י יִשְׂרָאֵ֖ל לָקַ֣ח אֶת־הַכָּ֑סֶף חֲמִשָּׁ֨ה וְשִׁשִּׁ֜ים וּשְׁלֹ֥שׁ מֵא֛וֹת וָאֶ֖לֶף בְּשֶׁ֥קֶל הַקֹּֽדֶשׁ׃ מִן, בְּכוֹרַיָּא דִּבְנֵי יִשְׂרָאֵל--נְסֵיב יָת כַּסְפָּא:  אֶלֶף, וּתְלָת מְאָה וְשִׁתִּין וַחֲמֵישׁ--סִלְעִין בְּסִלְעֵי קֻדְשָׁא. 
ג,נא וַיִּתֵּ֨ן מֹשֶׁ֜ה אֶת־כֶּ֧סֶף הַפְּדֻיִ֛ם לְאַֽהֲרֹ֥ן וּלְבָנָ֖יו עַל־פִּ֣י יְהוָ֑ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ {פ} וִיהַב מֹשֶׁה יָת כְּסַף פְּרִיקַיָּא, לְאַהֲרוֹן וְלִבְנוֹהִי--עַל מֵימְרָא דַּייָ:  כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. {פ}
ד,א וַיְדַבֵּ֣ר יְהוָ֔ה אֶל־מֹשֶׁ֥ה וְאֶֽל־אַהֲרֹ֖ן לֵאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה וּלְאַהֲרוֹן לְמֵימַר. 
ד,ב נָשֹׂ֗א אֶת־רֹאשׁ֙ בְּנֵ֣י קְהָ֔ת מִתּ֖וֹךְ בְּנֵ֣י לֵוִ֑י לְמִשְׁפְּחֹתָ֖ם לְבֵ֥ית אֲבֹתָֽם׃ קַבִּילוּ, יָת חֻשְׁבַּן בְּנֵי קְהָת, מִגּוֹ, בְּנֵי לֵוִי--לְזַרְעֲיָתְהוֹן, לְבֵית אֲבָהָתְהוֹן. 
ד,ג מִבֶּ֨ן שְׁלֹשִׁ֤ים שָׁנָה֙ וָמַ֔עְלָה וְעַ֖ד בֶּן־חֲמִשִּׁ֣ים שָׁנָ֑ה כָּל־בָּא֙ לַצָּבָ֔א לַֽעֲשׂ֥וֹת מְלָאכָ֖ה בְּאֹ֥הֶל מוֹעֵֽד׃ מִבַּר תְּלָתִין שְׁנִין וּלְעֵילָא, וְעַד בַּר חַמְשִׁין שְׁנִין--כָּל דְּאָתֵי, לְחֵילָא, לְמֶעֱבַד עֲבִידְתָא, בְּמַשְׁכַּן זִמְנָא. 
ד,ד זֹ֛את עֲבֹדַ֥ת בְּנֵֽי־קְהָ֖ת בְּאֹ֣הֶל מוֹעֵ֑ד קֹ֖דֶשׁ הַקֳּדָשִֽׁים׃ דֵּין פֻּלְחַן בְּנֵי קְהָת, בְּמַשְׁכַּן זִמְנָא--קֹדֶשׁ, קֻדְשַׁיָּא. 
ד,ה וּבָ֨א אַֽהֲרֹ֤ן וּבָנָיו֙ בִּנְסֹ֣עַ הַֽמַּחֲנֶ֔ה וְהוֹרִ֕דוּ אֵ֖ת פָּרֹ֣כֶת הַמָּסָ֑ךְ וְכִ֨סּוּ־בָ֔הּ אֵ֖ת אֲרֹ֥ן הָֽעֵדֻֽת׃ וְיֵיעוֹל אַהֲרוֹן וּבְנוֹהִי, בְּמִטַּל מַשְׁרִיתָא, וִיפָרְקוּן, יָת פָּרֻכְתָּא דִּפְרָסָא; וִיכַסּוֹן בַּהּ--יָת, אֲרוֹנָא דְּסָהֲדוּתָא. 
ד,ו וְנָֽתְנ֣וּ עָלָ֗יו כְּסוּי֙ ע֣וֹר תַּ֔חַשׁ וּפָֽרְשׂ֧וּ בֶֽגֶד־כְּלִ֛יל תְּכֵ֖לֶת מִלְמָ֑עְלָה וְשָׂמ֖וּ בַּדָּֽיו׃ וְיִתְּנוּן עֲלוֹהִי, חוּפָאָה דִּמְשַׁךְ סָסְגוֹנָא, וְיִפְרְסוּן לְבוּשׁ גְּמִיר תַּכְלָא, מִלְּעֵילָא; וִישַׁוּוֹן, אֲרִיחוֹהִי. 
ד,ז וְעַ֣ל ׀ שֻׁלְחַ֣ן הַפָּנִ֗ים יִפְרְשׂוּ֮ בֶּ֣גֶד תְּכֵלֶת֒ וְנָֽתְנ֣וּ עָ֠לָיו אֶת־הַקְּעָרֹ֤ת וְאֶת־הַכַּפֹּת֙ וְאֶת־הַמְּנַקִּיֹּ֔ת וְאֵ֖ת קְשׂ֣וֹת הַנָּ֑סֶךְ וְלֶ֥חֶם הַתָּמִ֖יד עָלָ֥יו יִֽהְיֶֽה׃ וְעַל פָּתוּרָא דִּלְחֵים אַפַּיָּא, יִפְרְסוּן לְבוּשׁ תַּכְלָא, וְיִתְּנוּן עֲלוֹהִי יָת מְגִסַּיָּא וְיָת בָּזִכַּיָּא וְיָת מְכִילָתָא, וְיָת קָסְוָת נִסּוּכָא; וְלַחְמָא תְּדִירָא, עֲלוֹהִי יְהֵי. 
ד,ח וּפָֽרְשׂ֣וּ עֲלֵיהֶ֗ם בֶּ֚גֶד תּוֹלַ֣עַת שָׁנִ֔י וְכִסּ֣וּ אֹת֔וֹ בְּמִכְסֵ֖ה ע֣וֹר תָּ֑חַשׁ וְשָׂמ֖וּ אֶת־בַּדָּֽיו׃ וְיִפְרְסוּן עֲלֵיהוֹן, לְבוּשׁ צְבַע זְהוֹרִי, וִיכַסּוֹן יָתֵיהּ, בְּחוּפָאָה דִּמְשַׁךְ סָסְגוֹנָא; וִישַׁוּוֹן, יָת אֲרִיחוֹהִי. 
ד,ט וְלָֽקְח֣וּ ׀ בֶּ֣גֶד תְּכֵ֗לֶת וְכִסּ֞וּ אֶת־מְנֹרַ֤ת הַמָּאוֹר֙ וְאֶת־נֵ֣רֹתֶ֔יהָ וְאֶת־מַלְקָחֶ֖יהָ וְאֶת־מַחְתֹּתֶ֑יהָ וְאֵת֙ כָּל־כְּלֵ֣י שַׁמְנָ֔הּ אֲשֶׁ֥ר יְשָֽׁרְתוּ־לָ֖הּ בָּהֶֽם׃ וְיִסְּבוּן לְבוּשׁ תַּכְלָא, וִיכַסּוֹן יָת מְנָרְתָא דְּאַנְהוֹרֵי וְיָת בּוֹצִינַהָא, וְיָת צֵיבְתַהָא, וְיָת מַחְתְּיָתַהָא; וְיָת כָּל מָנֵי שִׁמּוּשַׁהּ, דִּישַׁמְּשׁוּן לַהּ בְּהוֹן. 
ד,י וְנָֽתְנ֤וּ אֹתָהּ֙ וְאֶת־כָּל־כֵּלֶ֔יהָ אֶל־מִכְסֵ֖ה ע֣וֹר תָּ֑חַשׁ וְנָֽתְנ֖וּ עַל־הַמּֽוֹט׃ וְיִתְּנוּן יָתַהּ וְיָת כָּל מָנַהָא, לְחוּפָאָה דִּמְשַׁךְ סָסְגוֹנָא; וְיִתְּנוּן, עַל אֲרִיחָא. 
ד,יא וְעַ֣ל ׀ מִזְבַּ֣ח הַזָּהָ֗ב יִפְרְשׂוּ֙ בֶּ֣גֶד תְּכֵ֔לֶת וְכִסּ֣וּ אֹת֔וֹ בְּמִכְסֵ֖ה ע֣וֹר תָּ֑חַשׁ וְשָׂמ֖וּ אֶת־בַּדָּֽיו׃ וְעַל מַדְבְּחָא דְּדַהְבָּא, יִפְרְסוּן לְבוּשׁ תַּכְלָא, וִיכַסּוֹן יָתֵיהּ, בְּחוּפָאָה דִּמְשַׁךְ סָסְגוֹנָא; וִישַׁוּוֹן, יָת אֲרִיחוֹהִי. 
ד,יב וְלָֽקְחוּ֩ אֶת־כָּל־כְּלֵ֨י הַשָּׁרֵ֜ת אֲשֶׁ֧ר יְשָֽׁרְתוּ־בָ֣ם בַּקֹּ֗דֶשׁ וְנָֽתְנוּ֙ אֶל־בֶּ֣גֶד תְּכֵ֔לֶת וְכִסּ֣וּ אוֹתָ֔ם בְּמִכְסֵ֖ה ע֣וֹר תָּ֑חַשׁ וְנָֽתְנ֖וּ עַל־הַמּֽוֹט׃ וְיִסְּבוּן יָת כָּל מָנֵי שִׁמּוּשָׁא דִּישַׁמְּשׁוּן בְּהוֹן בְּקֻדְשָׁא, וְיִתְּנוּן לִלְבוּשׁ תַּכְלָא, וִיכַסּוֹן יָתְהוֹן, בְּחוּפָאָה דִּמְשַׁךְ סָסְגוֹנָא; וְיִתְּנוּן, עַל אֲרִיחָא. 
ד,יג וְדִשְּׁנ֖וּ אֶת־הַמִּזְבֵּ֑חַ וּפָֽרְשׂ֣וּ עָלָ֔יו בֶּ֖גֶד אַרְגָּמָֽן׃ וְיִסְפּוֹן, יָת קִטְמָא דְּמַדְבְּחָא; וְיִפְרְסוּן עֲלוֹהִי, לְבוּשׁ אַרְגְּוָן. 
ד,יד וְנָֽתְנ֣וּ עָ֠לָיו אֶֽת־כָּל־כֵּלָ֞יו אֲשֶׁ֣ר יְֽשָׁרְת֧וּ עָלָ֣יו בָּהֶ֗ם אֶת־הַמַּחְתֹּ֤ת אֶת־הַמִּזְלָגֹת֙ וְאֶת־הַיָּעִ֣ים וְאֶת־הַמִּזְרָקֹ֔ת כֹּ֖ל כְּלֵ֣י הַמִּזְבֵּ֑חַ וּפָֽרְשׂ֣וּ עָלָ֗יו כְּס֛וּי ע֥וֹר תַּ֖חַשׁ וְשָׂמ֥וּ בַדָּֽיו׃ וְיִתְּנוּן עֲלוֹהִי יָת כָּל מָנוֹהִי דִּישַׁמְּשׁוּן עֲלוֹהִי בְּהוֹן, יָת מַחְתְּיָתָא וְיָת צִנּוֹרְיָתָא וְיָת מַגְרוֹפְיָתָא וְיָת מִזְרְקַיָּא--כֹּל, מָנֵי מַדְבְּחָא; וְיִפְרְסוֹן עֲלוֹהִי, חוּפָאָה דִּמְשַׁךְ סָסְגוֹנָא--וִישַׁוּוֹן אֲרִיחוֹהִי. 
ד,טו וְכִלָּ֣ה אַֽהֲרֹן־וּ֠בָנָיו לְכַסֹּ֨ת אֶת־הַקֹּ֜דֶשׁ וְאֶת־כָּל־כְּלֵ֣י הַקֹּדֶשׁ֮ בִּנְסֹ֣עַ הַֽמַּחֲנֶה֒ וְאַֽחֲרֵי־כֵ֗ן יָבֹ֤אוּ בְנֵֽי־קְהָת֙ לָשֵׂ֔את וְלֹֽא־יִגְּע֥וּ אֶל־הַקֹּ֖דֶשׁ וָמֵ֑תוּ אֵ֛לֶּה מַשָּׂ֥א בְנֵֽי־קְהָ֖ת בְּאֹ֥הֶל מוֹעֵֽד׃ וִישֵׁיצֵי אַהֲרוֹן וּבְנוֹהִי לְכַסָּאָה יָת קֻדְשָׁא וְיָת כָּל מָנֵי קֻדְשָׁא, בְּמִטַּל מַשְׁרִיתָא, וּבָתַר כֵּין יֵיעֲלוּן בְּנֵי קְהָת לְמִטַּל, וְלָא יִקְרְבוּן לְקֻדְשָׁא וְלָא יְמוּתוּן; אִלֵּין מַטוּל בְּנֵי קְהָת, בְּמַשְׁכַּן זִמְנָא. 
ד,טז וּפְקֻדַּ֞ת אֶלְעָזָ֣ר ׀ בֶּן־אַֽהֲרֹ֣ן הַכֹּהֵ֗ן שֶׁ֤מֶן הַמָּאוֹר֙ וּקְטֹ֣רֶת הַסַּמִּ֔ים וּמִנְחַ֥ת הַתָּמִ֖יד וְשֶׁ֣מֶן הַמִּשְׁחָ֑ה פְּקֻדַּ֗ת כָּל־הַמִּשְׁכָּן֙ וְכָל־אֲשֶׁר־בּ֔וֹ בְּקֹ֖דֶשׁ וּבְכֵלָֽיו׃ {פ} וְדִמְסִיר לְאֶלְעָזָר בַּר אַהֲרוֹן כָּהֲנָא, מִשְׁחָא דְּאַנְהָרוּתָא וּקְטֹרֶת בֻּסְמַיָּא, וּמִנְחָתָא תְּדִירָא, וּמִשְׁחָא דִּרְבוּתָא:  מָסְרַת, כָּל מַשְׁכְּנָא וְכָל דְּבֵיהּ, בְּקֻדְשָׁא, וּבְמָנוֹהִי. {פ}
ד,יז וַיְדַבֵּ֣ר יְהוָ֔ה אֶל־מֹשֶׁ֥ה וְאֶֽל־אַהֲרֹ֖ן לֵאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה וּלְאַהֲרוֹן לְמֵימַר. 
ד,יח אַל־תַּכְרִ֕יתוּ אֶת־שֵׁ֖בֶט מִשְׁפְּחֹ֣ת הַקְּהָתִ֑י מִתּ֖וֹךְ הַלְוִיִּֽם׃ לָא תְּשֵׁיצוֹן, יָת שֵׁיבַט זַרְעֲיָת קְהָת, מִגּוֹ, לֵיוָאֵי. 
ד,יט וְזֹ֣את ׀ עֲשׂ֣וּ לָהֶ֗ם וְחָיוּ֙ וְלֹ֣א יָמֻ֔תוּ בְּגִשְׁתָּ֖ם אֶת־קֹ֣דֶשׁ הַקֳּדָשִׁ֑ים אַֽהֲרֹ֤ן וּבָנָיו֙ יָבֹ֔אוּ וְשָׂמ֣וּ אוֹתָ֗ם אִ֥ישׁ אִ֛ישׁ עַל־עֲבֹֽדָת֖וֹ וְאֶל־מַשָּׂאֽוֹ׃ וְדָא עֲבִידוּ לְהוֹן, וְיֵיחוֹן וְלָא יְמוּתוּן, בְּמִקְרַבְהוֹן, לְקֹדֶשׁ קֻדְשַׁיָּא:  אַהֲרוֹן וּבְנוֹהִי, יֵיעֲלוּן, וִימַנּוֹן יָתְהוֹן גְּבַר גְּבַר עַל פֻּלְחָנֵיהּ, וּלְמַטוּלֵיהּ. 
ד,כ וְלֹֽא־יָבֹ֧אוּ לִרְא֛וֹת כְּבַלַּ֥ע אֶת־הַקֹּ֖דֶשׁ וָמֵֽתוּ׃ {פ} וְלָא יֵיעֲלוּן לְמִחְזֵי כַּד מְכַסַּן יָת מָנֵי קֻדְשָׁא, וְלָא יְמוּתוּן. {פ}
