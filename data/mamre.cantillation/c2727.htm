איוב פרק כז

א                וַיֹּ֣סֶף אִ֭יּוֹב שְׂאֵ֥ת מְשָׁל֗וֹ וַיֹּאמַֽר׃
ב חַי־אֵ֭ל הֵסִ֣יר מִשְׁפָּטִ֑י    וְ֝שַׁדַּ֗י הֵמַ֥ר נַפְשִֽׁי׃
ג כִּֽי־כָל־ע֣וֹד נִשְׁמָתִ֣י בִ֑י    וְר֖וּחַ אֱל֣וֹהַּ בְּאַפִּֽי׃
ד אִם־תְּדַבֵּ֣רְנָה שְׂפָתַ֣י עַוְלָ֑ה    וּ֝לְשׁוֹנִ֗י אִם־יֶהְגֶּ֥ה רְמִיָּֽה׃
ה חָלִ֣ילָה לִּי֮    אִם־אַצְדִּ֪יק אֶ֫תְכֶ֥ם
עַד־אֶגְוָ֑ע    לֹֽא־אָסִ֖יר תֻּמָּתִ֣י מִמֶּֽנִּי׃
ו בְּצִדְקָתִ֣י הֶֽ֭חֱזַקְתִּי וְלֹ֣א אַרְפֶּ֑הָ    לֹֽא־יֶחֱרַ֥ף לְ֝בָבִ֗י מִיָּמָֽי׃
ז יְהִ֣י כְ֭רָשָׁע אֹ֣יְבִ֑י    וּמִתְקֽוֹמְמִ֥י כְעַוָּֽל׃
ח כִּ֤י מַה־תִּקְוַ֣ת חָ֭נֵף כִּ֣י יִבְצָ֑ע    כִּ֤י יֵ֖שֶׁל אֱל֣וֹהַּ נַפְשֽׁוֹ׃
ט הֲֽ֭צַעֲקָתוֹ יִשְׁמַ֥ע ׀ אֵ֑ל    כִּֽי־תָב֖וֹא עָלָ֣יו צָרָֽה׃
י אִם־עַל־שַׁדַּ֥י יִתְעַנָּ֑ג    יִקְרָ֖א אֱל֣וֹהַּ בְּכָל־עֵֽת׃
יא אוֹרֶ֣ה אֶתְכֶ֣ם בְּיַד־אֵ֑ל    אֲשֶׁ֥ר עִם־שַׁ֝דַּ֗י לֹ֣א אֲכַחֵֽד׃
יב הֵן־אַתֶּ֣ם כֻּלְּכֶ֣ם חֲזִיתֶ֑ם    וְלָמָּה־זֶּ֝֗ה הֶ֣בֶל תֶּהְבָּֽלוּ׃
יג זֶ֤ה ׀ חֵֽלֶק־אָדָ֖ם רָשָׁ֥ע ׀ עִם־אֵ֑ל    וְֽנַחֲלַ֥ת עָֽ֝רִיצִ֗ים מִשַּׁדַּ֥י יִקָּֽחוּ׃
יד אִם־יִרְבּ֣וּ בָנָ֣יו לְמוֹ־חָ֑רֶב    וְ֝צֶֽאֱצָאָ֗יו לֹ֣א יִשְׂבְּעוּ־לָֽחֶם׃
טו שְׂ֭רִידָו בַּמָּ֣וֶת יִקָּבֵ֑רוּ    וְ֝אַלְמְנֹתָ֗יו לֹ֣א תִבְכֶּֽינָה׃
טז אִם־יִצְבֹּ֣ר כֶּֽעָפָ֣ר כָּ֑סֶף    וְ֝כַחֹ֗מֶר יָכִ֥ין מַלְבּֽוּשׁ׃
יז יָ֭כִין וְצַדִּ֣יק יִלְבָּ֑שׁ    וְ֝כֶ֗סֶף נָקִ֥י יַֽחֲלֹֽק׃
יח בָּנָ֣ה כָעָ֣שׁ בֵּית֑וֹ    וּ֝כְסֻכָּ֗ה עָשָׂ֥ה נֹצֵֽר׃
יט עָשִׁ֣יר יִ֭שְׁכַּב וְלֹ֣א יֵֽאָסֵ֑ף    עֵינָ֖יו פָּקַ֣ח וְאֵינֶֽנּוּ׃
כ תַּשִּׂיגֵ֣הוּ כַ֭מַּיִם בַּלָּה֑וֹת    לַ֝֗יְלָה גְּנָבַ֥תּוּ סוּפָֽה׃
כא יִשָּׂאֵ֣הוּ קָדִ֣ים וְיֵלַ֑ךְ    וִ֝ישָֽׂעֲרֵ֗הוּ מִמְּקֹמֽוֹ׃
כב וְיַשְׁלֵ֣ךְ עָ֭לָיו וְלֹ֣א יַחְמֹ֑ל    מִ֝יָּד֗וֹ בָּר֥וֹחַ יִבְרָֽח׃
כג יִשְׂפֹּ֣ק עָלֵ֣ימוֹ כַפֵּ֑ימוֹ    וְיִשְׁרֹ֥ק עָ֝לָ֗יו מִמְּקֹמֽוֹ׃
