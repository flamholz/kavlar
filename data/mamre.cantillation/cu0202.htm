ואירא

ו,ב וַיְדַבֵּ֥ר אֱלֹהִ֖ים אֶל־מֹשֶׁ֑ה וַיֹּ֥אמֶר אֵלָ֖יו אֲנִ֥י יְהוָֽה׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה; וַאֲמַר לֵיהּ, אֲנָא יְיָ. 
ו,ג וָֽאֵרָ֗א אֶל־אַבְרָהָ֛ם אֶל־יִצְחָ֥ק וְאֶֽל־יַעֲקֹ֖ב בְּאֵ֣ל שַׁדָּ֑י וּשְׁמִ֣י יְהוָ֔ה לֹ֥א נוֹדַ֖עְתִּי לָהֶֽם׃ וְאִתְגְּלִיתִי, לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקוֹב--בְּאֵל שַׁדָּי; וּשְׁמִי יְיָ, לָא הוֹדַעִית לְהוֹן. 
ו,ד וְגַ֨ם הֲקִמֹ֤תִי אֶת־בְּרִיתִי֙ אִתָּ֔ם לָתֵ֥ת לָהֶ֖ם אֶת־אֶ֣רֶץ כְּנָ֑עַן אֵ֛ת אֶ֥רֶץ מְגֻֽרֵיהֶ֖ם אֲשֶׁר־גָּ֥רוּ בָֽהּ׃ וְאַף אֲקֵימִית יָת קְיָמִי עִמְּהוֹן, לְמִתַּן לְהוֹן יָת אַרְעָא דִּכְנָעַן--יָת אֲרַע תּוֹתָבוּתְהוֹן, דְּאִתּוֹתַבוּ בַּהּ. 
ו,ה וְגַ֣ם ׀ אֲנִ֣י שָׁמַ֗עְתִּי אֶֽת־נַאֲקַת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל אֲשֶׁ֥ר מִצְרַ֖יִם מַֽעֲבִדִ֣ים אֹתָ֑ם וָֽאֶזְכֹּ֖ר אֶת־בְּרִיתִֽי׃ וְאַף קֳדָמַי שְׁמִיעַ, יָת קְבִילַת בְּנֵי יִשְׂרָאֵל, דְּמִצְרָאֵי, מַפְלְחִין בְּהוֹן; וּדְכִירְנָא, יָת קְיָמִי. 
ו,ו לָכֵ֞ן אֱמֹ֥ר לִבְנֵֽי־יִשְׂרָאֵל֮ אֲנִ֣י יְהוָה֒ וְהֽוֹצֵאתִ֣י אֶתְכֶ֗ם מִתַּ֨חַת֙ סִבְלֹ֣ת מִצְרַ֔יִם וְהִצַּלְתִּ֥י אֶתְכֶ֖ם מֵעֲבֹֽדָתָ֑ם וְגָֽאַלְתִּ֤י אֶתְכֶם֙ בִּזְר֣וֹעַ נְטוּיָ֔ה וּבִשְׁפָטִ֖ים גְּדֹלִֽים׃ בְּכֵין אֵימַר לִבְנֵי יִשְׂרָאֵל, אֲנָא יְיָ, וְאַפֵּיק יָתְכוֹן מִגּוֹ דְּחוֹק פֻּלְחַן מִצְרָאֵי, וַאֲשֵׁיזֵיב יָתְכוֹן מִפֻּלְחָנְהוֹן; וְאֶפְרוֹק יָתְכוֹן בִּדְרָע מְרָמַם, וּבְדִינִין רַבְרְבִין. 
ו,ז וְלָֽקַחְתִּ֨י אֶתְכֶ֥ם לִי֙ לְעָ֔ם וְהָיִ֥יתִי לָכֶ֖ם לֵֽאלֹהִ֑ים וִֽידַעְתֶּ֗ם כִּ֣י אֲנִ֤י יְהוָה֙ אֱלֹ֣הֵיכֶ֔ם הַמּוֹצִ֣יא אֶתְכֶ֔ם מִתַּ֖חַת סִבְל֥וֹת מִצְרָֽיִם׃ וַאֲקָרֵיב יָתְכוֹן קֳדָמַי לְעַם, וְאֶהְוֵי לְכוֹן לֶאֱלָהּ; וְתִדְּעוּן, אֲרֵי אֲנָא יְיָ אֱלָהֲכוֹן, דְּאַפֵּיק יָתְכוֹן, מִגּוֹ דְּחוֹק פֻּלְחַן מִצְרָאֵי. 
ו,ח וְהֵֽבֵאתִ֤י אֶתְכֶם֙ אֶל־הָאָ֔רֶץ אֲשֶׁ֤ר נָשָׂ֨אתִי֙ אֶת־יָדִ֔י לָתֵ֣ת אֹתָ֔הּ לְאַבְרָהָ֥ם לְיִצְחָ֖ק וּֽלְיַעֲקֹ֑ב וְנָֽתַתִּ֨י אֹתָ֥הּ לָכֶ֛ם מֽוֹרָשָׁ֖ה אֲנִ֥י יְהוָֽה׃ וְאַעֵיל יָתְכוֹן, לְאַרְעָא, דְּקַיֵּימִית בְּמֵימְרִי, לְמִתַּן יָתַהּ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקוֹב; וְאֶתֵּין יָתַהּ לְכוֹן יְרֻתָּא, אֲנָא יְיָ. 
ו,ט וַיְדַבֵּ֥ר מֹשֶׁ֛ה כֵּ֖ן אֶל־בְּנֵ֣י יִשְׂרָאֵ֑ל וְלֹ֤א שָֽׁמְעוּ֙ אֶל־מֹשֶׁ֔ה מִקֹּ֣צֶר ר֔וּחַ וּמֵֽעֲבֹדָ֖ה קָשָֽׁה׃ {פ} וּמַלֵּיל מֹשֶׁה כֵּין, עִם בְּנֵי יִשְׂרָאֵל; וְלָא קַבִּילוּ, מִן מֹשֶׁה, מֵעֲיָק רוּחַ, וּמִפֻּלְחָנָא דַּהֲוָה קְשֵׁי עֲלֵיהוֹן. {פ}
ו,י וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ו,יא בֹּ֣א דַבֵּ֔ר אֶל־פַּרְעֹ֖ה מֶ֣לֶךְ מִצְרָ֑יִם וִֽישַׁלַּ֥ח אֶת־בְּנֵֽי־יִשְׂרָאֵ֖ל מֵֽאַרְצֽוֹ׃ עוֹל מַלֵּיל, עִם פַּרְעֹה מַלְכָּא דְּמִצְרָיִם; וִישַׁלַּח יָת בְּנֵי יִשְׂרָאֵל, מֵאַרְעֵיהּ. 
ו,יב וַיְדַבֵּ֣ר מֹשֶׁ֔ה לִפְנֵ֥י יְהוָ֖ה לֵאמֹ֑ר הֵ֤ן בְּנֵֽי־יִשְׂרָאֵל֙ לֹֽא־שָׁמְע֣וּ אֵלַ֔י וְאֵיךְ֙ יִשְׁמָעֵ֣נִי פַרְעֹ֔ה וַֽאֲנִ֖י עֲרַ֥ל שְׂפָתָֽיִם׃ {פ} וּמַלֵּיל מֹשֶׁה, קֳדָם יְיָ לְמֵימַר:  הָא בְּנֵי יִשְׂרָאֵל, לָא קַבִּילוּ מִנִּי, וְאֵיכְדֵין יְקַבֵּיל מִנִּי פַּרְעֹה, וַאֲנָא יַקִּיר מַמְלַל. {פ}
ו,יג וַיְדַבֵּ֣ר יְהוָה֮ אֶל־מֹשֶׁ֣ה וְאֶֽל־אַהֲרֹן֒ וַיְצַוֵּם֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל וְאֶל־פַּרְעֹ֖ה מֶ֣לֶךְ מִצְרָ֑יִם לְהוֹצִ֥יא אֶת־בְּנֵֽי־יִשְׂרָאֵ֖ל מֵאֶ֥רֶץ מִצְרָֽיִם׃ {ס} וּמַלֵּיל יְיָ, עִם מֹשֶׁה וּלְאַהֲרוֹן, וּפַקֵּידִנּוּן לְוָת בְּנֵי יִשְׂרָאֵל, וּלְוָת פַּרְעֹה מַלְכָּא דְּמִצְרָיִם--לְאַפָּקָא יָת בְּנֵי יִשְׂרָאֵל, מֵאַרְעָא דְּמִצְרָיִם. {ס}
ו,יד אֵ֖לֶּה רָאשֵׁ֣י בֵית־אֲבֹתָ֑ם בְּנֵ֨י רְאוּבֵ֜ן בְּכֹ֣ר יִשְׂרָאֵ֗ל חֲנ֤וֹךְ וּפַלּוּא֙ חֶצְרֹ֣ן וְכַרְמִ֔י אֵ֖לֶּה מִשְׁפְּחֹ֥ת רְאוּבֵֽן׃ אִלֵּין, רֵישֵׁי בֵּית אֲבָהָתְהוֹן:  בְּנֵי רְאוּבֵן בֻּכְרָא דְּיִשְׂרָאֵל, חֲנוֹךְ וּפַלּוּא חֶצְרוֹן וְכַרְמִי--אִלֵּין, זַרְעֲיָת רְאוּבֵן. 
ו,טו וּבְנֵ֣י שִׁמְע֗וֹן יְמוּאֵ֨ל וְיָמִ֤ין וְאֹ֨הַד֙ וְיָכִ֣ין וְצֹ֔חַר וְשָׁא֖וּל בֶּן־הַֽכְּנַעֲנִ֑ית אֵ֖לֶּה מִשְׁפְּחֹ֥ת שִׁמְעֽוֹן׃ וּבְנֵי שִׁמְעוֹן, יְמוּאֵל וְיָמִין וְאֹהַד וְיָכִין וְצֹחַר, וְשָׁאוּל, בַּר כְּנַעֲנֵיתָא; אִלֵּין, זַרְעֲיָת שִׁמְעוֹן. 
ו,טז וְאֵ֨לֶּה שְׁמ֤וֹת בְּנֵֽי־לֵוִי֙ לְתֹ֣לְדֹתָ֔ם גֵּֽרְשׁ֕וֹן וּקְהָ֖ת וּמְרָרִ֑י וּשְׁנֵי֙ חַיֵּ֣י לֵוִ֔י שֶׁ֧בַע וּשְׁלֹשִׁ֛ים וּמְאַ֖ת שָׁנָֽה׃ וְאִלֵּין שְׁמָהָת בְּנֵי לֵוִי, לְתוֹלְדָתְהוֹן--גֵּרְשׁוֹן, וּקְהָת וּמְרָרִי; וּשְׁנֵי חַיֵּי לֵוִי, מְאָה וּתְלָתִין וּשְׁבַע שְׁנִין. 
ו,יז בְּנֵ֥י גֵֽרְשׁ֛וֹן לִבְנִ֥י וְשִׁמְעִ֖י לְמִשְׁפְּחֹתָֽם׃ בְּנֵי גֵּרְשׁוֹן לִבְנִי וְשִׁמְעִי, לְזַרְעֲיָתְהוֹן. 
ו,יח וּבְנֵ֣י קְהָ֔ת עַמְרָ֣ם וְיִצְהָ֔ר וְחֶבְר֖וֹן וְעֻזִּיאֵ֑ל וּשְׁנֵי֙ חַיֵּ֣י קְהָ֔ת שָׁלֹ֧שׁ וּשְׁלֹשִׁ֛ים וּמְאַ֖ת שָׁנָֽה׃ וּבְנֵי קְהָת--עַמְרָם וְיִצְהָר, וְחֶבְרוֹן וְעֻזִּיאֵל; וּשְׁנֵי חַיֵּי קְהָת, מְאָה וּתְלָתִין וּתְלָת שְׁנִין. 
ו,יט וּבְנֵ֥י מְרָרִ֖י מַחְלִ֣י וּמוּשִׁ֑י אֵ֛לֶּה מִשְׁפְּחֹ֥ת הַלֵּוִ֖י לְתֹֽלְדֹתָֽם׃ וּבְנֵי מְרָרִי, מַחְלִי וּמוּשִׁי; אִלֵּין זַרְעֲיָת לֵוִי, לְתוֹלְדָתְהוֹן. 
ו,כ וַיִּקַּ֨ח עַמְרָ֜ם אֶת־יוֹכֶ֤בֶד דֹּֽדָתוֹ֙ ל֣וֹ לְאִשָּׁ֔ה וַתֵּ֣לֶד ל֔וֹ אֶֽת־אַהֲרֹ֖ן וְאֶת־מֹשֶׁ֑ה וּשְׁנֵי֙ חַיֵּ֣י עַמְרָ֔ם שֶׁ֧בַע וּשְׁלֹשִׁ֛ים וּמְאַ֖ת שָׁנָֽה׃ וּנְסֵיב עַמְרָם יָת יוֹכֶבֶד אֲחָת אֲבוּהִי, לֵיהּ לְאִתּוּ, וִילֵידַת לֵיהּ, יָת אַהֲרוֹן וְיָת מֹשֶׁה; וּשְׁנֵי חַיֵּי עַמְרָם, מְאָה וּתְלָתִין וּשְׁבַע שְׁנִין. 
ו,כא וּבְנֵ֖י יִצְהָ֑ר קֹ֥רַח וָנֶ֖פֶג וְזִכְרִֽי׃ וּבְנֵי, יִצְהָר--קֹרַח וָנֶפֶג, וְזִכְרִי. 
ו,כב וּבְנֵ֖י עֻזִּיאֵ֑ל מִֽישָׁאֵ֥ל וְאֶלְצָפָ֖ן וְסִתְרִֽי׃ וּבְנֵי, עֻזִּיאֵל--מִישָׁאֵל וְאֶלְצָפָן, וְסִתְרִי. 
ו,כג וַיִּקַּ֨ח אַֽהֲרֹ֜ן אֶת־אֱלִישֶׁ֧בַע בַּת־עַמִּֽינָדָ֛ב אֲח֥וֹת נַחְשׁ֖וֹן ל֣וֹ לְאִשָּׁ֑ה וַתֵּ֣לֶד ל֗וֹ אֶת־נָדָב֙ וְאֶת־אֲבִיה֔וּא אֶת־אֶלְעָזָ֖ר וְאֶת־אִֽיתָמָֽר׃ וּנְסֵיב אַהֲרוֹן יָת אֱלִישֶׁבַע בַּת עַמִּינָדָב, אֲחָתֵיהּ דְּנַחְשׁוֹן--לֵיהּ לְאִתּוּ; וִילֵידַת לֵיהּ, יָת נָדָב וְיָת אֲבִיהוּא, יָת אֶלְעָזָר, וְיָת אִיתָמָר. 
ו,כד וּבְנֵ֣י קֹ֔רַח אַסִּ֥יר וְאֶלְקָנָ֖ה וַאֲבִֽיאָסָ֑ף אֵ֖לֶּה מִשְׁפְּחֹ֥ת הַקָּרְחִֽי׃ וּבְנֵי קֹרַח, אַסִּיר וְאֶלְקָנָה וַאֲבִיאָסָף; אִלֵּין, זַרְעֲיָת קֹרַח. 
ו,כה וְאֶלְעָזָ֨ר בֶּֽן־אַהֲרֹ֜ן לָקַֽח־ל֨וֹ מִבְּנ֤וֹת פּֽוּטִיאֵל֙ ל֣וֹ לְאִשָּׁ֔ה וַתֵּ֥לֶד ל֖וֹ אֶת־פִּֽינְחָ֑ס אֵ֗לֶּה רָאשֵׁ֛י אֲב֥וֹת הַלְוִיִּ֖ם לְמִשְׁפְּחֹתָֽם׃ וְאֶלְעָזָר בַּר אַהֲרוֹן נְסֵיב לֵיהּ מִבְּנָת פּוּטִיאֵל, לֵיהּ לְאִתּוּ, וִילֵידַת לֵיהּ, יָת פִּינְחָס; אִלֵּין, רֵישֵׁי אֲבָהָת לֵיוָאֵי--לְזַרְעֲיָתְהוֹן. 
ו,כו ה֥וּא אַֽהֲרֹ֖ן וּמֹשֶׁ֑ה אֲשֶׁ֨ר אָמַ֤ר יְהוָה֙ לָהֶ֔ם הוֹצִ֜יאוּ אֶת־בְּנֵ֧י יִשְׂרָאֵ֛ל מֵאֶ֥רֶץ מִצְרַ֖יִם עַל־צִבְאֹתָֽם׃ הוּא אַהֲרוֹן, וּמֹשֶׁה--דַּאֲמַר יְיָ, לְהוֹן, אַפִּיקוּ יָת בְּנֵי יִשְׂרָאֵל מֵאַרְעָא דְּמִצְרַיִם, עַל חֵילֵיהוֹן. 
ו,כז הֵ֗ם הַֽמְדַבְּרִים֙ אֶל־פַּרְעֹ֣ה מֶֽלֶךְ־מִצְרַ֔יִם לְהוֹצִ֥יא אֶת־בְּנֵֽי־יִשְׂרָאֵ֖ל מִמִּצְרָ֑יִם ה֥וּא מֹשֶׁ֖ה וְאַֽהֲרֹֽן׃ אִנּוּן, דִּמְמַלְּלִין עִם פַּרְעֹה מַלְכָּא דְּמִצְרַיִם, לְאַפָּקָא יָת בְּנֵי יִשְׂרָאֵל, מִמִּצְרָיִם; הוּא מֹשֶׁה, וְאַהֲרוֹן. 
ו,כח וַיְהִ֗י בְּי֨וֹם דִּבֶּ֧ר יְהוָ֛ה אֶל־מֹשֶׁ֖ה בְּאֶ֥רֶץ מִצְרָֽיִם׃ {ס} וַהֲוָה, בְּיוֹמָא דְּמַלֵּיל יְיָ עִם מֹשֶׁה--בְּאַרְעָא דְּמִצְרָיִם. {ס}
ו,כט וַיְדַבֵּ֧ר יְהוָ֛ה אֶל־מֹשֶׁ֥ה לֵּאמֹ֖ר אֲנִ֣י יְהוָ֑ה דַּבֵּ֗ר אֶל־פַּרְעֹה֙ מֶ֣לֶךְ מִצְרַ֔יִם אֵ֛ת כָּל־אֲשֶׁ֥ר אֲנִ֖י דֹּבֵ֥ר אֵלֶֽיךָ׃ וּמַלֵּיל יְיָ עִם מֹשֶׁה לְמֵימַר, אֲנָא יְיָ; מַלֵּיל, עִם פַּרְעֹה מַלְכָּא דְּמִצְרַיִם, יָת כָּל דַּאֲנָא, מְמַלֵּיל עִמָּךְ. 
ו,ל וַיֹּ֥אמֶר מֹשֶׁ֖ה לִפְנֵ֣י יְהוָ֑ה הֵ֤ן אֲנִי֙ עֲרַ֣ל שְׂפָתַ֔יִם וְאֵ֕יךְ יִשְׁמַ֥ע אֵלַ֖י פַּרְעֹֽה׃ {פ} וַאֲמַר מֹשֶׁה, קֳדָם יְיָ:  הָא אֲנָא, יַקִּיר מַמְלַל, וְאֵיכְדֵין, יְקַבֵּיל מִנִּי פַּרְעֹה. {פ}
ז,א וַיֹּ֤אמֶר יְהוָה֙ אֶל־מֹשֶׁ֔ה רְאֵ֛ה נְתַתִּ֥יךָ אֱלֹהִ֖ים לְפַרְעֹ֑ה וְאַֽהֲרֹ֥ן אָחִ֖יךָ יִֽהְיֶ֥ה נְבִיאֶֽךָ׃ וַאֲמַר יְיָ לְמֹשֶׁה, חֲזִי דְּמַנִּיתָךְ רָב לְפַרְעֹה; וְאַהֲרוֹן אֲחוּךְ, יְהֵי מְתֻרְגְּמָנָךְ. 
ז,ב אַתָּ֣ה תְדַבֵּ֔ר אֵ֖ת כָּל־אֲשֶׁ֣ר אֲצַוֶּ֑ךָּ וְאַֽהֲרֹ֤ן אָחִ֨יךָ֙ יְדַבֵּ֣ר אֶל־פַּרְעֹ֔ה וְשִׁלַּ֥ח אֶת־בְּנֵֽי־יִשְׂרָאֵ֖ל מֵֽאַרְצֽוֹ׃ אַתְּ תְּמַלֵּיל, יָת כָּל דַּאֲפַקְּדִנָּךְ; וְאַהֲרוֹן אֲחוּךְ יְמַלֵּיל עִם פַּרְעֹה, וִישַׁלַּח יָת בְּנֵי יִשְׂרָאֵל מֵאַרְעֵיהּ. 
ז,ג וַֽאֲנִ֥י אַקְשֶׁ֖ה אֶת־לֵ֣ב פַּרְעֹ֑ה וְהִרְבֵּיתִ֧י אֶת־אֹֽתֹתַ֛י וְאֶת־מֽוֹפְתַ֖י בְּאֶ֥רֶץ מִצְרָֽיִם׃ וַאֲנָא אַקְשֵׁי, יָת לִבָּא דְּפַרְעֹה; וְאַסְגֵּי יָת אָתְוָתַי וְיָת מוֹפְתַי, בְּאַרְעָא דְּמִצְרָיִם. 
ז,ד וְלֹֽא־יִשְׁמַ֤ע אֲלֵכֶם֙ פַּרְעֹ֔ה וְנָֽתַתִּ֥י אֶת־יָדִ֖י בְּמִצְרָ֑יִם וְהֽוֹצֵאתִ֨י אֶת־צִבְאֹתַ֜י אֶת־עַמִּ֤י בְנֵֽי־יִשְׂרָאֵל֙ מֵאֶ֣רֶץ מִצְרַ֔יִם בִּשְׁפָטִ֖ים גְּדֹלִֽים׃ וְלָא יְקַבֵּיל מִנְּכוֹן פַּרְעֹה, וְאֶתֵּין יָת מַחַת גְּבוּרְתִי בְּמִצְרָיִם; וְאַפֵּיק יָת חֵילַי יָת עַמִּי בְּנֵי יִשְׂרָאֵל, מֵאַרְעָא דְּמִצְרַיִם, בְּדִינִין, רַבְרְבִין. 
ז,ה וְיָֽדְע֤וּ מִצְרַ֨יִם֙ כִּֽי־אֲנִ֣י יְהוָ֔ה בִּנְטֹתִ֥י אֶת־יָדִ֖י עַל־מִצְרָ֑יִם וְהֽוֹצֵאתִ֥י אֶת־בְּנֵֽי־יִשְׂרָאֵ֖ל מִתּוֹכָֽם׃ וְיִדְּעוּן מִצְרָאֵי אֲרֵי אֲנָא יְיָ, כַּד אֲרֵים יָת מַחַת גְּבוּרְתִי עַל מִצְרָיִם; וְאַפֵּיק יָת בְּנֵי יִשְׂרָאֵל, מִבֵּינֵיהוֹן. 
ז,ו וַיַּ֥עַשׂ מֹשֶׁ֖ה וְאַֽהֲרֹ֑ן כַּֽאֲשֶׁ֨ר צִוָּ֧ה יְהוָ֛ה אֹתָ֖ם כֵּ֥ן עָשֽׂוּ׃ וַעֲבַד מֹשֶׁה, וְאַהֲרוֹן--כְּמָא דְּפַקֵּיד יְיָ יָתְהוֹן, כֵּין עֲבַדוּ. 
ז,ז וּמֹשֶׁה֙ בֶּן־שְׁמֹנִ֣ים שָׁנָ֔ה וְאַֽהֲרֹ֔ן בֶּן־שָׁלֹ֥שׁ וּשְׁמֹנִ֖ים שָׁנָ֑ה בְּדַבְּרָ֖ם אֶל־פַּרְעֹֽה׃ {פ} וּמֹשֶׁה, בַּר תְּמָנַן שְׁנִין, וְאַהֲרוֹן, בַּר תְּמָנַן וּתְלָת שְׁנִין--בְּמַלָּלוּתְהוֹן, עִם פַּרְעֹה. {פ}
ז,ח וַיֹּ֣אמֶר יְהוָ֔ה אֶל־מֹשֶׁ֥ה וְאֶֽל־אַהֲרֹ֖ן לֵאמֹֽר׃ וַאֲמַר יְיָ, לְמֹשֶׁה וּלְאַהֲרוֹן לְמֵימַר. 
ז,ט כִּי֩ יְדַבֵּ֨ר אֲלֵכֶ֤ם פַּרְעֹה֙ לֵאמֹ֔ר תְּנ֥וּ לָכֶ֖ם מוֹפֵ֑ת וְאָֽמַרְתָּ֣ אֶֽל־אַהֲרֹ֗ן קַ֧ח אֶֽת־מַטְּךָ֛ וְהַשְׁלֵ֥ךְ לִפְנֵֽי־פַרְעֹ֖ה יְהִ֥י לְתַנִּֽין׃ אֲרֵי יְמַלֵּיל עִמְּכוֹן פַּרְעֹה לְמֵימַר, הַבוּ לְכוֹן אָתָא; וְתֵימַר לְאַהֲרוֹן, סַב יָת חֻטְרָךְ וּרְמִי קֳדָם פַּרְעֹה--יְהֵי לְתַנִּינָא. 
ז,י וַיָּבֹ֨א מֹשֶׁ֤ה וְאַֽהֲרֹן֙ אֶל־פַּרְעֹ֔ה וַיַּ֣עֲשׂוּ כֵ֔ן כַּֽאֲשֶׁ֖ר צִוָּ֣ה יְהוָ֑ה וַיַּשְׁלֵ֨ךְ אַֽהֲרֹ֜ן אֶת־מַטֵּ֗הוּ לִפְנֵ֥י פַרְעֹ֛ה וְלִפְנֵ֥י עֲבָדָ֖יו וַיְהִ֥י לְתַנִּֽין׃ וְעָאל מֹשֶׁה וְאַהֲרוֹן, לְוָת פַּרְעֹה, וַעֲבַדוּ כֵּין, כְּמָא דְּפַקֵּיד יְיָ; וּרְמָא אַהֲרוֹן יָת חֻטְרֵיהּ, קֳדָם פַּרְעֹה וּקְדָם עַבְדּוֹהִי--וַהֲוָה לְתַנִּינָא. 
ז,יא וַיִּקְרָא֙ גַּם־פַּרְעֹ֔ה לַֽחֲכָמִ֖ים וְלַֽמְכַשְּׁפִ֑ים וַיַּֽעֲשׂ֨וּ גַם־הֵ֜ם חַרְטֻמֵּ֥י מִצְרַ֛יִם בְּלַֽהֲטֵיהֶ֖ם כֵּֽן׃ וּקְרָא, אַף פַּרְעֹה, לְחַכִּימַיָּא, וּלְחָרָשַׁיָּא; וַעֲבַדוּ אַף אִנּוּן חָרָשֵׁי מִצְרַיִם, בְּלַחֲשֵׁיהוֹן--כֵּין. 
ז,יב וַיַּשְׁלִ֨יכוּ֙ אִ֣ישׁ מַטֵּ֔הוּ וַיִּֽהְי֖וּ לְתַנִּינִ֑ם וַיִּבְלַ֥ע מַטֵּֽה־אַהֲרֹ֖ן אֶת־מַטֹּתָֽם׃ וּרְמוֹ גְּבַר חֻטְרֵיהּ, וַהֲווֹ לְתַנִּינִין; וּבְלַע חֻטְרָא דְּאַהֲרוֹן, יָת חֻטְרֵיהוֹן. 
ז,יג וַיֶּֽחֱזַק֙ לֵ֣ב פַּרְעֹ֔ה וְלֹ֥א שָׁמַ֖ע אֲלֵהֶ֑ם כַּֽאֲשֶׁ֖ר דִּבֶּ֥ר יְהוָֽה׃ {ס} וְאִתַּקַּף לִבָּא דְּפַרְעֹה, וְלָא קַבֵּיל מִנְּהוֹן:  כְּמָא דְּמַלֵּיל יְיָ. {ס}
ז,יד וַיֹּ֤אמֶר יְהוָה֙ אֶל־מֹשֶׁ֔ה כָּבֵ֖ד לֵ֣ב פַּרְעֹ֑ה מֵאֵ֖ן לְשַׁלַּ֥ח הָעָֽם׃ וַאֲמַר יְיָ לְמֹשֶׁה, אִתְיַקַּר לִבָּא דְּפַרְעֹה; סָרֵיב, לְשַׁלָּחָא עַמָּא. 
ז,טו לֵ֣ךְ אֶל־פַּרְעֹ֞ה בַּבֹּ֗קֶר הִנֵּה֙ יֹצֵ֣א הַמַּ֔יְמָה וְנִצַּבְתָּ֥ לִקְרָאת֖וֹ עַל־שְׂפַ֣ת הַיְאֹ֑ר וְהַמַּטֶּ֛ה אֲשֶׁר־נֶהְפַּ֥ךְ לְנָחָ֖שׁ תִּקַּ֥ח בְּיָדֶֽךָ׃ אִיזֵיל לְוָת פַּרְעֹה בְּצַפְרָא, הָא נָפֵיק לְמַיָּא, וְתִתְעַתַּד לְקַדָּמוּתֵיהּ, עַל כֵּיף נַהְרָא; וְחֻטְרָא דְּאִתְהֲפֵיךְ לְחִוְיָא, תִּסַּב בִּידָךְ. 
ז,טז וְאָֽמַרְתָּ֣ אֵלָ֗יו יְהוָ֞ה אֱלֹהֵ֤י הָֽעִבְרִים֙ שְׁלָחַ֤נִי אֵלֶ֨יךָ֙ לֵאמֹ֔ר שַׁלַּח֙ אֶת־עַמִּ֔י וְיַֽעַבְדֻ֖נִי בַּמִּדְבָּ֑ר וְהִנֵּ֥ה לֹֽא־שָׁמַ֖עְתָּ עַד־כֹּֽה׃ וְתֵימַר לֵיהּ, יְיָ אֱלָהָא דִּיהוּדָאֵי שַׁלְחַנִי לְוָתָךְ לְמֵימַר, שַׁלַּח יָת עַמִּי, וְיִפְלְחוּן קֳדָמַי בְּמַדְבְּרָא; וְהָא לָא קַבֵּילְתָּא, עַד כְּעַן. 
ז,יז כֹּ֚ה אָמַ֣ר יְהוָ֔ה בְּזֹ֣את תֵּדַ֔ע כִּ֖י אֲנִ֣י יְהוָ֑ה הִנֵּ֨ה אָֽנֹכִ֜י מַכֶּ֣ה ׀ בַּמַּטֶּ֣ה אֲשֶׁר־בְּיָדִ֗י עַל־הַמַּ֛יִם אֲשֶׁ֥ר בַּיְאֹ֖ר וְנֶֽהֶפְכ֥וּ לְדָֽם׃ כִּדְנָן, אֲמַר יְיָ, בְּדָא תִּדַּע, אֲרֵי אֲנָא יְיָ:  הָא אֲנָא מָחֵי בְּחֻטְרָא דְּבִידִי, עַל מַיָּא דִּבְנַהְרָא--וְיִתְהַפְכוּן לִדְמָא. 
ז,יח וְהַדָּגָ֧ה אֲשֶׁר־בַּיְאֹ֛ר תָּמ֖וּת וּבָאַ֣שׁ הַיְאֹ֑ר וְנִלְא֣וּ מִצְרַ֔יִם לִשְׁתּ֥וֹת מַ֖יִם מִן־הַיְאֹֽר׃ {ס} וְנוּנֵי דִּבְנַהְרָא יְמוּתוּן, וְיִסְרֵי נַהְרָא; וְיִלְאוֹן מִצְרָאֵי, לְמִשְׁתֵּי מַיָּא מִן נַהְרָא. {ס}
ז,יט וַיֹּ֨אמֶר יְהוָ֜ה אֶל־מֹשֶׁ֗ה אֱמֹ֣ר אֶֽל־אַהֲרֹ֡ן קַ֣ח מַטְּךָ֣ וּנְטֵֽה־יָדְךָ֩ עַל־מֵימֵ֨י מִצְרַ֜יִם עַֽל־נַהֲרֹתָ֣ם ׀ עַל־יְאֹֽרֵיהֶ֣ם וְעַל־אַגְמֵיהֶ֗ם וְעַ֛ל כָּל־מִקְוֵ֥ה מֵֽימֵיהֶ֖ם וְיִֽהְיוּ־דָ֑ם וְהָ֤יָה דָם֙ בְּכָל־אֶ֣רֶץ מִצְרַ֔יִם וּבָֽעֵצִ֖ים וּבָֽאֲבָנִֽים׃ וַאֲמַר יְיָ לְמֹשֶׁה, אֵימַר לְאַהֲרוֹן סַב חֻטְרָךְ וַאֲרֵים יְדָךְ עַל מַיָּא דְּמִצְרָאֵי עַל נַהְרֵיהוֹן עַל אֲרִתֵּיהוֹן וְעַל אַגְמֵיהוֹן וְעַל כָּל בֵּית כְּנֵישָׁת מֵימֵיהוֹן--וִיהוֹן דְּמָא; וִיהֵי דְּמָא בְּכָל אַרְעָא דְּמִצְרַיִם, וּבְמָנֵי אָעָא וּבְמָנֵי אַבְנָא. 
ז,כ וַיַּֽעֲשׂוּ־כֵן֩ מֹשֶׁ֨ה וְאַֽהֲרֹ֜ן כַּֽאֲשֶׁ֣ר ׀ צִוָּ֣ה יְהוָ֗ה וַיָּ֤רֶם בַּמַּטֶּה֙ וַיַּ֤ךְ אֶת־הַמַּ֨יִם֙ אֲשֶׁ֣ר בַּיְאֹ֔ר לְעֵינֵ֣י פַרְעֹ֔ה וּלְעֵינֵ֖י עֲבָדָ֑יו וַיֵּהָ֥פְכ֛וּ כָּל־הַמַּ֥יִם אֲשֶׁר־בַּיְאֹ֖ר לְדָֽם׃ וַעֲבַדוּ כֵּין מֹשֶׁה וְאַהֲרוֹן כְּמָא דְּפַקֵּיד יְיָ, וַאֲרֵים בְּחֻטְרָא וּמְחָא יָת מַיָּא דִּבְנַהְרָא, לְעֵינֵי פַּרְעֹה, וּלְעֵינֵי עַבְדּוֹהִי; וְאִתְהֲפִיכוּ כָּל מַיָּא דִּבְנַהְרָא, לִדְמָא. 
ז,כא וְהַדָּגָ֨ה אֲשֶׁר־בַּיְאֹ֥ר מֵ֨תָה֙ וַיִּבְאַ֣שׁ הַיְאֹ֔ר וְלֹֽא־יָכְל֣וּ מִצְרַ֔יִם לִשְׁתּ֥וֹת מַ֖יִם מִן־הַיְאֹ֑ר וַיְהִ֥י הַדָּ֖ם בְּכָל־אֶ֥רֶץ מִצְרָֽיִם׃ וְנוּנֵי דִּבְנַהְרָא מִיתוּ, וּסְרִי נַהְרָא, וְלָא יְכִילוּ מִצְרָאֵי, לְמִשְׁתֵּי מַיָּא מִן נַהְרָא; וַהֲוָה דְּמָא, בְּכָל אַרְעָא דְּמִצְרָיִם. 
ז,כב וַיַּֽעֲשׂוּ־כֵ֛ן חַרְטֻמֵּ֥י מִצְרַ֖יִם בְּלָֽטֵיהֶ֑ם וַיֶּֽחֱזַ֤ק לֵב־פַּרְעֹה֙ וְלֹֽא־שָׁמַ֣ע אֲלֵהֶ֔ם כַּֽאֲשֶׁ֖ר דִּבֶּ֥ר יְהוָֽה׃ וַעֲבַדוּ כֵּין חָרָשֵׁי מִצְרַיִם, בְּלַחֲשֵׁיהוֹן; וְאִתַּקַּף לִבָּא דְּפַרְעֹה וְלָא קַבֵּיל מִנְּהוֹן, כְּמָא דְּמַלֵּיל יְיָ. 
ז,כג וַיִּ֣פֶן פַּרְעֹ֔ה וַיָּבֹ֖א אֶל־בֵּית֑וֹ וְלֹא־שָׁ֥ת לִבּ֖וֹ גַּם־לָזֹֽאת׃ וְאִתְפְּנִי פַּרְעֹה, וְעָאל לְבֵיתֵיהּ; וְלָא שַׁוִּי לִבֵּיהּ, אַף לְדָא. 
ז,כד וַיַּחְפְּר֧וּ כָל־מִצְרַ֛יִם סְבִיבֹ֥ת הַיְאֹ֖ר מַ֣יִם לִשְׁתּ֑וֹת כִּ֣י לֹ֤א יָֽכְלוּ֙ לִשְׁתֹּ֔ת מִמֵּימֵ֖י הַיְאֹֽר׃ וַחֲפַרוּ כָּל מִצְרָאֵי סַחְרָנוּת נַהְרָא, מַיָּא לְמִשְׁתֵּי:  אֲרֵי לָא יְכִילוּ לְמִשְׁתֵּי, מִמַּיָּא דִּבְנַהְרָא. 
ז,כה וַיִּמָּלֵ֖א שִׁבְעַ֣ת יָמִ֑ים אַֽחֲרֵ֥י הַכּוֹת־יְהוָ֖ה אֶת־הַיְאֹֽר׃ {פ} וּשְׁלִימוּ, שִׁבְעָא יוֹמִין, בָּתַר דִּמְחָא יְיָ, יָת נַהְרָא. {פ}
ז,כו וַיֹּ֤אמֶר יְהוָה֙ אֶל־מֹשֶׁ֔ה בֹּ֖א אֶל־פַּרְעֹ֑ה וְאָֽמַרְתָּ֣ אֵלָ֗יו כֹּ֚ה אָמַ֣ר יְהוָ֔ה שַׁלַּ֥ח אֶת־עַמִּ֖י וְיַֽעַבְדֻֽנִי׃ וַאֲמַר יְיָ לְמֹשֶׁה, עוֹל לְוָת פַּרְעֹה; וְתֵימַר לֵיהּ, כִּדְנָן אֲמַר יְיָ, שַׁלַּח יָת עַמִּי, וְיִפְלְחוּן קֳדָמָי. 
ז,כז וְאִם־מָאֵ֥ן אַתָּ֖ה לְשַׁלֵּ֑חַ הִנֵּ֣ה אָֽנֹכִ֗י נֹגֵ֛ף אֶת־כָּל־גְּבֽוּלְךָ֖ בַּֽצְפַרְדְּעִֽים׃ וְאִם מְסָרֵיב אַתְּ, לְשַׁלָּחָא:  הָא אֲנָא, מָחֵי יָת כָּל תְּחוּמָךְ--בְּעֻרְדְּעָנַיָּא. 
ז,כח וְשָׁרַ֣ץ הַיְאֹר֮ צְפַרְדְּעִים֒ וְעָלוּ֙ וּבָ֣אוּ בְּבֵיתֶ֔ךָ וּבַֽחֲדַ֥ר מִשְׁכָּֽבְךָ֖ וְעַל־מִטָּתֶ֑ךָ וּבְבֵ֤ית עֲבָדֶ֨יךָ֙ וּבְעַמֶּ֔ךָ וּבְתַנּוּרֶ֖יךָ וּבְמִשְׁאֲרוֹתֶֽיךָ׃ וִירַבֵּי נַהְרָא, עֻרְדְּעָנַיָּא, וְיִסְּקוּן וְיֵיעֲלוּן בְּבֵיתָךְ, וּבְאִדְּרוֹן בֵּית מִשְׁכְּבָךְ וְעַל עַרְסָתָךְ; וּבְבֵית עַבְדָךְ וּבְעַמָּךְ, וּבְתַנּוּרָךְ וּבְאָצְוָתָךְ. 
ז,כט וּבְכָ֥ה וּֽבְעַמְּךָ֖ וּבְכָל־עֲבָדֶ֑יךָ יַֽעֲל֖וּ הַֽצְפַרְדְּעִֽים׃ וּבָךְ וּבְעַמָּךְ, וּבְכָל עַבְדָךְ--יִסְּקוּן, עֻרְדְּעָנַיָּא. 
ח,א וַיֹּ֣אמֶר יְהוָה֮ אֶל־מֹשֶׁה֒ אֱמֹ֣ר אֶֽל־אַהֲרֹ֗ן נְטֵ֤ה אֶת־יָֽדְךָ֙ בְּמַטֶּ֔ךָ עַ֨ל־הַנְּהָרֹ֔ת עַל־הַיְאֹרִ֖ים וְעַל־הָֽאֲגַמִּ֑ים וְהַ֥עַל אֶת־הַֽצְפַרְדְּעִ֖ים עַל־אֶ֥רֶץ מִצְרָֽיִם׃ וַאֲמַר יְיָ, לְמֹשֶׁה, אֵימַר לְאַהֲרוֹן אֲרֵים יָת יְדָךְ בְּחֻטְרָךְ, עַל נַהְרַיָּא עַל אֲרִתַּיָּא וְעַל אַגְמַיָּא; וְאַסֵּיק יָת עֻרְדְּעָנַיָּא, עַל אַרְעָא דְּמִצְרָיִם. 
ח,ב וַיֵּ֤ט אַֽהֲרֹן֙ אֶת־יָד֔וֹ עַ֖ל מֵימֵ֣י מִצְרָ֑יִם וַתַּ֨עַל֙ הַצְּפַרְדֵּ֔עַ וַתְּכַ֖ס אֶת־אֶ֥רֶץ מִצְרָֽיִם׃ וַאֲרֵים אַהֲרוֹן יָת יְדֵיהּ, עַל מַיָּא דְּמִצְרָאֵי; וּסְלִיקוּ, עֻרְדְּעָנַיָּא, וַחֲפוֹ, יָת אַרְעָא דְּמִצְרָיִם. 
ח,ג וַיַּֽעֲשׂוּ־כֵ֥ן הַֽחַרְטֻמִּ֖ים בְּלָֽטֵיהֶ֑ם וַיַּֽעֲל֥וּ אֶת־הַֽצְפַרְדְּעִ֖ים עַל־אֶ֥רֶץ מִצְרָֽיִם׃ וַעֲבַדוּ כֵּין חָרָשַׁיָּא, בְּלַחֲשֵׁיהוֹן; וְאַסִּיקוּ יָת עֻרְדְּעָנַיָּא, עַל אַרְעָא דְּמִצְרָיִם. 
ח,ד וַיִּקְרָ֨א פַרְעֹ֜ה לְמֹשֶׁ֣ה וּֽלְאַהֲרֹ֗ן וַיֹּ֨אמֶר֙ הַעְתִּ֣ירוּ אֶל־יְהוָ֔ה וְיָסֵר֙ הַֽצְפַרְדְּעִ֔ים מִמֶּ֖נִּי וּמֵֽעַמִּ֑י וַֽאֲשַׁלְּחָה֙ אֶת־הָעָ֔ם וְיִזְבְּח֖וּ לַֽיהוָֽה׃ וּקְרָא פַּרְעֹה לְמֹשֶׁה וּלְאַהֲרוֹן, וַאֲמַר צַלּוֹ קֳדָם יְיָ, וְיַעְדֵּי עֻרְדְּעָנַיָּא, מִנִּי וּמֵעַמִּי; וַאֲשַׁלַּח, יָת עַמָּא, וִידַבְּחוּן, קֳדָם יְיָ. 
ח,ה וַיֹּ֨אמֶר מֹשֶׁ֣ה לְפַרְעֹה֮ הִתְפָּאֵ֣ר עָלַי֒ לְמָתַ֣י ׀ אַעְתִּ֣יר לְךָ֗ וְלַֽעֲבָדֶ֨יךָ֙ וּֽלְעַמְּךָ֔ לְהַכְרִית֙ הַֽצְפַרְדְּעִ֔ים מִמְּךָ֖ וּמִבָּתֶּ֑יךָ רַ֥ק בַּיְאֹ֖ר תִּשָּׁאַֽרְנָה׃ וַאֲמַר מֹשֶׁה לְפַרְעֹה, שְׁאַל לָךְ גְּבוּרָא הַב לָךְ זְמָן, לְאִמַּתִּי אֲצַלֵּי עֲלָךְ וְעַל עַבְדָךְ וְעַל עַמָּךְ, לְשֵׁיצָאָה עֻרְדְּעָנַיָּא מִנָּךְ וּמִבָּתָּךְ:  לְחוֹד דִּבְנַהְרָא, יִשְׁתְּאַרוּן. 
ח,ו וַיֹּ֖אמֶר לְמָחָ֑ר וַיֹּ֨אמֶר֙ כִּדְבָ֣רְךָ֔ לְמַ֣עַן תֵּדַ֔ע כִּי־אֵ֖ין כַּֽיהוָ֥ה אֱלֹהֵֽינוּ׃ וַאֲמַר, לִמְחַר; וַאֲמַר, כְּפִתְגָמָךְ--בְּדִיל דְּתִדַּע, אֲרֵי לֵית כַּייָ אֱלָהַנָא. 
ח,ז וְסָר֣וּ הַֽצְפַרְדְּעִ֗ים מִמְּךָ֙ וּמִבָּ֣תֶּ֔יךָ וּמֵֽעֲבָדֶ֖יךָ וּמֵֽעַמֶּ֑ךָ רַ֥ק בַּיְאֹ֖ר תִּשָּׁאַֽרְנָה׃ וְיִעְדּוֹן עֻרְדְּעָנַיָּא, מִנָּךְ וּמִבָּתָּךְ, וּמֵעַבְדָךְ, וּמֵעַמָּךְ:  לְחוֹד דִּבְנַהְרָא, יִשְׁתְּאַרוּן. 
ח,ח וַיֵּצֵ֥א מֹשֶׁ֛ה וְאַֽהֲרֹ֖ן מֵעִ֣ם פַּרְעֹ֑ה וַיִּצְעַ֤ק מֹשֶׁה֙ אֶל־יְהוָ֔ה עַל־דְּבַ֥ר הַֽצְפַרְדְּעִ֖ים אֲשֶׁר־שָׂ֥ם לְפַרְעֹֽה׃ וּנְפַק מֹשֶׁה וְאַהֲרוֹן, מִלְּוָת פַּרְעֹה; וְצַלִּי מֹשֶׁה קֳדָם יְיָ, עַל עֵיסַק עֻרְדְּעָנַיָּא דְּשַׁוִּי לְפַרְעֹה. 
ח,ט וַיַּ֥עַשׂ יְהוָ֖ה כִּדְבַ֣ר מֹשֶׁ֑ה וַיָּמֻ֨תוּ֙ הַֽצְפַרְדְּעִ֔ים מִן־הַבָּתִּ֥ים מִן־הַֽחֲצֵרֹ֖ת וּמִן־הַשָּׂדֹֽת׃ וַעֲבַד יְיָ, כְּפִתְגָמָא דְּמֹשֶׁה; וּמִיתוּ, עֻרְדְּעָנַיָּא, מִן בָּתַּיָּא מִן דָּרָתָא, וּמִן חַקְלָתָא. 
ח,י וַיִּצְבְּר֥וּ אֹתָ֖ם חֳמָרִ֣ם חֳמָרִ֑ם וַתִּבְאַ֖שׁ הָאָֽרֶץ׃ וּכְנַשׁוּ יָתְהוֹן, דְּגוֹרִין דְּגוֹרִין; וּסְרִיאוּ, עַל אַרְעָא. 
ח,יא וַיַּ֣רְא פַּרְעֹ֗ה כִּ֤י הָֽיְתָה֙ הָֽרְוָחָ֔ה וְהַכְבֵּד֙ אֶת־לִבּ֔וֹ וְלֹ֥א שָׁמַ֖ע אֲלֵהֶ֑ם כַּֽאֲשֶׁ֖ר דִּבֶּ֥ר יְהוָֽה׃ {ס} וַחֲזָא פַּרְעֹה, אֲרֵי הֲוָת רְוַחְתָּא, וְיַקַּר יָת לִבֵּיהּ, וְלָא קַבֵּיל מִנְּהוֹן:  כְּמָא דְּמַלֵּיל יְיָ. {ס}
ח,יב וַיֹּ֣אמֶר יְהוָה֮ אֶל־מֹשֶׁה֒ אֱמֹר֙ אֶֽל־אַהֲרֹ֔ן נְטֵ֣ה אֶֽת־מַטְּךָ֔ וְהַ֖ךְ אֶת־עֲפַ֣ר הָאָ֑רֶץ וְהָיָ֥ה לְכִנִּ֖ם בְּכָל־אֶ֥רֶץ מִצְרָֽיִם׃ וַאֲמַר יְיָ, לְמֹשֶׁה, אֵימַר לְאַהֲרוֹן, אֲרֵים יָת חֻטְרָךְ וּמְחִי יָת עַפְרָא דְּאַרְעָא; וִיהֵי לְקַלְמְתָא, בְּכָל אַרְעָא דְּמִצְרָיִם. 
ח,יג וַיַּֽעֲשׂוּ־כֵ֗ן וַיֵּט֩ אַֽהֲרֹ֨ן אֶת־יָד֤וֹ בְמַטֵּ֨הוּ֙ וַיַּךְ֙ אֶת־עֲפַ֣ר הָאָ֔רֶץ וַתְּהִי֙ הַכִּנָּ֔ם בָּֽאָדָ֖ם וּבַבְּהֵמָ֑ה כָּל־עֲפַ֥ר הָאָ֛רֶץ הָיָ֥ה כִנִּ֖ים בְּכָל־אֶ֥רֶץ מִצְרָֽיִם׃ וַעֲבַדוּ כֵּין, וַאֲרֵים אַהֲרוֹן יָת יְדֵיהּ בְּחֻטְרֵיהּ וּמְחָא יָת עַפְרָא דְּאַרְעָא, וַהֲוָת קַלְמְתָא, בַּאֲנָשָׁא וּבִבְעִירָא:  כָּל עַפְרָא דְּאַרְעָא הֲוָת קַלְמְתָא, בְּכָל אַרְעָא דְּמִצְרָיִם. 
ח,יד וַיַּֽעֲשׂוּ־כֵ֨ן הַֽחַרְטֻמִּ֧ים בְּלָֽטֵיהֶ֛ם לְהוֹצִ֥יא אֶת־הַכִּנִּ֖ים וְלֹ֣א יָכֹ֑לוּ וַתְּהִי֙ הַכִּנָּ֔ם בָּֽאָדָ֖ם וּבַבְּהֵמָֽה׃ וַעֲבַדוּ כֵּין חָרָשַׁיָּא בְּלַחֲשֵׁיהוֹן לְאַפָּקָא יָת קַלְמְתָא, וְלָא יְכִילוּ; וַהֲוָת, קַלְמְתָא, בַּאֲנָשָׁא, וּבִבְעִירָא. 
ח,טו וַיֹּֽאמְר֤וּ הַֽחַרְטֻמִּם֙ אֶל־פַּרְעֹ֔ה אֶצְבַּ֥ע אֱלֹהִ֖ים הִ֑וא וַיֶּֽחֱזַ֤ק לֵב־פַּרְעֹה֙ וְלֹֽא־שָׁמַ֣ע אֲלֵהֶ֔ם כַּֽאֲשֶׁ֖ר דִּבֶּ֥ר יְהוָֽה׃ {ס} וַאֲמַרוּ חָרָשַׁיָּא לְפַרְעֹה, מַחָא מִן קֳדָם יְיָ הִיא; וְאִתַּקַּף לִבָּא דְּפַרְעֹה וְלָא קַבֵּיל מִנְּהוֹן, כְּמָא דְּמַלֵּיל יְיָ. {ס}
ח,טז וַיֹּ֨אמֶר יְהוָ֜ה אֶל־מֹשֶׁ֗ה הַשְׁכֵּ֤ם בַּבֹּ֨קֶר֙ וְהִתְיַצֵּב֙ לִפְנֵ֣י פַרְעֹ֔ה הִנֵּ֖ה יוֹצֵ֣א הַמָּ֑יְמָה וְאָֽמַרְתָּ֣ אֵלָ֗יו כֹּ֚ה אָמַ֣ר יְהוָ֔ה שַׁלַּ֥ח עַמִּ֖י וְיַֽעַבְדֻֽנִי׃ וַאֲמַר יְיָ לְמֹשֶׁה, אַקְדֵּים בְּצַפְרָא וְאִתְעַתַּד קֳדָם פַּרְעֹה--הָא, נָפֵיק לְמַיָּא; וְתֵימַר לֵיהּ, כִּדְנָן אֲמַר יְיָ, שַׁלַּח עַמִּי, וְיִפְלְחוּן קֳדָמָי. 
ח,יז כִּ֣י אִם־אֵֽינְךָ֮ מְשַׁלֵּ֣חַ אֶת־עַמִּי֒ הִנְנִי֩ מַשְׁלִ֨יחַ בְּךָ֜ וּבַֽעֲבָדֶ֧יךָ וּֽבְעַמְּךָ֛ וּבְבָתֶּ֖יךָ אֶת־הֶֽעָרֹ֑ב וּמָ֨לְא֜וּ בָּתֵּ֤י מִצְרַ֨יִם֙ אֶת־הֶ֣עָרֹ֔ב וְגַ֥ם הָֽאֲדָמָ֖ה אֲשֶׁר־הֵ֥ם עָלֶֽיהָ׃ אֲרֵי אִם לָיְתָךְ, מְשַׁלַּח יָת עַמִּי--הָאֲנָא מַשְׁלַח בָּךְ וּבְעַבְדָךְ וּבְעַמָּךְ וּבְבָתָּךְ, יָת עָרוֹבָא; וְיִתְמְלוֹן בָּתֵּי מִצְרַיִם, יָת עָרוֹבָא, וְאַף אַרְעָא, דְּאִנּוּן עֲלַהּ. 
ח,יח וְהִפְלֵיתִי֩ בַיּ֨וֹם הַה֜וּא אֶת־אֶ֣רֶץ גֹּ֗שֶׁן אֲשֶׁ֤ר עַמִּי֙ עֹמֵ֣ד עָלֶ֔יהָ לְבִלְתִּ֥י הֱיֽוֹת־שָׁ֖ם עָרֹ֑ב לְמַ֣עַן תֵּדַ֔ע כִּ֛י אֲנִ֥י יְהוָ֖ה בְּקֶ֥רֶב הָאָֽרֶץ׃ וְאַפְרֵישׁ בְּיוֹמָא הַהוּא יָת אַרְעָא דְּגֹשֶׁן, דְּעַמִּי שָׁרֵי עֲלַהּ, בְּדִיל דְּלָא לְמִהְוֵי תַּמָּן, עָרוֹבָא--בְּדִיל דְּתִדַּע, אֲרֵי אֲנָא יְיָ שַׁלִּיט בְּגוֹ אַרְעָא. 
ח,יט וְשַׂמְתִּ֣י פְדֻ֔ת בֵּ֥ין עַמִּ֖י וּבֵ֣ין עַמֶּ֑ךָ לְמָחָ֥ר יִֽהְיֶ֖ה הָאֹ֥ת הַזֶּֽה׃ וַאֲשַׁוֵּי פֻּרְקָן לְעַמִּי, וְעַל עַמָּךְ אַיְתִי מַחָא; לִמְחַר יְהֵי, אָתָא הָדֵין. 
ח,כ וַיַּ֤עַשׂ יְהוָה֙ כֵּ֔ן וַיָּבֹא֙ עָרֹ֣ב כָּבֵ֔ד בֵּ֥יתָה פַרְעֹ֖ה וּבֵ֣ית עֲבָדָ֑יו וּבְכָל־אֶ֧רֶץ מִצְרַ֛יִם תִּשָּׁחֵ֥ת הָאָ֖רֶץ מִפְּנֵ֥י הֶֽעָרֹֽב׃ וַעֲבַד יְיָ, כֵּין, וַאֲתָא עָרוֹבָא תַּקִּיף, לְבֵית פַּרְעֹה וּלְבֵית עַבְדּוֹהִי; וּבְכָל אַרְעָא דְּמִצְרַיִם אִתְחַבַּלַת אַרְעָא, מִן קֳדָם עָרוֹבָא. 
ח,כא וַיִּקְרָ֣א פַרְעֹ֔ה אֶל־מֹשֶׁ֖ה וּֽלְאַהֲרֹ֑ן וַיֹּ֗אמֶר לְכ֛וּ זִבְח֥וּ לֵֽאלֹהֵיכֶ֖ם בָּאָֽרֶץ׃ וּקְרָא פַּרְעֹה, לְמֹשֶׁה וּלְאַהֲרוֹן; וַאֲמַר, אִיזִילוּ דַּבַּחוּ קֳדָם אֱלָהֲכוֹן--בְּאַרְעָא. 
ח,כב וַיֹּ֣אמֶר מֹשֶׁ֗ה לֹ֤א נָכוֹן֙ לַֽעֲשׂ֣וֹת כֵּ֔ן כִּ֚י תּֽוֹעֲבַ֣ת מִצְרַ֔יִם נִזְבַּ֖ח לַֽיהוָ֣ה אֱלֹהֵ֑ינוּ הֵ֣ן נִזְבַּ֞ח אֶת־תּֽוֹעֲבַ֥ת מִצְרַ֛יִם לְעֵֽינֵיהֶ֖ם וְלֹ֥א יִסְקְלֻֽנוּ׃ וַאֲמַר מֹשֶׁה, לָא תָּקֵין לְמֶעֱבַד כֵּין, אֲרֵי בְּעִירָא דְּמִצְרָאֵי דָּחֲלִין לֵיהּ, מִנֵּיהּ אֲנַחְנָא נָסְבִין לְדַבָּחָא קֳדָם יְיָ אֱלָהַנָא:  הָא נְדַבַּח יָת בְּעִירָא דְּמִצְרָאֵי דָּחֲלִין לֵיהּ, וְאִנּוּן יְהוֹן חָזַן--הֲלָא יֵימְרוּן לְמִרְגְּמַנָא. 
ח,כג דֶּ֚רֶךְ שְׁלֹ֣שֶׁת יָמִ֔ים נֵלֵ֖ךְ בַּמִּדְבָּ֑ר וְזָבַ֨חְנוּ֙ לַֽיהוָ֣ה אֱלֹהֵ֔ינוּ כַּֽאֲשֶׁ֖ר יֹאמַ֥ר אֵלֵֽינוּ׃ מַהְלַךְ תְּלָתָא יוֹמִין, נֵיזֵיל בְּמַדְבְּרָא; וּנְדַבַּח קֳדָם יְיָ אֱלָהַנָא, כְּמָא דְּיֵימַר לַנָא. 
ח,כד וַיֹּ֣אמֶר פַּרְעֹ֗ה אָֽנֹכִ֞י אֲשַׁלַּ֤ח אֶתְכֶם֙ וּזְבַחְתֶּ֞ם לַֽיהוָ֤ה אֱלֹֽהֵיכֶם֙ בַּמִּדְבָּ֔ר רַ֛ק הַרְחֵ֥ק לֹֽא־תַרְחִ֖יקוּ לָלֶ֑כֶת הַעְתִּ֖ירוּ בַּֽעֲדִֽי׃ וַאֲמַר פַּרְעֹה, אֲנָא אֲשַׁלַּח יָתְכוֹן וּתְדַבְּחוּן קֳדָם יְיָ אֱלָהֲכוֹן בְּמַדְבְּרָא--לְחוֹד אַרְחָקָא לָא תְּרַחֲקוּן, לְמֵיזַל; צַלּוֹ, עֲלָי. 
ח,כה וַיֹּ֣אמֶר מֹשֶׁ֗ה הִנֵּ֨ה אָֽנֹכִ֜י יוֹצֵ֤א מֵֽעִמָּךְ֙ וְהַעְתַּרְתִּ֣י אֶל־יְהוָ֔ה וְסָ֣ר הֶֽעָרֹ֗ב מִפַּרְעֹ֛ה מֵֽעֲבָדָ֥יו וּמֵֽעַמּ֖וֹ מָחָ֑ר רַ֗ק אַל־יֹסֵ֤ף פַּרְעֹה֙ הָתֵ֔ל לְבִלְתִּי֙ שַׁלַּ֣ח אֶת־הָעָ֔ם לִזְבֹּ֖חַ לַֽיהוָֽה׃ וַאֲמַר מֹשֶׁה, הָא אֲנָא נָפֵיק מֵעִמָּךְ וַאֲצַלֵּי קֳדָם יְיָ, וְיִעְדֵּי עָרוֹבָא מִפַּרְעֹה מֵעַבְדּוֹהִי וּמֵעַמֵּיהּ, מְחַר:  לְחוֹד, לָא יוֹסֵיף פַּרְעֹה לְשַׁקָּרָא, בְּדִיל דְּלָא לְשַׁלָּחָא יָת עַמָּא, לְדַבָּחָא קֳדָם יְיָ. 
ח,כו וַיֵּצֵ֥א מֹשֶׁ֖ה מֵעִ֣ם פַּרְעֹ֑ה וַיֶּעְתַּ֖ר אֶל־יְהוָֽה׃ וּנְפַק מֹשֶׁה, מִלְּוָת פַּרְעֹה; וְצַלִּי, קֳדָם יְיָ. 
ח,כז וַיַּ֤עַשׂ יְהוָה֙ כִּדְבַ֣ר מֹשֶׁ֔ה וַיָּ֨סַר֙ הֶֽעָרֹ֔ב מִפַּרְעֹ֖ה מֵֽעֲבָדָ֣יו וּמֵֽעַמּ֑וֹ לֹ֥א נִשְׁאַ֖ר אֶחָֽד׃ וַעֲבַד יְיָ, כְּפִתְגָמָא דְּמֹשֶׁה, וְאַעְדִּי עָרוֹבָא, מִפַּרְעֹה מֵעַבְדּוֹהִי וּמֵעַמֵּיהּ:  לָא אִשְׁתְּאַר, חַד. 
ח,כח וַיַּכְבֵּ֤ד פַּרְעֹה֙ אֶת־לִבּ֔וֹ גַּ֖ם בַּפַּ֣עַם הַזֹּ֑את וְלֹ֥א שִׁלַּ֖ח אֶת־הָעָֽם׃ {פ} וְיַקַּר פַּרְעֹה יָת לִבֵּיהּ, אַף בְּזִמְנָא הָדָא; וְלָא שַׁלַּח, יָת עַמָּא. {פ}
ט,א וַיֹּ֤אמֶר יְהוָה֙ אֶל־מֹשֶׁ֔ה בֹּ֖א אֶל־פַּרְעֹ֑ה וְדִבַּרְתָּ֣ אֵלָ֗יו כֹּֽה־אָמַ֤ר יְהוָה֙ אֱלֹהֵ֣י הָֽעִבְרִ֔ים שַׁלַּ֥ח אֶת־עַמִּ֖י וְיַֽעַבְדֻֽנִי׃ וַאֲמַר יְיָ לְמֹשֶׁה, עוֹל לְוָת פַּרְעֹה; וּתְמַלֵּיל עִמֵּיהּ, כִּדְנָן אֲמַר יְיָ אֱלָהָא דִּיהוּדָאֵי, שַׁלַּח יָת עַמִּי, וְיִפְלְחוּן קֳדָמָי. 
ט,ב כִּ֛י אִם־מָאֵ֥ן אַתָּ֖ה לְשַׁלֵּ֑חַ וְעֽוֹדְךָ֖ מַֽחֲזִ֥יק בָּֽם׃ אֲרֵי אִם מְסָרֵיב אַתְּ, לְשַׁלָּחָא, וְעַד כְּעַן אַתְּ, מַתְקֵיף בְּהוֹן. 
ט,ג הִנֵּ֨ה יַד־יְהוָ֜ה הוֹיָ֗ה בְּמִקְנְךָ֙ אֲשֶׁ֣ר בַּשָּׂדֶ֔ה בַּסּוּסִ֤ים בַּֽחֲמֹרִים֙ בַּגְּמַלִּ֔ים בַּבָּקָ֖ר וּבַצֹּ֑אן דֶּ֖בֶר כָּבֵ֥ד מְאֹֽד׃ הָא מַחָא מִן קֳדָם יְיָ הָוְיָא, בִּבְעִירָךְ דִּבְחַקְלָא, בְּסוּסָוָתָא בִּחְמָרֵי בְּגַמְלֵי, בְּתוֹרֵי וּבְעָנָא--מוֹתָא, סַגִּי לַחְדָּא. 
ט,ד וְהִפְלָ֣ה יְהוָ֔ה בֵּ֚ין מִקְנֵ֣ה יִשְׂרָאֵ֔ל וּבֵ֖ין מִקְנֵ֣ה מִצְרָ֑יִם וְלֹ֥א יָמ֛וּת מִכָּל־לִבְנֵ֥י יִשְׂרָאֵ֖ל דָּבָֽר׃ וְיַפְרֵישׁ יְיָ--בֵּין בְּעִירָא דְּיִשְׂרָאֵל, וּבֵין בְּעִירָא דְּמִצְרָאֵי; וְלָא יְמוּת מִכּוֹלָא לִבְנֵי יִשְׂרָאֵל, מִדָּעַם. 
ט,ה וַיָּ֥שֶׂם יְהוָ֖ה מוֹעֵ֣ד לֵאמֹ֑ר מָחָ֗ר יַֽעֲשֶׂ֧ה יְהוָ֛ה הַדָּבָ֥ר הַזֶּ֖ה בָּאָֽרֶץ׃ וְשַׁוִּי יְיָ, זִמְנָא לְמֵימַר:  מְחַר, יַעֲבֵיד יְיָ פִּתְגָמָא הָדֵין--בְּאַרְעָא. 
ט,ו וַיַּ֨עַשׂ יְהוָ֜ה אֶת־הַדָּבָ֤ר הַזֶּה֙ מִֽמָּחֳרָ֔ת וַיָּ֕מָת כֹּ֖ל מִקְנֵ֣ה מִצְרָ֑יִם וּמִמִּקְנֵ֥ה בְנֵֽי־יִשְׂרָאֵ֖ל לֹא־מֵ֥ת אֶחָֽד׃ וַעֲבַד יְיָ יָת פִּתְגָמָא הָדֵין, בְּיוֹמָא דְּבָתְרוֹהִי, וּמִית, כֹּל בְּעִירָא דְּמִצְרָאֵי; וּמִבְּעִירָא דִּבְנֵי יִשְׂרָאֵל, לָא מִית חַד. 
ט,ז וַיִּשְׁלַ֣ח פַּרְעֹ֔ה וְהִנֵּ֗ה לֹא־מֵ֛ת מִמִּקְנֵ֥ה יִשְׂרָאֵ֖ל עַד־אֶחָ֑ד וַיִּכְבַּד֙ לֵ֣ב פַּרְעֹ֔ה וְלֹ֥א שִׁלַּ֖ח אֶת־הָעָֽם׃ {פ} וּשְׁלַח פַּרְעֹה--וְהָא לָא מִית מִבְּעִירָא דְּיִשְׂרָאֵל, עַד חַד; וְאִתְיַקַּר לִבָּא דְּפַרְעֹה, וְלָא שַׁלַּח יָת עַמָּא. {פ}
ט,ח וַיֹּ֣אמֶר יְהוָה֮ אֶל־מֹשֶׁ֣ה וְאֶֽל־אַהֲרֹן֒ קְח֤וּ לָכֶם֙ מְלֹ֣א חָפְנֵיכֶ֔ם פִּ֖יחַ כִּבְשָׁ֑ן וּזְרָק֥וֹ מֹשֶׁ֛ה הַשָּׁמַ֖יְמָה לְעֵינֵ֥י פַרְעֹֽה׃ וַאֲמַר יְיָ, לְמֹשֶׁה וּלְאַהֲרוֹן, סַבוּ לְכוֹן מְלֵי חֻפְנֵיכוֹן, פִּיחַ דְּאַתּוּנָא; וְיִזְרְקִנֵּיהּ מֹשֶׁה לְצֵית שְׁמַיָּא, לְעֵינֵי פַּרְעֹה. 
ט,ט וְהָיָ֣ה לְאָבָ֔ק עַ֖ל כָּל־אֶ֣רֶץ מִצְרָ֑יִם וְהָיָ֨ה עַל־הָֽאָדָ֜ם וְעַל־הַבְּהֵמָ֗ה לִשְׁחִ֥ין פֹּרֵ֛חַ אֲבַעְבֻּעֹ֖ת בְּכָל־אֶ֥רֶץ מִצְרָֽיִם׃ וִיהֵי לְאַבְקָא, עַל כָּל אַרְעָא דְּמִצְרָיִם; וִיהֵי עַל אֲנָשָׁא וְעַל בְּעִירָא, לִשְׁחִין סָגֵי אֲבַעְבּוֹעֲיָן--בְּכָל אַרְעָא דְּמִצְרָיִם. 
ט,י וַיִּקְח֞וּ אֶת־פִּ֣יחַ הַכִּבְשָׁ֗ן וַיַּֽעַמְדוּ֙ לִפְנֵ֣י פַרְעֹ֔ה וַיִּזְרֹ֥ק אֹת֛וֹ מֹשֶׁ֖ה הַשָּׁמָ֑יְמָה וַיְהִ֗י שְׁחִין֙ אֲבַעְבֻּעֹ֔ת פֹּרֵ֕חַ בָּֽאָדָ֖ם וּבַבְּהֵמָֽה׃ וּנְסִיבוּ יָת פִּיחַ דְּאַתּוּנָא, וְקָמוּ קֳדָם פַּרְעֹה, וּזְרַק יָתֵיהּ מֹשֶׁה, לְצֵית שְׁמַיָּא; וַהֲוָה, שְׁחִין אֲבַעְבּוֹעֲיָן, סָגֵי, בַּאֲנָשָׁא וּבִבְעִירָא. 
ט,יא וְלֹֽא־יָכְל֣וּ הַֽחַרְטֻמִּ֗ים לַֽעֲמֹ֛ד לִפְנֵ֥י מֹשֶׁ֖ה מִפְּנֵ֣י הַשְּׁחִ֑ין כִּֽי־הָיָ֣ה הַשְּׁחִ֔ין בַּֽחַרְטֻמִּ֖ם וּבְכָל־מִצְרָֽיִם׃ וְלָא יְכִילוּ חָרָשַׁיָּא, לִמְקָם קֳדָם מֹשֶׁה--מִן קֳדָם שִׁחְנָא:  אֲרֵי הֲוָה שִׁחְנָא, בְּחָרָשַׁיָּא וּבְכָל מִצְרָאֵי. 
ט,יב וַיְחַזֵּ֤ק יְהוָה֙ אֶת־לֵ֣ב פַּרְעֹ֔ה וְלֹ֥א שָׁמַ֖ע אֲלֵהֶ֑ם כַּֽאֲשֶׁ֛ר דִּבֶּ֥ר יְהוָ֖ה אֶל־מֹשֶֽׁה׃ {ס} וְתַקֵּיף יְיָ יָת לִבָּא דְּפַרְעֹה, וְלָא קַבֵּיל מִנְּהוֹן:  כְּמָא דְּמַלֵּיל יְיָ, עִם מֹשֶׁה. {ס}
ט,יג וַיֹּ֤אמֶר יְהוָה֙ אֶל־מֹשֶׁ֔ה הַשְׁכֵּ֣ם בַּבֹּ֔קֶר וְהִתְיַצֵּ֖ב לִפְנֵ֣י פַרְעֹ֑ה וְאָֽמַרְתָּ֣ אֵלָ֗יו כֹּֽה־אָמַ֤ר יְהוָה֙ אֱלֹהֵ֣י הָֽעִבְרִ֔ים שַׁלַּ֥ח אֶת־עַמִּ֖י וְיַֽעַבְדֻֽנִי׃ וַאֲמַר יְיָ, לְמֹשֶׁה, אַקְדֵּים בְּצַפְרָא, וְאִתְעַתַּד קֳדָם פַּרְעֹה; וְתֵימַר לֵיהּ, כִּדְנָן אֲמַר יְיָ אֱלָהָא דִּיהוּדָאֵי, שַׁלַּח יָת עַמִּי, וְיִפְלְחוּן קֳדָמָי. 
ט,יד כִּ֣י ׀ בַּפַּ֣עַם הַזֹּ֗את אֲנִ֨י שֹׁלֵ֜חַ אֶת־כָּל־מַגֵּֽפֹתַי֙ אֶֽל־לִבְּךָ֔ וּבַֽעֲבָדֶ֖יךָ וּבְעַמֶּ֑ךָ בַּֽעֲב֣וּר תֵּדַ֔ע כִּ֛י אֵ֥ין כָּמֹ֖נִי בְּכָל־הָאָֽרֶץ׃ אֲרֵי בְּזִמְנָא הָדָא, אֲנָא שָׁלַח יָת כָּל מַחָתַי בְּלִבָּךְ, וּבְעַבְדָךְ, וּבְעַמָּךְ--בְּדִיל דְּתִדַּע, אֲרֵי לֵית דִּכְוָתִי בְּכָל אַרְעָא. 
ט,טו כִּ֤י עַתָּה֙ שָׁלַ֣חְתִּי אֶת־יָדִ֔י וָאַ֥ךְ אֽוֹתְךָ֛ וְאֶֽת־עַמְּךָ֖ בַּדָּ֑בֶר וַתִּכָּחֵ֖ד מִן־הָאָֽרֶץ׃ אֲרֵי כְּעַן קָרִיב קֳדָמַי דִּשְׁלַחִית פּוֹן יָת מַחַת גְּבוּרְתִי, וּמְחֵית יָתָךְ וְיָת עַמָּךְ בְּמוֹתָא; וְאִשְׁתֵּיצִיתָא, מִן אַרְעָא. 
ט,טז וְאוּלָ֗ם בַּֽעֲב֥וּר זֹאת֙ הֶֽעֱמַדְתִּ֔יךָ בַּֽעֲב֖וּר הַרְאֹֽתְךָ֣ אֶת־כֹּחִ֑י וּלְמַ֛עַן סַפֵּ֥ר שְׁמִ֖י בְּכָל־הָאָֽרֶץ׃ וּבְרַם, בְּדִיל דָּא קַיֵּימְתָּךְ, בְּדִיל, לְאַחְזָיוּתָךְ יָת חֵילִי; וּבְדִיל דִּיהוֹן מִשְׁתָּעַן גְּבוּרַת שְׁמִי, בְּכָל אַרְעָא. 
ט,יז עֽוֹדְךָ֖ מִסְתּוֹלֵ֣ל בְּעַמִּ֑י לְבִלְתִּ֖י שַׁלְּחָֽם׃ עַד כְּעַן, כְּבֵישְׁתְּ לֵיהּ בְּעַמִּי, בְּדִיל דְּלָא, לְשַׁלָּחוּתְהוֹן. 
ט,יח הִנְנִ֤י מַמְטִיר֙ כָּעֵ֣ת מָחָ֔ר בָּרָ֖ד כָּבֵ֣ד מְאֹ֑ד אֲשֶׁ֨ר לֹֽא־הָיָ֤ה כָמֹ֨הוּ֙ בְּמִצְרַ֔יִם לְמִן־הַיּ֥וֹם הִוָּֽסְדָ֖ה וְעַד־עָֽתָּה׃ הָאֲנָא מַחֵית כְּעִדָּנָא הָדֵין מְחַר, בַּרְדָּא תַּקִּיף לַחְדָּא, דְּלָא הֲוָה דִּכְוָתֵיהּ בְּמִצְרַיִם, לְמִן יוֹמָא דְּאִשְׁתַּכְלַלַת וְעַד כְּעַן. 
ט,יט וְעַתָּ֗ה שְׁלַ֤ח הָעֵז֙ אֶֽת־מִקְנְךָ֔ וְאֵ֛ת כָּל־אֲשֶׁ֥ר לְךָ֖ בַּשָּׂדֶ֑ה כָּל־הָֽאָדָ֨ם וְהַבְּהֵמָ֜ה אֲשֶֽׁר־יִמָּצֵ֣א בַשָּׂדֶ֗ה וְלֹ֤א יֵֽאָסֵף֙ הַבַּ֔יְתָה וְיָרַ֧ד עֲלֵהֶ֛ם הַבָּרָ֖ד וָמֵֽתוּ׃ וּכְעַן, שְׁלַח כְּנוֹשׁ יָת בְּעִירָךְ, וְיָת כָּל דְּלָךְ, בְּחַקְלָא:  כָּל אֲנָשָׁא וּבְעִירָא דְּיִשְׁתְּכַח בְּחַקְלָא, וְלָא יִתְכְּנֵישׁ לְבֵיתָא--וְיֵיחוֹת עֲלֵיהוֹן בַּרְדָּא, וִימוּתוּן. 
ט,כ הַיָּרֵא֙ אֶת־דְּבַ֣ר יְהוָ֔ה מֵֽעַבְדֵ֖י פַּרְעֹ֑ה הֵנִ֛יס אֶת־עֲבָדָ֥יו וְאֶת־מִקְנֵ֖הוּ אֶל־הַבָּתִּֽים׃ דְּדָחֵיל מִפִּתְגָמָא דַּייָ, מֵעַבְדֵי פַּרְעֹה--כְּנַשׁ יָת עַבְדּוֹהִי וְיָת בִּעִירֵיהּ, לְבָתַּיָּא. 
ט,כא וַֽאֲשֶׁ֥ר לֹא־שָׂ֛ם לִבּ֖וֹ אֶל־דְּבַ֣ר יְהוָ֑ה וַֽיַּעֲזֹ֛ב אֶת־עֲבָדָ֥יו וְאֶת־מִקְנֵ֖הוּ בַּשָּׂדֶֽה׃ {פ} וּדְלָא שַׁוִּי לִבֵּיהּ, לְפִתְגָמָא דַּייָ--שְׁבַק יָת עַבְדּוֹהִי וְיָת בְּעִירֵיהּ, בְּחַקְלָא. {פ}
ט,כב וַיֹּ֨אמֶר יְהוָ֜ה אֶל־מֹשֶׁ֗ה נְטֵ֤ה אֶת־יָֽדְךָ֙ עַל־הַשָּׁמַ֔יִם וִיהִ֥י בָרָ֖ד בְּכָל־אֶ֣רֶץ מִצְרָ֑יִם עַל־הָֽאָדָ֣ם וְעַל־הַבְּהֵמָ֗ה וְעַ֛ל כָּל־עֵ֥שֶׂב הַשָּׂדֶ֖ה בְּאֶ֥רֶץ מִצְרָֽיִם׃ וַאֲמַר יְיָ לְמֹשֶׁה, אֲרֵים יָת יְדָךְ עַל צֵית שְׁמַיָּא, וִיהֵי בַּרְדָּא, בְּכָל אַרְעָא דְּמִצְרָיִם:  עַל אֲנָשָׁא וְעַל בְּעִירָא, וְעַל כָּל עִסְבָּא דְּחַקְלָא--בְּאַרְעָא דְּמִצְרָיִם. 
ט,כג וַיֵּ֨ט מֹשֶׁ֣ה אֶת־מַטֵּהוּ֮ עַל־הַשָּׁמַיִם֒ וַֽיהוָ֗ה נָתַ֤ן קֹלֹת֙ וּבָרָ֔ד וַתִּ֥הֲלַךְ אֵ֖שׁ אָ֑רְצָה וַיַּמְטֵ֧ר יְהוָ֛ה בָּרָ֖ד עַל־אֶ֥רֶץ מִצְרָֽיִם׃ וַאֲרֵים מֹשֶׁה יָת חֻטְרֵיהּ, עַל צֵית שְׁמַיָּא, וַייָ יְהַב קָלִין וּבְרַד, וּמְהַלְּכָא אִישָׁתָא עַל אַרְעָא; וְאַמְטַר יְיָ בַּרְדָּא, עַל אַרְעָא דְּמִצְרָיִם. 
ט,כד וַיְהִ֣י בָרָ֔ד וְאֵ֕שׁ מִתְלַקַּ֖חַת בְּת֣וֹךְ הַבָּרָ֑ד כָּבֵ֣ד מְאֹ֔ד אֲ֠שֶׁר לֹֽא־הָיָ֤ה כָמֹ֨הוּ֙ בְּכָל־אֶ֣רֶץ מִצְרַ֔יִם מֵאָ֖ז הָֽיְתָ֥ה לְגֽוֹי׃ וַהֲוָה בַּרְדָּא--וְאִישָׁתָא, מִשְׁתַּלְהֲבָא בְּגוֹ בַּרְדָּא:  תַּקִּיף לַחְדָּא--דְּלָא הֲוָה דִּכְוָתֵיהּ בְּכָל אַרְעָא דְּמִצְרַיִם, מֵעִדָּן דַּהֲוָת לְעַם. 
ט,כה וַיַּ֨ךְ הַבָּרָ֜ד בְּכָל־אֶ֣רֶץ מִצְרַ֗יִם אֵ֚ת כָּל־אֲשֶׁ֣ר בַּשָּׂדֶ֔ה מֵֽאָדָ֖ם וְעַד־בְּהֵמָ֑ה וְאֵ֨ת כָּל־עֵ֤שֶׂב הַשָּׂדֶה֙ הִכָּ֣ה הַבָּרָ֔ד וְאֶת־כָּל־עֵ֥ץ הַשָּׂדֶ֖ה שִׁבֵּֽר׃ וּמְחָא בַּרְדָּא בְּכָל אַרְעָא דְּמִצְרַיִם, יָת כָּל דִּבְחַקְלָא, מֵאֲנָשָׁא, וְעַד בְּעִירָא; וְיָת כָּל עִסְבָּא דְּחַקְלָא מְחָא בַּרְדָּא, וְיָת כָּל אִילָנֵי חַקְלָא תַּבַּר. 
ט,כו רַ֚ק בְּאֶ֣רֶץ גֹּ֔שֶׁן אֲשֶׁר־שָׁ֖ם בְּנֵ֣י יִשְׂרָאֵ֑ל לֹ֥א הָיָ֖ה בָּרָֽד׃ לְחוֹד בְּאַרְעָא דְּגֹשֶׁן, דְּתַמָּן בְּנֵי יִשְׂרָאֵל--לָא הֲוָה, בַּרְדָּא. 
ט,כז וַיִּשְׁלַ֣ח פַּרְעֹ֗ה וַיִּקְרָא֙ לְמֹשֶׁ֣ה וּֽלְאַהֲרֹ֔ן וַיֹּ֥אמֶר אֲלֵהֶ֖ם חָטָ֣אתִי הַפָּ֑עַם יְהוָה֙ הַצַּדִּ֔יק וַֽאֲנִ֥י וְעַמִּ֖י הָֽרְשָׁעִֽים׃ וּשְׁלַח פַּרְעֹה, וּקְרָא לְמֹשֶׁה וּלְאַהֲרוֹן, וַאֲמַר לְהוֹן, חַבִית זִמְנָא הָדָא:  יְיָ, זַכָּאָה, וַאֲנָא וְעַמִּי, חַיָּבִין. 
ט,כח הַעְתִּ֨ירוּ֙ אֶל־יְהוָ֔ה וְרַ֕ב מִֽהְיֹ֛ת קֹלֹ֥ת אֱלֹהִ֖ים וּבָרָ֑ד וַֽאֲשַׁלְּחָ֣ה אֶתְכֶ֔ם וְלֹ֥א תֹֽסִפ֖וּן לַֽעֲמֹֽד׃ צַלּוֹ, קֳדָם יְיָ, וְסַגִּי קֳדָמוֹהִי רְוַח, דְּלָא יְהוֹן עֲלַנָא קָלִין דִּלְוָט כְּאִלֵּין מִן קֳדָם יְיָ וּבְרַד; וַאֲשַׁלַּח יָתְכוֹן, וְלָא תֵּיסְפוּן לְאִתְעַכָּבָא. 
ט,כט וַיֹּ֤אמֶר אֵלָיו֙ מֹשֶׁ֔ה כְּצֵאתִי֙ אֶת־הָעִ֔יר אֶפְרֹ֥שׂ אֶת־כַּפַּ֖י אֶל־יְהוָ֑ה הַקֹּל֣וֹת יֶחְדָּל֗וּן וְהַבָּרָד֙ לֹ֣א יִֽהְיֶה־ע֔וֹד לְמַ֣עַן תֵּדַ֔ע כִּ֥י לַֽיהוָ֖ה הָאָֽרֶץ׃ וַאֲמַר לֵיהּ, מֹשֶׁה, כְּמִפְּקִי מִן קַרְתָּא, אֶפְרוֹס יָת יְדַי בִּצְלוֹ קֳדָם יְיָ; קָלַיָּא יִתְמַנְעוּן, וּבַרְדָּא לָא יְהֵי עוֹד, בְּדִיל דְּתִדַּע, אֲרֵי דַּייָ אַרְעָא. 
ט,ל וְאַתָּ֖ה וַֽעֲבָדֶ֑יךָ יָדַ֕עְתִּי כִּ֚י טֶ֣רֶם תִּֽירְא֔וּן מִפְּנֵ֖י יְהוָ֥ה אֱלֹהִֽים׃ וְאַתְּ, וְעַבְדָךְ:  יָדַעְנָא--אֲרֵי עַד כְּעַן לָא אִתְכְּנַעְתּוּן, מִן קֳדָם יְיָ אֱלֹהִים. 
ט,לא וְהַפִּשְׁתָּ֥ה וְהַשְּׂעֹרָ֖ה נֻכָּ֑תָה כִּ֤י הַשְּׂעֹרָה֙ אָבִ֔יב וְהַפִּשְׁתָּ֖ה גִּבְעֹֽל׃ וְכִתָּנָא וּסְעָרֵי, לְקוֹ:  אֲרֵי סְעָרַיָּא אֲבִיב, וְכִתָּנָא גַּבְעֻלִּין. 
ט,לב וְהַחִטָּ֥ה וְהַכֻּסֶּ֖מֶת לֹ֣א נֻכּ֑וּ כִּ֥י אֲפִילֹ֖ת הֵֽנָּה׃ וְחִטַּיָּא וְכוּנָתַיָּא, לָא לְקַאָה:  אֲרֵי אַפְלָתָא, אִנִּין. 
ט,לג וַיֵּצֵ֨א מֹשֶׁ֜ה מֵעִ֤ם פַּרְעֹה֙ אֶת־הָעִ֔יר וַיִּפְרֹ֥שׂ כַּפָּ֖יו אֶל־יְהוָ֑ה וַֽיַּחְדְּל֤וּ הַקֹּלוֹת֙ וְהַבָּרָ֔ד וּמָטָ֖ר לֹֽא־נִתַּ֥ךְ אָֽרְצָה׃ וּנְפַק מֹשֶׁה מִלְּוָת פַּרְעֹה, יָת קַרְתָּא, וּפְרַס יְדוֹהִי בִּצְלוֹ, קֳדָם יְיָ; וְאִתְמְנַעוּ קָלַיָּא וּבַרְדָּא, וּמִטְרָא דַּהֲוָה נָחֵית לָא מְטָא עַל אַרְעָא. 
ט,לד וַיַּ֣רְא פַּרְעֹ֗ה כִּֽי־חָדַ֨ל הַמָּטָ֧ר וְהַבָּרָ֛ד וְהַקֹּלֹ֖ת וַיֹּ֣סֶף לַֽחֲטֹ֑א וַיַּכְבֵּ֥ד לִבּ֖וֹ ה֥וּא וַֽעֲבָדָֽיו׃ וַחֲזָא פַּרְעֹה, אֲרֵי אִתְמְנַע מִטְרָא וּבַרְדָּא וְקָלַיָּא--וְאוֹסֵיף לְמִחְטֵי; וְיַקְּרֵיהּ לְלִבֵּיהּ, הוּא וְעַבְדּוֹהִי. 
ט,לה וַֽיֶּחֱזַק֙ לֵ֣ב פַּרְעֹ֔ה וְלֹ֥א שִׁלַּ֖ח אֶת־בְּנֵ֣י יִשְׂרָאֵ֑ל כַּֽאֲשֶׁ֛ר דִּבֶּ֥ר יְהוָ֖ה בְּיַד־מֹשֶֽׁה׃ {פ} וְאִתַּקַּף לִבָּא דְּפַרְעֹה, וְלָא שַׁלַּח יָת בְּנֵי יִשְׂרָאֵל:  כְּמָא דְּמַלֵּיל יְיָ, בִּידָא דְּמֹשֶׁה. {פ}
