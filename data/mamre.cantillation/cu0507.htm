כי-תבוא

כו,א וְהָיָה֙ כִּֽי־תָב֣וֹא אֶל־הָאָ֔רֶץ אֲשֶׁר֙ יְהוָ֣ה אֱלֹהֶ֔יךָ נֹתֵ֥ן לְךָ֖ נַֽחֲלָ֑ה וִֽירִשְׁתָּ֖הּ וְיָשַׁ֥בְתָּ בָּֽהּ׃ וִיהֵי, אֲרֵי תֵּיעוֹל לְאַרְעָא, דַּייָ אֱלָהָךְ, יָהֵיב לָךְ אַחְסָנָא; וְתֵירְתַהּ, וְתִתֵּיב בַּהּ. 
כו,ב וְלָֽקַחְתָּ֞ מֵֽרֵאשִׁ֣ית ׀ כָּל־פְּרִ֣י הָֽאֲדָמָ֗ה אֲשֶׁ֨ר תָּבִ֧יא מֵֽאַרְצְךָ֛ אֲשֶׁ֨ר יְהוָ֧ה אֱלֹהֶ֛יךָ נֹתֵ֥ן לָ֖ךְ וְשַׂמְתָּ֣ בַטֶּ֑נֶא וְהָֽלַכְתָּ֙ אֶל־הַמָּק֔וֹם אֲשֶׁ֤ר יִבְחַר֙ יְהוָ֣ה אֱלֹהֶ֔יךָ לְשַׁכֵּ֥ן שְׁמ֖וֹ שָֽׁם׃ וְתִסַּב מֵרֵישׁ כָּל אִבָּא דְּאַרְעָא, דְּתַעֵיל מֵאַרְעָךְ דַּייָ אֱלָהָךְ יָהֵיב לָךְ--וּתְשַׁוֵּי בְּסַלָּא; וּתְהָךְ, לְאַתְרָא, דְּיִתְרְעֵי יְיָ אֱלָהָךְ, לְאַשְׁרָאָה שְׁכִינְתֵיהּ תַּמָּן. 
כו,ג וּבָאתָ֙ אֶל־הַכֹּהֵ֔ן אֲשֶׁ֥ר יִֽהְיֶ֖ה בַּיָּמִ֣ים הָהֵ֑ם וְאָֽמַרְתָּ֣ אֵלָ֗יו הִגַּ֤דְתִּי הַיּוֹם֙ לַֽיהוָ֣ה אֱלֹהֶ֔יךָ כִּי־בָ֨אתִי֙ אֶל־הָאָ֔רֶץ אֲשֶׁ֨ר נִשְׁבַּ֧ע יְהוָ֛ה לַֽאֲבֹתֵ֖ינוּ לָ֥תֶת לָֽנוּ׃ וְתֵיתֵי, לְוָת כָּהֲנָא, דִּיהֵי, בְּיוֹמַיָּא הָאִנּוּן; וְתֵימַר לֵיהּ, חַוִּיתִי יוֹמָא דֵּין קֳדָם יְיָ אֱלָהָךְ, אֲרֵי עַאלִית לְאַרְעָא, דְּקַיֵּים יְיָ לַאֲבָהָתַנָא לְמִתַּן לַנָא. 
כו,ד וְלָקַ֧ח הַכֹּהֵ֛ן הַטֶּ֖נֶא מִיָּדֶ֑ךָ וְהִ֨נִּיח֔וֹ לִפְנֵ֕י מִזְבַּ֖ח יְהוָ֥ה אֱלֹהֶֽיךָ׃ וְיִסַּב כָּהֲנָא סַלָּא, מִן יְדָךְ; וְיַחֲתִנֵּיהּ--קֳדָם, מַדְבְּחָא דַּייָ אֱלָהָךְ. 
כו,ה וְעָנִ֨יתָ וְאָֽמַרְתָּ֜ לִפְנֵ֣י ׀ יְהוָ֣ה אֱלֹהֶ֗יךָ אֲרַמִּי֙ אֹבֵ֣ד אָבִ֔י וַיֵּ֣רֶד מִצְרַ֔יְמָה וַיָּ֥גָר שָׁ֖ם בִּמְתֵ֣י מְעָ֑ט וַֽיְהִי־שָׁ֕ם לְג֥וֹי גָּד֖וֹל עָצ֥וּם וָרָֽב׃ וְתָתִיב וְתֵימַר קֳדָם יְיָ אֱלָהָךְ, לָבָן אֲרַמָּאָה בְּעָא לְאַבָּדָא יָת אַבָּא, וּנְחַת לְמִצְרַיִם, וְדָר תַּמָּן בְּעַם זְעֵיר; וַהֲוָה תַּמָּן, לְעַם רַב תַּקִּיף וְסַגִּי. 
כו,ו וַיָּרֵ֧עוּ אֹתָ֛נוּ הַמִּצְרִ֖ים וַיְעַנּ֑וּנוּ וַיִּתְּנ֥וּ עָלֵ֖ינוּ עֲבֹדָ֥ה קָשָֽׁה׃ וְאַבְאִישׁוּ לַנָא מִצְרָאֵי, וְעַנְּיוּנָא; וִיהַבוּ עֲלַנָא, פֻּלְחָנָא קַשְׁיָא. 
כו,ז וַנִּצְעַ֕ק אֶל־יְהוָ֖ה אֱלֹהֵ֣י אֲבֹתֵ֑ינוּ וַיִּשְׁמַ֤ע יְהוָה֙ אֶת־קֹלֵ֔נוּ וַיַּ֧רְא אֶת־עָנְיֵ֛נוּ וְאֶת־עֲמָלֵ֖נוּ וְאֶת־לַֽחֲצֵֽנוּ׃ וְצַלִּינָא, קֳדָם יְיָ אֱלָהָא דַּאֲבָהָתַנָא; וְקַבֵּיל יְיָ צְלוֹתַנָא, וּגְלֵי קֳדָמוֹהִי עַמְלַנָא וְלֵיאוּתַנָא וְדֻחְקַנָא. 
כו,ח וַיּֽוֹצִאֵ֤נוּ יְהוָה֙ מִמִּצְרַ֔יִם בְּיָ֤ד חֲזָקָה֙ וּבִזְרֹ֣עַ נְטוּיָ֔ה וּבְמֹרָ֖א גָּדֹ֑ל וּבְאֹת֖וֹת וּבְמֹֽפְתִֽים׃ וְאַפְּקַנָא יְיָ, מִמִּצְרַיִם, בְּיַד תַּקִּיפָא וּבִדְרָע מְרָמַם, וּבְחֶזְוָנָא רַבָּא--וּבְאָתִין, וּבְמוֹפְתִין. 
כו,ט וַיְבִאֵ֖נוּ אֶל־הַמָּק֣וֹם הַזֶּ֑ה וַיִּתֶּן־לָ֨נוּ֙ אֶת־הָאָ֣רֶץ הַזֹּ֔את אֶ֛רֶץ זָבַ֥ת חָלָ֖ב וּדְבָֽשׁ׃ וְאַיְתְיַנָא, לְאַתְרָא הָדֵין; וִיהַב לַנָא יָת אַרְעָא הָדָא, אֲרַע עָבְדָא חֲלָב וּדְבַשׁ. 
כו,י וְעַתָּ֗ה הִנֵּ֤ה הֵבֵ֨אתִי֙ אֶת־רֵאשִׁית֙ פְּרִ֣י הָֽאֲדָמָ֔ה אֲשֶׁר־נָתַ֥תָּה לִּ֖י יְהוָ֑ה וְהִנַּחְתּ֗וֹ לִפְנֵי֙ יְהוָ֣ה אֱלֹהֶ֔יךָ וְהִֽשְׁתַּחֲוִ֔יתָ לִפְנֵ֖י יְהוָ֥ה אֱלֹהֶֽיךָ׃ וּכְעַן, הָא אֵיתִיתִי יָת רֵישׁ אִבָּא דְּאַרְעָא, דִּיהַבְתְּ לִי, יְיָ; וְתַחֲתִנֵּיהּ, קֳדָם יְיָ אֱלָהָךְ, וְתִסְגּוֹד, קֳדָם יְיָ אֱלָהָךְ. 
כו,יא וְשָֽׂמַחְתָּ֣ בְכָל־הַטּ֗וֹב אֲשֶׁ֧ר נָֽתַן־לְךָ֛ יְהוָ֥ה אֱלֹהֶ֖יךָ וּלְבֵיתֶ֑ךָ אַתָּה֙ וְהַלֵּוִ֔י וְהַגֵּ֖ר אֲשֶׁ֥ר בְּקִרְבֶּֽךָ׃ {ס} וְתִחְדֵּי בְּכָל טָבְתָא, דִּיהַב לָךְ יְיָ אֱלָהָךְ--וְלַאֲנָשׁ בֵּיתָךְ:  אַתְּ, וְלֵיוָאָה, וְגִיּוֹרָא, דְּבֵינָךְ. {ס}
כו,יב כִּ֣י תְכַלֶּ֞ה לַ֠עְשֵׂר אֶת־כָּל־מַעְשַׂ֧ר תְּבוּאָֽתְךָ֛ בַּשָּׁנָ֥ה הַשְּׁלִישִׁ֖ת שְׁנַ֣ת הַֽמַּעֲשֵׂ֑ר וְנָֽתַתָּ֣ה לַלֵּוִ֗י לַגֵּר֙ לַיָּת֣וֹם וְלָֽאַלְמָנָ֔ה וְאָֽכְל֥וּ בִשְׁעָרֶ֖יךָ וְשָׂבֵֽעוּ׃ אֲרֵי תְּשֵׁיצֵי לְעַסָּרָא יָת כָּל מַעְסַר עֲלַלְתָּךְ, בְּשַׁתָּא תְּלִיתֵיתָא--שְׁנַת מַעְסְרָא:  וְתִתֵּין לְלֵיוָאָה, לְגִיּוֹרָא לְיַתְמָא וּלְאַרְמַלְתָּא, וְיֵיכְלוּן בְּקִרְוָךְ, וְיִסְבְּעוּן. 
כו,יג וְאָֽמַרְתָּ֡ לִפְנֵי֩ יְהוָ֨ה אֱלֹהֶ֜יךָ בִּעַ֧רְתִּי הַקֹּ֣דֶשׁ מִן־הַבַּ֗יִת וְגַ֨ם נְתַתִּ֤יו לַלֵּוִי֙ וְלַגֵּר֙ לַיָּת֣וֹם וְלָֽאַלְמָנָ֔ה כְּכָל־מִצְוָֽתְךָ֖ אֲשֶׁ֣ר צִוִּיתָ֑נִי לֹֽא־עָבַ֥רְתִּי מִמִּצְו‍ֹתֶ֖יךָ וְלֹ֥א שָׁכָֽחְתִּי׃ וְתֵימַר קֳדָם יְיָ אֱלָהָךְ פַּלִּיתִי קֹדֶשׁ מַעְסְרָא מִן בֵּיתָא, וְאַף יְהַבְתֵּיהּ לְלֵיוָאָה וּלְגִיּוֹרָא לְיַתְמָא וּלְאַרְמַלְתָּא, כְּכָל תַּפְקֵידְתָּךְ, דְּפַקֵּידְתָּנִי:  לָא עֲבַרִית מִפִּקּוֹדָךְ, וְלָא אִתְנְשִׁיתִי. 
כו,יד לֹֽא־אָכַ֨לְתִּי בְאֹנִ֜י מִמֶּ֗נּוּ וְלֹֽא־בִעַ֤רְתִּי מִמֶּ֨נּוּ֙ בְּטָמֵ֔א וְלֹֽא־נָתַ֥תִּי מִמֶּ֖נּוּ לְמֵ֑ת שָׁמַ֗עְתִּי בְּקוֹל֙ יְהוָ֣ה אֱלֹהָ֔י עָשִׂ֕יתִי כְּכֹ֖ל אֲשֶׁ֥ר צִוִּיתָֽנִי׃ לָא אֲכַלִית בְּאֶבְלִי מִנֵּיהּ, וְלָא חַלֵּיפִית מִנֵּיהּ בִּמְסָאַב, וְלָא יְהַבִית מִנֵּיהּ, לְמִית; קַבֵּילִית, לְמֵימְרָא דַּייָ אֱלָהִי--עֲבַדִית, כְּכֹל דְּפַקֵּידְתָּנִי. 
כו,טו הַשְׁקִיפָה֩ מִמְּע֨וֹן קָדְשְׁךָ֜ מִן־הַשָּׁמַ֗יִם וּבָרֵ֤ךְ אֶֽת־עַמְּךָ֙ אֶת־יִשְׂרָאֵ֔ל וְאֵת֙ הָֽאֲדָמָ֔ה אֲשֶׁ֥ר נָתַ֖תָּה לָ֑נוּ כַּֽאֲשֶׁ֤ר נִשְׁבַּ֨עְתָּ֙ לַֽאֲבֹתֵ֔ינוּ אֶ֛רֶץ זָבַ֥ת חָלָ֖ב וּדְבָֽשׁ׃ {ס} אִסְתַּכִי מִמְּדוֹר קֻדְשָׁךְ מִן שְׁמַיָּא, וּבָרֵיךְ יָת עַמָּךְ יָת יִשְׂרָאֵל, וְיָת אַרְעָא, דִּיהַבְתְּ לַנָא--כְּמָא דְּקַיֵּימְתָּא לַאֲבָהָתַנָא, אֲרַע עָבְדָא חֲלָב וּדְבַשׁ. {ס}
כו,טז הַיּ֣וֹם הַזֶּ֗ה יְהוָ֨ה אֱלֹהֶ֜יךָ מְצַוְּךָ֧ לַֽעֲשׂ֛וֹת אֶת־הַֽחֻקִּ֥ים הָאֵ֖לֶּה וְאֶת־הַמִּשְׁפָּטִ֑ים וְשָֽׁמַרְתָּ֤ וְעָשִׂ֨יתָ֙ אוֹתָ֔ם בְּכָל־לְבָֽבְךָ֖ וּבְכָל־נַפְשֶֽׁךָ׃ יוֹמָא הָדֵין, יְיָ אֱלָהָךְ מְפַקְּדָךְ לְמֶעֱבַד יָת קְיָמַיָּא הָאִלֵּין--וְיָת דִּינַיָּא; וְתִטַּר וְתַעֲבֵיד יָתְהוֹן, בְּכָל לִבָּךְ וּבְכָל נַפְשָׁךְ. 
כו,יז אֶת־יְהוָ֥ה הֶֽאֱמַ֖רְתָּ הַיּ֑וֹם לִֽהְיוֹת֩ לְךָ֨ לֵֽאלֹהִ֜ים וְלָלֶ֣כֶת בִּדְרָכָ֗יו וְלִשְׁמֹ֨ר חֻקָּ֧יו וּמִצְו‍ֹתָ֛יו וּמִשְׁפָּטָ֖יו וְלִשְׁמֹ֥עַ בְּקֹלֽוֹ׃ יָת יְיָ חֲטַבְתְּ, יוֹמָא דֵּין:  לְמִהְוֵי לָךְ לֶאֱלָהּ וְלִמְהָךְ בְּאוֹרְחָן דְּתָקְנָן קֳדָמוֹהִי, וּלְמִטַּר קְיָמוֹהִי וּפִקּוֹדוֹהִי וְדִינוֹהִי--וּלְקַבָּלָא לְמֵימְרֵיהּ. 
כו,יח וַֽיהוָ֞ה הֶאֱמִֽירְךָ֣ הַיּ֗וֹם לִֽהְי֥וֹת לוֹ֙ לְעַ֣ם סְגֻלָּ֔ה כַּֽאֲשֶׁ֖ר דִּבֶּר־לָ֑ךְ וְלִשְׁמֹ֖ר כָּל־מִצְו‍ֹתָֽיו׃ וַייָ חַטְבָךְ יוֹמָא דֵּין, לְמִהְוֵי לֵיהּ לְעַם חַבִּיב--כְּמָא דְּמַלֵּיל לָךְ; וּלְמִטַּר, כָּל פִּקּוֹדוֹהִי. 
כו,יט וּֽלְתִתְּךָ֣ עֶלְי֗וֹן עַ֤ל כָּל־הַגּוֹיִם֙ אֲשֶׁ֣ר עָשָׂ֔ה לִתְהִלָּ֖ה וּלְשֵׁ֣ם וּלְתִפְאָ֑רֶת וְלִֽהְיֹתְךָ֧ עַם־קָדֹ֛שׁ לַֽיהוָ֥ה אֱלֹהֶ֖יךָ כַּֽאֲשֶׁ֥ר דִּבֵּֽר׃ {פ} וּלְמִתְּנָךְ עִלַּאי, עַל כָּל עַמְמַיָּא דַּעֲבַד, לְתֻשְׁבְּחָא, וּלְשׁוֹם וְלִרְבוּ; וּלְמִהְוָךְ עַם קַדִּישׁ קֳדָם יְיָ אֱלָהָךְ, כְּמָא דְּמַלֵּיל. {פ}
כז,א וַיְצַ֤ו מֹשֶׁה֙ וְזִקְנֵ֣י יִשְׂרָאֵ֔ל אֶת־הָעָ֖ם לֵאמֹ֑ר שָׁמֹר֙ אֶת־כָּל־הַמִּצְוָ֔ה אֲשֶׁ֧ר אָֽנֹכִ֛י מְצַוֶּ֥ה אֶתְכֶ֖ם הַיּֽוֹם׃ וּפַקֵּיד מֹשֶׁה וְסָבֵי יִשְׂרָאֵל, יָת עַמָּא לְמֵימַר:  טַרוּ, יָת כָּל תַּפְקֵידְתָּא, דַּאֲנָא מְפַקֵּיד יָתְכוֹן, יוֹמָא דֵּין. 
כז,ב וְהָיָ֗ה בַּיּוֹם֮ אֲשֶׁ֣ר תַּֽעַבְר֣וּ אֶת־הַיַּרְדֵּן֒ אֶל־הָאָ֕רֶץ אֲשֶׁר־יְהוָ֥ה אֱלֹהֶ֖יךָ נֹתֵ֣ן לָ֑ךְ וַהֲקֵֽמֹתָ֤ לְךָ֙ אֲבָנִ֣ים גְּדֹל֔וֹת וְשַׂדְתָּ֥ אֹתָ֖ם בַּשִּֽׂיד׃ וִיהֵי, בְּיוֹמָא דְּתִעְבְּרוּן יָת יַרְדְּנָא, לְאַרְעָא, דַּייָ אֱלָהָךְ יָהֵיב לָךְ--וּתְקִים לָךְ אַבְנִין רַבְרְבָן, וּתְסוּד יָתְהוֹן בְּסִידָא. 
כז,ג וְכָֽתַבְתָּ֣ עֲלֵיהֶ֗ן אֶֽת־כָּל־דִּבְרֵ֛י הַתּוֹרָ֥ה הַזֹּ֖את בְּעָבְרֶ֑ךָ לְמַ֡עַן אֲשֶׁר֩ תָּבֹ֨א אֶל־הָאָ֜רֶץ אֲֽשֶׁר־יְהוָ֥ה אֱלֹהֶ֣יךָ ׀ נֹתֵ֣ן לְךָ֗ אֶ֣רֶץ זָבַ֤ת חָלָב֙ וּדְבַ֔שׁ כַּֽאֲשֶׁ֥ר דִּבֶּ֛ר יְהוָ֥ה אֱלֹהֵֽי־אֲבֹתֶ֖יךָ לָֽךְ׃ וְתִכְתּוֹב עֲלֵיהוֹן, יָת כָּל פִּתְגָמֵי אוֹרָיְתָא הָדָא--בְּמִעְבְּרָךְ:  בְּדִיל דְּתֵיעוֹל לְאַרְעָא דַּייָ אֱלָהָךְ יָהֵיב לָךְ, אֲרַע עָבְדָא חֲלָב וּדְבַשׁ, כְּמָא דְּמַלֵּיל יְיָ אֱלָהָא דַּאֲבָהָתָךְ, לָךְ. 
כז,ד וְהָיָה֮ בְּעָבְרְכֶ֣ם אֶת־הַיַּרְדֵּן֒ תָּקִ֜ימוּ אֶת־הָֽאֲבָנִ֣ים הָאֵ֗לֶּה אֲשֶׁ֨ר אָֽנֹכִ֜י מְצַוֶּ֥ה אֶתְכֶ֛ם הַיּ֖וֹם בְּהַ֣ר עֵיבָ֑ל וְשַׂדְתָּ֥ אוֹתָ֖ם בַּשִּֽׂיד׃ וִיהֵי, בְּמִעְבַּרְכוֹן יָת יַרְדְּנָא, תְּקִימוּן יָת אַבְנַיָּא הָאִלֵּין דַּאֲנָא מְפַקֵּיד יָתְכוֹן יוֹמָא דֵּין, בְּטוּרָא דְּעֵיבָל; וּתְסוּד יָתְהוֹן, בְּסִידָא. 
כז,ה וּבָנִ֤יתָ שָּׁם֙ מִזְבֵּ֔חַ לַֽיהוָ֖ה אֱלֹהֶ֑יךָ מִזְבַּ֣ח אֲבָנִ֔ים לֹֽא־תָנִ֥יף עֲלֵיהֶ֖ם בַּרְזֶֽל׃ וְתִבְנֵי תַּמָּן מַדְבְּחָא, קֳדָם יְיָ אֱלָהָךְ:  מַדְבַּח אַבְנִין, לָא תְּרִים עֲלֵיהוֹן בַּרְזְלָא. 
כז,ו אֲבָנִ֤ים שְׁלֵמוֹת֙ תִּבְנֶ֔ה אֶת־מִזְבַּ֖ח יְהוָ֣ה אֱלֹהֶ֑יךָ וְהַֽעֲלִ֤יתָ עָלָיו֙ עוֹלֹ֔ת לַֽיהוָ֖ה אֱלֹהֶֽיךָ׃ אַבְנִין שַׁלְמָן תִּבְנֵי, יָת מַדְבְּחָא דַּייָ אֱלָהָךְ; וְתַסֵּיק עֲלוֹהִי עֲלָוָן, קֳדָם יְיָ אֱלָהָךְ. 
כז,ז וְזָֽבַחְתָּ֥ שְׁלָמִ֖ים וְאָכַ֣לְתָּ שָּׁ֑ם וְשָׂ֣מַחְתָּ֔ לִפְנֵ֖י יְהוָ֥ה אֱלֹהֶֽיךָ׃ וְתִכּוֹס נִכְסַת קֻדְשִׁין, וְתֵיכוֹל תַּמָּן; וְתִחְדֵּי, קֳדָם יְיָ אֱלָהָךְ. 
כז,ח וְכָֽתַבְתָּ֣ עַל־הָֽאֲבָנִ֗ים אֶֽת־כָּל־דִּבְרֵ֛י הַתּוֹרָ֥ה הַזֹּ֖את בַּאֵ֥ר הֵיטֵֽב׃ {ס} וְתִכְתּוֹב עַל אַבְנַיָּא, יָת כָּל פִּתְגָמֵי אוֹרָיְתָא הָדָא--פָּרֵישׁ יָאוּת. {ס}
כז,ט וַיְדַבֵּ֤ר מֹשֶׁה֙ וְהַכֹּֽהֲנִ֣ים הַלְוִיִּ֔ם אֶ֥ל כָּל־יִשְׂרָאֵ֖ל לֵאמֹ֑ר הַסְכֵּ֤ת ׀ וּשְׁמַע֙ יִשְׂרָאֵ֔ל הַיּ֤וֹם הַזֶּה֙ נִֽהְיֵ֣יתָֽ לְעָ֔ם לַֽיהוָ֖ה אֱלֹהֶֽיךָ׃ וּמַלֵּיל מֹשֶׁה וְכָהֲנַיָּא לֵיוָאֵי, עִם כָּל יִשְׂרָאֵל לְמֵימַר:  אַצֵּית וּשְׁמַע, יִשְׂרָאֵל, יוֹמָא הָדֵין הֲוֵיתָא לְעַם, קֳדָם יְיָ אֱלָהָךְ. 
כז,י וְשָׁ֣מַעְתָּ֔ בְּק֖וֹל יְהוָ֣ה אֱלֹהֶ֑יךָ וְעָשִׂ֤יתָ אֶת־מִצְו‍ֹתָו֙ וְאֶת־חֻקָּ֔יו אֲשֶׁ֛ר אָֽנֹכִ֥י מְצַוְּךָ֖ הַיּֽוֹם׃ {ס} וּתְקַבֵּיל, לְמֵימְרָא דַּייָ אֱלָהָךְ; וְתַעֲבֵיד יָת פִּקּוֹדוֹהִי וְיָת קְיָמוֹהִי, דַּאֲנָא מְפַקֵּיד לָךְ יוֹמָא דֵּין. {ס}
כז,יא וַיְצַ֤ו מֹשֶׁה֙ אֶת־הָעָ֔ם בַּיּ֥וֹם הַה֖וּא לֵאמֹֽר׃ וּפַקֵּיד מֹשֶׁה יָת עַמָּא, בְּיוֹמָא הַהוּא לְמֵימַר. 
כז,יב אֵ֠לֶּה יַֽעַמְד֞וּ לְבָרֵ֤ךְ אֶת־הָעָם֙ עַל־הַ֣ר גְּרִזִ֔ים בְּעָבְרְכֶ֖ם אֶת־הַיַּרְדֵּ֑ן שִׁמְעוֹן֙ וְלֵוִ֣י וִֽיהוּדָ֔ה וְיִשָּׂשכָ֖ר וְיוֹסֵ֥ף וּבִנְיָמִֽן׃ אִלֵּין יְקוּמוּן לְבָרָכָא יָת עַמָּא, עַל טוּרָא דִּגְרִיזִין, בְּמִעְבַּרְכוֹן, יָת יַרְדְּנָא:  שִׁמְעוֹן וְלֵוִי וִיהוּדָה, וְיִשָּׂשכָר וְיוֹסֵף וּבִנְיָמִין. 
כז,יג וְאֵ֛לֶּה יַֽעַמְד֥וּ עַל־הַקְּלָלָ֖ה בְּהַ֣ר עֵיבָ֑ל רְאוּבֵן֙ גָּ֣ד וְאָשֵׁ֔ר וּזְבוּלֻ֖ן דָּ֥ן וְנַפְתָּלִֽי׃ וְאִלֵּין יְקוּמוּן עַל לְוָטַיָּא, בְּטוּרָא דְּעֵיבָל:  רְאוּבֵן גָּד וְאָשֵׁר, וּזְבוּלוּן דָּן וְנַפְתָּלִי. 
כז,יד וְעָנ֣וּ הַלְוִיִּ֗ם וְאָ֥מְר֛וּ אֶל־כָּל־אִ֥ישׁ יִשְׂרָאֵ֖ל ק֥וֹל רָֽם׃ {ס} וְיָתִיבוּן לֵיוָאֵי, וְיֵימְרוּן לְכָל אֲנָשׁ יִשְׂרָאֵל--קָל רַם. {ס}
כז,טו אָר֣וּר הָאִ֡ישׁ אֲשֶׁ֣ר יַֽעֲשֶׂה֩ פֶ֨סֶל וּמַסֵּכָ֜ה תּֽוֹעֲבַ֣ת יְהוָ֗ה מַֽעֲשֵׂ֛ה יְדֵ֥י חָרָ֖שׁ וְשָׂ֣ם בַּסָּ֑תֶר וְעָנ֧וּ כָל־הָעָ֛ם וְאָֽמְר֖וּ אָמֵֽן׃ {ס} לִיט גֻּבְרָא דְּיַעֲבֵיד צֵילַם וּמַתְּכָא מְרַחַק קֳדָם יְיָ, עוֹבָד יְדֵי אֻמָּנָא--וִישַׁוֵּי בְּסִתְרָא; וְיָתִיבוּן כָּל עַמָּא וְיֵימְרוּן, אָמֵן. {ס}
כז,טז אָר֕וּר מַקְלֶ֥ה אָבִ֖יו וְאִמּ֑וֹ וְאָמַ֥ר כָּל־הָעָ֖ם אָמֵֽן׃ {ס} לִיט, דְּיַקְלֵי אֲבוּהִי וְאִמֵּיהּ; וְיֵימַר כָּל עַמָּא, אָמֵן. {ס}
כז,יז אָר֕וּר מַסִּ֖יג גְּב֣וּל רֵעֵ֑הוּ וְאָמַ֥ר כָּל־הָעָ֖ם אָמֵֽן׃ {ס} לִיט, דִּישַׁנֵּי תְּחוּמָא דְּחַבְרֵיהּ; וְיֵימַר כָּל עַמָּא, אָמֵן. {ס}
כז,יח אָר֕וּר מַשְׁגֶּ֥ה עִוֵּ֖ר בַּדָּ֑רֶךְ וְאָמַ֥ר כָּל־הָעָ֖ם אָמֵֽן׃ {ס} לִיט, דְּיַטְעֵי עֲוִירָא בְּאוֹרְחָא; וְיֵימַר כָּל עַמָּא, אָמֵן. {ס}
כז,יט אָר֗וּר מַטֶּ֛ה מִשְׁפַּ֥ט גֵּר־יָת֖וֹם וְאַלְמָנָ֑ה וְאָמַ֥ר כָּל־הָעָ֖ם אָמֵֽן׃ לִיט, דְּיַצְלֵי דִּין גִּיּוֹר יִיתַם--וְאַרְמְלָא; וְיֵימַר כָּל עַמָּא, אָמֵן. 
כז,כ אָר֗וּר שֹׁכֵב֙ עִם־אֵ֣שֶׁת אָבִ֔יו כִּ֥י גִלָּ֖ה כְּנַ֣ף אָבִ֑יו וְאָמַ֥ר כָּל־הָעָ֖ם אָמֵֽן׃ {ס} לִיט, דְּיִשְׁכּוֹב עִם אִתַּת אֲבוּהִי--אֲרֵי גַּלִּי, כַּנְפָא דַּאֲבוּהִי; וְיֵימַר כָּל עַמָּא, אָמֵן. {ס}
כז,כא אָר֕וּר שֹׁכֵ֖ב עִם־כָּל־בְּהֵמָ֑ה וְאָמַ֥ר כָּל־הָעָ֖ם אָמֵֽן׃ {ס} לִיט, דְּיִשְׁכּוֹב עִם כָּל בְּעִירָא; וְיֵימַר כָּל עַמָּא, אָמֵן. {ס}
כז,כב אָר֗וּר שֹׁכֵב֙ עִם־אֲחֹת֔וֹ בַּת־אָבִ֖יו א֣וֹ בַת־אִמּ֑וֹ וְאָמַ֥ר כָּל־הָעָ֖ם אָמֵֽן׃ {ס} לִיט, דְּיִשְׁכּוֹב עִם אֲחָתֵיהּ--בַּת אֲבוּהִי, אוֹ בַּת אִמֵּיהּ; וְיֵימַר כָּל עַמָּא, אָמֵן. {ס}
כז,כג אָר֕וּר שֹׁכֵ֖ב עִם־חֹֽתַנְתּ֑וֹ וְאָמַ֥ר כָּל־הָעָ֖ם אָמֵֽן׃ {ס} לִיט, דְּיִשְׁכּוֹב עִם חֲמָתֵיהּ; וְיֵימַר כָּל עַמָּא, אָמֵן. {ס}
כז,כד אָר֕וּר מַכֵּ֥ה רֵעֵ֖הוּ בַּסָּ֑תֶר וְאָמַ֥ר כָּל־הָעָ֖ם אָמֵֽן׃ {ס} לִיט, דְּיִמְחֵי חַבְרֵיהּ בְּסִתְרָא; וְיֵימַר כָּל עַמָּא, אָמֵן. {ס}
כז,כה אָרוּר֙ לֹקֵ֣חַ שֹׁ֔חַד לְהַכּ֥וֹת נֶ֖פֶשׁ דָּ֣ם נָקִ֑י וְאָמַ֥ר כָּל־הָעָ֖ם אָמֵֽן׃ {ס} לִיט דִּיקַבֵּיל שֻׁחְדָּא, לְמִקְטַל נְפַשׁ דַּם זַכַּאי; וְיֵימַר כָּל עַמָּא, אָמֵן. {ס}
כז,כו אָר֗וּר אֲשֶׁ֧ר לֹֽא־יָקִ֛ים אֶת־דִּבְרֵ֥י הַתּוֹרָֽה־הַזֹּ֖את לַֽעֲשׂ֣וֹת אוֹתָ֑ם וְאָמַ֥ר כָּל־הָעָ֖ם אָמֵֽן׃ {פ} לִיט, דְּלָא יְקַיֵּים יָת פִּתְגָמֵי אוֹרָיְתָא הָדָא--לְמֶעֱבַד יָתְהוֹן; וְיֵימַר כָּל עַמָּא, אָמֵן. {פ}
כח,א וְהָיָ֗ה אִם־שָׁמ֤וֹעַ תִּשְׁמַע֙ בְּקוֹל֙ יְהוָ֣ה אֱלֹהֶ֔יךָ לִשְׁמֹ֤ר לַֽעֲשׂוֹת֙ אֶת־כָּל־מִצְו‍ֹתָ֔יו אֲשֶׁ֛ר אָֽנֹכִ֥י מְצַוְּךָ֖ הַיּ֑וֹם וּנְתָ֨נְךָ֜ יְהוָ֤ה אֱלֹהֶ֨יךָ֙ עֶלְי֔וֹן עַ֖ל כָּל־גּוֹיֵ֥י הָאָֽרֶץ׃ וִיהֵי, אִם קַבָּלָא תְּקַבֵּיל לְמֵימְרָא דַּייָ אֱלָהָךְ, לְמִטַּר לְמֶעֱבַד יָת כָּל פִּקּוֹדוֹהִי, דַּאֲנָא מְפַקֵּיד לָךְ יוֹמָא דֵּין--וְיִתְּנִנָּךְ יְיָ אֱלָהָךְ, עִלַּאי, עַל, כָּל עַמְמֵי אַרְעָא. 
כח,ב וּבָ֧אוּ עָלֶ֛יךָ כָּל־הַבְּרָכ֥וֹת הָאֵ֖לֶּה וְהִשִּׂיגֻ֑ךָ כִּ֣י תִשְׁמַ֔ע בְּק֖וֹל יְהוָ֥ה אֱלֹהֶֽיךָ׃ וְיֵיתוֹן עֲלָךְ כָּל בִּרְכָתָא הָאִלֵּין, וְיִדְבְּקוּנָּךְ:  אֲרֵי תְּקַבֵּיל, לְמֵימְרָא דַּייָ אֱלָהָךְ. 
כח,ג בָּר֥וּךְ אַתָּ֖ה בָּעִ֑יר וּבָר֥וּךְ אַתָּ֖ה בַּשָּׂדֶֽה׃ בְּרִיךְ אַתְּ, בְּקַרְתָּא; וּבְרִיךְ אַתְּ, בְּחַקְלָא. 
כח,ד בָּר֧וּךְ פְּרִֽי־בִטְנְךָ֛ וּפְרִ֥י אַדְמָֽתְךָ֖ וּפְרִ֣י בְהֶמְתֶּ֑ךָ שְׁגַ֥ר אֲלָפֶ֖יךָ וְעַשְׁתְּר֥וֹת צֹאנֶֽךָ׃ בְּרִיךְ וַלְדָּא דִּמְעָךְ וְאִבָּא דְּאַרְעָךְ, וּוַלְדָּא דִּבְעִירָךְ--בַּקְרֵי תּוֹרָךְ, וְעֶדְרֵי עָנָךְ. 
כח,ה בָּר֥וּךְ טַנְאֲךָ֖ וּמִשְׁאַרְתֶּֽךָ׃ בְּרִיךְ סַלָּךְ, וְאָצוּתָךְ. 
כח,ו בָּר֥וּךְ אַתָּ֖ה בְּבֹאֶ֑ךָ וּבָר֥וּךְ אַתָּ֖ה בְּצֵאתֶֽךָ׃ בְּרִיךְ אַתְּ, בְּמֵיעֲלָךְ; וּבְרִיךְ אַתְּ, בְּמִפְּקָךְ. 
כח,ז יִתֵּ֨ן יְהוָ֤ה אֶת־אֹֽיְבֶ֨יךָ֙ הַקָּמִ֣ים עָלֶ֔יךָ נִגָּפִ֖ים לְפָנֶ֑יךָ בְּדֶ֤רֶךְ אֶחָד֙ יֵֽצְא֣וּ אֵלֶ֔יךָ וּבְשִׁבְעָ֥ה דְרָכִ֖ים יָנ֥וּסוּ לְפָנֶֽיךָ׃ יִתֵּין יְיָ יָת בַּעֲלֵי דְּבָבָךְ דְּקָיְמִין עֲלָךְ, תְּבִירִין קֳדָמָךְ:  בְּאוֹרְחָא חֲדָא יִפְּקוּן לְוָתָךְ, וּבִשְׁבַע אוֹרְחָן יִעְרְקוּן מִן קֳדָמָךְ. 
כח,ח יְצַ֨ו יְהוָ֤ה אִתְּךָ֙ אֶת־הַבְּרָכָ֔ה בַּֽאֲסָמֶ֕יךָ וּבְכֹ֖ל מִשְׁלַ֣ח יָדֶ֑ךָ וּבֵ֣רַכְךָ֔ בָּאָ֕רֶץ אֲשֶׁר־יְהוָ֥ה אֱלֹהֶ֖יךָ נֹתֵ֥ן לָֽךְ׃ יְפַקֵּיד יְיָ לָךְ, יָת בִּרְכָן, בְּאוֹצְרָךְ, וּבְכָל אוֹשָׁטוּת יְדָךְ; וִיבָרְכִנָּךְ--בְּאַרְעָא, דַּייָ אֱלָהָךְ יָהֵיב לָךְ. 
כח,ט יְקִֽימְךָ֨ יְהוָ֥ה לוֹ֙ לְעַ֣ם קָד֔וֹשׁ כַּֽאֲשֶׁ֖ר נִֽשְׁבַּֽע־לָ֑ךְ כִּ֣י תִשְׁמֹ֗ר אֶת־מִצְו‍ֹת֙ יְהוָ֣ה אֱלֹהֶ֔יךָ וְהָֽלַכְתָּ֖ בִּדְרָכָֽיו׃ יְקִימִנָּךְ יְיָ קֳדָמוֹהִי לְעַם קַדִּישׁ, כְּמָא דְּקַיֵּים לָךְ:  אֲרֵי תִּטַּר, יָת פִּקּוֹדַיָּא דַּייָ אֱלָהָךְ, וּתְהָךְ, בְּאוֹרְחָן דְּתָקְנָן קֳדָמוֹהִי. 
כח,י וְרָאוּ֙ כָּל־עַמֵּ֣י הָאָ֔רֶץ כִּ֛י שֵׁ֥ם יְהוָ֖ה נִקְרָ֣א עָלֶ֑יךָ וְיָֽרְא֖וּ מִמֶּֽךָּ׃ וְיִחְזוֹן כָּל עַמְמֵי אַרְעָא, אֲרֵי שְׁמָא דַּייָ אִתְקְרִי עֲלָךְ; וְיִדְחֲלוּן, מִנָּךְ. 
כח,יא וְהוֹתִֽרְךָ֤ יְהוָה֙ לְטוֹבָ֔ה בִּפְרִ֧י בִטְנְךָ֛ וּבִפְרִ֥י בְהֶמְתְּךָ֖ וּבִפְרִ֣י אַדְמָתֶ֑ךָ עַ֚ל הָֽאֲדָמָ֔ה אֲשֶׁ֨ר נִשְׁבַּ֧ע יְהוָ֛ה לַֽאֲבֹתֶ֖יךָ לָ֥תֶת לָֽךְ׃ וְיוֹתְרִנָּךְ יְיָ לְטָבָא, בְּוַלְדָּא דִּמְעָךְ וּבְוַלְדָּא דִּבְעִירָךְ וּבְאִבָּא דְּאַרְעָךְ--עַל, אַרְעָא, דְּקַיֵּים יְיָ לַאֲבָהָתָךְ, לְמִתַּן לָךְ. 
כח,יב יִפְתַּ֣ח יְהוָ֣ה ׀ לְ֠ךָ אֶת־אֽוֹצָר֨וֹ הַטּ֜וֹב אֶת־הַשָּׁמַ֗יִם לָתֵ֤ת מְטַֽר־אַרְצְךָ֙ בְּעִתּ֔וֹ וּלְבָרֵ֕ךְ אֵ֖ת כָּל־מַֽעֲשֵׂ֣ה יָדֶ֑ךָ וְהִלְוִ֨יתָ֙ גּוֹיִ֣ם רַבִּ֔ים וְאַתָּ֖ה לֹ֥א תִלְוֶֽה׃ יִפְתַּח יְיָ לָךְ יָת אוֹצְרֵיהּ טָבָא יָת שְׁמַיָּא, לְמִתַּן מְטַר אַרְעָךְ בְּעִדָּנֵיהּ, וּלְבָרָכָא, יָת כָּל עוּבָדֵי יְדָךְ; וְתוֹזֵיף לְעַמְמִין סַגִּיאִין, וְאַתְּ לָא תְּזִיף. 
כח,יג וּנְתָֽנְךָ֨ יְהוָ֤ה לְרֹאשׁ֙ וְלֹ֣א לְזָנָ֔ב וְהָיִ֨יתָ֙ רַ֣ק לְמַ֔עְלָה וְלֹ֥א תִֽהְיֶ֖ה לְמָ֑טָּה כִּֽי־תִשְׁמַ֞ע אֶל־מִצְו֣‍ֹת ׀ יְהוָ֣ה אֱלֹהֶ֗יךָ אֲשֶׁ֨ר אָֽנֹכִ֧י מְצַוְּךָ֛ הַיּ֖וֹם לִשְׁמֹ֥ר וְלַֽעֲשֽׂוֹת׃ וְיִתְּנִנָּךְ יְיָ תַּקִּיף, וְלָא חַלָּשׁ, וּתְהֵי בְּרַם לְעֵילָא, וְלָא תְּהֵי לְתַחְתָּא:  אֲרֵי תְּקַבֵּיל לְפִקּוֹדַיָּא דַּייָ אֱלָהָךְ, דַּאֲנָא מְפַקֵּיד לָךְ יוֹמָא דֵּין--לְמִטַּר וּלְמֶעֱבַד. 
כח,יד וְלֹ֣א תָס֗וּר מִכָּל־הַדְּבָרִים֙ אֲשֶׁ֨ר אָֽנֹכִ֜י מְצַוֶּ֥ה אֶתְכֶ֛ם הַיּ֖וֹם יָמִ֣ין וּשְׂמֹ֑אול לָלֶ֗כֶת אַֽחֲרֵ֛י אֱלֹהִ֥ים אֲחֵרִ֖ים לְעָבְדָֽם׃ {פ} וְלָא תִּסְטוֹן, מִכָּל פִּתְגָמַיָּא דַּאֲנָא מְפַקֵּיד יָתְכוֹן יוֹמָא דֵּין--לְיַמִּינָא וְלִסְמָאלָא:  לִמְהָךְ, בָּתַר טָעֲוָת עַמְמַיָּא--לְמִפְלַחְהוֹן. {פ}
כח,טו וְהָיָ֗ה אִם־לֹ֤א תִשְׁמַע֙ בְּקוֹל֙ יְהוָ֣ה אֱלֹהֶ֔יךָ לִשְׁמֹ֤ר לַֽעֲשׂוֹת֙ אֶת־כָּל־מִצְו‍ֹתָ֣יו וְחֻקֹּתָ֔יו אֲשֶׁ֛ר אָֽנֹכִ֥י מְצַוְּךָ֖ הַיּ֑וֹם וּבָ֧אוּ עָלֶ֛יךָ כָּל־הַקְּלָל֥וֹת הָאֵ֖לֶּה וְהִשִּׂיגֽוּךָ׃ וִיהֵי, אִם לָא תְּקַבֵּיל לְמֵימְרָא דַּייָ אֱלָהָךְ, לְמִטַּר לְמֶעֱבַד יָת כָּל פִּקּוֹדוֹהִי וּקְיָמוֹהִי, דַּאֲנָא מְפַקֵּיד לָךְ יוֹמָא דֵּין--וְיֵיתוֹן עֲלָךְ כָּל לְוָטַיָּא הָאִלֵּין, וְיִדְבְּקוּנָּךְ. 
כח,טז אָר֥וּר אַתָּ֖ה בָּעִ֑יר וְאָר֥וּר אַתָּ֖ה בַּשָּׂדֶֽה׃ לִיט אַתְּ, בְּקַרְתָּא; וְלִיט אַתְּ, בְּחַקְלָא. 
כח,יז אָר֥וּר טַנְאֲךָ֖ וּמִשְׁאַרְתֶּֽךָ׃ לִיט סַלָּךְ, וְאָצוּתָךְ. 
כח,יח אָר֥וּר פְּרִֽי־בִטְנְךָ֖ וּפְרִ֣י אַדְמָתֶ֑ךָ שְׁגַ֥ר אֲלָפֶ֖יךָ וְעַשְׁתְּרֹ֥ת צֹאנֶֽךָ׃ לִיט וַלְדָּא דִּמְעָךְ, וְאִבָּא דְּאַרְעָךְ--בַּקְרֵי תּוֹרָךְ, וְעֶדְרֵי עָנָךְ. 
כח,יט אָר֥וּר אַתָּ֖ה בְּבֹאֶ֑ךָ וְאָר֥וּר אַתָּ֖ה בְּצֵאתֶֽךָ׃ לִיט אַתְּ, בְּמֵיעֲלָךְ; וְלִיט אַתְּ, בְּמִפְּקָךְ. 
כח,כ יְשַׁלַּ֣ח יְהוָ֣ה ׀ בְּ֠ךָ אֶת־הַמְּאֵרָ֤ה אֶת־הַמְּהוּמָה֙ וְאֶת־הַמִּגְעֶ֔רֶת בְּכָל־מִשְׁלַ֥ח יָֽדְךָ֖ אֲשֶׁ֣ר תַּֽעֲשֶׂ֑ה עַ֣ד הִשָּֽׁמֶדְךָ֤ וְעַד־אֲבָדְךָ֙ מַהֵ֔ר מִפְּנֵ֛י רֹ֥עַ מַֽעֲלָלֶ֖יךָ אֲשֶׁ֥ר עֲזַבְתָּֽנִי׃ יְשַׁלַּח יְיָ בָּךְ יָת מְאֵירְתָא יָת שִׁגֻּשְׁיָא, וְיָת מַזּוֹפִיתָא, בְּכָל אוֹשָׁטוּת יְדָךְ, דְּתַעֲבֵיד--עַד דְּתִשְׁתֵּיצֵי וְעַד דְּתֵיבַד בִּפְרִיעַ, מִן קֳדָם בִּישׁוּת עוּבָדָךְ דִּשְׁבַקְתָּא דַּחְלְתִי. 
כח,כא יַדְבֵּ֧ק יְהוָ֛ה בְּךָ֖ אֶת־הַדָּ֑בֶר עַ֚ד כַּלֹּת֣וֹ אֹֽתְךָ֔ מֵעַל֙ הָֽאֲדָמָ֔ה אֲשֶׁר־אַתָּ֥ה בָא־שָׁ֖מָּה לְרִשְׁתָּֽהּ׃ יַדְבֵּיק יְיָ בָּךְ, יָת מוֹתָא--עַד, דִּישֵׁיצֵי יָתָךְ, מֵעַל אַרְעָא, דְּאַתְּ עָלֵיל לְתַמָּן לְמֵירְתַהּ. 
כח,כב יַכְּכָ֣ה יְ֠הוָה בַּשַּׁחֶ֨פֶת וּבַקַּדַּ֜חַת וּבַדַּלֶּ֗קֶת וּבַֽחַרְחֻר֙ וּבַחֶ֔רֶב וּבַשִּׁדָּפ֖וֹן וּבַיֵּֽרָק֑וֹן וּרְדָפ֖וּךָ עַ֥ד אָבְדֶֽךָ׃ יִמְחֵינָךְ יְיָ בְּשַׁחַפְתָּא וּבְקַדַּחְתָּא וּבִדְלֵיקְתָא, וּבְחַרְחוּרָא וּבְחַרְבָּא, וּבְשַׁדְפָנָא, וּבְיַרְקָנָא; וְיִרְדְּפוּנָּךְ, עַד דְּתֵיבַד. 
כח,כג וְהָי֥וּ שָׁמֶ֛יךָ אֲשֶׁ֥ר עַל־רֹֽאשְׁךָ֖ נְחֹ֑שֶׁת וְהָאָ֥רֶץ אֲשֶׁר־תַּחְתֶּ֖יךָ בַּרְזֶֽל׃ וִיהוֹן שְׁמַיָּא דְּעִלָּוֵי רֵישָׁךְ, חַסִּינִין כִּנְחָשָׁא מִלְּאַחָתָא מִטְרָא; וְאַרְעָא דִּתְחוֹתָךְ, תַּקִּיפָא כְּבַרְזְלָא מִלְּמֶעֱבַד פֵּירִין. 
כח,כד יִתֵּ֧ן יְהוָ֛ה אֶת־מְטַ֥ר אַרְצְךָ֖ אָבָ֣ק וְעָפָ֑ר מִן־הַשָּׁמַ֨יִם֙ יֵרֵ֣ד עָלֶ֔יךָ עַ֖ד הִשָּֽׁמְדָֽךְ׃ יִתֵּין יְיָ יָת מְטַר אַרְעָךְ, אַבְקָא וְעַפְרָא:  מִן שְׁמַיָּא יֵיחוֹת עֲלָךְ, עַד דְּתִשְׁתֵּיצֵי. 
כח,כה יִתֶּנְךָ֨ יְהוָ֥ה ׀ נִגָּף֮ לִפְנֵ֣י אֹֽיְבֶיךָ֒ בְּדֶ֤רֶךְ אֶחָד֙ תֵּצֵ֣א אֵלָ֔יו וּבְשִׁבְעָ֥ה דְרָכִ֖ים תָּנ֣וּס לְפָנָ֑יו וְהָיִ֣יתָ לְזַֽעֲוָ֔ה לְכֹ֖ל מַמְלְכ֥וֹת הָאָֽרֶץ׃ יִתְּנִנָּךְ יְיָ תְּבִיר, קֳדָם סָנְאָךְ--בְּאוֹרְחָא חֲדָא תִּפּוֹק לְוָתֵיהּ, וּבִשְׁבַע אוֹרְחָן תִּעְרוֹק מִן קֳדָמוֹהִי; וּתְהֵי לִזְיָע, לְכֹל מַלְכְוָת אַרְעָא. 
כח,כו וְהָֽיְתָ֤ה נִבְלָֽתְךָ֙ לְמַֽאֲכָ֔ל לְכָל־ע֥וֹף הַשָּׁמַ֖יִם וּלְבֶֽהֱמַ֣ת הָאָ֑רֶץ וְאֵ֖ין מַֽחֲרִֽיד׃ וּתְהֵי נְבִילְתָךְ מְשֻׁגְּרָא לְמֵיכַל, לְכָל עוֹפָא דִּשְׁמַיָּא וְלִבְעִירָא דְּאַרְעָא; וְלֵית, דְּמַנִּיד. 
כח,כז יַכְּכָ֨ה יְהוָ֜ה בִּשְׁחִ֤ין מִצְרַ֨יִם֙ ובעפלים (וּבַטְּחֹרִ֔ים) וּבַגָּרָ֖ב וּבֶחָ֑רֶס אֲשֶׁ֥ר לֹֽא־תוּכַ֖ל לְהֵֽרָפֵֽא׃ יִמְחֵינָךְ יְיָ בְּשִׁחְנָא דְּמִצְרַיִם, וּבִטְחוֹרִין, וּבְגַרְבָּא, וּבַחֲרַס יַבִּישׁ--דְּלָא תִּכּוֹל, לְאִתַּסָּאָה. 
כח,כח יַכְּכָ֣ה יְהוָ֔ה בְּשִׁגָּע֖וֹן וּבְעִוָּר֑וֹן וּבְתִמְה֖וֹן לֵבָֽב׃ יִמְחֵינָךְ יְיָ, בְּטַפְשׁוּתָא וּבְסַמְיוּתָא; וּבְשַׁעְמְמוּת, לִבָּא. 
כח,כט וְהָיִ֜יתָ מְמַשֵּׁ֣שׁ בַּֽצָּהֳרַ֗יִם כַּֽאֲשֶׁ֨ר יְמַשֵּׁ֤שׁ הַֽעִוֵּר֙ בָּֽאֲפֵלָ֔ה וְלֹ֥א תַצְלִ֖יחַ אֶת־דְּרָכֶ֑יךָ וְהָיִ֜יתָ אַ֣ךְ עָשׁ֧וּק וְגָז֛וּל כָּל־הַיָּמִ֖ים וְאֵ֥ין מוֹשִֽׁיעַ׃ וּתְהֵי מְמַשֵּׁישׁ בְּטֵיהֲרָא, כְּמָא דִּימַשֵּׁישׁ עֲוִירָא בְּקַבְלָא, וְלָא תַּצְלַח, יָת אוֹרְחָתָךְ; וּתְהֵי בְּרַם עֲשִׁיק וַאֲנִיס, כָּל יוֹמַיָּא--וְלֵית דְּפָרִיק. 
כח,ל אִשָּׁ֣ה תְאָרֵ֗שׂ וְאִ֤ישׁ אַחֵר֙ ישגלנה (יִשְׁכָּבֶ֔נָּה) בַּ֥יִת תִּבְנֶ֖ה וְלֹֽא־תֵשֵׁ֣ב בּ֑וֹ כֶּ֥רֶם תִּטַּ֖ע וְלֹ֥א תְחַלְּלֶֽנּוּ׃ אִתְּתָא תֵּירוֹס, וּגְבַר אָחֳרָן יִשְׁכְּבִנַּהּ--בֵּיתָא תִּבְנֵי, וְלָא תִּתֵּיב בֵּיהּ; כַּרְמָא תִּצּוּב, וְלָא תַּחֲלִנֵּיהּ. 
כח,לא שֽׁוֹרְךָ֞ טָב֣וּחַ לְעֵינֶ֗יךָ וְלֹ֣א תֹאכַל֮ מִמֶּנּוּ֒ חֲמֹֽרְךָ֙ גָּז֣וּל מִלְּפָנֶ֔יךָ וְלֹ֥א יָשׁ֖וּב לָ֑ךְ צֹֽאנְךָ֙ נְתֻנ֣וֹת לְאֹֽיְבֶ֔יךָ וְאֵ֥ין לְךָ֖ מוֹשִֽׁיעַ׃ תּוֹרָךְ יְהֵי נְכִיס לְעֵינָךְ, וְלָא תֵּיכוֹל מִנֵּיהּ--חֲמָרָךְ יְהֵי אֲנִיס מִן קֳדָמָךְ, וְלָא יְתוּב לָךְ; עָנָךְ מְסִירָן לְבַעֲלֵי דְּבָבָךְ, וְלֵית לָךְ פָּרִיק. 
כח,לב בָּנֶ֨יךָ וּבְנֹתֶ֜יךָ נְתֻנִ֨ים לְעַ֤ם אַחֵר֙ וְעֵינֶ֣יךָ רֹא֔וֹת וְכָל֥וֹת אֲלֵיהֶ֖ם כָּל־הַיּ֑וֹם וְאֵ֥ין לְאֵ֖ל יָדֶֽךָ׃ בְּנָךְ וּבְנָתָךְ מְסִירִין לְעַם אָחֳרָן, וְעֵינָךְ חָזְיָן, וִיסוּפָן בִּגְלַלְהוֹן, כָּל יוֹמָא; וְלֵית חֵילָא, בִּידָךְ. 
כח,לג פְּרִ֤י אַדְמָֽתְךָ֙ וְכָל־יְגִ֣יעֲךָ֔ יֹאכַ֥ל עַ֖ם אֲשֶׁ֣ר לֹֽא־יָדָ֑עְתָּ וְהָיִ֗יתָ רַ֛ק עָשׁ֥וּק וְרָצ֖וּץ כָּל־הַיָּמִֽים׃ אִבָּא דְּאַרְעָךְ וְכָל לֵיאוּתָךְ, יֵיכוֹל עַם דְּלָא יְדַעְתָּא; וּתְהֵי, בְּרַם עֲשִׁיק וּרְעִיעַ--כָּל יוֹמַיָּא. 
כח,לד וְהָיִ֖יתָ מְשֻׁגָּ֑ע מִמַּרְאֵ֥ה עֵינֶ֖יךָ אֲשֶׁ֥ר תִּרְאֶֽה׃ וּתְהֵי, מִשְׁתַּטֵּי, מֵחֵיזוּ עֵינָךְ, דִּתְהֵי חָזֵי. 
כח,לה יַכְּכָ֨ה יְהוָ֜ה בִּשְׁחִ֣ין רָ֗ע עַל־הַבִּרְכַּ֨יִם֙ וְעַל־הַשֹּׁקַ֔יִם אֲשֶׁ֥ר לֹֽא־תוּכַ֖ל לְהֵֽרָפֵ֑א מִכַּ֥ף רַגְלְךָ֖ וְעַ֥ד קָדְקֳדֶֽךָ׃ יִמְחֵינָךְ יְיָ בְּשִׁחְנָא בִּישָׁא, עַל רְכֻבִּין וְעַל שָׁקָן, דְּלָא תִּכּוֹל, לְאִתַּסָּאָה--מִפַּרְסַת רַגְלָךְ, וְעַד מוֹחָךְ. 
כח,לו יוֹלֵ֨ךְ יְהוָ֜ה אֹֽתְךָ֗ וְאֶֽת־מַלְכְּךָ֙ אֲשֶׁ֣ר תָּקִ֣ים עָלֶ֔יךָ אֶל־גּ֕וֹי אֲשֶׁ֥ר לֹֽא־יָדַ֖עְתָּ אַתָּ֣ה וַֽאֲבֹתֶ֑יךָ וְעָבַ֥דְתָּ שָּׁ֛ם אֱלֹהִ֥ים אֲחֵרִ֖ים עֵ֥ץ וָאָֽבֶן׃ יַגְלֵי יְיָ יָתָךְ, וְיָת מַלְכָּךְ דִּתְקִים עֲלָךְ, לְעַם, דְּלָא יְדַעְתָּא אַתְּ וַאֲבָהָתָךְ; וְתִפְלַח תַּמָּן לְעַמְמַיָּא פָּלְחֵי טָעֲוָתָא, אָעָא וְאַבְנָא. 
כח,לז וְהָיִ֣יתָ לְשַׁמָּ֔ה לְמָשָׁ֖ל וְלִשְׁנִינָ֑ה בְּכֹל֙ הָֽעַמִּ֔ים אֲשֶׁר־יְנַֽהֶגְךָ֥ יְהוָ֖ה שָֽׁמָּה׃ וּתְהֵי לְצָדוּ, לִמְתָל וּלְשׁוֹעִי--בְּכֹל, עַמְמַיָּא, דִּידַבְּרִנָּךְ יְיָ, לְתַמָּן. 
כח,לח זֶ֥רַע רַ֖ב תּוֹצִ֣יא הַשָּׂדֶ֑ה וּמְעַ֣ט תֶּֽאֱסֹ֔ף כִּ֥י יַחְסְלֶ֖נּוּ הָֽאַרְבֶּֽה׃ בַּר זְרַע סַגִּי, תַּפֵּיק לְחַקְלָא; וּזְעֵיר תִּכְנוֹשׁ, אֲרֵי יַחְסְלִנֵּיהּ גּוֹבָא. 
כח,לט כְּרָמִ֥ים תִּטַּ֖ע וְעָבָ֑דְתָּ וְיַ֤יִן לֹֽא־תִשְׁתֶּה֙ וְלֹ֣א תֶֽאֱגֹ֔ר כִּ֥י תֹֽאכְלֶ֖נּוּ הַתֹּלָֽעַת׃ כַּרְמִין תִּצּוּב, וְתִפְלַח; וַחֲמַר לָא תִּשְׁתֵּי וְלָא תִּכְנוֹשׁ, אֲרֵי תֵּיכְלִנֵּיהּ תּוֹלַעְתָּא. 
כח,מ זֵיתִ֛ים יִֽהְי֥וּ לְךָ֖ בְּכָל־גְּבוּלֶ֑ךָ וְשֶׁ֨מֶן֙ לֹ֣א תָס֔וּךְ כִּ֥י יִשַּׁ֖ל זֵיתֶֽךָ׃ זֵיתִין יְהוֹן לָךְ, בְּכָל תְּחוּמָךְ; וּמִשְׁחָא לָא תְּסוּךְ, אֲרֵי יִתְּרָן זֵיתָךְ. 
כח,מא בָּנִ֥ים וּבָנ֖וֹת תּוֹלִ֑יד וְלֹֽא־יִהְי֣וּ לָ֔ךְ כִּ֥י יֵֽלְכ֖וּ בַּשֶּֽׁבִי׃ בְּנִין וּבְנָן, תְּלִיד; וְלָא יְהוֹן לָךְ, אֲרֵי יְהָכוּן בְּשִׁבְיָא. 
כח,מב כָּל־עֵֽצְךָ֖ וּפְרִ֣י אַדְמָתֶ֑ךָ יְיָרֵ֖שׁ הַצְּלָצַֽל׃ כָּל אִילָנָךְ, וְאִבָּא דְּאַרְעָךְ, יַחְסְנִנֵּיהּ, סַקָּאָה. 
כח,מג הַגֵּר֙ אֲשֶׁ֣ר בְּקִרְבְּךָ֔ יַֽעֲלֶ֥ה עָלֶ֖יךָ מַ֣עְלָה מָּ֑עְלָה וְאַתָּ֥ה תֵרֵ֖ד מַ֥טָּה מָּֽטָּה׃ תּוֹתַב עֲרַל דְּבֵינָךְ, יְהֵי סָלֵיק עֵיל מִנָּךְ לְעֵילָא לְעֵילָא; וְאַתְּ תְּהֵי נָחֵית, לְתַחְתָּא לְתַחְתָּא. 
כח,מד ה֣וּא יַלְוְךָ֔ וְאַתָּ֖ה לֹ֣א תַלְוֶ֑נּוּ ה֚וּא יִֽהְיֶ֣ה לְרֹ֔אשׁ וְאַתָּ֖ה תִּֽהְיֶ֥ה לְזָנָֽב׃ הוּא יוֹזְפִנָּךְ, וְאַתְּ לָא תּוֹזְפִנֵּיהּ; הוּא יְהֵי תַּקִּיף, וְאַתְּ תְּהֵי חַלָּשׁ. 
כח,מה וּבָ֨אוּ עָלֶ֜יךָ כָּל־הַקְּלָל֣וֹת הָאֵ֗לֶּה וּרְדָפ֨וּךָ֙ וְהִשִּׂיג֔וּךָ עַ֖ד הִשָּֽׁמְדָ֑ךְ כִּי־לֹ֣א שָׁמַ֗עְתָּ בְּקוֹל֙ יְהוָ֣ה אֱלֹהֶ֔יךָ לִשְׁמֹ֛ר מִצְו‍ֹתָ֥יו וְחֻקֹּתָ֖יו אֲשֶׁ֥ר צִוָּֽךְ׃ וְיֵיתוֹן עֲלָךְ כָּל לְוָטַיָּא הָאִלֵּין, וְיִרְדְּפוּנָּךְ וְיִדְבְּקוּנָּךְ, עַד, דְּתִשְׁתֵּיצֵי:  אֲרֵי לָא קַבֵּילְתָּא, לְמֵימְרָא דַּייָ אֱלָהָךְ--לְמִטַּר פִּקּוֹדוֹהִי וּקְיָמוֹהִי, דְּפַקְּדָךְ. 
כח,מו וְהָי֣וּ בְךָ֔ לְא֖וֹת וּלְמוֹפֵ֑ת וּֽבְזַרְעֲךָ֖ עַד־עוֹלָֽם׃ וִיהוֹן בָּךְ, לְאָת וּלְמוֹפָת; וּבִבְנָךְ, עַד עָלְמָא. 
כח,מז תַּ֗חַת אֲשֶׁ֤ר לֹֽא־עָבַ֨דְתָּ֙ אֶת־יְהוָ֣ה אֱלֹהֶ֔יךָ בְּשִׂמְחָ֖ה וּבְט֣וּב לֵבָ֑ב מֵרֹ֖ב כֹּֽל׃ חֲלָף, דְּלָא פְּלַחְתָּא קֳדָם יְיָ אֱלָהָךְ, בְּחֶדְוָא, וּבְשַׁפִּירוּת לִבָּא--מִסְּגֵי, כּוֹלָא. 
כח,מח וְעָֽבַדְתָּ֣ אֶת־אֹֽיְבֶ֗יךָ אֲשֶׁ֨ר יְשַׁלְּחֶ֤נּוּ יְהוָה֙ בָּ֔ךְ בְּרָעָ֧ב וּבְצָמָ֛א וּבְעֵירֹ֖ם וּבְחֹ֣סֶר כֹּ֑ל וְנָתַ֞ן עֹ֤ל בַּרְזֶל֙ עַל־צַוָּארֶ֔ךָ עַ֥ד הִשְׁמִיד֖וֹ אֹתָֽךְ׃ וְתִפְלַח יָת בַּעֲלֵי דְּבָבָךְ, דִּיגָרִנּוּן יְיָ בָּךְ, בְּכַפְנָא וּבְצָהוּתָא וּבְעַרְטִלָּיוּתָא, וּבְחַסִּירוּת כּוֹלָא; וְיִתֵּין נִיר דְּבַרְזֶל, עַל צַוְרָךְ, עַד דִּישֵׁיצֵי, יָתָךְ. 
כח,מט יִשָּׂ֣א יְהוָה֩ עָלֶ֨יךָ גּ֤וֹי מֵֽרָחֹק֙ מִקְצֵ֣ה הָאָ֔רֶץ כַּֽאֲשֶׁ֥ר יִדְאֶ֖ה הַנָּ֑שֶׁר גּ֕וֹי אֲשֶׁ֥ר לֹֽא־תִשְׁמַ֖ע לְשֹׁנֽוֹ׃ יַיְתֵי יְיָ עֲלָךְ עַם מֵרַחִיק מִסְּיָפֵי אַרְעָא, כְּמָא דְּמִשְׁתְּדֵי נִשְׁרָא:  עַמָּא, דְּלָא תִּשְׁמַע לִישָׁנֵיהּ. 
כח,נ גּ֖וֹי עַ֣ז פָּנִ֑ים אֲשֶׁ֨ר לֹֽא־יִשָּׂ֤א פָנִים֙ לְזָקֵ֔ן וְנַ֖עַר לֹ֥א יָחֹֽן׃ עַם, תַּקִּיף אַפִּין, דְּלָא נָסֵיב אַפֵּי סָבָא, וְעַל יָנְקָא לָא מְרַחֵים. 
כח,נא וְ֠אָכַל פְּרִ֨י בְהֶמְתְּךָ֥ וּפְרִֽי־אַדְמָתְךָ֮ עַ֣ד הִשָּֽׁמְדָךְ֒ אֲשֶׁ֨ר לֹֽא־יַשְׁאִ֜יר לְךָ֗ דָּגָן֙ תִּיר֣וֹשׁ וְיִצְהָ֔ר שְׁגַ֥ר אֲלָפֶ֖יךָ וְעַשְׁתְּרֹ֣ת צֹאנֶ֑ךָ עַ֥ד הַֽאֲבִיד֖וֹ אֹתָֽךְ׃ וְיֵיכוֹל וַלְדָּא דִּבְעִירָךְ וְאִבָּא דְּאַרְעָךְ, עַד דְּתִשְׁתֵּיצֵי, דְּלָא יַשְׁאַר לָךְ עֲבוּרָא חַמְרָא וּמִשְׁחָא, בַּקְרֵי תּוֹרָךְ וְעֶדְרֵי עָנָךְ--עַד דְּיוֹבֵיד, יָתָךְ. 
כח,נב וְהֵצַ֨ר לְךָ֜ בְּכָל־שְׁעָרֶ֗יךָ עַ֣ד רֶ֤דֶת חֹֽמֹתֶ֨יךָ֙ הַגְּבֹהֹ֣ת וְהַבְּצֻר֔וֹת אֲשֶׁ֥ר אַתָּ֛ה בֹּטֵ֥חַ בָּהֵ֖ן בְּכָל־אַרְצֶ֑ךָ וְהֵצַ֤ר לְךָ֙ בְּכָל־שְׁעָרֶ֔יךָ בְּכָ֨ל־אַרְצְךָ֔ אֲשֶׁ֥ר נָתַ֛ן יְהוָ֥ה אֱלֹהֶ֖יךָ לָֽךְ׃ וִיעִיק לָךְ בְּכָל קִרְוָךְ, עַד דִּיכַבֵּישׁ שׁוּרָךְ רָמַיָּא וּכְּרִיכַיָּא, דְּאַתְּ רְחִיץ לְאִשְׁתֵּיזָבָא בְּהוֹן, בְּכָל אַרְעָךְ; וִיעִיק לָךְ, בְּכָל קִרְוָךְ, בְּכָל אַרְעָךְ, דִּיהַב יְיָ אֱלָהָךְ לָךְ. 
כח,נג וְאָֽכַלְתָּ֣ פְרִֽי־בִטְנְךָ֗ בְּשַׂ֤ר בָּנֶ֨יךָ֙ וּבְנֹתֶ֔יךָ אֲשֶׁ֥ר נָֽתַן־לְךָ֖ יְהוָ֣ה אֱלֹהֶ֑יךָ בְּמָצוֹר֙ וּבְמָצ֔וֹק אֲשֶׁר־יָצִ֥יק לְךָ֖ אֹֽיְבֶֽךָ׃ וְתֵיכוֹל וַלְדָּא דִּמְעָךְ, בְּסַר בְּנָךְ וּבְנָתָךְ, דִּיהַב לָךְ, יְיָ אֱלָהָךְ--בִּצְיָרָא, וּבְעָקְתָא, דִּיעִיק לָךְ, סָנְאָךְ. 
כח,נד הָאִישׁ֙ הָרַ֣ךְ בְּךָ֔ וְהֶֽעָנֹ֖ג מְאֹ֑ד תֵּרַ֨ע עֵינ֤וֹ בְאָחִיו֙ וּבְאֵ֣שֶׁת חֵיק֔וֹ וּבְיֶ֥תֶר בָּנָ֖יו אֲשֶׁ֥ר יוֹתִֽיר׃ גֻּבְרָא דְּרַכִּיךְ בָּךְ, וְדִמְפֻנַּק לַחְדָּא--תִּבְאַשׁ עֵינֵיהּ בַּאֲחוּהִי וּבְאִתַּת קְיָמֵיהּ, וּבִשְׁאָר בְּנוֹהִי דְּיַשְׁאַר. 
כח,נה מִתֵּ֣ת ׀ לְאַחַ֣ד מֵהֶ֗ם מִבְּשַׂ֤ר בָּנָיו֙ אֲשֶׁ֣ר יֹאכֵ֔ל מִבְּלִ֥י הִשְׁאִֽיר־ל֖וֹ כֹּ֑ל בְּמָצוֹר֙ וּבְמָצ֔וֹק אֲשֶׁ֨ר יָצִ֥יק לְךָ֛ אֹֽיִבְךָ֖ בְּכָל־שְׁעָרֶֽיךָ׃ מִלְּמִתַּן לְחַד מִנְּהוֹן, מִבְּסַר בְּנוֹהִי דְּיֵיכוֹל, מִדְּלָא אִשְׁתְּאַר לֵיהּ, כּוֹלָא--בִּצְיָרָא, וּבְעָקְתָא, דִּיעִיק לָךְ סָנְאָךְ, בְּכָל קִרְוָךְ. 
כח,נו הָֽרַכָּ֨ה בְךָ֜ וְהָֽעֲנֻגָּ֗ה אֲשֶׁ֨ר לֹֽא־נִסְּתָ֤ה כַף־רַגְלָהּ֙ הַצֵּ֣ג עַל־הָאָ֔רֶץ מֵֽהִתְעַנֵּ֖ג וּמֵרֹ֑ךְ תֵּרַ֤ע עֵינָהּ֙ בְּאִ֣ישׁ חֵיקָ֔הּ וּבִבְנָ֖הּ וּבְבִתָּֽהּ׃ דְּרַכִּיכָא בָּךְ וְדִמְפֻנְּקָא, דְּלָא נַסִּיאַת פַּרְסַת רַגְלַהּ לְאַחָתָא עַל אַרְעָא, מִמְּפֻנְּקוּ, וּמֵרַכִּיכוּ--תִּבְאַשׁ עֵינַהּ בִּגְבַר קְיָמַהּ, וּבִבְרַהּ וּבִבְרַתַּהּ. 
כח,נז וּֽבְשִׁלְיָתָ֞הּ הַיּוֹצֵ֣ת ׀ מִבֵּ֣ין רַגְלֶ֗יהָ וּבְבָנֶ֨יהָ֙ אֲשֶׁ֣ר תֵּלֵ֔ד כִּֽי־תֹאכְלֵ֥ם בְּחֹֽסֶר־כֹּ֖ל בַּסָּ֑תֶר בְּמָצוֹר֙ וּבְמָצ֔וֹק אֲשֶׁ֨ר יָצִ֥יק לְךָ֛ אֹֽיִבְךָ֖ בִּשְׁעָרֶֽיךָ׃ וּבִזְעֵיר בְּנַהָא דְּיִפְּקוּן מִנַּהּ, וּבִבְנַהָא דִּתְלִיד, אֲרֵי תֵּיכְלִנּוּן בְּחַסִּירוּת כּוֹלָא, בְּסִתְרָא--בִּצְיָרָא, וּבְעָקְתָא, דִּיעִיק לָךְ סָנְאָךְ, בְּקִרְוָךְ. 
כח,נח אִם־לֹ֨א תִשְׁמֹ֜ר לַֽעֲשׂ֗וֹת אֶת־כָּל־דִּבְרֵי֙ הַתּוֹרָ֣ה הַזֹּ֔את הַכְּתֻבִ֖ים בַּסֵּ֣פֶר הַזֶּ֑ה לְ֠יִרְאָה אֶת־הַשֵּׁ֞ם הַנִּכְבָּ֤ד וְהַנּוֹרָא֙ הַזֶּ֔ה אֵ֖ת יְהוָ֥ה אֱלֹהֶֽיךָ׃ אִם לָא תִּטַּר לְמֶעֱבַד, יָת כָּל פִּתְגָמֵי אוֹרָיְתָא הָדָא, דִּכְתִיבִין, בְּסִפְרָא הָדֵין:  לְמִדְחַל יָת שְׁמָא יַקִּירָא וּדְחִילָא, הָדֵין--יָת, יְיָ אֱלָהָךְ. 
כח,נט וְהִפְלָ֤א יְהוָה֙ אֶת־מַכֹּ֣תְךָ֔ וְאֵ֖ת מַכּ֣וֹת זַרְעֶ֑ךָ מַכּ֤וֹת גְּדֹלֹת֙ וְנֶ֣אֱמָנ֔וֹת וָֽחֳלָיִ֖ם רָעִ֥ים וְנֶֽאֱמָנִֽים׃ וְיַפְרֵישׁ יְיָ יָת מַחָתָךְ, וְיָת מַחָת בְּנָךְ:  מַחָן רַבְרְבָן וּמְהֵימְנָן, וּמַכְתָּשִׁין בִּישִׁין וּמְהֵימְנִין. 
כח,ס וְהֵשִׁ֣יב בְּךָ֗ אֵ֚ת כָּל־מַדְוֵ֣ה מִצְרַ֔יִם אֲשֶׁ֥ר יָגֹ֖רְתָּ מִפְּנֵיהֶ֑ם וְדָֽבְק֖וּ בָּֽךְ׃ וְיָתִיב בָּךְ, יָת כָּל מַכְתָּשֵׁי מִצְרַיִם, דִּדְחֵילְתָּא, מִן קֳדָמֵיהוֹן; וְיִדְבְּקוּן, בָּךְ. 
כח,סא גַּ֤ם כָּל־חֳלִי֙ וְכָל־מַכָּ֔ה אֲשֶׁר֙ לֹ֣א כָת֔וּב בְּסֵ֖פֶר הַתּוֹרָ֣ה הַזֹּ֑את יַעְלֵ֤ם יְהוָה֙ עָלֶ֔יךָ עַ֖ד הִשָּֽׁמְדָֽךְ׃ אַף כָּל מְרַע, וְכָל מַחָא, דְּלָא כְּתִיבִין, בִּסְפַר אוֹרָיְתָא הָדָא--יַיְתֵינוּן יְיָ עֲלָךְ, עַד דְּתִשְׁתֵּיצֵי. 
כח,סב וְנִשְׁאַרְתֶּם֙ בִּמְתֵ֣י מְעָ֔ט תַּ֚חַת אֲשֶׁ֣ר הֱיִיתֶ֔ם כְּכֽוֹכְבֵ֥י הַשָּׁמַ֖יִם לָרֹ֑ב כִּֽי־לֹ֣א שָׁמַ֔עְתָּ בְּק֖וֹל יְהוָ֥ה אֱלֹהֶֽיךָ׃ וְתִשְׁתְּאַרוּן, בְּעַם זְעֵיר, חֲלָף דַּהֲוֵיתוֹן, כְּכוֹכְבֵי שְׁמַיָּא לִסְגֵּי:  אֲרֵי לָא קַבֵּילְתָּא, לְמֵימְרָא דַּייָ אֱלָהָךְ. 
כח,סג וְ֠הָיָה כַּֽאֲשֶׁר־שָׂ֨שׂ יְהוָ֜ה עֲלֵיכֶ֗ם לְהֵיטִ֣יב אֶתְכֶם֮ וּלְהַרְבּ֣וֹת אֶתְכֶם֒ כֵּ֣ן יָשִׂ֤ישׂ יְהוָה֙ עֲלֵיכֶ֔ם לְהַֽאֲבִ֥יד אֶתְכֶ֖ם וּלְהַשְׁמִ֣יד אֶתְכֶ֑ם וְנִסַּחְתֶּם֙ מֵעַ֣ל הָֽאֲדָמָ֔ה אֲשֶׁר־אַתָּ֥ה בָא־שָׁ֖מָּה לְרִשְׁתָּֽהּ׃ וִיהֵי כְּמָא דַּחְדִּי יְיָ עֲלֵיכוֹן, לְאֵיטָבָא לְכוֹן וּלְאַסְגָּאָה יָתְכוֹן--כֵּין יִחְדֵּי יְיָ עֲלֵיכוֹן, לְאוֹבָדָא יָתְכוֹן וּלְשֵׁיצָאָה יָתְכוֹן; וְתִטַּלְטְלוּן מֵעַל אַרְעָא, דְּאַתְּ עָלֵיל לְתַמָּן לְמֵירְתַהּ. 
כח,סד וֶהֱפִֽיצְךָ֤ יְהוָה֙ בְּכָל־הָ֣עַמִּ֔ים מִקְצֵ֥ה הָאָ֖רֶץ וְעַד־קְצֵ֣ה הָאָ֑רֶץ וְעָבַ֨דְתָּ שָּׁ֜ם אֱלֹהִ֣ים אֲחֵרִ֗ים אֲשֶׁ֧ר לֹֽא־יָדַ֛עְתָּ אַתָּ֥ה וַֽאֲבֹתֶ֖יךָ עֵ֥ץ וָאָֽבֶן׃ וִיבַדְּרִנָּךְ יְיָ בְּכָל עַמְמַיָּא, מִסְּיָפֵי אַרְעָא וְעַד סְיָפֵי אַרְעָא; וְתִפְלַח תַּמָּן לְעַמְמַיָּא פָּלְחֵי טָעֲוָתָא, דְּלָא יְדַעְתָּא אַתְּ וַאֲבָהָתָךְ--אָעָא וְאַבְנָא. 
כח,סה וּבַגּוֹיִ֤ם הָהֵם֙ לֹ֣א תַרְגִּ֔יעַ וְלֹֽא־יִהְיֶ֥ה מָנ֖וֹחַ לְכַף־רַגְלֶ֑ךָ וְנָתַן֩ יְהוָ֨ה לְךָ֥ שָׁם֙ לֵ֣ב רַגָּ֔ז וְכִלְי֥וֹן עֵינַ֖יִם וְדַֽאֲב֥וֹן נָֽפֶשׁ׃ וּבְעַמְמַיָּא הָאִנּוּן לָא תְּנוּחַ, וְלָא יְהֵי מְנָח לְפַרְסַת רַגְלָךְ; וְיִתֵּין יְיָ לָךְ תַּמָּן לֵב דָּחוּל, וְחַשְׁכוּת עַיְנִין וּמַפְּחָן נְפַשׁ. 
כח,סו וְהָי֣וּ חַיֶּ֔יךָ תְּלֻאִ֥ים לְךָ֖ מִנֶּ֑גֶד וּפָֽחַדְתָּ֙ לַ֣יְלָה וְיוֹמָ֔ם וְלֹ֥א תַֽאֲמִ֖ין בְּחַיֶּֽיךָ׃ וִיהוֹן חַיָּךְ, תְּלָן לָךְ מִקֳּבֵיל; וּתְהֵי תָּוַהּ בְּלֵילְיָא וּבִימָמָא, וְלָא תְּהֵימֵין בְּחַיָּךְ. 
כח,סז בַּבֹּ֤קֶר תֹּאמַר֙ מִֽי־יִתֵּ֣ן עֶ֔רֶב וּבָעֶ֥רֶב תֹּאמַ֖ר מִֽי־יִתֵּ֣ן בֹּ֑קֶר מִפַּ֤חַד לְבָֽבְךָ֙ אֲשֶׁ֣ר תִּפְחָ֔ד וּמִמַּרְאֵ֥ה עֵינֶ֖יךָ אֲשֶׁ֥ר תִּרְאֶֽה׃ בְּצַפְרָא תֵּימַר מַאן יִתֵּין רַמְשָׁא, וּבְרַמְשָׁא תֵּימַר מַאן יִתֵּין צַפְרָא--מִתִּוְהַת לִבָּךְ דִּתְהֵי תָּוַהּ, וּמֵחֵיזוּ עֵינָךְ דִּתְהֵי חָזֵי. 
כח,סח וֶהֱשִֽׁיבְךָ֨ יְהוָ֥ה ׀ מִצְרַיִם֮ בָּֽאֳנִיּוֹת֒ בַּדֶּ֨רֶךְ֙ אֲשֶׁ֣ר אָמַ֣רְתִּֽי לְךָ֔ לֹֽא־תֹסִ֥יף ע֖וֹד לִרְאֹתָ֑הּ וְהִתְמַכַּרְתֶּ֨ם שָׁ֧ם לְאֹֽיְבֶ֛יךָ לַֽעֲבָדִ֥ים וְלִשְׁפָח֖וֹת וְאֵ֥ין קֹנֶֽה׃ {ס} וְיָתִיבִנָּךְ יְיָ לְמִצְרַיִם, בִּסְפִינָן, בְּאוֹרְחָא דַּאֲמַרִית לָךְ, לָא תּוֹסֵיף עוֹד לְמִחְזַהּ; וְתִזְדַּבְּנוּן תַּמָּן לְבַעֲלֵי דְּבָבֵיכוֹן לְעַבְדִּין וּלְאַמְהָן, וְלֵית דְּקָנֵי. {ס}
כח,סט אֵלֶּה֩ דִבְרֵ֨י הַבְּרִ֜ית אֲֽשֶׁר־צִוָּ֧ה יְהוָ֣ה אֶת־מֹשֶׁ֗ה לִכְרֹ֛ת אֶת־בְּנֵ֥י יִשְׂרָאֵ֖ל בְּאֶ֣רֶץ מוֹאָ֑ב מִלְּבַ֣ד הַבְּרִ֔ית אֲשֶׁר־כָּרַ֥ת אִתָּ֖ם בְּחֹרֵֽב׃ {פ} אִלֵּין פִּתְגָמֵי קְיָמָא דְּפַקֵּיד יְיָ יָת מֹשֶׁה, לְמִגְזַר עִם בְּנֵי יִשְׂרָאֵל--בְּאַרְעָא דְּמוֹאָב:  בָּר מִקְּיָמָא, דִּגְזַר עִמְּהוֹן בְּחוֹרֵב. {פ}
כט,א וַיִּקְרָ֥א מֹשֶׁ֛ה אֶל־כָּל־יִשְׂרָאֵ֖ל וַיֹּ֣אמֶר אֲלֵהֶ֑ם אַתֶּ֣ם רְאִיתֶ֗ם אֵ֣ת כָּל־אֲשֶׁר֩ עָשָׂ֨ה יְהוָ֤ה לְעֵֽינֵיכֶם֙ בְּאֶ֣רֶץ מִצְרַ֔יִם לְפַרְעֹ֥ה וּלְכָל־עֲבָדָ֖יו וּלְכָל־אַרְצֽוֹ׃ וּקְרָא מֹשֶׁה לְכָל יִשְׂרָאֵל, וַאֲמַר לְהוֹן:  אַתּוּן חֲזֵיתוֹן, יָת כָּל דַּעֲבַד יְיָ לְעֵינֵיכוֹן בְּאַרְעָא דְּמִצְרַיִם, לְפַרְעֹה וּלְכָל עַבְדּוֹהִי, וּלְכָל אַרְעֵיהּ. 
כט,ב הַמַּסּוֹת֙ הַגְּדֹלֹ֔ת אֲשֶׁ֥ר רָא֖וּ עֵינֶ֑יךָ הָֽאֹתֹ֧ת וְהַמֹּֽפְתִ֛ים הַגְּדֹלִ֖ים הָהֵֽם׃ נִסִּין, רַבְרְבָן, דַּחֲזַאָה, עֵינָךְ--אָתַיָּא וּמוֹפְתַיָּא, רַבְרְבַיָּא הָאִנּוּן. 
כט,ג וְלֹֽא־נָתַן֩ יְהוָ֨ה לָכֶ֥ם לֵב֙ לָדַ֔עַת וְעֵינַ֥יִם לִרְא֖וֹת וְאָזְנַ֣יִם לִשְׁמֹ֑עַ עַ֖ד הַיּ֥וֹם הַזֶּֽה׃ וְלָא יְהַב יְיָ לְכוֹן לִבָּא לְמִדַּע, וְעַיְנִין לְמִחְזֵי וְאֻדְנִין לְמִשְׁמַע, עַד, יוֹמָא הָדֵין. 
כט,ד וָֽאוֹלֵ֥ךְ אֶתְכֶ֛ם אַרְבָּעִ֥ים שָׁנָ֖ה בַּמִּדְבָּ֑ר לֹֽא־בָל֤וּ שַׂלְמֹֽתֵיכֶם֙ מֵֽעֲלֵיכֶ֔ם וְנַֽעַלְךָ֥ לֹֽא־בָלְתָ֖ה מֵעַ֥ל רַגְלֶֽךָ׃ וְדַבַּרִית יָתְכוֹן אַרְבְּעִין שְׁנִין, בְּמַדְבְּרָא; לָא בְּלִיאַת כְּסוּתְכוֹן מִנְּכוֹן, וּמְסָנָךְ לָא עֲדוֹ מֵעַל רַגְלָךְ. 
כט,ה לֶ֚חֶם לֹ֣א אֲכַלְתֶּ֔ם וְיַ֥יִן וְשֵׁכָ֖ר לֹ֣א שְׁתִיתֶ֑ם לְמַ֨עַן֙ תֵּֽדְע֔וּ כִּ֛י אֲנִ֥י יְהוָ֖ה אֱלֹֽהֵיכֶֽם׃ לַחְמָא לָא אֲכַלְתּוּן, וַחֲמַר חֲדַת וְעַתִּיק לָא שְׁתֵיתוֹן--בְּדִיל, דְּתִדְּעוּן, אֲרֵי אֲנָא יְיָ, אֱלָהֲכוֹן. 
כט,ו וַתָּבֹ֖אוּ אֶל־הַמָּק֣וֹם הַזֶּ֑ה וַיֵּצֵ֣א סִיחֹ֣ן מֶֽלֶךְ־חֶ֠שְׁבּוֹן וְע֨וֹג מֶֽלֶךְ־הַבָּשָׁ֧ן לִקְרָאתֵ֛נוּ לַמִּלְחָמָ֖ה וַנַּכֵּֽם׃ וַאֲתֵיתוֹן, לְאַתְרָא הָדֵין; וּנְפַק סִיחוֹן מַלְכָּא דְּחֶשְׁבּוֹן וְעוֹג מַלְכָּא דְּמַתְנַן לְקַדָּמוּתַנָא, לְאָגָחָא קְרָבָא--וּמְחֵינָנוּן. 
כט,ז וַנִּקַּח֙ אֶת־אַרְצָ֔ם וַנִּתְּנָ֣הּ לְנַֽחֲלָ֔ה לָרֽאוּבֵנִ֖י וְלַגָּדִ֑י וְלַֽחֲצִ֖י שֵׁ֥בֶט הַֽמְנַשִּֽׁי׃ וּנְסֵיבְנָא, יָת אֲרַעְהוֹן, וִיהַבְנַהּ לְאַחְסָנָא, לְשֵׁיבַט רְאוּבֵן וּלְשֵׁיבַט גָּד--וּלְפַלְגוּת, שִׁבְטָא דִּמְנַשֶּׁה. 
כט,ח וּשְׁמַרְתֶּ֗ם אֶת־דִּבְרֵי֙ הַבְּרִ֣ית הַזֹּ֔את וַֽעֲשִׂיתֶ֖ם אֹתָ֑ם לְמַ֣עַן תַּשְׂכִּ֔ילוּ אֵ֖ת כָּל־אֲשֶׁ֥ר תַּֽעֲשֽׂוּן׃ {פ} וְתִטְּרוּן, יָת פִּתְגָמֵי קְיָמָא הָדֵין, וְתַעְבְּדוּן, יָתְהוֹן--בְּדִיל דְּתַצְלְחוּן, יָת כָּל דְּתַעְבְּדוּן. {פ}
