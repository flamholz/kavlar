שמיני

ט,א וַֽיְהִי֙ בַּיּ֣וֹם הַשְּׁמִינִ֔י קָרָ֣א מֹשֶׁ֔ה לְאַֽהֲרֹ֖ן וּלְבָנָ֑יו וּלְזִקְנֵ֖י יִשְׂרָאֵֽל׃ וַהֲוָה, בְּיוֹמָא תְּמִינָאָה, קְרָא מֹשֶׁה, לְאַהֲרוֹן וְלִבְנוֹהִי--וּלְסָבֵי, יִשְׂרָאֵל. 
ט,ב וַיֹּ֣אמֶר אֶֽל־אַהֲרֹ֗ן קַח־לְ֠ךָ עֵ֣גֶל בֶּן־בָּקָ֧ר לְחַטָּ֛את וְאַ֥יִל לְעֹלָ֖ה תְּמִימִ֑ם וְהַקְרֵ֖ב לִפְנֵ֥י יְהוָֽה׃ וַאֲמַר לְאַהֲרוֹן, סַב לָךְ עֵיגַל בַּר תּוֹרֵי לְחַטָּתָא וּדְכַר לַעֲלָתָא--שַׁלְמִין; וְקָרֵיב, קֳדָם יְיָ. 
ט,ג וְאֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל תְּדַבֵּ֣ר לֵאמֹ֑ר קְח֤וּ שְׂעִיר־עִזִּים֙ לְחַטָּ֔את וְעֵ֨גֶל וָכֶ֧בֶשׂ בְּנֵֽי־שָׁנָ֛ה תְּמִימִ֖ם לְעֹלָֽה׃ וְעִם בְּנֵי יִשְׂרָאֵל, תְּמַלֵּיל לְמֵימַר:  סַבוּ צְפִיר בַּר עִזִּין לְחַטָּתָא, וְעֵיגַל וְאִמַּר בְּנֵי שְׁנָא שַׁלְמִין לַעֲלָתָא. 
ט,ד וְשׁ֨וֹר וָאַ֜יִל לִשְׁלָמִ֗ים לִזְבֹּ֨חַ֙ לִפְנֵ֣י יְהוָ֔ה וּמִנְחָ֖ה בְּלוּלָ֣ה בַשָּׁ֑מֶן כִּ֣י הַיּ֔וֹם יְהוָ֖ה נִרְאָ֥ה אֲלֵיכֶֽם׃ וְתוֹר וּדְכַר לְנִכְסַת קֻדְשַׁיָּא, לְדַבָּחָא קֳדָם יְיָ, וּמִנְחָתָא, דְּפִילָא בִּמְשַׁח:  אֲרֵי יוֹמָא דֵּין, יְקָרָא דַּייָ מִתְגְּלֵי לְכוֹן. 
ט,ה וַיִּקְח֗וּ אֵ֚ת אֲשֶׁ֣ר צִוָּ֣ה מֹשֶׁ֔ה אֶל־פְּנֵ֖י אֹ֣הֶל מוֹעֵ֑ד וַֽיִּקְרְבוּ֙ כָּל־הָ֣עֵדָ֔ה וַיַּֽעַמְד֖וּ לִפְנֵ֥י יְהוָֽה׃ וְקָרִיבוּ, יָת דְּפַקֵּיד מֹשֶׁה, לִקְדָם, מַשְׁכַּן זִמְנָא; וּקְרִיבוּ, כָּל כְּנִשְׁתָּא, וְקָמוּ, קֳדָם יְיָ. 
ט,ו וַיֹּ֣אמֶר מֹשֶׁ֔ה זֶ֧ה הַדָּבָ֛ר אֲשֶׁר־צִוָּ֥ה יְהוָ֖ה תַּֽעֲשׂ֑וּ וְיֵרָ֥א אֲלֵיכֶ֖ם כְּב֥וֹד יְהוָֽה׃ וַאֲמַר מֹשֶׁה, דֵּין פִּתְגָמָא דְּפַקֵּיד יְיָ תַּעְבְּדוּן--וְיִתְגְּלֵי לְכוֹן, יְקָרָא דַּייָ. 
ט,ז וַיֹּ֨אמֶר מֹשֶׁ֜ה אֶֽל־אַהֲרֹ֗ן קְרַ֤ב אֶל־הַמִּזְבֵּ֨חַ֙ וַֽעֲשֵׂ֞ה אֶת־חַטָּֽאתְךָ֙ וְאֶת־עֹ֣לָתֶ֔ךָ וְכַפֵּ֥ר בַּֽעַדְךָ֖ וּבְעַ֣ד הָעָ֑ם וַֽעֲשֵׂ֞ה אֶת־קָרְבַּ֤ן הָעָם֙ וְכַפֵּ֣ר בַּֽעֲדָ֔ם כַּֽאֲשֶׁ֖ר צִוָּ֥ה יְהוָֽה׃ וַאֲמַר מֹשֶׁה לְאַהֲרוֹן, קְרַב לְמַדְבְּחָא וַעֲבֵיד יָת חַטָּתָךְ וְיָת עֲלָתָךְ, וְכַפַּר עֲלָךְ, וְעַל עַמָּא; וַעֲבֵיד יָת קֻרְבַּן עַמָּא, וְכַפַּר עֲלֵיהוֹן--כְּמָא דְּפַקֵּיד יְיָ. 
ט,ח וַיִּקְרַ֥ב אַֽהֲרֹ֖ן אֶל־הַמִּזְבֵּ֑חַ וַיִּשְׁחַ֛ט אֶת־עֵ֥גֶל הַֽחַטָּ֖את אֲשֶׁר־לֽוֹ׃ וּקְרֵיב אַהֲרוֹן, לְמַדְבְּחָא; וּנְכַס יָת עִגְלָא דְּחַטָּתָא, דִּילֵיהּ. 
ט,ט וַ֠יַּקְרִבוּ בְּנֵ֨י אַֽהֲרֹ֣ן אֶת־הַדָּם֮ אֵלָיו֒ וַיִּטְבֹּ֤ל אֶצְבָּעוֹ֙ בַּדָּ֔ם וַיִּתֵּ֖ן עַל־קַרְנ֣וֹת הַמִּזְבֵּ֑חַ וְאֶת־הַדָּ֣ם יָצַ֔ק אֶל־יְס֖וֹד הַמִּזְבֵּֽחַ׃ וְקָרִיבוּ בְּנֵי אַהֲרוֹן יָת דְּמָא, לֵיהּ, וּטְבַל אֶצְבְּעֵיהּ בִּדְמָא, וִיהַב עַל קַרְנָת מַדְבְּחָא; וְיָת דְּמָא אֲרֵיק, לִיסוֹדָא דְּמַדְבְּחָא. 
ט,י וְאֶת־הַחֵ֨לֶב וְאֶת־הַכְּלָיֹ֜ת וְאֶת־הַיֹּתֶ֤רֶת מִן־הַכָּבֵד֙ מִן־הַ֣חַטָּ֔את הִקְטִ֖יר הַמִּזְבֵּ֑חָה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ וְיָת תַּרְבָּא וְיָת כּוֹלְיָתָא וְיָת חַצְרָא מִן כַּבְדָּא, מִן חַטָּתָא--אַסֵּיק, לְמַדְבְּחָא:  כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. 
ט,יא וְאֶת־הַבָּשָׂ֖ר וְאֶת־הָע֑וֹר שָׂרַ֣ף בָּאֵ֔שׁ מִח֖וּץ לַֽמַּחֲנֶֽה׃ וְיָת בִּסְרָא, וְיָת מַשְׁכָּא, אוֹקֵיד בְּנוּרָא, מִבַּרָא לְמַשְׁרִיתָא. 
ט,יב וַיִּשְׁחַ֖ט אֶת־הָֽעֹלָ֑ה וַ֠יַּמְצִאוּ בְּנֵ֨י אַֽהֲרֹ֤ן אֵלָיו֙ אֶת־הַדָּ֔ם וַיִּזְרְקֵ֥הוּ עַל־הַמִּזְבֵּ֖חַ סָבִֽיב׃ וּנְכַס, יָת עֲלָתָא; וְאַמְטִיאוּ בְּנֵי אַהֲרוֹן לֵיהּ, יָת דְּמָא, וְזַרְקֵיהּ עַל מַדְבְּחָא, סְחוֹר סְחוֹר. 
ט,יג וְאֶת־הָֽעֹלָ֗ה הִמְצִ֧יאוּ אֵלָ֛יו לִנְתָחֶ֖יהָ וְאֶת־הָרֹ֑אשׁ וַיַּקְטֵ֖ר עַל־הַמִּזְבֵּֽחַ׃ וְיָת עֲלָתָא, אַמְטִיאוּ לֵיהּ לְאֶבְרַהָא--וְיָת רֵישָׁא; וְאַסֵּיק, עַל מַדְבְּחָא. 
ט,יד וַיִּרְחַ֥ץ אֶת־הַקֶּ֖רֶב וְאֶת־הַכְּרָעָ֑יִם וַיַּקְטֵ֥ר עַל־הָֽעֹלָ֖ה הַמִּזְבֵּֽחָה׃ וְחַלֵּיל יָת גַּוָּא, וְיָת כְּרָעַיָּא; וְאַסֵּיק עַל עֲלָתָא, לְמַדְבְּחָא. 
ט,טו וַיַּקְרֵ֕ב אֵ֖ת קָרְבַּ֣ן הָעָ֑ם וַיִּקַּ֞ח אֶת־שְׂעִ֤יר הַֽחַטָּאת֙ אֲשֶׁ֣ר לָעָ֔ם וַיִּשְׁחָטֵ֥הוּ וַֽיְחַטְּאֵ֖הוּ כָּֽרִאשֽׁוֹן׃ וְקָרֵיב, יָת קֻרְבַּן עַמָּא; וּנְסֵיב יָת צְפִירָא דְּחַטָּתָא, דִּלְעַמָּא, וְנַכְסֵיהּ וְכַפַּר בִּדְמֵיהּ, כְּקַדְמָאָה. 
ט,טז וַיַּקְרֵ֖ב אֶת־הָֽעֹלָ֑ה וַֽיַּעֲשֶׂ֖הָ כַּמִּשְׁפָּֽט׃ וְקָרֵיב, יָת עֲלָתָא; וְעַבְדַהּ, כְּדַחְזֵי. 
ט,יז וַיַּקְרֵב֮ אֶת־הַמִּנְחָה֒ וַיְמַלֵּ֤א כַפּוֹ֙ מִמֶּ֔נָּה וַיַּקְטֵ֖ר עַל־הַמִּזְבֵּ֑חַ מִלְּבַ֖ד עֹלַ֥ת הַבֹּֽקֶר׃ וְקָרֵיב, יָת מִנְחָתָא, וּמְלָא יְדֵיהּ מִנַּהּ, וְאַסֵּיק עַל מַדְבְּחָא--בָּר מֵעֲלַת צַפְרָא. 
ט,יח וַיִּשְׁחַ֤ט אֶת־הַשּׁוֹר֙ וְאֶת־הָאַ֔יִל זֶ֥בַח הַשְּׁלָמִ֖ים אֲשֶׁ֣ר לָעָ֑ם וַ֠יַּמְצִאוּ בְּנֵ֨י אַֽהֲרֹ֤ן אֶת־הַדָּם֙ אֵלָ֔יו וַיִּזְרְקֵ֥הוּ עַל־הַמִּזְבֵּ֖חַ סָבִֽיב׃ וּנְכַס יָת תּוֹרָא וְיָת דִּכְרָא, נִכְסַת קֻדְשַׁיָּא דִּלְעַמָּא; וְאַמְטִיאוּ בְּנֵי אַהֲרוֹן יָת דְּמָא, לֵיהּ, וְזַרְקֵיהּ עַל מַדְבְּחָא, סְחוֹר סְחוֹר. 
ט,יט וְאֶת־הַֽחֲלָבִ֖ים מִן־הַשּׁ֑וֹר וּמִ֨ן־הָאַ֔יִל הָֽאַלְיָ֤ה וְהַֽמְכַסֶּה֙ וְהַכְּלָיֹ֔ת וְיֹתֶ֖רֶת הַכָּבֵֽד׃ וְיָת תַּרְבַּיָּא, מִן תּוֹרָא; וּמִן דִּכְרָא--אַלְיְתָא וְחָפֵי גַּוָּא וְכוֹלְיָתָא, וַחֲצַר כַּבְדָּא. 
ט,כ וַיָּשִׂ֥ימוּ אֶת־הַֽחֲלָבִ֖ים עַל־הֶֽחָז֑וֹת וַיַּקְטֵ֥ר הַֽחֲלָבִ֖ים הַמִּזְבֵּֽחָה׃ וְשַׁוִּיאוּ יָת תַּרְבַּיָּא, עַל חֲדָוָתָא; וְאַסֵּיק תַּרְבַּיָּא, לְמַדְבְּחָא. 
ט,כא וְאֵ֣ת הֶֽחָז֗וֹת וְאֵת֙ שׁ֣וֹק הַיָּמִ֔ין הֵנִ֧יף אַֽהֲרֹ֛ן תְּנוּפָ֖ה לִפְנֵ֣י יְהוָ֑ה כַּֽאֲשֶׁ֖ר צִוָּ֥ה מֹשֶֽׁה׃ וְיָת חֲדָוָתָא, וְיָת שָׁקָא דְּיַמִּינָא, אֲרֵים אַהֲרוֹן אֲרָמָא, קֳדָם יְיָ--כְּמָא דְּפַקֵּיד מֹשֶׁה. 
ט,כב וַיִּשָּׂ֨א אַֽהֲרֹ֧ן אֶת־יָדָ֛ו אֶל־הָעָ֖ם וַֽיְבָרְכֵ֑ם וַיֵּ֗רֶד מֵֽעֲשֹׂ֧ת הַֽחַטָּ֛את וְהָֽעֹלָ֖ה וְהַשְּׁלָמִֽים׃ וַאֲרֵים אַהֲרוֹן יָת יְדוֹהִי לְעַמָּא, וּבָרֵיכִנּוּן; וּנְחַת, מִלְּמֶעֱבַד חַטָּתָא וַעֲלָתָא--וְנִכְסַת קֻדְשַׁיָּא. 
ט,כג וַיָּבֹ֨א מֹשֶׁ֤ה וְאַֽהֲרֹן֙ אֶל־אֹ֣הֶל מוֹעֵ֔ד וַיֵּ֣צְא֔וּ וַֽיְבָרְכ֖וּ אֶת־הָעָ֑ם וַיֵּרָ֥א כְבוֹד־יְהוָ֖ה אֶל־כָּל־הָעָֽם׃ וְעָאל מֹשֶׁה וְאַהֲרוֹן, לְמַשְׁכַּן זִמְנָא, וּנְפַקוּ, וּבָרִיכוּ יָת עַמָּא; וְאִתְגְּלִי יְקָרָא דַּייָ, לְכָל עַמָּא. 
ט,כד וַתֵּ֤צֵא אֵשׁ֙ מִלִּפְנֵ֣י יְהוָ֔ה וַתֹּ֨אכַל֙ עַל־הַמִּזְבֵּ֔חַ אֶת־הָֽעֹלָ֖ה וְאֶת־הַֽחֲלָבִ֑ים וַיַּ֤רְא כָּל־הָעָם֙ וַיָּרֹ֔נּוּ וַֽיִּפְּל֖וּ עַל־פְּנֵיהֶֽם׃ וּנְפַקַת אִישָׁתָא, מִן קֳדָם יְיָ, וַאֲכַלַת עַל מַדְבְּחָא, יָת עֲלָתָא וְיָת תַּרְבַּיָּא; וַחֲזָא כָּל עַמָּא וְשַׁבַּחוּ, וּנְפַלוּ עַל אַפֵּיהוֹן. 
י,א וַיִּקְח֣וּ בְנֵֽי־אַ֠הֲרֹן נָדָ֨ב וַֽאֲבִיה֜וּא אִ֣ישׁ מַחְתָּת֗וֹ וַיִּתְּנ֤וּ בָהֵן֙ אֵ֔שׁ וַיָּשִׂ֥ימוּ עָלֶ֖יהָ קְטֹ֑רֶת וַיַּקְרִ֜יבוּ לִפְנֵ֤י יְהוָה֙ אֵ֣שׁ זָרָ֔ה אֲשֶׁ֧ר לֹ֦א צִוָּ֖ה אֹתָֽם׃ וּנְסִיבוּ בְּנֵי אַהֲרוֹן נָדָב וַאֲבִיהוּא גְּבַר מַחְתִּיתֵיהּ, וִיהַבוּ בְּהוֹן אִישָׁתָא, וְשַׁוִּיאוּ עֲלַהּ, קְטֹרֶת בֻּסְמִין; וְקָרִיבוּ קֳדָם יְיָ, אִישָׁתָא נֻכְרֵיתָא--דְּלָא פַּקֵּיד, יָתְהוֹן. 
י,ב וַתֵּ֥צֵא אֵ֛שׁ מִלִּפְנֵ֥י יְהוָ֖ה וַתֹּ֣אכַל אוֹתָ֑ם וַיָּמֻ֖תוּ לִפְנֵ֥י יְהוָֽה׃ וּנְפַקַת אִישָׁתָא מִן קֳדָם יְיָ, וַאֲכַלַת יָתְהוֹן; וּמִיתוּ, קֳדָם יְיָ. 
י,ג וַיֹּ֨אמֶר מֹשֶׁ֜ה אֶֽל־אַהֲרֹ֗ן הוּא֩ אֲשֶׁר־דִּבֶּ֨ר יְהוָ֤ה ׀ לֵאמֹר֙ בִּקְרֹבַ֣י אֶקָּדֵ֔שׁ וְעַל־פְּנֵ֥י כָל־הָעָ֖ם אֶכָּבֵ֑ד וַיִּדֹּ֖ם אַֽהֲרֹֽן׃ וַאֲמַר מֹשֶׁה לְאַהֲרוֹן, הוּא דְּמַלֵּיל יְיָ לְמֵימַר בְּקָרִיבַי אֶתְקַדַּשׁ, וְעַל אַפֵּי כָּל עַמָּא, אֶתְיַקַּר; וּשְׁתֵיק, אַהֲרוֹן. 
י,ד וַיִּקְרָ֣א מֹשֶׁ֗ה אֶל־מִֽישָׁאֵל֙ וְאֶ֣ל אֶלְצָפָ֔ן בְּנֵ֥י עֻזִּיאֵ֖ל דֹּ֣ד אַֽהֲרֹ֑ן וַיֹּ֣אמֶר אֲלֵהֶ֗ם קִ֠רְב֞וּ שְׂא֤וּ אֶת־אֲחֵיכֶם֙ מֵאֵ֣ת פְּנֵֽי־הַקֹּ֔דֶשׁ אֶל־מִח֖וּץ לַֽמַּחֲנֶֽה׃ וּקְרָא מֹשֶׁה, לְמִישָׁאֵל וּלְאֶלְצָפָן, בְּנֵי עֻזִּיאֵל, אַחְבּוּהִי דְּאַהֲרוֹן; וַאֲמַר לְהוֹן, קְרוּבוּ טוּלוּ יָת אֲחֵיכוֹן מִן קֳדָם אַפֵּי קֻדְשָׁא, לְמִבַּרָא, לְמַשְׁרִיתָא. 
י,ה וַֽיִּקְרְב֗וּ וַיִּשָּׂאֻם֙ בְּכֻתֳּנֹתָ֔ם אֶל־מִח֖וּץ לַֽמַּחֲנֶ֑ה כַּֽאֲשֶׁ֖ר דִּבֶּ֥ר מֹשֶֽׁה׃ וּקְרִיבוּ, וּנְטַלוּנִין בְּכִתּוּנֵיהוֹן, לְמִבַּרָא, לְמַשְׁרִיתָא--כְּמָא דְּמַלֵּיל מֹשֶׁה. 
י,ו וַיֹּ֣אמֶר מֹשֶׁ֣ה אֶֽל־אַהֲרֹ֡ן וּלְאֶלְעָזָר֩ וּלְאִֽיתָמָ֨ר ׀ בָּנָ֜יו רָֽאשֵׁיכֶ֥ם אַל־תִּפְרָ֣עוּ ׀ וּבִגְדֵיכֶ֤ם לֹֽא־תִפְרֹ֨מוּ֙ וְלֹ֣א תָמֻ֔תוּ וְעַ֥ל כָּל־הָֽעֵדָ֖ה יִקְצֹ֑ף וַֽאֲחֵיכֶם֙ כָּל־בֵּ֣ית יִשְׂרָאֵ֔ל יִבְכּוּ֙ אֶת־הַשְּׂרֵפָ֔ה אֲשֶׁ֖ר שָׂרַ֥ף יְהוָֽה׃ וַאֲמַר מֹשֶׁה לְאַהֲרוֹן וּלְאֶלְעָזָר וּלְאִיתָמָר בְּנוֹהִי רֵישֵׁיכוֹן לָא תְּרַבּוֹן פֵּירוּעַ וּלְבוּשֵׁיכוֹן לָא תְּבַזְּעוּן, וְלָא תְּמוּתוּן, וְעַל כָּל כְּנִשְׁתָּא, יְהֵי רֻגְזָא; וַאֲחֵיכוֹן, כָּל בֵּית יִשְׂרָאֵל--יִבְכּוֹן יָת יְקֵידְתָא, דְּאוֹקֵיד יְיָ. 
י,ז וּמִפֶּתַח֩ אֹ֨הֶל מוֹעֵ֜ד לֹ֤א תֵֽצְאוּ֙ פֶּן־תָּמֻ֔תוּ כִּי־שֶׁ֛מֶן מִשְׁחַ֥ת יְהוָ֖ה עֲלֵיכֶ֑ם וַֽיַּעֲשׂ֖וּ כִּדְבַ֥ר מֹשֶֽׁה׃ {פ} וּמִתְּרַע מַשְׁכַּן זִמְנָא לָא תִּפְּקוּן, דִּלְמָא תְּמוּתוּן--אֲרֵי מְשַׁח רְבוּתָא דַּייָ, עֲלֵיכוֹן; וַעֲבַדוּ, כְּפִתְגָמָא דְּמֹשֶׁה. {פ}
י,ח וַיְדַבֵּ֣ר יְהוָ֔ה אֶֽל־אַהֲרֹ֖ן לֵאמֹֽר׃ וּמַלֵּיל יְיָ, עִם אַהֲרוֹן לְמֵימַר. 
י,ט יַ֣יִן וְשֵׁכָ֞ר אַל־תֵּ֣שְׁתְּ ׀ אַתָּ֣ה ׀ וּבָנֶ֣יךָ אִתָּ֗ךְ בְּבֹֽאֲכֶ֛ם אֶל־אֹ֥הֶל מוֹעֵ֖ד וְלֹ֣א תָמֻ֑תוּ חֻקַּ֥ת עוֹלָ֖ם לְדֹרֹֽתֵיכֶֽם׃ חֲמַר וּמְרַוֵּי לָא תִּשְׁתֵּי אַתְּ וּבְנָךְ עִמָּךְ, בְּמֵיעַלְכוֹן לְמַשְׁכַּן זִמְנָא--וְלָא תְּמוּתוּן:  קְיָם עָלַם, לְדָרֵיכוֹן. 
י,י וּֽלְהַבְדִּ֔יל בֵּ֥ין הַקֹּ֖דֶשׁ וּבֵ֣ין הַחֹ֑ל וּבֵ֥ין הַטָּמֵ֖א וּבֵ֥ין הַטָּהֽוֹר׃ וּלְאַפְרָשָׁא, בֵּין קֻדְשָׁא וּבֵין חֻלָּא, וּבֵין מְסָאֲבָא, וּבֵין דָּכְיָא. 
י,יא וּלְהוֹרֹ֖ת אֶת־בְּנֵ֣י יִשְׂרָאֵ֑ל אֵ֚ת כָּל־הַ֣חֻקִּ֔ים אֲשֶׁ֨ר דִּבֶּ֧ר יְהוָ֛ה אֲלֵיהֶ֖ם בְּיַד־מֹשֶֽׁה׃ {פ} וּלְאַלָּפָא, יָת בְּנֵי יִשְׂרָאֵל--יָת, כָּל קְיָמַיָּא, דְּמַלֵּיל יְיָ לְהוֹן, בִּידָא דְּמֹשֶׁה. {פ}
י,יב וַיְדַבֵּ֨ר מֹשֶׁ֜ה אֶֽל־אַהֲרֹ֗ן וְאֶ֣ל אֶ֠לְעָזָר וְאֶל־אִ֨יתָמָ֥ר ׀ בָּנָיו֮ הַנּֽוֹתָרִים֒ קְח֣וּ אֶת־הַמִּנְחָ֗ה הַנּוֹתֶ֨רֶת֙ מֵֽאִשֵּׁ֣י יְהוָ֔ה וְאִכְל֥וּהָ מַצּ֖וֹת אֵ֣צֶל הַמִּזְבֵּ֑חַ כִּ֛י קֹ֥דֶשׁ קָֽדָשִׁ֖ים הִֽוא׃ וּמַלֵּיל מֹשֶׁה עִם אַהֲרוֹן, וְעִם אֶלְעָזָר וְעִם אִיתָמָר בְּנוֹהִי דְּאִשְׁתְּאַרוּ, סַבוּ יָת מִנְחָתָא דְּאִשְׁתְּאַרַת מִקֻּרְבָּנַיָּא דַּייָ, וְאִכְלוּהָא פַּטִּיר בִּסְטַר מַדְבְּחָא:  אֲרֵי קֹדֶשׁ קֻדְשִׁין, הִיא. 
י,יג וַֽאֲכַלְתֶּ֤ם אֹתָהּ֙ בְּמָק֣וֹם קָד֔וֹשׁ כִּ֣י חָקְךָ֤ וְחָק־בָּנֶ֨יךָ֙ הִ֔וא מֵֽאִשֵּׁ֖י יְהוָ֑ה כִּי־כֵ֖ן צֻוֵּֽיתִי׃ וְתֵיכְלוּן יָתַהּ, בַּאֲתַר קַדִּישׁ, אֲרֵי חוּלָקָךְ וְחוּלָק בְּנָךְ הִיא, מִקֻּרְבָּנַיָּא דַּייָ:  אֲרֵי כֵּין, אִתְפַּקַּדִית. 
י,יד וְאֵת֩ חֲזֵ֨ה הַתְּנוּפָ֜ה וְאֵ֣ת ׀ שׁ֣וֹק הַתְּרוּמָ֗ה תֹּֽאכְלוּ֙ בְּמָק֣וֹם טָה֔וֹר אַתָּ֕ה וּבָנֶ֥יךָ וּבְנֹתֶ֖יךָ אִתָּ֑ךְ כִּֽי־חָקְךָ֤ וְחָק־בָּנֶ֨יךָ֙ נִתְּנ֔וּ מִזִּבְחֵ֥י שַׁלְמֵ֖י בְּנֵ֥י יִשְׂרָאֵֽל׃ וְיָת חַדְיָא דַּאֲרָמוּתָא וְיָת שָׁקָא דְּאַפְרָשׁוּתָא, תֵּיכְלוּן בַּאֲתַר דְּכֵי--אַתְּ, וּבְנָךְ וּבְנָתָךְ עִמָּךְ:  אֲרֵי חוּלָקָךְ וְחוּלָק בְּנָךְ אִתְיְהִיבוּ, מִנִּכְסַת קֻדְשַׁיָּא דִּבְנֵי יִשְׂרָאֵל. 
י,טו שׁ֣וֹק הַתְּרוּמָ֞ה וַֽחֲזֵ֣ה הַתְּנוּפָ֗ה עַ֣ל אִשֵּׁ֤י הַֽחֲלָבִים֙ יָבִ֔יאוּ לְהָנִ֥יף תְּנוּפָ֖ה לִפְנֵ֣י יְהוָ֑ה וְהָיָ֨ה לְךָ֜ וּלְבָנֶ֤יךָ אִתְּךָ֙ לְחָק־עוֹלָ֔ם כַּֽאֲשֶׁ֖ר צִוָּ֥ה יְהוָֽה׃ שָׁקָא דְּאַפְרָשׁוּתָא וְחַדְיָא דַּאֲרָמוּתָא, עַל קֻרְבָּנֵי תַּרְבַּיָּא יִתֵּיתוֹן, לְאָרָמָא אֲרָמָא, קֳדָם יְיָ; וִיהֵי לָךְ וְלִבְנָךְ עִמָּךְ, לִקְיָם עָלַם--כְּמָא דְּפַקֵּיד יְיָ. 
י,טז וְאֵ֣ת ׀ שְׂעִ֣יר הַֽחַטָּ֗את דָּרֹ֥שׁ דָּרַ֛שׁ מֹשֶׁ֖ה וְהִנֵּ֣ה שֹׂרָ֑ף וַ֠יִּקְצֹף עַל־אֶלְעָזָ֤ר וְעַל־אִֽיתָמָר֙ בְּנֵ֣י אַֽהֲרֹ֔ן הַנּֽוֹתָרִ֖ם לֵאמֹֽר׃ וְיָת צְפִירָא דְּחַטָּתָא, מִתְבָּע תַּבְעֵיהּ מֹשֶׁה--וְהָא אִתּוֹקַד; וּרְגֵיז עַל אֶלְעָזָר וְעַל אִיתָמָר, בְּנֵי אַהֲרוֹן, דְּאִשְׁתְּאַרוּ, לְמֵימַר. 
י,יז מַדּ֗וּעַ לֹֽא־אֲכַלְתֶּ֤ם אֶת־הַֽחַטָּאת֙ בִּמְק֣וֹם הַקֹּ֔דֶשׁ כִּ֛י קֹ֥דֶשׁ קָֽדָשִׁ֖ים הִ֑וא וְאֹתָ֣הּ ׀ נָתַ֣ן לָכֶ֗ם לָשֵׂאת֙ אֶת־עֲו֣‍ֹן הָֽעֵדָ֔ה לְכַפֵּ֥ר עֲלֵיהֶ֖ם לִפְנֵ֥י יְהוָֽה׃ מָדֵין, לָא אֲכַלְתּוּן יָת חַטָּתָא בַּאֲתַר קַדִּישׁ--אֲרֵי קֹדֶשׁ קֻדְשִׁין, הִיא; וְיָתַהּ יְהַב לְכוֹן, לְסַלָּחָא עַל חוֹבֵי כְּנִשְׁתָּא, לְכַפָּרָא עֲלֵיהוֹן, קֳדָם יְיָ. 
י,יח הֵ֚ן לֹֽא־הוּבָ֣א אֶת־דָּמָ֔הּ אֶל־הַקֹּ֖דֶשׁ פְּנִ֑ימָה אָכ֨וֹל תֹּֽאכְל֥וּ אֹתָ֛הּ בַּקֹּ֖דֶשׁ כַּֽאֲשֶׁ֥ר צִוֵּֽיתִי׃ הָא לָא אִתָּעַל מִדְּמַהּ, לְבֵית קֻדְשָׁא גַּוָּאָה; מֵיכָל תֵּיכְלוּן יָתַהּ בְּקֻדְשָׁא, כְּמָא דְּפַקֵּידִית. 
י,יט וַיְדַבֵּ֨ר אַֽהֲרֹ֜ן אֶל־מֹשֶׁ֗ה הֵ֣ן הַ֠יּוֹם הִקְרִ֨יבוּ אֶת־חַטָּאתָ֤ם וְאֶת־עֹֽלָתָם֙ לִפְנֵ֣י יְהוָ֔ה וַתִּקְרֶ֥אנָה אֹתִ֖י כָּאֵ֑לֶּה וְאָכַ֤לְתִּי חַטָּאת֙ הַיּ֔וֹם הַיִּיטַ֖ב בְּעֵינֵ֥י יְהוָֽה׃ וּמַלֵּיל אַהֲרוֹן עִם מֹשֶׁה, הָא יוֹמָא דֵּין קָרִיבוּ יָת חַטָּוָתְהוֹן וְיָת עֲלָוָתְהוֹן קֳדָם יְיָ, וְעָרַעָא יָתִי, עָקָן כְּאִלֵּין; אִלּוּ פּוֹן אֲכַלִית חַטָּתָא יוֹמָא דֵּין, הֲתָקֵין קֳדָם יְיָ. 
י,כ וַיִּשְׁמַ֣ע מֹשֶׁ֔ה וַיִּיטַ֖ב בְּעֵינָֽיו׃ {פ} וּשְׁמַע מֹשֶׁה, וּשְׁפַר בְּעֵינוֹהִי. {פ}
יא,א וַיְדַבֵּ֧ר יְהוָ֛ה אֶל־מֹשֶׁ֥ה וְאֶֽל־אַהֲרֹ֖ן לֵאמֹ֥ר אֲלֵהֶֽם׃ וּמַלֵּיל יְיָ עִם מֹשֶׁה וּלְאַהֲרוֹן, לְמֵימַר לְהוֹן. 
יא,ב דַּבְּר֛וּ אֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר זֹ֤את הַֽחַיָּה֙ אֲשֶׁ֣ר תֹּֽאכְל֔וּ מִכָּל־הַבְּהֵמָ֖ה אֲשֶׁ֥ר עַל־הָאָֽרֶץ׃ מַלִּילוּ עִם בְּנֵי יִשְׂרָאֵל, לְמֵימַר:  דָּא חַיְתָא דְּתֵיכְלוּן, מִכָּל בְּעִירָא דְּעַל אַרְעָא. 
יא,ג כֹּ֣ל ׀ מַפְרֶ֣סֶת פַּרְסָ֗ה וְשֹׁסַ֤עַת שֶׁ֨סַע֙ פְּרָסֹ֔ת מַֽעֲלַ֥ת גֵּרָ֖ה בַּבְּהֵמָ֑ה אֹתָ֖הּ תֹּאכֵֽלוּ׃ כֹּל דִּסְדִיקָא פַּרְסְתַהּ, וּמַטְלְפָן טִלְפִין פַּרְסָתַהּ, מַסְּקָא פִּשְׁרָא, בִּבְעִירָא--יָתַהּ, תֵּיכְלוּן. 
יא,ד אַ֤ךְ אֶת־זֶה֙ לֹ֣א תֹֽאכְל֔וּ מִֽמַּעֲלֵי֙ הַגֵּרָ֔ה וּמִמַּפְרִסֵ֖י הַפַּרְסָ֑ה אֶֽת־הַ֠גָּמָל כִּֽי־מַעֲלֵ֨ה גֵרָ֜ה ה֗וּא וּפַרְסָה֙ אֵינֶ֣נּוּ מַפְרִ֔יס טָמֵ֥א ה֖וּא לָכֶֽם׃ בְּרַם יָת דֵּין, לָא תֵּיכְלוּן, מִמַּסְּקֵי פִּשְׁרָא, וּמִסְּדִיקֵי פַּרְסְתָא:  יָת גַּמְלָא אֲרֵי מַסֵּיק פִּשְׁרָא הוּא, וּפַרְסְתֵיהּ לָא סְדִיקָא--מְסָאַב הוּא, לְכוֹן. 
יא,ה וְאֶת־הַשָּׁפָ֗ן כִּֽי־מַעֲלֵ֤ה גֵרָה֙ ה֔וּא וּפַרְסָ֖ה לֹ֣א יַפְרִ֑יס טָמֵ֥א ה֖וּא לָכֶֽם׃ וְיָת טַבְזָא, אֲרֵי מַסֵּיק פִּשְׁרָא הוּא, וּפַרְסְתֵיהּ, לָא סְדִיקָא; מְסָאַב הוּא, לְכוֹן. 
יא,ו וְאֶת־הָֽאַרְנֶ֗בֶת כִּֽי־מַעֲלַ֤ת גֵּרָה֙ הִ֔וא וּפַרְסָ֖ה לֹ֣א הִפְרִ֑יסָה טְמֵאָ֥ה הִ֖וא לָכֶֽם׃ וְיָת אַרְנְבָא, אֲרֵי מַסְּקָא פִּשְׁרָא הִיא, וּפַרְסְתַהּ, לָא סְדִיקָא; מְסָאֲבָא הִיא, לְכוֹן. 
יא,ז וְאֶת־הַֽ֠חֲזִיר כִּֽי־מַפְרִ֨יס פַּרְסָ֜ה ה֗וּא וְשֹׁסַ֥ע שֶׁ֨סַע֙ פַּרְסָ֔ה וְה֖וּא גֵּרָ֣ה לֹֽא־יִגָּ֑ר טָמֵ֥א ה֖וּא לָכֶֽם׃ וְיָת חֲזִירָא אֲרֵי סְדִיק פַּרְסְתָא הוּא, וּמַטְלְפָן טִלְפִין פַּרְסָתֵיהּ, וְהוּא, פִּשְׁרָא לָא פָּשַׁר; מְסָאַב הוּא, לְכוֹן. 
יא,ח מִבְּשָׂרָם֙ לֹ֣א תֹאכֵ֔לוּ וּבְנִבְלָתָ֖ם לֹ֣א תִגָּ֑עוּ טְמֵאִ֥ים הֵ֖ם לָכֶֽם׃ מִבִּסְרְהוֹן לָא תֵּיכְלוּן, וּבִנְבִילַתְהוֹן לָא תִּקְרְבוּן; מְסָאֲבִין אִנּוּן, לְכוֹן. 
יא,ט אֶת־זֶה֙ תֹּֽאכְל֔וּ מִכֹּ֖ל אֲשֶׁ֣ר בַּמָּ֑יִם כֹּ֣ל אֲשֶׁר־לוֹ֩ סְנַפִּ֨יר וְקַשְׂקֶ֜שֶׂת בַּמַּ֗יִם בַּיַּמִּ֛ים וּבַנְּחָלִ֖ים אֹתָ֥ם תֹּאכֵֽלוּ׃ יָת דֵּין, תֵּיכְלוּן, מִכֹּל, דִּבְמַיָּא:  כֹּל דְּלֵיהּ צִיצִין וְקַלְפִין בְּמַיָּא, בְּיַמְמַיָּא וּבְנַחְלַיָּא--יָתְהוֹן תֵּיכְלוּן. 
יא,י וְכֹל֩ אֲשֶׁ֨ר אֵֽין־ל֜וֹ סְנַפִּ֣יר וְקַשְׂקֶ֗שֶׂת בַּיַּמִּים֙ וּבַנְּחָלִ֔ים מִכֹּל֙ שֶׁ֣רֶץ הַמַּ֔יִם וּמִכֹּ֛ל נֶ֥פֶשׁ הַֽחַיָּ֖ה אֲשֶׁ֣ר בַּמָּ֑יִם שֶׁ֥קֶץ הֵ֖ם לָכֶֽם׃ וְכֹל דְּלֵית לֵיהּ צִיצִין וְקַלְפִין, בְּיַמְמַיָּא וּבְנַחְלַיָּא, מִכֹּל רִחְשָׁא דְּמַיָּא, וּמִכֹּל נַפְשָׁא חַיְתָא דִּבְמַיָּא--שִׁקְצָא אִנּוּן, לְכוֹן. 
יא,יא וְשֶׁ֖קֶץ יִֽהְי֣וּ לָכֶ֑ם מִבְּשָׂרָם֙ לֹ֣א תֹאכֵ֔לוּ וְאֶת־נִבְלָתָ֖ם תְּשַׁקֵּֽצוּ׃ וְשִׁקְצָא, יְהוֹן לְכוֹן; מִבִּסְרְהוֹן לָא תֵּיכְלוּן, וְיָת נְבִילַתְהוֹן תְּשַׁקְּצוּן. 
יא,יב כֹּ֣ל אֲשֶׁ֥ר אֵֽין־ל֛וֹ סְנַפִּ֥יר וְקַשְׂקֶ֖שֶׂת בַּמָּ֑יִם שֶׁ֥קֶץ ה֖וּא לָכֶֽם׃ כֹּל דְּלֵית לֵיהּ צִיצִין וְקַלְפִין, בְּמַיָּא--שִׁקְצָא הוּא, לְכוֹן. 
יא,יג וְאֶת־אֵ֨לֶּה֙ תְּשַׁקְּצ֣וּ מִן־הָע֔וֹף לֹ֥א יֵאָֽכְל֖וּ שֶׁ֣קֶץ הֵ֑ם אֶת־הַנֶּ֨שֶׁר֙ וְאֶת־הַפֶּ֔רֶס וְאֵ֖ת הָֽעָזְנִיָּֽה׃ וְיָת אִלֵּין תְּשַׁקְּצוּן מִן עוֹפָא, לָא יִתְאַכְלוּן שִׁקְצָא אִנּוּן:  נִשְׁרָא, וְעָר--וְעָזְיָא. 
יא,יד וְאֶ֨ת־הַדָּאָ֔ה וְאֶת־הָֽאַיָּ֖ה לְמִינָֽהּ׃ וְדַיְתָא--וְטָרָפִיתָא, לִזְנַהּ. 
יא,טו אֵ֥ת כָּל־עֹרֵ֖ב לְמִינֽוֹ׃ יָת כָּל עוֹרְבָא, לִזְנֵיהּ. 
יא,טז וְאֵת֙ בַּ֣ת הַֽיַּעֲנָ֔ה וְאֶת־הַתַּחְמָ֖ס וְאֶת־הַשָּׁ֑חַף וְאֶת־הַנֵּ֖ץ לְמִינֵֽהוּ׃ וְיָת בַּת נַעָמִיתָא, וְצִיצָא וְצִפַּר שַׁחְפָּא; וְנַצָּא, לִזְנוֹהִי. 
יא,יז וְאֶת־הַכּ֥וֹס וְאֶת־הַשָּׁלָ֖ךְ וְאֶת־הַיַּנְשֽׁוּף׃ וְקָדְיָא וְשָׁלֵינוּנָא, וְקִפּוּפָא. 
יא,יח וְאֶת־הַתִּנְשֶׁ֥מֶת וְאֶת־הַקָּאָ֖ת וְאֶת־הָֽרָחָֽם׃ וּבָוְתָא וְקָתָא, וִירַקְרֵיקָא. 
יא,יט וְאֵת֙ הַֽחֲסִידָ֔ה הָֽאֲנָפָ֖ה לְמִינָ֑הּ וְאֶת־הַדּֽוּכִיפַ֖ת וְאֶת־הָֽעֲטַלֵּֽף׃ וְחָוָרִיתָא--וְאִבּוֹ, לִזְנַהּ; וְנַגָּר טוּרָא, וַעֲטַלֵּיפָא. 
יא,כ כֹּ֚ל שֶׁ֣רֶץ הָע֔וֹף הַֽהֹלֵ֖ךְ עַל־אַרְבַּ֑ע שֶׁ֥קֶץ ה֖וּא לָכֶֽם׃ כֹּל רִחְשָׁא דְּעוֹפָא, דִּמְהַלֵּיךְ עַל אַרְבַּע--שִׁקְצָא הוּא, לְכוֹן. 
יא,כא אַ֤ךְ אֶת־זֶה֙ תֹּֽאכְל֔וּ מִכֹּל֙ שֶׁ֣רֶץ הָע֔וֹף הַֽהֹלֵ֖ךְ עַל־אַרְבַּ֑ע אֲשֶׁר־לא (ל֤וֹ) כְרָעַ֨יִם֙ מִמַּ֣עַל לְרַגְלָ֔יו לְנַתֵּ֥ר בָּהֵ֖ן עַל־הָאָֽרֶץ׃ בְּרַם יָת דֵּין, תֵּיכְלוּן, מִכֹּל רִחְשָׁא דְּעוֹפָא, דִּמְהַלֵּיךְ עַל אַרְבַּע:  דְּלֵיהּ קַרְסֻלִּין מֵעִלָּוֵי רַגְלוֹהִי, לְקַפָּצָא בְּהוֹן עַל אַרְעָא. 
יא,כב אֶת־אֵ֤לֶּה מֵהֶם֙ תֹּאכֵ֔לוּ אֶת־הָֽאַרְבֶּ֣ה לְמִינ֔וֹ וְאֶת־הַסָּלְעָ֖ם לְמִינֵ֑הוּ וְאֶת־הַֽחַרְגֹּ֣ל לְמִינֵ֔הוּ וְאֶת־הֶֽחָגָ֖ב לְמִינֵֽהוּ׃ יָת אִלֵּין מִנְּהוֹן, תֵּיכְלוּן--יָת גּוֹבָא לִזְנֵיהּ, וְיָת רָשׁוֹנָא לִזְנוֹהִי; וְיָת חַרְגְּלָא לִזְנוֹהִי, וְיָת חַגְבָּא לִזְנוֹהִי. 
יא,כג וְכֹל֙ שֶׁ֣רֶץ הָע֔וֹף אֲשֶׁר־ל֖וֹ אַרְבַּ֣ע רַגְלָ֑יִם שֶׁ֥קֶץ ה֖וּא לָכֶֽם׃ וְכֹל רִחְשָׁא דְּעוֹפָא, דְּלֵיהּ אַרְבַּע רַגְלִין--שִׁקְצָא הוּא, לְכוֹן. 
יא,כד וּלְאֵ֖לֶּה תִּטַּמָּ֑אוּ כָּל־הַנֹּגֵ֥עַ בְּנִבְלָתָ֖ם יִטְמָ֥א עַד־הָעָֽרֶב׃ וּלְאִלֵּין, תִּסְתָּאֲבוּן; כָּל דְּיִקְרַב בִּנְבִילַתְהוֹן, יְהֵי מְסָאַב עַד רַמְשָׁא. 
יא,כה וְכָל־הַנֹּשֵׂ֖א מִנִּבְלָתָ֑ם יְכַבֵּ֥ס בְּגָדָ֖יו וְטָמֵ֥א עַד־הָעָֽרֶב׃ וְכָל דְּיִטּוֹל, מִנְּבִילַתְהוֹן--יְצַבַּע לְבוּשׁוֹהִי, וִיהֵי מְסָאַב עַד רַמְשָׁא. 
יא,כו לְֽכָל־הַבְּהֵמָ֡ה אֲשֶׁ֣ר הִוא֩ מַפְרֶ֨סֶת פַּרְסָ֜ה וְשֶׁ֣סַע ׀ אֵינֶ֣נָּה שֹׁסַ֗עַת וְגֵרָה֙ אֵינֶ֣נָּה מַֽעֲלָ֔ה טְמֵאִ֥ים הֵ֖ם לָכֶ֑ם כָּל־הַנֹּגֵ֥עַ בָּהֶ֖ם יִטְמָֽא׃ לְכָל בְּעִירָא דְּהִיא סְדִיקָא פַּרְסְתַהּ וְטִלְפִין לָיְתַהָא מַטְלְפָא, וּפִשְׁרָא לָיְתַהָא מַסְּקָא--מְסָאֲבִין אִנּוּן, לְכוֹן; כָּל דְּיִקְרַב בְּהוֹן, יְהֵי מְסָאַב. 
יא,כז וְכֹ֣ל ׀ הוֹלֵ֣ךְ עַל־כַּפָּ֗יו בְּכָל־הַֽחַיָּה֙ הַֽהֹלֶ֣כֶת עַל־אַרְבַּ֔ע טְמֵאִ֥ים הֵ֖ם לָכֶ֑ם כָּל־הַנֹּגֵ֥עַ בְּנִבְלָתָ֖ם יִטְמָ֥א עַד־הָעָֽרֶב׃ וְכֹל דִּמְהַלֵּיךְ עַל יְדוֹהִי, בְּכָל חַיְתָא דִּמְהַלְּכָא עַל אַרְבַּע--מְסָאֲבִין אִנּוּן, לְכוֹן; כָּל דְּיִקְרַב בִּנְבִילַתְהוֹן, יְהֵי מְסָאַב עַד רַמְשָׁא. 
יא,כח וְהַנֹּשֵׂא֙ אֶת־נִבְלָתָ֔ם יְכַבֵּ֥ס בְּגָדָ֖יו וְטָמֵ֣א עַד־הָעָ֑רֶב טְמֵאִ֥ים הֵ֖מָּה לָכֶֽם׃ {ס} וּדְיִטּוֹל, יָת נְבִילַתְהוֹן--יְצַבַּע לְבוּשׁוֹהִי, וִיהֵי מְסָאַב עַד רַמְשָׁא; מְסָאֲבִין אִנּוּן, לְכוֹן. {ס}
יא,כט וְזֶ֤ה לָכֶם֙ הַטָּמֵ֔א בַּשֶּׁ֖רֶץ הַשֹּׁרֵ֣ץ עַל־הָאָ֑רֶץ הַחֹ֥לֶד וְהָֽעַכְבָּ֖ר וְהַצָּ֥ב לְמִינֵֽהוּ׃ וְדֵין לְכוֹן דִּמְסָאַב, בְּרִחְשָׁא דְּרָחֵישׁ עַל אַרְעָא:  חֻלְדָּא וְעַכְבְּרָא, וְצַבָּא לִזְנוֹהִי. 
יא,ל וְהָֽאֲנָקָ֥ה וְהַכֹּ֖חַ וְהַלְּטָאָ֑ה וְהַחֹ֖מֶט וְהַתִּנְשָֽׁמֶת׃ וְיַלָּא וְכוֹחָא, וְהַלְטְתָא; וְחֻמְטָא, וְאָשׁוּתָא. 
יא,לא אֵ֛לֶּה הַטְּמֵאִ֥ים לָכֶ֖ם בְּכָל־הַשָּׁ֑רֶץ כָּל־הַנֹּגֵ֧עַ בָּהֶ֛ם בְּמֹתָ֖ם יִטְמָ֥א עַד־הָעָֽרֶב׃ אִלֵּין דִּמְסָאֲבִין לְכוֹן, בְּכָל רִחְשָׁא; כָּל דְּיִקְרַב בְּהוֹן בְּמוֹתְהוֹן, יְהֵי מְסָאַב עַד רַמְשָׁא. 
יא,לב וְכֹ֣ל אֲשֶׁר־יִפֹּל־עָלָיו֩ מֵהֶ֨ם ׀ בְּמֹתָ֜ם יִטְמָ֗א מִכָּל־כְּלִי־עֵץ֙ א֣וֹ בֶ֤גֶד אוֹ־עוֹר֙ א֣וֹ שָׂ֔ק כָּל־כְּלִ֕י אֲשֶׁר־יֵֽעָשֶׂ֥ה מְלָאכָ֖ה בָּהֶ֑ם בַּמַּ֧יִם יוּבָ֛א וְטָמֵ֥א עַד־הָעֶ֖רֶב וְטָהֵֽר׃ וְכֹל דְּיִפּוֹל עֲלוֹהִי מִנְּהוֹן בְּמוֹתְהוֹן יְהֵי מְסָאַב, מִכָּל מָן דְּאָע אוֹ לְבוּשׁ אוֹ מְשַׁךְ אוֹ שָׂק, כָּל מָן, דְּיִתְעֲבֵיד עֲבִידָא בְּהוֹן; בְּמַיָּא יִתָּעַל וִיהֵי מְסָאַב עַד רַמְשָׁא, וְיִדְכֵּי. 
יא,לג וְכָ֨ל־כְּלִי־חֶ֔רֶשׂ אֲשֶׁר־יִפֹּ֥ל מֵהֶ֖ם אֶל־תּוֹכ֑וֹ כֹּ֣ל אֲשֶׁ֧ר בְּתוֹכ֛וֹ יִטְמָ֖א וְאֹת֥וֹ תִשְׁבֹּֽרוּ׃ וְכָל מָן דַּחֲסַף--דְּיִפּוֹל מִנְּהוֹן, לְגַוֵּיהּ:  כֹּל דִּבְגַוֵּיהּ יִסְתָּאַב, וְיָתֵיהּ תִּתְבְּרוּן. 
יא,לד מִכָּל־הָאֹ֜כֶל אֲשֶׁ֣ר יֵֽאָכֵ֗ל אֲשֶׁ֨ר יָב֥וֹא עָלָ֛יו מַ֖יִם יִטְמָ֑א וְכָל־מַשְׁקֶה֙ אֲשֶׁ֣ר יִשָּׁתֶ֔ה בְּכָל־כְּלִ֖י יִטְמָֽא׃ מִכָּל מֵיכָל דְּמִתְאֲכִיל, דְּיֵיעֲלוּן עֲלוֹהִי מַיָּא--יְהֵי מְסָאַב; וְכָל מַשְׁקְיָא דְּיִשְׁתְּתֵי, בְּכָל מָן יְהֵי מְסָאַב. 
יא,לה וְ֠כֹל אֲשֶׁר־יִפֹּ֨ל מִנִּבְלָתָ֥ם ׀ עָלָיו֮ יִטְמָא֒ תַּנּ֧וּר וְכִירַ֛יִם יֻתָּ֖ץ טְמֵאִ֣ים הֵ֑ם וּטְמֵאִ֖ים יִֽהְי֥וּ לָכֶֽם׃ וְכֹל דְּיִפּוֹל מִנְּבִילַתְהוֹן עֲלוֹהִי, יְהֵי מְסָאַב--תַּנּוּר וְכִירַיִם יִתָּרְעוּן, מְסָאֲבִין אִנּוּן; וּמְסָאֲבִין, יְהוֹן לְכוֹן. 
יא,לו אַ֣ךְ מַעְיָ֥ן וּב֛וֹר מִקְוֵה־מַ֖יִם יִֽהְיֶ֣ה טָה֑וֹר וְנֹגֵ֥עַ בְּנִבְלָתָ֖ם יִטְמָֽא׃ בְּרַם מַעְיָן וְגוּב בֵּית כְּנֵישַׁת מַיָּא, יְהֵי דְּכֵי; וּדְיִקְרַב בִּנְבִילַתְהוֹן, יְהֵי מְסָאַב. 
יא,לז וְכִ֤י יִפֹּל֙ מִנִּבְלָתָ֔ם עַל־כָּל־זֶ֥רַע זֵר֖וּעַ אֲשֶׁ֣ר יִזָּרֵ֑עַ טָה֖וֹר הֽוּא׃ וַאֲרֵי יִפּוֹל מִנְּבִילַתְהוֹן, עַל כָּל בַּר זְרַע זֵירוּעַ דְּיִזְדְּרַע--דְּכֵי, הוּא. 
יא,לח וְכִ֤י יֻתַּן־מַ֨יִם֙ עַל־זֶ֔רַע וְנָפַ֥ל מִנִּבְלָתָ֖ם עָלָ֑יו טָמֵ֥א ה֖וּא לָכֶֽם׃ {ס} וַאֲרֵי יִתְיַהְבּוּן מַיָּא עַל בַּר זַרְעָא, וְיִפּוֹל מִנְּבִילַתְהוֹן עֲלוֹהִי--מְסָאַב הוּא, לְכוֹן. {ס}
יא,לט וְכִ֤י יָמוּת֙ מִן־הַבְּהֵמָ֔ה אֲשֶׁר־הִ֥יא לָכֶ֖ם לְאָכְלָ֑ה הַנֹּגֵ֥עַ בְּנִבְלָתָ֖הּ יִטְמָ֥א עַד־הָעָֽרֶב׃ וַאֲרֵי יְמוּת מִן בְּעִירָא, דְּהִיא לְכוֹן לְמֵיכַל--דְּיִקְרַב בִּנְבִילְתַהּ, יְהֵי מְסָאַב עַד רַמְשָׁא. 
יא,מ וְהָֽאֹכֵל֙ מִנִּבְלָתָ֔הּ יְכַבֵּ֥ס בְּגָדָ֖יו וְטָמֵ֣א עַד־הָעָ֑רֶב וְהַנֹּשֵׂא֙ אֶת־נִבְלָתָ֔הּ יְכַבֵּ֥ס בְּגָדָ֖יו וְטָמֵ֥א עַד־הָעָֽרֶב׃ וּדְיֵיכוֹל, מִנְּבִילְתַהּ--יְצַבַּע לְבוּשׁוֹהִי, וִיהֵי מְסָאַב עַד רַמְשָׁא; וּדְיִטּוֹל, יָת נְבִילְתַהּ--יְצַבַּע לְבוּשׁוֹהִי, וִיהֵי מְסָאַב עַד רַמְשָׁא. 
יא,מא וְכָל־הַשֶּׁ֖רֶץ הַשֹּׁרֵ֣ץ עַל־הָאָ֑רֶץ שֶׁ֥קֶץ ה֖וּא לֹ֥א יֵֽאָכֵֽל׃ וְכָל רִחְשָׁא, דְּרָחֵישׁ עַל אַרְעָא--שִׁקְצָא הוּא, לָא יִתְאֲכִיל. 
יא,מב כֹּל֩ הוֹלֵ֨ךְ עַל־גָּח֜וֹן וְכֹ֣ל ׀ הוֹלֵ֣ךְ עַל־אַרְבַּ֗ע עַ֚ד כָּל־מַרְבֵּ֣ה רַגְלַ֔יִם לְכָל־הַשֶּׁ֖רֶץ הַשֹּׁרֵ֣ץ עַל־הָאָ֑רֶץ לֹ֥א תֹֽאכְל֖וּם כִּי־שֶׁ֥קֶץ הֵֽם׃ כֹּל דִּמְהַלֵּיךְ עַל מְעוֹהִי וְכֹל דִּמְהַלֵּיךְ עַל אַרְבַּע, עַד כָּל סַגְיוּת רַגְלִין, לְכָל רִחְשָׁא, דְּרָחֵישׁ עַל אַרְעָא--לָא תֵּיכְלוּנוּנוּן, אֲרֵי שִׁקְצָא אִנּוּן. 
יא,מג אַל־תְּשַׁקְּצוּ֙ אֶת־נַפְשֹׁ֣תֵיכֶ֔ם בְּכָל־הַשֶּׁ֖רֶץ הַשֹּׁרֵ֑ץ וְלֹ֤א תִֽטַּמְּאוּ֙ בָּהֶ֔ם וְנִטְמֵתֶ֖ם בָּֽם׃ לָא תְּשַׁקְּצוּן, יָת נַפְשָׁתְכוֹן, בְּכָל רִחְשָׁא, דְּרָחֵישׁ; וְלָא תִּסְתָּאֲבוּן בְּהוֹן, וְתִסְתָּאֲבוּן פּוֹן בְּהוֹן. 
יא,מד כִּ֣י אֲנִ֣י יְהוָה֮ אֱלֹֽהֵיכֶם֒ וְהִתְקַדִּשְׁתֶּם֙ וִֽהְיִיתֶ֣ם קְדֹשִׁ֔ים כִּ֥י קָד֖וֹשׁ אָ֑נִי וְלֹ֤א תְטַמְּאוּ֙ אֶת־נַפְשֹׁ֣תֵיכֶ֔ם בְּכָל־הַשֶּׁ֖רֶץ הָֽרֹמֵ֥שׂ עַל־הָאָֽרֶץ׃ אֲרֵי אֲנָא יְיָ, אֱלָהֲכוֹן, וְתִתְקַדְּשׁוּן וּתְהוֹן קַדִּישִׁין, אֲרֵי קַדִּישׁ אֲנָא; וְלָא תְּסָאֲבוּן יָת נַפְשָׁתְכוֹן, בְּכָל רִחְשָׁא דְּרָחֵישׁ עַל אַרְעָא. 
יא,מה כִּ֣י ׀ אֲנִ֣י יְהוָ֗ה הַֽמַּעֲלֶ֤ה אֶתְכֶם֙ מֵאֶ֣רֶץ מִצְרַ֔יִם לִֽהְיֹ֥ת לָכֶ֖ם לֵֽאלֹהִ֑ים וִֽהְיִיתֶ֣ם קְדֹשִׁ֔ים כִּ֥י קָד֖וֹשׁ אָֽנִי׃ אֲרֵי אֲנָא יְיָ, דְּאַסֵּיק יָתְכוֹן מֵאַרְעָא דְּמִצְרַיִם, לְמִהְוֵי לְכוֹן, לֶאֱלָהּ; וּתְהוֹן קַדִּישִׁין, אֲרֵי קַדִּישׁ אֲנָא. 
יא,מו זֹ֣את תּוֹרַ֤ת הַבְּהֵמָה֙ וְהָע֔וֹף וְכֹל֙ נֶ֣פֶשׁ הַֽחַיָּ֔ה הָֽרֹמֶ֖שֶׂת בַּמָּ֑יִם וּלְכָל־נֶ֖פֶשׁ הַשֹּׁרֶ֥צֶת עַל־הָאָֽרֶץ׃ דָּא אוֹרָיְתָא דִּבְעִירָא, וּדְעוֹפָא, וּלְכֹל נַפְשָׁא חַיְתָא, דְּרָחֲשָׁא בְּמַיָּא; וּלְכָל נַפְשָׁא, דְּרָחֲשָׁא עַל אַרְעָא. 
יא,מז לְהַבְדִּ֕יל בֵּ֥ין הַטָּמֵ֖א וּבֵ֣ין הַטָּהֹ֑ר וּבֵ֤ין הַֽחַיָּה֙ הַֽנֶּאֱכֶ֔לֶת וּבֵין֙ הַֽחַיָּ֔ה אֲשֶׁ֖ר לֹ֥א תֵֽאָכֵֽל׃ {פ} לְאַפְרָשָׁא, בֵּין מְסָאֲבָא וּבֵין דָּכְיָא; וּבֵין חַיְתָא, דְּמִתְאַכְלָא, וּבֵין חַיְתָא, דְּלָא מִתְאַכְלָא. {פ}
