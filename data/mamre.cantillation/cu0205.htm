יתרו

יח,א וַיִּשְׁמַ֞ע יִתְר֨וֹ כֹהֵ֤ן מִדְיָן֙ חֹתֵ֣ן מֹשֶׁ֔ה אֵת֩ כָּל־אֲשֶׁ֨ר עָשָׂ֤ה אֱלֹהִים֙ לְמֹשֶׁ֔ה וּלְיִשְׂרָאֵ֖ל עַמּ֑וֹ כִּֽי־הוֹצִ֧יא יְהוָ֛ה אֶת־יִשְׂרָאֵ֖ל מִמִּצְרָֽיִם׃ וּשְׁמַע יִתְרוֹ רַבָּא דְּמִדְיָן, חֲמוּהִי דְּמֹשֶׁה, יָת כָּל דַּעֲבַד יְיָ לְמֹשֶׁה, וּלְיִשְׂרָאֵל עַמֵּיהּ:  אֲרֵי אַפֵּיק יְיָ יָת יִשְׂרָאֵל, מִמִּצְרָיִם. 
יח,ב וַיִּקַּ֗ח יִתְרוֹ֙ חֹתֵ֣ן מֹשֶׁ֔ה אֶת־צִפֹּרָ֖ה אֵ֣שֶׁת מֹשֶׁ֑ה אַחַ֖ר שִׁלּוּחֶֽיהָ׃ וּדְבַר, יִתְרוֹ חֲמוּהִי דְּמֹשֶׁה, יָת צִפּוֹרָה, אִתַּת מֹשֶׁה--בָּתַר, דְּשַׁלְּחַהּ. 
יח,ג וְאֵ֖ת שְׁנֵ֣י בָנֶ֑יהָ אֲשֶׁ֨ר שֵׁ֤ם הָֽאֶחָד֙ גֵּֽרְשֹׁ֔ם כִּ֣י אָמַ֔ר גֵּ֣ר הָיִ֔יתִי בְּאֶ֖רֶץ נָכְרִיָּֽה׃ וְיָת, תְּרֵין בְּנַהָא:  דְּשׁוֹם חַד, גֵּרְשׁוֹם--אֲרֵי אֲמַר, דַּיָּר הֲוֵיתִי בַּאֲרַע נֻכְרָאָה. 
יח,ד וְשֵׁ֥ם הָֽאֶחָ֖ד אֱלִיעֶ֑זֶר כִּֽי־אֱלֹהֵ֤י אָבִי֙ בְּעֶזְרִ֔י וַיַּצִּלֵ֖נִי מֵחֶ֥רֶב פַּרְעֹֽה׃ וְשׁוֹם חַד, אֱלִיעֶזֶר--אֲרֵי אֱלָהֵיהּ דְּאַבָּא הֲוָה בְּסַעֲדִי, וְשֵׁיזְבַנִי מֵחַרְבָּא דְּפַרְעֹה. 
יח,ה וַיָּבֹ֞א יִתְר֨וֹ חֹתֵ֥ן מֹשֶׁ֛ה וּבָנָ֥יו וְאִשְׁתּ֖וֹ אֶל־מֹשֶׁ֑ה אֶל־הַמִּדְבָּ֗ר אֲשֶׁר־ה֛וּא חֹנֶ֥ה שָׁ֖ם הַ֥ר הָֽאֱלֹהִֽים׃ וַאֲתָא יִתְרוֹ חֲמוּהִי דְּמֹשֶׁה, וּבְנוֹהִי וְאִתְּתֵיהּ--לְוָת מֹשֶׁה:  לְמַדְבְּרָא, דְּהוּא שָׁרֵי תַּמָּן--לְטוּרָא דְּאִתְגְּלִי עֲלוֹהִי יְקָרָא דַּייָ. 
יח,ו וַיֹּ֨אמֶר֙ אֶל־מֹשֶׁ֔ה אֲנִ֛י חֹֽתֶנְךָ֥ יִתְר֖וֹ בָּ֣א אֵלֶ֑יךָ וְאִ֨שְׁתְּךָ֔ וּשְׁנֵ֥י בָנֶ֖יהָ עִמָּֽהּ׃ וַאֲמַר, לְמֹשֶׁה, אֲנָא חֲמוּךְ יִתְרוֹ, אָתֵי לְוָתָךְ; וְאִתְּתָךְ--וּתְרֵין בְּנַהָא, עִמַּהּ. 
יח,ז וַיֵּצֵ֨א מֹשֶׁ֜ה לִקְרַ֣את חֹֽתְנ֗וֹ וַיִּשְׁתַּ֨חוּ֙ וַיִּשַּׁק־ל֔וֹ וַיִּשְׁאֲל֥וּ אִישׁ־לְרֵעֵ֖הוּ לְשָׁל֑וֹם וַיָּבֹ֖אוּ הָאֹֽהֱלָה׃ וּנְפַק מֹשֶׁה לְקַדָּמוּת חֲמוּהִי, וּסְגֵיד וְנַשֵּׁיק לֵיהּ, וּשְׁאִילוּ גְּבַר לְחַבְרֵיהּ, לִשְׁלָם; וְעָאלוּ, לְמַשְׁכְּנָא. 
יח,ח וַיְסַפֵּ֤ר מֹשֶׁה֙ לְחֹ֣תְנ֔וֹ אֵת֩ כָּל־אֲשֶׁ֨ר עָשָׂ֤ה יְהוָה֙ לְפַרְעֹ֣ה וּלְמִצְרַ֔יִם עַ֖ל אוֹדֹ֣ת יִשְׂרָאֵ֑ל אֵ֤ת כָּל־הַתְּלָאָה֙ אֲשֶׁ֣ר מְצָאָ֣תַם בַּדֶּ֔רֶךְ וַיַּצִּלֵ֖ם יְהוָֽה׃ וְאִשְׁתַּעִי מֹשֶׁה, לַחֲמוּהִי, יָת כָּל דַּעֲבַד יְיָ לְפַרְעֹה וּלְמִצְרָאֵי, עַל עֵיסַק יִשְׂרָאֵל:  יָת כָּל עָקְתָא דְּאַשְׁכַּחַתְנוּן בְּאוֹרְחָא, וְשֵׁיזֵיבִנּוּן יְיָ. 
יח,ט וַיִּ֣חַדְּ יִתְר֔וֹ עַ֚ל כָּל־הַטּוֹבָ֔ה אֲשֶׁר־עָשָׂ֥ה יְהוָ֖ה לְיִשְׂרָאֵ֑ל אֲשֶׁ֥ר הִצִּיל֖וֹ מִיַּ֥ד מִצְרָֽיִם׃ וַחְדִּי יִתְרוֹ--עַל כָּל טָבְתָא, דַּעֲבַד יְיָ לְיִשְׂרָאֵל:  דְּשֵׁיזֵיבִנּוּן, מִיְּדָא דְּמִצְרָאֵי. 
יח,י וַיֹּאמֶר֮ יִתְרוֹ֒ בָּר֣וּךְ יְהוָ֔ה אֲשֶׁ֨ר הִצִּ֥יל אֶתְכֶ֛ם מִיַּ֥ד מִצְרַ֖יִם וּמִיַּ֣ד פַּרְעֹ֑ה אֲשֶׁ֤ר הִצִּיל֙ אֶת־הָעָ֔ם מִתַּ֖חַת יַד־מִצְרָֽיִם׃ וַאֲמַר, יִתְרוֹ, בְּרִיךְ יְיָ, דְּשֵׁיזֵיב יָתְכוֹן מִיְּדָא דְּמִצְרָאֵי וּמִיְּדָא דְּפַרְעֹה:  דְּשֵׁיזֵיב יָת עַמָּא, מִתְּחוֹת מַרְוַת מִצְרָאֵי. 
יח,יא עַתָּ֣ה יָדַ֔עְתִּי כִּֽי־גָד֥וֹל יְהוָ֖ה מִכָּל־הָֽאֱלֹהִ֑ים כִּ֣י בַדָּבָ֔ר אֲשֶׁ֥ר זָד֖וּ עֲלֵיהֶֽם׃ כְּעַן יָדַעְנָא, אֲרֵי רָב יְיָ וְלֵית אֱלָהּ בָּר מִנֵּיהּ:  אֲרֵי בְּפִתְגָמָא, דְּחַשִּׁיבוּ מִצְרָאֵי לִמְדָּן יָת יִשְׂרָאֵל בֵּיהּ דָּנִנּוּן. 
יח,יב וַיִּקַּ֞ח יִתְר֨וֹ חֹתֵ֥ן מֹשֶׁ֛ה עֹלָ֥ה וּזְבָחִ֖ים לֵֽאלֹהִ֑ים וַיָּבֹ֨א אַֽהֲרֹ֜ן וְכֹ֣ל ׀ זִקְנֵ֣י יִשְׂרָאֵ֗ל לֶֽאֱכָל־לֶ֛חֶם עִם־חֹתֵ֥ן מֹשֶׁ֖ה לִפְנֵ֥י הָֽאֱלֹהִֽים׃ וְקָרֵיב יִתְרוֹ חֲמוּהִי דְּמֹשֶׁה, עֲלָוָן וְנִכְסַת קֻדְשִׁין--קֳדָם יְיָ; וַאֲתָא אַהֲרוֹן וְכֹל סָבֵי יִשְׂרָאֵל, לְמֵיכַל לַחְמָא עִם חֲמוּהִי דְּמֹשֶׁה--קֳדָם יְיָ. 
יח,יג וַֽיְהִי֙ מִֽמָּחֳרָ֔ת וַיֵּ֥שֶׁב מֹשֶׁ֖ה לִשְׁפֹּ֣ט אֶת־הָעָ֑ם וַיַּֽעֲמֹ֤ד הָעָם֙ עַל־מֹשֶׁ֔ה מִן־הַבֹּ֖קֶר עַד־הָעָֽרֶב׃ וַהֲוָה, בְּיוֹמָא דְּבָתְרוֹהִי, וִיתֵיב מֹשֶׁה, לִמְדָּן יָת עַמָּא; וְקָם עַמָּא עִלָּווֹהִי דְּמֹשֶׁה, מִן צַפְרָא עַד רַמְשָׁא. 
יח,יד וַיַּרְא֙ חֹתֵ֣ן מֹשֶׁ֔ה אֵ֛ת כָּל־אֲשֶׁר־ה֥וּא עֹשֶׂ֖ה לָעָ֑ם וַיֹּ֗אמֶר מָֽה־הַדָּבָ֤ר הַזֶּה֙ אֲשֶׁ֨ר אַתָּ֤ה עֹשֶׂה֙ לָעָ֔ם מַדּ֗וּעַ אַתָּ֤ה יוֹשֵׁב֙ לְבַדֶּ֔ךָ וְכָל־הָעָ֛ם נִצָּ֥ב עָלֶ֖יךָ מִן־בֹּ֥קֶר עַד־עָֽרֶב׃ וַחֲזָא חֲמוּהִי דְּמֹשֶׁה, יָת כָּל דְּהוּא עָבֵיד לְעַמָּא; וַאֲמַר, מָא פִּתְגָמָא הָדֵין דְּאַתְּ עָבֵיד לְעַמָּא--מָדֵין אַתְּ יָתֵיב בִּלְחוֹדָךְ, וְכָל עַמָּא קָיְמִין עִלָּוָךְ מִן צַפְרָא עַד רַמְשָׁא. 
יח,טו וַיֹּ֥אמֶר מֹשֶׁ֖ה לְחֹֽתְנ֑וֹ כִּֽי־יָבֹ֥א אֵלַ֛י הָעָ֖ם לִדְרֹ֥שׁ אֱלֹהִֽים׃ וַאֲמַר מֹשֶׁה, לַחֲמוּהִי:  אֲרֵי אָתַן לְוָתִי עַמָּא, לְמִתְבַּע אֻלְפָן מִן קֳדָם יְיָ. 
יח,טז כִּֽי־יִהְיֶ֨ה לָהֶ֤ם דָּבָר֙ בָּ֣א אֵלַ֔י וְשָׁ֣פַטְתִּ֔י בֵּ֥ין אִ֖ישׁ וּבֵ֣ין רֵעֵ֑הוּ וְהֽוֹדַעְתִּ֛י אֶת־חֻקֵּ֥י הָֽאֱלֹהִ֖ים וְאֶת־תּֽוֹרֹתָֽיו׃ כַּד הָוֵי לְהוֹן דִּינָא, אָתַן לְוָתִי, וְדָאֵינְנָא, בֵּין גֻּבְרָא וּבֵין חַבְרֵיהּ; וּמְהוֹדַעְנָא לְהוֹן יָת קְיָמַיָּא דַּייָ, וְיָת אוֹרָיָתֵיהּ. 
יח,יז וַיֹּ֛אמֶר חֹתֵ֥ן מֹשֶׁ֖ה אֵלָ֑יו לֹא־טוֹב֙ הַדָּבָ֔ר אֲשֶׁ֥ר אַתָּ֖ה עֹשֶֽׂה׃ וַאֲמַר חֲמוּהִי דְּמֹשֶׁה, לֵיהּ:  לָא תָּקֵין, פִּתְגָמָא, דְּאַתְּ, עָבֵיד. 
יח,יח נָבֹ֣ל תִּבֹּ֔ל גַּם־אַתָּ֕ה גַּם־הָעָ֥ם הַזֶּ֖ה אֲשֶׁ֣ר עִמָּ֑ךְ כִּֽי־כָבֵ֤ד מִמְּךָ֙ הַדָּבָ֔ר לֹֽא־תוּכַ֥ל עֲשֹׂ֖הוּ לְבַדֶּֽךָ׃ מִלְאָה תִּלְאֵי--אַף אַתְּ, אַף עַמָּא הָדֵין דְּעִמָּךְ:  אֲרֵי יַקִּיר מִנָּךְ פִּתְגָמָא, לָא תִּכּוֹל לְמִעְבְּדֵיהּ בִּלְחוֹדָךְ. 
יח,יט עַתָּ֞ה שְׁמַ֤ע בְּקֹלִי֙ אִיעָ֣צְךָ֔ וִיהִ֥י אֱלֹהִ֖ים עִמָּ֑ךְ הֱיֵ֧ה אַתָּ֣ה לָעָ֗ם מ֚וּל הָֽאֱלֹהִ֔ים וְהֵֽבֵאתָ֥ אַתָּ֛ה אֶת־הַדְּבָרִ֖ים אֶל־הָֽאֱלֹהִֽים׃ כְּעַן קַבֵּיל מִנִּי, אַמְלְכִנָּךְ, וִיהֵי מֵימְרָא דַּייָ, בְּסַעֲדָךְ; הֱוִי אַתְּ לְעַמָּא, תָּבַע אֻלְפָן מִן קֳדָם יְיָ, וּתְהֵי מֵיתֵי אַתְּ יָת פִּתְגָמַיָּא, לִקְדָם יְיָ. 
יח,כ וְהִזְהַרְתָּ֣ה אֶתְהֶ֔ם אֶת־הַֽחֻקִּ֖ים וְאֶת־הַתּוֹרֹ֑ת וְהֽוֹדַעְתָּ֣ לָהֶ֗ם אֶת־הַדֶּ֨רֶךְ֙ יֵ֣לְכוּ בָ֔הּ וְאֶת־הַֽמַּעֲשֶׂ֖ה אֲשֶׁ֥ר יַֽעֲשֽׂוּן׃ וְתַזְהַר יָתְהוֹן, יָת קְיָמַיָּא וְיָת אוֹרָיָתָא; וּתְהוֹדַע לְהוֹן, יָת אוֹרְחָא דִּיהָכוּן בַּהּ, וְיָת עוּבָדָא, דְּיַעְבְּדוּן. 
יח,כא וְאַתָּ֣ה תֶֽחֱזֶ֣ה מִכָּל־הָ֠עָם אַנְשֵׁי־חַ֜יִל יִרְאֵ֧י אֱלֹהִ֛ים אַנְשֵׁ֥י אֱמֶ֖ת שֹׂ֣נְאֵי בָ֑צַע וְשַׂמְתָּ֣ עֲלֵהֶ֗ם שָׂרֵ֤י אֲלָפִים֙ שָׂרֵ֣י מֵא֔וֹת שָׂרֵ֥י חֲמִשִּׁ֖ים וְשָׂרֵ֥י עֲשָׂרֹֽת׃ וְאַתְּ תִּחְזֵי מִכָּל עַמָּא גֻּבְרִין דְּחֵילָא דָּחֲלַיָּא דַּייָ, גֻּבְרִין דִּקְשׁוֹט--דְּסָנַן לְקַבָּלָא מָמוֹן; וּתְמַנֵּי עֲלֵיהוֹן, רַבָּנֵי אַלְפֵי רַבָּנֵי מָאוָתָא, רַבָּנֵי חַמְשִׁין, וְרַבָּנֵי עֲסוֹרְיָתָא. 
יח,כב וְשָֽׁפְט֣וּ אֶת־הָעָם֮ בְּכָל־עֵת֒ וְהָיָ֞ה כָּל־הַדָּבָ֤ר הַגָּדֹל֙ יָבִ֣יאוּ אֵלֶ֔יךָ וְכָל־הַדָּבָ֥ר הַקָּטֹ֖ן יִשְׁפְּטוּ־הֵ֑ם וְהָקֵל֙ מֵֽעָלֶ֔יךָ וְנָֽשְׂא֖וּ אִתָּֽךְ׃ וִידִינוּן יָת עַמָּא, בְּכָל עִדָּן, וִיהֵי כָּל פִּתְגָם רַב יַיְתוֹן לְוָתָךְ, וְכָל פִּתְגָם זְעֵיר יְדִינוּן אִנּוּן; וְיֵיקְלוּן, מִנָּךְ, וִיסוֹבְרוּן, עִמָּךְ. 
יח,כג אִ֣ם אֶת־הַדָּבָ֤ר הַזֶּה֙ תַּֽעֲשֶׂ֔ה וְצִוְּךָ֣ אֱלֹהִ֔ים וְיָֽכָלְתָּ֖ עֲמֹ֑ד וְגַם֙ כָּל־הָעָ֣ם הַזֶּ֔ה עַל־מְקֹמ֖וֹ יָבֹ֥א בְשָׁלֽוֹם׃ אִם יָת פִּתְגָמָא הָדֵין, תַּעֲבֵיד, וִיפַקְּדִנָּךְ יְיָ, וְתִכּוֹל לִמְקָם; וְאַף כָּל עַמָּא הָדֵין, עַל אַתְרֵיהּ יְהָךְ בִּשְׁלָם. 
יח,כד וַיִּשְׁמַ֥ע מֹשֶׁ֖ה לְק֣וֹל חֹֽתְנ֑וֹ וַיַּ֕עַשׂ כֹּ֖ל אֲשֶׁ֥ר אָמָֽר׃ וְקַבֵּיל מֹשֶׁה, לְמֵימַר חֲמוּהִי; וַעֲבַד, כֹּל דַּאֲמַר. 
יח,כה וַיִּבְחַ֨ר מֹשֶׁ֤ה אַנְשֵׁי־חַ֨יִל֙ מִכָּל־יִשְׂרָאֵ֔ל וַיִּתֵּ֥ן אֹתָ֛ם רָאשִׁ֖ים עַל־הָעָ֑ם שָׂרֵ֤י אֲלָפִים֙ שָׂרֵ֣י מֵא֔וֹת שָׂרֵ֥י חֲמִשִּׁ֖ים וְשָׂרֵ֥י עֲשָׂרֹֽת׃ וּבְחַר מֹשֶׁה גֻּבְרִין דְּחֵילָא מִכָּל יִשְׂרָאֵל, וּמַנִּי יָתְהוֹן רֵישִׁין עַל עַמָּא--רַבָּנֵי אַלְפֵי רַבָּנֵי מָאוָתָא, רַבָּנֵי חַמְשִׁין וְרַבָּנֵי עֲסוֹרְיָתָא. 
יח,כו וְשָֽׁפְט֥וּ אֶת־הָעָ֖ם בְּכָל־עֵ֑ת אֶת־הַדָּבָ֤ר הַקָּשֶׁה֙ יְבִיא֣וּן אֶל־מֹשֶׁ֔ה וְכָל־הַדָּבָ֥ר הַקָּטֹ֖ן יִשְׁפּוּט֥וּ הֵֽם׃ וְדָיְנִין יָת עַמָּא, בְּכָל עִדָּן:  יָת פִּתְגָם קְשֵׁי מֵיתַן לְוָת מֹשֶׁה, וְכָל פִּתְגָם זְעֵיר דָּיְנִין אִנּוּן. 
יח,כז וַיְשַׁלַּ֥ח מֹשֶׁ֖ה אֶת־חֹֽתְנ֑וֹ וַיֵּ֥לֶךְ ל֖וֹ אֶל־אַרְצֽוֹ׃ {פ} וְשַׁלַּח מֹשֶׁה, יָת חֲמוּהִי; וַאֲזַל לֵיהּ, לְאַרְעֵיהּ. {פ}
יט,א בַּחֹ֨דֶשׁ֙ הַשְּׁלִישִׁ֔י לְצֵ֥את בְּנֵֽי־יִשְׂרָאֵ֖ל מֵאֶ֣רֶץ מִצְרָ֑יִם בַּיּ֣וֹם הַזֶּ֔ה בָּ֖אוּ מִדְבַּ֥ר סִינָֽי׃ בְּיַרְחָא, תְּלִיתָאָה, לְמִפַּק בְּנֵי יִשְׂרָאֵל, מֵאַרְעָא דְּמִצְרָיִם--בְּיוֹמָא הָדֵין, אֲתוֹ לְמַדְבְּרָא דְּסִינָי. 
יט,ב וַיִּסְע֣וּ מֵֽרְפִידִ֗ים וַיָּבֹ֨אוּ֙ מִדְבַּ֣ר סִינַ֔י וַֽיַּחֲנ֖וּ בַּמִּדְבָּ֑ר וַיִּֽחַן־שָׁ֥ם יִשְׂרָאֵ֖ל נֶ֥גֶד הָהָֽר׃ וּנְטַלוּ מֵרְפִידִים, וַאֲתוֹ לְמַדְבְּרָא דְּסִינַי, וּשְׁרוֹ, בְּמַדְבְּרָא; וּשְׁרָא תַּמָּן יִשְׂרָאֵל, לָקֳבֵיל טוּרָא. 
יט,ג וּמֹשֶׁ֥ה עָלָ֖ה אֶל־הָֽאֱלֹהִ֑ים וַיִּקְרָ֨א אֵלָ֤יו יְהוָה֙ מִן־הָהָ֣ר לֵאמֹ֔ר כֹּ֤ה תֹאמַר֙ לְבֵ֣ית יַֽעֲקֹ֔ב וְתַגֵּ֖יד לִבְנֵ֥י יִשְׂרָאֵֽל׃ וּמֹשֶׁה סְלֵיק, לִקְדָם יְיָ; וּקְרָא לֵיהּ יְיָ, מִן טוּרָא לְמֵימַר, כְּדֵין תֵּימַר לְבֵית יַעֲקוֹב, וּתְחַוֵּי לִבְנֵי יִשְׂרָאֵל. 
יט,ד אַתֶּ֣ם רְאִיתֶ֔ם אֲשֶׁ֥ר עָשִׂ֖יתִי לְמִצְרָ֑יִם וָֽאֶשָּׂ֤א אֶתְכֶם֙ עַל־כַּנְפֵ֣י נְשָׁרִ֔ים וָֽאָבִ֥א אֶתְכֶ֖ם אֵלָֽי׃ אַתּוּן חֲזֵיתוֹן, דַּעֲבַדִית לְמִצְרָאֵי; וְנַטֵּילִית יָתְכוֹן כִּד עַל גַּדְפֵּי נִשְׁרִין, וְקָרֵיבִית יָתְכוֹן לְפֻלְחָנִי. 
יט,ה וְעַתָּ֗ה אִם־שָׁמ֤וֹעַ תִּשְׁמְעוּ֙ בְּקֹלִ֔י וּשְׁמַרְתֶּ֖ם אֶת־בְּרִיתִ֑י וִֽהְיִ֨יתֶם לִ֤י סְגֻלָּה֙ מִכָּל־הָ֣עַמִּ֔ים כִּי־לִ֖י כָּל־הָאָֽרֶץ׃ וּכְעַן, אִם קַבָּלָא תְּקַבְּלוּן לְמֵימְרִי, וְתִטְּרוּן, יָת קְיָמִי--וּתְהוֹן קֳדָמַי חַבִּיבִין מִכָּל עַמְמַיָּא, אֲרֵי דִּילִי כָּל אַרְעָא. 
יט,ו וְאַתֶּ֧ם תִּֽהְיוּ־לִ֛י מַמְלֶ֥כֶת כֹּֽהֲנִ֖ים וְג֣וֹי קָד֑וֹשׁ אֵ֚לֶּה הַדְּבָרִ֔ים אֲשֶׁ֥ר תְּדַבֵּ֖ר אֶל־בְּנֵ֥י יִשְׂרָאֵֽל׃ וְאַתּוּן תְּהוֹן קֳדָמַי מַלְכִין כָּהֲנִין, וְעַם קַדִּישׁ:  אִלֵּין, פִּתְגָמַיָּא, דִּתְמַלֵּיל, עִם בְּנֵי יִשְׂרָאֵל. 
יט,ז וַיָּבֹ֣א מֹשֶׁ֔ה וַיִּקְרָ֖א לְזִקְנֵ֣י הָעָ֑ם וַיָּ֣שֶׂם לִפְנֵיהֶ֗ם אֵ֚ת כָּל־הַדְּבָרִ֣ים הָאֵ֔לֶּה אֲשֶׁ֥ר צִוָּ֖הוּ יְהוָֽה׃ וַאֲתָא מֹשֶׁה, וּקְרָא לְסָבֵי עַמָּא; וְסַדַּר קֳדָמֵיהוֹן, יָת כָּל פִּתְגָמַיָּא הָאִלֵּין, דְּפַקְּדֵיהּ, יְיָ. 
יט,ח וַיַּֽעֲנ֨וּ כָל־הָעָ֤ם יַחְדָּו֙ וַיֹּ֣אמְר֔וּ כֹּ֛ל אֲשֶׁר־דִּבֶּ֥ר יְהוָ֖ה נַֽעֲשֶׂ֑ה וַיָּ֧שֶׁב מֹשֶׁ֛ה אֶת־דִּבְרֵ֥י הָעָ֖ם אֶל־יְהוָֽה׃ וַאֲתִיבוּ כָּל עַמָּא כַּחְדָּא וַאֲמַרוּ, כֹּל דְּמַלֵּיל יְיָ נַעֲבֵיד; וַאֲתֵיב מֹשֶׁה יָת פִּתְגָמֵי עַמָּא, לִקְדָם יְיָ. 
יט,ט וַיֹּ֨אמֶר יְהוָ֜ה אֶל־מֹשֶׁ֗ה הִנֵּ֨ה אָֽנֹכִ֜י בָּ֣א אֵלֶיךָ֮ בְּעַ֣ב הֶֽעָנָן֒ בַּֽעֲב֞וּר יִשְׁמַ֤ע הָעָם֙ בְּדַבְּרִ֣י עִמָּ֔ךְ וְגַם־בְּךָ֖ יַֽאֲמִ֣ינוּ לְעוֹלָ֑ם וַיַּגֵּ֥ד מֹשֶׁ֛ה אֶת־דִּבְרֵ֥י הָעָ֖ם אֶל־יְהוָֽה׃ וַאֲמַר יְיָ לְמֹשֶׁה, הָא אֲנָא מִתְגְּלֵי לָךְ בְּעֵיבָא דַּעֲנָנָא, בְּדִיל דְּיִשְׁמַע עַמָּא בְּמַלָּלוּתִי עִמָּךְ, וְאַף בָּךְ יְהֵימְנוּן לְעָלַם; וְחַוִּי מֹשֶׁה יָת פִּתְגָמֵי עַמָּא, לִקְדָם יְיָ. 
יט,י וַיֹּ֨אמֶר יְהוָ֤ה אֶל־מֹשֶׁה֙ לֵ֣ךְ אֶל־הָעָ֔ם וְקִדַּשְׁתָּ֥ם הַיּ֖וֹם וּמָחָ֑ר וְכִבְּס֖וּ שִׂמְלֹתָֽם׃ וַאֲמַר יְיָ לְמֹשֶׁה אִיזֵיל לְוָת עַמָּא, וּתְזָמֵינִנּוּן יוֹמָא דֵּין וּמְחַר; וִיחַוְּרוּן, לְבוּשֵׁיהוֹן. 
יט,יא וְהָי֥וּ נְכֹנִ֖ים לַיּ֣וֹם הַשְּׁלִישִׁ֑י כִּ֣י ׀ בַּיּ֣וֹם הַשְּׁלִשִׁ֗י יֵרֵ֧ד יְהוָ֛ה לְעֵינֵ֥י כָל־הָעָ֖ם עַל־הַ֥ר סִינָֽי׃ וִיהוֹן זְמִינִין, לְיוֹמָא תְּלִיתָאָה:  אֲרֵי בְּיוֹמָא תְּלִיתָאָה, יִתְגְּלֵי יְיָ לְעֵינֵי כָּל עַמָּא--עַל טוּרָא דְּסִינָי. 
יט,יב וְהִגְבַּלְתָּ֤ אֶת־הָעָם֙ סָבִ֣יב לֵאמֹ֔ר הִשָּֽׁמְר֥וּ לָכֶ֛ם עֲל֥וֹת בָּהָ֖ר וּנְגֹ֣עַ בְּקָצֵ֑הוּ כָּל־הַנֹּגֵ֥עַ בָּהָ֖ר מ֥וֹת יוּמָֽת׃ וּתְתַּחֵים יָת עַמָּא סְחוֹר סְחוֹר לְמֵימַר, אִסְתְּמַרוּ לְכוֹן מִלְּמִסַּק בְּטוּרָא וּלְמִקְרַב בְּסוֹפֵיהּ:  כָּל דְּיִקְרַב בְּטוּרָא, אִתְקְטָלָא יִתְקְטִיל. 
יט,יג לֹֽא־תִגַּ֨ע בּ֜וֹ יָ֗ד כִּֽי־סָק֤וֹל יִסָּקֵל֙ אֽוֹ־יָרֹ֣ה יִיָּרֶ֔ה אִם־בְּהֵמָ֥ה אִם־אִ֖ישׁ לֹ֣א יִֽחְיֶ֑ה בִּמְשֹׁךְ֙ הַיֹּבֵ֔ל הֵ֖מָּה יַֽעֲל֥וּ בָהָֽר׃ לָא תִּקְרַב בֵּיהּ יַד, אֲרֵי אִתְרְגָמָא יִתְרְגֵים אוֹ אִשְׁתְּדָאָה יִשְׁתְּדֵי--אִם בְּעִירָא אִם אֲנָשָׁא, לָא יִתְקַיַּם; בְּמֵיגַד, שׁוֹפָרָא, אִנּוּן, מֻרְשַׁן לְמִסַּק בְּטוּרָא. 
יט,יד וַיֵּ֧רֶד מֹשֶׁ֛ה מִן־הָהָ֖ר אֶל־הָעָ֑ם וַיְקַדֵּשׁ֙ אֶת־הָעָ֔ם וַֽיְכַבְּס֖וּ שִׂמְלֹתָֽם׃ וּנְחַת מֹשֶׁה מִן טוּרָא, לְוָת עַמָּא; וְזָמֵין, יָת עַמָּא, וְחַוַּרוּ, לְבוּשֵׁיהוֹן. 
יט,טו וַיֹּ֨אמֶר֙ אֶל־הָעָ֔ם הֱי֥וּ נְכֹנִ֖ים לִשְׁלֹ֣שֶׁת יָמִ֑ים אַֽל־תִּגְּשׁ֖וּ אֶל־אִשָּֽׁה׃ וַאֲמַר, לְעַמָּא, הֲווֹ זְמִינִין, לִתְלָתָא יוֹמִין:  לָא תִּקְרְבוּן, לְצַד אִתְּתָא. 
יט,טז וַיְהִי֩ בַיּ֨וֹם הַשְּׁלִישִׁ֜י בִּֽהְיֹ֣ת הַבֹּ֗קֶר וַיְהִי֩ קֹלֹ֨ת וּבְרָקִ֜ים וְעָנָ֤ן כָּבֵד֙ עַל־הָהָ֔ר וְקֹ֥ל שֹׁפָ֖ר חָזָ֣ק מְאֹ֑ד וַיֶּֽחֱרַ֥ד כָּל־הָעָ֖ם אֲשֶׁ֥ר בַּֽמַּחֲנֶֽה׃ וַהֲוָה בְּיוֹמָא תְּלִיתָאָה בְּמִהְוֵי צַפְרָא, וַהֲווֹ קָלִין וּבַרְקִין וַעֲנָנָא תַּקִּיף עַל טוּרָא, וְקָל שׁוֹפָרָא, תַּקִּיף לַחְדָּא; וְזָע כָּל עַמָּא, דִּבְמַשְׁרִיתָא. 
יט,יז וַיּוֹצֵ֨א מֹשֶׁ֧ה אֶת־הָעָ֛ם לִקְרַ֥את הָֽאֱלֹהִ֖ים מִן־הַֽמַּחֲנֶ֑ה וַיִּֽתְיַצְּב֖וּ בְּתַחְתִּ֥ית הָהָֽר׃ וְאַפֵּיק מֹשֶׁה יָת עַמָּא לְקַדָּמוּת מֵימְרָא דַּייָ, מִן מַשְׁרִיתָא; וְאִתְעַתַּדוּ, בְּשִׁפּוֹלֵי טוּרָא. 
יט,יח וְהַ֤ר סִינַי֙ עָשַׁ֣ן כֻּלּ֔וֹ מִ֠פְּנֵי אֲשֶׁ֨ר יָרַ֥ד עָלָ֛יו יְהוָ֖ה בָּאֵ֑שׁ וַיַּ֤עַל עֲשָׁנוֹ֙ כְּעֶ֣שֶׁן הַכִּבְשָׁ֔ן וַיֶּֽחֱרַ֥ד כָּל־הָהָ֖ר מְאֹֽד׃ וְטוּרָא דְּסִינַי, תָּנַן כֻּלֵּיהּ, מִן קֳדָם דְּאִתְגְּלִי עֲלוֹהִי יְיָ, בְּאִישָׁתָא; וּסְלֵיק תַּנְנֵיהּ כְּתַנְנָא דְּאַתּוּנָא, וְזָע כָּל טוּרָא לַחְדָּא. 
יט,יט וַֽיְהִי֙ ק֣וֹל הַשֹּׁפָ֔ר הוֹלֵ֖ךְ וְחָזֵ֣ק מְאֹ֑ד מֹשֶׁ֣ה יְדַבֵּ֔ר וְהָֽאֱלֹהִ֖ים יַֽעֲנֶ֥נּֽוּ בְקֽוֹל׃ וַהֲוָה קָל שׁוֹפָרָא, אָזֵיל וְתָקֵיף לַחְדָּא; מֹשֶׁה מְמַלֵּיל, וּמִן קֳדָם יְיָ מִתְעֲנֵי לֵיהּ בְּקָל. 
יט,כ וַיֵּ֧רֶד יְהוָ֛ה עַל־הַ֥ר סִינַ֖י אֶל־רֹ֣אשׁ הָהָ֑ר וַיִּקְרָ֨א יְהוָ֧ה לְמֹשֶׁ֛ה אֶל־רֹ֥אשׁ הָהָ֖ר וַיַּ֥עַל מֹשֶֽׁה׃ וְאִתְגְּלִי יְיָ עַל טוּרָא דְּסִינַי, לְרֵישׁ טוּרָא; וּקְרָא יְיָ לְמֹשֶׁה לְרֵישׁ טוּרָא, וּסְלֵיק מֹשֶׁה. 
יט,כא וַיֹּ֤אמֶר יְהוָה֙ אֶל־מֹשֶׁ֔ה רֵ֖ד הָעֵ֣ד בָּעָ֑ם פֶּן־יֶֽהֶרְס֤וּ אֶל־יְהוָה֙ לִרְא֔וֹת וְנָפַ֥ל מִמֶּ֖נּוּ רָֽב׃ וַאֲמַר יְיָ לְמֹשֶׁה, חוֹת אַסְהֵיד בְּעַמָּא:  דִּלְמָא יְפַגְּרוּן קֳדָם יְיָ לְמִחְזֵי, וְיִפּוֹל מִנְּהוֹן סַגִּי. 
יט,כב וְגַ֧ם הַכֹּֽהֲנִ֛ים הַנִּגָּשִׁ֥ים אֶל־יְהוָ֖ה יִתְקַדָּ֑שׁוּ פֶּן־יִפְרֹ֥ץ בָּהֶ֖ם יְהוָֽה׃ וְאַף כָּהֲנַיָּא דְּקָרִיבִין לְשַׁמָּשָׁא קֳדָם יְיָ, יִתְקַדְּשׁוּן:  דִּלְמָא יִקְטוֹל בְּהוֹן, יְיָ. 
יט,כג וַיֹּ֤אמֶר מֹשֶׁה֙ אֶל־יְהוָ֔ה לֹֽא־יוּכַ֣ל הָעָ֔ם לַֽעֲלֹ֖ת אֶל־הַ֣ר סִינָ֑י כִּֽי־אַתָּ֞ה הַֽעֵדֹ֤תָה בָּ֨נוּ֙ לֵאמֹ֔ר הַגְבֵּ֥ל אֶת־הָהָ֖ר וְקִדַּשְׁתּֽוֹ׃ וַאֲמַר מֹשֶׁה, קֳדָם יְיָ, לָא יִכּוֹל עַמָּא, לְמִסַּק לְטוּרָא דְּסִינָי:  אֲרֵי אַתְּ אַסְהֵידְתְּ בַּנָא, לְמֵימַר, תַּחֵים יָת טוּרָא, וְקַדֵּישְׁהִי. 
יט,כד וַיֹּ֨אמֶר אֵלָ֤יו יְהוָה֙ לֶךְ־רֵ֔ד וְעָלִ֥יתָ אַתָּ֖ה וְאַֽהֲרֹ֣ן עִמָּ֑ךְ וְהַכֹּֽהֲנִ֣ים וְהָעָ֗ם אַל־יֶֽהֶרְס֛וּ לַֽעֲלֹ֥ת אֶל־יְהוָ֖ה פֶּן־יִפְרָץ־בָּֽם׃ וַאֲמַר לֵיהּ יְיָ אִיזֵיל חוֹת, וְתִסַּק אַתְּ וְאַהֲרוֹן עִמָּךְ; וְכָהֲנַיָּא וְעַמָּא, לָא יְפַגְּרוּן לְמִסַּק לִקְדָם יְיָ--דִּלְמָא יִקְטוֹל בְּהוֹן. 
יט,כה וַיֵּ֥רֶד מֹשֶׁ֖ה אֶל־הָעָ֑ם וַיֹּ֖אמֶר אֲלֵהֶֽם׃ {ס} וּנְחַת מֹשֶׁה, לְוָת עַמָּא; וַאֲמַר, לְהוֹן. {ס}
כ,א וַיְדַבֵּ֣ר אֱלֹהִ֔ים אֵ֛ת כָּל־הַדְּבָרִ֥ים הָאֵ֖לֶּה לֵאמֹֽר׃ {ס} וּמַלֵּיל יְיָ, יָת כָּל פִּתְגָמַיָּא הָאִלֵּין לְמֵימַר. {ס}
כ,ב אָֽנֹכִי֙ יְהוָ֣ה אֱלֹהֶ֔יךָ אֲשֶׁ֧ר הֽוֹצֵאתִ֛יךָ מֵאֶ֥רֶץ מִצְרַ֖יִם מִבֵּ֣ית עֲבָדִ֑ים לֹֽא־יִהְיֶ֥ה לְךָ֛ אֱלֹהִ֥ים אֲחֵרִ֖ים עַל־פָּנָֽי׃ אֲנָא יְיָ אֱלָהָךְ, דְּאַפֵּיקְתָּךְ מֵאַרְעָא דְּמִצְרַיִם מִבֵּית עַבְדוּתָא:  לָא יִהְוֵי לָךְ אֱלָהּ אָחֳרָן, בָּר מִנִּי. 
כ,ג לֹֽא־תַעֲשֶׂ֨ה לְךָ֥ פֶ֨סֶל֙ וְכָל־תְּמוּנָ֔ה אֲשֶׁ֤ר בַּשָּׁמַ֨יִם֙ מִמַּ֔עַל וַֽאֲשֶׁ֥ר בָּאָ֖רֶץ מִתָּ֑חַת וַֽאֲשֶׁ֥ר בַּמַּ֖יִם מִתַּ֥חַת לָאָֽרֶץ׃ לָא תַּעֲבֵיד לָךְ צֵילַם, וְכָל דְּמוּ, דְּבִשְׁמַיָּא מִלְּעֵילָא, וְדִבְאַרְעָא מִלְּרַע--וְדִבְמַיָּא, מִלְּרַע לְאַרְעָא. 
כ,ד לֹֽא־תִשְׁתַּחֲוֶ֥ה לָהֶ֖ם וְלֹ֣א תָֽעָבְדֵ֑ם כִּ֣י אָֽנֹכִ֞י יְהוָ֤ה אֱלֹהֶ֨יךָ֙ אֵ֣ל קַנָּ֔א פֹּ֠קֵד עֲו֨‍ֹן אָבֹ֧ת עַל־בָּנִ֛ים עַל־שִׁלֵּשִׁ֥ים וְעַל־רִבֵּעִ֖ים לְשֹֽׂנְאָֽי׃ לָא תִּסְגּוֹד לְהוֹן, וְלָא תִּפְלְחִנִּין:  אֲרֵי אֲנָא יְיָ אֱלָהָךְ, אֵל קַנָּא--מַסְעַר חוֹבֵי אֲבָהָן עַל בְּנִין מָרָדִין עַל דָּר תְּלִיתַאי וְעַל דָּר רְבִיעַאי, לְסָנְאָי כַּד מַשְׁלְמִין בְּנַיָּא לְמִחְטֵי בָּתַר אֲבָהָתְהוֹן. 
כ,ה וְעֹ֥שֶׂה חֶ֖סֶד לַֽאֲלָפִ֑ים לְאֹֽהֲבַ֖י וּלְשֹֽׁמְרֵ֥י מִצְו‍ֹתָֽי׃ {ס} וְעָבֵיד טֵיבוּ, לְאַלְפֵי דָּרִין--לְרָחֲמַי, וּלְנָטְרֵי פִּקּוֹדָי. {ס}
כ,ו לֹ֥א תִשָּׂ֛א אֶת־שֵֽׁם־יְהוָ֥ה אֱלֹהֶ֖יךָ לַשָּׁ֑וְא כִּ֣י לֹ֤א יְנַקֶּה֙ יְהוָ֔ה אֵ֛ת אֲשֶׁר־יִשָּׂ֥א אֶת־שְׁמ֖וֹ לַשָּֽׁוְא׃ {פ} לָא תֵּימֵי בִּשְׁמָא דַּייָ אֱלָהָךְ, לְמַגָּנָא:  אֲרֵי לָא יְזַכֵּי יְיָ, יָת דְּיֵימֵי בִּשְׁמֵיהּ לְשִׁקְרָא. {פ}
כ,ז זָכ֛וֹר אֶת־י֥וֹם הַשַּׁבָּ֖ת לְקַדְּשֽׁוֹ׃ הֱוִי דְּכִיר יָת יוֹמָא דְּשַׁבְּתָא, לְקַדָּשׁוּתֵיהּ. 
כ,ח שֵׁ֤שֶׁת יָמִים֙ תַּֽעֲבֹ֔ד וְעָשִׂ֖יתָ כָּל־מְלַאכְתֶּֽךָ׃ שִׁתָּא יוֹמִין תִּפְלַח, וְתַעֲבֵיד כָּל עֲבִידְתָךְ. 
כ,ט וְיוֹם֙ הַשְּׁבִיעִ֔י שַׁבָּ֖ת לַֽיהוָ֣ה אֱלֹהֶ֑יךָ לֹֽא־תַעֲשֶׂ֨ה כָל־מְלָאכָ֜ה אַתָּ֣ה ׀ וּבִנְךָ֣ וּבִתֶּ֗ךָ עַבְדְּךָ֤ וַאֲמָֽתְךָ֙ וּבְהֶמְתֶּ֔ךָ וְגֵֽרְךָ֖ אֲשֶׁ֥ר בִּשְׁעָרֶֽיךָ׃ וְיוֹמָא, שְׁבִיעָאָה--שַׁבְּתָא, קֳדָם יְיָ אֱלָהָךְ:  לָא תַּעֲבֵיד כָּל עֲבִידָא אַתְּ וּבְרָךְ וּבְרַתָּךְ, עַבְדָּךְ וְאַמְתָּךְ וּבְעִירָךְ, וְגִיּוֹרָךְ, דִּבְקִרְוָךְ. 
כ,י כִּ֣י שֵֽׁשֶׁת־יָמִים֩ עָשָׂ֨ה יְהוָ֜ה אֶת־הַשָּׁמַ֣יִם וְאֶת־הָאָ֗רֶץ אֶת־הַיָּם֙ וְאֶת־כָּל־אֲשֶׁר־בָּ֔ם וַיָּ֖נַח בַּיּ֣וֹם הַשְּׁבִיעִ֑י עַל־כֵּ֗ן בֵּרַ֧ךְ יְהוָ֛ה אֶת־י֥וֹם הַשַּׁבָּ֖ת וַֽיְקַדְּשֵֽׁהוּ׃ {ס} אֲרֵי שִׁתָּא יוֹמִין עֲבַד יְיָ יָת שְׁמַיָּא וְיָת אַרְעָא, יָת יַמָּא וְיָת כָּל דִּבְהוֹן, וְנָח, בְּיוֹמָא שְׁבִיעָאָה; עַל כֵּין, בָּרֵיךְ יְיָ יָת יוֹמָא דְּשַׁבְּתָא--וְקַדְּשֵׁיהּ. {ס}
כ,יא כַּבֵּ֥ד אֶת־אָבִ֖יךָ וְאֶת־אִמֶּ֑ךָ לְמַ֨עַן֙ יַֽאֲרִכ֣וּן יָמֶ֔יךָ עַ֚ל הָֽאֲדָמָ֔ה אֲשֶׁר־יְהוָ֥ה אֱלֹהֶ֖יךָ נֹתֵ֥ן לָֽךְ׃ {ס} יַקַּר יָת אֲבוּךְ, וְיָת אִמָּךְ--בְּדִיל, דְּיֵירְכוּן יוֹמָךְ, עַל אַרְעָא, דַּייָ אֱלָהָךְ יָהֵיב לָךְ. {ס}
כ,יב לֹ֥א תִרְצָ֖ח {ס} לֹ֣א תִנְאָ֑ף {ס} לֹ֣א תִגְנֹ֔ב {ס} לֹֽא־תַעֲנֶ֥ה בְרֵֽעֲךָ֖ עֵ֥ד שָֽׁקֶר׃ {ס} לָא תִּקְטוּל נְפַשׁ,  {ס}  לָא תְּגוּף;  {ס}  לָא תִּגְנוּב,  {ס}  לָא תַּסְהֵיד בְּחַבְרָךְ סָהֲדוּתָא דְּשִׁקְרָא. {ס}
כ,יג לֹ֥א תַחְמֹ֖ד בֵּ֣ית רֵעֶ֑ךָ {ס} לֹֽא־תַחְמֹ֞ד אֵ֣שֶׁת רֵעֶ֗ךָ וְעַבְדּ֤וֹ וַֽאֲמָתוֹ֙ וְשׁוֹר֣וֹ וַֽחֲמֹר֔וֹ וְכֹ֖ל אֲשֶׁ֥ר לְרֵעֶֽךָ׃ {פ} לָא תַּחְמֵיד, בֵּית חַבְרָךְ;  {ס}  לָא תַּחְמֵיד אִתַּת חַבְרָךְ, וְעַבְדֵּיהּ וְאַמְתֵּיהּ וְתוֹרֵיהּ וּחְמָרֵיהּ, וְכֹל, דִּלְחַבְרָךְ. {פ}
כ,יד וְכָל־הָעָם֩ רֹאִ֨ים אֶת־הַקּוֹלֹ֜ת וְאֶת־הַלַּפִּידִ֗ם וְאֵת֙ ק֣וֹל הַשֹּׁפָ֔ר וְאֶת־הָהָ֖ר עָשֵׁ֑ן וַיַּ֤רְא הָעָם֙ וַיָּנֻ֔עוּ וַיַּֽעַמְד֖וּ מֵֽרָחֹֽק׃ וְכָל עַמָּא חָזַן יָת קָלַיָּא וְיָת בָּעוֹרַיָּא, וְיָת קָל שׁוֹפָרָא, וְיָת טוּרָא, דְּתָנַן; וַחֲזָא עַמָּא וְזָעוּ, וְקָמוּ מֵרַחִיק. 
כ,טו וַיֹּֽאמְרוּ֙ אֶל־מֹשֶׁ֔ה דַּבֵּר־אַתָּ֥ה עִמָּ֖נוּ וְנִשְׁמָ֑עָה וְאַל־יְדַבֵּ֥ר עִמָּ֛נוּ אֱלֹהִ֖ים פֶּן־נָמֽוּת׃ וַאֲמַרוּ, לְמֹשֶׁה, מַלֵּיל אַתְּ עִמַּנָא, וּנְקַבֵּיל; וְלָא יִתְמַלַּל עִמַּנָא מִן קֳדָם יְיָ, דִּלְמָא נְמוּת. 
כ,טז וַיֹּ֨אמֶר מֹשֶׁ֣ה אֶל־הָעָם֮ אַל־תִּירָאוּ֒ כִּ֗י לְבַֽעֲבוּר֙ נַסּ֣וֹת אֶתְכֶ֔ם בָּ֖א הָֽאֱלֹהִ֑ים וּבַֽעֲב֗וּר תִּֽהְיֶ֧ה יִרְאָת֛וֹ עַל־פְּנֵיכֶ֖ם לְבִלְתִּ֥י תֶֽחֱטָֽאוּ׃ וַאֲמַר מֹשֶׁה לְעַמָּא, לָא תִּדְחֲלוּן, אֲרֵי בְּדִיל לְנַסָּאָה יָתְכוֹן, אִתְגְּלִי לְכוֹן יְקָרָא דַּייָ; וּבְדִיל, דִּתְהֵי דַּחְלְתֵיהּ עַל אַפֵּיכוֹן--בְּדִיל דְּלָא תְּחוּבוּן. 
כ,יז וַיַּֽעֲמֹ֥ד הָעָ֖ם מֵֽרָחֹ֑ק וּמֹשֶׁה֙ נִגַּ֣שׁ אֶל־הָֽעֲרָפֶ֔ל אֲשֶׁר־שָׁ֖ם הָֽאֱלֹהִֽים׃ {ס} וְקָם עַמָּא, מֵרַחִיק; וּמֹשֶׁה קְרֵיב לְצַד אֲמִטְּתָא, דְּתַמָּן יְקָרָא דַּייָ. {ס}
כ,יח וַיֹּ֤אמֶר יְהוָה֙ אֶל־מֹשֶׁ֔ה כֹּ֥ה תֹאמַ֖ר אֶל־בְּנֵ֣י יִשְׂרָאֵ֑ל אַתֶּ֣ם רְאִיתֶ֔ם כִּ֚י מִן־הַשָּׁמַ֔יִם דִּבַּ֖רְתִּי עִמָּכֶֽם׃ וַאֲמַר יְיָ לְמֹשֶׁה, כִּדְנָן תֵּימַר לִבְנֵי יִשְׂרָאֵל:  אַתּוּן חֲזֵיתוֹן--אֲרֵי מִן שְׁמַיָּא, מַלֵּילִית עִמְּכוֹן. 
כ,יט לֹ֥א תַֽעֲשׂ֖וּן אִתִּ֑י אֱלֹ֤הֵי כֶ֨סֶף֙ וֵֽאלֹהֵ֣י זָהָ֔ב לֹ֥א תַֽעֲשׂ֖וּ לָכֶֽם׃ לָא תַּעְבְּדוּן, קֳדָמָי:  דַּחְלָן דִּכְסַף וְדַחְלָן דִּדְהַב, לָא תַּעְבְּדוּן לְכוֹן. 
כ,כ מִזְבַּ֣ח אֲדָמָה֮ תַּֽעֲשֶׂה־לִּי֒ וְזָֽבַחְתָּ֣ עָלָ֗יו אֶת־עֹֽלֹתֶ֨יךָ֙ וְאֶת־שְׁלָמֶ֔יךָ אֶת־צֹֽאנְךָ֖ וְאֶת־בְּקָרֶ֑ךָ בְּכָל־הַמָּקוֹם֙ אֲשֶׁ֣ר אַזְכִּ֣יר אֶת־שְׁמִ֔י אָב֥וֹא אֵלֶ֖יךָ וּבֵֽרַכְתִּֽיךָ׃ מַדְבַּח אֲדַמְתָּא, תַּעֲבֵיד קֳדָמַי, וּתְהֵי דָּבַח עֲלוֹהִי יָת עֲלָוָתָךְ וְיָת נִכְסַת קֻדְשָׁךְ, מִן עָנָךְ וּמִן תּוֹרָךְ; בְּכָל אֲתַר דְּאַשְׁרֵי שְׁכִינְתִי, לְתַמָּן אֶשְׁלַח בִּרְכְתִי לָךְ וַאֲבָרְכִנָּךְ. 
כ,כא וְאִם־מִזְבַּ֤ח אֲבָנִים֙ תַּֽעֲשֶׂה־לִּ֔י לֹֽא־תִבְנֶ֥ה אֶתְהֶ֖ן גָּזִ֑ית כִּ֧י חַרְבְּךָ֛ הֵנַ֥פְתָּ עָלֶ֖יהָ וַתְּחַֽלְלֶֽהָ׃ וְאִם מַדְבַּח אַבְנִין תַּעֲבֵיד קֳדָמַי, לָא תִּבְנֵי יָתְהוֹן פְּסִילָן:  דִּלְמָא תְּרִים חַרְבָּךְ עֲלַהּ, וְתַחֲלִנַּהּ. 
כ,כב וְלֹֽא־תַעֲלֶ֥ה בְמַֽעֲלֹ֖ת עַֽל־מִזְבְּחִ֑י אֲשֶׁ֛ר לֹֽא־תִגָּלֶ֥ה עֶרְוָֽתְךָ֖ עָלָֽיו׃ {פ} וְלָא תִּסַּק בְּדַרְגִּין, עַל מַדְבְּחִי:  דְּלָא תִּתְגַּלֵּי עַרְיְתָךְ, עֲלוֹהִי. {פ}
