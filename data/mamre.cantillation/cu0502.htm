ואתחנן

ג,כג וָֽאֶתְחַנַּ֖ן אֶל־יְהוָ֑ה בָּעֵ֥ת הַהִ֖וא לֵאמֹֽר׃ וְצַלִּיתִי, קֳדָם יְיָ, בְּעִדָּנָא הַהוּא, לְמֵימַר. 
ג,כד אֲדֹנָ֣י יְהוִ֗ה אַתָּ֤ה הַֽחִלּ֨וֹתָ֙ לְהַרְא֣וֹת אֶֽת־עַבְדְּךָ֔ אֶ֨ת־גָּדְלְךָ֔ וְאֶת־יָֽדְךָ֖ הַֽחֲזָקָ֑ה אֲשֶׁ֤ר מִי־אֵל֙ בַּשָּׁמַ֣יִם וּבָאָ֔רֶץ אֲשֶׁר־יַֽעֲשֶׂ֥ה כְמַֽעֲשֶׂ֖יךָ וְכִגְבֽוּרֹתֶֽךָ׃ יְיָ אֱלֹהִים, אַתְּ שָׁרִיתָא לְאַחְזָאָה יָת עַבְדָּךְ, יָת רְבוּתָךְ, וְיָת יְדָךְ תַּקִּיפְתָא--דְּאַתְּ הוּא אֱלָהָא דִּשְׁכִינְתָךְ בִּשְׁמַיָּא מִלְּעֵילָא וְשַׁלִּיט בְּאַרְעָא, לֵית דְּיַעֲבֵיד כְּעוּבָדָךְ וּכְגִבָּרְוָתָךְ. 
ג,כה אֶעְבְּרָה־נָּ֗א וְאֶרְאֶה֙ אֶת־הָאָ֣רֶץ הַטּוֹבָ֔ה אֲשֶׁ֖ר בְּעֵ֣בֶר הַיַּרְדֵּ֑ן הָהָ֥ר הַטּ֛וֹב הַזֶּ֖ה וְהַלְּבָנֹֽן׃ אֶעְבַּר כְּעַן, וְאֶחְזֵי יָת אַרְעָא טָבְתָא--דִּבְעִבְרָא דְּיַרְדְּנָא:  טוּרָא טָבָא הָדֵין, וּבֵית מַקְדְּשָׁא. 
ג,כו וַיִּתְעַבֵּ֨ר יְהוָ֥ה בִּי֙ לְמַ֣עַנְכֶ֔ם וְלֹ֥א שָׁמַ֖ע אֵלָ֑י וַיֹּ֨אמֶר יְהוָ֤ה אֵלַי֙ רַב־לָ֔ךְ אַל־תּ֗וֹסֶף דַּבֵּ֥ר אֵלַ֛י ע֖וֹד בַּדָּבָ֥ר הַזֶּֽה׃ וַהֲוָה רְגַז מִן קֳדָם יְיָ עֲלַי בְּדִילְכוֹן, וְלָא קַבֵּיל מִנִּי; וַאֲמַר יְיָ לִי, סַגִּי לָךְ--לָא תּוֹסֵיף לְמַלָּלָא קֳדָמַי עוֹד, בְּפִתְגָמָא הָדֵין. 
ג,כז עֲלֵ֣ה ׀ רֹ֣אשׁ הַפִּסְגָּ֗ה וְשָׂ֥א עֵינֶ֛יךָ יָ֧מָּה וְצָפֹ֛נָה וְתֵימָ֥נָה וּמִזְרָ֖חָה וּרְאֵ֣ה בְעֵינֶ֑יךָ כִּי־לֹ֥א תַֽעֲבֹ֖ר אֶת־הַיַּרְדֵּ֥ן הַזֶּֽה׃ סַק לְרֵישׁ רָמְתָא, וּזְקַף עֵינָךְ לְמַעְרְבָא וּלְצִפּוּנָא וּלְדָרוֹמָא וּלְמַדְנְחָא--וַחְזִי בְּעֵינָךְ:  אֲרֵי לָא תִּעְבַּר, יָת יַרְדְּנָא הָדֵין. 
ג,כח וְצַ֥ו אֶת־יְהוֹשֻׁ֖עַ וְחַזְּקֵ֣הוּ וְאַמְּצֵ֑הוּ כִּי־ה֣וּא יַֽעֲבֹ֗ר לִפְנֵי֙ הָעָ֣ם הַזֶּ֔ה וְהוּא֙ יַנְחִ֣יל אוֹתָ֔ם אֶת־הָאָ֖רֶץ אֲשֶׁ֥ר תִּרְאֶֽה׃ וּפַקֵּיד יָת יְהוֹשׁוּעַ, וְתַקֵּיפְהִי וְאַלֵּימְהִי:  אֲרֵי הוּא יִעְבַּר, קֳדָם עַמָּא הָדֵין, וְהוּא יַחְסֵין יָתְהוֹן, יָת אַרְעָא דְּתִחְזֵי. 
ג,כט וַנֵּ֣שֶׁב בַּגָּ֔יְא מ֖וּל בֵּ֥ית פְּעֽוֹר׃ {פ} וִיתֵיבְנָא בְּחִלְּתָא, לָקֳבֵיל בֵּית פְּעוֹר. {פ}
ד,א וְעַתָּ֣ה יִשְׂרָאֵ֗ל שְׁמַ֤ע אֶל־הַֽחֻקִּים֙ וְאֶל־הַמִּשְׁפָּטִ֔ים אֲשֶׁ֧ר אָֽנֹכִ֛י מְלַמֵּ֥ד אֶתְכֶ֖ם לַֽעֲשׂ֑וֹת לְמַ֣עַן תִּֽחְי֗וּ וּבָאתֶם֙ וִֽירִשְׁתֶּ֣ם אֶת־הָאָ֔רֶץ אֲשֶׁ֧ר יְהוָ֛ה אֱלֹהֵ֥י אֲבֹֽתֵיכֶ֖ם נֹתֵ֥ן לָכֶֽם׃ וּכְעַן יִשְׂרָאֵל, שְׁמַע לִקְיָמַיָּא וּלְדִינַיָּא, דַּאֲנָא מַלֵּיף יָתְכוֹן, לְמֶעֱבַד--בְּדִיל דְּתֵיחוֹן, וְתֵיעֲלוּן וְתֵירְתוּן יָת אַרְעָא, דַּייָ אֱלָהָא דַּאֲבָהָתְכוֹן, יָהֵיב לְכוֹן. 
ד,ב לֹ֣א תֹסִ֗פוּ עַל־הַדָּבָר֙ אֲשֶׁ֤ר אָֽנֹכִי֙ מְצַוֶּ֣ה אֶתְכֶ֔ם וְלֹ֥א תִגְרְע֖וּ מִמֶּ֑נּוּ לִשְׁמֹ֗ר אֶת־מִצְו‍ֹת֙ יְהוָ֣ה אֱלֹֽהֵיכֶ֔ם אֲשֶׁ֥ר אָֽנֹכִ֖י מְצַוֶּ֥ה אֶתְכֶֽם׃ לָא תֵּיסְפוּן, עַל פִּתְגָמָא דַּאֲנָא מְפַקֵּיד יָתְכוֹן, וְלָא תִּמְנְעוּן, מִנֵּיהּ--לְמִטַּר, יָת פִּקּוֹדַיָּא דַּייָ אֱלָהֲכוֹן, דַּאֲנָא, מְפַקֵּיד יָתְכוֹן. 
ד,ג עֵֽינֵיכֶם֙ הָֽרֹא֔וֹת אֵ֛ת אֲשֶׁר־עָשָׂ֥ה יְהוָ֖ה בְּבַ֣עַל פְּע֑וֹר כִּ֣י כָל־הָאִ֗ישׁ אֲשֶׁ֤ר הָלַךְ֙ אַֽחֲרֵ֣י בַֽעַל־פְּע֔וֹר הִשְׁמִיד֛וֹ יְהוָ֥ה אֱלֹהֶ֖יךָ מִקִּרְבֶּֽךָ׃ עֵינֵיכוֹן, חֲזַאָה, יָת דַּעֲבַד יְיָ, בְּפָלְחֵי בַּעְלָא פְּעוֹר:  אֲרֵי כָּל גֻּבְרָא, דְּהַלֵּיךְ בָּתַר בַּעְלָא פְּעוֹר--שֵׁיצְיֵיהּ יְיָ אֱלָהָךְ, מִבֵּינָךְ. 
ד,ד וְאַתֶּם֙ הַדְּבֵקִ֔ים בַּֽיהוָ֖ה אֱלֹֽהֵיכֶ֑ם חַיִּ֥ים כֻּלְּכֶ֖ם הַיּֽוֹם׃ וְאַתּוּן, דְּאִדְּבֵיקְתּוּן, בְּדַחְלְתָא דַּייָ, אֱלָהֲכוֹן--קַיָּמִין כֻּלְּכוֹן, יוֹמָא דֵּין. 
ד,ה רְאֵ֣ה ׀ לִמַּ֣דְתִּי אֶתְכֶ֗ם חֻקִּים֙ וּמִשְׁפָּטִ֔ים כַּֽאֲשֶׁ֥ר צִוַּ֖נִי יְהוָ֣ה אֱלֹהָ֑י לַֽעֲשׂ֣וֹת כֵּ֔ן בְּקֶ֣רֶב הָאָ֔רֶץ אֲשֶׁ֥ר אַתֶּ֛ם בָּאִ֥ים שָׁ֖מָּה לְרִשְׁתָּֽהּ׃ חֲזוֹ דְּאַלֵּיפִית יָתְכוֹן, קְיָמִין וְדִינִין, כְּמָא דְּפַקְּדַנִי, יְיָ אֱלָהִי:  לְמֶעֱבַד כֵּין--בְּגוֹ אַרְעָא, דְּאַתּוּן עָאלִין לְתַמָּן לְמֵירְתַהּ. 
ד,ו וּשְׁמַרְתֶּם֮ וַֽעֲשִׂיתֶם֒ כִּ֣י הִ֤וא חָכְמַתְכֶם֙ וּבִ֣ינַתְכֶ֔ם לְעֵינֵ֖י הָֽעַמִּ֑ים אֲשֶׁ֣ר יִשְׁמְע֗וּן אֵ֚ת כָּל־הַֽחֻקִּ֣ים הָאֵ֔לֶּה וְאָֽמְר֗וּ רַ֚ק עַם־חָכָ֣ם וְנָב֔וֹן הַגּ֥וֹי הַגָּד֖וֹל הַזֶּֽה׃ וְתִטְּרוּן, וְתַעְבְּדוּן--אֲרֵי הִיא חָכְמַתְכוֹן וְסֻכְלְתָנוּתְכוֹן, לְעֵינֵי עַמְמַיָּא:  דְּיִשְׁמְעוּן, יָת כָּל קְיָמַיָּא הָאִלֵּין, וְיֵימְרוּן לְחוֹד עַם חַכִּים וְסֻכְלְתָן, עַמָּא רַבָּא הָדֵין. 
ד,ז כִּ֚י מִי־ג֣וֹי גָּד֔וֹל אֲשֶׁר־ל֥וֹ אֱלֹהִ֖ים קְרֹבִ֣ים אֵלָ֑יו כַּֽיהוָ֣ה אֱלֹהֵ֔ינוּ בְּכָל־קָרְאֵ֖נוּ אֵלָֽיו׃ אֲרֵי מַאן עַם רַב, דְּלֵיהּ אֱלָהּ קָרִיב לֵיהּ לְקַבָּלָא צְלוֹתֵיהּ בְּעִדָּן עָקְתֵיהּ, כַּייָ אֱלָהַנָא, בְּכָל עִדָּן דַּאֲנַחְנָא מְצַלַּן קֳדָמוֹהִי. 
ד,ח וּמִי֙ גּ֣וֹי גָּד֔וֹל אֲשֶׁר־ל֛וֹ חֻקִּ֥ים וּמִשְׁפָּטִ֖ים צַדִּיקִ֑ם כְּכֹל֙ הַתּוֹרָ֣ה הַזֹּ֔את אֲשֶׁ֧ר אָֽנֹכִ֛י נֹתֵ֥ן לִפְנֵיכֶ֖ם הַיּֽוֹם׃ וּמַאן עַם רַב, דְּלֵיהּ קְיָמִין וְדִינִין קַשִּׁיטִין, כְּכֹל אוֹרָיְתָא הָדָא, דַּאֲנָא יָהֵיב קֳדָמֵיכוֹן יוֹמָא דֵּין. 
ד,ט רַ֡ק הִשָּׁ֣מֶר לְךָ֩ וּשְׁמֹ֨ר נַפְשְׁךָ֜ מְאֹ֗ד פֶּן־תִּשְׁכַּ֨ח אֶת־הַדְּבָרִ֜ים אֲשֶׁר־רָא֣וּ עֵינֶ֗יךָ וּפֶן־יָס֨וּרוּ֙ מִלְּבָ֣בְךָ֔ כֹּ֖ל יְמֵ֣י חַיֶּ֑יךָ וְהֽוֹדַעְתָּ֥ם לְבָנֶ֖יךָ וְלִבְנֵ֥י בָנֶֽיךָ׃ לְחוֹד אִסְתְּמַר לָךְ וְטַר נַפְשָׁךְ לַחְדָּא, דִּלְמָא תִּתְנְשֵׁי יָת פִּתְגָמַיָּא דַּחֲזַאָה עֵינָךְ וְדִלְמָא יִעְדּוֹן מִלִּבָּךְ, כֹּל, יוֹמֵי חַיָּךְ; וּתְהוֹדְעִנּוּן לִבְנָךְ, וְלִבְנֵי בְּנָךְ. 
ד,י י֗וֹם אֲשֶׁ֨ר עָמַ֜דְתָּ לִפְנֵ֨י יְהוָ֣ה אֱלֹהֶיךָ֮ בְּחֹרֵב֒ בֶּֽאֱמֹ֨ר יְהוָ֜ה אֵלַ֗י הַקְהֶל־לִי֙ אֶת־הָעָ֔ם וְאַשְׁמִעֵ֖ם אֶת־דְּבָרָ֑י אֲשֶׁ֨ר יִלְמְד֜וּן לְיִרְאָ֣ה אֹתִ֗י כָּל־הַיָּמִים֙ אֲשֶׁ֨ר הֵ֤ם חַיִּים֙ עַל־הָ֣אֲדָמָ֔ה וְאֶת־בְּנֵיהֶ֖ם יְלַמֵּדֽוּן׃ יוֹמָא, דְּקַמְתָּא קֳדָם יְיָ אֱלָהָךְ בְּחוֹרֵב, כַּד אֲמַר יְיָ לִי כְּנוֹשׁ קֳדָמַי יָת עַמָּא, וְאַשְׁמְעִנּוּן יָת פִּתְגָמָי:  דְּיֵילְפוּן לְמִדְחַל קֳדָמַי, כָּל יוֹמַיָּא דְּאִנּוּן קַיָּמִין עַל אַרְעָא, וְיָת בְּנֵיהוֹן, יַלְּפוּן. 
ד,יא וַתִּקְרְב֥וּן וַתַּֽעַמְד֖וּן תַּ֣חַת הָהָ֑ר וְהָהָ֞ר בֹּעֵ֤ר בָּאֵשׁ֙ עַד־לֵ֣ב הַשָּׁמַ֔יִם חֹ֖שֶׁךְ עָנָ֥ן וַֽעֲרָפֶֽל׃ וּקְרֵיבְתּוּן וְקַמְתּוּן, בְּשִׁפּוֹלֵי טוּרָא; וְטוּרָא בָּעֵר בְּאִישָׁתָא, עַד צֵית שְׁמַיָּא--חֲשׁוֹכָא, עֲנָנָא וַאֲמִטְּתָא. 
ד,יב וַיְדַבֵּ֧ר יְהוָ֛ה אֲלֵיכֶ֖ם מִתּ֣וֹךְ הָאֵ֑שׁ ק֤וֹל דְּבָרִים֙ אַתֶּ֣ם שֹֽׁמְעִ֔ים וּתְמוּנָ֛ה אֵֽינְכֶ֥ם רֹאִ֖ים זֽוּלָתִ֥י קֽוֹל׃ וּמַלֵּיל יְיָ עִמְּכוֹן, מִגּוֹ אִישָׁתָא:  קָל פִּתְגָמִין אַתּוּן שָׁמְעִין, וּדְמוּ לֵיתֵיכוֹן חָזַן אֱלָהֵין קָלָא. 
ד,יג וַיַּגֵּ֨ד לָכֶ֜ם אֶת־בְּרִית֗וֹ אֲשֶׁ֨ר צִוָּ֤ה אֶתְכֶם֙ לַֽעֲשׂ֔וֹת עֲשֶׂ֖רֶת הַדְּבָרִ֑ים וַֽיִּכְתְּבֵ֔ם עַל־שְׁנֵ֖י לֻח֥וֹת אֲבָנִֽים׃ וְחַוִּי לְכוֹן יָת קְיָמֵיהּ, דְּפַקֵּיד יָתְכוֹן לְמֶעֱבַד--עַסְרָא, פִּתְגָמִין; וּכְתַבִנּוּן, עַל תְּרֵין לוּחֵי אַבְנַיָּא. 
ד,יד וְאֹתִ֞י צִוָּ֤ה יְהוָה֙ בָּעֵ֣ת הַהִ֔וא לְלַמֵּ֣ד אֶתְכֶ֔ם חֻקִּ֖ים וּמִשְׁפָּטִ֑ים לַעֲשֹֽׂתְכֶ֣ם אֹתָ֔ם בָּאָ֕רֶץ אֲשֶׁ֥ר אַתֶּ֛ם עֹֽבְרִ֥ים שָׁ֖מָּה לְרִשְׁתָּֽהּ׃ וְיָתִי פַּקֵּיד יְיָ, בְּעִדָּנָא הַהוּא, לְאַלָּפָא יָתְכוֹן, קְיָמִין וְדִינִין:  לְמֶעֱבַדְכוֹן יָתְהוֹן--בְּאַרְעָא, דְּאַתּוּן עָבְרִין לְתַמָּן לְמֵירְתַהּ. 
ד,טו וְנִשְׁמַרְתֶּ֥ם מְאֹ֖ד לְנַפְשֹֽׁתֵיכֶ֑ם כִּ֣י לֹ֤א רְאִיתֶם֙ כָּל־תְּמוּנָ֔ה בְּי֗וֹם דִּבֶּ֨ר יְהוָ֧ה אֲלֵיכֶ֛ם בְּחֹרֵ֖ב מִתּ֥וֹךְ הָאֵֽשׁ׃ וְתִסְתַּמְרוּן לַחְדָּא, לְנַפְשָׁתְכוֹן:  אֲרֵי לָא חֲזֵיתוֹן, כָּל דְּמוּ, בְּיוֹמָא דְּמַלֵּיל יְיָ עִמְּכוֹן בְּחוֹרֵב, מִגּוֹ אִישָׁתָא. 
ד,טז פֶּ֨ן־תַּשְׁחִת֔וּן וַֽעֲשִׂיתֶ֥ם לָכֶ֛ם פֶּ֖סֶל תְּמוּנַ֣ת כָּל־סָ֑מֶל תַּבְנִ֥ית זָכָ֖ר א֥וֹ נְקֵבָֽה׃ דִּלְמָא תְּחַבְּלוּן--וְתַעְבְּדוּן לְכוֹן צֵילַם, דְּמוּת כָּל צוּרָא:  דְּמוּת דְּכַר, אוֹ נֻקְבָּא. 
ד,יז תַּבְנִ֕ית כָּל־בְּהֵמָ֖ה אֲשֶׁ֣ר בָּאָ֑רֶץ תַּבְנִית֙ כָּל־צִפּ֣וֹר כָּנָ֔ף אֲשֶׁ֥ר תָּע֖וּף בַּשָּׁמָֽיִם׃ דְּמוּת, כָּל בְּעִירָא דִּבְאַרְעָא; דְּמוּת כָּל צִפַּר גַּפָּא, דְּפָרַח בְּאַוֵּיר רְקִיעַ שְׁמַיָּא. 
ד,יח תַּבְנִ֕ית כָּל־רֹמֵ֖שׂ בָּֽאֲדָמָ֑ה תַּבְנִ֛ית כָּל־דָּגָ֥ה אֲשֶׁר־בַּמַּ֖יִם מִתַּ֥חַת לָאָֽרֶץ׃ דְּמוּת, כָּל רִחְשָׁא דִּבְאַרְעָא; דְּמוּת כָּל נוּנֵי דִּבְמַיָּא, מִלְּרַע לְאַרְעָא. 
ד,יט וּפֶן־תִּשָּׂ֨א עֵינֶ֜יךָ הַשָּׁמַ֗יְמָה וְֽ֠רָאִיתָ אֶת־הַשֶּׁ֨מֶשׁ וְאֶת־הַיָּרֵ֜חַ וְאֶת־הַכּֽוֹכָבִ֗ים כֹּ֚ל צְבָ֣א הַשָּׁמַ֔יִם וְנִדַּחְתָּ֛ וְהִשְׁתַּֽחֲוִ֥יתָ לָהֶ֖ם וַֽעֲבַדְתָּ֑ם אֲשֶׁ֨ר חָלַ֜ק יְהוָ֤ה אֱלֹהֶ֨יךָ֙ אֹתָ֔ם לְכֹל֙ הָֽעַמִּ֔ים תַּ֖חַת כָּל־הַשָּׁמָֽיִם׃ וְדִלְמָא תִּזְקוֹף עֵינָךְ לִשְׁמַיָּא, וְתִחְזֵי יָת שִׁמְשָׁא וְיָת סֵיהֲרָא וְיָת כּוֹכְבַיָּא כֹּל חֵילֵי שְׁמַיָּא, וְתִטְעֵי וְתִסְגּוֹד לְהוֹן, וְתִפְלְחִנִּין--דְּזָמֵין יְיָ אֱלָהָךְ, יָתְהוֹן, לְכֹל עַמְמַיָּא, תְּחוֹת כָּל שְׁמַיָּא. 
ד,כ וְאֶתְכֶם֙ לָקַ֣ח יְהוָ֔ה וַיּוֹצִ֥א אֶתְכֶ֛ם מִכּ֥וּר הַבַּרְזֶ֖ל מִמִּצְרָ֑יִם לִֽהְי֥וֹת ל֛וֹ לְעַ֥ם נַֽחֲלָ֖ה כַּיּ֥וֹם הַזֶּֽה׃ וְיָתְכוֹן קָרֵיב יְיָ לְדַחְלְתֵיהּ, וְאַפֵּיק יָתְכוֹן מִכּוּרָא דְּבַרְזְלָא מִמִּצְרָיִם, לְמִהְוֵי לֵיהּ לְעַם אַחְסָנָא, כְּיוֹמָא הָדֵין. 
ד,כא וַֽיהוָ֥ה הִתְאַנַּף־בִּ֖י עַל־דִּבְרֵיכֶ֑ם וַיִּשָּׁבַ֗ע לְבִלְתִּ֤י עָבְרִי֙ אֶת־הַיַּרְדֵּ֔ן וּלְבִלְתִּי־בֹא֙ אֶל־הָאָ֣רֶץ הַטּוֹבָ֔ה אֲשֶׁר֙ יְהוָ֣ה אֱלֹהֶ֔יךָ נֹתֵ֥ן לְךָ֖ נַֽחֲלָֽה׃ וּמִן קֳדָם יְיָ הֲוָה רְגַז עֲלַי, עַל פִּתְגָמֵיכוֹן; וְקַיֵּים, בְּדִיל דְּלָא לְמִעְבְּרִי יָת יַרְדְּנָא, וּבְדִיל דְּלָא לְמֵיעַל לְאַרְעָא טָבְתָא, דַּייָ אֱלָהָךְ יָהֵיב לָךְ אַחְסָנָא. 
ד,כב כִּ֣י אָֽנֹכִ֥י מֵת֙ בָּאָ֣רֶץ הַזֹּ֔את אֵינֶ֥נִּי עֹבֵ֖ר אֶת־הַיַּרְדֵּ֑ן וְאַתֶּם֙ עֹֽבְרִ֔ים וִֽירִשְׁתֶּ֕ם אֶת־הָאָ֥רֶץ הַטּוֹבָ֖ה הַזֹּֽאת׃ אֲרֵי אֲנָא מָאִית בְּאַרְעָא הָדָא, לֵית אֲנָא עָבַר יָת יַרְדְּנָא; וְאַתּוּן, עָבְרִין, וְתֵירְתוּן, יָת אַרְעָא טָבְתָא הָדָא. 
ד,כג הִשָּֽׁמְר֣וּ לָכֶ֗ם פֶּֽן־תִּשְׁכְּחוּ֙ אֶת־בְּרִ֤ית יְהוָה֙ אֱלֹ֣הֵיכֶ֔ם אֲשֶׁ֥ר כָּרַ֖ת עִמָּכֶ֑ם וַֽעֲשִׂיתֶ֨ם לָכֶ֥ם פֶּ֨סֶל֙ תְּמ֣וּנַת כֹּ֔ל אֲשֶׁ֥ר צִוְּךָ֖ יְהוָ֥ה אֱלֹהֶֽיךָ׃ אִסְתְּמַרוּ לְכוֹן, דִּלְמָא תִּתְנְשׁוּן יָת קְיָמָא דַּייָ אֱלָהֲכוֹן, דִּגְזַר, עִמְּכוֹן; וְתַעְבְּדוּן לְכוֹן צֵילַם דְּמוּת כּוֹלָא, דְּפַקְּדָךְ יְיָ אֱלָהָךְ. 
ד,כד כִּ֚י יְהוָ֣ה אֱלֹהֶ֔יךָ אֵ֥שׁ אֹֽכְלָ֖ה ה֑וּא אֵ֖ל קַנָּֽא׃ {פ} אֲרֵי יְיָ אֱלָהָךְ, מֵימְרֵיהּ אִישָׁא אָכְלָא הוּא:  אֵל, קַנָּא. {פ}
ד,כה כִּֽי־תוֹלִ֤יד בָּנִים֙ וּבְנֵ֣י בָנִ֔ים וְנֽוֹשַׁנְתֶּ֖ם בָּאָ֑רֶץ וְהִשְׁחַתֶּ֗ם וַֽעֲשִׂ֤יתֶם פֶּ֨סֶל֙ תְּמ֣וּנַת כֹּ֔ל וַֽעֲשִׂיתֶ֥ם הָרַ֛ע בְּעֵינֵ֥י יְהוָֽה־אֱלֹהֶ֖יךָ לְהַכְעִיסֽוֹ׃ אֲרֵי תֵּילְדוּן בְּנִין וּבְנֵי בְּנִין, וְתִתְעַתְּקוּן בְּאַרְעָא; וּתְחַבְּלוּן, וְתַעְבְּדוּן צֵילַם דְּמוּת כּוֹלָא, וְתַעְבְּדוּן דְּבִישׁ קֳדָם יְיָ אֱלָהָךְ, לְאַרְגָּזָא קֳדָמוֹהִי. 
ד,כו הַֽעִידֹתִי֩ בָכֶ֨ם הַיּ֜וֹם אֶת־הַשָּׁמַ֣יִם וְאֶת־הָאָ֗רֶץ כִּֽי־אָבֹ֣ד תֹּֽאבֵדוּן֮ מַהֵר֒ מֵעַ֣ל הָאָ֔רֶץ אֲשֶׁ֨ר אַתֶּ֜ם עֹֽבְרִ֧ים אֶת־הַיַּרְדֵּ֛ן שָׁ֖מָּה לְרִשְׁתָּ֑הּ לֹֽא־תַאֲרִיכֻ֤ן יָמִים֙ עָלֶ֔יהָ כִּ֥י הִשָּׁמֵ֖ד תִּשָּֽׁמֵדֽוּן׃ אַסְהֵידִית בְּכוֹן יוֹמָא דֵּין יָת שְׁמַיָּא וְיָת אַרְעָא, אֲרֵי מֵיבָד תֵּיבְדוּן בִּפְרִיעַ, מֵעַל אַרְעָא, דְּאַתּוּן עָבְרִין יָת יַרְדְּנָא לְתַמָּן לְמֵירְתַהּ:  לָא תֵּירְכוּן יוֹמִין עֲלַהּ, אֲרֵי אִשְׁתֵּיצָאָה תִּשְׁתֵּיצוֹן. 
ד,כז וְהֵפִ֧יץ יְהוָ֛ה אֶתְכֶ֖ם בָּֽעַמִּ֑ים וְנִשְׁאַרְתֶּם֙ מְתֵ֣י מִסְפָּ֔ר בַּגּוֹיִ֕ם אֲשֶׁ֨ר יְנַהֵ֧ג יְהוָ֛ה אֶתְכֶ֖ם שָֽׁמָּה׃ וִיבַדַּר יְיָ יָתְכוֹן, בֵּינֵי עַמְמַיָּא; וְתִשְׁתְּאַרוּן, עַם דְּמִנְיָן, בְּעַמְמַיָּא, דִּידַבַּר יְיָ יָתְכוֹן לְתַמָּן. 
ד,כח וַֽעֲבַדְתֶּם־שָׁ֣ם אֱלֹהִ֔ים מַֽעֲשֵׂ֖ה יְדֵ֣י אָדָ֑ם עֵ֣ץ וָאֶ֔בֶן אֲשֶׁ֤ר לֹֽא־יִרְאוּן֙ וְלֹ֣א יִשְׁמְע֔וּן וְלֹ֥א יֹֽאכְל֖וּן וְלֹ֥א יְרִיחֻֽן׃ וְתִפְלְחוּן תַּמָּן לְעַמְמַיָּא פָּלְחֵי טָעֲוָתָא, עוֹבָד יְדֵי אֲנָשָׁא:  אָעָא וְאַבְנָא--דְּלָא חָזַן וְלָא שָׁמְעִין, וְלָא אָכְלִין וְלָא מְרִיחִין. 
ד,כט וּבִקַּשְׁתֶּ֥ם מִשָּׁ֛ם אֶת־יְהוָ֥ה אֱלֹהֶ֖יךָ וּמָצָ֑אתָ כִּ֣י תִדְרְשֶׁ֔נּוּ בְּכָל־לְבָֽבְךָ֖ וּבְכָל־נַפְשֶֽׁךָ׃ וְתִתְבְּעוּן מִתַּמָּן יָת דַּחְלְתָא דַּייָ אֱלָהָךְ, וְתַשְׁכַּח:  אֲרֵי תִּבְעֵי מִן קֳדָמוֹהִי, בְּכָל לִבָּךְ וּבְכָל נַפְשָׁךְ. 
ד,ל בַּצַּ֣ר לְךָ֔ וּמְצָא֕וּךָ כֹּ֖ל הַדְּבָרִ֣ים הָאֵ֑לֶּה בְּאַֽחֲרִית֙ הַיָּמִ֔ים וְשַׁבְתָּ֙ עַד־יְהוָ֣ה אֱלֹהֶ֔יךָ וְשָֽׁמַעְתָּ֖ בְּקֹלֽוֹ׃ כַּד תֵּיעוֹק לָךְ--וְיַשְׁכְּחוּנָּךְ, כֹּל פִּתְגָמַיָּא הָאִלֵּין; בְּסוֹף, יוֹמַיָּא, וּתְתוּב לְדַחְלְתָא דַּייָ אֱלָהָךְ, וּתְקַבֵּיל לְמֵימְרֵיהּ. 
ד,לא כִּ֣י אֵ֤ל רַחוּם֙ יְהוָ֣ה אֱלֹהֶ֔יךָ לֹ֥א יַרְפְּךָ֖ וְלֹ֣א יַשְׁחִיתֶ֑ךָ וְלֹ֤א יִשְׁכַּח֙ אֶת־בְּרִ֣ית אֲבֹתֶ֔יךָ אֲשֶׁ֥ר נִשְׁבַּ֖ע לָהֶֽם׃ אֲרֵי אֱלָהָא רַחְמָנָא יְיָ אֱלָהָךְ, לָא יִשְׁבְּקִנָּךְ וְלָא יְחַבְּלִנָּךְ; וְלָא יִתְנְשֵׁי יָת קְיָמָא דַּאֲבָהָתָךְ, דְּקַיֵּים לְהוֹן. 
ד,לב כִּ֣י שְׁאַל־נָא֩ לְיָמִ֨ים רִֽאשֹׁנִ֜ים אֲשֶׁר־הָי֣וּ לְפָנֶ֗יךָ לְמִן־הַיּוֹם֙ אֲשֶׁר֩ בָּרָ֨א אֱלֹהִ֤ים ׀ אָדָם֙ עַל־הָאָ֔רֶץ וּלְמִקְצֵ֥ה הַשָּׁמַ֖יִם וְעַד־קְצֵ֣ה הַשָּׁמָ֑יִם הֲנִֽהְיָ֗ה כַּדָּבָ֤ר הַגָּדוֹל֙ הַזֶּ֔ה א֖וֹ הֲנִשְׁמַ֥ע כָּמֹֽהוּ׃ אֲרֵי שְׁאַל כְּעַן לְיוֹמַיָּא קַדְמָאֵי דַּהֲווֹ קֳדָמָךְ, לְמִן יוֹמָא דִּבְרָא יְיָ אָדָם עַל אַרְעָא, וּלְמִסְּיָפֵי שְׁמַיָּא, וְעַד סְיָפֵי שְׁמַיָּא:  הַהֲוָה, כְּפִתְגָמָא רַבָּא הָדֵין, אוֹ, הֲאִשְׁתְּמַע דִּכְוָתֵיהּ. 
ד,לג הֲשָׁ֣מַֽע עָם֩ ק֨וֹל אֱלֹהִ֜ים מְדַבֵּ֧ר מִתּוֹךְ־הָאֵ֛שׁ כַּֽאֲשֶׁר־שָׁמַ֥עְתָּ אַתָּ֖ה וַיֶּֽחִי׃ הַשְׁמַע עַמָּא קָל מֵימְרָא דַּייָ מְמַלֵּיל מִגּוֹ אִישָׁתָא, כְּמָא דִּשְׁמַעְתָּא אַתְּ--וְאִתְקַיַּם. 
ד,לד א֣וֹ ׀ הֲנִסָּ֣ה אֱלֹהִ֗ים לָ֠בוֹא לָקַ֨חַת ל֣וֹ גוֹי֮ מִקֶּ֣רֶב גּוֹי֒ בְּמַסֹּת֩ בְּאֹתֹ֨ת וּבְמֽוֹפְתִ֜ים וּבְמִלְחָמָ֗ה וּבְיָ֤ד חֲזָקָה֙ וּבִזְר֣וֹעַ נְטוּיָ֔ה וּבְמֽוֹרָאִ֖ים גְּדֹלִ֑ים כְּ֠כֹל אֲשֶׁר־עָשָׂ֨ה לָכֶ֜ם יְהוָ֧ה אֱלֹֽהֵיכֶ֛ם בְּמִצְרַ֖יִם לְעֵינֶֽיךָ׃ אוֹ נִסִּין עֲבַד יְיָ, לְאִתְגְּלָאָה לְמִפְרַק לֵיהּ עַם מִגּוֹ עַם, בְּנִסִּין בְּאָתִין וּבְמוֹפְתִין וּבִקְרָבָא וּבִיד תַּקִּיפָא וּבִדְרָעָא מְרָמְמָא, וּבְחֶזְוָנִין רַבְרְבִין:  כְּכֹל דַּעֲבַד לְכוֹן יְיָ אֱלָהֲכוֹן, בְּמִצְרַיִם--לְעֵינֵיכוֹן. 
ד,לה אַתָּה֙ הָרְאֵ֣תָ לָדַ֔עַת כִּ֥י יְהוָ֖ה ה֣וּא הָֽאֱלֹהִ֑ים אֵ֥ין ע֖וֹד מִלְּבַדּֽוֹ׃ אַתְּ אִתַּחְזִיתָא לְמִדַּע, אֲרֵי יְיָ הוּא אֱלֹהִים:  לֵית עוֹד, בָּר מִנֵּיהּ. 
ד,לו מִן־הַשָּׁמַ֛יִם הִשְׁמִֽיעֲךָ֥ אֶת־קֹל֖וֹ לְיַסְּרֶ֑ךָּ וְעַל־הָאָ֗רֶץ הֶרְאֲךָ֙ אֶת־אִשּׁ֣וֹ הַגְּדוֹלָ֔ה וּדְבָרָ֥יו שָׁמַ֖עְתָּ מִתּ֥וֹךְ הָאֵֽשׁ׃ מִן שְׁמַיּא אַשְׁמְעָךְ יָת קָל מֵימְרֵיהּ, לְאַלָּפוּתָךְ; וְעַל אַרְעָא, אַחְזְיָךְ יָת אִישָׁתֵיהּ רַבְּתָא, וּפִתְגָמוֹהִי שְׁמַעְתָּא, מִגּוֹ אִישָׁתָא. 
ד,לז וְתַ֗חַת כִּ֤י אָהַב֙ אֶת־אֲבֹתֶ֔יךָ וַיִּבְחַ֥ר בְּזַרְע֖וֹ אַֽחֲרָ֑יו וַיּוֹצִֽאֲךָ֧ בְּפָנָ֛יו בְּכֹח֥וֹ הַגָּדֹ֖ל מִמִּצְרָֽיִם׃ וּחְלָף, אֲרֵי רְחֵים יָת אֲבָהָתָךְ, וְאִתְרְעִי בִּבְנֵיהוֹן, בָּתְרֵיהוֹן; וְאַפְּקָךְ בְּמֵימְרֵיהּ בְּחֵילֵיהּ רַבָּא, מִמִּצְרָיִם. 
ד,לח לְהוֹרִ֗ישׁ גּוֹיִ֛ם גְּדֹלִ֧ים וַֽעֲצֻמִ֛ים מִמְּךָ֖ מִפָּנֶ֑יךָ לַהֲבִֽיאֲךָ֗ לָֽתֶת־לְךָ֧ אֶת־אַרְצָ֛ם נַֽחֲלָ֖ה כַּיּ֥וֹם הַזֶּֽה׃ לְתָרָכָא, עַמְמִין רַבְרְבִין וְתַקִּיפִין מִנָּךְ--מִן קֳדָמָךְ; לְאַעָלוּתָךְ, לְמִתַּן לָךְ יָת אֲרַעְהוֹן אַחְסָנָא--כְּיוֹמָא הָדֵין. 
ד,לט וְיָֽדַעְתָּ֣ הַיּ֗וֹם וַהֲשֵֽׁבֹתָ֮ אֶל־לְבָבֶךָ֒ כִּ֤י יְהוָה֙ ה֣וּא הָֽאֱלֹהִ֔ים בַּשָּׁמַ֣יִם מִמַּ֔עַל וְעַל־הָאָ֖רֶץ מִתָּ֑חַת אֵ֖ין עֽוֹד׃ וְתִדַּע יוֹמָא דֵּין, וְתָתִיב לְלִבָּךְ, אֲרֵי יְיָ הוּא אֱלֹהִים, דִּשְׁכִינְתֵיהּ בִּשְׁמַיָּא מִלְּעֵילָא וְשַׁלִּיט עַל אַרְעָא מִלְּרַע:  לֵית, עוֹד. 
ד,מ וְשָֽׁמַרְתָּ֞ אֶת־חֻקָּ֣יו וְאֶת־מִצְו‍ֹתָ֗יו אֲשֶׁ֨ר אָֽנֹכִ֤י מְצַוְּךָ֙ הַיּ֔וֹם אֲשֶׁר֙ יִיטַ֣ב לְךָ֔ וּלְבָנֶ֖יךָ אַֽחֲרֶ֑יךָ וּלְמַ֨עַן תַּֽאֲרִ֤יךְ יָמִים֙ עַל־הָ֣אֲדָמָ֔ה אֲשֶׁ֨ר יְהוָ֧ה אֱלֹהֶ֛יךָ נֹתֵ֥ן לְךָ֖ כָּל־הַיָּמִֽים׃ {פ} וְתִטַּר יָת קְיָמוֹהִי וְיָת פִּקּוֹדוֹהִי, דַּאֲנָא מְפַקֵּיד לָךְ יוֹמָא דֵּין, דְּיִיטַב לָךְ, וְלִבְנָךְ בָּתְרָךְ--וּבְדִיל דְּתוֹרֵיךְ יוֹמִין עַל אַרְעָא, דַּייָ אֱלָהָךְ יָהֵיב לָךְ כָּל יוֹמַיָּא. {פ}
ד,מא אָ֣ז יַבְדִּ֤יל מֹשֶׁה֙ שָׁלֹ֣שׁ עָרִ֔ים בְּעֵ֖בֶר הַיַּרְדֵּ֑ן מִזְרְחָ֖ה שָֽׁמֶשׁ׃ בְּכֵין אַפְרֵישׁ מֹשֶׁה תְּלָת קִרְוִין, בְּעִבְרָא דְּיַרְדְּנָא, מַדְנַח, שִׁמְשָׁא. 
ד,מב לָנֻ֨ס שָׁ֜מָּה רוֹצֵ֗חַ אֲשֶׁ֨ר יִרְצַ֤ח אֶת־רֵעֵ֨הוּ֙ בִּבְלִי־דַ֔עַת וְה֛וּא לֹֽא־שֹׂנֵ֥א ל֖וֹ מִתְּמֹ֣ל שִׁלְשֹׁ֑ם וְנָ֗ס אֶל־אַחַ֛ת מִן־הֶֽעָרִ֥ים הָאֵ֖ל וָחָֽי׃ לְמִעְרַק לְתַמָּן קָטוֹלָא, דְּיִקְטוֹל יָת חַבְרֵיהּ בְּלָא מַדְּעֵיהּ, וְהוּא לָא סָנֵי לֵיהּ, מֵאֶתְמָלִי וּמִדְּקַמּוֹהִי; וְיִעְרוֹק, לַחֲדָא מִן קִרְוַיָּא הָאִלֵּין--וְיִתְקַיַּם. 
ד,מג אֶת־בֶּ֧צֶר בַּמִּדְבָּ֛ר בְּאֶ֥רֶץ הַמִּישֹׁ֖ר לָרֽאוּבֵנִ֑י וְאֶת־רָאמֹ֤ת בַּגִּלְעָד֙ לַגָּדִ֔י וְאֶת־גּוֹלָ֥ן בַּבָּשָׁ֖ן לַֽמְנַשִּֽׁי׃ יָת בֶּצֶר בְּמַדְבְּרָא בַּאֲרַע מֵישְׁרָא, לְשֵׁיבַט רְאוּבֵן; וְיָת רָאמוֹת בַּגִּלְעָד לְשֵׁיבַט גָּד, וְיָת גּוֹלָן בְּמַתְנַן לְשִׁבְטָא דִּמְנַשֶּׁה. 
ד,מד וְזֹ֖את הַתּוֹרָ֑ה אֲשֶׁר־שָׂ֣ם מֹשֶׁ֔ה לִפְנֵ֖י בְּנֵ֥י יִשְׂרָאֵֽל׃ וְדָא, אוֹרָיְתָא, דִּסְדַר מֹשֶׁה, קֳדָם בְּנֵי יִשְׂרָאֵל. 
ד,מה אֵ֚לֶּה הָֽעֵדֹ֔ת וְהַֽחֻקִּ֖ים וְהַמִּשְׁפָּטִ֑ים אֲשֶׁ֨ר דִּבֶּ֤ר מֹשֶׁה֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל בְּצֵאתָ֖ם מִמִּצְרָֽיִם׃ אִלֵּין, סָהִדְוָתָא, וּקְיָמַיָּא, וְדִינַיָּא--דְּמַלֵּיל מֹשֶׁה עִם בְּנֵי יִשְׂרָאֵל, בְּמִפַּקְהוֹן מִמִּצְרָיִם. 
ד,מו בְּעֵ֨בֶר הַיַּרְדֵּ֜ן בַּגַּ֗יְא מ֚וּל בֵּ֣ית פְּע֔וֹר בְּאֶ֗רֶץ סִיחֹן֙ מֶ֣לֶךְ הָֽאֱמֹרִ֔י אֲשֶׁ֥ר יוֹשֵׁ֖ב בְּחֶשְׁבּ֑וֹן אֲשֶׁ֨ר הִכָּ֤ה מֹשֶׁה֙ וּבְנֵ֣י יִשְׂרָאֵ֔ל בְּצֵאתָ֖ם מִמִּצְרָֽיִם׃ בִּעִבְרָא דְּיַרְדְּנָא בְּחִלְּתָא, לָקֳבֵיל בֵּית פְּעוֹר, בַּאֲרַע סִיחוֹן מַלְכָּא אֱמוֹרָאָה, דְּיָתֵיב בְּחֶשְׁבּוֹן--דִּמְחָא מֹשֶׁה וּבְנֵי יִשְׂרָאֵל, בְּמִפַּקְהוֹן מִמִּצְרָיִם. 
ד,מז וַיִּֽירְשׁ֨וּ אֶת־אַרְצ֜וֹ וְאֶת־אֶ֣רֶץ ׀ ע֣וֹג מֶֽלֶךְ־הַבָּשָׁ֗ן שְׁנֵי֙ מַלְכֵ֣י הָֽאֱמֹרִ֔י אֲשֶׁ֖ר בְּעֵ֣בֶר הַיַּרְדֵּ֑ן מִזְרַ֖ח שָֽׁמֶשׁ׃ וִירִיתוּ יָת אַרְעֵיהּ וְיָת אֲרַע עוֹג מַלְכָּא דְּמַתְנַן, תְּרֵין מַלְכֵי אֶמוֹרָאָה, דִּבְעִבְרָא, דְּיַרְדְּנָא--מַדְנַח, שִׁמְשָׁא. 
ד,מח מֵֽעֲרֹעֵ֞ר אֲשֶׁ֨ר עַל־שְׂפַת־נַ֧חַל אַרְנֹ֛ן וְעַד־הַ֥ר שִׂיאֹ֖ן ה֥וּא חֶרְמֽוֹן׃ מֵעֲרוֹעֵר דְּעַל כֵּיף נַחְלָא דְּאַרְנוֹן, וְעַד טוּרָא דְּשִׂיאוֹן--הוּא חֶרְמוֹן. 
ד,מט וְכָל־הָ֨עֲרָבָ֜ה עֵ֤בֶר הַיַּרְדֵּן֙ מִזְרָ֔חָה וְעַ֖ד יָ֣ם הָֽעֲרָבָ֑ה תַּ֖חַת אַשְׁדֹּ֥ת הַפִּסְגָּֽה׃ {פ} וְכָל מֵישְׁרָא עִבְרָא דְּיַרְדְּנָא, מַדְנְחָא, וְעַד, יַמָּא דְּמֵישְׁרָא--תְּחוֹת, מַשְׁפַּךְ מֵי רָמְתָא. {פ}
ה,א וַיִּקְרָ֣א מֹשֶׁה֮ אֶל־כָּל־יִשְׂרָאֵל֒ וַיֹּ֣אמֶר אֲלֵהֶ֗ם שְׁמַ֤ע יִשְׂרָאֵל֙ אֶת־הַֽחֻקִּ֣ים וְאֶת־הַמִּשְׁפָּטִ֔ים אֲשֶׁ֧ר אָֽנֹכִ֛י דֹּבֵ֥ר בְּאָזְנֵיכֶ֖ם הַיּ֑וֹם וּלְמַדְתֶּ֣ם אֹתָ֔ם וּשְׁמַרְתֶּ֖ם לַֽעֲשֹׂתָֽם׃ וּקְרָא מֹשֶׁה, לְכָל יִשְׂרָאֵל, וַאֲמַר לְהוֹן שְׁמַע יִשְׂרָאֵל יָת קְיָמַיָּא וְיָת דִּינַיָּא, דַּאֲנָא מְמַלֵּיל קֳדָמֵיכוֹן יוֹמָא דֵּין; וְתֵילְפוּן יָתְהוֹן, וְתִטְּרוּן לְמֶעֱבַדְהוֹן. 
ה,ב יְהוָ֣ה אֱלֹהֵ֗ינוּ כָּרַ֥ת עִמָּ֛נוּ בְּרִ֖ית בְּחֹרֵֽב׃ יְיָ אֱלָהַנָא, גְּזַר עִמַּנָא קְיָם--בְּחוֹרֵב. 
ה,ג לֹ֣א אֶת־אֲבֹתֵ֔ינוּ כָּרַ֥ת יְהוָ֖ה אֶת־הַבְּרִ֣ית הַזֹּ֑את כִּ֣י אִתָּ֔נוּ אֲנַ֨חְנוּ אֵ֥לֶּה פֹ֛ה הַיּ֖וֹם כֻּלָּ֥נוּ חַיִּֽים׃ לָא עִם אֲבָהָתַנָא, גְּזַר יְיָ יָת קְיָמָא הָדֵין:  אֱלָהֵין עִמַּנָא, אֲנַחְנָא אִלֵּין כָּא יוֹמָא דֵּין כֻּלַּנָא קַיָּמִין. 
ה,ד פָּנִ֣ים ׀ בְּפָנִ֗ים דִּבֶּ֨ר יְהוָ֧ה עִמָּכֶ֛ם בָּהָ֖ר מִתּ֥וֹךְ הָאֵֽשׁ׃ מַמְלַל עִם מַמְלַל, מַלֵּיל יְיָ עִמְּכוֹן בְּטוּרָא--מִגּוֹ אִישָׁתָא. 
ה,ה אָֽ֠נֹכִי עֹמֵ֨ד בֵּין־יְהוָ֤ה וּבֵֽינֵיכֶם֙ בָּעֵ֣ת הַהִ֔וא לְהַגִּ֥יד לָכֶ֖ם אֶת־דְּבַ֣ר יְהוָ֑ה כִּ֤י יְרֵאתֶם֙ מִפְּנֵ֣י הָאֵ֔שׁ וְלֹֽא־עֲלִיתֶ֥ם בָּהָ֖ר לֵאמֹֽר׃ {ס} אֲנָא הֲוֵיתִי קָאֵים בֵּין מֵימְרָא דַּייָ וּבֵינֵיכוֹן, בְּעִדָּנָא הַהוּא, לְחַוָּאָה לְכוֹן, יָת פִּתְגָמָא דַּייָ:  אֲרֵי דְּחֵילְתּוּן מִן קֳדָם אִישָׁתָא, וְלָא סְלֵיקְתּוּן בְּטוּרָא לְמֵימַר. {ס}
ה,ו אָֽנֹכִי֙ יְהוָ֣ה אֱלֹהֶ֔יךָ אֲשֶׁ֧ר הֽוֹצֵאתִ֛יךָ מֵאֶ֥רֶץ מִצְרַ֖יִם מִבֵּ֣ית עֲבָדִ֑ים לֹֽא־יִהְיֶ֥ה לְךָ֛ אֱלֹהִ֥ים אֲחֵרִ֖ים עַל־פָּנָֽי׃ אֲנָא יְיָ אֱלָהָךְ, דְּאַפֵּיקְתָּךְ מֵאַרְעָא דְּמִצְרַיִם מִבֵּית עַבְדוּתָא:  לָא יִהְוֵי לָךְ אֱלָהּ אָחֳרָן, בָּר מִנִּי. 
ה,ז לֹֽא־תַעֲשֶׂ֨ה לְךָ֥ פֶ֨סֶל֙ כָּל־תְּמוּנָ֔ה אֲשֶׁ֤ר בַּשָּׁמַ֨יִם֙ מִמַּ֔עַל וַֽאֲשֶׁ֥ר בָּאָ֖רֶץ מִתָּ֑חַת וַֽאֲשֶׁ֥ר בַּמַּ֖יִם מִתַּ֥חַת לָאָֽרֶץ׃ לָא תַּעֲבֵיד לָךְ צֵילַם, כָּל דְּמוּ, דְּבִשְׁמַיָּא מִלְּעֵילָא, וְדִבְאַרְעָא מִלְּרַע--וְדִבְמַיָּא, מִלְּרַע לְאַרְעָא. 
ה,ח לֹֽא־תִשְׁתַּחֲוֶ֥ה לָהֶ֖ם וְלֹ֣א תָֽעָבְדֵ֑ם כִּ֣י אָֽנֹכִ֞י יְהוָ֤ה אֱלֹהֶ֨יךָ֙ אֵ֣ל קַנָּ֔א פֹּ֠קֵד עֲו֨‍ֹן אָב֧וֹת עַל־בָּנִ֛ים וְעַל־שִׁלֵּשִׁ֥ים וְעַל־רִבֵּעִ֖ים לְשֹֽׂנְאָֽי׃ לָא תִּסְגּוֹד לְהוֹן, וְלָא תִּפְלְחִנִּין:  אֲרֵי אֲנָא יְיָ אֱלָהָךְ, אֵל קַנָּא--מַסְעַר חוֹבֵי אֲבָהָן עַל בְּנִין מָרָדִין וְעַל דָּר תְּלִיתַאי וְעַל דָּר רְבִיעַאי, לְסָנְאָי כַּד מַשְׁלְמִין בְּנַיָּא לְמִחְטֵי בָּתַר אֲבָהָתְהוֹן. 
ה,ט וְעֹ֥שֶׂה חֶ֖סֶד לַֽאֲלָפִ֑ים לְאֹֽהֲבַ֖י וּלְשֹֽׁמְרֵ֥י מצותו (מִצְו‍ֹתָֽי)׃ {ס} וְעָבֵיד טֵיבוּ, לְאַלְפֵי דָּרִין--לְרָחֲמַי, וּלְנָטְרֵי פִּקּוֹדָי. {ס}
ה,י לֹ֥א תִשָּׂ֛א אֶת־שֵֽׁם־יְהוָ֥ה אֱלֹהֶ֖יךָ לַשָּׁ֑וְא כִּ֣י לֹ֤א יְנַקֶּה֙ יְהוָ֔ה אֵ֛ת אֲשֶׁר־יִשָּׂ֥א אֶת־שְׁמ֖וֹ לַשָּֽׁוְא׃ {ס} לָא תֵּימֵי בִּשְׁמָא דַּייָ אֱלָהָךְ, לְמַגָּנָא:  אֲרֵי לָא יְזַכֵּי יְיָ, יָת דְּיֵימֵי בִּשְׁמֵיהּ לְשִׁקְרָא. {ס}
ה,יא שָׁמ֛וֹר אֶת־י֥וֹם הַשַּׁבָּ֖ת לְקַדְּשׁ֑וֹ כַּֽאֲשֶׁ֥ר צִוְּךָ֖ יְהוָ֥ה אֱלֹהֶֽיךָ׃ טַר יָת יוֹמָא דְּשַׁבְּתָא, לְקַדָּשׁוּתֵיהּ, כְּמָא דְּפַקְּדָךְ, יְיָ אֱלָהָךְ. 
ה,יב שֵׁ֤שֶׁת יָמִים֙ תַּֽעֲבֹ֔ד וְעָשִׂ֖יתָ כָּל־מְלַאכְתֶּֽךָ׃ שִׁתָּא יוֹמִין תִּפְלַח, וְתַעֲבֵיד כָּל עֲבִידְתָךְ. 
ה,יג וְיוֹם֙ הַשְּׁבִיעִ֔י שַׁבָּ֖ת לַֽיהוָ֣ה אֱלֹהֶ֑יךָ לֹ֣א תַֽעֲשֶׂ֣ה כָל־מְלָאכָ֡ה אַתָּ֣ה וּבִנְךָֽ־וּבִתֶּ֣ךָ וְעַבְדְּךָֽ־וַ֠אֲמָתֶךָ וְשֽׁוֹרְךָ֨ וַחֲמֹֽרְךָ֜ וְכָל־בְּהֶמְתֶּ֗ךָ וְגֵֽרְךָ֙ אֲשֶׁ֣ר בִּשְׁעָרֶ֔יךָ לְמַ֗עַן יָנ֛וּחַ עַבְדְּךָ֥ וַאֲמָֽתְךָ֖ כָּמֽוֹךָ׃ וְיוֹמָא, שְׁבִיעָאָה--שַׁבְּתָא, קֳדָם יְיָ אֱלָהָךְ:  לָא תַּעֲבֵיד כָּל עֲבִידָא אַתְּ וּבְרָךְ וּבְרַתָּךְ וְעַבְדָּךְ וְאַמְתָּךְ וְתוֹרָךְ וּחְמָרָךְ וְכָל בְּעִירָךְ, וְגִיּוֹרָךְ דִּבְקִרְוָךְ--בְּדִיל דִּינוּחַ עַבְדָּךְ וְאַמְתָּךְ, כְּוָתָךְ. 
ה,יד וְזָֽכַרְתָּ֗ כִּ֣י עֶ֤בֶד הָיִ֨יתָ֙ בְּאֶ֣רֶץ מִצְרַ֔יִם וַיֹּצִ֨אֲךָ֜ יְהוָ֤ה אֱלֹהֶ֨יךָ֙ מִשָּׁ֔ם בְּיָ֥ד חֲזָקָ֖ה וּבִזְרֹ֣עַ נְטוּיָ֑ה עַל־כֵּ֗ן צִוְּךָ֙ יְהוָ֣ה אֱלֹהֶ֔יךָ לַֽעֲשׂ֖וֹת אֶת־י֥וֹם הַשַּׁבָּֽת׃ {ס} וְתִדְכַר, אֲרֵי עַבְדָּא הֲוֵיתָא בְּאַרְעָא דְּמִצְרַיִם, וְאַפְּקָךְ יְיָ אֱלָהָךְ מִתַּמָּן, בְּיַד תַּקִּיפָא וּבִדְרָע מְרָמַם; עַל כֵּין, פַּקְּדָךְ יְיָ אֱלָהָךְ, לְמֶעֱבַד, יָת יוֹמָא דְּשַׁבְּתָא. {ס}
ה,טו כַּבֵּ֤ד אֶת־אָבִ֨יךָ֙ וְאֶת־אִמֶּ֔ךָ כַּֽאֲשֶׁ֥ר צִוְּךָ֖ יְהוָ֣ה אֱלֹהֶ֑יךָ לְמַ֣עַן ׀ יַֽאֲרִיכֻ֣ן יָמֶ֗יךָ וּלְמַ֨עַן֙ יִ֣יטַב לָ֔ךְ עַ֚ל הָֽאֲדָמָ֔ה אֲשֶׁר־יְהוָ֥ה אֱלֹהֶ֖יךָ נֹתֵ֥ן לָֽךְ׃ {ס} יַקַּר יָת אֲבוּךְ וְיָת אִמָּךְ, כְּמָא דְּפַקְּדָךְ יְיָ אֱלָהָךְ--בְּדִיל דְּיֵירְכוּן יוֹמָךְ, וּבְדִיל דְּיִיטַב לָךְ, עַל אַרְעָא, דַּייָ אֱלָהָךְ יָהֵיב לָךְ. {ס}
ה,טז לֹ֥א תִרְצָ֖ח {ס} וְלֹ֣א תִנְאָ֑ף {ס} וְלֹ֣א תִגְנֹ֔ב {ס} וְלֹֽא־תַעֲנֶ֥ה בְרֵֽעֲךָ֖ עֵ֥ד שָֽׁוְא׃ {ס} לָא תִּקְטוּל נְפַשׁ,  {ס}  וְלָא תְּגוּף;  {ס}  וְלָא תִּגְנוּב,  {ס}  וְלָא תַּסְהֵיד בְּחַבְרָךְ סָהֲדוּתָא דְּשִׁקְרָא. {ס}
ה,יז וְלֹ֥א תַחְמֹ֖ד אֵ֣שֶׁת רֵעֶ֑ךָ {ס} וְלֹ֨א תִתְאַוֶּ֜ה בֵּ֣ית רֵעֶ֗ךָ שָׂדֵ֜הוּ וְעַבְדּ֤וֹ וַֽאֲמָתוֹ֙ שׁוֹר֣וֹ וַֽחֲמֹר֔וֹ וְכֹ֖ל אֲשֶׁ֥ר לְרֵעֶֽךָ׃ {ס} וְלָא תַּחְמֵיד, אִתַּת חַבְרָךְ;  {ס}  וְלָא תִּרּוֹג בֵּית חַבְרָךְ, חַקְלֵיהּ וְעַבְדֵּיהּ וְאַמְתֵּיהּ תּוֹרֵיהּ וּחְמָרֵיהּ, וְכֹל, דִּלְחַבְרָךְ. {ס}
ה,יח אֶֽת־הַדְּבָרִ֣ים הָאֵ֡לֶּה דִּבֶּר֩ יְהוָ֨ה אֶל־כָּל־קְהַלְכֶ֜ם בָּהָ֗ר מִתּ֤וֹךְ הָאֵשׁ֙ הֶֽעָנָ֣ן וְהָֽעֲרָפֶ֔ל ק֥וֹל גָּד֖וֹל וְלֹ֣א יָסָ֑ף וַֽיִּכְתְּבֵ֗ם עַל־שְׁנֵי֙ לֻחֹ֣ת אֲבָנִ֔ים וַֽיִּתְּנֵ֖ם אֵלָֽי׃ יָת פִּתְגָמַיָּא הָאִלֵּין מַלֵּיל יְיָ עִם כָּל קְהַלְכוֹן בְּטוּרָא, מִגּוֹ אִישָׁתָא עֲנָנָא וַאֲמִטְּתָא--קָל רַב, וְלָא פָּסִיק; וּכְתַבִנּוּן, עַל תְּרֵין לוּחֵי אַבְנַיָּא, וִיהַבִנּוּן, לִי. 
ה,יט וַיְהִ֗י כְּשָׁמְעֲכֶ֤ם אֶת־הַקּוֹל֙ מִתּ֣וֹךְ הַחֹ֔שֶׁךְ וְהָהָ֖ר בֹּעֵ֣ר בָּאֵ֑שׁ וַתִּקְרְב֣וּן אֵלַ֔י כָּל־רָאשֵׁ֥י שִׁבְטֵיכֶ֖ם וְזִקְנֵיכֶֽם׃ וַהֲוָה, כְּמִשְׁמַעְכוֹן יָת קָלָא מִגּוֹ חֲשׁוֹכָא, וְטוּרָא, בָּעֵר בְּאִישָׁתָא; וּקְרֵיבְתּוּן לְוָתִי, כָּל רֵישֵׁי שִׁבְטֵיכוֹן וְסָבֵיכוֹן. 
ה,כ וַתֹּֽאמְר֗וּ הֵ֣ן הֶרְאָ֜נוּ יְהוָ֤ה אֱלֹהֵ֨ינוּ֙ אֶת־כְּבֹד֣וֹ וְאֶת־גָּדְל֔וֹ וְאֶת־קֹל֥וֹ שָׁמַ֖עְנוּ מִתּ֣וֹךְ הָאֵ֑שׁ הַיּ֤וֹם הַזֶּה֙ רָאִ֔ינוּ כִּֽי־יְדַבֵּ֧ר אֱלֹהִ֛ים אֶת־הָֽאָדָ֖ם וָחָֽי׃ וַאֲמַרְתּוּן, הָא אַחְזְיַנָא יְיָ אֱלָהַנָא יָת יְקָרֵיהּ וְיָת רְבוּתֵיהּ, וְיָת קָל מֵימְרֵיהּ שְׁמַעְנָא, מִגּוֹ אִישָׁתָא; יוֹמָא הָדֵין חֲזֵינָא, אֲרֵי מְמַלֵּיל יְיָ עִם אֲנָשָׁא וּמִתְקַיַּם. 
ה,כא וְעַתָּה֙ לָ֣מָּה נָמ֔וּת כִּ֣י תֹֽאכְלֵ֔נוּ הָאֵ֥שׁ הַגְּדֹלָ֖ה הַזֹּ֑את אִם־יֹֽסְפִ֣ים ׀ אֲנַ֗חְנוּ לִ֠שְׁמֹעַ אֶת־ק֨וֹל יְהוָ֧ה אֱלֹהֵ֛ינוּ ע֖וֹד וָמָֽתְנוּ׃ וּכְעַן, לְמָא נְמוּת, אֲרֵי תֵּיכְלִנַּנָא, אִישָׁתָא רַבְּתָא הָדָא; אִם מוֹסְפִין אֲנַחְנָא, לְמִשְׁמַע יָת קָל מֵימְרָא דַּייָ אֱלָהַנָא עוֹד--מָיְתִין אֲנַחְנָא. 
ה,כב כִּ֣י מִ֣י כָל־בָּשָׂ֡ר אֲשֶׁ֣ר שָׁמַ֣ע קוֹל֩ אֱלֹהִ֨ים חַיִּ֜ים מְדַבֵּ֧ר מִתּוֹךְ־הָאֵ֛שׁ כָּמֹ֖נוּ וַיֶּֽחִי׃ אֲרֵי מַאן כָּל בִּסְרָא דִּשְׁמַע קָל מֵימְרָא דַּייָ קַיָּמָא מְמַלֵּיל מִגּוֹ אִישָׁתָא, כְּוָתַנָא--וְאִתְקַיַּם. 
ה,כג קְרַ֤ב אַתָּה֙ וּֽשְׁמָ֔ע אֵ֛ת כָּל־אֲשֶׁ֥ר יֹאמַ֖ר יְהוָ֣ה אֱלֹהֵ֑ינוּ וְאַ֣תְּ ׀ תְּדַבֵּ֣ר אֵלֵ֗ינוּ אֵת֩ כָּל־אֲשֶׁ֨ר יְדַבֵּ֜ר יְהוָ֧ה אֱלֹהֵ֛ינוּ אֵלֶ֖יךָ וְשָׁמַ֥עְנוּ וְעָשִֽׂינוּ׃ קְרַב אַתְּ וּשְׁמַע, יָת כָּל דְּיֵימַר יְיָ אֱלָהַנָא; וְאַתְּ תְּמַלֵּיל עִמַּנָא, יָת כָּל דִּימַלֵּיל יְיָ אֱלָהַנָא עִמָּךְ--וּנְקַבֵּיל וְנַעֲבֵיד. 
ה,כד וַיִּשְׁמַ֤ע יְהוָה֙ אֶת־ק֣וֹל דִּבְרֵיכֶ֔ם בְּדַבֶּרְכֶ֖ם אֵלָ֑י וַיֹּ֨אמֶר יְהוָ֜ה אֵלַ֗י שָׁ֠מַעְתִּי אֶת־ק֨וֹל דִּבְרֵ֜י הָעָ֤ם הַזֶּה֙ אֲשֶׁ֣ר דִּבְּר֣וּ אֵלֶ֔יךָ הֵיטִ֖יבוּ כָּל־אֲשֶׁ֥ר דִּבֵּֽרוּ׃ וּשְׁמִיעַ קֳדָם יְיָ יָת קָל פִּתְגָמֵיכוֹן, בְּמַלָּלוּתְכוֹן עִמִּי; וַאֲמַר יְיָ לִי, שְׁמִיעַ קֳדָמַי יָת קָל פִּתְגָמֵי עַמָּא הָדֵין דְּמַלִּילוּ עִמָּךְ--אַתְקִינוּ, כָּל דְּמַלִּילוּ. 
ה,כה מִֽי־יִתֵּ֡ן וְהָיָה֩ לְבָבָ֨ם זֶ֜ה לָהֶ֗ם לְיִרְאָ֥ה אֹתִ֛י וְלִשְׁמֹ֥ר אֶת־כָּל־מִצְו‍ֹתַ֖י כָּל־הַיָּמִ֑ים לְמַ֨עַן יִיטַ֥ב לָהֶ֛ם וְלִבְנֵיהֶ֖ם לְעֹלָֽם׃ לְוֵי דִּיהֵי לִבָּא הָדֵין לְהוֹן, לְמִדְחַל קֳדָמַי וּלְמִטַּר יָת כָּל פִּקּוֹדַי--כָּל יוֹמַיָּא:  בְּדִיל דְּיִיטַב לְהוֹן וְלִבְנֵיהוֹן, לְעָלַם. 
ה,כו לֵ֖ךְ אֱמֹ֣ר לָהֶ֑ם שׁ֥וּבוּ לָכֶ֖ם לְאָֽהֳלֵיכֶֽם׃ אִיזֵיל, אֵימַר לְהוֹן:  תּוּבוּ לְכוֹן, לְמַשְׁכְּנֵיכוֹן. 
ה,כז וְאַתָּ֗ה פֹּה֮ עֲמֹ֣ד עִמָּדִי֒ וַֽאֲדַבְּרָ֣ה אֵלֶ֗יךָ אֵ֧ת כָּל־הַמִּצְוָ֛ה וְהַֽחֻקִּ֥ים וְהַמִּשְׁפָּטִ֖ים אֲשֶׁ֣ר תְּלַמְּדֵ֑ם וְעָשׂ֣וּ בָאָ֔רֶץ אֲשֶׁ֧ר אָֽנֹכִ֛י נֹתֵ֥ן לָהֶ֖ם לְרִשְׁתָּֽהּ׃ וְאַתְּ, הָכָא קוּם קֳדָמַי, וַאֲמַלֵּיל עִמָּךְ יָת כָּל תַּפְקֵידְתָּא וּקְיָמַיָּא וְדִינַיָּא, דְּתַלֵּיפִנּוּן; וְיַעְבְּדוּן בְּאַרְעָא, דַּאֲנָא יָהֵיב לְהוֹן לְמֵירְתַהּ. 
ה,כח וּשְׁמַרְתֶּ֣ם לַֽעֲשׂ֔וֹת כַּֽאֲשֶׁ֥ר צִוָּ֛ה יְהוָ֥ה אֱלֹֽהֵיכֶ֖ם אֶתְכֶ֑ם לֹ֥א תָסֻ֖רוּ יָמִ֥ין וּשְׂמֹֽאל׃ וְתִטְּרוּן לְמֶעֱבַד, כְּמָא דְּפַקֵּיד יְיָ אֱלָהֲכוֹן יָתְכוֹן:  לָא תִּסְטוֹן, לְיַמִּינָא וְלִסְמָאלָא. 
ה,כט בְּכָל־הַדֶּ֗רֶךְ אֲשֶׁ֨ר צִוָּ֜ה יְהוָ֧ה אֱלֹֽהֵיכֶ֛ם אֶתְכֶ֖ם תֵּלֵ֑כוּ לְמַ֤עַן תִּֽחְיוּן֙ וְט֣וֹב לָכֶ֔ם וְהַֽאֲרַכְתֶּ֣ם יָמִ֔ים בָּאָ֖רֶץ אֲשֶׁ֥ר תִּֽירָשֽׁוּן׃ בְּכָל אוֹרְחָא, דְּפַקֵּיד יְיָ אֱלָהֲכוֹן יָתְכוֹן--תְּהָכוּן:  בְּדִיל דְּתֵיחוֹן, וְיִיטַב לְכוֹן, וְתֵירְכוּן יוֹמִין, בְּאַרְעָא דְּתֵירְתוּן. 
ו,א וְזֹ֣את הַמִּצְוָ֗ה הַֽחֻקִּים֙ וְהַמִּשְׁפָּטִ֔ים אֲשֶׁ֥ר צִוָּ֛ה יְהוָ֥ה אֱלֹֽהֵיכֶ֖ם לְלַמֵּ֣ד אֶתְכֶ֑ם לַֽעֲשׂ֣וֹת בָּאָ֔רֶץ אֲשֶׁ֥ר אַתֶּ֛ם עֹֽבְרִ֥ים שָׁ֖מָּה לְרִשְׁתָּֽהּ׃ וְדָא תַּפְקֵידְתָּא, קְיָמַיָּא וְדִינַיָּא, דְּפַקֵּיד יְיָ אֱלָהֲכוֹן, לְאַלָּפָא יָתְכוֹן--לְמֶעֱבַד בְּאַרְעָא, דְּאַתּוּן עָבְרִין לְתַמָּן לְמֵירְתַהּ. 
ו,ב לְמַ֨עַן תִּירָ֜א אֶת־יְהוָ֣ה אֱלֹהֶ֗יךָ לִ֠שְׁמֹר אֶת־כָּל־חֻקֹּתָ֣יו וּמִצְו‍ֹתָיו֮ אֲשֶׁ֣ר אָֽנֹכִ֣י מְצַוֶּךָ֒ אַתָּה֙ וּבִנְךָ֣ וּבֶן־בִּנְךָ֔ כֹּ֖ל יְמֵ֣י חַיֶּ֑יךָ וּלְמַ֖עַן יַֽאֲרִכֻ֥ן יָמֶֽיךָ׃ בְּדִיל דְּתִדְחַל קֳדָם יְיָ אֱלָהָךְ, לְמִטַּר יָת כָּל קְיָמוֹהִי וּפִקּוֹדוֹהִי דַּאֲנָא מְפַקֵּיד לָךְ, אַתְּ וּבְרָךְ וּבַר בְּרָךְ, כֹּל יוֹמֵי חַיָּךְ--וּבְדִיל, דְּיֵירְכוּן יוֹמָךְ. 
ו,ג וְשָֽׁמַעְתָּ֤ יִשְׂרָאֵל֙ וְשָֽׁמַרְתָּ֣ לַֽעֲשׂ֔וֹת אֲשֶׁר֙ יִיטַ֣ב לְךָ֔ וַֽאֲשֶׁ֥ר תִּרְבּ֖וּן מְאֹ֑ד כַּֽאֲשֶׁר֩ דִּבֶּ֨ר יְהוָ֜ה אֱלֹהֵ֤י אֲבֹתֶ֨יךָ֙ לָ֔ךְ אֶ֛רֶץ זָבַ֥ת חָלָ֖ב וּדְבָֽשׁ׃ {פ} וּתְקַבֵּיל יִשְׂרָאֵל, וְתִטַּר לְמֶעֱבַד, דְּיִיטַב לָךְ, וּדְתִסְגוֹן לַחְדָּא:  כְּמָא דְּמַלֵּיל יְיָ אֱלָהָא דַּאֲבָהָתָךְ, לָךְ--אֲרַע עָבְדָא חֲלָב, וּדְבַשׁ. {פ}
ו,ד שְׁמַ֖ע יִשְׂרָאֵ֑ל יְהוָ֥ה אֱלֹהֵ֖ינוּ יְהוָ֥ה ׀ אֶחָֽד׃ שְׁמַע, יִשְׂרָאֵל:  יְיָ אֱלָהַנָא, יְיָ חַד. 
ו,ה וְאָ֣הַבְתָּ֔ אֵ֖ת יְהוָ֣ה אֱלֹהֶ֑יךָ בְּכָל־לְבָֽבְךָ֥ וּבְכָל־נַפְשְׁךָ֖ וּבְכָל־מְאֹדֶֽךָ׃ וְתִרְחַם, יָת יְיָ אֱלָהָךְ, בְּכָל לִבָּךְ וּבְכָל נַפְשָׁךְ, וּבְכָל נִכְסָךְ. 
ו,ו וְהָי֞וּ הַדְּבָרִ֣ים הָאֵ֗לֶּה אֲשֶׁ֨ר אָֽנֹכִ֧י מְצַוְּךָ֛ הַיּ֖וֹם עַל־לְבָבֶֽךָ׃ וִיהוֹן פִּתְגָמַיָּא הָאִלֵּין, דַּאֲנָא מְפַקֵּיד לָךְ יוֹמָא דֵּין--עַל לִבָּךְ. 
ו,ז וְשִׁנַּנְתָּ֣ם לְבָנֶ֔יךָ וְדִבַּרְתָּ֖ בָּ֑ם בְּשִׁבְתְּךָ֤ בְּבֵיתֶ֨ךָ֙ וּבְלֶכְתְּךָ֣ בַדֶּ֔רֶךְ וּֽבְשָׁכְבְּךָ֖ וּבְקוּמֶֽךָ׃ וּתְתַנֵּינוּן לִבְנָךְ, וּתְמַלֵּיל בְּהוֹן, בְּמִתְּבָךְ בְּבֵיתָךְ וּבִמְהָכָךְ בְּאוֹרְחָא, וּבְמִשְׁכְּבָךְ וּבִמְקִימָךְ. 
ו,ח וּקְשַׁרְתָּ֥ם לְא֖וֹת עַל־יָדֶ֑ךָ וְהָי֥וּ לְטֹֽטָפֹ֖ת בֵּ֥ין עֵינֶֽיךָ׃ וְתִקְטְרִנּוּן לְאָת, עַל יְדָךְ; וִיהוֹן לִתְפִלִּין, בֵּין עֵינָךְ. 
ו,ט וּכְתַבְתָּ֛ם עַל־מְזֻז֥וֹת בֵּיתֶ֖ךָ וּבִשְׁעָרֶֽיךָ׃ {ס} וְתִכְתּוֹבִנּוּן עַל מְזוֹזְיָן וְתִקְבְּעִנּוּן בְּסִפֵּי בֵּיתָךְ, וּבְתַרְעָךְ. {ס}
ו,י וְהָיָ֞ה כִּ֥י יְבִֽיאֲךָ֣ ׀ יְהוָ֣ה אֱלֹהֶ֗יךָ אֶל־הָאָ֜רֶץ אֲשֶׁ֨ר נִשְׁבַּ֧ע לַֽאֲבֹתֶ֛יךָ לְאַבְרָהָ֛ם לְיִצְחָ֥ק וּֽלְיַעֲקֹ֖ב לָ֣תֶת לָ֑ךְ עָרִ֛ים גְּדֹלֹ֥ת וְטֹבֹ֖ת אֲשֶׁ֥ר לֹֽא־בָנִֽיתָ׃ וִיהֵי אֲרֵי יַעֵילִנָּךְ יְיָ אֱלָהָךְ, לְאַרְעָא דְּקַיֵּים לַאֲבָהָתָךְ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקוֹב--לְמִתַּן לָךְ:  קִרְוִין רַבְרְבָן וְטָבָן, דְּלָא בְּנֵיתָא. 
ו,יא וּבָ֨תִּ֜ים מְלֵאִ֣ים כָּל־טוּב֮ אֲשֶׁ֣ר לֹֽא־מִלֵּאתָ֒ וּבֹרֹ֤ת חֲצוּבִים֙ אֲשֶׁ֣ר לֹֽא־חָצַ֔בְתָּ כְּרָמִ֥ים וְזֵיתִ֖ים אֲשֶׁ֣ר לֹֽא־נָטָ֑עְתָּ וְאָֽכַלְתָּ֖ וְשָׂבָֽעְתָּ׃ וּבָתִּין מְלַן כָּל טוּב, דְּלָא מְלֵיתָא, וְגֻבִּין פְּסִילָן דְּלָא פְּסַלְתָּא, כַּרְמִין וְזֵיתִין דְּלָא נְצַבְתָּא; וְתֵיכוֹל, וְתִסְבַּע. 
ו,יב הִשָּׁ֣מֶר לְךָ֔ פֶּן־תִּשְׁכַּ֖ח אֶת־יְהוָ֑ה אֲשֶׁ֧ר הוֹצִֽיאֲךָ֛ מֵאֶ֥רֶץ מִצְרַ֖יִם מִבֵּ֥ית עֲבָדִֽים׃ אִסְתְּמַר לָךְ, דִּלְמָא תִּתְנְשֵׁי יָת דַּחְלְתָא דַּייָ, דְּאַפְּקָךְ מֵאַרְעָא דְּמִצְרַיִם, מִבֵּית עַבְדּוּתָא. 
ו,יג אֶת־יְהוָ֧ה אֱלֹהֶ֛יךָ תִּירָ֖א וְאֹת֣וֹ תַֽעֲבֹ֑ד וּבִשְׁמ֖וֹ תִּשָּׁבֵֽעַ׃ יָת יְיָ אֱלָהָךְ תִּדְחַל, וּקְדָמוֹהִי תִּפְלַח; וּבִשְׁמֵיהּ, תְּקַיֵּים. 
ו,יד לֹ֣א תֵֽלְכ֔וּן אַֽחֲרֵ֖י אֱלֹהִ֣ים אֲחֵרִ֑ים מֵֽאֱלֹהֵי֙ הָֽעַמִּ֔ים אֲשֶׁ֖ר סְבִיבֽוֹתֵיכֶֽם׃ לָא תְּהָכוּן, בָּתַר טָעֲוָת עַמְמַיָּא:  מִטָּעֲוָת, עַמְמַיָּא--דִּבְסַחְרָנֵיכוֹן. 
ו,טו כִּ֣י אֵ֥ל קַנָּ֛א יְהוָ֥ה אֱלֹהֶ֖יךָ בְּקִרְבֶּ֑ךָ פֶּן־יֶֽ֠חֱרֶה אַף־יְהוָ֤ה אֱלֹהֶ֨יךָ֙ בָּ֔ךְ וְהִשְׁמִ֣ידְךָ֔ מֵעַ֖ל פְּנֵ֥י הָֽאֲדָמָֽה׃ {ס} אֲרֵי אֵל קַנָּא יְיָ אֱלָהָךְ, שְׁכִינְתֵיהּ בֵּינָךְ:  דִּלְמָא יִתְקַף רֻגְזָא דַּייָ אֱלָהָךְ, בָּךְ, וִישֵׁיצֵינָךְ, מֵעַל אַפֵּי אַרְעָא. {ס}
ו,טז לֹ֣א תְנַסּ֔וּ אֶת־יְהוָ֖ה אֱלֹֽהֵיכֶ֑ם כַּֽאֲשֶׁ֥ר נִסִּיתֶ֖ם בַּמַּסָּֽה׃ לָא תְּנַסּוֹן, קֳדָם יְיָ אֱלָהֲכוֹן, כְּמָא דְּנַסֵּיתוֹן, בְּנִסֵּיתָא. 
ו,יז שָׁמ֣וֹר תִּשְׁמְר֔וּן אֶת־מִצְו֖‍ֹת יְהוָ֣ה אֱלֹֽהֵיכֶ֑ם וְעֵֽדֹתָ֥יו וְחֻקָּ֖יו אֲשֶׁ֥ר צִוָּֽךְ׃ מִטָּר תִּטְּרוּן, יָת פִּקּוֹדַיָּא דַּייָ אֱלָהֲכוֹן, וְסָהִדְוָתֵיהּ וּקְיָמוֹהִי, דְּפַקְּדָךְ. 
ו,יח וְעָשִׂ֛יתָ הַיָּשָׁ֥ר וְהַטּ֖וֹב בְּעֵינֵ֣י יְהוָ֑ה לְמַ֨עַן֙ יִ֣יטַב לָ֔ךְ וּבָ֗אתָ וְיָֽרַשְׁתָּ֙ אֶת־הָאָ֣רֶץ הַטֹּבָ֔ה אֲשֶׁר־נִשְׁבַּ֥ע יְהוָ֖ה לַֽאֲבֹתֶֽיךָ׃ וְתַעֲבֵיד דְּכָשַׁר וּדְתָקֵין, קֳדָם יְיָ--בְּדִיל, דְּיִיטַב לָךְ, וְתֵיעוֹל וְתֵירַת יָת אַרְעָא טָבְתָא, דְּקַיֵּים יְיָ לַאֲבָהָתָךְ. 
ו,יט לַֽהֲדֹ֥ף אֶת־כָּל־אֹֽיְבֶ֖יךָ מִפָּנֶ֑יךָ כַּֽאֲשֶׁ֖ר דִּבֶּ֥ר יְהוָֽה׃ {ס} לְמִתְבַּר יָת כָּל בַּעֲלֵי דְּבָבָךְ, מִן קֳדָמָךְ--כְּמָא דְּמַלֵּיל יְיָ. {ס}
ו,כ כִּֽי־יִשְׁאָלְךָ֥ בִנְךָ֛ מָחָ֖ר לֵאמֹ֑ר מָ֣ה הָֽעֵדֹ֗ת וְהַֽחֻקִּים֙ וְהַמִּשְׁפָּטִ֔ים אֲשֶׁ֥ר צִוָּ֛ה יְהוָ֥ה אֱלֹהֵ֖ינוּ אֶתְכֶֽם׃ אֲרֵי יִשְׁאֲלִנָּךְ בְּרָךְ מְחַר, לְמֵימַר:  מָא סָהִדְוָתָא, וּקְיָמַיָּא וְדִינַיָּא, דְּפַקֵּיד יְיָ אֱלָהֲנָא, יָתְכוֹן. 
ו,כא וְאָֽמַרְתָּ֣ לְבִנְךָ֔ עֲבָדִ֛ים הָיִ֥ינוּ לְפַרְעֹ֖ה בְּמִצְרָ֑יִם וַיֹּֽצִיאֵ֧נוּ יְהוָ֛ה מִמִּצְרַ֖יִם בְּיָ֥ד חֲזָקָֽה׃ וְתֵימַר לִבְרָךְ, עַבְדִּין הֲוֵינָא לְפַרְעֹה בְּמִצְרָיִם; וְאַפְּקַנָא יְיָ מִמִּצְרַיִם, בְּיַד תַּקִּיפָא. 
ו,כב וַיִּתֵּ֣ן יְהוָ֡ה אוֹתֹ֣ת וּ֠מֹֽפְתִים גְּדֹלִ֨ים וְרָעִ֧ים ׀ בְּמִצְרַ֛יִם בְּפַרְעֹ֥ה וּבְכָל־בֵּית֖וֹ לְעֵינֵֽינוּ׃ וִיהַב יְיָ אָתִין וּמוֹפְתִין רַבְרְבִין וּבִישִׁין בְּמִצְרַיִם, בְּפַרְעֹה וּבְכָל אֲנָשׁ בֵּיתֵיהּ--לְעֵינַנָא. 
ו,כג וְאוֹתָ֖נוּ הוֹצִ֣יא מִשָּׁ֑ם לְמַ֨עַן֙ הָבִ֣יא אֹתָ֔נוּ לָ֤תֶת לָ֨נוּ֙ אֶת־הָאָ֔רֶץ אֲשֶׁ֥ר נִשְׁבַּ֖ע לַֽאֲבֹתֵֽינוּ׃ וְיָתַנָא, אַפֵּיק מִתַּמָּן--בְּדִיל, לְאַעָלָא יָתַנָא, לְמִתַּן לַנָא יָת אַרְעָא, דְּקַיֵּים לַאֲבָהָתַנָא. 
ו,כד וַיְצַוֵּ֣נוּ יְהוָ֗ה לַֽעֲשׂוֹת֙ אֶת־כָּל־הַֽחֻקִּ֣ים הָאֵ֔לֶּה לְיִרְאָ֖ה אֶת־יְהוָ֣ה אֱלֹהֵ֑ינוּ לְט֥וֹב לָ֨נוּ֙ כָּל־הַיָּמִ֔ים לְחַיֹּתֵ֖נוּ כְּהַיּ֥וֹם הַזֶּֽה׃ וּפַקְּדַנָא יְיָ, לְמֶעֱבַד יָת כָּל קְיָמַיָּא הָאִלֵּין, לְמִדְחַל, קֳדָם יְיָ אֱלָהַנָא--דְּיִיטַב לַנָא כָּל יוֹמַיָּא, לְקַיָּמוּתַנָא כְּיוֹמָא הָדֵין. 
ו,כה וּצְדָקָ֖ה תִּֽהְיֶה־לָּ֑נוּ כִּֽי־נִשְׁמֹ֨ר לַֽעֲשׂ֜וֹת אֶת־כָּל־הַמִּצְוָ֣ה הַזֹּ֗את לִפְנֵ֛י יְהוָ֥ה אֱלֹהֵ֖ינוּ כַּֽאֲשֶׁ֥ר צִוָּֽנוּ׃ {ס} וְזָכוּתָא, תְּהֵי לַנָא:  אֲרֵי נִטַּר לְמֶעֱבַד יָת כָּל תַּפְקֵידְתָּא הָדָא, קֳדָם יְיָ אֱלָהַנָא--כְּמָא דְּפַקְּדַנָא. {ס}
ז,א כִּ֤י יְבִֽיאֲךָ֙ יְהוָ֣ה אֱלֹהֶ֔יךָ אֶל־הָאָ֕רֶץ אֲשֶׁר־אַתָּ֥ה בָא־שָׁ֖מָּה לְרִשְׁתָּ֑הּ וְנָשַׁ֣ל גּֽוֹיִם־רַבִּ֣ים ׀ מִפָּנֶ֡יךָ הַֽחִתִּי֩ וְהַגִּרְגָּשִׁ֨י וְהָֽאֱמֹרִ֜י וְהַכְּנַֽעֲנִ֣י וְהַפְּרִזִּ֗י וְהַֽחִוִּי֙ וְהַיְבוּסִ֔י שִׁבְעָ֣ה גוֹיִ֔ם רַבִּ֥ים וַֽעֲצוּמִ֖ים מִמֶּֽךָּ׃ אֲרֵי יַעֵילִנָּךְ, יְיָ אֱלָהָךְ, לְאַרְעָא, דְּאַתְּ עָלֵיל לְתַמָּן לְמֵירְתַהּ; וִיתָרֵיךְ עַמְמִין סַגִּיאִין מִן קֳדָמָךְ חִתָּאֵי וְגַרְגִּישָׁאֵי וֶאֱמוֹרָאֵי וּכְנַעֲנָאֵי וּפְרִזָּאֵי, וְחִוָּאֵי וִיבוּסָאֵי--שִׁבְעָא עַמְמִין, סַגִּיאִין וְתַקִּיפִין מִנָּךְ. 
ז,ב וּנְתָנָ֞ם יְהוָ֧ה אֱלֹהֶ֛יךָ לְפָנֶ֖יךָ וְהִכִּיתָ֑ם הַֽחֲרֵ֤ם תַּֽחֲרִים֙ אֹתָ֔ם לֹֽא־תִכְרֹ֥ת לָהֶ֛ם בְּרִ֖ית וְלֹ֥א תְחָנֵּֽם׃ וְיִמְסְרִנּוּן יְיָ אֱלָהָךְ, קֳדָמָךְ--וְתִמְחֵינוּן:  גַּמָּרָא תְּגַּמַּר יָתְהוֹן, לָא תִּגְזַר לְהוֹן קְיָם וְלָא תְּרַחֵים עֲלֵיהוֹן. 
ז,ג וְלֹ֥א תִתְחַתֵּ֖ן בָּ֑ם בִּתְּךָ֙ לֹֽא־תִתֵּ֣ן לִבְנ֔וֹ וּבִתּ֖וֹ לֹֽא־תִקַּ֥ח לִבְנֶֽךָ׃ וְלָא תִּתְחַתַּן, בְּהוֹן:  בְּרַתָּךְ לָא תִּתֵּין לִבְרֵיהּ, וּבְרַתֵּיהּ לָא תִּסַּב לִבְרָךְ. 
ז,ד כִּֽי־יָסִ֤יר אֶת־בִּנְךָ֙ מֵֽאַחֲרַ֔י וְעָֽבְד֖וּ אֱלֹהִ֣ים אֲחֵרִ֑ים וְחָרָ֤ה אַף־יְהוָה֙ בָּכֶ֔ם וְהִשְׁמִֽידְךָ֖ מַהֵֽר׃ אֲרֵי יַטְעֲיָן יָת בְּרָךְ מִבָּתַר פֻּלְחָנִי, וְיִפְלְחוּן לְטָעֲוָת עַמְמַיָּא; וְיִתְקַף רֻגְזָא דַּייָ בְּכוֹן, וִישֵׁיצֵינָךְ בִּפְרִיעַ. 
ז,ה כִּֽי־אִם־כֹּ֤ה תַֽעֲשׂוּ֙ לָהֶ֔ם מִזְבְּחֹֽתֵיהֶ֣ם תִּתֹּ֔צוּ וּמַצֵּֽבֹתָ֖ם תְּשַׁבֵּ֑רוּ וַאֲשֵֽׁירֵהֶם֙ תְּגַדֵּע֔וּן וּפְסִֽילֵיהֶ֖ם תִּשְׂרְפ֥וּן בָּאֵֽשׁ׃ אֲרֵי אִם כְּדֵין תַּעְבְּדוּן, לְהוֹן--אֵיגוֹרֵיהוֹן תְּתָרְעוּן, וְקָמָתְהוֹן תְּתַבְּרוּן; וַאֲשֵׁירֵיהוֹן, תְּקָצְצוּן, וְצַלְמֵיהוֹן, תֵּיקְדוּן בְּנוּרָא. 
ז,ו כִּ֣י עַ֤ם קָדוֹשׁ֙ אַתָּ֔ה לַֽיהוָ֖ה אֱלֹהֶ֑יךָ בְּךָ֞ בָּחַ֣ר ׀ יְהוָ֣ה אֱלֹהֶ֗יךָ לִֽהְי֥וֹת לוֹ֙ לְעַ֣ם סְגֻלָּ֔ה מִכֹּל֙ הָֽעַמִּ֔ים אֲשֶׁ֖ר עַל־פְּנֵ֥י הָֽאֲדָמָֽה׃ אֲרֵי עַם קַדִּישׁ אַתְּ, קֳדָם יְיָ אֱלָהָךְ:  בָּךְ אִתְרְעִי יְיָ אֱלָהָךְ, לְמִהְוֵי לֵיהּ לְעַם חַבִּיב, מִכֹּל עַמְמַיָּא, דְּעַל אַפֵּי אַרְעָא. 
ז,ז לֹ֣א מֵֽרֻבְּכֶ֞ם מִכָּל־הָֽעַמִּ֗ים חָשַׁ֧ק יְהוָ֛ה בָּכֶ֖ם וַיִּבְחַ֣ר בָּכֶ֑ם כִּֽי־אַתֶּ֥ם הַמְעַ֖ט מִכָּל־הָֽעַמִּֽים׃ לָא מִדְּסַגִּיאִין אַתּוּן מִכָּל עַמְמַיָּא, צְבִי יְיָ בְּכוֹן--וְאִתְרְעִי בְּכוֹן:  אֲרֵי אַתּוּן זְעֵירִין, מִכָּל עַמְמַיָּא. 
ז,ח כִּי֩ מֵֽאַהֲבַ֨ת יְהוָ֜ה אֶתְכֶ֗ם וּמִשָּׁמְר֤וֹ אֶת־הַשְּׁבֻעָה֙ אֲשֶׁ֤ר נִשְׁבַּע֙ לַאֲבֹ֣תֵיכֶ֔ם הוֹצִ֧יא יְהוָ֛ה אֶתְכֶ֖ם בְּיָ֣ד חֲזָקָ֑ה וַֽיִּפְדְּךָ֙ מִבֵּ֣ית עֲבָדִ֔ים מִיַּ֖ד פַּרְעֹ֥ה מֶֽלֶךְ־מִצְרָֽיִם׃ אֲרֵי מִדְּרָחֵים יְיָ יָתְכוֹן, וּמִדְּנָטַר יָת קְיָמָא דְּקַיֵּים לַאֲבָהָתְכוֹן, אַפֵּיק יְיָ יָתְכוֹן, בְּיַד תַּקִּיפָא; וּפַרְקָךְ מִבֵּית עַבְדּוּתָא, מִיְּדָא דְּפַרְעֹה מַלְכָּא דְּמִצְרָיִם. 
ז,ט וְיָ֣דַעְתָּ֔ כִּֽי־יְהוָ֥ה אֱלֹהֶ֖יךָ ה֣וּא הָֽאֱלֹהִ֑ים הָאֵל֙ הַֽנֶּאֱמָ֔ן שֹׁמֵ֧ר הַבְּרִ֣ית וְהַחֶ֗סֶד לְאֹֽהֲבָ֛יו וּלְשֹֽׁמְרֵ֥י מִצְו‍ֹתָ֖ו לְאֶ֥לֶף דּֽוֹר׃ וְתִדַּע, אֲרֵי יְיָ אֱלָהָךְ הוּא אֱלֹהִים:  אֱלָהָא, מְהֵימְנָא--נָטַר קְיָמָא וְחִסְדָּא לְרָחֲמוֹהִי וּלְנָטְרֵי פִּקּוֹדוֹהִי, לְאֶלֶף דָּרִין. 
ז,י וּמְשַׁלֵּ֧ם לְשֹֽׂנְאָ֛יו אֶל־פָּנָ֖יו לְהַֽאֲבִיד֑וֹ לֹ֤א יְאַחֵר֙ לְשֹׂ֣נְא֔וֹ אֶל־פָּנָ֖יו יְשַׁלֶּם־לֽוֹ׃ וּמְשַׁלֵּים לְסָנְאוֹהִי טָבְוָן דְּאִנּוּן עָבְדִין קֳדָמוֹהִי בְּחַיֵּיהוֹן, לְאוֹבוֹדֵיהוֹן:  לָא מְאַחַר עוֹבָד טָב לְסָנְאוֹהִי, טָבְוָן דְּאִנּוּן עָבְדִין קֳדָמוֹהִי בְּחַיֵּיהוֹן מְשַׁלֵּים לְהוֹן. 
ז,יא וְשָֽׁמַרְתָּ֨ אֶת־הַמִּצְוָ֜ה וְאֶת־הַֽחֻקִּ֣ים וְאֶת־הַמִּשְׁפָּטִ֗ים אֲשֶׁ֨ר אָֽנֹכִ֧י מְצַוְּךָ֛ הַיּ֖וֹם לַֽעֲשׂוֹתָֽם׃ {פ} וְתִטַּר יָת תַּפְקֵידְתָּא וְיָת קְיָמַיָּא וְיָת דִּינַיָּא, דַּאֲנָא מְפַקֵּיד לָךְ יוֹמָא דֵּין--לְמֶעֱבַדְהוֹן. {פ}
