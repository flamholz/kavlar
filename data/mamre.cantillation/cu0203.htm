בוא

י,א וַיֹּ֤אמֶר יְהוָה֙ אֶל־מֹשֶׁ֔ה בֹּ֖א אֶל־פַּרְעֹ֑ה כִּֽי־אֲנִ֞י הִכְבַּ֤דְתִּי אֶת־לִבּוֹ֙ וְאֶת־לֵ֣ב עֲבָדָ֔יו לְמַ֗עַן שִׁתִ֛י אֹֽתֹתַ֥י אֵ֖לֶּה בְּקִרְבּֽוֹ׃ וַאֲמַר יְיָ לְמֹשֶׁה, עוֹל לְוָת פַּרְעֹה:  אֲרֵי אֲנָא יַקַּרִית יָת לִבֵּיהּ, וְיָת לִבָּא דְּעַבְדּוֹהִי, בְּדִיל לְשַׁוָּאָה אָתַי אִלֵּין, בֵּינֵיהוֹן. 
י,ב וּלְמַ֡עַן תְּסַפֵּר֩ בְּאָזְנֵ֨י בִנְךָ֜ וּבֶן־בִּנְךָ֗ אֵ֣ת אֲשֶׁ֤ר הִתְעַלַּ֨לְתִּי֙ בְּמִצְרַ֔יִם וְאֶת־אֹֽתֹתַ֖י אֲשֶׁר־שַׂ֣מְתִּי בָ֑ם וִֽידַעְתֶּ֖ם כִּֽי־אֲנִ֥י יְהוָֽה׃ וּבְדִיל דְּתִשְׁתַּעֵי קֳדָם בְּרָךְ וּבַר בְּרָךְ, יָת נִסִּין דַּעֲבַדִית בְּמִצְרַיִם, וְיָת אָתְוָתַי, דְּשַׁוִּיתִי בְּהוֹן; וְתִדְּעוּן, אֲרֵי אֲנָא יְיָ. 
י,ג וַיָּבֹ֨א מֹשֶׁ֣ה וְאַֽהֲרֹן֮ אֶל־פַּרְעֹה֒ וַיֹּֽאמְר֣וּ אֵלָ֗יו כֹּֽה־אָמַ֤ר יְהוָה֙ אֱלֹהֵ֣י הָֽעִבְרִ֔ים עַד־מָתַ֣י מֵאַ֔נְתָּ לֵֽעָנֹ֖ת מִפָּנָ֑י שַׁלַּ֥ח עַמִּ֖י וְיַֽעַבְדֻֽנִי׃ וְעָאל מֹשֶׁה וְאַהֲרוֹן, לְוָת פַּרְעֹה, וַאֲמַרוּ לֵיהּ כִּדְנָן אֲמַר יְיָ אֱלָהָא דִּיהוּדָאֵי, עַד אִמַּתִּי מְסָרֵיב אַתְּ לְאִתְכְּנָעָא מִן קֳדָמָי; שַׁלַּח עַמִּי, וְיִפְלְחוּן קֳדָמָי. 
י,ד כִּ֛י אִם־מָאֵ֥ן אַתָּ֖ה לְשַׁלֵּ֣חַ אֶת־עַמִּ֑י הִנְנִ֨י מֵבִ֥יא מָחָ֛ר אַרְבֶּ֖ה בִּגְבֻלֶֽךָ׃ אֲרֵי אִם מְסָרֵיב אַתְּ, לְשַׁלָּחָא יָת עַמִּי--הָאֲנָא מֵיתֵי מְחַר גּוֹבָא, בִּתְחוּמָךְ. 
י,ה וְכִסָּה֙ אֶת־עֵ֣ין הָאָ֔רֶץ וְלֹ֥א יוּכַ֖ל לִרְאֹ֣ת אֶת־הָאָ֑רֶץ וְאָכַ֣ל ׀ אֶת־יֶ֣תֶר הַפְּלֵטָ֗ה הַנִּשְׁאֶ֤רֶת לָכֶם֙ מִן־הַבָּרָ֔ד וְאָכַל֙ אֶת־כָּל־הָעֵ֔ץ הַצֹּמֵ֥חַ לָכֶ֖ם מִן־הַשָּׂדֶֽה׃ וְיִחְפֵי יָת עֵין שִׁמְשָׁא דְּאַרְעָא, וְלָא יִכּוֹל לְמִחְזֵי יָת אַרְעָא; וְיֵיכוֹל יָת שְׁאָר שֵׁיזָבְתָא, דְּאִשְׁתְּאַרַת לְכוֹן מִן בַּרְדָּא, וְיֵיכוֹל יָת כָּל אִילָנָא, דְּאַצְמַח לְכוֹן מִן חַקְלָא. 
י,ו וּמָֽלְא֨וּ בָתֶּ֜יךָ וּבָתֵּ֣י כָל־עֲבָדֶיךָ֮ וּבָתֵּ֣י כָל־מִצְרַיִם֒ אֲשֶׁ֨ר לֹֽא־רָא֤וּ אֲבֹתֶ֨יךָ֙ וַֽאֲב֣וֹת אֲבֹתֶ֔יךָ מִיּ֗וֹם הֱיוֹתָם֙ עַל־הָ֣אֲדָמָ֔ה עַ֖ד הַיּ֣וֹם הַזֶּ֑ה וַיִּ֥פֶן וַיֵּצֵ֖א מֵעִ֥ם פַּרְעֹֽה׃ וְיִתְמְלוֹן בָּתָּךְ וּבָתֵּי כָּל עַבְדָךְ, וּבָתֵּי כָּל מִצְרָאֵי, דְּלָא חֲזוֹ אֲבָהָתָךְ וַאֲבָהָת אֲבָהָתָךְ, מִיּוֹם מִהְוֵיהוֹן עַל אַרְעָא עַד יוֹמָא הָדֵין; וְאִתְפְּנִי וּנְפַק, מִלְּוָת פַּרְעֹה. 
י,ז וַיֹּֽאמְרוּ֩ עַבְדֵ֨י פַרְעֹ֜ה אֵלָ֗יו עַד־מָתַי֙ יִֽהְיֶ֨ה זֶ֥ה לָ֨נוּ֙ לְמוֹקֵ֔שׁ שַׁלַּח֙ אֶת־הָ֣אֲנָשִׁ֔ים וְיַֽעַבְד֖וּ אֶת־יְהוָ֣ה אֱלֹֽהֵיהֶ֑ם הֲטֶ֣רֶם תֵּדַ֔ע כִּ֥י אָֽבְדָ֖ה מִצְרָֽיִם׃ וַאֲמַרוּ עַבְדֵי פַּרְעֹה לֵיהּ, עַד אִמַּתִּי יְהֵי דֵּין לַנָא לְתַקְלָא--שַׁלַּח יָת גֻּבְרַיָּא, וְיִפְלְחוּן קֳדָם יְיָ אֱלָהֲהוֹן; הַעַד כְּעַן לָא יְדַעְתָּא, אֲרֵי אֲבַדַת מִצְרָיִם. 
י,ח וַיּוּשַׁ֞ב אֶת־מֹשֶׁ֤ה וְאֶֽת־אַהֲרֹן֙ אֶל־פַּרְעֹ֔ה וַיֹּ֣אמֶר אֲלֵהֶ֔ם לְכ֥וּ עִבְד֖וּ אֶת־יְהוָ֣ה אֱלֹֽהֵיכֶ֑ם מִ֥י וָמִ֖י הַהֹֽלְכִֽים׃ וְאִתָּתַב יָת מֹשֶׁה וְיָת אַהֲרוֹן, לְוָת פַּרְעֹה, וַאֲמַר לְהוֹן, אִיזִילוּ פְּלַחוּ קֳדָם יְיָ אֱלָהֲכוֹן; מַאן וּמַאן, אָזְלִין. 
י,ט וַיֹּ֣אמֶר מֹשֶׁ֔ה בִּנְעָרֵ֥ינוּ וּבִזְקֵנֵ֖ינוּ נֵלֵ֑ךְ בְּבָנֵ֨ינוּ וּבִבְנוֹתֵ֜נוּ בְּצֹאנֵ֤נוּ וּבִבְקָרֵ֨נוּ֙ נֵלֵ֔ךְ כִּ֥י חַג־יְהוָ֖ה לָֽנוּ׃ וַאֲמַר מֹשֶׁה, בְּעוּלֵימַנָא וּבְסָבַנָא נֵיזֵיל; בִּבְנַנָא וּבִבְנָתַנָא בְּעָנַנָא וּבְתוֹרַנָא, נֵיזֵיל--אֲרֵי חַגָּא קֳדָם יְיָ, לַנָא. 
י,י וַיֹּ֣אמֶר אֲלֵהֶ֗ם יְהִ֨י כֵ֤ן יְהוָה֙ עִמָּכֶ֔ם כַּֽאֲשֶׁ֛ר אֲשַׁלַּ֥ח אֶתְכֶ֖ם וְאֶֽת־טַפְּכֶ֑ם רְא֕וּ כִּ֥י רָעָ֖ה נֶ֥גֶד פְּנֵיכֶֽם׃ וַאֲמַר לְהוֹן, יְהֵי כֵּין מֵימְרָא דַּייָ בְּסַעְדְּכוֹן, כַּד אֲשַׁלַּח יָתְכוֹן, וְיָת טַפְלְכוֹן; חֲזוֹ, אֲרֵי בִּישָׁא אַתּוּן סְבִירִין לְמֶעֱבַד לֵית קֳבֵיל אַפֵּיכוֹן לְאִסְתְּחָרָא. 
י,יא לֹ֣א כֵ֗ן לְכֽוּ־נָ֤א הַגְּבָרִים֙ וְעִבְד֣וּ אֶת־יְהוָ֔ה כִּ֥י אֹתָ֖הּ אַתֶּ֣ם מְבַקְשִׁ֑ים וַיְגָ֣רֶשׁ אֹתָ֔ם מֵאֵ֖ת פְּנֵ֥י פַרְעֹֽה׃ {ס} לָא כֵּין, אִיזִילוּ כְּעַן גֻּבְרַיָּא וּפְלַחוּ קֳדָם יְיָ--אֲרֵי יָתַהּ, אַתּוּן בָּעַן; וְתָרֵיךְ יָתְהוֹן, מִן קֳדָם פַּרְעֹה. {ס}
י,יב וַיֹּ֨אמֶר יְהוָ֜ה אֶל־מֹשֶׁ֗ה נְטֵ֨ה יָֽדְךָ֜ עַל־אֶ֤רֶץ מִצְרַ֨יִם֙ בָּֽאַרְבֶּ֔ה וְיַ֖עַל עַל־אֶ֣רֶץ מִצְרָ֑יִם וְיֹאכַל֙ אֶת־כָּל־עֵ֣שֶׂב הָאָ֔רֶץ אֵ֛ת כָּל־אֲשֶׁ֥ר הִשְׁאִ֖יר הַבָּרָֽד׃ וַאֲמַר יְיָ לְמֹשֶׁה, אֲרֵים יְדָךְ עַל אַרְעָא דְּמִצְרַיִם וְיֵיתֵי גּוֹבָא, וְיִסַּק, עַל אַרְעָא דְּמִצְרָיִם; וְיֵיכוֹל יָת כָּל עִסְבָּא דְּאַרְעָא, יָת כָּל דְּאַשְׁאַר בַּרְדָּא. 
י,יג וַיֵּ֨ט מֹשֶׁ֣ה אֶת־מַטֵּהוּ֮ עַל־אֶ֣רֶץ מִצְרַיִם֒ וַֽיהוָ֗ה נִהַ֤ג רֽוּחַ־קָדִים֙ בָּאָ֔רֶץ כָּל־הַיּ֥וֹם הַה֖וּא וְכָל־הַלָּ֑יְלָה הַבֹּ֣קֶר הָיָ֔ה וְר֨וּחַ֙ הַקָּדִ֔ים נָשָׂ֖א אֶת־הָֽאַרְבֶּֽה׃ וַאֲרֵים מֹשֶׁה יָת חֻטְרֵיהּ, עַל אַרְעָא דְּמִצְרַיִם, וַייָ דַּבַּר רוּחַ קִדּוּמָא בְּאַרְעָא, כָּל יוֹמָא הַהוּא וְכָל לֵילְיָא; צַפְרָא הֲוָה--וְרוּחַ קִדּוּמָא, נְטַל יָת גּוֹבָא. 
י,יד וַיַּ֣עַל הָֽאַרְבֶּ֗ה עַ֚ל כָּל־אֶ֣רֶץ מִצְרַ֔יִם וַיָּ֕נַח בְּכֹ֖ל גְּב֣וּל מִצְרָ֑יִם כָּבֵ֣ד מְאֹ֔ד לְ֠פָנָיו לֹא־הָ֨יָה כֵ֤ן אַרְבֶּה֙ כָּמֹ֔הוּ וְאַֽחֲרָ֖יו לֹ֥א יִֽהְיֶה־כֵּֽן׃ וּסְלֵיק גּוֹבָא, עַל כָּל אַרְעָא דְּמִצְרַיִם, וּשְׁרָא, בְּכֹל תְּחוּם מִצְרָיִם:  תַּקִּיף לַחְדָּא--קֳדָמוֹהִי לָא הֲוָה כֵּין גּוֹבָא דִּכְוָתֵיהּ, וּבָתְרוֹהִי לָא יְהֵי כֵּין. 
י,טו וַיְכַ֞ס אֶת־עֵ֣ין כָּל־הָאָרֶץ֮ וַתֶּחְשַׁ֣ךְ הָאָרֶץ֒ וַיֹּ֜אכַל אֶת־כָּל־עֵ֣שֶׂב הָאָ֗רֶץ וְאֵת֙ כָּל־פְּרִ֣י הָעֵ֔ץ אֲשֶׁ֥ר הוֹתִ֖יר הַבָּרָ֑ד וְלֹֽא־נוֹתַ֨ר כָּל־יֶ֧רֶק בָּעֵ֛ץ וּבְעֵ֥שֶׂב הַשָּׂדֶ֖ה בְּכָל־אֶ֥רֶץ מִצְרָֽיִם׃ וַחֲפָא יָת עֵין שִׁמְשָׁא דְּכָל אַרְעָא, וַחֲשׁוֹכַת אַרְעָא, וַאֲכַל יָת כָּל עִסְבָּא דְּאַרְעָא וְיָת כָּל פֵּירֵי אִילָנָא, דְּאַשְׁאַר בַּרְדָּא; וְלָא אִשְׁתְּאַר כָּל יָרוֹק בְּאִילָנָא וּבְעִסְבָּא דְּחַקְלָא, בְּכָל אַרְעָא דְּמִצְרָיִם. 
י,טז וַיְמַהֵ֣ר פַּרְעֹ֔ה לִקְרֹ֖א לְמֹשֶׁ֣ה וּֽלְאַהֲרֹ֑ן וַיֹּ֗אמֶר חָטָ֛אתִי לַֽיהוָ֥ה אֱלֹֽהֵיכֶ֖ם וְלָכֶֽם׃ וְאוֹחִי פַּרְעֹה, לְמִקְרֵי לְמֹשֶׁה וּלְאַהֲרוֹן; וַאֲמַר, חַבִית קֳדָם יְיָ אֱלָהֲכוֹן--וּלְכוֹן. 
י,יז וְעַתָּ֗ה שָׂ֣א נָ֤א חַטָּאתִי֙ אַ֣ךְ הַפַּ֔עַם וְהַעְתִּ֖ירוּ לַֽיהוָ֣ה אֱלֹֽהֵיכֶ֑ם וְיָסֵר֙ מֵֽעָלַ֔י רַ֖ק אֶת־הַמָּ֥וֶת הַזֶּֽה׃ וּכְעַן, שְׁבוֹק כְּעַן לְחוֹבִי בְּרַם זִמְנָא הָדָא, וְצַלּוֹ, קֳדָם יְיָ אֱלָהֲכוֹן; וְיַעְדֵּי, מִנִּי, לְחוֹד, יָת מוֹתָא הָדֵין. 
י,יח וַיֵּצֵ֖א מֵעִ֣ם פַּרְעֹ֑ה וַיֶּעְתַּ֖ר אֶל־יְהוָֽה׃ וּנְפַק, מִלְּוָת פַּרְעֹה; וְצַלִּי, קֳדָם יְיָ. 
י,יט וַיַּֽהֲפֹ֨ךְ יְהוָ֤ה רֽוּחַ־יָם֙ חָזָ֣ק מְאֹ֔ד וַיִּשָּׂא֙ אֶת־הָ֣אַרְבֶּ֔ה וַיִּתְקָעֵ֖הוּ יָ֣מָּה סּ֑וּף לֹ֤א נִשְׁאַר֙ אַרְבֶּ֣ה אֶחָ֔ד בְּכֹ֖ל גְּב֥וּל מִצְרָֽיִם׃ וַהֲפַךְ יְיָ רוּחַ מַעְרְבָא, תַּקִּיף לַחְדָּא, וּנְטַל יָת גּוֹבָא, וּרְמָהִי לְיַמָּא דְּסוּף:  לָא אִשְׁתְּאַר גּוֹבָא חַד, בְּכֹל תְּחוּם מִצְרָיִם. 
י,כ וַיְחַזֵּ֥ק יְהוָ֖ה אֶת־לֵ֣ב פַּרְעֹ֑ה וְלֹ֥א שִׁלַּ֖ח אֶת־בְּנֵ֥י יִשְׂרָאֵֽל׃ {פ} וְתַקֵּיף יְיָ, יָת לִבָּא דְּפַרְעֹה; וְלָא שַׁלַּח, יָת בְּנֵי יִשְׂרָאֵל. {פ}
י,כא וַיֹּ֨אמֶר יְהוָ֜ה אֶל־מֹשֶׁ֗ה נְטֵ֤ה יָֽדְךָ֙ עַל־הַשָּׁמַ֔יִם וִ֥יהִי חֹ֖שֶׁךְ עַל־אֶ֣רֶץ מִצְרָ֑יִם וְיָמֵ֖שׁ חֹֽשֶׁךְ׃ וַאֲמַר יְיָ לְמֹשֶׁה, אֲרֵים יְדָךְ עַל צֵית שְׁמַיָּא, וִיהֵי חֲשׁוֹכָא, עַל אַרְעָא דְּמִצְרָיִם; בָּתַר דְּיִעְדֵּי, קְבַל לֵילְיָא. 
י,כב וַיֵּ֥ט מֹשֶׁ֛ה אֶת־יָד֖וֹ עַל־הַשָּׁמָ֑יִם וַיְהִ֧י חֹֽשֶׁךְ־אֲפֵלָ֛ה בְּכָל־אֶ֥רֶץ מִצְרַ֖יִם שְׁלֹ֥שֶׁת יָמִֽים׃ וַאֲרֵים מֹשֶׁה יָת יְדֵיהּ, עַל צֵית שְׁמַיָּא; וַהֲוָה חֲשׁוֹךְ קְבַל בְּכָל אַרְעָא דְּמִצְרַיִם, תְּלָתָא יוֹמִין. 
י,כג לֹֽא־רָא֞וּ אִ֣ישׁ אֶת־אָחִ֗יו וְלֹא־קָ֛מוּ אִ֥ישׁ מִתַּחְתָּ֖יו שְׁלֹ֣שֶׁת יָמִ֑ים וּֽלְכָל־בְּנֵ֧י יִשְׂרָאֵ֛ל הָ֥יָה א֖וֹר בְּמֽוֹשְׁבֹתָֽם׃ לָא חֲזוֹ גְּבַר יָת אֲחוּהִי, וְלָא קָמוּ אֲנָשׁ מִתְּחוֹתוֹהִי--תְּלָתָא יוֹמִין; וּלְכָל בְּנֵי יִשְׂרָאֵל הֲוָה נְהוֹרָא, בְּמוֹתְבָנֵיהוֹן. 
י,כד וַיִּקְרָ֨א פַרְעֹ֜ה אֶל־מֹשֶׁ֗ה וַיֹּ֨אמֶר֙ לְכוּ֙ עִבְד֣וּ אֶת־יְהוָ֔ה רַ֛ק צֹֽאנְכֶ֥ם וּבְקַרְכֶ֖ם יֻצָּ֑ג גַּֽם־טַפְּכֶ֖ם יֵלֵ֥ךְ עִמָּכֶֽם׃ וּקְרָא פַּרְעֹה לְמֹשֶׁה, וַאֲמַר אִיזִילוּ פְּלַחוּ קֳדָם יְיָ--לְחוֹד עָנְכוֹן וְתוֹרֵיכוֹן, שְׁבוּקוּ:  אַף טַפְלְכוֹן, יֵיזֵיל עִמְּכוֹן. 
י,כה וַיֹּ֣אמֶר מֹשֶׁ֔ה גַּם־אַתָּ֛ה תִּתֵּ֥ן בְּיָדֵ֖נוּ זְבָחִ֣ים וְעֹלֹ֑ת וְעָשִׂ֖ינוּ לַֽיהוָ֥ה אֱלֹהֵֽינוּ׃ וַאֲמַר מֹשֶׁה, אַף אַתְּ תִּתֵּין בִּידַנָא נִכְסַת קֻדְשִׁין וַעֲלָוָן; וְנַעֲבֵיד, קֳדָם יְיָ אֱלָהַנָא. 
י,כו וְגַם־מִקְנֵ֜נוּ יֵלֵ֣ךְ עִמָּ֗נוּ לֹ֤א תִשָּׁאֵר֙ פַּרְסָ֔ה כִּ֚י מִמֶּ֣נּוּ נִקַּ֔ח לַֽעֲבֹ֖ד אֶת־יְהוָ֣ה אֱלֹהֵ֑ינוּ וַֽאֲנַ֣חְנוּ לֹֽא־נֵדַ֗ע מַֽה־נַּעֲבֹד֙ אֶת־יְהוָ֔ה עַד־בֹּאֵ֖נוּ שָֽׁמָּה׃ וְאַף בְּעִירַנָא יֵיזֵיל עִמַּנָא, לָא נַשְׁאַר מִנֵּיהּ מִדָּעַם--אֲרֵי מִנֵּיהּ אֲנַחְנָא נָסְבִין, לְמִפְלַח קֳדָם יְיָ אֱלָהַנָא; וַאֲנַחְנָא לֵית אֲנַחְנָא יָדְעִין, מָא נִפְלַח קֳדָם יְיָ, עַד מֵיתַנָא, לְתַמָּן. 
י,כז וַיְחַזֵּ֥ק יְהוָ֖ה אֶת־לֵ֣ב פַּרְעֹ֑ה וְלֹ֥א אָבָ֖ה לְשַׁלְּחָֽם׃ וְתַקֵּיף יְיָ, יָת לִבָּא דְּפַרְעֹה; וְלָא אֲבָא, לְשַׁלָּחוּתְהוֹן. 
י,כח וַיֹּֽאמֶר־ל֥וֹ פַרְעֹ֖ה לֵ֣ךְ מֵֽעָלָ֑י הִשָּׁ֣מֶר לְךָ֗ אַל־תֹּ֨סֶף֙ רְא֣וֹת פָּנַ֔י כִּ֗י בְּי֛וֹם רְאֹֽתְךָ֥ פָנַ֖י תָּמֽוּת׃ וַאֲמַר לֵיהּ פַּרְעֹה, אִיזֵיל מֵעִלָּוָי; אִסְתְּמַר לָךְ, לָא תּוֹסֵיף לְמִחְזֵי אַפַּי--אֲרֵי בְּיוֹמָא דְּתִחְזֵי אַפַּי, תְּמוּת. 
י,כט וַיֹּ֥אמֶר מֹשֶׁ֖ה כֵּ֣ן דִּבַּ֑רְתָּ לֹֽא־אֹסִ֥ף ע֖וֹד רְא֥וֹת פָּנֶֽיךָ׃ {פ} וַאֲמַר מֹשֶׁה, יָאוּת מַלֵּילְתָּא:  לָא אוֹסֵיף עוֹד, לְמִחְזֵי אַפָּךְ. {פ}
יא,א וַיֹּ֨אמֶר יְהוָ֜ה אֶל־מֹשֶׁ֗ה ע֣וֹד נֶ֤גַע אֶחָד֙ אָבִ֤יא עַל־פַּרְעֹה֙ וְעַל־מִצְרַ֔יִם אַֽחֲרֵי־כֵ֕ן יְשַׁלַּ֥ח אֶתְכֶ֖ם מִזֶּ֑ה כְּשַׁ֨לְּח֔וֹ כָּלָ֕ה גָּרֵ֛שׁ יְגָרֵ֥שׁ אֶתְכֶ֖ם מִזֶּֽה׃ וַאֲמַר יְיָ לְמֹשֶׁה, עוֹד מַכְתָּשׁ חַד אַיְתִי עַל פַּרְעֹה וְעַל מִצְרָאֵי--בָּתַר כֵּין, יְשַׁלַּח יָתְכוֹן מִכָּא:  כְּשַׁלָּחוּתֵיהּ--גְּמֵירָא, תָּרָכָא יְתָרֵיךְ יָתְכוֹן מִכָּא. 
יא,ב דַּבֶּר־נָ֖א בְּאָזְנֵ֣י הָעָ֑ם וְיִשְׁאֲל֞וּ אִ֣ישׁ ׀ מֵאֵ֣ת רֵעֵ֗הוּ וְאִשָּׁה֙ מֵאֵ֣ת רְעוּתָ֔הּ כְּלֵי־כֶ֖סֶף וּכְלֵ֥י זָהָֽב׃ מַלֵּיל כְּעַן, קֳדָם עַמָּא; וְיִשְׁאֲלוּן גְּבַר מִן חַבְרֵיהּ, וְאִתְּתָא מִן חֲבִרְתַּהּ, מָנִין דִּכְסַף, וּמָנִין דִּדְהַב. 
יא,ג וַיִּתֵּ֧ן יְהוָ֛ה אֶת־חֵ֥ן הָעָ֖ם בְּעֵינֵ֣י מִצְרָ֑יִם גַּ֣ם ׀ הָאִ֣ישׁ מֹשֶׁ֗ה גָּד֤וֹל מְאֹד֙ בְּאֶ֣רֶץ מִצְרַ֔יִם בְּעֵינֵ֥י עַבְדֵֽי־פַרְעֹ֖ה וּבְעֵינֵ֥י הָעָֽם׃ {ס} וִיהַב יְיָ יָת עַמָּא לְרַחֲמִין, בְּעֵינֵי מִצְרָאֵי; אַף גֻּבְרָא מֹשֶׁה, רַב לַחְדָּא בְּאַרְעָא דְּמִצְרַיִם, בְּעֵינֵי עַבְדֵי פַּרְעֹה, וּבְעֵינֵי עַמָּא. {ס}
יא,ד וַיֹּ֣אמֶר מֹשֶׁ֔ה כֹּ֖ה אָמַ֣ר יְהוָ֑ה כַּֽחֲצֹ֣ת הַלַּ֔יְלָה אֲנִ֥י יוֹצֵ֖א בְּת֥וֹךְ מִצְרָֽיִם׃ וַאֲמַר מֹשֶׁה, כִּדְנָן אֲמַר יְיָ:  כְּפַלְגוּת לֵילְיָא, אֲנָא מִתְגְּלֵי בְּגוֹ מִצְרָיִם. 
יא,ה וּמֵ֣ת כָּל־בְּכוֹר֮ בְּאֶ֣רֶץ מִצְרַיִם֒ מִבְּכ֤וֹר פַּרְעֹה֙ הַיֹּשֵׁ֣ב עַל־כִּסְא֔וֹ עַ֚ד בְּכ֣וֹר הַשִּׁפְחָ֔ה אֲשֶׁ֖ר אַחַ֣ר הָֽרֵחָ֑יִם וְכֹ֖ל בְּכ֥וֹר בְּהֵמָֽה׃ וִימוּת כָּל בֻּכְרָא, בְּאַרְעָא דְּמִצְרַיִם--מִבֻּכְרָא דְּפַרְעֹה דַּעֲתִיד לְמִתַּב עַל כֻּרְסֵי מַלְכוּתֵיהּ, עַד בֻּכְרָא דְּאַמְתָּא דִּבְבָּתַר רִחְיָא; וְכֹל, בֻּכְרָא דִּבְעִירָא. 
יא,ו וְהָ֥יְתָ֛ה צְעָקָ֥ה גְדֹלָ֖ה בְּכָל־אֶ֣רֶץ מִצְרָ֑יִם אֲשֶׁ֤ר כָּמֹ֨הוּ֙ לֹ֣א נִֽהְיָ֔תָה וְכָמֹ֖הוּ לֹ֥א תֹסִֽף׃ וּתְהֵי צְוַחְתָּא רַבְּתָא, בְּכָל אַרְעָא דְּמִצְרָיִם, דִּכְוָתַהּ לָא הֲוָת, וְדִכְוָתַהּ לָא תּוֹסֵיף. 
יא,ז וּלְכֹ֣ל ׀ בְּנֵ֣י יִשְׂרָאֵ֗ל לֹ֤א יֶֽחֱרַץ־כֶּ֨לֶב֙ לְשֹׁנ֔וֹ לְמֵאִ֖ישׁ וְעַד־בְּהֵמָ֑ה לְמַ֨עַן֙ תֵּֽדְע֔וּן אֲשֶׁר֙ יַפְלֶ֣ה יְהוָ֔ה בֵּ֥ין מִצְרַ֖יִם וּבֵ֥ין יִשְׂרָאֵֽל׃ וּלְכֹל בְּנֵי יִשְׂרָאֵל, לָא יַנְזֵיק כַּלְבָּא בְּלִישָׁנֵיהּ לְמִבַּח, לְמֵאֲנָשָׁא, וְעַד בְּעִירָא--בְּדִיל, דְּתִדְּעוּן, דְּיַפְרֵישׁ יְיָ, בֵּין מִצְרָאֵי וּבֵין יִשְׂרָאֵל. 
יא,ח וְיָֽרְד֣וּ כָל־עֲבָדֶיךָ֩ אֵ֨לֶּה אֵלַ֜י וְהִשְׁתַּֽחֲווּ־לִ֣י לֵאמֹ֗ר צֵ֤א אַתָּה֙ וְכָל־הָעָ֣ם אֲשֶׁר־בְּרַגְלֶ֔יךָ וְאַֽחֲרֵי־כֵ֖ן אֵצֵ֑א וַיֵּצֵ֥א מֵֽעִם־פַּרְעֹ֖ה בָּֽחֳרִי־אָֽף׃ {ס} וְיֵיחֲתוּן כָּל עַבְדָךְ אִלֵּין לְוָתִי וְיִבְעוֹן מִנִּי לְמֵימַר, פּוֹק אַתְּ וְכָל עַמָּא דְּעִמָּךְ, וּבָתַר כֵּין, אֶפּוֹק; וּנְפַק מִלְּוָת פַּרְעֹה, בִּתְקוֹף רְגַז. {ס}
יא,ט וַיֹּ֤אמֶר יְהוָה֙ אֶל־מֹשֶׁ֔ה לֹֽא־יִשְׁמַ֥ע אֲלֵיכֶ֖ם פַּרְעֹ֑ה לְמַ֛עַן רְב֥וֹת מֽוֹפְתַ֖י בְּאֶ֥רֶץ מִצְרָֽיִם׃ וַאֲמַר יְיָ לְמֹשֶׁה, לָא יְקַבֵּיל מִנְּכוֹן פַּרְעֹה--בְּדִיל לְאַסְגָּאָה מוֹפְתַי, בְּאַרְעָא דְּמִצְרָיִם. 
יא,י וּמֹשֶׁ֣ה וְאַֽהֲרֹ֗ן עָשׂ֛וּ אֶת־כָּל־הַמֹּֽפְתִ֥ים הָאֵ֖לֶּה לִפְנֵ֣י פַרְעֹ֑ה וַיְחַזֵּ֤ק יְהוָה֙ אֶת־לֵ֣ב פַּרְעֹ֔ה וְלֹֽא־שִׁלַּ֥ח אֶת־בְּנֵֽי־יִשְׂרָאֵ֖ל מֵֽאַרְצֽוֹ׃ {ס} וּמֹשֶׁה וְאַהֲרוֹן, עֲבַדוּ יָת כָּל מוֹפְתַיָּא הָאִלֵּין--קֳדָם פַּרְעֹה; וְתַקֵּיף יְיָ יָת לִבָּא דְּפַרְעֹה, וְלָא שַׁלַּח יָת בְּנֵי יִשְׂרָאֵל מֵאַרְעֵיהּ. {ס}
יב,א וַיֹּ֤אמֶר יְהוָה֙ אֶל־מֹשֶׁ֣ה וְאֶֽל־אַהֲרֹ֔ן בְּאֶ֥רֶץ מִצְרַ֖יִם לֵאמֹֽר׃ וַאֲמַר יְיָ לְמֹשֶׁה וּלְאַהֲרוֹן, בְּאַרְעָא דְּמִצְרַיִם לְמֵימַר. 
יב,ב הַחֹ֧דֶשׁ הַזֶּ֛ה לָכֶ֖ם רֹ֣אשׁ חֳדָשִׁ֑ים רִאשׁ֥וֹן הוּא֙ לָכֶ֔ם לְחָדְשֵׁ֖י הַשָּׁנָֽה׃ יַרְחָא הָדֵין לְכוֹן, רֵישׁ יַרְחַיָּא:  קַדְמַאי הוּא לְכוֹן, לְיַרְחֵי שַׁתָּא. 
יב,ג דַּבְּר֗וּ אֶֽל־כָּל־עֲדַ֤ת יִשְׂרָאֵל֙ לֵאמֹ֔ר בֶּֽעָשֹׂ֖ר לַחֹ֣דֶשׁ הַזֶּ֑ה וְיִקְח֣וּ לָהֶ֗ם אִ֛ישׁ שֶׂ֥ה לְבֵית־אָבֹ֖ת שֶׂ֥ה לַבָּֽיִת׃ מַלִּילוּ, עִם כָּל כְּנִשְׁתָּא דְּיִשְׂרָאֵל לְמֵימַר, בְּעַסְרָא, לְיַרְחָא הָדֵין:  וְיִסְּבוּן לְהוֹן, גְּבַר אִמַּר לְבֵית אַבָּא--אִמְּרָא לְבֵיתָא. 
יב,ד וְאִם־יִמְעַ֣ט הַבַּיִת֮ מִֽהְי֣וֹת מִשֶּׂה֒ וְלָקַ֣ח ה֗וּא וּשְׁכֵנ֛וֹ הַקָּרֹ֥ב אֶל־בֵּית֖וֹ בְּמִכְסַ֣ת נְפָשֹׁ֑ת אִ֚ישׁ לְפִ֣י אָכְל֔וֹ תָּכֹ֖סּוּ עַל־הַשֶּֽׂה׃ וְאִם זְעֵיר בֵּיתָא, מִלְּאִתְמְנָאָה עַל אִמְּרָא--וְיִסַּב הוּא וְשֵׁיבָבֵיהּ דְּקָרִיב לְבֵיתֵיהּ, בְּמִנְיַן נַפְשָׁתָא:  גְּבַר לְפֹם מֵיכְלֵיהּ, תִּתְמְנוֹן עַל אִמְּרָא. 
יב,ה שֶׂ֥ה תָמִ֛ים זָכָ֥ר בֶּן־שָׁנָ֖ה יִֽהְיֶ֣ה לָכֶ֑ם מִן־הַכְּבָשִׂ֥ים וּמִן־הָֽעִזִּ֖ים תִּקָּֽחוּ׃ אִמַּר שְׁלִים דְּכַר בַּר שַׁתֵּיהּ, יְהֵי לְכוֹן; מִן אִמְּרַיָּא וּמִן בְּנֵי עִזַּיָּא, תִּסְּבוּן. 
יב,ו וְהָיָ֤ה לָכֶם֙ לְמִשְׁמֶ֔רֶת עַ֣ד אַרְבָּעָ֥ה עָשָׂ֛ר י֖וֹם לַחֹ֣דֶשׁ הַזֶּ֑ה וְשָֽׁחֲט֣וּ אֹת֗וֹ כֹּ֛ל קְהַ֥ל עֲדַֽת־יִשְׂרָאֵ֖ל בֵּ֥ין הָֽעַרְבָּֽיִם׃ וִיהֵי לְכוֹן לְמַטְּרָא, עַד אַרְבְּעַת עַסְרָא יוֹמָא לְיַרְחָא הָדֵין; וְיִכְּסוּן יָתֵיהּ, כֹּל קְהָלָא כְּנִשְׁתָּא דְּיִשְׂרָאֵל--בֵּין שִׁמְשַׁיָּא. 
יב,ז וְלָֽקְחוּ֙ מִן־הַדָּ֔ם וְנָ֥תְנ֛וּ עַל־שְׁתֵּ֥י הַמְּזוּזֹ֖ת וְעַל־הַמַּשְׁק֑וֹף עַ֚ל הַבָּ֣תִּ֔ים אֲשֶׁר־יֹֽאכְל֥וּ אֹת֖וֹ בָּהֶֽם׃ וְיִסְּבוּן, מִן דְּמָא, וְיִתְּנוּן עַל תְּרֵין סִפַּיָּא, וְעַל שָׁקְפָא--עַל, בָּתַּיָּא, דְּיֵיכְלוּן יָתֵיהּ, בְּהוֹן. 
יב,ח וְאָֽכְל֥וּ אֶת־הַבָּשָׂ֖ר בַּלַּ֣יְלָה הַזֶּ֑ה צְלִי־אֵ֣שׁ וּמַצּ֔וֹת עַל־מְרֹרִ֖ים יֹֽאכְלֻֽהוּ׃ וְיֵיכְלוּן יָת בִּסְרָא, בְּלֵילְיָא הָדֵין:  טְוֵי נוּר וּפַטִּיר, עַל מְרָרִין יֵיכְלוּנֵּיהּ. 
יב,ט אַל־תֹּֽאכְל֤וּ מִמֶּ֨נּוּ֙ נָ֔א וּבָשֵׁ֥ל מְבֻשָּׁ֖ל בַּמָּ֑יִם כִּ֣י אִם־צְלִי־אֵ֔שׁ רֹאשׁ֥וֹ עַל־כְּרָעָ֖יו וְעַל־קִרְבּֽוֹ׃ לָא תֵּיכְלוּן מִנֵּיהּ כִּד חַי, וְאַף לָא כִּד בַּשָּׁלָא מְבֻשַּׁל בְּמַיָּא:  אֱלָהֵין טְוֵי נוּר, רֵישֵׁיהּ עַל כְּרָעוֹהִי וְעַל גַּוֵּיהּ. 
יב,י וְלֹֽא־תוֹתִ֥ירוּ מִמֶּ֖נּוּ עַד־בֹּ֑קֶר וְהַנֹּתָ֥ר מִמֶּ֛נּוּ עַד־בֹּ֖קֶר בָּאֵ֥שׁ תִּשְׂרֹֽפוּ׃ וְלָא תַּשְׁאֲרוּן מִנֵּיהּ, עַד צַפְרָא; וּדְיִשְׁתְּאַר מִנֵּיהּ עַד צַפְרָא, בְּנוּרָא תֵּיקְדוּן. 
יב,יא וְכָכָה֮ תֹּֽאכְל֣וּ אֹתוֹ֒ מָתְנֵיכֶ֣ם חֲגֻרִ֔ים נַֽעֲלֵיכֶם֙ בְּרַגְלֵיכֶ֔ם וּמַקֶּלְכֶ֖ם בְּיֶדְכֶ֑ם וַֽאֲכַלְתֶּ֤ם אֹתוֹ֙ בְּחִפָּז֔וֹן פֶּ֥סַח ה֖וּא לַֽיהוָֽה׃ וּכְדֵין, תֵּיכְלוּן יָתֵיהּ--חַרְצֵיכוֹן יְהוֹן אֲסִירִין, מְסָנֵיכוֹן בְּרַגְלֵיכוֹן וְחֻטְרֵיכוֹן בְּיַדְכוֹן; וְתֵיכְלוּן יָתֵיהּ בִּבְהִילוּ, פִּסְחָא הוּא קֳדָם יְיָ. 
יב,יב וְעָֽבַרְתִּ֣י בְאֶֽרֶץ־מִצְרַיִם֮ בַּלַּ֣יְלָה הַזֶּה֒ וְהִכֵּיתִ֤י כָל־בְּכוֹר֙ בְּאֶ֣רֶץ מִצְרַ֔יִם מֵֽאָדָ֖ם וְעַד־בְּהֵמָ֑ה וּבְכָל־אֱלֹהֵ֥י מִצְרַ֛יִם אֶֽעֱשֶׂ֥ה שְׁפָטִ֖ים אֲנִ֥י יְהוָֽה׃ וְאֶתְגְּלֵי בְּאַרְעָא דְּמִצְרַיִם, בְּלֵילְיָא הָדֵין, וְאֶקְטוֹל כָּל בֻּכְרָא בְּאַרְעָא דְּמִצְרַיִם, מֵאֲנָשָׁא וְעַד בְּעִירָא; וּבְכָל טָעֲוָת מִצְרָאֵי אַעֲבֵיד דִּינִין, אֲנָא יְיָ. 
יב,יג וְהָיָה֩ הַדָּ֨ם לָכֶ֜ם לְאֹ֗ת עַ֤ל הַבָּתִּים֙ אֲשֶׁ֣ר אַתֶּ֣ם שָׁ֔ם וְרָאִ֨יתִי֙ אֶת־הַדָּ֔ם וּפָֽסַחְתִּ֖י עֲלֵכֶ֑ם וְלֹֽא־יִֽהְיֶ֨ה בָכֶ֥ם נֶ֨גֶף֙ לְמַשְׁחִ֔ית בְּהַכֹּתִ֖י בְּאֶ֥רֶץ מִצְרָֽיִם׃ וִיהֵי דְּמָא לְכוֹן לְאָת, עַל בָּתַּיָּא דְּאַתּוּן תַּמָּן, וְאֶחְזֵי יָת דְּמָא, וַאֲחוּס עֲלֵיכוֹן; וְלָא יְהֵי בְּכוֹן מוֹתָא לְחַבָּלָא, בְּמִקְטְלִי בְּאַרְעָא דְּמִצְרָיִם. 
יב,יד וְהָיָה֩ הַיּ֨וֹם הַזֶּ֤ה לָכֶם֙ לְזִכָּר֔וֹן וְחַגֹּתֶ֥ם אֹת֖וֹ חַ֣ג לַֽיהוָ֑ה לְדֹרֹ֣תֵיכֶ֔ם חֻקַּ֥ת עוֹלָ֖ם תְּחָגֻּֽהוּ׃ וִיהֵי יוֹמָא הָדֵין לְכוֹן לְדֻכְרָנָא, וְתֵיחֲגוּן יָתֵיהּ חַגָּא קֳדָם יְיָ:  לְדָרֵיכוֹן, קְיָם עָלַם תֵּיחֲגוּנֵּיהּ. 
יב,טו שִׁבְעַ֤ת יָמִים֙ מַצּ֣וֹת תֹּאכֵ֔לוּ אַ֚ךְ בַּיּ֣וֹם הָֽרִאשׁ֔וֹן תַּשְׁבִּ֥יתוּ שְּׂאֹ֖ר מִבָּֽתֵּיכֶ֑ם כִּ֣י ׀ כָּל־אֹכֵ֣ל חָמֵ֗ץ וְנִכְרְתָ֞ה הַנֶּ֤פֶשׁ הַהִוא֙ מִיִּשְׂרָאֵ֔ל מִיּ֥וֹם הָֽרִאשֹׁ֖ן עַד־י֥וֹם הַשְּׁבִעִֽי׃ שִׁבְעָא יוֹמִין, פַּטִּירָא תֵּיכְלוּן--בְּרַם בְּיוֹמָא קַדְמָאָה, תְּבַטְּלוּן חֲמִירָא מִבָּתֵּיכוֹן:  אֲרֵי כָּל דְּיֵיכוֹל חֲמִיעַ, וְיִשְׁתֵּיצֵי אֲנָשָׁא הַהוּא מִיִּשְׂרָאֵל--מִיּוֹמָא קַדְמָאָה, עַד יוֹמָא שְׁבִיעָאָה. 
יב,טז וּבַיּ֤וֹם הָֽרִאשׁוֹן֙ מִקְרָא־קֹ֔דֶשׁ וּבַיּוֹם֙ הַשְּׁבִיעִ֔י מִקְרָא־קֹ֖דֶשׁ יִֽהְיֶ֣ה לָכֶ֑ם כָּל־מְלָאכָה֙ לֹֽא־יֵעָשֶׂ֣ה בָהֶ֔ם אַ֚ךְ אֲשֶׁ֣ר יֵֽאָכֵ֣ל לְכָל־נֶ֔פֶשׁ ה֥וּא לְבַדּ֖וֹ יֵֽעָשֶׂ֥ה לָכֶֽם׃ וּבְיוֹמָא קַדְמָאָה, מְעָרַע קַדִּישׁ, וּבְיוֹמָא שְׁבִיעָאָה, מְעָרַע קַדִּישׁ יְהֵי לְכוֹן:  כָּל עֲבִידָא, לָא יִתְעֲבֵיד בְּהוֹן--בְּרַם מָא דְּמִתְאֲכִיל לְכָל נְפַשׁ, הוּא בִּלְחוֹדוֹהִי יִתְעֲבֵיד לְכוֹן. 
יב,יז וּשְׁמַרְתֶּם֮ אֶת־הַמַּצּוֹת֒ כִּ֗י בְּעֶ֨צֶם֙ הַיּ֣וֹם הַזֶּ֔ה הוֹצֵ֥אתִי אֶת־צִבְאֽוֹתֵיכֶ֖ם מֵאֶ֣רֶץ מִצְרָ֑יִם וּשְׁמַרְתֶּ֞ם אֶת־הַיּ֥וֹם הַזֶּ֛ה לְדֹרֹֽתֵיכֶ֖ם חֻקַּ֥ת עוֹלָֽם׃ וְתִטְּרוּן, יָת פַּטִּירָא, אֲרֵי בִּכְרַן יוֹמָא הָדֵין, אַפֵּיקִית יָת חֵילֵיכוֹן מֵאַרְעָא דְּמִצְרָיִם; וְתִטְּרוּן יָת יוֹמָא הָדֵין, לְדָרֵיכוֹן--קְיָם עָלַם. 
יב,יח בָּֽרִאשֹׁ֡ן בְּאַרְבָּעָה֩ עָשָׂ֨ר י֤וֹם לַחֹ֨דֶשׁ֙ בָּעֶ֔רֶב תֹּֽאכְל֖וּ מַצֹּ֑ת עַ֠ד י֣וֹם הָֽאֶחָ֧ד וְעֶשְׂרִ֛ים לַחֹ֖דֶשׁ בָּעָֽרֶב׃ בְּנִיסָן בְּאַרְבְּעַת עַסְרָא יוֹמָא לְיַרְחָא, בְּרַמְשָׁא, תֵּיכְלוּן, פַּטִּירָא:  עַד יוֹמָא חַד וְעַסְרִין, לְיַרְחָא--בְּרַמְשָׁא. 
יב,יט שִׁבְעַ֣ת יָמִ֔ים שְׂאֹ֕ר לֹ֥א יִמָּצֵ֖א בְּבָֽתֵּיכֶ֑ם כִּ֣י ׀ כָּל־אֹכֵ֣ל מַחְמֶ֗צֶת וְנִכְרְתָ֞ה הַנֶּ֤פֶשׁ הַהִוא֙ מֵֽעֲדַ֣ת יִשְׂרָאֵ֔ל בַּגֵּ֖ר וּבְאֶזְרַ֥ח הָאָֽרֶץ׃ שִׁבְעָא יוֹמִין--חֲמִירָא, לָא יִשְׁתְּכַח בְּבָתֵּיכוֹן:  אֲרֵי כָּל דְּיֵיכוֹל מַחְמְעָא, וְיִשְׁתֵּיצֵי אֲנָשָׁא הַהוּא מִכְּנִשְׁתָּא דְּיִשְׂרָאֵל--בְּגִיּוֹרַיָּא, וּבְיַצִּיבַיָּא דְּאַרְעָא. 
יב,כ כָּל־מַחְמֶ֖צֶת לֹ֣א תֹאכֵ֑לוּ בְּכֹל֙ מוֹשְׁבֹ֣תֵיכֶ֔ם תֹּֽאכְל֖וּ מַצּֽוֹת׃ {פ} כָּל מַחְמְעָא, לָא תֵּיכְלוּן; בְּכֹל, מוֹתְבָנֵיכוֹן, תֵּיכְלוּן, פַּטִּירָא. {פ}
יב,כא וַיִּקְרָ֥א מֹשֶׁ֛ה לְכָל־זִקְנֵ֥י יִשְׂרָאֵ֖ל וַיֹּ֣אמֶר אֲלֵהֶ֑ם מִֽשְׁכ֗וּ וּקְח֨וּ לָכֶ֥ם צֹ֛אן לְמִשְׁפְּחֹֽתֵיכֶ֖ם וְשַֽׁחֲט֥וּ הַפָּֽסַח׃ וּקְרָא מֹשֶׁה לְכָל סָבֵי יִשְׂרָאֵל, וַאֲמַר לְהוֹן:  אִתְנְגִידוּ, וְסַבוּ לְכוֹן מִן בְּנֵי עָנָא לְזַרְעֲיָתְכוֹן--וְכוּסוּ פִּסְחָא. 
יב,כב וּלְקַחְתֶּ֞ם אֲגֻדַּ֣ת אֵז֗וֹב וּטְבַלְתֶּם֮ בַּדָּ֣ם אֲשֶׁר־בַּסַּף֒ וְהִגַּעְתֶּ֤ם אֶל־הַמַּשְׁקוֹף֙ וְאֶל־שְׁתֵּ֣י הַמְּזוּזֹ֔ת מִן־הַדָּ֖ם אֲשֶׁ֣ר בַּסָּ֑ף וְאַתֶּ֗ם לֹ֥א תֵֽצְא֛וּ אִ֥ישׁ מִפֶּֽתַח־בֵּית֖וֹ עַד־בֹּֽקֶר׃ וְתִסְּבוּן אֲסָרַת אֵיזוֹבָא, וְתִטְבְּלוּן בִּדְמָא דִּבְמָנָא, וְתַדּוֹן לְשָׁקְפָא וְלִתְרֵין סִפַּיָּא, מִן דְּמָא דִּבְמָנָא; וְאַתּוּן, לָא תִּפְּקוּן אֲנָשׁ מִתְּרַע בֵּיתֵיהּ--עַד צַפְרָא. 
יב,כג וְעָבַ֣ר יְהוָה֮ לִנְגֹּ֣ף אֶת־מִצְרַיִם֒ וְרָאָ֤ה אֶת־הַדָּם֙ עַל־הַמַּשְׁק֔וֹף וְעַ֖ל שְׁתֵּ֣י הַמְּזוּזֹ֑ת וּפָסַ֤ח יְהוָה֙ עַל־הַפֶּ֔תַח וְלֹ֤א יִתֵּן֙ הַמַּשְׁחִ֔ית לָבֹ֥א אֶל־בָּֽתֵּיכֶ֖ם לִנְגֹּֽף׃ וְיִתְגְּלֵי יְיָ, לְמִמְחֵי יָת מִצְרָאֵי, וְיִחְזֵי יָת דְּמָא עַל שָׁקְפָא, וְעַל תְּרֵין סִפַּיָּא; וְיֵיחוּס יְיָ, עַל תַּרְעָא, וְלָא יִשְׁבּוֹק מְחַבְּלָא, לְמֵיעַל לְבָתֵּיכוֹן לְמִמְחֵי. 
יב,כד וּשְׁמַרְתֶּ֖ם אֶת־הַדָּבָ֣ר הַזֶּ֑ה לְחָק־לְךָ֥ וּלְבָנֶ֖יךָ עַד־עוֹלָֽם׃ וְתִטְּרוּן, יָת פִּתְגָמָא הָדֵין, לִקְיָם לָךְ וְלִבְנָךְ, עַד עָלְמָא. 
יב,כה וְהָיָ֞ה כִּֽי־תָבֹ֣אוּ אֶל־הָאָ֗רֶץ אֲשֶׁ֨ר יִתֵּ֧ן יְהוָ֛ה לָכֶ֖ם כַּֽאֲשֶׁ֣ר דִּבֵּ֑ר וּשְׁמַרְתֶּ֖ם אֶת־הָֽעֲבֹדָ֥ה הַזֹּֽאת׃ וִיהֵי אֲרֵי תֵּיעֲלוּן לְאַרְעָא, דְּיִתֵּין יְיָ לְכוֹן--כְּמָא דְּמַלֵּיל; וְתִטְּרוּן, יָת פֻּלְחָנָא הָדֵין. 
יב,כו וְהָיָ֕ה כִּֽי־יֹאמְר֥וּ אֲלֵיכֶ֖ם בְּנֵיכֶ֑ם מָ֛ה הָֽעֲבֹדָ֥ה הַזֹּ֖את לָכֶֽם׃ וִיהֵי, אֲרֵי יֵימְרוּן לְכוֹן בְּנֵיכוֹן:  מָא פֻּלְחָנָא הָדֵין, לְכוֹן. 
יב,כז וַֽאֲמַרְתֶּ֡ם זֶֽבַח־פֶּ֨סַח ה֜וּא לַֽיהוָ֗ה אֲשֶׁ֣ר פָּ֠סַח עַל־בָּתֵּ֤י בְנֵֽי־יִשְׂרָאֵל֙ בְּמִצְרַ֔יִם בְּנָגְפּ֥וֹ אֶת־מִצְרַ֖יִם וְאֶת־בָּתֵּ֣ינוּ הִצִּ֑יל וַיִּקֹּ֥ד הָעָ֖ם וַיִּֽשְׁתַּחֲוֽוּ׃ וְתֵימְרוּן דֵּיבַח חֲיָס הוּא קֳדָם יְיָ, דְּחָס עַל בָּתֵּי בְּנֵי יִשְׂרָאֵל בְּמִצְרַיִם, כַּד מְחָא יָת מִצְרָאֵי, וְיָת בָּתַּנָא שֵׁיזֵיב; וּכְרַע עַמָּא, וּסְגִידוּ. 
יב,כח וַיֵּֽלְכ֥וּ וַֽיַּעֲשׂ֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל כַּֽאֲשֶׁ֨ר צִוָּ֧ה יְהוָ֛ה אֶת־מֹשֶׁ֥ה וְאַֽהֲרֹ֖ן כֵּ֥ן עָשֽׂוּ׃ {ס} וַאֲזַלוּ וַעֲבַדוּ, בְּנֵי יִשְׂרָאֵל:  כְּמָא דְּפַקֵּיד יְיָ יָת מֹשֶׁה וְאַהֲרוֹן, כֵּין עֲבַדוּ. {ס}
יב,כט וַיְהִ֣י ׀ בַּֽחֲצִ֣י הַלַּ֗יְלָה וַֽיהוָה֮ הִכָּ֣ה כָל־בְּכוֹר֮ בְּאֶ֣רֶץ מִצְרַיִם֒ מִבְּכֹ֤ר פַּרְעֹה֙ הַיֹּשֵׁ֣ב עַל־כִּסְא֔וֹ עַ֚ד בְּכ֣וֹר הַשְּׁבִ֔י אֲשֶׁ֖ר בְּבֵ֣ית הַבּ֑וֹר וְכֹ֖ל בְּכ֥וֹר בְּהֵמָֽה׃ וַהֲוָה בְּפַלְגוּת לֵילְיָא, וַייָ קְטַל כָּל בֻּכְרָא בְּאַרְעָא דְּמִצְרַיִם, מִבֻּכְרָא דְּפַרְעֹה דַּעֲתִיד לְמִתַּב עַל כֻּרְסֵי מַלְכוּתֵיהּ, עַד בֻּכְרָא דְּשִׁבְיָא דִּבְבֵית אֲסִירֵי; וְכֹל, בֻּכְרָא דִּבְעִירָא. 
יב,ל וַיָּ֨קָם פַּרְעֹ֜ה לַ֗יְלָה ה֤וּא וְכָל־עֲבָדָיו֙ וְכָל־מִצְרַ֔יִם וַתְּהִ֛י צְעָקָ֥ה גְדֹלָ֖ה בְּמִצְרָ֑יִם כִּֽי־אֵ֣ין בַּ֔יִת אֲשֶׁ֥ר אֵֽין־שָׁ֖ם מֵֽת׃ וְקָם פַּרְעֹה בְּלֵילְיָא, הוּא וְכָל עַבְדּוֹהִי וְכָל מִצְרָאֵי, וַהֲוָת צְוַחְתָּא רַבְּתָא, בְּמִצְרָיִם:  אֲרֵי לֵית בֵּיתָא תַּמָּן, דְּלָא הֲוָה בֵּיהּ מִיתָא. 
יב,לא וַיִּקְרָא֩ לְמֹשֶׁ֨ה וּֽלְאַהֲרֹ֜ן לַ֗יְלָה וַיֹּ֨אמֶר֙ ק֤וּמוּ צְּאוּ֙ מִתּ֣וֹךְ עַמִּ֔י גַּם־אַתֶּ֖ם גַּם־בְּנֵ֣י יִשְׂרָאֵ֑ל וּלְכ֛וּ עִבְד֥וּ אֶת־יְהוָ֖ה כְּדַבֶּרְכֶֽם׃ וּקְרָא לְמֹשֶׁה וּלְאַהֲרוֹן בְּלֵילְיָא, וַאֲמַר קוּמוּ פּוּקוּ מִגּוֹ עַמִּי--אַף אַתּוּן, אַף בְּנֵי יִשְׂרָאֵל; וְאִיזִילוּ פְּלַחוּ קֳדָם יְיָ, כְּמָא דַּהֲוֵיתוֹן אָמְרִין. 
יב,לב גַּם־צֹֽאנְכֶ֨ם גַּם־בְּקַרְכֶ֥ם קְח֛וּ כַּֽאֲשֶׁ֥ר דִּבַּרְתֶּ֖ם וָלֵ֑כוּ וּבֵֽרַכְתֶּ֖ם גַּם־אֹתִֽי׃ אַף עָנְכוֹן אַף תּוֹרֵיכוֹן דְּבַרוּ כְּמָא דְּמַלֵּילְתּוּן, וְאִיזִילוּ; וְצַלּוֹ, אַף עֲלָי. 
יב,לג וַתֶּֽחֱזַ֤ק מִצְרַ֨יִם֙ עַל־הָעָ֔ם לְמַהֵ֖ר לְשַׁלְּחָ֣ם מִן־הָאָ֑רֶץ כִּ֥י אָֽמְר֖וּ כֻּלָּ֥נוּ מֵתִֽים׃ וּתְקִיפוּ מִצְרָאֵי עַל עַמָּא, לְאוֹחָאָה לְשַׁלָּחוּתְהוֹן מִן אַרְעָא:  אֲרֵי אֲמַרוּ, כֻּלַּנָא מָיְתִין. 
יב,לד וַיִּשָּׂ֥א הָעָ֛ם אֶת־בְּצֵק֖וֹ טֶ֣רֶם יֶחְמָ֑ץ מִשְׁאֲרֹתָ֛ם צְרֻרֹ֥ת בְּשִׂמְלֹתָ֖ם עַל־שִׁכְמָֽם׃ וּנְטַל עַמָּא יָת לֵישְׁהוֹן, עַד לָא חֲמַע; מוֹתַר אָצְוָתְהוֹן צְרִיר בִּלְבוּשֵׁיהוֹן, עַל כַּתְפֵיהוֹן. 
יב,לה וּבְנֵֽי־יִשְׂרָאֵ֥ל עָשׂ֖וּ כִּדְבַ֣ר מֹשֶׁ֑ה וַֽיִּשְׁאֲלוּ֙ מִמִּצְרַ֔יִם כְּלֵי־כֶ֛סֶף וּכְלֵ֥י זָהָ֖ב וּשְׂמָלֹֽת׃ וּבְנֵי יִשְׂרָאֵל עֲבַדוּ, כְּפִתְגָמָא דְּמֹשֶׁה; וּשְׁאִילוּ, מִמִּצְרַיִם, מָנִין דִּכְסַף וּמָנִין דִּדְהַב, וּלְבוּשִׁין. 
יב,לו וַֽיהוָ֞ה נָתַ֨ן אֶת־חֵ֥ן הָעָ֛ם בְּעֵינֵ֥י מִצְרַ֖יִם וַיַּשְׁאִל֑וּם וַֽיְנַצְּל֖וּ אֶת־מִצְרָֽיִם׃ {פ} וַייָ יְהַב יָת עַמָּא לְרַחֲמִין, בְּעֵינֵי מִצְרָאֵי--וְאַשְׁאִילוּנוּן; וְרוֹקִינוּ, יָת מִצְרָיִם. {פ}
יב,לז וַיִּסְע֧וּ בְנֵֽי־יִשְׂרָאֵ֛ל מֵֽרַעְמְסֵ֖ס סֻכֹּ֑תָה כְּשֵׁשׁ־מֵא֨וֹת אֶ֧לֶף רַגְלִ֛י הַגְּבָרִ֖ים לְבַ֥ד מִטָּֽף׃ וּנְטַלוּ בְּנֵי יִשְׂרָאֵל מֵרַעְמְסֵס, לְסֻכּוֹת, כְּשֵׁית מְאָה אַלְפִין גֻּבְרָא, רִגְלָאָה בָּר מִטַּפְלָא. 
יב,לח וְגַם־עֵ֥רֶב רַ֖ב עָלָ֣ה אִתָּ֑ם וְצֹ֣אן וּבָקָ֔ר מִקְנֶ֖ה כָּבֵ֥ד מְאֹֽד׃ וְאַף נֻכְרָאִין סַגִּיאִין, סְלִיקוּ עִמְּהוֹן, וְעָנָא וְתוֹרֵי, בְּעִירָא סַגִּי לַחְדָּא. 
יב,לט וַיֹּאפ֨וּ אֶת־הַבָּצֵ֜ק אֲשֶׁ֨ר הוֹצִ֧יאוּ מִמִּצְרַ֛יִם עֻגֹ֥ת מַצּ֖וֹת כִּ֣י לֹ֣א חָמֵ֑ץ כִּֽי־גֹרְשׁ֣וּ מִמִּצְרַ֗יִם וְלֹ֤א יָֽכְלוּ֙ לְהִתְמַהְמֵ֔הַּ וְגַם־צֵדָ֖ה לֹֽא־עָשׂ֥וּ לָהֶֽם׃ וַאֲפוֹ יָת לֵישָׁא דְּאַפִּיקוּ מִמִּצְרַיִם, גְּרִיצָן פַּטִּירָן--אֲרֵי לָא חֲמַע:  אֲרֵי אִתָּרַכוּ מִמִּצְרַיִם, וְלָא יְכִילוּ לְאִתְעַכָּבָא, וְאַף זְוָדִין, לָא עֲבַדוּ לְהוֹן. 
יב,מ וּמוֹשַׁב֙ בְּנֵ֣י יִשְׂרָאֵ֔ל אֲשֶׁ֥ר יָֽשְׁב֖וּ בְּמִצְרָ֑יִם שְׁלֹשִׁ֣ים שָׁנָ֔ה וְאַרְבַּ֥ע מֵא֖וֹת שָׁנָֽה׃ וּמוֹתַב בְּנֵי יִשְׂרָאֵל, דִּיתִיבוּ בְּמִצְרָיִם--אַרְבַּע מְאָה, וּתְלָתִין שְׁנִין. 
יב,מא וַיְהִ֗י מִקֵּץ֙ שְׁלֹשִׁ֣ים שָׁנָ֔ה וְאַרְבַּ֥ע מֵא֖וֹת שָׁנָ֑ה וַיְהִ֗י בְּעֶ֨צֶם֙ הַיּ֣וֹם הַזֶּ֔ה יָ֥צְא֛וּ כָּל־צִבְא֥וֹת יְהוָ֖ה מֵאֶ֥רֶץ מִצְרָֽיִם׃ וַהֲוָה, מִסּוֹף אַרְבַּע מְאָה, וּתְלָתִין, שְׁנִין; וַהֲוָה, בִּכְרַן יוֹמָא הָדֵין, נְפַקוּ כָּל חֵילַיָּא דַּייָ, מֵאַרְעָא דְּמִצְרָיִם. 
יב,מב לֵ֣יל שִׁמֻּרִ֥ים הוּא֙ לַֽיהוָ֔ה לְהֽוֹצִיאָ֖ם מֵאֶ֣רֶץ מִצְרָ֑יִם הֽוּא־הַלַּ֤יְלָה הַזֶּה֙ לַֽיהוָ֔ה שִׁמֻּרִ֛ים לְכָל־בְּנֵ֥י יִשְׂרָאֵ֖ל לְדֹֽרֹתָֽם׃ {פ} לֵילֵי נְטִיר הוּא קֳדָם יְיָ, לְאַפָּקוּתְהוֹן מֵאַרְעָא דְּמִצְרָיִם:  הוּא לֵילְיָא הָדֵין קֳדָם יְיָ, נְטִיר לְכָל בְּנֵי יִשְׂרָאֵל לְדָרֵיהוֹן. {פ}
יב,מג וַיֹּ֤אמֶר יְהוָה֙ אֶל־מֹשֶׁ֣ה וְאַֽהֲרֹ֔ן זֹ֖את חֻקַּ֣ת הַפָּ֑סַח כָּל־בֶּן־נֵכָ֖ר לֹא־יֹ֥אכַל בּֽוֹ׃ וַאֲמַר יְיָ לְמֹשֶׁה וְאַהֲרוֹן, דָּא גְּזֵירַת פִּסְחָא:  כָּל בַּר יִשְׂרָאֵל דְּיִשְׁתַּמַּד, לָא יֵיכוֹל בֵּיהּ. 
יב,מד וְכָל־עֶ֥בֶד אִ֖ישׁ מִקְנַת־כָּ֑סֶף וּמַלְתָּ֣ה אֹת֔וֹ אָ֖ז יֹ֥אכַל בּֽוֹ׃ וְכָל עֶבֶד גְּבַר, זְבִין כַּסְפָּא--וְתִגְזַר יָתֵיהּ, בְּכֵין יֵיכוֹל בֵּיהּ. 
יב,מה תּוֹשָׁ֥ב וְשָׂכִ֖יר לֹא־יֹ֥אכַל בּֽוֹ׃ תּוֹתָבָא וַאֲגִירָא, לָא יֵיכוֹל בֵּיהּ. 
יב,מו בְּבַ֤יִת אֶחָד֙ יֵֽאָכֵ֔ל לֹֽא־תוֹצִ֧יא מִן־הַבַּ֛יִת מִן־הַבָּשָׂ֖ר ח֑וּצָה וְעֶ֖צֶם לֹ֥א תִשְׁבְּרוּ־בֽוֹ׃ בַּחֲבוּרָא חֲדָא יִתְאֲכִיל, לָא תַּפְּקוּן מִן בֵּיתָא מִן בִּסְרָא לְבָרָא; וְגַרְמָא, לָא תִּתְבְּרוּן בֵּיהּ. 
יב,מז כָּל־עֲדַ֥ת יִשְׂרָאֵ֖ל יַֽעֲשׂ֥וּ אֹתֽוֹ׃ כָּל כְּנִשְׁתָּא דְּיִשְׂרָאֵל, יַעְבְּדוּן יָתֵיהּ. 
יב,מח וְכִֽי־יָג֨וּר אִתְּךָ֜ גֵּ֗ר וְעָ֣שָׂה פֶסַח֮ לַֽיהוָה֒ הִמּ֧וֹל ל֣וֹ כָל־זָכָ֗ר וְאָז֙ יִקְרַ֣ב לַֽעֲשֹׂת֔וֹ וְהָיָ֖ה כְּאֶזְרַ֣ח הָאָ֑רֶץ וְכָל־עָרֵ֖ל לֹֽא־יֹ֥אכַל בּֽוֹ׃ וַאֲרֵי יִתְגַיַּר עִמְּכוֹן גִּיּוֹרָא, וְיַעֲבֵיד פִּסְחָא קֳדָם יְיָ--מִגְזַר לֵיהּ כָּל דְּכוּרָא וּבְכֵין יִקְרַב לְמִעְבְּדֵיהּ, וִיהֵי כְּיַצִּיבֵי אַרְעָא; וְכָל עַרְלָא, לָא יֵיכוֹל בֵּיהּ. 
יב,מט תּוֹרָ֣ה אַחַ֔ת יִֽהְיֶ֖ה לָֽאֶזְרָ֑ח וְלַגֵּ֖ר הַגָּ֥ר בְּתֽוֹכְכֶֽם׃ אוֹרָיְתָא חֲדָא, תְּהֵי לְיַצִּיבַיָּא, וּלְגִיּוֹרַיָּא, דְּיִתְגַיְּרוּן בֵּינֵיכוֹן. 
יב,נ וַֽיַּעֲשׂ֖וּ כָּל־בְּנֵ֣י יִשְׂרָאֵ֑ל כַּֽאֲשֶׁ֨ר צִוָּ֧ה יְהוָ֛ה אֶת־מֹשֶׁ֥ה וְאֶֽת־אַהֲרֹ֖ן כֵּ֥ן עָשֽׂוּ׃ {ס} וַעֲבַדוּ, כָּל בְּנֵי יִשְׂרָאֵל:  כְּמָא דְּפַקֵּיד יְיָ יָת מֹשֶׁה וְיָת אַהֲרוֹן, כֵּין עֲבַדוּ. {ס}
יב,נא וַיְהִ֕י בְּעֶ֖צֶם הַיּ֣וֹם הַזֶּ֑ה הוֹצִ֨יא יְהוָ֜ה אֶת־בְּנֵ֧י יִשְׂרָאֵ֛ל מֵאֶ֥רֶץ מִצְרַ֖יִם עַל־צִבְאֹתָֽם׃ {פ} וַהֲוָה, בִּכְרַן יוֹמָא הָדֵין:  אַפֵּיק יְיָ יָת בְּנֵי יִשְׂרָאֵל, מֵאַרְעָא דְּמִצְרַיִם--עַל חֵילֵיהוֹן. {פ}
יג,א וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
יג,ב קַדֶּשׁ־לִ֨י כָל־בְּכ֜וֹר פֶּ֤טֶר כָּל־רֶ֨חֶם֙ בִּבְנֵ֣י יִשְׂרָאֵ֔ל בָּֽאָדָ֖ם וּבַבְּהֵמָ֑ה לִ֖י הֽוּא׃ אַקְדֵּישׁ קֳדָמַי כָּל בֻּכְרָא פָּתַח כָּל וַלְדָּא, בִּבְנֵי יִשְׂרָאֵל--בַּאֲנָשָׁא, וּבִבְעִירָא:  דִּילִי, הוּא. 
יג,ג וַיֹּ֨אמֶר מֹשֶׁ֜ה אֶל־הָעָ֗ם זָכ֞וֹר אֶת־הַיּ֤וֹם הַזֶּה֙ אֲשֶׁ֨ר יְצָאתֶ֤ם מִמִּצְרַ֨יִם֙ מִבֵּ֣ית עֲבָדִ֔ים כִּ֚י בְּחֹ֣זֶק יָ֔ד הוֹצִ֧יא יְהוָ֛ה אֶתְכֶ֖ם מִזֶּ֑ה וְלֹ֥א יֵֽאָכֵ֖ל חָמֵֽץ׃ וַאֲמַר מֹשֶׁה לְעַמָּא, הֲווֹ דְּכִירִין יָת יוֹמָא הָדֵין דִּנְפַקְתּוּן מִמִּצְרַיִם מִבֵּית עַבְדּוּתָא, אֲרֵי בִּתְקוֹף יַד, אַפֵּיק יְיָ יָתְכוֹן מִכָּא; וְלָא יִתְאֲכִיל, חֲמִיעַ. 
יג,ד הַיּ֖וֹם אַתֶּ֣ם יֹֽצְאִ֑ים בְּחֹ֖דֶשׁ הָֽאָבִֽיב׃ יוֹמָא דֵּין, אַתּוּן נָפְקִין, בְּיַרְחָא, דַּאֲבִיבָא. 
יג,ה וְהָיָ֣ה כִֽי־יְבִֽיאֲךָ֣ יְהוָ֡ה אֶל־אֶ֣רֶץ הַֽ֠כְּנַעֲנִי וְהַֽחִתִּ֨י וְהָֽאֱמֹרִ֜י וְהַֽחִוִּ֣י וְהַיְבוּסִ֗י אֲשֶׁ֨ר נִשְׁבַּ֤ע לַֽאֲבֹתֶ֨יךָ֙ לָ֣תֶת לָ֔ךְ אֶ֛רֶץ זָבַ֥ת חָלָ֖ב וּדְבָ֑שׁ וְעָֽבַדְתָּ֛ אֶת־הָֽעֲבֹדָ֥ה הַזֹּ֖את בַּחֹ֥דֶשׁ הַזֶּֽה׃ וִיהֵי אֲרֵי יַעֵילִנָּךְ יְיָ לַאֲרַע כְּנַעֲנָאֵי וְחִתָּאֵי וֶאֱמוֹרָאֵי וְחִוָּאֵי וִיבוּסָאֵי, דְּקַיֵּים לַאֲבָהָתָךְ לְמִתַּן לָךְ, אֲרַע עָבְדָא חֲלָב, וּדְבַשׁ; וְתִפְלַח יָת פֻּלְחָנָא הָדָא, בְּיַרְחָא הָדֵין. 
יג,ו שִׁבְעַ֥ת יָמִ֖ים תֹּאכַ֣ל מַצֹּ֑ת וּבַיּוֹם֙ הַשְּׁבִיעִ֔י חַ֖ג לַֽיהוָֽה׃ שִׁבְעָא יוֹמִין, תֵּיכוֹל פַּטִּירָא; וּבְיוֹמָא, שְׁבִיעָאָה, חַגָּא, קֳדָם יְיָ. 
יג,ז מַצּוֹת֙ יֵֽאָכֵ֔ל אֵ֖ת שִׁבְעַ֣ת הַיָּמִ֑ים וְלֹֽא־יֵרָאֶ֨ה לְךָ֜ חָמֵ֗ץ וְלֹֽא־יֵרָאֶ֥ה לְךָ֛ שְׂאֹ֖ר בְּכָל־גְּבֻלֶֽךָ׃ פַּטִּירָא, יִתְאֲכִיל, יָת, שִׁבְעָא יוֹמִין; וְלָא יִתַּחְזֵי לָךְ חֲמִיעַ, וְלָא יִתַּחְזֵי לָךְ חֲמִיר--בְּכָל תְּחוּמָךְ. 
יג,ח וְהִגַּדְתָּ֣ לְבִנְךָ֔ בַּיּ֥וֹם הַה֖וּא לֵאמֹ֑ר בַּֽעֲב֣וּר זֶ֗ה עָשָׂ֤ה יְהוָה֙ לִ֔י בְּצֵאתִ֖י מִמִּצְרָֽיִם׃ וּתְחַוֵּי לִבְרָךְ, בְּיוֹמָא הַהוּא לְמֵימַר:  בְּדִיל דָּא, עֲבַד יְיָ לִי, בְּמִפְּקִי, מִמִּצְרָיִם. 
יג,ט וְהָיָה֩ לְךָ֨ לְא֜וֹת עַל־יָֽדְךָ֗ וּלְזִכָּרוֹן֙ בֵּ֣ין עֵינֶ֔יךָ לְמַ֗עַן תִּֽהְיֶ֛ה תּוֹרַ֥ת יְהוָ֖ה בְּפִ֑יךָ כִּ֚י בְּיָ֣ד חֲזָקָ֔ה הוֹצִֽאֲךָ֥ יְהוָ֖ה מִמִּצְרָֽיִם׃ וִיהֵי לָךְ לְאָת עַל יְדָךְ, וּלְדֻכְרָן בֵּין עֵינָךְ, בְּדִיל דִּתְהֵי אוֹרָיְתָא דַּייָ, בְּפֻמָּךְ:  אֲרֵי בְּיַד תַּקִּיפָא, אַפְּקָךְ יְיָ מִמִּצְרָיִם. 
יג,י וְשָֽׁמַרְתָּ֛ אֶת־הַחֻקָּ֥ה הַזֹּ֖את לְמֽוֹעֲדָ֑הּ מִיָּמִ֖ים יָמִֽימָה׃ {פ} וְתִטַּר יָת קְיָמָא הָדֵין, בְּזִמְנֵיהּ, מִזְּמָן, לִזְמָן. {פ}
יג,יא וְהָיָ֞ה כִּֽי־יְבִֽאֲךָ֤ יְהוָה֙ אֶל־אֶ֣רֶץ הַֽכְּנַעֲנִ֔י כַּֽאֲשֶׁ֛ר נִשְׁבַּ֥ע לְךָ֖ וְלַֽאֲבֹתֶ֑יךָ וּנְתָנָ֖הּ לָֽךְ׃ וִיהֵי אֲרֵי יַעֵילִנָּךְ יְיָ, לַאֲרַע כְּנַעֲנָאֵי, כְּמָא דְּקַיֵּים לָךְ, וְלַאֲבָהָתָךְ; וְיִתְּנַהּ, לָךְ. 
יג,יב וְהַֽעֲבַרְתָּ֥ כָל־פֶּֽטֶר־רֶ֖חֶם לַֽיהוָ֑ה וְכָל־פֶּ֣טֶר ׀ שֶׁ֣גֶר בְּהֵמָ֗ה אֲשֶׁ֨ר יִֽהְיֶ֥ה לְךָ֛ הַזְּכָרִ֖ים לַֽיהוָֽה׃ וְתַעְבַּר כָּל פָּתַח וַלְדָּא, קֳדָם יְיָ; וְכָל פָּתַח וְלַד בְּעִירָא, דִּיהוֹן לָךְ דִּכְרִין--תַּקְדֵּישׁ קֳדָם יְיָ. 
יג,יג וְכָל־פֶּ֤טֶר חֲמֹר֙ תִּפְדֶּ֣ה בְשֶׂ֔ה וְאִם־לֹ֥א תִפְדֶּ֖ה וַֽעֲרַפְתּ֑וֹ וְכֹ֨ל בְּכ֥וֹר אָדָ֛ם בְּבָנֶ֖יךָ תִּפְדֶּֽה׃ וְכָל בֻּכְרָא דִּחְמָרָא תִּפְרוּק בְּאִמְּרָא, וְאִם לָא תִּפְרוּק וְתִקְפֵיהּ; וְכֹל בֻּכְרָא דַּאֲנָשָׁא בִּבְנָךְ, תִּפְרוּק. 
יג,יד וְהָיָ֞ה כִּֽי־יִשְׁאָלְךָ֥ בִנְךָ֛ מָחָ֖ר לֵאמֹ֣ר מַה־זֹּ֑את וְאָֽמַרְתָּ֣ אֵלָ֔יו בְּחֹ֣זֶק יָ֗ד הֽוֹצִיאָ֧נוּ יְהוָ֛ה מִמִּצְרַ֖יִם מִבֵּ֥ית עֲבָדִֽים׃ וִיהֵי אֲרֵי יִשְׁאֲלִנָּךְ בְּרָךְ, מְחַר--לְמֵימַר מָא דָּא:  וְתֵימַר לֵיהּ--בִּתְקוֹף יַד אַפְּקַנָא יְיָ מִמִּצְרַיִם, מִבֵּית עַבְדּוּתָא. 
יג,טו וַיְהִ֗י כִּֽי־הִקְשָׁ֣ה פַרְעֹה֮ לְשַׁלְּחֵנוּ֒ וַיַּֽהֲרֹ֨ג יְהוָ֤ה כָּל־בְּכוֹר֙ בְּאֶ֣רֶץ מִצְרַ֔יִם מִבְּכֹ֥ר אָדָ֖ם וְעַד־בְּכ֣וֹר בְּהֵמָ֑ה עַל־כֵּן֩ אֲנִ֨י זֹבֵ֜חַ לַֽיהוָ֗ה כָּל־פֶּ֤טֶר רֶ֨חֶם֙ הַזְּכָרִ֔ים וְכָל־בְּכ֥וֹר בָּנַ֖י אֶפְדֶּֽה׃ וַהֲוָה, כַּד אַקְשִׁי פַּרְעֹה לְשַׁלָּחוּתַנָא, וּקְטַל יְיָ כָּל בֻּכְרָא בְּאַרְעָא דְּמִצְרַיִם, מִבֻּכְרָא דַּאֲנָשָׁא וְעַד בֻּכְרָא דִּבְעִירָא; עַל כֵּין אֲנָא דָּבַח קֳדָם יְיָ, כָּל פָּתַח וְלַד דִּכְרַיָּא, וְכָל בֻּכְרָא דִּבְנַי, אֶפְרוּק. 
יג,טז וְהָיָ֤ה לְאוֹת֙ עַל־יָ֣דְכָ֔ה וּלְטֽוֹטָפֹ֖ת בֵּ֣ין עֵינֶ֑יךָ כִּ֚י בְּחֹ֣זֶק יָ֔ד הֽוֹצִיאָ֥נוּ יְהוָ֖ה מִמִּצְרָֽיִם׃ {ס} וִיהֵי לְאָת עַל יְדָךְ, וְלִתְפִלִּין בֵּין עֵינָךְ:  אֲרֵי בִּתְקוֹף יַד, אַפְּקַנָא יְיָ מִמִּצְרָיִם. {ס}
