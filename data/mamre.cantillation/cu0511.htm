וזאת הברכה

לג,א וְזֹ֣את הַבְּרָכָ֗ה אֲשֶׁ֨ר בֵּרַ֥ךְ מֹשֶׁ֛ה אִ֥ישׁ הָֽאֱלֹהִ֖ים אֶת־בְּנֵ֣י יִשְׂרָאֵ֑ל לִפְנֵ֖י מוֹתֽוֹ׃ וְדָא בִּרְכְתָא, דְּבָרֵיךְ מֹשֶׁה נְבִיָּא דַּייָ--יָת בְּנֵי יִשְׂרָאֵל:  קֳדָם, מוֹתֵיהּ. 
לג,ב וַיֹּאמַ֗ר יְהוָ֞ה מִסִּינַ֥י בָּא֙ וְזָרַ֤ח מִשֵּׂעִיר֙ לָ֔מוֹ הוֹפִ֨יעַ֙ מֵהַ֣ר פָּארָ֔ן וְאָתָ֖ה מֵֽרִבְבֹ֣ת קֹ֑דֶשׁ מִֽימִינ֕וֹ אשדת (אֵ֥שׁ דָּ֖ת) לָֽמוֹ׃ וַאֲמַר, יְיָ מִסִּינַי אִתְגְּלִי וְזֵיהוֹר יְקָרֵיהּ מִשֵּׂעִיר אִתַּחְזִי לַנָא--אִתְגְּלִי בִּגְבוּרְתֵיהּ עַל טוּרָא דְּפָארָן, וְעִמֵּיהּ רִבְּוָת קַדִּישִׁין; כְּתָב יַמִּינֵיהּ, מִגּוֹ אִישָׁתָא אוֹרָיְתָא יְהַב לַנָא. 
לג,ג אַ֚ף חֹבֵ֣ב עַמִּ֔ים כָּל־קְדֹשָׁ֖יו בְּיָדֶ֑ךָ וְהֵם֙ תֻּכּ֣וּ לְרַגְלֶ֔ךָ יִשָּׂ֖א מִדַּבְּרֹתֶֽיךָ׃ אַף חַבֵּיבִנּוּן לְשִׁבְטַיָּא, כָּל קַדִּישׁוֹהִי בֵּית יִשְׂרָאֵל בִּגְבוּרָא אַפֵּיקִנּוּן מִמִּצְרָיִם; וְאִנּוּן מִדַּבְּרִין תְּחוֹת עֲנָנָךְ, נָטְלִין עַל מֵימְרָךְ. 
לג,ד תּוֹרָ֥ה צִוָּה־לָ֖נוּ מֹשֶׁ֑ה מֽוֹרָשָׁ֖ה קְהִלַּ֥ת יַֽעֲקֹֽב׃ אוֹרָיְתָא יְהַב לַנָא, מֹשֶׁה:  מַסְרַהּ יְרֻתָּא, לְכִנְשָׁת יַעֲקוֹב. 
לג,ה וַיְהִ֥י בִֽישֻׁר֖וּן מֶ֑לֶךְ בְּהִתְאַסֵּף֙ רָ֣אשֵׁי עָ֔ם יַ֖חַד שִׁבְטֵ֥י יִשְׂרָאֵֽל׃ וַהֲוָה בְּיִשְׂרָאֵל, מַלְכָּא, בְּאִתְכַּנָּשׁוּת רֵישֵׁי עַמָּא, כַּחְדָּא שִׁבְטַיָּא דְּיִשְׂרָאֵל. 
לג,ו יְחִ֥י רְאוּבֵ֖ן וְאַל־יָמֹ֑ת וִיהִ֥י מְתָ֖יו מִסְפָּֽר׃ {ס} יֵיחֵי רְאוּבֵן בְּחַיֵּי עָלְמָא, וּמוֹתָא תִּנְיָנָא לָא יְמוּת; וִיקַבְּלוּן בְּנוֹהִי, אַחְסָנַתְהוֹן בְּמִנְיָנְהוֹן. {ס}
לג,ז וְזֹ֣את לִֽיהוּדָה֮ וַיֹּאמַר֒ שְׁמַ֤ע יְהוָה֙ ק֣וֹל יְהוּדָ֔ה וְאֶל־עַמּ֖וֹ תְּבִיאֶ֑נּוּ יָדָיו֙ רָ֣ב ל֔וֹ וְעֵ֥זֶר מִצָּרָ֖יו תִּֽהְיֶֽה׃ {פ} וְדָא לִיהוּדָה, וַאֲמַר, קַבֵּיל יְיָ צְלוֹתֵיהּ דִּיהוּדָה בְּמִפְּקֵיהּ לִקְרָבָא, וּלְעַמֵּיהּ תָּתִיבִנֵּיהּ בִּשְׁלָם; יְדוֹהִי יַעְבְּדָן לֵיהּ פֻּרְעָנוּתָא מִסָּנְאוֹהִי, וְסָעֵיד מִבְּעֵיל דְּבָבָא הֱוִי לֵיהּ. {פ}
לג,ח וּלְלֵוִ֣י אָמַ֔ר תֻּמֶּ֥יךָ וְאוּרֶ֖יךָ לְאִ֣ישׁ חֲסִידֶ֑ךָ אֲשֶׁ֤ר נִסִּיתוֹ֙ בְּמַסָּ֔ה תְּרִיבֵ֖הוּ עַל־מֵ֥י מְרִיבָֽה׃ וּלְלֵוִי אֲמַר, תֻּמַּיָּא וְאוּרַיָּא אַלְבֵּישְׁתָּא לִגְבַר דְּאִשְׁתְּכַח חַסִּיד קֳדָמָךְ, דְּנַסִּיתָהִי בְּנִסֵּיתָא וַהֲוָה שְׁלִים, בְּחַנְתָּהִי עַל מֵי מַצּוּתָא וְאִשְׁתְּכַח מְהֵימָן. 
לג,ט הָֽאֹמֵ֞ר לְאָבִ֤יו וּלְאִמּוֹ֙ לֹ֣א רְאִיתִ֔יו וְאֶת־אֶחָיו֙ לֹ֣א הִכִּ֔יר וְאֶת־בָּנָ֖ו לֹ֣א יָדָ֑ע כִּ֤י שָֽׁמְרוּ֙ אִמְרָתֶ֔ךָ וּבְרִֽיתְךָ֖ יִנְצֹֽרוּ׃ דְּעַל אֲבוּהִי וְעַל אִמֵּיהּ, לָא רְחֵים כַּד חָבוּ מִן דִּינָא, וְאַפֵּי אֲחוֹהִי וּבְנוֹהִי, לָא נְסֵיב:  אֲרֵי נְטַרוּ מַטְּרַת מֵימְרָךְ, וּקְיָמָךְ לָא אַשְׁנִיאוּ. 
לג,י יוֹר֤וּ מִשְׁפָּטֶ֨יךָ֙ לְיַֽעֲקֹ֔ב וְתוֹרָֽתְךָ֖ לְיִשְׂרָאֵ֑ל יָשִׂ֤ימוּ קְטוֹרָה֙ בְּאַפֶּ֔ךָ וְכָלִ֖יל עַֽל־מִזְבְּחֶֽךָ׃ כָּשְׁרִין אִלֵּין דְּיַלְּפוּן דִּינָךְ לְיַעֲקוֹב, וְאוֹרָיְתָךְ לְיִשְׂרָאֵל; יְשַׁוּוֹן קְטֹרֶת בֻּסְמִין קֳדָמָךְ, וּגְמִיר לְרַעֲוָא עַל מַדְבְּחָךְ. 
לג,יא בָּרֵ֤ךְ יְהוָה֙ חֵיל֔וֹ וּפֹ֥עַל יָדָ֖יו תִּרְצֶ֑ה מְחַ֨ץ מָתְנַ֧יִם קָמָ֛יו וּמְשַׂנְאָ֖יו מִן־יְקוּמֽוּן׃ {ס} בָּרֵיךְ יְיָ נִכְסוֹהִי, וְקֻרְבַּן יְדוֹהִי קַבֵּיל בְּרַעֲוָא; תַּבַּר חַרְצָא דְּסָנְאוֹהִי, וּדְבַעֲלֵי דְּבָבוֹהִי דְּלָא יְקוּמוּן. {ס}
לג,יב לְבִנְיָמִ֣ן אָמַ֔ר יְדִ֣יד יְהוָ֔ה יִשְׁכֹּ֥ן לָבֶ֖טַח עָלָ֑יו חֹפֵ֤ף עָלָיו֙ כָּל־הַיּ֔וֹם וּבֵ֥ין כְּתֵפָ֖יו שָׁכֵֽן׃ {ס} לְבִנְיָמִין אֲמַר--רְחִימָא דַּייָ, יִשְׁרֵי לְרֻחְצָן עֲלוֹהִי; יְהֵי מַגֵּין עֲלוֹהִי כָּל יוֹמָא, וּבְאַרְעֵיהּ תִּשְׁרֵי שְׁכִינְתָא. {ס}
לג,יג וּלְיוֹסֵ֣ף אָמַ֔ר מְבֹרֶ֥כֶת יְהוָ֖ה אַרְצ֑וֹ מִמֶּ֤גֶד שָׁמַ֨יִם֙ מִטָּ֔ל וּמִתְּה֖וֹם רֹבֶ֥צֶת תָּֽחַת׃ וּלְיוֹסֵף אֲמַר, מְבָרְכָא מִן קֳדָם יְיָ אַרְעֵיהּ, עָבְדָא מַגְדָּנִין מִטַּלָּא דִּשְׁמַיָּא מִלְּעֵילָא, וּמִמַּבּוּעֵי עֵינָוָן וּתְהוֹמִין דְּנָגְדִין מִמַּעֲמַקֵּי אַרְעָא מִלְּרַע. 
לג,יד וּמִמֶּ֖גֶד תְּבוּאֹ֣ת שָׁ֑מֶשׁ וּמִמֶּ֖גֶד גֶּ֥רֶשׁ יְרָחִֽים׃ עָבְדָא מַגְדָּנִין, וְעַלְלָן מִיְּבוּל שִׁמְשָׁא; עָבְדָא מַגְדָּנִין, מֵרֵישׁ יְרַח בִּירַח. 
לג,טו וּמֵרֹ֖אשׁ הַרְרֵי־קֶ֑דֶם וּמִמֶּ֖גֶד גִּבְע֥וֹת עוֹלָֽם׃ וּמֵרֵישׁ, טוּרַיָּא בַּכִּירַיָּא; וּמִטּוּב, רָמָן דְּלָא פָּסְקָן. 
לג,טז וּמִמֶּ֗גֶד אֶ֚רֶץ וּמְלֹאָ֔הּ וּרְצ֥וֹן שֹֽׁכְנִ֖י סְנֶ֑ה תָּב֨וֹאתָה֙ לְרֹ֣אשׁ יוֹסֵ֔ף וּלְקָדְקֹ֖ד נְזִ֥יר אֶחָֽיו׃ וּמִטּוּב, אַרְעָא וּמְלָאַהּ, רְעֵי לֵיהּ דִּשְׁכִינְתֵיהּ בִּשְׁמַיָּא, וְעַל מֹשֶׁה אִתְגְּלִי בַּאֲסַנָּא; יֵיתְיָן כָּל אִלֵּין לְרֵישָׁא דְּיוֹסֵף, גֻּבְרָא פְּרִישָׁא דַּאֲחוֹהִי. 
לג,יז בְּכ֨וֹר שׁוֹר֜וֹ הָדָ֣ר ל֗וֹ וְקַרְנֵ֤י רְאֵם֙ קַרְנָ֔יו בָּהֶ֗ם עַמִּ֛ים יְנַגַּ֥ח יַחְדָּ֖ו אַפְסֵי־אָ֑רֶץ וְהֵם֙ רִבְב֣וֹת אֶפְרַ֔יִם וְהֵ֖ם אַלְפֵ֥י מְנַשֶּֽׁה׃ {ס} רַבָּא דִּבְנוֹהִי זִיוָא לֵיהּ, וּגְבוּרָן אִתְעֲבִידָא לֵיהּ מִן קֳדָם דְּתֻקְפָא וְרוּמָא דִּילֵיהּ--בִּגְבוּרְתֵיהּ עַמְמַיָּא יְקַטֵּיל כַּחְדָּא, עַד סְיָפֵי אַרְעָא; וְאִנּוּן רִבְּוָתָא דְּבֵית אֶפְרַיִם, וְאִנּוּן אַלְפַיָּא דְּבֵית מְנַשֶּׁה. {ס}
לג,יח וְלִזְבוּלֻ֣ן אָמַ֔ר שְׂמַ֥ח זְבוּלֻ֖ן בְּצֵאתֶ֑ךָ וְיִשָּׂשכָ֖ר בְּאֹֽהָלֶֽיךָ׃ וְלִזְבוּלוּן אֲמַר, חֲדִי זְבוּלוּן בְּמִפְּקָךְ לְאָגָחָא קְרָבָא עַל בַּעֲלֵי דְּבָבָךְ; וְיִשָּׂשכָר, בִּמְהָכָךְ לְמֶעֱבַד זִמְנֵי מוֹעֲדַיָּא בִּירוּשְׁלֶם. 
לג,יט עַמִּים֙ הַר־יִקְרָ֔אוּ שָׁ֖ם יִזְבְּח֣וּ זִבְחֵי־צֶ֑דֶק כִּ֣י שֶׁ֤פַע יַמִּים֙ יִינָ֔קוּ וּשְׂפֻנֵ֖י טְמ֥וּנֵי חֽוֹל׃ {ס} שִׁבְטַיָּא דְּיִשְׂרָאֵל, לְטוּר בֵּית מַקְדְּשָׁא יִתְכַּנְשׁוּן--תַּמָּן, יִכְּסוּן נִכְסַת קֻדְשִׁין לְרַעֲוָא:  אֲרֵי נִכְסֵי עַמְמַיָּא יֵיכְלוּן, וְסִימָן דְּמִטַּמְרָן בְּחָלָא מִתְגַּלְיָן לְהוֹן. {ס}
לג,כ וּלְגָ֣ד אָמַ֔ר בָּר֖וּךְ מַרְחִ֣יב גָּ֑ד כְּלָבִ֣יא שָׁכֵ֔ן וְטָרַ֥ף זְר֖וֹעַ אַף־קָדְקֹֽד׃ וּלְגָד אֲמַר, בְּרִיךְ דְּאַפְתֵּי לְגָד:  כְּלֵיתָא שְׁרֵי, וִיקַטֵּיל שִׁלְטוֹנִין עִם מַלְכִין. 
לג,כא וַיַּ֤רְא רֵאשִׁית֙ ל֔וֹ כִּי־שָׁ֛ם חֶלְקַ֥ת מְחֹקֵ֖ק סָפ֑וּן וַיֵּתֵא֙ רָ֣אשֵׁי עָ֔ם צִדְקַ֤ת יְהוָה֙ עָשָׂ֔ה וּמִשְׁפָּטָ֖יו עִם־יִשְׂרָאֵֽל׃ {ס} וְאִתְקַבַּל בְּקַדְמֵיתָא דִּילֵיהּ, אֲרֵי תַּמָּן בְּאַחְסָנְתֵיהּ מֹשֶׁה סָפְרָא רַבָּא דְּיִשְׂרָאֵל קְבִיר; וְהוּא נְפַק וְעָאל, בְּרֵישׁ עַמָּא--זָכְוָן קֳדָם יְיָ עֲבַד, וְדִינוֹהִי עִם יִשְׂרָאֵל. {ס}
לג,כב וּלְדָ֣ן אָמַ֔ר דָּ֖ן גּ֣וּר אַרְיֵ֑ה יְזַנֵּ֖ק מִן־הַבָּשָֽׁן׃ וּלְדָן אֲמַר, דָּן תַּקִּיף כְּגוּר אַרְיָוָן; אַרְעֵיהּ שָׁתְיָא, מִן נַחְלַיָּא דְּנָגְדִין מִן מַתְנַן. 
לג,כג וּלְנַפְתָּלִ֣י אָמַ֔ר נַפְתָּלִי֙ שְׂבַ֣ע רָצ֔וֹן וּמָלֵ֖א בִּרְכַּ֣ת יְהוָ֑ה יָ֥ם וְדָר֖וֹם יְרָֽשָׁה׃ {ס} וּלְנַפְתָּלִי אֲמַר--נַפְתָּלִי סְבַע רַעֲוָא, וּמְלִי בִּרְכָן מִן קֳדָם יְיָ; מַעְרַב יָם גִּנֵּיסַר וְדָרוֹמוֹהִי, יֵירַת. {ס}
לג,כד וּלְאָשֵׁ֣ר אָמַ֔ר בָּר֥וּךְ מִבָּנִ֖ים אָשֵׁ֑ר יְהִ֤י רְצוּי֙ אֶחָ֔יו וְטֹבֵ֥ל בַּשֶּׁ֖מֶן רַגְלֽוֹ׃ וּלְאָשֵׁר אֲמַר, בְּרִיךְ מִבִּרְכַת בְּנַיָּא אָשֵׁר; יְהֵי רַעֲוָא לַאֲחוֹהִי, וְיִתְרַבַּא בְּתַפְנוּקֵי מַלְכִין. 
לג,כה בַּרְזֶ֥ל וּנְחֹ֖שֶׁת מִנְעָלֶ֑ךָ וּכְיָמֶ֖יךָ דָּבְאֶֽךָ׃ תַּקִּיף כְּבַרְזְלָא וְכִנְחָשָׁא, בֵּית מוֹתְבָךְ; וּכְיוֹמֵי עוֹלֵימוּתָךְ, תֻּקְפָךְ. 
לג,כו אֵ֥ין כָּאֵ֖ל יְשֻׁר֑וּן רֹכֵ֤ב שָׁמַ֨יִם֙ בְּעֶזְרֶ֔ךָ וּבְגַֽאֲוָת֖וֹ שְׁחָקִֽים׃ לֵית אֱלָהּ, אֵלָא אֱלָהָא דְּיִשְׂרָאֵל:  דִּשְׁכִינְתֵיהּ בִּשְׁמַיָּא בְּסַעֲדָךְ, וְתֻקְפֵיהּ בִּשְׁמֵי שְׁמַיָּא. 
לג,כז מְעֹנָה֙ אֱלֹ֣הֵי קֶ֔דֶם וּמִתַּ֖חַת זְרֹעֹ֣ת עוֹלָ֑ם וַיְגָ֧רֶשׁ מִפָּנֶ֛יךָ אוֹיֵ֖ב וַיֹּ֥אמֶר הַשְׁמֵֽד׃ מְדוֹר אֱלָהָא דְּמִלְּקַדְמִין, דִּבְמֵימְרֵיהּ אִתְעֲבֵיד עָלְמָא; וְתָרֵיךְ מִן קֳדָמָךְ סָנְאָה, וַאֲמַר שֵׁיצִי. 
לג,כח וַיִּשְׁכֹּן֩ יִשְׂרָאֵ֨ל בֶּ֤טַח בָּדָד֙ עֵ֣ין יַֽעֲקֹ֔ב אֶל־אֶ֖רֶץ דָּגָ֣ן וְתִיר֑וֹשׁ אַף־שָׁמָ֖יו יַ֥עַרְפוּ טָֽל׃ וּשְׁרָא יִשְׂרָאֵל לְרֻחְצָן בִּלְחוֹדֵיהוֹן כְּעֵין בִּרְכְתָא דְּבָרֵיכִנּוּן יַעֲקוֹב אֲבוּהוֹן, בַּאֲרַע עָבְדָא עֲבוּר וַחֲמַר; אַף שְׁמַיָּא דְּעִלָּוֵיהוֹן, יְשַׁמְּשׁוּנוּנוּן בְּטַלָּא. 
לג,כט אַשְׁרֶ֨יךָ יִשְׂרָאֵ֜ל מִ֣י כָמ֗וֹךָ עַ֚ם נוֹשַׁ֣ע בַּֽיהוָ֔ה מָגֵ֣ן עֶזְרֶ֔ךָ וַֽאֲשֶׁר־חֶ֖רֶב גַּֽאֲוָתֶ֑ךָ וְיִכָּֽחֲשׁ֤וּ אֹֽיְבֶ֨יךָ֙ לָ֔ךְ וְאַתָּ֖ה עַל־בָּֽמוֹתֵ֥ימוֹ תִדְרֹֽךְ׃ {ס} טוּבָךְ יִשְׂרָאֵל לֵית דִּכְוָתָךְ, עַמָּא דְּפֻרְקָנֵיהּ מִן קֳדָם יְיָ, תְּקוֹף בְּסַעֲדָךְ, וּדְמִן קֳדָמוֹהִי נִצְחָן גִּבָּרְוָתָךְ; וְיִתְכַדְּבוּן סָנְאָךְ לָךְ, וְאַתְּ עַל פִּרְקַת צַוְרֵי מַלְכֵיהוֹן תִּדְרוֹךְ. {ס}
לד,א וַיַּ֨עַל מֹשֶׁ֜ה מֵֽעַרְבֹ֤ת מוֹאָב֙ אֶל־הַ֣ר נְב֔וֹ רֹ֚אשׁ הַפִּסְגָּ֔ה אֲשֶׁ֖ר עַל־פְּנֵ֣י יְרֵח֑וֹ וַיַּרְאֵ֨הוּ יְהוָ֧ה אֶת־כָּל־הָאָ֛רֶץ אֶת־הַגִּלְעָ֖ד עַד־דָּֽן׃ וּסְלֵיק מֹשֶׁה מִמֵּישְׁרַיָּא דְּמוֹאָב, לְטוּרָא דִּנְבוֹ, רֵישׁ רָמְתָא, דְּעַל אַפֵּי יְרֵיחוֹ; וְאַחְזְיֵיהּ יְיָ יָת כָּל אַרְעָא יָת גִּלְעָד, עַד דָּן. 
לד,ב וְאֵת֙ כָּל־נַפְתָּלִ֔י וְאֶת־אֶ֥רֶץ אֶפְרַ֖יִם וּמְנַשֶּׁ֑ה וְאֵת֙ כָּל־אֶ֣רֶץ יְהוּדָ֔ה עַ֖ד הַיָּ֥ם הָאַֽחֲרֽוֹן׃ וְיָת, כָּל נַפְתָּלִי, וְיָת אֲרַע אֶפְרַיִם, וּמְנַשֶּׁה; וְיָת כָּל אֲרַע יְהוּדָה, עַד יַמָּא מַעְרְבָאָה. 
לד,ג וְאֶת־הַנֶּ֗גֶב וְֽאֶת־הַכִּכָּ֞ר בִּקְעַ֧ת יְרֵח֛וֹ עִ֥יר הַתְּמָרִ֖ים עַד־צֹֽעַר׃ וְיָת דָּרוֹמָא, וְיָת מֵישְׁרָא בִּקְעַת יְרֵיחוֹ קִרְיַת דִּקְלַיָּא--עַד צֹעַר. 
לד,ד וַיֹּ֨אמֶר יְהוָ֜ה אֵלָ֗יו זֹ֤את הָאָ֨רֶץ֙ אֲשֶׁ֣ר נִ֠שְׁבַּעְתִּי לְאַבְרָהָ֨ם לְיִצְחָ֤ק וּֽלְיַעֲקֹב֙ לֵאמֹ֔ר לְזַרְעֲךָ֖ אֶתְּנֶ֑נָּה הֶרְאִיתִ֣יךָ בְעֵינֶ֔יךָ וְשָׁ֖מָּה לֹ֥א תַֽעֲבֹֽר׃ וַאֲמַר יְיָ לֵיהּ, דָּא אַרְעָא דְּקַיֵּימִית לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקוֹב לְמֵימַר, לִבְנָךְ, אֶתְּנִנַּהּ; אַחְזִיתָךְ בְּעֵינָךְ, וּלְתַמָּן לָא תִּעְבַּר. 
לד,ה וַיָּ֨מָת שָׁ֜ם מֹשֶׁ֧ה עֶֽבֶד־יְהוָ֛ה בְּאֶ֥רֶץ מוֹאָ֖ב עַל־פִּ֥י יְהוָֽה׃ וּמִית תַּמָּן מֹשֶׁה עַבְדָּא דַּייָ, בְּאַרְעָא דְּמוֹאָב--עַל מֵימְרָא דַּייָ. 
לד,ו וַיִּקְבֹּ֨ר אֹת֤וֹ בַגַּי֙ בְּאֶ֣רֶץ מוֹאָ֔ב מ֖וּל בֵּ֣ית פְּע֑וֹר וְלֹֽא־יָדַ֥ע אִישׁ֙ אֶת־קְבֻ֣רָת֔וֹ עַ֖ד הַיּ֥וֹם הַזֶּֽה׃ וּקְבַר יָתֵיהּ בְּחִלְּתָא בְּאַרְעָא דְּמוֹאָב, לָקֳבֵיל בֵּית פְּעוֹר; וְלָא יְדַע אֲנָשׁ יָת קְבוּרְתֵיהּ, עַד יוֹמָא הָדֵין. 
לד,ז וּמֹשֶׁ֗ה בֶּן־מֵאָ֧ה וְעֶשְׂרִ֛ים שָׁנָ֖ה בְּמֹת֑וֹ לֹֽא־כָהֲתָ֥ה עֵינ֖וֹ וְלֹא־נָ֥ס לֵחֹֽה׃ וּמֹשֶׁה, בַּר מְאָה וְעַסְרִין שְׁנִין--כַּד מִית; לָא כְּהָת עֵינֵיהּ, וְלָא שְׁנָא זִיו יְקָרָא דְּאַפּוֹהִי. 
לד,ח וַיִּבְכּוּ֩ בְנֵ֨י יִשְׂרָאֵ֧ל אֶת־מֹשֶׁ֛ה בְּעַֽרְבֹ֥ת מוֹאָ֖ב שְׁלֹשִׁ֣ים י֑וֹם וַֽיִּתְּמ֔וּ יְמֵ֥י בְכִ֖י אֵ֥בֶל מֹשֶֽׁה׃ וּבְכוֹ בְּנֵי יִשְׂרָאֵל יָת מֹשֶׁה בְּמֵישְׁרַיָּא דְּמוֹאָב, תְּלָתִין יוֹמִין; וּשְׁלִימוּ, יוֹמֵי בְּכִיתָא אֶבְלָא דְּמֹשֶׁה. 
לד,ט וִֽיהוֹשֻׁ֣עַ בִּן־נ֗וּן מָלֵא֙ ר֣וּחַ חָכְמָ֔ה כִּֽי־סָמַ֥ךְ מֹשֶׁ֛ה אֶת־יָדָ֖יו עָלָ֑יו וַיִּשְׁמְע֨וּ אֵלָ֤יו בְּנֵֽי־יִשְׂרָאֵל֙ וַֽיַּעֲשׂ֔וּ כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ וִיהוֹשׁוּעַ בַּר נוּן, אִתְמְלִי רוּחַ חָכְמָא--אֲרֵי סְמַךְ מֹשֶׁה יָת יְדוֹהִי, עֲלוֹהִי; וְקַבִּילוּ מִנֵּיהּ בְּנֵי יִשְׂרָאֵל וַעֲבַדוּ, כְּמָא דְּפַקֵּיד יְיָ יָת מֹשֶׁה. 
לד,י וְלֹא־קָ֨ם נָבִ֥יא ע֛וֹד בְּיִשְׂרָאֵ֖ל כְּמֹשֶׁ֑ה אֲשֶׁר֙ יְדָע֣וֹ יְהוָ֔ה פָּנִ֖ים אֶל־פָּנִֽים׃ וְלָא קָם נְבִיָּא עוֹד בְּיִשְׂרָאֵל, כְּמֹשֶׁה, דְּאִתְגְּלִי לֵיהּ יְיָ, אַפִּין בְּאַפִּין. 
לד,יא לְכָל־הָ֨אֹתֹ֜ת וְהַמּֽוֹפְתִ֗ים אֲשֶׁ֤ר שְׁלָחוֹ֙ יְהוָ֔ה לַֽעֲשׂ֖וֹת בְּאֶ֣רֶץ מִצְרָ֑יִם לְפַרְעֹ֥ה וּלְכָל־עֲבָדָ֖יו וּלְכָל־אַרְצֽוֹ׃ לְכָל אָתַיָּא וּמוֹפְתַיָּא, דְּשַׁלְּחֵיהּ יְיָ, לְמֶעֱבַד, בְּאַרְעָא דְּמִצְרָיִם--לְפַרְעֹה וּלְכָל עַבְדּוֹהִי, וּלְכָל אַרְעֵיהּ. 
לד,יב וּלְכֹל֙ הַיָּ֣ד הַֽחֲזָקָ֔ה וּלְכֹ֖ל הַמּוֹרָ֣א הַגָּד֑וֹל אֲשֶׁר֙ עָשָׂ֣ה מֹשֶׁ֔ה לְעֵינֵ֖י כָּל־יִשְׂרָאֵֽל׃ {ש} וּלְכֹל יְדָא תַּקִּיפְתָא, וּלְכֹל חֶזְוָנָא רַבָּא, דַּעֲבַד מֹשֶׁה, לְעֵינֵי כָּל יִשְׂרָאֵל. {ש}
