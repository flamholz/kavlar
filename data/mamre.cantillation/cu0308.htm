אמור

כא,א וַיֹּ֤אמֶר יְהוָה֙ אֶל־מֹשֶׁ֔ה אֱמֹ֥ר אֶל־הַכֹּֽהֲנִ֖ים בְּנֵ֣י אַֽהֲרֹ֑ן וְאָֽמַרְתָּ֣ אֲלֵהֶ֔ם לְנֶ֥פֶשׁ לֹֽא־יִטַּמָּ֖א בְּעַמָּֽיו׃ וַאֲמַר יְיָ לְמֹשֶׁה, אֵימַר לְכָהֲנַיָּא בְּנֵי אַהֲרוֹן; וְתֵימַר לְהוֹן, עַל מִית לָא יִסְתָּאַב בְּעַמֵּיהּ. 
כא,ב כִּ֚י אִם־לִשְׁאֵר֔וֹ הַקָּרֹ֖ב אֵלָ֑יו לְאִמּ֣וֹ וּלְאָבִ֔יו וְלִבְנ֥וֹ וּלְבִתּ֖וֹ וּלְאָחִֽיו׃ אֱלָהֵין לְקָרִיבֵיהּ--דְּקָרִיב, לֵיהּ:  לְאִמֵּיהּ וְלַאֲבוּהִי, וְלִבְרֵיהּ וְלִבְרַתֵּיהּ וְלַאֲחוּהִי. 
כא,ג וְלַֽאֲחֹת֤וֹ הַבְּתוּלָה֙ הַקְּרוֹבָ֣ה אֵלָ֔יו אֲשֶׁ֥ר לֹֽא־הָיְתָ֖ה לְאִ֑ישׁ לָ֖הּ יִטַּמָּֽא׃ וְלַאֲחָתֵיהּ בְּתוּלְתָא דְּקָרִיבָא לֵיהּ, דְּלָא הֲוָת לִגְבַר--לַהּ, יִסְתָּאַב. 
כא,ד לֹ֥א יִטַּמָּ֖א בַּ֣עַל בְּעַמָּ֑יו לְהֵ֖חַלּֽוֹ׃ לָא יִסְתָּאַב, בְּרַבָּא בְּעַמֵּיהּ--לְאַחָלוּתֵיהּ. 
כא,ה לֹֽא־יקרחה (יִקְרְח֤וּ) קָרְחָה֙ בְּרֹאשָׁ֔ם וּפְאַ֥ת זְקָנָ֖ם לֹ֣א יְגַלֵּ֑חוּ וּבִ֨בְשָׂרָ֔ם לֹ֥א יִשְׂרְט֖וּ שָׂרָֽטֶת׃ לָא יִמְרְטוּן מְרַט בְּרֵישְׁהוֹן, וּפָתָא דְּדִקְנְהוֹן לָא יְגַלְּחוּן; וּבִבְסַרְהוֹן--לָא יְחַבְּלוּן, חִבּוּל. 
כא,ו קְדֹשִׁ֤ים יִֽהְיוּ֙ לֵאלֹ֣הֵיהֶ֔ם וְלֹ֣א יְחַלְּל֔וּ שֵׁ֖ם אֱלֹֽהֵיהֶ֑ם כִּי֩ אֶת־אִשֵּׁ֨י יְהוָ֜ה לֶ֧חֶם אֱלֹֽהֵיהֶ֛ם הֵ֥ם מַקְרִיבִ֖ם וְהָ֥יוּ קֹֽדֶשׁ׃ קַדִּישִׁין יְהוֹן, קֳדָם אֱלָהֲהוֹן, וְלָא יַחֲלוּן, שְׁמָא דֶּאֱלָהֲהוֹן:  אֲרֵי יָת קֻרְבָּנַיָּא דַּייָ קֻרְבַּן אֱלָהֲהוֹן, אִנּוּן מְקָרְבִין--וִיהוֹן קַדִּישִׁין. 
כא,ז אִשָּׁ֨ה זֹנָ֤ה וַֽחֲלָלָה֙ לֹ֣א יִקָּ֔חוּ וְאִשָּׁ֛ה גְּרוּשָׁ֥ה מֵֽאִישָׁ֖הּ לֹ֣א יִקָּ֑חוּ כִּֽי־קָדֹ֥שׁ ה֖וּא לֵֽאלֹהָֽיו׃ אִתְּתָא מַטְעֲיָא וּמַחֲלָא לָא יִסְּבוּן, וְאִתְּתָא דִּמְתָרְכָא מִבַּעְלַהּ לָא יִסְּבוּן:  אֲרֵי קַדִּישׁ הוּא, קֳדָם אֱלָהֵיהּ. 
כא,ח וְקִ֨דַּשְׁתּ֔וֹ כִּֽי־אֶת־לֶ֥חֶם אֱלֹהֶ֖יךָ ה֣וּא מַקְרִ֑יב קָדֹשׁ֙ יִֽהְיֶה־לָּ֔ךְ כִּ֣י קָד֔וֹשׁ אֲנִ֥י יְהוָ֖ה מְקַדִּשְׁכֶֽם׃ וּתְקַדְּשִׁנֵּיהּ--אֲרֵי יָת קֻרְבַּן אֱלָהָךְ, הוּא מְקָרֵיב; קַדִּישׁ, יְהֵי לָךְ--אֲרֵי קַדִּישׁ, אֲנָא יְיָ מְקַדִּשְׁכוֹן. 
כא,ט וּבַת֙ אִ֣ישׁ כֹּהֵ֔ן כִּ֥י תֵחֵ֖ל לִזְנ֑וֹת אֶת־אָבִ֨יהָ֙ הִ֣יא מְחַלֶּ֔לֶת בָּאֵ֖שׁ תִּשָּׂרֵֽף׃ {ס} וּבַת גְּבַר כָּהִין, אֲרֵי תִּתַּחַל לְמִטְעֵי--מִקְּדֻשַּׁת אֲבוּהָא הִיא מִתַּחֲלָא, בְּנוּרָא תִּתּוֹקַד. {ס}
כא,י וְהַכֹּהֵן֩ הַגָּד֨וֹל מֵֽאֶחָ֜יו אֲֽשֶׁר־יוּצַ֥ק עַל־רֹאשׁ֣וֹ ׀ שֶׁ֤מֶן הַמִּשְׁחָה֙ וּמִלֵּ֣א אֶת־יָד֔וֹ לִלְבֹּ֖שׁ אֶת־הַבְּגָדִ֑ים אֶת־רֹאשׁוֹ֙ לֹ֣א יִפְרָ֔ע וּבְגָדָ֖יו לֹ֥א יִפְרֹֽם׃ וְכָהֲנָא דְּיִתְרַבַּא מֵאֲחוֹהִי דְּיִתָּרַק עַל רֵישֵׁיהּ מִשְׁחָא דִּרְבוּתָא, וְדִיקָרֵיב יָת קֻרְבָּנֵיהּ, לְמִלְבַּשׁ, יָת לְבוּשַׁיָּא--יָת רֵישֵׁיהּ לָא יְרַבֵּי פֵּירוּעַ, וּלְבוּשׁוֹהִי לָא יְבַזַּע. 
כא,יא וְעַ֛ל כָּל־נַפְשֹׁ֥ת מֵ֖ת לֹ֣א יָבֹ֑א לְאָבִ֥יו וּלְאִמּ֖וֹ לֹ֥א יִטַּמָּֽא׃ וְעַל כָּל נַפְשָׁת מִיתָא, לָא יֵיעוֹל:  לַאֲבוּהִי וּלְאִמֵּיהּ, לָא יִסְתָּאַב. 
כא,יב וּמִן־הַמִּקְדָּשׁ֙ לֹ֣א יֵצֵ֔א וְלֹ֣א יְחַלֵּ֔ל אֵ֖ת מִקְדַּ֣שׁ אֱלֹהָ֑יו כִּ֡י נֵ֠זֶר שֶׁ֣מֶן מִשְׁחַ֧ת אֱלֹהָ֛יו עָלָ֖יו אֲנִ֥י יְהוָֽה׃ וּמִן מַקְדְּשָׁא, לָא יִפּוֹק, וְלָא יַחֵיל, יָת מַקְדְּשָׁא דֶּאֱלָהֵיהּ:  אֲרֵי כְּלִיל מְשַׁח רְבוּתָא דֶּאֱלָהֵיהּ, עֲלוֹהִי--אֲנָא יְיָ. 
כא,יג וְה֕וּא אִשָּׁ֥ה בִבְתוּלֶ֖יהָ יִקָּֽח׃ וְהוּא, אִתְּתָא בִּבְתוּלַהָא יִסַּב. 
כא,יד אַלְמָנָ֤ה וּגְרוּשָׁה֙ וַֽחֲלָלָ֣ה זֹנָ֔ה אֶת־אֵ֖לֶּה לֹ֣א יִקָּ֑ח כִּ֛י אִם־בְּתוּלָ֥ה מֵֽעַמָּ֖יו יִקַּ֥ח אִשָּֽׁה׃ אַרְמְלָא וּמְתָרְכָא וַחֲלִילָא מַטְעֲיָא, יָת אִלֵּין לָא יִסַּב:  אֱלָהֵין בְּתוּלְתָא מֵעַמֵּיהּ, יִסַּב אִתְּתָא. 
כא,טו וְלֹֽא־יְחַלֵּ֥ל זַרְע֖וֹ בְּעַמָּ֑יו כִּ֛י אֲנִ֥י יְהוָ֖ה מְקַדְּשֽׁוֹ׃ {ס} וְלָא יַחֵיל זַרְעֵיהּ, בְּעַמֵּיהּ:  אֲרֵי אֲנָא יְיָ, מְקַדְּשֵׁיהּ. {ס}
כא,טז וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
כא,יז דַּבֵּ֥ר אֶֽל־אַהֲרֹ֖ן לֵאמֹ֑ר אִ֣ישׁ מִֽזַּרְעֲךָ֞ לְדֹֽרֹתָ֗ם אֲשֶׁ֨ר יִֽהְיֶ֥ה בוֹ֙ מ֔וּם לֹ֣א יִקְרַ֔ב לְהַקְרִ֖יב לֶ֥חֶם אֱלֹהָֽיו׃ מַלֵּיל עִם אַהֲרוֹן, לְמֵימַר:  גְּבַר מִבְּנָךְ לְדָרֵיהוֹן, דִּיהֵי בֵּיהּ מוּמָא--לָא יִקְרַב, לְקָרָבָא קֻרְבָּנָא קֳדָם אֱלָהֵיהּ. 
כא,יח כִּ֥י כָל־אִ֛ישׁ אֲשֶׁר־בּ֥וֹ מ֖וּם לֹ֣א יִקְרָ֑ב אִ֤ישׁ עִוֵּר֙ א֣וֹ פִסֵּ֔חַ א֥וֹ חָרֻ֖ם א֥וֹ שָׂרֽוּעַ׃ אֲרֵי כָּל גְּבַר דְּבֵיהּ מוּמָא, לָא יִקְרַב:  גְּבַר עֲוִיר אוֹ חֲגִיר, אוֹ חֲרִים אוֹ סְרִיעַ. 
כא,יט א֣וֹ אִ֔ישׁ אֲשֶׁר־יִֽהְיֶ֥ה ב֖וֹ שֶׁ֣בֶר רָ֑גֶל א֖וֹ שֶׁ֥בֶר יָֽד׃ אוֹ גְּבַר, דִּיהֵי בֵּיהּ תְּבָר רִגְלָא, אוֹ, תְּבָר יְדָא. 
כא,כ אֽוֹ־גִבֵּ֣ן אוֹ־דַ֔ק א֖וֹ תְּבַלֻּ֣ל בְּעֵינ֑וֹ א֤וֹ גָרָב֙ א֣וֹ יַלֶּ֔פֶת א֖וֹ מְר֥וֹחַ אָֽשֶׁךְ׃ אוֹ גְּבִין אוֹ דֻּקָּא, אוֹ חִלִּיז בְּעֵינוֹהִי, אוֹ גַּרְבָּן אוֹ חֲזָזָן, אוֹ מְרִיס פַּחְדִּין. 
כא,כא כָּל־אִ֞ישׁ אֲשֶׁר־בּ֣וֹ מ֗וּם מִזֶּ֨רַע֙ אַֽהֲרֹ֣ן הַכֹּהֵ֔ן לֹ֣א יִגַּ֔שׁ לְהַקְרִ֖יב אֶת־אִשֵּׁ֣י יְהוָ֑ה מ֣וּם בּ֔וֹ אֵ֚ת לֶ֣חֶם אֱלֹהָ֔יו לֹ֥א יִגַּ֖שׁ לְהַקְרִֽיב׃ כָּל גְּבַר דְּבֵיהּ מוּמָא, מִזַּרְעָא דְּאַהֲרוֹן כָּהֲנָא--לָא יִקְרַב, לְקָרָבָא יָת קֻרְבָּנַיָּא דַּייָ:  מוּמָא בֵּיהּ--יָת קֻרְבַּן אֱלָהֵיהּ, לָא יִקְרַב לְקָרָבָא. 
כא,כב לֶ֣חֶם אֱלֹהָ֔יו מִקָּדְשֵׁ֖י הַקֳּדָשִׁ֑ים וּמִן־הַקֳּדָשִׁ֖ים יֹאכֵֽל׃ קֻרְבַּן אֱלָהֵיהּ, מִקֻּדְשֵׁי קֻדְשַׁיָּא, וּמִן קֻדְשַׁיָּא, יֵיכוֹל. 
כא,כג אַ֣ךְ אֶל־הַפָּרֹ֜כֶת לֹ֣א יָבֹ֗א וְאֶל־הַמִּזְבֵּ֛חַ לֹ֥א יִגַּ֖שׁ כִּֽי־מ֣וּם בּ֑וֹ וְלֹ֤א יְחַלֵּל֙ אֶת־מִקְדָּשַׁ֔י כִּ֛י אֲנִ֥י יְהוָ֖ה מְקַדְּשָֽׁם׃ בְּרַם לְפָרֻכְתָּא לָא יֵיעוֹל, וּלְמַדְבְּחָא לָא יִקְרַב--אֲרֵי מוּמָא בֵּיהּ; וְלָא יַחֵיל יָת מַקְדָּשַׁי, אֲרֵי אֲנָא יְיָ מְקַדִּשְׁהוֹן. 
כא,כד וַיְדַבֵּ֣ר מֹשֶׁ֔ה אֶֽל־אַהֲרֹ֖ן וְאֶל־בָּנָ֑יו וְאֶֽל־כָּל־בְּנֵ֖י יִשְׂרָאֵֽל׃ {פ} וּמַלֵּיל מֹשֶׁה, עִם אַהֲרוֹן וְעִם בְּנוֹהִי, וְעִם כָּל בְּנֵי, יִשְׂרָאֵל. {פ}
כב,א וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
כב,ב דַּבֵּ֨ר אֶֽל־אַהֲרֹ֜ן וְאֶל־בָּנָ֗יו וְיִנָּֽזְרוּ֙ מִקָּדְשֵׁ֣י בְנֵֽי־יִשְׂרָאֵ֔ל וְלֹ֥א יְחַלְּל֖וּ אֶת־שֵׁ֣ם קָדְשִׁ֑י אֲשֶׁ֨ר הֵ֧ם מַקְדִּשִׁ֛ים לִ֖י אֲנִ֥י יְהוָֽה׃ מַלֵּיל עִם אַהֲרוֹן וְעִם בְּנוֹהִי, וְיִפְרְשׁוּן מִקֻּדְשַׁיָּא דִּבְנֵי יִשְׂרָאֵל, וְלָא יַחֲלוּן, יָת שְׁמָא דְּקֻדְשִׁי--דְּאִנּוּן מַקְדְּשִׁין קֳדָמַי, אֲנָא יְיָ. 
כב,ג אֱמֹ֣ר אֲלֵהֶ֗ם לְדֹרֹ֨תֵיכֶ֜ם כָּל־אִ֣ישׁ ׀ אֲשֶׁר־יִקְרַ֣ב מִכָּל־זַרְעֲכֶ֗ם אֶל־הַקֳּדָשִׁים֙ אֲשֶׁ֨ר יַקְדִּ֤ישׁוּ בְנֵֽי־יִשְׂרָאֵל֙ לַֽיהוָ֔ה וְטֻמְאָת֖וֹ עָלָ֑יו וְנִכְרְתָ֞ה הַנֶּ֧פֶשׁ הַהִ֛וא מִלְּפָנַ֖י אֲנִ֥י יְהוָֽה׃ אֵימַר לְהוֹן, לְדָרֵיכוֹן כָּל גְּבַר דְּיִקְרַב מִכָּל זַרְעֲכוֹן לְקֻדְשַׁיָּא דְּיַקְדְּשׁוּן בְּנֵי יִשְׂרָאֵל קֳדָם יְיָ, וּסְאוֹבְתֵיהּ, עֲלוֹהִי:  וְיִשְׁתֵּיצֵי אֲנָשָׁא הַהוּא, מִן קֳדָמַי--אֲנָא יְיָ. 
כב,ד אִ֣ישׁ אִ֞ישׁ מִזֶּ֣רַע אַֽהֲרֹ֗ן וְה֤וּא צָר֨וּעַ֙ א֣וֹ זָ֔ב בַּקֳּדָשִׁים֙ לֹ֣א יֹאכַ֔ל עַ֖ד אֲשֶׁ֣ר יִטְהָ֑ר וְהַנֹּגֵ֨עַ֙ בְּכָל־טְמֵא־נֶ֔פֶשׁ א֣וֹ אִ֔ישׁ אֲשֶׁר־תֵּצֵ֥א מִמֶּ֖נּוּ שִׁכְבַת־זָֽרַע׃ גְּבַר גְּבַר מִזַּרְעָא דְּאַהֲרוֹן, וְהוּא סְגִיר אוֹ דָּאִיב--בְּקֻדְשַׁיָּא לָא יֵיכוֹל, עַד דְּיִדְכֵּי; וּדְיִקְרַב, בְּכָל טְמֵי נַפְשָׁא, אוֹ גְּבַר, דְּתִפּוֹק מִנֵּיהּ שִׁכְבַת זַרְעָא. 
כב,ה אוֹ־אִישׁ֙ אֲשֶׁ֣ר יִגַּ֔ע בְּכָל־שֶׁ֖רֶץ אֲשֶׁ֣ר יִטְמָא־ל֑וֹ א֤וֹ בְאָדָם֙ אֲשֶׁ֣ר יִטְמָא־ל֔וֹ לְכֹ֖ל טֻמְאָתֽוֹ׃ אוֹ גְּבַר דְּיִקְרַב, בְּכָל רִחְשָׁא דְּיִסְתָּאַב לֵיהּ; אוֹ בַּאֲנָשָׁא דְּיִסְתָּאַב לֵיהּ, לְכֹל סְאוֹבְתֵיהּ. 
כב,ו נֶ֚פֶשׁ אֲשֶׁ֣ר תִּגַּע־בּ֔וֹ וְטָֽמְאָ֖ה עַד־הָעָ֑רֶב וְלֹ֤א יֹאכַל֙ מִן־הַקֳּדָשִׁ֔ים כִּ֛י אִם־רָחַ֥ץ בְּשָׂר֖וֹ בַּמָּֽיִם׃ אֲנָשׁ דְּיִקְרַב בֵּיהּ, וִיהֵי מְסָאַב עַד רַמְשָׁא; וְלָא יֵיכוֹל מִן קֻדְשַׁיָּא, אֱלָהֵין אַסְחִי בִּסְרֵיהּ בְּמַיָּא. 
כב,ז וּבָ֥א הַשֶּׁ֖מֶשׁ וְטָהֵ֑ר וְאַחַר֙ יֹאכַ֣ל מִן־הַקֳּדָשִׁ֔ים כִּ֥י לַחְמ֖וֹ הֽוּא׃ וּכְמֵיעַל שִׁמְשָׁא, וְיִדְכֵּי; וּבָתַר כֵּין יֵיכוֹל מִן קֻדְשַׁיָּא, אֲרֵי לַחְמֵיהּ הוּא. 
כב,ח נְבֵלָ֧ה וּטְרֵפָ֛ה לֹ֥א יֹאכַ֖ל לְטָמְאָה־בָ֑הּ אֲנִ֖י יְהוָֽה׃ נְבִילָא וּתְבִירָא לָא יֵיכוֹל, לְאִסְתַּאָבָא בַּהּ:  אֲנָא, יְיָ. 
כב,ט וְשָֽׁמְר֣וּ אֶת־מִשְׁמַרְתִּ֗י וְלֹֽא־יִשְׂא֤וּ עָלָיו֙ חֵ֔טְא וּמֵ֥תוּ ב֖וֹ כִּ֣י יְחַלְּלֻ֑הוּ אֲנִ֥י יְהוָ֖ה מְקַדְּשָֽׁם׃ וְיִטְּרוּן יָת מַטְּרַת מֵימְרִי, וְלָא יְקַבְּלוּן עֲלוֹהִי חוֹבָא, וִימוּתוּן בֵּיהּ, אֲרֵי יַחֲלוּנֵּיהּ:  אֲנָא יְיָ, מְקַדִּשְׁהוֹן. 
כב,י וְכָל־זָ֖ר לֹא־יֹ֣אכַל קֹ֑דֶשׁ תּוֹשַׁ֥ב כֹּהֵ֛ן וְשָׂכִ֖יר לֹא־יֹ֥אכַל קֹֽדֶשׁ׃ וְכָל חִילוֹנַי, לָא יֵיכוֹל קֻדְשָׁא; תּוֹתָבָא דְּכָהֲנָא וַאֲגִירָא, לָא יֵיכוֹל קֻדְשָׁא. 
כב,יא וְכֹהֵ֗ן כִּֽי־יִקְנֶ֥ה נֶ֨פֶשׁ֙ קִנְיַ֣ן כַּסְפּ֔וֹ ה֖וּא יֹ֣אכַל בּ֑וֹ וִילִ֣יד בֵּית֔וֹ הֵ֖ם יֹֽאכְל֥וּ בְלַחְמֽוֹ׃ וְכָהִין, אֲרֵי יִקְנֵי נְפַשׁ קִנְיַן כַּסְפֵּיהּ--הוּא, יֵיכוֹל בֵּיהּ; וִילִידֵי בֵּיתֵיהּ, אִנּוּן יֵיכְלוּן בְּלַחְמֵיהּ. 
כב,יב וּבַ֨ת־כֹּהֵ֔ן כִּ֥י תִֽהְיֶ֖ה לְאִ֣ישׁ זָ֑ר הִ֕וא בִּתְרוּמַ֥ת הַקֳּדָשִׁ֖ים לֹ֥א תֹאכֵֽל׃ וּבַת כָּהִין--אֲרֵי תְּהֵי, לִגְבַר חִילוֹנָי:  הִיא, בְּאַפְרָשׁוּת קֻדְשַׁיָּא לָא תֵּיכוֹל. 
כב,יג וּבַת־כֹּהֵן֩ כִּ֨י תִֽהְיֶ֜ה אַלְמָנָ֣ה וּגְרוּשָׁ֗ה וְזֶרַע֮ אֵ֣ין לָהּ֒ וְשָׁבָ֞ה אֶל־בֵּ֤ית אָבִ֨יהָ֙ כִּנְעוּרֶ֔יהָ מִלֶּ֥חֶם אָבִ֖יהָ תֹּאכֵ֑ל וְכָל־זָ֖ר לֹא־יֹ֥אכַל בּֽוֹ׃ וּבַת כָּהִין אֲרֵי תְּהֵי אַרְמְלָא וּמְתָרְכָא, וּבַר לֵית לַהּ--וּתְתוּב לְבֵית אֲבוּהָא כְּרַבְיוּתַהּ, מִלַּחְמָא דַּאֲבוּהָא תֵּיכוֹל; וְכָל חִילוֹנַי, לָא יֵיכוֹל בֵּיהּ. 
כב,יד וְאִ֕ישׁ כִּֽי־יֹאכַ֥ל קֹ֖דֶשׁ בִּשְׁגָגָ֑ה וְיָסַ֤ף חֲמִֽשִׁיתוֹ֙ עָלָ֔יו וְנָתַ֥ן לַכֹּהֵ֖ן אֶת־הַקֹּֽדֶשׁ׃ וּגְבַר, אֲרֵי יֵיכוֹל קֻדְשָׁא בְּשָׁלוּ--וְיוֹסֵיף חֻמְשֵׁיהּ עֲלוֹהִי, וְיִתֵּין לְכָהֲנָא יָת קֻדְשָׁא. 
כב,טו וְלֹ֣א יְחַלְּל֔וּ אֶת־קָדְשֵׁ֖י בְּנֵ֣י יִשְׂרָאֵ֑ל אֵ֥ת אֲשֶׁר־יָרִ֖ימוּ לַֽיהוָֽה׃ וְלָא יַחֲלוּן, יָת קֻדְשַׁיָּא דִּבְנֵי יִשְׂרָאֵל--יָת דְּיַפְרְשׁוּן, קֳדָם יְיָ. 
כב,טז וְהִשִּׂ֤יאוּ אוֹתָם֙ עֲו֣‍ֹן אַשְׁמָ֔ה בְּאָכְלָ֖ם אֶת־קָדְשֵׁיהֶ֑ם כִּ֛י אֲנִ֥י יְהוָ֖ה מְקַדְּשָֽׁם׃ {פ} וִיקַבְּלוּן עֲלֵיהוֹן עֲוָיָן וְחוֹבִין, בְּמֵיכַלְהוֹן בְּסַאָבָא יָת קֻדְשֵׁיהוֹן:  אֲרֵי אֲנָא יְיָ, מְקַדִּשְׁהוֹן. {פ}
כב,יז וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
כב,יח דַּבֵּ֨ר אֶֽל־אַהֲרֹ֜ן וְאֶל־בָּנָ֗יו וְאֶל֙ כָּל־בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָֽמַרְתָּ֖ אֲלֵהֶ֑ם אִ֣ישׁ אִישׁ֩ מִבֵּ֨ית יִשְׂרָאֵ֜ל וּמִן־הַגֵּ֣ר בְּיִשְׂרָאֵ֗ל אֲשֶׁ֨ר יַקְרִ֤יב קָרְבָּנוֹ֙ לְכָל־נִדְרֵיהֶם֙ וּלְכָל־נִדְבוֹתָ֔ם אֲשֶׁר־יַקְרִ֥יבוּ לַֽיהוָ֖ה לְעֹלָֽה׃ מַלֵּיל עִם אַהֲרוֹן וְעִם בְּנוֹהִי, וְעִם כָּל בְּנֵי יִשְׂרָאֵל, וְתֵימַר, לְהוֹן:  גְּבַר גְּבַר מִבֵּית יִשְׂרָאֵל וּמִן גִּיּוֹרַיָּא בְּיִשְׂרָאֵל, דִּיקָרֵיב קֻרְבָּנֵיהּ לְכָל נִדְרֵיהוֹן וּלְכָל נִדְבָתְהוֹן, דִּיקָרְבוּן קֳדָם יְיָ, לַעֲלָתָא. 
כב,יט לִֽרְצֹנְכֶ֑ם תָּמִ֣ים זָכָ֔ר בַּבָּקָ֕ר בַּכְּשָׂבִ֖ים וּבָֽעִזִּֽים׃ לְרַעֲוָא לְכוֹן:  שְׁלִים דְּכַר--בְּתוֹרַיָּא, בְּאִמְּרַיָּא וּבְעִזַּיָּא. 
כב,כ כֹּ֛ל אֲשֶׁר־בּ֥וֹ מ֖וּם לֹ֣א תַקְרִ֑יבוּ כִּי־לֹ֥א לְרָצ֖וֹן יִֽהְיֶ֥ה לָכֶֽם׃ כֹּל דְּבֵיהּ מוּמָא, לָא תְּקָרְבוּן:  אֲרֵי לָא לְרַעֲוָא, יְהֵי לְכוֹן. 
כב,כא וְאִ֗ישׁ כִּֽי־יַקְרִ֤יב זֶֽבַח־שְׁלָמִים֙ לַֽיהוָ֔ה לְפַלֵּא־נֶ֨דֶר֙ א֣וֹ לִנְדָבָ֔ה בַּבָּקָ֖ר א֣וֹ בַצֹּ֑אן תָּמִ֤ים יִֽהְיֶה֙ לְרָצ֔וֹן כָּל־מ֖וּם לֹ֥א יִֽהְיֶה־בּֽוֹ׃ וּגְבַר, אֲרֵי יְקָרֵיב נִכְסַת קֻדְשַׁיָּא קֳדָם יְיָ, לְפָרָשָׁא נִדְרָא אוֹ לִנְדַבְתָּא, בְּתוֹרֵי אוֹ בְּעָנָא--שְׁלִים יְהֵי לְרַעֲוָא, כָּל מוּמָא לָא יְהֵי בֵּיהּ. 
כב,כב עַוֶּרֶת֩ א֨וֹ שָׁב֜וּר אֽוֹ־חָר֣וּץ אֽוֹ־יַבֶּ֗לֶת א֤וֹ גָרָב֙ א֣וֹ יַלֶּ֔פֶת לֹֽא־תַקְרִ֥יבוּ אֵ֖לֶּה לַֽיהוָ֑ה וְאִשֶּׁ֗ה לֹֽא־תִתְּנ֥וּ מֵהֶ֛ם עַל־הַמִּזְבֵּ֖חַ לַֽיהוָֽה׃ עֲוִיר אוֹ תְּבִיר אוֹ פְּסִיק אוֹ יַבְלָן, אוֹ גַּרְבָּן אוֹ חֲזָזָן--לָא תְּקָרְבוּן אִלֵּין, קֳדָם יְיָ; וְקֻרְבָּנָא, לָא תִּתְּנוּן מִנְּהוֹן עַל מַדְבְּחָא--קֳדָם יְיָ. 
כב,כג וְשׁ֥וֹר וָשֶׂ֖ה שָׂר֣וּעַ וְקָל֑וּט נְדָבָה֙ תַּֽעֲשֶׂ֣ה אֹת֔וֹ וּלְנֵ֖דֶר לֹ֥א יֵֽרָצֶֽה׃ וְתוֹר וְאִמַּר, יַתִּיר וְחַסִּיר--נְדַבְתָּא תַּעֲבֵיד יָתֵיהּ, וּלְנִדְרָא לָא יִתְרְעֵי. 
כב,כד וּמָע֤וּךְ וְכָתוּת֙ וְנָת֣וּק וְכָר֔וּת לֹ֥א תַקְרִ֖יבוּ לַֽיהוָ֑ה וּֽבְאַרְצְכֶ֖ם לֹ֥א תַֽעֲשֽׂוּ׃ וְדִמְרִיס וְדִרְסִיס וְדִשְׁלִיף וְדִגְזִיר, לָא תְּקָרְבוּן קֳדָם יְיָ; וּבַאֲרַעְכוֹן, לָא תַּעְבְּדוּן. 
כב,כה וּמִיַּ֣ד בֶּן־נֵכָ֗ר לֹ֥א תַקְרִ֛יבוּ אֶת־לֶ֥חֶם אֱלֹֽהֵיכֶ֖ם מִכָּל־אֵ֑לֶּה כִּ֣י מָשְׁחָתָ֤ם בָּהֶם֙ מ֣וּם בָּ֔ם לֹ֥א יֵֽרָצ֖וּ לָכֶֽם׃ {ס} וּמִיַּד בַּר עַמְמִין, לָא תְּקָרְבוּן יָת קֻרְבַּן אֱלָהֲכוֹן--מִכָּל אִלֵּין:  אֲרֵי חִבֻּלְהוֹן בְּהוֹן מוּמָא בְּהוֹן, לָא לְרַעֲוָא יְהוֹן לְכוֹן. {ס}
כב,כו וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
כב,כז שׁ֣וֹר אוֹ־כֶ֤שֶׂב אוֹ־עֵז֙ כִּ֣י יִוָּלֵ֔ד וְהָיָ֛ה שִׁבְעַ֥ת יָמִ֖ים תַּ֣חַת אִמּ֑וֹ וּמִיּ֤וֹם הַשְּׁמִינִי֙ וָהָ֔לְאָה יֵֽרָצֶ֕ה לְקָרְבַּ֥ן אִשֶּׁ֖ה לַֽיהוָֽה׃ תּוֹר אוֹ אִמַּר אוֹ עֵז אֲרֵי יִתְיְלֵיד, וִיהֵי שִׁבְעָא יוֹמִין בָּתַר אִמֵּיהּ; וּמִיּוֹמָא תְּמִינָאָה, וּלְהַלְאָה, יִתְרְעֵי, לְקָרָבָא קֻרְבָּנָא קֳדָם יְיָ. 
כב,כח וְשׁ֖וֹר אוֹ־שֶׂ֑ה אֹת֣וֹ וְאֶת־בְּנ֔וֹ לֹ֥א תִשְׁחֲט֖וּ בְּי֥וֹם אֶחָֽד׃ וְתוֹרְתָא, אוֹ סִיתָא--לַהּ וְלִבְרַהּ, לָא תִּכְּסוּן בְּיוֹמָא חַד. 
כב,כט וְכִֽי־תִזְבְּח֥וּ זֶֽבַח־תּוֹדָ֖ה לַֽיהוָ֑ה לִֽרְצֹנְכֶ֖ם תִּזְבָּֽחוּ׃ וַאֲרֵי תִּכְּסוּן נִכְסַת תּוֹדְתָא, קֳדָם יְיָ--לְרַעֲוָא לְכוֹן, תִּכְּסוּן. 
כב,ל בַּיּ֤וֹם הַהוּא֙ יֵֽאָכֵ֔ל לֹֽא־תוֹתִ֥ירוּ מִמֶּ֖נּוּ עַד־בֹּ֑קֶר אֲנִ֖י יְהוָֽה׃ בְּיוֹמָא הַהוּא יִתְאֲכִיל, לָא תַּשְׁאֲרוּן מִנֵּיהּ עַד צַפְרָא:  אֲנָא, יְיָ. 
כב,לא וּשְׁמַרְתֶּם֙ מִצְו‍ֹתַ֔י וַֽעֲשִׂיתֶ֖ם אֹתָ֑ם אֲנִ֖י יְהוָֽה׃ וְתִטְּרוּן, פִּקּוֹדַי, וְתַעְבְּדוּן, יָתְהוֹן:  אֲנָא, יְיָ. 
כב,לב וְלֹ֤א תְחַלְּלוּ֙ אֶת־שֵׁ֣ם קָדְשִׁ֔י וְנִ֨קְדַּשְׁתִּ֔י בְּת֖וֹךְ בְּנֵ֣י יִשְׂרָאֵ֑ל אֲנִ֥י יְהוָ֖ה מְקַדִּשְׁכֶֽם׃ וְלָא תַּחֲלוּן, יָת שְׁמָא דְּקֻדְשִׁי, וְאֶתְקַדַּשׁ, בְּגוֹ בְּנֵי יִשְׂרָאֵל:  אֲנָא יְיָ, מְקַדִּשְׁכוֹן. 
כב,לג הַמּוֹצִ֤יא אֶתְכֶם֙ מֵאֶ֣רֶץ מִצְרַ֔יִם לִֽהְי֥וֹת לָכֶ֖ם לֵֽאלֹהִ֑ים אֲנִ֖י יְהוָֽה׃ {פ} דְּאַפֵּיק יָתְכוֹן מֵאַרְעָא דְּמִצְרַיִם, לְמִהְוֵי לְכוֹן לֶאֱלָהּ:  אֲנָא, יְיָ. {פ}
כג,א וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
כג,ב דַּבֵּ֞ר אֶל־בְּנֵ֤י יִשְׂרָאֵל֙ וְאָֽמַרְתָּ֣ אֲלֵהֶ֔ם מֽוֹעֲדֵ֣י יְהוָ֔ה אֲשֶׁר־תִּקְרְא֥וּ אֹתָ֖ם מִקְרָאֵ֣י קֹ֑דֶשׁ אֵ֥לֶּה הֵ֖ם מֽוֹעֲדָֽי׃ מַלֵּיל עִם בְּנֵי יִשְׂרָאֵל, וְתֵימַר לְהוֹן, מוֹעֲדַיָּא דַּייָ, דִּתְעָרְעוּן יָתְהוֹן מְעָרְעֵי קַדִּישׁ--אִלֵּין אִנּוּן, מוֹעֲדָי. 
כג,ג שֵׁ֣שֶׁת יָמִים֮ תֵּֽעָשֶׂ֣ה מְלָאכָה֒ וּבַיּ֣וֹם הַשְּׁבִיעִ֗י שַׁבַּ֤ת שַׁבָּתוֹן֙ מִקְרָא־קֹ֔דֶשׁ כָּל־מְלָאכָ֖ה לֹ֣א תַֽעֲשׂ֑וּ שַׁבָּ֥ת הִוא֙ לַֽיהוָ֔ה בְּכֹ֖ל מוֹשְׁבֹֽתֵיכֶֽם׃ {פ} שִׁתָּא יוֹמִין, תִּתְעֲבֵיד עֲבִידְתָא, וּבְיוֹמָא שְׁבִיעָאָה שַׁבָּא שַׁבָּתָא מְעָרַע קַדִּישׁ, כָּל עֲבִידָא לָא תַּעְבְּדוּן:  שַׁבְּתָא הִיא קֳדָם יְיָ, בְּכֹל מוֹתְבָנֵיכוֹן. {פ}
כג,ד אֵ֚לֶּה מֽוֹעֲדֵ֣י יְהוָ֔ה מִקְרָאֵ֖י קֹ֑דֶשׁ אֲשֶׁר־תִּקְרְא֥וּ אֹתָ֖ם בְּמֽוֹעֲדָֽם׃ אִלֵּין מוֹעֲדַיָּא דַּייָ, מְעָרְעֵי קַדִּישׁ, דִּתְעָרְעוּן יָתְהוֹן, בְּזִמְנֵיהוֹן. 
כג,ה בַּחֹ֣דֶשׁ הָֽרִאשׁ֗וֹן בְּאַרְבָּעָ֥ה עָשָׂ֛ר לַחֹ֖דֶשׁ בֵּ֣ין הָֽעַרְבָּ֑יִם פֶּ֖סַח לַֽיהוָֽה׃ בְּיַרְחָא קַדְמָאָה, בְּאַרְבְּעַת עַסְרָא לְיַרְחָא--בֵּין שִׁמְשַׁיָּא:  פִּסְחָא, קֳדָם יְיָ. 
כג,ו וּבַֽחֲמִשָּׁ֨ה עָשָׂ֥ר יוֹם֙ לַחֹ֣דֶשׁ הַזֶּ֔ה חַ֥ג הַמַּצּ֖וֹת לַֽיהוָ֑ה שִׁבְעַ֥ת יָמִ֖ים מַצּ֥וֹת תֹּאכֵֽלוּ׃ וּבַחֲמֵישְׁתְּ עַסְרָא יוֹמָא לְיַרְחָא הָדֵין, חַגָּא דְּפַטִּירַיָּא קֳדָם יְיָ:  שִׁבְעָא יוֹמִין, פַּטִּירָא תֵּיכְלוּן. 
כג,ז בַּיּוֹם֙ הָֽרִאשׁ֔וֹן מִקְרָא־קֹ֖דֶשׁ יִֽהְיֶ֣ה לָכֶ֑ם כָּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֥א תַֽעֲשֽׂוּ׃ בְּיוֹמָא, קַדְמָאָה, מְעָרַע קַדִּישׁ, יְהֵי לְכוֹן; כָּל עֲבִידַת פֻּלְחַן, לָא תַּעְבְּדוּן. 
כג,ח וְהִקְרַבְתֶּ֥ם אִשֶּׁ֛ה לַֽיהוָ֖ה שִׁבְעַ֣ת יָמִ֑ים בַּיּ֤וֹם הַשְּׁבִיעִי֙ מִקְרָא־קֹ֔דֶשׁ כָּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֥א תַֽעֲשֽׂוּ׃ {פ} וּתְקָרְבוּן קֻרְבָּנָא קֳדָם יְיָ, שִׁבְעָא יוֹמִין; בְּיוֹמָא שְׁבִיעָאָה מְעָרַע קַדִּישׁ, כָּל עֲבִידַת פֻּלְחַן לָא תַּעְבְּדוּן. {פ}
כג,ט וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
כג,י דַּבֵּ֞ר אֶל־בְּנֵ֤י יִשְׂרָאֵל֙ וְאָֽמַרְתָּ֣ אֲלֵהֶ֔ם כִּֽי־תָבֹ֣אוּ אֶל־הָאָ֗רֶץ אֲשֶׁ֤ר אֲנִי֙ נֹתֵ֣ן לָכֶ֔ם וּקְצַרְתֶּ֖ם אֶת־קְצִירָ֑הּ וַֽהֲבֵאתֶ֥ם אֶת־עֹ֛מֶר רֵאשִׁ֥ית קְצִֽירְכֶ֖ם אֶל־הַכֹּהֵֽן׃ מַלֵּיל עִם בְּנֵי יִשְׂרָאֵל, וְתֵימַר לְהוֹן, אֲרֵי תֵּיעֲלוּן לְאַרְעָא דַּאֲנָא יָהֵיב לְכוֹן, וְתִחְצְדוּן יָת חֲצָדַהּ--וְתַיְתוֹן יָת עֹמֶר רֵישׁ חֲצָדְכוֹן, לְוָת כָּהֲנָא. 
כג,יא וְהֵנִ֧יף אֶת־הָעֹ֛מֶר לִפְנֵ֥י יְהוָ֖ה לִֽרְצֹנְכֶ֑ם מִֽמָּחֳרַת֙ הַשַּׁבָּ֔ת יְנִיפֶ֖נּוּ הַכֹּהֵֽן׃ וִירִים יָת עֻמְרָא קֳדָם יְיָ, לְרַעֲוָא לְכוֹן; מִבָּתַר, יוֹמָא טָבָא, יְרִימִנֵּיהּ, כָּהֲנָא. 
כג,יב וַֽעֲשִׂיתֶ֕ם בְּי֥וֹם הֲנִֽיפְכֶ֖ם אֶת־הָעֹ֑מֶר כֶּ֣בֶשׂ תָּמִ֧ים בֶּן־שְׁנָת֛וֹ לְעֹלָ֖ה לַֽיהוָֽה׃ וְתַעְבְּדוּן, בְּיוֹם אֲרָמוּתְכוֹן יָת עֻמְרָא, אִמַּר שְׁלִים בַּר שַׁתֵּיהּ לַעֲלָתָא, קֳדָם יְיָ. 
כג,יג וּמִנְחָתוֹ֩ שְׁנֵ֨י עֶשְׂרֹנִ֜ים סֹ֣לֶת בְּלוּלָ֥ה בַשֶּׁ֛מֶן אִשֶּׁ֥ה לַֽיהוָ֖ה רֵ֣יחַ נִיחֹ֑חַ וְנִסְכֹּ֥ה יַ֖יִן רְבִיעִ֥ת הַהִֽין׃ וּמִנְחָתֵיהּ תְּרֵין עֶסְרוֹנִין סֻלְתָּא דְּפִילָא בִּמְשַׁח, קֻרְבָּנָא קֳדָם יְיָ--לְאִתְקַבָּלָא בְּרַעֲוָא; וְנִסְכֵּיהּ חַמְרָא, רַבְעוּת הִינָא. 
כג,יד וְלֶחֶם֩ וְקָלִ֨י וְכַרְמֶ֜ל לֹ֣א תֹֽאכְל֗וּ עַד־עֶ֨צֶם֙ הַיּ֣וֹם הַזֶּ֔ה עַ֚ד הֲבִ֣יאֲכֶ֔ם אֶת־קָרְבַּ֖ן אֱלֹֽהֵיכֶ֑ם חֻקַּ֤ת עוֹלָם֙ לְדֹרֹ֣תֵיכֶ֔ם בְּכֹ֖ל מֹשְׁבֹֽתֵיכֶֽם׃ {ס} וּלְחֵים וּקְלֵי וּפֵירוּכָן לָא תֵּיכְלוּן, עַד כְּרַן יוֹמָא הָדֵין--עַד אַיְתוֹאֵיכוֹן, יָת קֻרְבָּנָא דֶּאֱלָהֲכוֹן:  קְיָם עָלַם לְדָרֵיכוֹן, בְּכֹל מוֹתְבָנֵיכוֹן. {ס}
כג,טו וּסְפַרְתֶּ֤ם לָכֶם֙ מִמָּֽחֳרַ֣ת הַשַּׁבָּ֔ת מִיּוֹם֙ הֲבִ֣יאֲכֶ֔ם אֶת־עֹ֖מֶר הַתְּנוּפָ֑ה שֶׁ֥בַע שַׁבָּת֖וֹת תְּמִימֹ֥ת תִּֽהְיֶֽינָה׃ וְתִמְנוֹן לְכוֹן, מִבָּתַר יוֹמָא טָבָא, מִיּוֹם אַיְתוֹאֵיכוֹן, יָת עֻמְרָא דַּאֲרָמוּתָא:  שְׁבַע שָׁבוּעָן, שַׁלְמָן יִהְוְיָן. 
כג,טז עַ֣ד מִֽמָּחֳרַ֤ת הַשַּׁבָּת֙ הַשְּׁבִיעִ֔ת תִּסְפְּר֖וּ חֲמִשִּׁ֣ים י֑וֹם וְהִקְרַבְתֶּ֛ם מִנְחָ֥ה חֲדָשָׁ֖ה לַֽיהוָֽה׃ עַד מִבָּתַר שְׁבוּעֲתָא שְׁבִיעֵיתָא, תִּמְנוֹן חַמְשִׁין יוֹמִין; וּתְקָרְבוּן מִנְחָתָא חֲדַתָּא, קֳדָם יְיָ. 
כג,יז מִמּוֹשְׁבֹ֨תֵיכֶ֜ם תָּבִ֣יאּוּ ׀ לֶ֣חֶם תְּנוּפָ֗ה שְׁ֚תַּיִם שְׁנֵ֣י עֶשְׂרֹנִ֔ים סֹ֣לֶת תִּֽהְיֶ֔ינָה חָמֵ֖ץ תֵּֽאָפֶ֑ינָה בִּכּוּרִ֖ים לַֽיהוָֽה׃ מִמּוֹתְבָנֵיכוֹן תַּיְתוֹן לְחֵים אֲרָמוּתָא, תַּרְתֵּין גְּרִיצָן תְּרֵין עֶסְרוֹנִין--סֻלְתָּא יִהְוְיָן, חֲמִיעַ יִתְאַפְיָן:  בִּכּוּרִין, קֳדָם יְיָ. 
כג,יח וְהִקְרַבְתֶּ֣ם עַל־הַלֶּ֗חֶם שִׁבְעַ֨ת כְּבָשִׂ֤ים תְּמִימִם֙ בְּנֵ֣י שָׁנָ֔ה וּפַ֧ר בֶּן־בָּקָ֛ר אֶחָ֖ד וְאֵילִ֣ם שְׁנָ֑יִם יִֽהְי֤וּ עֹלָה֙ לַֽיהוָ֔ה וּמִנְחָתָם֙ וְנִסְכֵּיהֶ֔ם אִשֵּׁ֥ה רֵֽיחַ־נִיחֹ֖חַ לַֽיהוָֽה׃ וּתְקָרְבוּן עַל לַחְמָא, שִׁבְעָא אִמְּרִין שַׁלְמִין בְּנֵי שְׁנָא, וְתוֹר בַּר תּוֹרֵי חַד, וְדִכְרִין תְּרֵין:  יְהוֹן עֲלָתָא, קֳדָם יְיָ, וּמִנְחָתְהוֹן וְנִסְכֵּיהוֹן, קֻרְבַּן דְּמִתְקַבַּל בְּרַעֲוָא קֳדָם יְיָ. 
כג,יט וַֽעֲשִׂיתֶ֛ם שְׂעִיר־עִזִּ֥ים אֶחָ֖ד לְחַטָּ֑את וּשְׁנֵ֧י כְבָשִׂ֛ים בְּנֵ֥י שָׁנָ֖ה לְזֶ֥בַח שְׁלָמִֽים׃ וְתַעְבְּדוּן צְפִיר בַּר עִזִּין חַד, לְחַטָּתָא; וּתְרֵין אִמְּרִין בְּנֵי שְׁנָא, לְנִכְסַת קֻדְשַׁיָּא. 
כג,כ וְהֵנִ֣יף הַכֹּהֵ֣ן ׀ אֹתָ֡ם עַל֩ לֶ֨חֶם הַבִּכֻּרִ֤ים תְּנוּפָה֙ לִפְנֵ֣י יְהוָ֔ה עַל־שְׁנֵ֖י כְּבָשִׂ֑ים קֹ֛דֶשׁ יִֽהְי֥וּ לַֽיהוָ֖ה לַכֹּהֵֽן׃ וִירִים כָּהֲנָא יָתְהוֹן עַל לַחְמָא דְּבִכּוּרַיָּא אֲרָמָא, קֳדָם יְיָ, עַל תְּרֵין, אִמְּרִין; קֻדְשָׁא יְהוֹן קֳדָם יְיָ, לְכָהֲנָא. 
כג,כא וּקְרָאתֶ֞ם בְּעֶ֣צֶם ׀ הַיּ֣וֹם הַזֶּ֗ה מִֽקְרָא־קֹ֨דֶשׁ֙ יִֽהְיֶ֣ה לָכֶ֔ם כָּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֣א תַֽעֲשׂ֑וּ חֻקַּ֥ת עוֹלָ֛ם בְּכָל־מוֹשְׁבֹ֥תֵיכֶ֖ם לְדֹרֹֽתֵיכֶֽם׃ וּתְעָרְעוּן בִּכְרַן יוֹמָא הָדֵין, מְעָרַע קַדִּישׁ יְהֵי לְכוֹן--כָּל עֲבִידַת פֻּלְחַן, לָא תַּעְבְּדוּן:  קְיָם עָלַם בְּכָל מוֹתְבָנֵיכוֹן, לְדָרֵיכוֹן. 
כג,כב וּֽבְקֻצְרְכֶ֞ם אֶת־קְצִ֣יר אַרְצְכֶ֗ם לֹֽא־תְכַלֶּ֞ה פְּאַ֤ת שָֽׂדְךָ֙ בְּקֻצְרֶ֔ךָ וְלֶ֥קֶט קְצִֽירְךָ֖ לֹ֣א תְלַקֵּ֑ט לֶֽעָנִ֤י וְלַגֵּר֙ תַּֽעֲזֹ֣ב אֹתָ֔ם אֲנִ֖י יְהוָ֥ה אֱלֹֽהֵיכֶֽם׃ {פ} וּבְמִחְצַדְכוֹן יָת חֲצָדָא דַּאֲרַעְכוֹן, לָא תְּשֵׁיצֵי פָּתָא דְּחַקְלָךְ בִּחְצָדָךְ, וּלְקָטָא דִּחְצָדָךְ, לָא תְּלַקֵּיט; לְעַנְיֵי וּלְגִיּוֹרֵי תִּשְׁבּוֹק יָתְהוֹן, אֲנָא יְיָ אֱלָהֲכוֹן. {פ}
כג,כג וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
כג,כד דַּבֵּ֛ר אֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר בַּחֹ֨דֶשׁ הַשְּׁבִיעִ֜י בְּאֶחָ֣ד לַחֹ֗דֶשׁ יִֽהְיֶ֤ה לָכֶם֙ שַׁבָּת֔וֹן זִכְר֥וֹן תְּרוּעָ֖ה מִקְרָא־קֹֽדֶשׁ׃ מַלֵּיל עִם בְּנֵי יִשְׂרָאֵל, לְמֵימַר:  בְּיַרְחָא שְׁבִיעָאָה בְּחַד לְיַרְחָא, יְהֵי לְכוֹן נְיָחָא--דֻּכְרַן יַבָּבָא, מְעָרַע קַדִּישׁ. 
כג,כה כָּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֣א תַֽעֲשׂ֑וּ וְהִקְרַבְתֶּ֥ם אִשֶּׁ֖ה לַֽיהוָֽה׃ {ס} כָּל עֲבִידַת פֻּלְחַן, לָא תַּעְבְּדוּן; וּתְקָרְבוּן קֻרְבָּנָא, קֳדָם יְיָ. {ס}
כג,כו וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
כג,כז אַ֡ךְ בֶּֽעָשׂ֣וֹר לַחֹדֶשׁ֩ הַשְּׁבִיעִ֨י הַזֶּ֜ה י֧וֹם הַכִּפֻּרִ֣ים ה֗וּא מִֽקְרָא־קֹ֨דֶשׁ֙ יִֽהְיֶ֣ה לָכֶ֔ם וְעִנִּיתֶ֖ם אֶת־נַפְשֹֽׁתֵיכֶ֑ם וְהִקְרַבְתֶּ֥ם אִשֶּׁ֖ה לַֽיהוָֽה׃ בְּרַם בְּעַסְרָא לְיַרְחָא שְׁבִיעָאָה הָדֵין יוֹמָא דְּכִפּוּרַיָּא הוּא, מְעָרַע קַדִּישׁ יְהֵי לְכוֹן, וּתְעַנּוֹן, יָת נַפְשָׁתְכוֹן; וּתְקָרְבוּן קֻרְבָּנָא, קֳדָם יְיָ. 
כג,כח וְכָל־מְלָאכָה֙ לֹ֣א תַֽעֲשׂ֔וּ בְּעֶ֖צֶם הַיּ֣וֹם הַזֶּ֑ה כִּ֣י י֤וֹם כִּפֻּרִים֙ ה֔וּא לְכַפֵּ֣ר עֲלֵיכֶ֔ם לִפְנֵ֖י יְהוָ֥ה אֱלֹֽהֵיכֶֽם׃ וְכָל עֲבִידָא לָא תַּעְבְּדוּן, בִּכְרַן יוֹמָא הָדֵין:  אֲרֵי יוֹמָא דְּכִפּוּרַיָּא, הוּא, לְכַפָּרָא עֲלֵיכוֹן, קֳדָם יְיָ אֱלָהֲכוֹן. 
כג,כט כִּ֤י כָל־הַנֶּ֨פֶשׁ֙ אֲשֶׁ֣ר לֹֽא־תְעֻנֶּ֔ה בְּעֶ֖צֶם הַיּ֣וֹם הַזֶּ֑ה וְנִכְרְתָ֖ה מֵֽעַמֶּֽיהָ׃ אֲרֵי כָּל אֲנָשׁ דְּלָא יִתְעַנֵּי, בִּכְרַן יוֹמָא הָדֵין--וְיִשְׁתֵּיצֵי, מֵעַמֵּיהּ. 
כג,ל וְכָל־הַנֶּ֗פֶשׁ אֲשֶׁ֤ר תַּֽעֲשֶׂה֙ כָּל־מְלָאכָ֔ה בְּעֶ֖צֶם הַיּ֣וֹם הַזֶּ֑ה וְהַֽאֲבַדְתִּ֛י אֶת־הַנֶּ֥פֶשׁ הַהִ֖וא מִקֶּ֥רֶב עַמָּֽהּ׃ וְכָל אֲנָשׁ, דְּיַעֲבֵיד כָּל עֲבִידָא, בִּכְרַן, יוֹמָא הָדֵין--וְאוֹבֵיד יָת אֲנָשָׁא הַהוּא, מִגּוֹ עַמֵּיהּ. 
כג,לא כָּל־מְלָאכָ֖ה לֹ֣א תַֽעֲשׂ֑וּ חֻקַּ֤ת עוֹלָם֙ לְדֹרֹ֣תֵיכֶ֔ם בְּכֹ֖ל מֹשְׁבֹֽתֵיכֶֽם׃ כָּל עֲבִידָא, לָא תַּעְבְּדוּן:  קְיָם עָלַם לְדָרֵיכוֹן, בְּכֹל מוֹתְבָנֵיכוֹן. 
כג,לב שַׁבַּ֨ת שַׁבָּת֥וֹן הוּא֙ לָכֶ֔ם וְעִנִּיתֶ֖ם אֶת־נַפְשֹֽׁתֵיכֶ֑ם בְּתִשְׁעָ֤ה לַחֹ֨דֶשׁ֙ בָּעֶ֔רֶב מֵעֶ֣רֶב עַד־עֶ֔רֶב תִּשְׁבְּת֖וּ שַׁבַּתְּכֶֽם׃ {פ} שַׁבָּא שַׁבָּתָא הוּא לְכוֹן, וּתְעַנּוֹן יָת נַפְשָׁתְכוֹן; בְּתִשְׁעָא לְיַרְחָא, בְּרַמְשָׁא--מֵרַמְשָׁא עַד רַמְשָׁא, תְּנוּחוּן נְיָחֲכוֹן. {פ}
כג,לג וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
כג,לד דַּבֵּ֛ר אֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר בַּֽחֲמִשָּׁ֨ה עָשָׂ֜ר י֗וֹם לַחֹ֤דֶשׁ הַשְּׁבִיעִי֙ הַזֶּ֔ה חַ֧ג הַסֻּכּ֛וֹת שִׁבְעַ֥ת יָמִ֖ים לַֽיהוָֽה׃ מַלֵּיל עִם בְּנֵי יִשְׂרָאֵל, לְמֵימַר:  בַּחֲמֵישְׁתְּ עַסְרָא יוֹמָא, לְיַרְחָא שְׁבִיעָאָה הָדֵין, חַגָּא דִּמְטַלַּיָּא שִׁבְעָא יוֹמִין, קֳדָם יְיָ. 
כג,לה בַּיּ֥וֹם הָֽרִאשׁ֖וֹן מִקְרָא־קֹ֑דֶשׁ כָּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֥א תַֽעֲשֽׂוּ׃ בְּיוֹמָא קַדְמָאָה, מְעָרַע קַדִּישׁ; כָּל עֲבִידַת פֻּלְחַן, לָא תַּעְבְּדוּן. 
כג,לו שִׁבְעַ֣ת יָמִ֔ים תַּקְרִ֥יבוּ אִשֶּׁ֖ה לַֽיהוָ֑ה בַּיּ֣וֹם הַשְּׁמִינִ֡י מִקְרָא־קֹדֶשׁ֩ יִֽהְיֶ֨ה לָכֶ֜ם וְהִקְרַבְתֶּ֨ם אִשֶּׁ֤ה לַֽיהוָה֙ עֲצֶ֣רֶת הִ֔וא כָּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֥א תַֽעֲשֽׂוּ׃ שִׁבְעָא יוֹמִין, תְּקָרְבוּן קֻרְבָּנָא קֳדָם יְיָ; בְּיוֹמָא תְּמִינָאָה מְעָרַע קַדִּישׁ יְהֵי לְכוֹן וּתְקָרְבוּן קֻרְבָּנָא קֳדָם יְיָ, כְּנִישִׁין תְּהוֹן--כָּל עֲבִידַת פֻּלְחַן, לָא תַּעְבְּדוּן. 
כג,לז אֵ֚לֶּה מֽוֹעֲדֵ֣י יְהוָ֔ה אֲשֶׁר־תִּקְרְא֥וּ אֹתָ֖ם מִקְרָאֵ֣י קֹ֑דֶשׁ לְהַקְרִ֨יב אִשֶּׁ֜ה לַֽיהוָ֗ה עֹלָ֧ה וּמִנְחָ֛ה זֶ֥בַח וּנְסָכִ֖ים דְּבַר־י֥וֹם בְּיוֹמֽוֹ׃ אִלֵּין מוֹעֲדַיָּא דַּייָ, דִּתְעָרְעוּן יָתְהוֹן מְעָרְעֵי קַדִּישׁ:  לְקָרָבָא קֻרְבָּנָא קֳדָם יְיָ, עֲלָתָא וּמִנְחָתָא נִכְסַת קֻדְשִׁין וְנִסּוּכִין--פִּתְגָם יוֹם בְּיוֹמֵיהּ. 
כג,לח מִלְּבַ֖ד שַׁבְּתֹ֣ת יְהוָ֑ה וּמִלְּבַ֣ד מַתְּנֽוֹתֵיכֶ֗ם וּמִלְּבַ֤ד כָּל־נִדְרֵיכֶם֙ וּמִלְּבַד֙ כָּל־נִדְבֹ֣תֵיכֶ֔ם אֲשֶׁ֥ר תִּתְּנ֖וּ לַֽיהוָֽה׃ בָּר מִשַּׁבַּיָּא דַּייָ; וּבָר מִמַּתְּנָתְכוֹן, וּבָר מִכָּל נִדְרֵיכוֹן וּבָר מִכָּל נִדְבָתְכוֹן, דְּתִתְּנוּן, קֳדָם יְיָ. 
כג,לט אַ֡ךְ בַּֽחֲמִשָּׁה֩ עָשָׂ֨ר י֜וֹם לַחֹ֣דֶשׁ הַשְּׁבִיעִ֗י בְּאָסְפְּכֶם֙ אֶת־תְּבוּאַ֣ת הָאָ֔רֶץ תָּחֹ֥גּוּ אֶת־חַג־יְהוָ֖ה שִׁבְעַ֣ת יָמִ֑ים בַּיּ֤וֹם הָֽרִאשׁוֹן֙ שַׁבָּת֔וֹן וּבַיּ֥וֹם הַשְּׁמִינִ֖י שַׁבָּתֽוֹן׃ בְּרַם בַּחֲמֵישְׁתְּ עַסְרָא יוֹמָא לְיַרְחָא שְׁבִיעָאָה, בְּמִכְנַשְׁכוֹן יָת עֲלַלְתָּא דְּאַרְעָא, תֵּיחֲגוּן יָת חַגָּא קֳדָם יְיָ, שִׁבְעָא יוֹמִין; בְּיוֹמָא קַדְמָאָה נְיָחָא, וּבְיוֹמָא תְּמִינָאָה נְיָחָא. 
כג,מ וּלְקַחְתֶּ֨ם לָכֶ֜ם בַּיּ֣וֹם הָֽרִאשׁ֗וֹן פְּרִ֨י עֵ֤ץ הָדָר֙ כַּפֹּ֣ת תְּמָרִ֔ים וַֽעֲנַ֥ף עֵץ־עָבֹ֖ת וְעַרְבֵי־נָ֑חַל וּשְׂמַחְתֶּ֗ם לִפְנֵ֛י יְהוָ֥ה אֱלֹֽהֵיכֶ֖ם שִׁבְעַ֥ת יָמִֽים׃ וְתִסְּבוּן לְכוֹן בְּיוֹמָא קַדְמָאָה, פֵּירֵי אִילָנָא אֶתְּרוֹגִין לוֹלַבִּין, וַהֲדַסִּין, וְעַרְבִין דִּנְחַל; וְתִחְדוֹן, קֳדָם יְיָ אֱלָהֲכוֹן--שִׁבְעָא יוֹמִין. 
כג,מא וְחַגֹּתֶ֤ם אֹתוֹ֙ חַ֣ג לַֽיהוָ֔ה שִׁבְעַ֥ת יָמִ֖ים בַּשָּׁנָ֑ה חֻקַּ֤ת עוֹלָם֙ לְדֹרֹ֣תֵיכֶ֔ם בַּחֹ֥דֶשׁ הַשְּׁבִיעִ֖י תָּחֹ֥גּוּ אֹתֽוֹ׃ וְתֵיחֲגוּן יָתֵיהּ חַגָּא קֳדָם יְיָ, שִׁבְעָא יוֹמִין בְּשַׁתָּא:  קְיָם עָלַם לְדָרֵיכוֹן, בְּיַרְחָא שְׁבִיעָאָה תֵּיחֲגוּן יָתֵיהּ. 
כג,מב בַּסֻּכֹּ֥ת תֵּֽשְׁב֖וּ שִׁבְעַ֣ת יָמִ֑ים כָּל־הָֽאֶזְרָח֙ בְּיִשְׂרָאֵ֔ל יֵֽשְׁב֖וּ בַּסֻּכֹּֽת׃ בִּמְטַלַּיָּא תִּתְּבוּן, שִׁבְעָא יוֹמִין; כָּל יַצִּיבַיָּא, בְּיִשְׂרָאֵל, יִתְּבוּן, בִּמְטַלַּיָּא. 
כג,מג לְמַעַן֮ יֵֽדְע֣וּ דֹרֹֽתֵיכֶם֒ כִּ֣י בַסֻּכּ֗וֹת הוֹשַׁ֨בְתִּי֙ אֶת־בְּנֵ֣י יִשְׂרָאֵ֔ל בְּהֽוֹצִיאִ֥י אוֹתָ֖ם מֵאֶ֣רֶץ מִצְרָ֑יִם אֲנִ֖י יְהוָ֥ה אֱלֹֽהֵיכֶֽם׃ בְּדִיל, דְּיִדְּעוּן דָּרֵיכוֹן, אֲרֵי בִּמְטַלַּת עֲנָנִי אוֹתֵיבִית יָת בְּנֵי יִשְׂרָאֵל, בְּאַפָּקוּתִי יָתְהוֹן מֵאַרְעָא דְּמִצְרָיִם:  אֲנָא, יְיָ אֱלָהֲכוֹן. 
כג,מד וַיְדַבֵּ֣ר מֹשֶׁ֔ה אֶת־מֹֽעֲדֵ֖י יְהוָ֑ה אֶל־בְּנֵ֖י יִשְׂרָאֵֽל׃ {פ} וּמַלֵּיל מֹשֶׁה, יָת סְדַר מוֹעֲדַיָּא דַּייָ, וְאַלֵּיפִנּוּן לִבְנֵי, יִשְׂרָאֵל. {פ}
כד,א וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
כד,ב צַ֞ו אֶת־בְּנֵ֣י יִשְׂרָאֵ֗ל וְיִקְח֨וּ אֵלֶ֜יךָ שֶׁ֣מֶן זַ֥יִת זָ֛ךְ כָּתִ֖ית לַמָּא֑וֹר לְהַֽעֲלֹ֥ת נֵ֖ר תָּמִֽיד׃ פַּקֵּיד יָת בְּנֵי יִשְׂרָאֵל, וְיִסְּבוּן לָךְ מְשַׁח זֵיתָא דָּכְיָא כָּתִישָׁא--לְאַנְהָרָא:  לְאַדְלָקָא בּוֹצִינַיָּא, תְּדִירָא. 
כד,ג מִחוּץ֩ לְפָרֹ֨כֶת הָֽעֵדֻ֜ת בְּאֹ֣הֶל מוֹעֵ֗ד יַֽעֲרֹךְ֩ אֹת֨וֹ אַֽהֲרֹ֜ן מֵעֶ֧רֶב עַד־בֹּ֛קֶר לִפְנֵ֥י יְהוָ֖ה תָּמִ֑יד חֻקַּ֥ת עוֹלָ֖ם לְדֹרֹֽתֵיכֶֽם׃ מִבַּרָא לְפָרֻכְתָּא דְּסָהֲדוּתָא בְּמַשְׁכַּן זִמְנָא, יַסְדַּר יָתֵיהּ אַהֲרוֹן מֵרַמְשָׁא עַד צַפְרָא קֳדָם יְיָ--תְּדִירָא:  קְיָם עָלַם, לְדָרֵיכוֹן. 
כד,ד עַ֚ל הַמְּנֹרָ֣ה הַטְּהֹרָ֔ה יַֽעֲרֹ֖ךְ אֶת־הַנֵּר֑וֹת לִפְנֵ֥י יְהוָ֖ה תָּמִֽיד׃ {פ} עַל מְנָרְתָא דָּכִיתָא, יַסְדַּר יָת בּוֹצִינַיָּא, קֳדָם יְיָ, תְּדִירָא. {פ}
כד,ה וְלָֽקַחְתָּ֣ סֹ֔לֶת וְאָֽפִיתָ֣ אֹתָ֔הּ שְׁתֵּ֥ים עֶשְׂרֵ֖ה חַלּ֑וֹת שְׁנֵי֙ עֶשְׂרֹנִ֔ים יִֽהְיֶ֖ה הַֽחַלָּ֥ה הָֽאֶחָֽת׃ וְתִסַּב סֻלְתָּא--וְתֵיפֵי יָתַהּ, תַּרְתַּא עַסְרֵי גְּרִיצָן; תְּרֵין, עֶסְרוֹנִין, תְּהֵי הָוְיָא, גְּרִיצְתָא חֲדָא. 
כד,ו וְשַׂמְתָּ֥ אוֹתָ֛ם שְׁתַּ֥יִם מַֽעֲרָכ֖וֹת שֵׁ֣שׁ הַֽמַּעֲרָ֑כֶת עַ֛ל הַשֻּׁלְחָ֥ן הַטָּהֹ֖ר לִפְנֵ֥י יְהוָֽה׃ וּתְשַׁוֵּי יָתְהוֹן תַּרְתֵּין סִדְרִין, שֵׁית סִדְרָא, עַל פָּתוּרָא דָּכְיָא, קֳדָם יְיָ. 
כד,ז וְנָֽתַתָּ֥ עַל־הַֽמַּעֲרֶ֖כֶת לְבֹנָ֣ה זַכָּ֑ה וְהָֽיְתָ֤ה לַלֶּ֨חֶם֙ לְאַזְכָּרָ֔ה אִשֶּׁ֖ה לַֽיהוָֽה׃ וְתִתֵּין עַל סִדְרָא, לְבוֹנְתָא דָּכִיתָא; וּתְהֵי לִלְחֵים לְאַדְכָרָא, קֻרְבָּנָא קֳדָם יְיָ. 
כד,ח בְּי֨וֹם הַשַּׁבָּ֜ת בְּי֣וֹם הַשַּׁבָּ֗ת יַֽעַרְכֶ֛נּוּ לִפְנֵ֥י יְהוָ֖ה תָּמִ֑יד מֵאֵ֥ת בְּנֵֽי־יִשְׂרָאֵ֖ל בְּרִ֥ית עוֹלָֽם׃ בְּיוֹמָא דְּשַׁבְּתָא בְּיוֹמָא דְּשַׁבְּתָא, יַסְדְּרִנֵּיהּ קֳדָם יְיָ--תְּדִירָא:  מִן בְּנֵי יִשְׂרָאֵל, קְיָם עָלַם. 
כד,ט וְהָֽיְתָה֙ לְאַֽהֲרֹ֣ן וּלְבָנָ֔יו וַֽאֲכָלֻ֖הוּ בְּמָק֣וֹם קָדֹ֑שׁ כִּ֡י קֹדֶשׁ֩ קָֽדָשִׁ֨ים ה֥וּא ל֛וֹ מֵֽאִשֵּׁ֥י יְהוָ֖ה חָק־עוֹלָֽם׃ {ס} וּתְהֵי לְאַהֲרוֹן וְלִבְנוֹהִי, וְיֵיכְלוּנֵּיהּ בַּאֲתַר קַדִּישׁ:  אֲרֵי קֹדֶשׁ קֻדְשִׁין הוּא לֵיהּ, מִקֻּרְבָּנַיָּא דַּייָ--קְיָם עָלַם. {ס}
כד,י וַיֵּצֵא֙ בֶּן־אִשָּׁ֣ה יִשְׂרְאֵלִ֔ית וְהוּא֙ בֶּן־אִ֣ישׁ מִצְרִ֔י בְּת֖וֹךְ בְּנֵ֣י יִשְׂרָאֵ֑ל וַיִּנָּצוּ֙ בַּֽמַּחֲנֶ֔ה בֶּ֚ן הַיִּשְׂרְאֵלִ֔ית וְאִ֖ישׁ הַיִּשְׂרְאֵלִֽי׃ וּנְפַק, בַּר אִתְּתָא בַּת יִשְׂרָאֵל, וְהוּא בַּר גְּבַר מִצְרַאי, בְּגוֹ בְּנֵי יִשְׂרָאֵל; וְאִתְנְצִיאוּ, בְּמַשְׁרִיתָא, בַּר אִתְּתָא בַּת יִשְׂרָאֵל, וְגֻבְרָא בַּר יִשְׂרָאֵל. 
כד,יא וַ֠יִּקֹּב בֶּן־הָֽאִשָּׁ֨ה הַיִּשְׂרְאֵלִ֤ית אֶת־הַשֵּׁם֙ וַיְקַלֵּ֔ל וַיָּבִ֥יאוּ אֹת֖וֹ אֶל־מֹשֶׁ֑ה וְשֵׁ֥ם אִמּ֛וֹ שְׁלֹמִ֥ית בַּת־דִּבְרִ֖י לְמַטֵּה־דָֽן׃ וּפָרֵישׁ בַּר אִתְּתָא בַּת יִשְׂרָאֵל יָת שְׁמָא, וְאַרְגֵּיז, וְאֵיתִיאוּ יָתֵיהּ, לְוָת מֹשֶׁה; וְשׁוֹם אִמֵּיהּ שְׁלוֹמִית בַּת דִּבְרִי, לְשִׁבְטָא דְּדָן. 
כד,יב וַיַּנִּיחֻ֖הוּ בַּמִּשְׁמָ֑ר לִפְרֹ֥שׁ לָהֶ֖ם עַל־פִּ֥י יְהוָֽה׃ {פ} וְאַסְרוּהִי, בְּבֵית מַטְּרָא, עַד דְּיִתְפָּרַשׁ לְהוֹן, עַל גְּזֵירַת מֵימְרָא דַּייָ. {פ}
כד,יג וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
כד,יד הוֹצֵ֣א אֶת־הַֽמְקַלֵּ֗ל אֶל־מִחוּץ֙ לַֽמַּחֲנֶ֔ה וְסָֽמְכ֧וּ כָֽל־הַשֹּׁמְעִ֛ים אֶת־יְדֵיהֶ֖ם עַל־רֹאשׁ֑וֹ וְרָֽגְמ֥וּ אֹת֖וֹ כָּל־הָֽעֵדָֽה׃ אַפֵּיק יָת דְּאַרְגֵּיז, לְמִבַּרָא לְמַשְׁרִיתָא, וְיִסְמְכוּן כָּל דִּשְׁמַעוּ יָת יְדֵיהוֹן, עַל רֵישֵׁיהּ; וְיִרְגְּמוּן יָתֵיהּ, כָּל כְּנִשְׁתָּא. 
כד,טו וְאֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל תְּדַבֵּ֣ר לֵאמֹ֑ר אִ֥ישׁ אִ֛ישׁ כִּֽי־יְקַלֵּ֥ל אֱלֹהָ֖יו וְנָשָׂ֥א חֶטְאֽוֹ׃ וְעִם בְּנֵי יִשְׂרָאֵל, תְּמַלֵּיל לְמֵימַר:  גְּבַר גְּבַר אֲרֵי יַרְגֵּיז קֳדָם אֱלָהֵיהּ, וִיקַבֵּיל חוֹבֵיהּ. 
כד,טז וְנֹקֵ֤ב שֵׁם־יְהוָה֙ מ֣וֹת יוּמָ֔ת רָג֥וֹם יִרְגְּמוּ־ב֖וֹ כָּל־הָֽעֵדָ֑ה כַּגֵּר֙ כָּֽאֶזְרָ֔ח בְּנָקְבוֹ־שֵׁ֖ם יוּמָֽת׃ וְדִיפָרֵישׁ שְׁמָא דַּייָ אִתְקְטָלָא יִתְקְטִיל, מִרְגָּם יִרְגְּמוּן בֵּיהּ כָּל כְּנִשְׁתָּא:  גִּיּוֹרָא, כְּיַצִּיבָא--בְּפָרָשׁוּתֵיהּ שְׁמָא, יִתְקְטִיל. 
כד,יז וְאִ֕ישׁ כִּ֥י יַכֶּ֖ה כָּל־נֶ֣פֶשׁ אָדָ֑ם מ֖וֹת יוּמָֽת׃ וּגְבַר, אֲרֵי יִקְטוֹל כָּל נַפְשָׁא דַּאֲנָשָׁא--אִתְקְטָלָא, יִתְקְטִיל. 
כד,יח וּמַכֵּ֥ה נֶֽפֶשׁ־בְּהֵמָ֖ה יְשַׁלְּמֶ֑נָּה נֶ֖פֶשׁ תַּ֥חַת נָֽפֶשׁ׃ וּדְיִקְטוֹל נְפַשׁ בְּעִירָא, יְשַׁלְּמִנַּהּ--נַפְשָׁא, חֲלָף נַפְשָׁא. 
כד,יט וְאִ֕ישׁ כִּֽי־יִתֵּ֥ן מ֖וּם בַּֽעֲמִית֑וֹ כַּֽאֲשֶׁ֣ר עָשָׂ֔ה כֵּ֖ן יֵעָ֥שֶׂה לּֽוֹ׃ וּגְבַר, אֲרֵי יִתֵּין מוּמָא בְּחַבְרֵיהּ--כְּמָא דַּעֲבַד, כֵּין יִתְעֲבֵיד לֵיהּ. 
כד,כ שֶׁ֚בֶר תַּ֣חַת שֶׁ֔בֶר עַ֚יִן תַּ֣חַת עַ֔יִן שֵׁ֖ן תַּ֣חַת שֵׁ֑ן כַּֽאֲשֶׁ֨ר יִתֵּ֥ן מוּם֙ בָּֽאָדָ֔ם כֵּ֖ן יִנָּ֥תֶן בּֽוֹ׃ תְּבָרָא, חֲלָף תְּבָרָא, עֵינָא חֲלָף עֵינָא, שִׁנָּא חֲלָף שִׁנָּא--כְּמָא דִּיהַב מוּמָא בַּאֲנָשָׁא, כֵּין יִתְיְהֵיב בֵּיהּ. 
כד,כא וּמַכֵּ֥ה בְהֵמָ֖ה יְשַׁלְּמֶ֑נָּה וּמַכֵּ֥ה אָדָ֖ם יוּמָֽת׃ וּדְיִקְטוֹל בְּעִירָא, יְשַׁלְּמִנַּהּ; וּדְיִקְטוֹל אֲנָשָׁא, יִתְקְטִיל. 
כד,כב מִשְׁפַּ֤ט אֶחָד֙ יִֽהְיֶ֣ה לָכֶ֔ם כַּגֵּ֥ר כָּֽאֶזְרָ֖ח יִֽהְיֶ֑ה כִּ֛י אֲנִ֥י יְהוָ֖ה אֱלֹֽהֵיכֶֽם׃ דִּינָא חַד יְהֵי לְכוֹן, גִּיּוֹרָא כְּיַצִּיבָא יְהֵי:  אֲרֵי אֲנָא יְיָ, אֱלָהֲכוֹן. 
כד,כג וַיְדַבֵּ֣ר מֹשֶׁה֮ אֶל־בְּנֵ֣י יִשְׂרָאֵל֒ וַיּוֹצִ֣יאוּ אֶת־הַֽמְקַלֵּ֗ל אֶל־מִחוּץ֙ לַֽמַּחֲנֶ֔ה וַיִּרְגְּמ֥וּ אֹת֖וֹ אָ֑בֶן וּבְנֵֽי־יִשְׂרָאֵ֣ל עָשׂ֔וּ כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ {פ} וּמַלֵּיל מֹשֶׁה, עִם בְּנֵי יִשְׂרָאֵל, וְאַפִּיקוּ יָת דְּאַרְגֵּיז לְמִבַּרָא לְמַשְׁרִיתָא, וּרְגַמוּ יָתֵיהּ אַבְנָא; וּבְנֵי יִשְׂרָאֵל עֲבַדוּ, כְּמָא דְּפַקֵּיד יְיָ יָת מֹשֶׁה. {פ}
