בהעלותך

ח,א וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ח,ב דַּבֵּר֙ אֶֽל־אַהֲרֹ֔ן וְאָֽמַרְתָּ֖ אֵלָ֑יו בְּהַעֲלֹֽתְךָ֙ אֶת־הַנֵּרֹ֔ת אֶל־מוּל֙ פְּנֵ֣י הַמְּנוֹרָ֔ה יָאִ֖ירוּ שִׁבְעַ֥ת הַנֵּרֽוֹת׃ מַלֵּיל, עִם אַהֲרוֹן, וְתֵימַר, לֵיהּ:  בְּאַדְלָקוּתָךְ, יָת בּוֹצִינַיָּא, לָקֳבֵיל אַפֵּי מְנָרְתָא, יְהוֹן מְנָהֲרִין שִׁבְעָא בּוֹצִינַיָּא. 
ח,ג וַיַּ֤עַשׂ כֵּן֙ אַֽהֲרֹ֔ן אֶל־מוּל֙ פְּנֵ֣י הַמְּנוֹרָ֔ה הֶֽעֱלָ֖ה נֵֽרֹתֶ֑יהָ כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ וַעֲבַד כֵּין, אַהֲרוֹן--לָקֳבֵיל אַפֵּי מְנָרְתָא, אַדְלֵיק בּוֹצִינַהָא:  כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. 
ח,ד וְזֶ֨ה מַֽעֲשֵׂ֤ה הַמְּנֹרָה֙ מִקְשָׁ֣ה זָהָ֔ב עַד־יְרֵכָ֥הּ עַד־פִּרְחָ֖הּ מִקְשָׁ֣ה הִ֑וא כַּמַּרְאֶ֗ה אֲשֶׁ֨ר הֶרְאָ֤ה יְהוָה֙ אֶת־מֹשֶׁ֔ה כֵּ֥ן עָשָׂ֖ה אֶת־הַמְּנֹרָֽה׃ {פ} וְדֵין עוֹבָד מְנָרְתָא נְגִידָא דְּהַב, עַד שִׁדַּהּ עַד שׁוֹשַׁנַּהּ נְגִידָא הִיא:  כְּחֶזְוָא, דְּאַחְזִי יְיָ יָת מֹשֶׁה--כֵּין עֲבַד, יָת מְנָרְתָא. {פ}
ח,ה וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ח,ו קַ֚ח אֶת־הַלְוִיִּ֔ם מִתּ֖וֹךְ בְּנֵ֣י יִשְׂרָאֵ֑ל וְטִֽהַרְתָּ֖ אֹתָֽם׃ קָרֵיב, יָת לֵיוָאֵי, מִגּוֹ, בְּנֵי יִשְׂרָאֵל; וּתְדַכֵּי, יָתְהוֹן. 
ח,ז וְכֹֽה־תַעֲשֶׂ֤ה לָהֶם֙ לְטַֽהֲרָ֔ם הַזֵּ֥ה עֲלֵיהֶ֖ם מֵ֣י חַטָּ֑את וְהֶֽעֱבִ֤ירוּ תַ֨עַר֙ עַל־כָּל־בְּשָׂרָ֔ם וְכִבְּס֥וּ בִגְדֵיהֶ֖ם וְהִטֶּהָֽרוּ׃ וּכְדֵין תַּעֲבֵיד לְהוֹן לְדַכּוֹאֵיהוֹן, אַדִּי עֲלֵיהוֹן מַיָּא דְּחַטָּתָא; וְיַעְבְּרוּן מַסְפַּר עַל כָּל בִּסְרְהוֹן, וִיחַוְּרוּן לְבוּשֵׁיהוֹן וְיִדְכּוֹן. 
ח,ח וְלָֽקְחוּ֙ פַּ֣ר בֶּן־בָּקָ֔ר וּמִ֨נְחָת֔וֹ סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן וּפַר־שֵׁנִ֥י בֶן־בָּקָ֖ר תִּקַּ֥ח לְחַטָּֽאת׃ וְיִסְּבוּן, תּוֹר בַּר תּוֹרֵי, וּמִנְחָתֵיהּ, סֻלְתָּא דְּפִילָא בִּמְשַׁח; וְתוֹר תִּנְיָן בַּר תּוֹרֵי, תִּסַּב לְחַטָּתָא. 
ח,ט וְהִקְרַבְתָּ֙ אֶת־הַלְוִיִּ֔ם לִפְנֵ֖י אֹ֣הֶל מוֹעֵ֑ד וְהִ֨קְהַלְתָּ֔ אֶֽת־כָּל־עֲדַ֖ת בְּנֵ֥י יִשְׂרָאֵֽל׃ וּתְקָרֵיב, יָת לֵיוָאֵי, קֳדָם, מַשְׁכַּן זִמְנָא; וְתִכְנוֹשׁ--יָת כָּל כְּנִשְׁתָּא, דִּבְנֵי יִשְׂרָאֵל. 
ח,י וְהִקְרַבְתָּ֥ אֶת־הַלְוִיִּ֖ם לִפְנֵ֣י יְהוָ֑ה וְסָֽמְכ֧וּ בְנֵֽי־יִשְׂרָאֵ֛ל אֶת־יְדֵיהֶ֖ם עַל־הַלְוִיִּֽם׃ וּתְקָרֵיב יָת לֵיוָאֵי, קֳדָם יְיָ; וְיִסְמְכוּן בְּנֵי יִשְׂרָאֵל יָת יְדֵיהוֹן, עַל לֵיוָאֵי. 
ח,יא וְהֵנִיף֩ אַֽהֲרֹ֨ן אֶת־הַלְוִיִּ֤ם תְּנוּפָה֙ לִפְנֵ֣י יְהוָ֔ה מֵאֵ֖ת בְּנֵ֣י יִשְׂרָאֵ֑ל וְהָי֕וּ לַֽעֲבֹ֖ד אֶת־עֲבֹדַ֥ת יְהוָֽה׃ וִירִים אַהֲרוֹן יָת לֵיוָאֵי אֲרָמָא קֳדָם יְיָ, מִן בְּנֵי יִשְׂרָאֵל; וִיהוֹן, לְמִפְלַח יָת פֻּלְחָנָא דַּייָ. 
ח,יב וְהַלְוִיִּם֙ יִסְמְכ֣וּ אֶת־יְדֵיהֶ֔ם עַ֖ל רֹ֣אשׁ הַפָּרִ֑ים וַֽ֠עֲשֵׂה אֶת־הָֽאֶחָ֨ד חַטָּ֜את וְאֶת־הָֽאֶחָ֤ד עֹלָה֙ לַֽיהוָ֔ה לְכַפֵּ֖ר עַל־הַלְוִיִּֽם׃ וְלֵיוָאֵי יִסְמְכוּן יָת יְדֵיהוֹן, עַל רֵישׁ תּוֹרַיָּא; וַעֲבֵיד יָת חַד חַטָּתָא וְיָת חַד עֲלָתָא, קֳדָם יְיָ, לְכַפָּרָא, עַל לֵיוָאֵי. 
ח,יג וְהַֽעֲמַדְתָּ֙ אֶת־הַלְוִיִּ֔ם לִפְנֵ֥י אַֽהֲרֹ֖ן וְלִפְנֵ֣י בָנָ֑יו וְהֵֽנַפְתָּ֥ אֹתָ֛ם תְּנוּפָ֖ה לַֽיהוָֽה׃ וּתְקִים, יָת לֵיוָאֵי, קֳדָם אַהֲרוֹן, וּקְדָם בְּנוֹהִי; וּתְרִים יָתְהוֹן אֲרָמָא, קֳדָם יְיָ. 
ח,יד וְהִבְדַּלְתָּ֙ אֶת־הַלְוִיִּ֔ם מִתּ֖וֹךְ בְּנֵ֣י יִשְׂרָאֵ֑ל וְהָ֥יוּ לִ֖י הַלְוִיִּֽם׃ וְתַפְרֵישׁ, יָת לֵיוָאֵי, מִגּוֹ, בְּנֵי יִשְׂרָאֵל; וִיהוֹן מְשַׁמְּשִׁין קֳדָמַי, לֵיוָאֵי. 
ח,טו וְאַֽחֲרֵי־כֵן֙ יָבֹ֣אוּ הַלְוִיִּ֔ם לַֽעֲבֹ֖ד אֶת־אֹ֣הֶל מוֹעֵ֑ד וְטִֽהַרְתָּ֣ אֹתָ֔ם וְהֵֽנַפְתָּ֥ אֹתָ֖ם תְּנוּפָֽה׃ וּבָתַר כֵּין יֵיעֲלוּן לֵיוָאֵי, לְמִפְלַח יָת מַשְׁכַּן זִמְנָא; וּתְדַכֵּי יָתְהוֹן, וּתְרִים יָתְהוֹן אֲרָמָא. 
ח,טז כִּי֩ נְתֻנִ֨ים נְתֻנִ֥ים הֵ֨מָּה֙ לִ֔י מִתּ֖וֹךְ בְּנֵ֣י יִשְׂרָאֵ֑ל תַּחַת֩ פִּטְרַ֨ת כָּל־רֶ֜חֶם בְּכ֥וֹר כֹּל֙ מִבְּנֵ֣י יִשְׂרָאֵ֔ל לָקַ֥חְתִּי אֹתָ֖ם לִֽי׃ אֲרֵי אַפְרָשָׁא מֻפְרְשִׁין אִנּוּן קֳדָמַי, מִגּוֹ בְּנֵי יִשְׂרָאֵל:  חֲלָף פָּתַח כָּל וַלְדָּא בֻּכְרָא כּוֹלָא, מִבְּנֵי יִשְׂרָאֵל--קָרֵיבִית יָתְהוֹן, קֳדָמָי. 
ח,יז כִּ֣י לִ֤י כָל־בְּכוֹר֙ בִּבְנֵ֣י יִשְׂרָאֵ֔ל בָּֽאָדָ֖ם וּבַבְּהֵמָ֑ה בְּי֗וֹם הַכֹּתִ֤י כָל־בְּכוֹר֙ בְּאֶ֣רֶץ מִצְרַ֔יִם הִקְדַּ֥שְׁתִּי אֹתָ֖ם לִֽי׃ אֲרֵי דִּילִי כָּל בֻּכְרָא בִּבְנֵי יִשְׂרָאֵל, בַּאֲנָשָׁא וּבִבְעִירָא:  בְּיוֹמָא, דִּקְטַלִית כָּל בֻּכְרָא בְּאַרְעָא דְּמִצְרַיִם, אַקְדֵּישִׁית יָתְהוֹן, קֳדָמָי. 
ח,יח וָֽאֶקַּ֖ח אֶת־הַלְוִיִּ֑ם תַּ֥חַת כָּל־בְּכ֖וֹר בִּבְנֵ֥י יִשְׂרָאֵֽל׃ וְקָרֵיבִית, יָת לֵיוָאֵי, חֲלָף כָּל בֻּכְרָא, בִּבְנֵי יִשְׂרָאֵל. 
ח,יט וָֽאֶתְּנָ֨ה אֶת־הַלְוִיִּ֜ם נְתֻנִ֣ים ׀ לְאַֽהֲרֹ֣ן וּלְבָנָ֗יו מִתּוֹךְ֮ בְּנֵ֣י יִשְׂרָאֵל֒ לַֽעֲבֹ֞ד אֶת־עֲבֹדַ֤ת בְּנֵֽי־יִשְׂרָאֵל֙ בְּאֹ֣הֶל מוֹעֵ֔ד וּלְכַפֵּ֖ר עַל־בְּנֵ֣י יִשְׂרָאֵ֑ל וְלֹ֨א יִֽהְיֶ֜ה בִּבְנֵ֤י יִשְׂרָאֵל֙ נֶ֔גֶף בְּגֶ֥שֶׁת בְּנֵֽי־יִשְׂרָאֵ֖ל אֶל־הַקֹּֽדֶשׁ׃ וִיהַבִית יָת לֵיוָאֵי מְסִירִין לְאַהֲרוֹן וְלִבְנוֹהִי, מִגּוֹ בְּנֵי יִשְׂרָאֵל, לְמִפְלַח יָת פֻּלְחַן בְּנֵי יִשְׂרָאֵל בְּמַשְׁכַּן זִמְנָא, וּלְכַפָּרָא עַל בְּנֵי יִשְׂרָאֵל; וְלָא יְהֵי בִּבְנֵי יִשְׂרָאֵל, מוֹתָא, בְּמִקְרַב בְּנֵי יִשְׂרָאֵל, לְקֻדְשָׁא. 
ח,כ וַיַּ֨עַשׂ מֹשֶׁ֧ה וְאַֽהֲרֹ֛ן וְכָל־עֲדַ֥ת בְּנֵֽי־יִשְׂרָאֵ֖ל לַלְוִיִּ֑ם כְּ֠כֹל אֲשֶׁר־צִוָּ֨ה יְהוָ֤ה אֶת־מֹשֶׁה֙ לַלְוִיִּ֔ם כֵּן־עָשׂ֥וּ לָהֶ֖ם בְּנֵ֥י יִשְׂרָאֵֽל׃ וַעֲבַד מֹשֶׁה וְאַהֲרוֹן וְכָל כְּנִשְׁתָּא דִּבְנֵי יִשְׂרָאֵל, לְלֵיוָאֵי:  כְּכֹל דְּפַקֵּיד יְיָ יָת מֹשֶׁה, לְלֵיוָאֵי--כֵּין עֲבַדוּ לְהוֹן, בְּנֵי יִשְׂרָאֵל. 
ח,כא וַיִּֽתְחַטְּא֣וּ הַלְוִיִּ֗ם וַֽיְכַבְּסוּ֙ בִּגְדֵיהֶ֔ם וַיָּ֨נֶף אַֽהֲרֹ֥ן אֹתָ֛ם תְּנוּפָ֖ה לִפְנֵ֣י יְהוָ֑ה וַיְכַפֵּ֧ר עֲלֵיהֶ֛ם אַֽהֲרֹ֖ן לְטַֽהֲרָֽם׃ וְאִדַּכִּיאוּ לֵיוָאֵי, וְחַוַּרוּ לְבוּשֵׁיהוֹן, וַאֲרֵים אַהֲרוֹן יָתְהוֹן אֲרָמָא, קֳדָם יְיָ; וְכַפַּר עֲלֵיהוֹן אַהֲרוֹן, לְדַכּוֹאֵיהוֹן. 
ח,כב וְאַֽחֲרֵי־כֵ֞ן בָּ֣אוּ הַלְוִיִּ֗ם לַֽעֲבֹ֤ד אֶת־עֲבֹֽדָתָם֙ בְּאֹ֣הֶל מוֹעֵ֔ד לִפְנֵ֥י אַֽהֲרֹ֖ן וְלִפְנֵ֣י בָנָ֑יו כַּֽאֲשֶׁר֩ צִוָּ֨ה יְהוָ֤ה אֶת־מֹשֶׁה֙ עַל־הַלְוִיִּ֔ם כֵּ֖ן עָשׂ֥וּ לָהֶֽם׃ {ס} וּבָתַר כֵּין עָאלוּ לֵיוָאֵי, לְמִפְלַח יָת פֻּלְחָנְהוֹן בְּמַשְׁכַּן זִמְנָא, קֳדָם אַהֲרוֹן, וּקְדָם בְּנוֹהִי:  כְּמָא דְּפַקֵּיד יְיָ יָת מֹשֶׁה עַל לֵיוָאֵי, כֵּין עֲבַדוּ לְהוֹן. {ס}
ח,כג וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ח,כד זֹ֖את אֲשֶׁ֣ר לַלְוִיִּ֑ם מִבֶּן֩ חָמֵ֨שׁ וְעֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה יָבוֹא֙ לִצְבֹ֣א צָבָ֔א בַּֽעֲבֹדַ֖ת אֹ֥הֶל מוֹעֵֽד׃ דָּא, דִּלְלֵיוָאֵי:  מִבַּר עַסְרִין וַחֲמֵישׁ שְׁנִין, וּלְעֵילָא, יֵיתֵי לְחַיָּלָא חֵילָא, בְּפֻלְחַן מַשְׁכַּן זִמְנָא. 
ח,כה וּמִבֶּן֙ חֲמִשִּׁ֣ים שָׁנָ֔ה יָשׁ֖וּב מִצְּבָ֣א הָֽעֲבֹדָ֑ה וְלֹ֥א יַֽעֲבֹ֖ד עֽוֹד׃ וּמִבַּר חַמְשִׁין שְׁנִין, יְתוּב מֵחֵיל פֻּלְחָנָא; וְלָא יִפְלַח, עוֹד. 
ח,כו וְשֵׁרֵ֨ת אֶת־אֶחָ֜יו בְּאֹ֤הֶל מוֹעֵד֙ לִשְׁמֹ֣ר מִשְׁמֶ֔רֶת וַֽעֲבֹדָ֖ה לֹ֣א יַֽעֲבֹ֑ד כָּ֛כָה תַּֽעֲשֶׂ֥ה לַלְוִיִּ֖ם בְּמִשְׁמְרֹתָֽם׃ {פ} וִישַׁמֵּישׁ עִם אֲחוֹהִי בְּמַשְׁכַּן זִמְנָא, לְמִטַּר מַטְּרָא, וּפֻלְחָנָא, לָא יִפְלַח; כְּדֵין תַּעֲבֵיד לְלֵיוָאֵי, בְּמַטְּרָתְהוֹן. {פ}
ט,א וַיְדַבֵּ֣ר יְהוָ֣ה אֶל־מֹשֶׁ֣ה בְמִדְבַּר־סִ֠ינַי בַּשָּׁנָ֨ה הַשֵּׁנִ֜ית לְצֵאתָ֨ם מֵאֶ֧רֶץ מִצְרַ֛יִם בַּחֹ֥דֶשׁ הָֽרִאשׁ֖וֹן לֵאמֹֽר׃ וּמַלֵּיל יְיָ עִם מֹשֶׁה בְּמַדְבְּרָא דְּסִינַי בְּשַׁתָּא תִּנְיֵיתָא לְמִפַּקְהוֹן מֵאַרְעָא דְּמִצְרַיִם, בְּיַרְחָא קַדְמָאָה--לְמֵימַר. 
ט,ב וְיַֽעֲשׂ֧וּ בְנֵֽי־יִשְׂרָאֵ֛ל אֶת־הַפָּ֖סַח בְּמֽוֹעֲדֽוֹ׃ וְיַעְבְּדוּן בְּנֵי יִשְׂרָאֵל יָת פִּסְחָא, בְּזִמְנֵיהּ. 
ט,ג בְּאַרְבָּעָ֣ה עָשָֽׂר־י֠וֹם בַּחֹ֨דֶשׁ הַזֶּ֜ה בֵּ֧ין הָֽעַרְבַּ֛יִם תַּֽעֲשׂ֥וּ אֹת֖וֹ בְּמֹֽעֲד֑וֹ כְּכָל־חֻקֹּתָ֥יו וּכְכָל־מִשְׁפָּטָ֖יו תַּֽעֲשׂ֥וּ אֹתֽוֹ׃ בְּאַרְבְּעַת עַסְרָא יוֹמָא בְּיַרְחָא הָדֵין בֵּין שִׁמְשַׁיָּא, תַּעְבְּדוּן יָתֵיהּ--בְּזִמְנֵיהּ; כְּכָל גְּזֵירָתֵיהּ וּכְכָל דְּחָזֵי לֵיהּ, תַּעְבְּדוּן יָתֵיהּ. 
ט,ד וַיְדַבֵּ֥ר מֹשֶׁ֛ה אֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל לַֽעֲשֹׂ֥ת הַפָּֽסַח׃ וּמַלֵּיל מֹשֶׁה עִם בְּנֵי יִשְׂרָאֵל, לְמֶעֱבַד פִּסְחָא. 
ט,ה וַיַּֽעֲשׂ֣וּ אֶת־הַפֶּ֡סַח בָּֽרִאשׁ֡וֹן בְּאַרְבָּעָה֩ עָשָׂ֨ר י֥וֹם לַחֹ֛דֶשׁ בֵּ֥ין הָֽעַרְבַּ֖יִם בְּמִדְבַּ֣ר סִינָ֑י כְּ֠כֹל אֲשֶׁ֨ר צִוָּ֤ה יְהוָה֙ אֶת־מֹשֶׁ֔ה כֵּ֥ן עָשׂ֖וּ בְּנֵ֥י יִשְׂרָאֵֽל׃ וַעֲבַדוּ יָת פִּסְחָא בְּנִיסָן בְּאַרְבְּעַת עַסְרָא יוֹמָא לְיַרְחָא, בֵּין שִׁמְשַׁיָּא--בְּמַדְבְּרָא דְּסִינָי:  כְּכֹל דְּפַקֵּיד יְיָ, יָת מֹשֶׁה--כֵּין עֲבַדוּ, בְּנֵי יִשְׂרָאֵל. 
ט,ו וַיְהִ֣י אֲנָשִׁ֗ים אֲשֶׁ֨ר הָי֤וּ טְמֵאִים֙ לְנֶ֣פֶשׁ אָדָ֔ם וְלֹֽא־יָכְל֥וּ לַֽעֲשֹׂת־הַפֶּ֖סַח בַּיּ֣וֹם הַה֑וּא וַֽיִּקְרְב֞וּ לִפְנֵ֥י מֹשֶׁ֛ה וְלִפְנֵ֥י אַֽהֲרֹ֖ן בַּיּ֥וֹם הַהֽוּא׃ וַהֲווֹ גֻּבְרַיָּא, דַּהֲווֹ מְסָאֲבִין לִטְמֵי נַפְשָׁא דַּאֲנָשָׁא, וְלָא יְכִילוּ לְמֶעֱבַד פִּסְחָא, בְּיוֹמָא הַהוּא; וּקְרִיבוּ קֳדָם מֹשֶׁה, וּקְדָם אַהֲרוֹן--בְּיוֹמָא הַהוּא. 
ט,ז וַ֠יֹּֽאמְרוּ הָֽאֲנָשִׁ֤ים הָהֵ֨מָּה֙ אֵלָ֔יו אֲנַ֥חְנוּ טְמֵאִ֖ים לְנֶ֣פֶשׁ אָדָ֑ם לָ֣מָּה נִגָּרַ֗ע לְבִלְתִּ֨י הַקְרִ֜יב אֶת־קָרְבַּ֤ן יְהוָה֙ בְּמֹ֣עֲד֔וֹ בְּת֖וֹךְ בְּנֵ֥י יִשְׂרָאֵֽל׃ וַאֲמַרוּ גֻּבְרַיָּא הָאִנּוּן, לֵיהּ, אֲנַחְנָא מְסָאֲבִין, לִטְמֵי נַפְשָׁא דַּאֲנָשָׁא; לְמָא נִתְמְנַע, בְּדִיל דְּלָא לְקָרָבָא יָת קֻרְבָּנָא דַּייָ בְּזִמְנֵיהּ, בְּגוֹ, בְּנֵי יִשְׂרָאֵל. 
ט,ח וַיֹּ֥אמֶר אֲלֵהֶ֖ם מֹשֶׁ֑ה עִמְד֣וּ וְאֶשְׁמְעָ֔ה מַה־יְצַוֶּ֥ה יְהוָ֖ה לָכֶֽם׃ {פ} וַאֲמַר לְהוֹן, מֹשֶׁה:  אוֹרִיכוּ עַד דְּאֶשְׁמַע, מָא דְּאֶתְפַּקַּד מִן קֳדָם יְיָ עַל דִּילְכוֹן. {פ}
ט,ט וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ט,י דַּבֵּ֛ר אֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר אִ֣ישׁ אִ֣ישׁ כִּי־יִהְיֶֽה־טָמֵ֣א ׀ לָנֶ֡פֶשׁ אוֹ֩ בְדֶ֨רֶךְ רְחֹקָ֜הׄ לָכֶ֗ם א֚וֹ לְדֹרֹ֣תֵיכֶ֔ם וְעָ֥שָׂה פֶ֖סַח לַֽיהוָֽה׃ מַלֵּיל עִם בְּנֵי יִשְׂרָאֵל, לְמֵימַר:  גְּבַר גְּבַר אֲרֵי יְהֵי מְסָאַב לִטְמֵי נַפְשָׁא דַּאֲנָשָׁא אוֹ בְּאוֹרַח רַחִיקָא לְכוֹן, אוֹ לְדָרֵיכוֹן, וְיַעֲבֵיד פִּסְחָא, קֳדָם יְיָ. 
ט,יא בַּחֹ֨דֶשׁ הַשֵּׁנִ֜י בְּאַרְבָּעָ֨ה עָשָׂ֥ר י֛וֹם בֵּ֥ין הָֽעַרְבַּ֖יִם יַֽעֲשׂ֣וּ אֹת֑וֹ עַל־מַצּ֥וֹת וּמְרֹרִ֖ים יֹֽאכְלֻֽהוּ׃ בְּיַרְחָא תִּנְיָנָא בְּאַרְבְּעַת עַסְרָא יוֹמָא, בֵּין שִׁמְשַׁיָּא--יַעְבְּדוּן יָתֵיהּ:  עַל פַּטִּיר וּמְרָרִין, יֵיכְלוּנֵּיהּ. 
ט,יב לֹֽא־יַשְׁאִ֤ירוּ מִמֶּ֨נּוּ֙ עַד־בֹּ֔קֶר וְעֶ֖צֶם לֹ֣א יִשְׁבְּרוּ־ב֑וֹ כְּכָל־חֻקַּ֥ת הַפֶּ֖סַח יַֽעֲשׂ֥וּ אֹתֽוֹ׃ לָא יַשְׁאֲרוּן מִנֵּיהּ עַד צַפְרָא, וְגַרְמָא לָא יִתְבְּרוּן בֵּיהּ; כְּכָל גְּזֵירַת פִּסְחָא, יַעְבְּדוּן יָתֵיהּ. 
ט,יג וְהָאִישׁ֩ אֲשֶׁר־ה֨וּא טָה֜וֹר וּבְדֶ֣רֶךְ לֹֽא־הָיָ֗ה וְחָדַל֙ לַֽעֲשׂ֣וֹת הַפֶּ֔סַח וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מֵֽעַמֶּ֑יהָ כִּ֣י ׀ קָרְבַּ֣ן יְהוָ֗ה לֹ֤א הִקְרִיב֙ בְּמֹ֣עֲד֔וֹ חֶטְא֥וֹ יִשָּׂ֖א הָאִ֥ישׁ הַהֽוּא׃ וְגֻבְרָא דְּהוּא דְּכֵי וּבְאוֹרַח לָא הֲוָה, וְיִתְמְנַע מִלְּמֶעֱבַד פִּסְחָא--וְיִשְׁתֵּיצֵי אֲנָשָׁא הַהוּא, מֵעַמֵּיהּ:  אֲרֵי קֻרְבָּנָא דַּייָ, לָא קָרֵיב בְּזִמְנֵיהּ--חוֹבֵיהּ יְקַבֵּיל, גֻּבְרָא הַהוּא. 
ט,יד וְכִֽי־יָג֨וּר אִתְּכֶ֜ם גֵּ֗ר וְעָ֤שָֽׂה פֶ֨סַח֙ לַֽיהוָ֔ה כְּחֻקַּ֥ת הַפֶּ֛סַח וּכְמִשְׁפָּט֖וֹ כֵּ֣ן יַֽעֲשֶׂ֑ה חֻקָּ֤ה אַחַת֙ יִֽהְיֶ֣ה לָכֶ֔ם וְלַגֵּ֖ר וּלְאֶזְרַ֥ח הָאָֽרֶץ׃ {ס} וַאֲרֵי יִתְגַּיַּר עִמְּכוֹן גִּיּוֹרָא, וְיַעֲבֵיד פִּסְחָא קֳדָם יְיָ--כִּגְזֵירַת פִּסְחָא וּכְדַחְזֵי לֵיהּ, כֵּין יַעֲבֵיד:  קְיָמָא חַד יְהֵי לְכוֹן, וּלְגִיּוֹרַיָּא וּלְיַצִּיבַיָּא דְּאַרְעָא. {ס}
ט,טו וּבְיוֹם֙ הָקִ֣ים אֶת־הַמִּשְׁכָּ֔ן כִּסָּ֤ה הֶֽעָנָן֙ אֶת־הַמִּשְׁכָּ֔ן לְאֹ֖הֶל הָֽעֵדֻ֑ת וּבָעֶ֜רֶב יִֽהְיֶ֧ה עַֽל־הַמִּשְׁכָּ֛ן כְּמַרְאֵה־אֵ֖שׁ עַד־בֹּֽקֶר׃ וּבְיוֹמָא, דְּאִתָּקַם מַשְׁכְּנָא, חֲפָא עֲנָנָא יָת מַשְׁכְּנָא, לְמַשְׁכְּנָא דְּסָהֲדוּתָא; וּבְרַמְשָׁא הָוֵי עַל מַשְׁכְּנָא, כְּחֵיזוּ אִישָׁתָא--עַד צַפְרָא. 
ט,טז כֵּ֚ן יִֽהְיֶ֣ה תָמִ֔יד הֶֽעָנָ֖ן יְכַסֶּ֑נּוּ וּמַרְאֵה־אֵ֖שׁ לָֽיְלָה׃ כֵּין הָוֵי תְּדִירָא, עֲנָנָא חָפֵי לֵיהּ; וְחֵיזוּ אִישָׁתָא, בְּלֵילְיָא. 
ט,יז וּלְפִ֞י הֵֽעָל֤וֹת הֶֽעָנָן֙ מֵעַ֣ל הָאֹ֔הֶל וְאַ֣חֲרֵי כֵ֔ן יִסְע֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל וּבִמְק֗וֹם אֲשֶׁ֤ר יִשְׁכָּן־שָׁם֙ הֶֽעָנָ֔ן שָׁ֥ם יַֽחֲנ֖וּ בְּנֵ֥י יִשְׂרָאֵֽל׃ וּלְפֹם אִסְתַּלָּקוּת עֲנָנָא, מֵעִלָּוֵי מַשְׁכְּנָא--וּבָתַר כֵּין, נָטְלִין בְּנֵי יִשְׂרָאֵל; וּבַאֲתַר, דְּשָׁרֵי תַּמָּן עֲנָנָא--תַּמָּן שָׁרַן, בְּנֵי יִשְׂרָאֵל. 
ט,יח עַל־פִּ֣י יְהוָ֗ה יִסְעוּ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וְעַל־פִּ֥י יְהוָ֖ה יַֽחֲנ֑וּ כָּל־יְמֵ֗י אֲשֶׁ֨ר יִשְׁכֹּ֧ן הֶֽעָנָ֛ן עַל־הַמִּשְׁכָּ֖ן יַֽחֲנֽוּ׃ עַל מֵימְרָא דַּייָ, נָטְלִין בְּנֵי יִשְׂרָאֵל, וְעַל מֵימְרָא דַּייָ, שָׁרַן:  כָּל יוֹמִין, דְּשָׁרֵי עֲנָנָא עַל מַשְׁכְּנָא--שָׁרַן. 
ט,יט וּבְהַֽאֲרִ֧יךְ הֶֽעָנָ֛ן עַל־הַמִּשְׁכָּ֖ן יָמִ֣ים רַבִּ֑ים וְשָֽׁמְר֧וּ בְנֵֽי־יִשְׂרָאֵ֛ל אֶת־מִשְׁמֶ֥רֶת יְהוָ֖ה וְלֹ֥א יִסָּֽעוּ׃ וּבְאוֹרָכוּת עֲנָנָא עַל מַשְׁכְּנָא, יוֹמִין סַגִּיאִין--וְיִטְּרוּן בְּנֵי יִשְׂרָאֵל יָת מַטְּרַת מֵימְרָא דַּייָ, וְלָא נָטְלִין. 
ט,כ וְיֵ֞שׁ אֲשֶׁ֨ר יִֽהְיֶ֧ה הֶֽעָנָ֛ן יָמִ֥ים מִסְפָּ֖ר עַל־הַמִּשְׁכָּ֑ן עַל־פִּ֤י יְהוָה֙ יַֽחֲנ֔וּ וְעַל־פִּ֥י יְהוָ֖ה יִסָּֽעוּ׃ וְאִית דְּהָוֵי עֲנָנָא, יוֹמִין דְּמִנְיָן--עַל מַשְׁכְּנָא; עַל מֵימְרָא דַּייָ שָׁרַן, וְעַל מֵימְרָא דַּייָ נָטְלִין. 
ט,כא וְיֵ֞שׁ אֲשֶׁר־יִֽהְיֶ֤ה הֶֽעָנָן֙ מֵעֶ֣רֶב עַד־בֹּ֔קֶר וְנַֽעֲלָ֧ה הֶֽעָנָ֛ן בַּבֹּ֖קֶר וְנָסָ֑עוּ א֚וֹ יוֹמָ֣ם וָלַ֔יְלָה וְנַֽעֲלָ֥ה הֶֽעָנָ֖ן וְנָסָֽעוּ׃ וְאִית דְּהָוֵי עֲנָנָא, מֵרַמְשָׁא עַד צַפְרָא, וּמִסְתַּלַּק עֲנָנָא בְּצַפְרָא, וְנָטְלִין; אוֹ יֵימָם וְלֵילֵי, וּמִסְתַּלַּק עֲנָנָא וְנָטְלִין. 
ט,כב אֽוֹ־יֹמַ֜יִם אוֹ־חֹ֣דֶשׁ אֽוֹ־יָמִ֗ים בְּהַֽאֲרִ֨יךְ הֶֽעָנָ֤ן עַל־הַמִּשְׁכָּן֙ לִשְׁכֹּ֣ן עָלָ֔יו יַֽחֲנ֥וּ בְנֵֽי־יִשְׂרָאֵ֖ל וְלֹ֣א יִסָּ֑עוּ וּבְהֵעָֽלֹת֖וֹ יִסָּֽעוּ׃ אוֹ תְּרֵין יוֹמִין אוֹ יַרְחָא אוֹ עִדָּן בְּעִדָּן, בְּאוֹרָכוּת עֲנָנָא עַל מַשְׁכְּנָא לְמִשְׁרֵי עֲלוֹהִי, שָׁרַן בְּנֵי יִשְׂרָאֵל, וְלָא נָטְלִין; וּבְאִסְתַּלָּקוּתֵיהּ, נָטְלִין. 
ט,כג עַל־פִּ֤י יְהוָה֙ יַֽחֲנ֔וּ וְעַל־פִּ֥י יְהוָ֖ה יִסָּ֑עוּ אֶת־מִשְׁמֶ֤רֶת יְהוָה֙ שָׁמָ֔רוּ עַל־פִּ֥י יְהוָ֖ה בְּיַד־מֹשֶֽׁה׃ {פ} עַל מֵימְרָא דַּייָ שָׁרַן, וְעַל מֵימְרָא דַּייָ נָטְלִין:  יָת מַטְּרַת מֵימְרָא דַּייָ נָטְרִין, עַל מֵימְרָא דַּייָ בִּידָא דְּמֹשֶׁה. {פ}
י,א וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
י,ב עֲשֵׂ֣ה לְךָ֗ שְׁתֵּי֙ חֲצֽוֹצְרֹ֣ת כֶּ֔סֶף מִקְשָׁ֖ה תַּֽעֲשֶׂ֣ה אֹתָ֑ם וְהָי֤וּ לְךָ֙ לְמִקְרָ֣א הָֽעֵדָ֔ה וּלְמַסַּ֖ע אֶת־הַֽמַּחֲנֽוֹת׃ עֲבֵיד לָךְ, תַּרְתֵּין חֲצוֹצְרָן דִּכְסַף--נְגִיד, תַּעֲבֵיד יָתְהוֹן; וְיִהְוְיָן לָךְ לְעָרָעָא כְּנִשְׁתָּא, וּלְאַטָּלָא יָת מַשְׁרְיָתָא. 
י,ג וְתָֽקְע֖וּ בָּהֵ֑ן וְנֽוֹעֲד֤וּ אֵלֶ֨יךָ֙ כָּל־הָ֣עֵדָ֔ה אֶל־פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ וְיִתְקְעוּן, בְּהוֹן--וְיִזְדָּמְנוּן לְוָתָךְ כָּל כְּנִשְׁתָּא, לִתְרַע מַשְׁכַּן זִמְנָא. 
י,ד וְאִם־בְּאַחַ֖ת יִתְקָ֑עוּ וְנֽוֹעֲד֤וּ אֵלֶ֨יךָ֙ הַנְּשִׂיאִ֔ים רָאשֵׁ֖י אַלְפֵ֥י יִשְׂרָאֵֽל׃ וְאִם בַּחֲדָא, יִתְקְעוּן--וְיִזְדָּמְנוּן לְוָתָךְ רַבְרְבַיָּא, רֵישֵׁי אַלְפַיָּא דְּיִשְׂרָאֵל. 
י,ה וּתְקַעְתֶּ֖ם תְּרוּעָ֑ה וְנָֽסְעוּ֙ הַֽמַּחֲנ֔וֹת הַֽחֹנִ֖ים קֵֽדְמָה׃ וְתִתְקְעוּן, יַבָּבְתָא--וְיִטְּלוּן, מַשְׁרְיָתָא, דְּשָׁרַן, קִדּוּמָא. 
י,ו וּתְקַעְתֶּ֤ם תְּרוּעָה֙ שֵׁנִ֔ית וְנָֽסְעוּ֙ הַֽמַּחֲנ֔וֹת הַֽחֹנִ֖ים תֵּימָ֑נָה תְּרוּעָ֥ה יִתְקְע֖וּ לְמַסְעֵיהֶֽם׃ וְתִתְקְעוּן יַבָּבְתָא, תִּנְיָנוּת--וְיִטְּלוּן מַשְׁרְיָתָא, דְּשָׁרַן דָּרוֹמָא; יַבָּבְתָא יִתְקְעוּן, לְמַטְּלָנֵיהוֹן. 
י,ז וּבְהַקְהִ֖יל אֶת־הַקָּהָ֑ל תִּתְקְע֖וּ וְלֹ֥א תָרִֽיעוּ׃ וּבְמִכְנַשׁ, יָת קְהָלָא--תִּתְקְעוּן, וְלָא תְּיַבְּבוּן. 
י,ח וּבְנֵ֤י אַֽהֲרֹן֙ הַכֹּ֣הֲנִ֔ים יִתְקְע֖וּ בַּחֲצֹֽצְר֑וֹת וְהָי֥וּ לָכֶ֛ם לְחֻקַּ֥ת עוֹלָ֖ם לְדֹרֹֽתֵיכֶֽם׃ וּבְנֵי אַהֲרוֹן כָּהֲנַיָּא, יִתְקְעוּן בַּחֲצוֹצְרָתָא; וִיהוֹן לְכוֹן לִקְיָם עָלַם, לְדָרֵיכוֹן. 
י,ט וְכִֽי־תָבֹ֨אוּ מִלְחָמָ֜ה בְּאַרְצְכֶ֗ם עַל־הַצַּר֙ הַצֹּרֵ֣ר אֶתְכֶ֔ם וַהֲרֵֽעֹתֶ֖ם בַּחֲצֹֽצְרֹ֑ת וְנִזְכַּרְתֶּ֗ם לִפְנֵי֙ יְהוָ֣ה אֱלֹֽהֵיכֶ֔ם וְנֽוֹשַׁעְתֶּ֖ם מֵאֹֽיְבֵיכֶֽם׃ וַאֲרֵי תֵּיעֲלוּן לְאָגָחָא קְרָבָא בַּאֲרַעְכוֹן, עַל מָעִיקֵי דִּמְעִיקִין לְכוֹן--וּתְיַבְּבוּן, בַּחֲצוֹצְרָתָא; וְיֵיעוֹל דֻּכְרָנְכוֹן לְטָבָא, קֳדָם יְיָ אֱלָהֲכוֹן, וְתִתְפַּרְקוּן, מִסָּנְאֵיכוֹן. 
י,י וּבְי֨וֹם שִׂמְחַתְכֶ֥ם וּֽבְמוֹעֲדֵיכֶם֮ וּבְרָאשֵׁ֣י חָדְשֵׁיכֶם֒ וּתְקַעְתֶּ֣ם בַּחֲצֹֽצְרֹ֗ת עַ֚ל עֹלֹ֣תֵיכֶ֔ם וְעַ֖ל זִבְחֵ֣י שַׁלְמֵיכֶ֑ם וְהָי֨וּ לָכֶ֤ם לְזִכָּרוֹן֙ לִפְנֵ֣י אֱלֹֽהֵיכֶ֔ם אֲנִ֖י יְהוָ֥ה אֱלֹֽהֵיכֶֽם׃ {פ} וּבְיוֹם חֶדְוַתְכוֹן וּבְמוֹעֲדֵיכוֹן, וּבְרֵישֵׁי יַרְחֵיכוֹן--וְתִתְקְעוּן בַּחֲצוֹצְרָתָא עַל עֲלָוָתְכוֹן, וְעַל נִכְסַת קֻדְשֵׁיכוֹן; וִיהוֹן לְכוֹן לְדֻכְרָנָא קֳדָם אֱלָהֲכוֹן, אֲנָא יְיָ אֱלָהֲכוֹן. {פ}
י,יא וַיְהִ֞י בַּשָּׁנָ֧ה הַשֵּׁנִ֛ית בַּחֹ֥דֶשׁ הַשֵּׁנִ֖י בְּעֶשְׂרִ֣ים בַּחֹ֑דֶשׁ נַֽעֲלָה֙ הֶֽעָנָ֔ן מֵעַ֖ל מִשְׁכַּ֥ן הָֽעֵדֻֽת׃ וַהֲוָה בְּשַׁתָּא תִּנְיֵיתָא, בְּיַרְחָא תִּנְיָנָא--בְּעַסְרִין בְּיַרְחָא; אִסְתַּלַּק, עֲנָנָא, מֵעִלָּוֵי, מַשְׁכְּנָא דְּסָהֲדוּתָא. 
י,יב וַיִּסְע֧וּ בְנֵֽי־יִשְׂרָאֵ֛ל לְמַסְעֵיהֶ֖ם מִמִּדְבַּ֣ר סִינָ֑י וַיִּשְׁכֹּ֥ן הֶֽעָנָ֖ן בְּמִדְבַּ֥ר פָּארָֽן׃ וּנְטַלוּ בְּנֵי יִשְׂרָאֵל לְמַטְּלָנֵיהוֹן, מִמַּדְבְּרָא דְּסִינָי; וּשְׁרָא עֲנָנָא, בְּמַדְבְּרָא דְּפָארָן. 
י,יג וַיִּסְע֖וּ בָּרִֽאשֹׁנָ֑ה עַל־פִּ֥י יְהוָ֖ה בְּיַד־מֹשֶֽׁה׃ וּנְטַלוּ, בְּקַדְמֵיתָא, עַל מֵימְרָא דַּייָ, בִּידָא דְּמֹשֶׁה. 
י,יד וַיִּסַּ֞ע דֶּ֣גֶל מַֽחֲנֵ֧ה בְנֵֽי־יְהוּדָ֛ה בָּרִֽאשֹׁנָ֖ה לְצִבְאֹתָ֑ם וְעַ֨ל־צְבָא֔וֹ נַחְשׁ֖וֹן בֶּן־עַמִּֽינָדָֽב׃ וּנְטַל טֵיקַס מַשְׁרִית בְּנֵי יְהוּדָה, בְּקַדְמֵיתָא--לְחֵילֵיהוֹן; וְעַל חֵילֵיהּ--נַחְשׁוֹן, בַּר עַמִּינָדָב. 
י,טו וְעַ֨ל־צְבָ֔א מַטֵּ֖ה בְּנֵ֣י יִשָּׂשכָ֑ר נְתַנְאֵ֖ל בֶּן־צוּעָֽר׃ וְעַל חֵילָא--דְּשִׁבְטָא, דִּבְנֵי יִשָּׂשכָר:  נְתַנְאֵל, בַּר צוּעָר. 
י,טז וְעַ֨ל־צְבָ֔א מַטֵּ֖ה בְּנֵ֣י זְבוּלֻ֑ן אֱלִיאָ֖ב בֶּן־חֵלֹֽן׃ וְעַל חֵילָא--דְּשִׁבְטָא, דִּבְנֵי זְבוּלוּן:  אֱלִיאָב, בַּר חֵילוֹן. 
י,יז וְהוּרַ֖ד הַמִּשְׁכָּ֑ן וְנָֽסְע֤וּ בְנֵֽי־גֵרְשׁוֹן֙ וּבְנֵ֣י מְרָרִ֔י נֹֽשְׂאֵ֖י הַמִּשְׁכָּֽן׃ וּמִתְפָּרַק, מַשְׁכְּנָא; וְנָטְלִין בְּנֵי גֵּרְשׁוֹן וּבְנֵי מְרָרִי, נָטְלֵי מַשְׁכְּנָא. 
י,יח וְנָסַ֗ע דֶּ֛גֶל מַֽחֲנֵ֥ה רְאוּבֵ֖ן לְצִבְאֹתָ֑ם וְעַ֨ל־צְבָא֔וֹ אֱלִיצ֖וּר בֶּן־שְׁדֵיאֽוּר׃ וְנָטֵיל, טֵיקַס מַשְׁרִית רְאוּבֵן--לְחֵילֵיהוֹן; וְעַל חֵילֵיהּ--אֱלִיצוּר, בַּר שְׁדֵיאוּר. 
י,יט וְעַ֨ל־צְבָ֔א מַטֵּ֖ה בְּנֵ֣י שִׁמְע֑וֹן שְׁלֻֽמִיאֵ֖ל בֶּן־צוּרִֽישַׁדָּֽי׃ וְעַל חֵילָא--דְּשִׁבְטָא, דִּבְנֵי שִׁמְעוֹן:  שְׁלוּמִיאֵל, בַּר צוּרִישַׁדָּי. 
י,כ וְעַל־צְבָ֖א מַטֵּ֣ה בְנֵי־גָ֑ד אֶלְיָסָ֖ף בֶּן־דְּעוּאֵֽל׃ וְעַל חֵילָא, דְּשִׁבְטָא דִּבְנֵי גָּד:  אֶלְיָסָף, בַּר דְּעוּאֵל. 
י,כא וְנָֽסְעוּ֙ הַקְּהָתִ֔ים נֹֽשְׂאֵ֖י הַמִּקְדָּ֑שׁ וְהֵקִ֥ימוּ אֶת־הַמִּשְׁכָּ֖ן עַד־בֹּאָֽם׃ וְנָטְלִין, בְּנֵי קְהָת, נָטְלֵי, מַקְדְּשָׁא; וּמְקִימִין יָת מַשְׁכְּנָא, עַד מֵיתֵיהוֹן. 
י,כב וְנָסַ֗ע דֶּ֛גֶל מַֽחֲנֵ֥ה בְנֵֽי־אֶפְרַ֖יִם לְצִבְאֹתָ֑ם וְעַ֨ל־צְבָא֔וֹ אֱלִֽישָׁמָ֖ע בֶּן־עַמִּיהֽוּד׃ וְנָטֵיל, טֵיקַס מַשְׁרִית בְּנֵי אֶפְרַיִם--לְחֵילֵיהוֹן; וְעַל חֵילֵיהּ--אֱלִישָׁמָע, בַּר עַמִּיהוּד. 
י,כג וְעַ֨ל־צְבָ֔א מַטֵּ֖ה בְּנֵ֣י מְנַשֶּׁ֑ה גַּמְלִיאֵ֖ל בֶּן־פְּדָהצֽוּר׃ וְעַל חֵילָא--דְּשִׁבְטָא, דִּבְנֵי מְנַשֶּׁה:  גַּמְלִיאֵל, בַּר פְּדָהצוּר. 
י,כד וְעַ֨ל־צְבָ֔א מַטֵּ֖ה בְּנֵ֣י בִנְיָמִ֑ן אֲבִידָ֖ן בֶּן־גִּדְעוֹנִֽי׃ וְעַל חֵילָא--דְּשִׁבְטָא, דִּבְנֵי בִּנְיָמִין:  אֲבִידָן, בַּר גִּדְעוֹנִי. 
י,כה וְנָסַ֗ע דֶּ֚גֶל מַֽחֲנֵ֣ה בְנֵי־דָ֔ן מְאַסֵּ֥ף לְכָל־הַֽמַּחֲנֹ֖ת לְצִבְאֹתָ֑ם וְעַ֨ל־צְבָא֔וֹ אֲחִיעֶ֖זֶר בֶּן־עַמִּֽישַׁדָּֽי׃ וְנָטֵיל, טֵיקַס מַשְׁרִית בְּנֵי דָּן--מְכַנֵּישׁ לְכָל מַשְׁרְיָתָא, לְחֵילֵיהוֹן; וְעַל חֵילֵיהּ--אֲחִיעֶזֶר, בַּר עַמִּישַׁדָּי. 
י,כו וְעַ֨ל־צְבָ֔א מַטֵּ֖ה בְּנֵ֣י אָשֵׁ֑ר פַּגְעִיאֵ֖ל בֶּן־עָכְרָֽן׃ וְעַל חֵילָא--דְּשִׁבְטָא, דִּבְנֵי אָשֵׁר:  פַּגְעִיאֵל, בַּר עָכְרָן. 
י,כז וְעַ֨ל־צְבָ֔א מַטֵּ֖ה בְּנֵ֣י נַפְתָּלִ֑י אֲחִירַ֖ע בֶּן־עֵינָֽן׃ וְעַל חֵילָא--דְּשִׁבְטָא, דִּבְנֵי נַפְתָּלִי:  אֲחִירַע, בַּר עֵינָן. 
י,כח אֵ֛לֶּה מַסְעֵ֥י בְנֵֽי־יִשְׂרָאֵ֖ל לְצִבְאֹתָ֑ם וַיִּסָּֽעוּ׃ {ס} אִלֵּין מַטְּלָנֵי בְּנֵי יִשְׂרָאֵל, לְחֵילֵיהוֹן; וּנְטַלוּ. {ס}
י,כט וַיֹּ֣אמֶר מֹשֶׁ֗ה לְ֠חֹבָב בֶּן־רְעוּאֵ֣ל הַמִּדְיָנִי֮ חֹתֵ֣ן מֹשֶׁה֒ נֹֽסְעִ֣ים ׀ אֲנַ֗חְנוּ אֶל־הַמָּקוֹם֙ אֲשֶׁ֣ר אָמַ֣ר יְהוָ֔ה אֹת֖וֹ אֶתֵּ֣ן לָכֶ֑ם לְכָ֤ה אִתָּ֨נוּ֙ וְהֵטַ֣בְנוּ לָ֔ךְ כִּֽי־יְהוָ֥ה דִּבֶּר־ט֖וֹב עַל־יִשְׂרָאֵֽל׃ וַאֲמַר מֹשֶׁה, לְחוֹבָב בַּר רְעוּאֵל מִדְיַנָאָה חֲמוּהִי דְּמֹשֶׁה, נָטְלִין אֲנַחְנָא לְאַתְרָא דַּאֲמַר יְיָ, יָתֵיהּ אֶתֵּין לְכוֹן; אֵיתַא עִמַּנָא וְנוֹטֵיב לָךְ, אֲרֵי יְיָ מַלֵּיל לְאֵיתָאָה טָבְתָא עַל יִשְׂרָאֵל. 
י,ל וַיֹּ֥אמֶר אֵלָ֖יו לֹ֣א אֵלֵ֑ךְ כִּ֧י אִם־אֶל־אַרְצִ֛י וְאֶל־מֽוֹלַדְתִּ֖י אֵלֵֽךְ׃ וַאֲמַר לֵיהּ, לָא אֵיזֵיל:  אֱלָהֵין לְאַרְעִי וּלְיַלָּדוּתִי, אֵיזֵיל. 
י,לא וַיֹּ֕אמֶר אַל־נָ֖א תַּֽעֲזֹ֣ב אֹתָ֑נוּ כִּ֣י ׀ עַל־כֵּ֣ן יָדַ֗עְתָּ חֲנֹתֵ֨נוּ֙ בַּמִּדְבָּ֔ר וְהָיִ֥יתָ לָּ֖נוּ לְעֵינָֽיִם׃ וַאֲמַר, לָא כְּעַן תִּשְׁבּוֹק יָתַנָא:  אֲרֵי עַל כֵּין יְדַעְתָּא, כַּד הֲוֵינָא שָׁרַן בְּמַדְבְּרָא, וּגְבוּרָן דְּאִתְעֲבִידָא לַנָא, חֲזֵיתָא בְּעֵינָךְ. 
י,לב וְהָיָ֖ה כִּֽי־תֵלֵ֣ךְ עִמָּ֑נוּ וְהָיָ֣ה ׀ הַטּ֣וֹב הַה֗וּא אֲשֶׁ֨ר יֵיטִ֧יב יְהוָ֛ה עִמָּ֖נוּ וְהֵטַ֥בְנוּ לָֽךְ׃ וִיהֵי, אֲרֵי תֵּיזֵיל עִמַּנָא:  וִיהֵי טָבָא הַהוּא, דְּיוֹטֵיב יְיָ עִמַּנָא--וְנוֹטֵיב לָךְ. 
י,לג וַיִּסְעוּ֙ מֵהַ֣ר יְהוָ֔ה דֶּ֖רֶךְ שְׁלֹ֣שֶׁת יָמִ֑ים וַֽאֲר֨וֹן בְּרִית־יְהוָ֜ה נֹסֵ֣עַ לִפְנֵיהֶ֗ם דֶּ֚רֶךְ שְׁלֹ֣שֶׁת יָמִ֔ים לָת֥וּר לָהֶ֖ם מְנוּחָֽה׃ וּנְטַלוּ מִטּוּרָא דְּאִתְגְּלִי עֲלוֹהִי יְקָרָא דַּייָ, מַהְלַךְ תְּלָתָא יוֹמִין; וַאֲרוֹן קְיָמָא דַּייָ נָטֵיל קֳדָמֵיהוֹן, מַהְלַךְ תְּלָתָא יוֹמִין, לְאַתְקָנָא לְהוֹן, אֲתַר בֵּית מִשְׁרֵי. 
י,לד וַֽעֲנַ֧ן יְהוָ֛ה עֲלֵיהֶ֖ם יוֹמָ֑ם בְּנָסְעָ֖ם מִן־הַֽמַּחֲנֶֽה׃ {ס} ׆ {ס} וַעֲנַן יְקָרָא דַּייָ מַטַּל עִלָּוֵיהוֹן, בִּימָמָא, בְּמִטַּלְהוֹן, מִן מַשְׁרִיתָא. {ס}  ]  {ס} 
י,לה וַיְהִ֛י בִּנְסֹ֥עַ הָֽאָרֹ֖ן וַיֹּ֣אמֶר מֹשֶׁ֑ה קוּמָ֣ה ׀ יְהוָ֗ה וְיָפֻ֨צוּ֙ אֹֽיְבֶ֔יךָ וְיָנֻ֥סוּ מְשַׂנְאֶ֖יךָ מִפָּנֶֽיךָ׃ וַהֲוָה בְּמִטַּל אֲרוֹנָא, וַאֲמַר מֹשֶׁה:  אִתְגְּלִי יְיָ, וְיִתְבַּדְּרוּן סָנְאָךְ, וְיִעְרְקוּן בַּעֲלֵי דְּבָבָךְ, מִן קֳדָמָךְ. 
י,לו וּבְנֻחֹ֖ה יֹאמַ֑ר שׁוּבָ֣ה יְהוָ֔ה רִֽבְב֖וֹת אַלְפֵ֥י יִשְׂרָאֵֽל׃ {ס} ׆ {פ} וּבְמִשְׁרוֹהִי, אֲמַר:  תּוּב יְיָ, שְׁרִי בִּיקָרָךְ בְּגוֹ רִבְּוָת אַלְפַיָּא דְּיִשְׂרָאֵל. {ס}  ]  {פ}
יא,א וַיְהִ֤י הָעָם֙ כְּמִתְאֹ֣נְנִ֔ים רַ֖ע בְּאָזְנֵ֣י יְהוָ֑ה וַיִּשְׁמַ֤ע יְהוָה֙ וַיִּ֣חַר אַפּ֔וֹ וַתִּבְעַר־בָּם֙ אֵ֣שׁ יְהוָ֔ה וַתֹּ֖אכַל בִּקְצֵ֥ה הַֽמַּחֲנֶֽה׃ וַהֲוָה עַמָּא כַּד מִסְתַּקְּפִין, בִּישׁ קֳדָם יְיָ; וּשְׁמִיעַ קֳדָם יְיָ, וּתְקֵיף רֻגְזֵיהּ, וּדְלֵיקַת בְּהוֹן אִישָׁתָא מִן קֳדָם יְיָ, וְשֵׁיצִיאַת בִּסְיָפֵי מַשְׁרִיתָא. 
יא,ב וַיִּצְעַ֥ק הָעָ֖ם אֶל־מֹשֶׁ֑ה וַיִּתְפַּלֵּ֤ל מֹשֶׁה֙ אֶל־יְהוָ֔ה וַתִּשְׁקַ֖ע הָאֵֽשׁ׃ וּצְוַח עַמָּא, עַל מֹשֶׁה; וְצַלִּי מֹשֶׁה קֳדָם יְיָ, וְאִשְׁתְּקַעַת אִישָׁתָא. 
יא,ג וַיִּקְרָ֛א שֵֽׁם־הַמָּק֥וֹם הַה֖וּא תַּבְעֵרָ֑ה כִּֽי־בָעֲרָ֥ה בָ֖ם אֵ֥שׁ יְהוָֽה׃ וּקְרָא שְׁמֵיהּ דְּאַתְרָא הַהוּא, דְּלֵיקְתָא:  אֲרֵי דְּלֵיקַת בְּהוֹן, אִישָׁתָא מִן קֳדָם יְיָ. 
יא,ד וְהָֽאסַפְסֻף֙ אֲשֶׁ֣ר בְּקִרְבּ֔וֹ הִתְאַוּ֖וּ תַּֽאֲוָ֑ה וַיָּשֻׁ֣בוּ וַיִּבְכּ֗וּ גַּ֚ם בְּנֵ֣י יִשְׂרָאֵ֔ל וַיֹּ֣אמְר֔וּ מִ֥י יַֽאֲכִלֵ֖נוּ בָּשָֽׂר׃ וְרַבְרְבִין דְּבֵינֵיהוֹן, שַׁאִילוּ שְׁאֵילְתָא; וְתָבוּ וּבְכוֹ, אַף בְּנֵי יִשְׂרָאֵל, וַאֲמַרוּ, מַאן יוֹכֵילִנַּנָא בִּסְרָא. 
יא,ה זָכַ֨רְנוּ֙ אֶת־הַדָּגָ֔ה אֲשֶׁר־נֹאכַ֥ל בְּמִצְרַ֖יִם חִנָּ֑ם אֵ֣ת הַקִּשֻּׁאִ֗ים וְאֵת֙ הָֽאֲבַטִּחִ֔ים וְאֶת־הֶֽחָצִ֥יר וְאֶת־הַבְּצָלִ֖ים וְאֶת־הַשּׁוּמִֽים׃ דְּכִירִין אֲנַחְנָא, יָת נוּנַיָּא, דַּהֲוֵינָא אָכְלִין בְּמִצְרַיִם, מַגָּן; יָת בּוֹצִינַיָּא, וְיָת אֲבַטִּיחַיָּא, וְכָרָתֵי וּבֻצְלֵי, וְתוּמֵי. 
יא,ו וְעַתָּ֛ה נַפְשֵׁ֥נוּ יְבֵשָׁ֖ה אֵ֣ין כֹּ֑ל בִּלְתִּ֖י אֶל־הַמָּ֥ן עֵינֵֽינוּ׃ וּכְעַן נַפְשַׁנָא תְּאִיבָא, לֵית כָּל מִדָּעַם--אֱלָהֵין, לְמַנָּא עֵינַנָא. 
יא,ז וְהַמָּ֕ן כִּזְרַע־גַּ֖ד ה֑וּא וְעֵינ֖וֹ כְּעֵ֥ין הַבְּדֹֽלַח׃ וּמַנָּא, כְּבַר זְרַע גִּדָּא הוּא; וְחֶזְוֵיהּ, כְּחֵיזוּ בְּדֻלְחָא. 
יא,ח שָׁטוּ֩ הָעָ֨ם וְלָֽקְט֜וּ וְטָֽחֲנ֣וּ בָֽרֵחַ֗יִם א֤וֹ דָכוּ֙ בַּמְּדֹכָ֔ה וּבִשְּׁלוּ֙ בַּפָּר֔וּר וְעָשׂ֥וּ אֹת֖וֹ עֻג֑וֹת וְהָיָ֣ה טַעְמ֔וֹ כְּטַ֖עַם לְשַׁ֥ד הַשָּֽׁמֶן׃ שָׁיְטִין עַמָּא וְלָקְטִין דְּצָבֵי טָחֵין בְּרִחְיָא, וּדְצָבֵי דָּאֵיךְ בִּמְדוֹכְתָא, וּמְבַשְּׁלִין לֵיהּ בְּקִדְרָא, וְעָבְדִין יָתֵיהּ גְּרִיצָן; וְהָוֵי טַעְמֵיהּ, כִּטְעֵים דְּלִישׁ בְּמִשְׁחָא. 
יא,ט וּבְרֶ֧דֶת הַטַּ֛ל עַל־הַֽמַּחֲנֶ֖ה לָ֑יְלָה יֵרֵ֥ד הַמָּ֖ן עָלָֽיו׃ וְכַד נָחֵית טַלָּא עַל מַשְׁרִיתָא, בְּלֵילְיָא, נָחֵית מַנָּא, עֲלוֹהִי. 
יא,י וַיִּשְׁמַ֨ע מֹשֶׁ֜ה אֶת־הָעָ֗ם בֹּכֶה֙ לְמִשְׁפְּחֹתָ֔יו אִ֖ישׁ לְפֶ֣תַח אָֽהֳל֑וֹ וַיִּֽחַר־אַ֤ף יְהוָה֙ מְאֹ֔ד וּבְעֵינֵ֥י מֹשֶׁ֖ה רָֽע׃ וּשְׁמַע מֹשֶׁה יָת עַמָּא, בָּכַן לְזַרְעֲיָתְהוֹן--גְּבַר, בִּתְרַע מַשְׁכְּנֵיהּ; וּתְקֵיף רֻגְזָא דַּייָ לַחְדָּא, וּבְעֵינֵי מֹשֶׁה בִּישׁ. 
יא,יא וַיֹּ֨אמֶר מֹשֶׁ֜ה אֶל־יְהוָ֗ה לָמָ֤ה הֲרֵעֹ֨תָ֙ לְעַבְדֶּ֔ךָ וְלָ֛מָּה לֹֽא־מָצָ֥תִי חֵ֖ן בְּעֵינֶ֑יךָ לָשׂ֗וּם אֶת־מַשָּׂ֛א כָּל־הָעָ֥ם הַזֶּ֖ה עָלָֽי׃ וַאֲמַר מֹשֶׁה קֳדָם יְיָ, לְמָא אַבְאֵישְׁתָּא לְעַבְדָּךְ, וּלְמָא לָא אַשְׁכַּחִית רַחֲמִין, קֳדָמָךְ:  לְשַׁוָּאָה, יָת מַטוּל כָּל עַמָּא הָדֵין--עֲלָי. 
יא,יב הֶאָֽנֹכִ֣י הָרִ֗יתִי אֵ֚ת כָּל־הָעָ֣ם הַזֶּ֔ה אִם־אָֽנֹכִ֖י יְלִדְתִּ֑יהוּ כִּֽי־תֹאמַ֨ר אֵלַ֜י שָׂאֵ֣הוּ בְחֵיקֶ֗ךָ כַּֽאֲשֶׁ֨ר יִשָּׂ֤א הָֽאֹמֵן֙ אֶת־הַיֹּנֵ֔ק עַ֚ל הָֽאֲדָמָ֔ה אֲשֶׁ֥ר נִשְׁבַּ֖עְתָּ לַֽאֲבֹתָֽיו׃ הֲאָב אֲנָא, לְכָל עַמָּא הָדֵין--אִם בְּנַי, אִנּוּן:  דַּאֲמַרְתְּ לִי סוֹבַרְהִי בְּתֻקְפָךְ, כְּמָא דִּמְסוֹבַר תֻּרְבְּיָנָא יָת יָנְקָא, עַל אַרְעָא, דְּקַיֵּימְתָּא לַאֲבָהָתוֹהִי. 
יא,יג מֵאַ֤יִן לִי֙ בָּשָׂ֔ר לָתֵ֖ת לְכָל־הָעָ֣ם הַזֶּ֑ה כִּֽי־יִבְכּ֤וּ עָלַי֙ לֵאמֹ֔ר תְּנָה־לָּ֥נוּ בָשָׂ֖ר וְנֹאכֵֽלָה׃ מְנָן לִי בִּסְרָא, לְמִתַּן לְכָל עַמָּא הָדֵין:  אֲרֵי בָּכַן עֲלַי לְמֵימַר, הַב לַנָא בִּסְרָא וְנֵיכוֹל. 
יא,יד לֹֽא־אוּכַ֤ל אָֽנֹכִי֙ לְבַדִּ֔י לָשֵׂ֖את אֶת־כָּל־הָעָ֣ם הַזֶּ֑ה כִּ֥י כָבֵ֖ד מִמֶּֽנִּי׃ לֵית אֲנָא יָכֵיל בִּלְחוֹדַי, לְסוֹבָרָא יָת כָּל עַמָּא הָדֵין:  אֲרֵי יַקִּיר, מִנִּי. 
יא,טו וְאִם־כָּ֣כָה ׀ אַתְּ־עֹ֣שֶׂה לִּ֗י הָרְגֵ֤נִי נָא֙ הָרֹ֔ג אִם־מָצָ֥אתִי חֵ֖ן בְּעֵינֶ֑יךָ וְאַל־אֶרְאֶ֖ה בְּרָֽעָתִֽי׃ {פ} וְאִם כְּדֵין אַתְּ עָבֵיד לִי, קְטוֹלְנִי כְּעַן מִקְטָל--אִם אַשְׁכַּחִית רַחֲמִין, קֳדָמָךְ; וְלָא אֶחְזֵי, בְּבִשְׁתִּי. {פ}
יא,טז וַיֹּ֨אמֶר יְהוָ֜ה אֶל־מֹשֶׁ֗ה אֶסְפָה־לִּ֞י שִׁבְעִ֣ים אִישׁ֮ מִזִּקְנֵ֣י יִשְׂרָאֵל֒ אֲשֶׁ֣ר יָדַ֔עְתָּ כִּי־הֵ֛ם זִקְנֵ֥י הָעָ֖ם וְשֹֽׁטְרָ֑יו וְלָֽקַחְתָּ֤ אֹתָם֙ אֶל־אֹ֣הֶל מוֹעֵ֔ד וְהִֽתְיַצְּב֥וּ שָׁ֖ם עִמָּֽךְ׃ וַאֲמַר יְיָ לְמֹשֶׁה, כְּנוֹשׁ קֳדָמַי שִׁבְעִין גֻּבְרָא מִסָּבֵי יִשְׂרָאֵל, דִּידַעְתָּא, אֲרֵי אִנּוּן סָבֵי עַמָּא וְסָרְכוֹהִי; וְתִדְבַּר יָתְהוֹן לְמַשְׁכַּן זִמְנָא, וְיִתְעַתְּדוּן תַּמָּן עִמָּךְ. 
יא,יז וְיָֽרַדְתִּ֗י וְדִבַּרְתִּ֣י עִמְּךָ֮ שָׁם֒ וְאָֽצַלְתִּ֗י מִן־הָר֛וּחַ אֲשֶׁ֥ר עָלֶ֖יךָ וְשַׂמְתִּ֣י עֲלֵיהֶ֑ם וְנָֽשְׂא֤וּ אִתְּךָ֙ בְּמַשָּׂ֣א הָעָ֔ם וְלֹֽא־תִשָּׂ֥א אַתָּ֖ה לְבַדֶּֽךָ׃ וְאֶתְגְּלֵי, וַאֲמַלֵּיל עִמָּךְ תַּמָּן, וַאֲרַבֵּי מִן רוּחָא דַּעֲלָךְ, וַאֲשַׁוֵּי עֲלֵיהוֹן; וִיסוֹבְרוּן עִמָּךְ בְּמַטוּל עַמָּא, וְלָא תְּסוֹבַר אַתְּ בִּלְחוֹדָךְ. 
יא,יח וְאֶל־הָעָ֨ם תֹּאמַ֜ר הִתְקַדְּשׁ֣וּ לְמָחָר֮ וַֽאֲכַלְתֶּ֣ם בָּשָׂר֒ כִּ֡י בְּכִיתֶם֩ בְּאָזְנֵ֨י יְהוָ֜ה לֵאמֹ֗ר מִ֤י יַֽאֲכִלֵ֨נוּ֙ בָּשָׂ֔ר כִּי־ט֥וֹב לָ֖נוּ בְּמִצְרָ֑יִם וְנָתַ֨ן יְהוָ֥ה לָכֶ֛ם בָּשָׂ֖ר וַֽאֲכַלְתֶּֽם׃ וּלְעַמָּא תֵּימַר אִזְדָּמַנוּ לִמְחַר, וְתֵיכְלוּן בִּסְרָא--אֲרֵי בְּכֵיתוֹן קֳדָם יְיָ לְמֵימַר מַאן יוֹכֵילִנַּנָה בִּסְרָא, אֲרֵי טָב לַנָא בְּמִצְרָיִם; וְיִתֵּין יְיָ לְכוֹן בִּסְרָא, וְתֵיכְלוּן. 
יא,יט לֹ֣א י֥וֹם אֶחָ֛ד תֹּֽאכְל֖וּן וְלֹ֣א יוֹמָ֑יִם וְלֹ֣א ׀ חֲמִשָּׁ֣ה יָמִ֗ים וְלֹא֙ עֲשָׂרָ֣ה יָמִ֔ים וְלֹ֖א עֶשְׂרִ֥ים יֽוֹם׃ לָא יוֹמָא חַד תֵּיכְלוּן, וְלָא תְּרֵין יוֹמִין; וְלָא חַמְשָׁא יוֹמִין, וְלָא עַסְרָא יוֹמִין, וְלָא, עַסְרִין יוֹם. 
יא,כ עַ֣ד ׀ חֹ֣דֶשׁ יָמִ֗ים עַ֤ד אֲשֶׁר־יֵצֵא֙ מֵֽאַפְּכֶ֔ם וְהָיָ֥ה לָכֶ֖ם לְזָרָ֑א יַ֗עַן כִּֽי־מְאַסְתֶּ֤ם אֶת־יְהוָה֙ אֲשֶׁ֣ר בְּקִרְבְּכֶ֔ם וַתִּבְכּ֤וּ לְפָנָיו֙ לֵאמֹ֔ר לָ֥מָּה זֶּ֖ה יָצָ֥אנוּ מִמִּצְרָֽיִם׃ עַד יְרַח יוֹמִין, עַד דִּתְקוּצוּן בֵּיהּ, וִיהֵי לְכוֹן, לְתַקְלָא:  חֲלָף, דְּקַצְתּוּן בְּמֵימְרָא דַּייָ דִּשְׁכִינְתֵיהּ שָׁרְיָא בֵּינֵיכוֹן, וּבְכֵיתוֹן קֳדָמוֹהִי לְמֵימַר, לְמָא דְּנָן נְפַקְנָא מִמִּצְרָיִם. 
יא,כא וַיֹּאמֶר֮ מֹשֶׁה֒ שֵׁשׁ־מֵא֥וֹת אֶ֨לֶף֙ רַגְלִ֔י הָעָ֕ם אֲשֶׁ֥ר אָֽנֹכִ֖י בְּקִרְבּ֑וֹ וְאַתָּ֣ה אָמַ֗רְתָּ בָּשָׂר֙ אֶתֵּ֣ן לָהֶ֔ם וְאָֽכְל֖וּ חֹ֥דֶשׁ יָמִֽים׃ וַאֲמַר, מֹשֶׁה, שֵׁית מְאָה אַלְפִין גֻּבְרָא רִגְלָאָה, עַמָּא דַּאֲנָא בֵּינֵיהוֹן; וְאַתְּ אֲמַרְתְּ, בִּסְרָא אֶתֵּין לְהוֹן, וְיֵיכְלוּן, יְרַח יוֹמִין. 
יא,כב הֲצֹ֧אן וּבָקָ֛ר יִשָּׁחֵ֥ט לָהֶ֖ם וּמָצָ֣א לָהֶ֑ם אִ֣ם אֶֽת־כָּל־דְּגֵ֥י הַיָּ֛ם יֵֽאָסֵ֥ף לָהֶ֖ם וּמָצָ֥א לָהֶֽם׃ {פ} הֲעָן וְתוֹרִין יִתְנַכְסוּן לְהוֹן, הַיְסַפְּקוּן לְהוֹן; אִם יָת כָּל נוּנֵי יַמָּא יִתְכַּנְשׁוּן לְהוֹן, הַיְסַפְּקוּן לְהוֹן. {פ}
יא,כג וַיֹּ֤אמֶר יְהוָה֙ אֶל־מֹשֶׁ֔ה הֲיַ֥ד יְהוָ֖ה תִּקְצָ֑ר עַתָּ֥ה תִרְאֶ֛ה הֲיִקְרְךָ֥ דְבָרִ֖י אִם־לֹֽא׃ וַאֲמַר יְיָ לְמֹשֶׁה, הֲמֵימְרָא דַּייָ מִתְעַכַּב; כְּעַן תִּחְזֵי הַיְעָרְעִנָּךְ פִּתְגָמִי, אִם לָא. 
יא,כד וַיֵּצֵ֣א מֹשֶׁ֔ה וַיְדַבֵּר֙ אֶל־הָעָ֔ם אֵ֖ת דִּבְרֵ֣י יְהוָ֑ה וַיֶּֽאֱסֹ֞ף שִׁבְעִ֥ים אִישׁ֙ מִזִּקְנֵ֣י הָעָ֔ם וַֽיַּעֲמֵ֥ד אֹתָ֖ם סְבִיבֹ֥ת הָאֹֽהֶל׃ וּנְפַק מֹשֶׁה--וּמַלֵּיל עִם עַמָּא, יָת פִּתְגָמַיָּא דַּייָ; וּכְנַשׁ שִׁבְעִין גֻּבְרָא, מִסָּבֵי עַמָּא, וַאֲקֵים יָתְהוֹן, סְחוֹר סְחוֹר לְמַשְׁכְּנָא. 
יא,כה וַיֵּ֨רֶד יְהוָ֥ה ׀ בֶּֽעָנָן֮ וַיְדַבֵּ֣ר אֵלָיו֒ וַיָּ֗אצֶל מִן־הָר֨וּחַ֙ אֲשֶׁ֣ר עָלָ֔יו וַיִּתֵּ֕ן עַל־שִׁבְעִ֥ים אִ֖ישׁ הַזְּקֵנִ֑ים וַיְהִ֗י כְּנ֤וֹחַ עֲלֵיהֶם֙ הָר֔וּחַ וַיִּֽתְנַבְּא֖וּ וְלֹ֥א יָסָֽפוּ׃ וְאִתְגְּלִי יְיָ בַּעֲנָנָא, וּמַלֵּיל עִמֵּיהּ, וְרַבִּי מִן רוּחָא דַּעֲלוֹהִי, וִיהַב עַל שִׁבְעִין גֻּבְרָא סָבַיָּא; וַהֲוָה, כַּד שְׁרָת עֲלֵיהוֹן רוּחַ נְבוּאָה, וּמִתְנַבַּן, וְלָא פָּסְקִין. 
יא,כו וַיִּשָּֽׁאֲר֣וּ שְׁנֵֽי־אֲנָשִׁ֣ים ׀ בַּֽמַּחֲנֶ֡ה שֵׁ֣ם הָֽאֶחָ֣ד ׀ אֶלְדָּ֡ד וְשֵׁם֩ הַשֵּׁנִ֨י מֵידָ֜ד וַתָּ֧נַח עֲלֵהֶ֣ם הָר֗וּחַ וְהֵ֨מָּה֙ בַּכְּתֻבִ֔ים וְלֹ֥א יָֽצְא֖וּ הָאֹ֑הֱלָה וַיִּֽתְנַבְּא֖וּ בַּֽמַּחֲנֶֽה׃ וְאִשְׁתְּאַרוּ תְּרֵין גֻּבְרִין בְּמַשְׁרִיתָא שׁוֹם חַד אֶלְדָּד וְשׁוֹם תִּנְיָנָא מֵידָד וּשְׁרָת עֲלֵיהוֹן רוּחַ נְבוּאָה, וְאִנּוּן בִּכְתִיבַיָּא, וְלָא נְפַקוּ, לְמַשְׁכְּנָא; וְאִתְנַבִּיאוּ, בְּמַשְׁרִיתָא. 
יא,כז וַיָּ֣רָץ הַנַּ֔עַר וַיַּגֵּ֥ד לְמֹשֶׁ֖ה וַיֹּאמַ֑ר אֶלְדָּ֣ד וּמֵידָ֔ד מִֽתְנַבְּאִ֖ים בַּֽמַּחֲנֶֽה׃ וּרְהַט עוּלֵימָא, וְחַוִּי לְמֹשֶׁה וַאֲמַר:  אֶלְדָּד וּמֵידָד, מִתְנַבַּן בְּמַשְׁרִיתָא. 
יא,כח וַיַּ֜עַן יְהוֹשֻׁ֣עַ בִּן־נ֗וּן מְשָׁרֵ֥ת מֹשֶׁ֛ה מִבְּחֻרָ֖יו וַיֹּאמַ֑ר אֲדֹנִ֥י מֹשֶׁ֖ה כְּלָאֵֽם׃ וַאֲתֵיב יְהוֹשׁוּעַ בַּר נוּן, מְשֻׁמְשָׁנֵיהּ דְּמֹשֶׁה מֵעוּלֵימוֹהִי--וַאֲמַר:  רִבּוֹנִי מֹשֶׁה, אַסְרִנּוּן. 
יא,כט וַיֹּ֤אמֶר לוֹ֙ מֹשֶׁ֔ה הַֽמְקַנֵּ֥א אַתָּ֖ה לִ֑י וּמִ֨י יִתֵּ֜ן כָּל־עַ֤ם יְהוָה֙ נְבִיאִ֔ים כִּֽי־יִתֵּ֧ן יְהוָ֛ה אֶת־רוּח֖וֹ עֲלֵיהֶֽם׃ וַאֲמַר לֵיהּ מֹשֶׁה, הֲקִנְאֲתִי אַתְּ מְקַנֵּי; רְעֵינָא פּוֹן דִּיהוֹן כָּל עַמֵּיהּ דַּייָ, נְבִיִּין--אֲרֵי יִתֵּין יְיָ יָת רוּחַ נְבוּאֲתֵיהּ, עֲלֵיהוֹן. 
יא,ל וַיֵּֽאָסֵ֥ף מֹשֶׁ֖ה אֶל־הַֽמַּחֲנֶ֑ה ה֖וּא וְזִקְנֵ֥י יִשְׂרָאֵֽל׃ וְאִתְכְּנֵישׁ מֹשֶׁה, לְמַשְׁרִיתָא--הוּא, וְסָבֵי יִשְׂרָאֵל. 
יא,לא וְר֜וּחַ נָסַ֣ע ׀ מֵאֵ֣ת יְהוָ֗ה וַיָּ֣גָז שַׂלְוִים֮ מִן־הַיָּם֒ וַיִּטֹּ֨שׁ עַל־הַֽמַּחֲנֶ֜ה כְּדֶ֧רֶךְ י֣וֹם כֹּ֗ה וּכְדֶ֤רֶךְ יוֹם֙ כֹּ֔ה סְבִיב֖וֹת הַֽמַּחֲנֶ֑ה וּכְאַמָּתַ֖יִם עַל־פְּנֵ֥י הָאָֽרֶץ׃ וְרוּחָא נְטַל מִן קֳדָם יְיָ, וְאַפְרַח שְׂלָיו מִן יַמָּא, וּרְמָא עַל מַשְׁרִיתָא כְּמַהְלַךְ יוֹמָא לְכָא וּכְמַהְלַךְ יוֹמָא לְכָא, סְחוֹר סְחוֹר לְמַשְׁרִיתָא--וּכְרוּם תַּרְתֵּין אַמִּין, עַל אַפֵּי אַרְעָא. 
יא,לב וַיָּ֣קָם הָעָ֡ם כָּל־הַיּוֹם֩ הַה֨וּא וְכָל־הַלַּ֜יְלָה וְכֹ֣ל ׀ י֣וֹם הַֽמָּחֳרָ֗ת וַיַּֽאַסְפוּ֙ אֶת־הַשְּׂלָ֔ו הַמַּמְעִ֕יט אָסַ֖ף עֲשָׂרָ֣ה חֳמָרִ֑ים וַיִּשְׁטְח֤וּ לָהֶם֙ שָׁט֔וֹחַ סְבִיב֖וֹת הַֽמַּחֲנֶֽה׃ וְקָם עַמָּא כָּל יוֹמָא הַהוּא וְכָל לֵילְיָא וְכֹל יוֹמָא דְּבָתְרוֹהִי, וּכְנַשׁוּ יָת שְׂלָיו--דְּאַזְעַר, כְּנַשׁ עַסְרָא דְּגוֹרִין; וּשְׁטַחוּ לְהוֹן מַשְׁטְחִין, סְחוֹר סְחוֹר לְמַשְׁרִיתָא. 
יא,לג הַבָּשָׂ֗ר עוֹדֶ֨נּוּ֙ בֵּ֣ין שִׁנֵּיהֶ֔ם טֶ֖רֶם יִכָּרֵ֑ת וְאַ֤ף יְהוָה֙ חָרָ֣ה בָעָ֔ם וַיַּ֤ךְ יְהוָה֙ בָּעָ֔ם מַכָּ֖ה רַבָּ֥ה מְאֹֽד׃ בִּסְרָא, עַד כְּעַן בֵּין שִׁנֵּיהוֹן--עַד, לָא פְּסַק; וְרֻגְזָא דַּייָ, תְּקֵיף בְּעַמָּא, וּקְטַל יְיָ בְּעַמָּא, קְטוֹל סַגִּי לַחְדָּא. 
יא,לד וַיִּקְרָ֛א אֶת־שֵֽׁם־הַמָּק֥וֹם הַה֖וּא קִבְר֣וֹת הַֽתַּאֲוָ֑ה כִּי־שָׁם֙ קָֽבְר֔וּ אֶת־הָעָ֖ם הַמִּתְאַוִּֽים׃ וּקְרָא יָת שְׁמֵיהּ דְּאַתְרָא הַהוּא, קִבְרֵי דִּמְשַׁאֲלֵי:  אֲרֵי תַּמָּן, קְבַרוּ, יָת עַמָּא, דְּשַׁאִילוּ. 
יא,לה מִקִּבְר֧וֹת הַֽתַּאֲוָ֛ה נָֽסְע֥וּ הָעָ֖ם חֲצֵר֑וֹת וַיִּֽהְי֖וּ בַּֽחֲצֵרֽוֹת׃ {פ} מִקִּבְרֵי דִּמְשַׁאֲלֵי נְטַלוּ עַמָּא, לַחֲצֵרוֹת; וַהֲווֹ, בַּחֲצֵרוֹת. {פ}
יב,א וַתְּדַבֵּ֨ר מִרְיָ֤ם וְאַֽהֲרֹן֙ בְּמֹשֶׁ֔ה עַל־אֹד֛וֹת הָֽאִשָּׁ֥ה הַכֻּשִׁ֖ית אֲשֶׁ֣ר לָקָ֑ח כִּֽי־אִשָּׁ֥ה כֻשִׁ֖ית לָקָֽח׃ וּמַלֵּילַת מִרְיָם וְאַהֲרוֹן בְּמֹשֶׁה, עַל עֵיסַק אִתְּתָא שַׁפִּירְתָא דִּנְסֵיב:  אֲרֵי אִתְּתָא שַׁפִּירְתָא דִּנְסֵיב, רַחֵיק. 
יב,ב וַיֹּֽאמְר֗וּ הֲרַ֤ק אַךְ־בְּמֹשֶׁה֙ דִּבֶּ֣ר יְהוָ֔ה הֲלֹ֖א גַּם־בָּ֣נוּ דִבֵּ֑ר וַיִּשְׁמַ֖ע יְהוָֽה׃ וַאֲמַרוּ, הַלְחוֹד בְּרַם עִם מֹשֶׁה מַלֵּיל יְיָ--הֲלָא, אַף עִמַּנָא מַלֵּיל; וּשְׁמִיעַ, קֳדָם יְיָ. 
יב,ג וְהָאִ֥ישׁ מֹשֶׁ֖ה עָנָ֣ו מְאֹ֑ד מִכֹּל֙ הָֽאָדָ֔ם אֲשֶׁ֖ר עַל־פְּנֵ֥י הָֽאֲדָמָֽה׃ {ס} וְגֻבְרָא מֹשֶׁה, עִנְוְתָן לַחְדָּא:  מִכֹּל, אֲנָשָׁא--דְּעַל אַפֵּי אַרְעָא. {ס}
יב,ד וַיֹּ֨אמֶר יְהוָ֜ה פִּתְאֹ֗ם אֶל־מֹשֶׁ֤ה וְאֶֽל־אַהֲרֹן֙ וְאֶל־מִרְיָ֔ם צְא֥וּ שְׁלָשְׁתְּכֶ֖ם אֶל־אֹ֣הֶל מוֹעֵ֑ד וַיֵּֽצְא֖וּ שְׁלָשְׁתָּֽם׃ וַאֲמַר יְיָ בִּתְכֵּיף, לְמֹשֶׁה וּלְאַהֲרוֹן וּלְמִרְיָם, פּוּקוּ תְּלָתֵיכוֹן, לְמַשְׁכַּן זִמְנָא; וּנְפַקוּ, תְּלָתֵיהוֹן. 
יב,ה וַיֵּ֤רֶד יְהוָה֙ בְּעַמּ֣וּד עָנָ֔ן וַֽיַּעֲמֹ֖ד פֶּ֣תַח הָאֹ֑הֶל וַיִּקְרָא֙ אַֽהֲרֹ֣ן וּמִרְיָ֔ם וַיֵּֽצְא֖וּ שְׁנֵיהֶֽם׃ וְאִתְגְּלִי יְיָ בְּעַמּוּדָא דַּעֲנָנָא, וְקָם בִּתְרַע מַשְׁכְּנָא; וּקְרָא אַהֲרוֹן וּמִרְיָם, וּנְפַקוּ תַּרְוֵיהוֹן. 
יב,ו וַיֹּ֖אמֶר שִׁמְעוּ־נָ֣א דְבָרָ֑י אִם־יִֽהְיֶה֙ נְבִ֣יאֲכֶ֔ם יְהוָ֗ה בַּמַּרְאָה֙ אֵלָ֣יו אֶתְוַדָּ֔ע בַּֽחֲל֖וֹם אֲדַבֶּר־בּֽוֹ׃ וַאֲמַר, שְׁמַעוּ כְּעַן פִּתְגָמָי; אִם יְהוֹן, לְכוֹן נְבִיִּין--אֲנָא יְיָ בְּחֶזְוִין אֲנָא מִתְגְּלֵי לְהוֹן, בְּחֶלְמִין אֲנָא מְמַלֵּיל עִמְּהוֹן. 
יב,ז לֹא־כֵ֖ן עַבְדִּ֣י מֹשֶׁ֑ה בְּכָל־בֵּיתִ֖י נֶֽאֱמָ֥ן הֽוּא׃ לָא כֵּין, עַבְדִּי מֹשֶׁה:  בְּכָל עַמִּי, מְהֵימָן הוּא. 
יב,ח פֶּ֣ה אֶל־פֶּ֞ה אֲדַבֶּר־בּ֗וֹ וּמַרְאֶה֙ וְלֹ֣א בְחִידֹ֔ת וּתְמֻנַ֥ת יְהוָ֖ה יַבִּ֑יט וּמַדּ֨וּעַ֙ לֹ֣א יְרֵאתֶ֔ם לְדַבֵּ֖ר בְּעַבְדִּ֥י בְמֹשֶֽׁה׃ מַמְלַל עִם מַמְלַל מַלֵּילְנָא עִמֵּיהּ, וּבְחֵיזוּ וְלָא בְּחֶדְוָן, וּדְמוּת יְקָרָא דַּייָ, מִסְתַּכַּל; וּמָדֵין לָא דְּחֵילְתּוּן, לְמַלָּלָא בְּעַבְדִּי בְּמֹשֶׁה. 
יב,ט וַיִּֽחַר־אַ֧ף יְהוָ֛ה בָּ֖ם וַיֵּלַֽךְ׃ וּתְקֵיף רֻגְזָא דַּייָ בְּהוֹן, וְאִסְתַּלַּק. 
יב,י וְהֶֽעָנָ֗ן סָ֚ר מֵעַ֣ל הָאֹ֔הֶל וְהִנֵּ֥ה מִרְיָ֖ם מְצֹרַ֣עַת כַּשָּׁ֑לֶג וַיִּ֧פֶן אַֽהֲרֹ֛ן אֶל־מִרְיָ֖ם וְהִנֵּ֥ה מְצֹרָֽעַת׃ וַעֲנָנָא, אִסְתַּלַּק מֵעִלָּוֵי מַשְׁכְּנָא, וְהָא מִרְיָם, חָוְרָא כְּתַלְגָּא; וְאִתְפְּנִי אַהֲרוֹן לְוָת מִרְיָם, וְהָא סְגִירָא. 
יב,יא וַיֹּ֥אמֶר אַֽהֲרֹ֖ן אֶל־מֹשֶׁ֑ה בִּ֣י אֲדֹנִ֔י אַל־נָ֨א תָשֵׁ֤ת עָלֵ֨ינוּ֙ חַטָּ֔את אֲשֶׁ֥ר נוֹאַ֖לְנוּ וַֽאֲשֶׁ֥ר חָטָֽאנוּ׃ וַאֲמַר אַהֲרוֹן, לְמֹשֶׁה:  בְּבָעוּ רִבּוֹנִי--לָא כְּעַן תְּשַׁוֵּי עֲלַנָא חוֹבָא, דְּאִטְּפַשְׁנָא וְדִסְרַחְנָא. 
יב,יב אַל־נָ֥א תְהִ֖י כַּמֵּ֑ת אֲשֶׁ֤ר בְּצֵאתוֹ֙ מֵרֶ֣חֶם אִמּ֔וֹ וַיֵּֽאָכֵ֖ל חֲצִ֥י בְשָׂרֽוֹ׃ לָא כְּעַן תִּתְרַחַק דָּא מִבֵּינַנָא, אֲרֵי אֲחָתַנָא הִיא--צַלִּי כְּעַן עַל בִּסְרָא מִיתָא הָדֵין דְּבַהּ, וְיִתַּסֵּי. 
יב,יג וַיִּצְעַ֣ק מֹשֶׁ֔ה אֶל־יְהוָ֖ה לֵאמֹ֑ר אֵ֕ל נָ֛א רְפָ֥א נָ֖א לָֽהּ׃ {פ} וְצַלִּי מֹשֶׁה, קֳדָם יְיָ לְמֵימַר:  אֱלָהָא, בְּבָעוּ אַסִּי כְּעַן יָתַהּ. {פ}
יב,יד וַיֹּ֨אמֶר יְהוָ֜ה אֶל־מֹשֶׁ֗ה וְאָבִ֨יהָ֙ יָרֹ֤ק יָרַק֙ בְּפָנֶ֔יהָ הֲלֹ֥א תִכָּלֵ֖ם שִׁבְעַ֣ת יָמִ֑ים תִּסָּגֵ֞ר שִׁבְעַ֤ת יָמִים֙ מִח֣וּץ לַֽמַּחֲנֶ֔ה וְאַחַ֖ר תֵּֽאָסֵֽף׃ וַאֲמַר יְיָ לְמֹשֶׁה, וְאִלּוּ אֲבוּהָא מִנְזָף נְזַף בַּהּ--הֲלָא תִּתְכְּלֵים, שִׁבְעָא יוֹמִין; תִּסְתְּגַר שִׁבְעָא יוֹמִין, מִבַּרָא לְמַשְׁרִיתָא, וּבָתַר כֵּין, תִּתְכְּנֵישׁ. 
יב,טו וַתִּסָּגֵ֥ר מִרְיָ֛ם מִח֥וּץ לַֽמַּחֲנֶ֖ה שִׁבְעַ֣ת יָמִ֑ים וְהָעָם֙ לֹ֣א נָסַ֔ע עַד־הֵֽאָסֵ֖ף מִרְיָֽם׃ וְאִסְתְּגַרַת מִרְיָם מִבַּרָא לְמַשְׁרִיתָא, שִׁבְעָא יוֹמִין; וְעַמָּא לָא נְטַל, עַד דְּאִתְכְּנֵישַׁת מִרְיָם. 
יב,טז וְאַחַ֛ר נָֽסְע֥וּ הָעָ֖ם מֵֽחֲצֵר֑וֹת וַֽיַּחֲנ֖וּ בְּמִדְבַּ֥ר פָּארָֽן׃ {פ} וּבָתַר כֵּין נְטַלוּ עַמָּא, מֵחֲצֵרוֹת; וּשְׁרוֹ, בְּמַדְבְּרָא דְּפָארָן. {פ}
