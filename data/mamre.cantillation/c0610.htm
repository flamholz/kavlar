לְמֶ֥לֶךְ יְרִיחֽוֹ׃ {ס} לא וַיַּֽעֲבֹ֣ר יְ֠הוֹשֻׁעַ וְכָל־יִשְׂרָאֵ֥ל עִמּ֛וֹ מִלִּבְנָ֖ה לָכִ֑ישָׁה וַיִּ֣חַן עָלֶ֔יהָ וַיִּלָּ֖חֶם בָּֽהּ׃ לב וַיִּתֵּן֩ יְהוָ֨ה אֶת־לָכִ֜ישׁ בְּיַ֣ד יִשְׂרָאֵ֗ל וַֽיִּלְכְּדָהּ֙ בַּיּ֣וֹם הַשֵּׁנִ֔י וַיַּכֶּ֣הָ לְפִי־חֶ֔רֶב וְאֶת־כָּל־הַנֶּ֖פֶשׁ אֲשֶׁר־בָּ֑הּ כְּכֹ֥ל אֲשֶׁר־עָשָׂ֖ה לְלִבְנָֽה׃ {פ}
לג אָ֣ז עָלָ֗ה הֹרָם֙ מֶ֣לֶךְ גֶּ֔זֶר לַעְזֹ֖ר אֶת־לָכִ֑ישׁ וַיַּכֵּ֤הוּ יְהוֹשֻׁ֨עַ֙ וְאֶת־עַמּ֔וֹ עַד־בִּלְתִּ֥י הִשְׁאִֽיר־ל֖וֹ שָׂרִֽיד׃ לד וַיַּֽעֲבֹ֣ר יְ֠הוֹשֻׁעַ וְכָל־יִשְׂרָאֵ֥ל עִמּ֛וֹ מִלָּכִ֖ישׁ עֶגְלֹ֑נָה וַיַּֽחֲנ֣וּ עָלֶ֔יהָ וַיִּֽלָּחֲמ֖וּ עָלֶֽיהָ׃ לה וַֽיִּלְכְּד֜וּהָ בַּיּ֤וֹם הַהוּא֙ וַיַּכּ֣וּהָ לְפִי־חֶ֔רֶב וְאֵת֙ כָּל־הַנֶּ֣פֶשׁ אֲשֶׁר־בָּ֔הּ בַּיּ֥וֹם הַה֖וּא הֶֽחֱרִ֑ים כְּכֹ֥ל אֲשֶׁר־עָשָׂ֖ה לְלָכִֽישׁ׃ {פ}
לו וַיַּ֣עַל יְ֠הוֹשֻׁעַ וְכָֽל־יִשְׂרָאֵ֥ל עִמּ֛וֹ מֵֽעֶגְל֖וֹנָה חֶבְר֑וֹנָה וַיִּֽלָּחֲמ֖וּ עָלֶֽיהָ׃ לז וַיִּלְכְּד֣וּהָ וַיַּכּֽוּהָ־לְפִי־חֶ֠רֶב וְאֶת־מַלְכָּ֨הּ וְאֶת־כָּל־עָרֶ֜יהָ וְאֶת־כָּל־הַנֶּ֤פֶשׁ אֲשֶׁר־בָּהּ֙ לֹֽא־הִשְׁאִ֣יר שָׂרִ֔יד כְּכֹ֥ל אֲשֶׁר־עָשָׂ֖ה לְעֶגְל֑וֹן וַיַּֽחֲרֵ֣ם אוֹתָ֔הּ וְאֶת־כָּל־הַנֶּ֖פֶשׁ אֲשֶׁר־בָּֽהּ׃ {ס} לח וַיָּ֧שָׁב יְהוֹשֻׁ֛עַ וְכָל־יִשְׂרָאֵ֥ל עִמּ֖וֹ דְּבִ֑רָה וַיִּלָּ֖חֶם עָלֶֽיהָ׃ לט וַֽיִּלְכְּדָ֞הּ וְאֶת־מַלְכָּ֤הּ וְאֶת־כָּל־עָרֶ֨יהָ֙ וַיַּכּ֣וּם לְפִי־חֶ֔רֶב וַֽיַּחֲרִ֨ימוּ֙ אֶת־כָּל־נֶ֣פֶשׁ אֲשֶׁר־בָּ֔הּ לֹ֥א הִשְׁאִ֖יר שָׂרִ֑יד כַּֽאֲשֶׁ֨ר עָשָׂ֜ה לְחֶבְר֗וֹן כֵּן־עָשָׂ֤ה לִדְבִ֨רָה֙ וּלְמַלְכָּ֔הּ וְכַֽאֲשֶׁ֥ר עָשָׂ֛ה לְלִבְנָ֖ה וּלְמַלְכָּֽהּ׃ מ וַיַּכֶּ֣ה יְהוֹשֻׁ֣עַ אֶת־כָּל־הָאָ֡רֶץ הָהָר֩ וְהַנֶּ֨גֶב וְהַשְּׁפֵלָ֜ה וְהָֽאֲשֵׁד֗וֹת וְאֵת֙ כָּל־מַלְכֵיהֶ֔ם לֹ֥א הִשְׁאִ֖יר שָׂרִ֑יד וְאֵ֤ת כָּל־הַנְּשָׁמָה֙ הֶֽחֱרִ֔ים כַּֽאֲשֶׁ֣ר צִוָּ֔ה יְהוָ֖ה אֱלֹהֵ֥י יִשְׂרָאֵֽל׃ מא וַיַּכֵּ֧ם יְהוֹשֻׁ֛עַ מִקָּדֵ֥שׁ בַּרְנֵ֖עַ וְעַד־עַזָּ֑ה וְאֵ֛ת כָּל־אֶ֥רֶץ גֹּ֖שֶׁן וְעַד־גִּבְעֽוֹן׃ מב וְאֵ֨ת כָּל־הַמְּלָכִ֤ים הָאֵ֨לֶּה֙ וְאֶת־אַרְצָ֔ם לָכַ֥ד יְהוֹשֻׁ֖עַ פַּ֣עַם אֶחָ֑ת כִּ֗י יְהוָה֙ אֱלֹהֵ֣י יִשְׂרָאֵ֔ל נִלְחָ֖ם לְיִשְׂרָאֵֽל׃ מג וַיָּ֤שָׁב יְהוֹשֻׁ֨עַ֙ וְכָל־יִשְׂרָאֵ֣ל עִמּ֔וֹ אֶל־הַֽמַּחֲנֶ֖ה הַגִּלְגָּֽלָה׃ {פ}
