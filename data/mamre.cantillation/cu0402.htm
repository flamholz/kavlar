נשוא

ד,כא וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ד,כב נָשֹׂ֗א אֶת־רֹ֛אשׁ בְּנֵ֥י גֵֽרְשׁ֖וֹן גַּם־הֵ֑ם לְבֵ֥ית אֲבֹתָ֖ם לְמִשְׁפְּחֹתָֽם׃ קַבֵּיל, יָת חֻשְׁבַּן בְּנֵי גֵּרְשׁוֹן--אַף אִנּוּן:  לְבֵית אֲבָהָתְהוֹן, לְזַרְעֲיָתְהוֹן. 
ד,כג מִבֶּן֩ שְׁלֹשִׁ֨ים שָׁנָ֜ה וָמַ֗עְלָה עַ֛ד בֶּן־חֲמִשִּׁ֥ים שָׁנָ֖ה תִּפְקֹ֣ד אוֹתָ֑ם כָּל־הַבָּא֙ לִצְבֹ֣א צָבָ֔א לַֽעֲבֹ֥ד עֲבֹדָ֖ה בְּאֹ֥הֶל מוֹעֵֽד׃ מִבַּר תְּלָתִין שְׁנִין וּלְעֵילָא, עַד בַּר חַמְשִׁין שְׁנִין--תִּמְנֵי יָתְהוֹן:  כָּל דְּאָתֵי לְחַיָּלָא חֵילָא, לְמִפְלַח פֻּלְחָנָא בְּמַשְׁכַּן זִמְנָא. 
ד,כד זֹ֣את עֲבֹדַ֔ת מִשְׁפְּחֹ֖ת הַגֵּֽרְשֻׁנִּ֑י לַֽעֲבֹ֖ד וּלְמַשָּֽׂא׃ דֵּין פֻּלְחַן, זַרְעֲיָת גֵּרְשׁוֹן--לְמִפְלַח, וּלְמִטַּל. 
ד,כה וְנָ֨שְׂא֜וּ אֶת־יְרִיעֹ֤ת הַמִּשְׁכָּן֙ וְאֶת־אֹ֣הֶל מוֹעֵ֔ד מִכְסֵ֕הוּ וּמִכְסֵ֛ה הַתַּ֥חַשׁ אֲשֶׁר־עָלָ֖יו מִלְמָ֑עְלָה וְאֶ֨ת־מָסַ֔ךְ פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ וְיִטְּלוּן יָת יְרִיעָת מַשְׁכְּנָא, וְיָת מַשְׁכַּן זִמְנָא, חוּפָאֵיהּ, וְחוּפָאָה דְּסָסְגוֹנָא דַּעֲלוֹהִי מִלְּעֵילָא; וְיָת פְּרָסָא--דִּתְרַע, מַשְׁכַּן זִמְנָא. 
ד,כו וְאֵת֩ קַלְעֵ֨י הֶֽחָצֵ֜ר וְאֶת־מָסַ֣ךְ ׀ פֶּ֣תַח ׀ שַׁ֣עַר הֶֽחָצֵ֗ר אֲשֶׁ֨ר עַל־הַמִּשְׁכָּ֤ן וְעַל־הַמִּזְבֵּ֨חַ֙ סָבִ֔יב וְאֵת֙ מֵֽיתְרֵיהֶ֔ם וְאֶֽת־כָּל־כְּלֵ֖י עֲבֹֽדָתָ֑ם וְאֵ֨ת כָּל־אֲשֶׁ֧ר יֵֽעָשֶׂ֛ה לָהֶ֖ם וְעָבָֽדוּ׃ וְיָת סְרָדֵי דָּרְתָא וְיָת פְּרָסָא דְּמַעֲלַנָא דִּתְרַע דָּרְתָא, דְּעַל מַשְׁכְּנָא וְעַל מַדְבְּחָא סְחוֹר סְחוֹר, וְיָת אֲטוּנֵיהוֹן, וְיָת כָּל מָנֵי פֻּלְחָנְהוֹן; וְיָת כָּל דְּיִתְמְסַר לְהוֹן, וְיִפְלְחוּן. 
ד,כז עַל־פִּי֩ אַֽהֲרֹ֨ן וּבָנָ֜יו תִּֽהְיֶ֗ה כָּל־עֲבֹדַת֙ בְּנֵ֣י הַגֵּֽרְשֻׁנִּ֔י לְכָ֨ל־מַשָּׂאָ֔ם וּלְכֹ֖ל עֲבֹֽדָתָ֑ם וּפְקַדְתֶּ֤ם עֲלֵהֶם֙ בְּמִשְׁמֶ֔רֶת אֵ֖ת כָּל־מַשָּׂאָֽם׃ עַל מֵימַר אַהֲרוֹן וּבְנוֹהִי יְהֵי, כָּל פֻּלְחַן בְּנֵי גֵּרְשׁוֹן, לְכָל מַטוּלְהוֹן, וּלְכֹל פֻּלְחָנְהוֹן; וּתְמַנּוֹן עֲלֵיהוֹן בְּמַטְּרָא, יָת כָּל מַטוּלְהוֹן. 
ד,כח זֹ֣את עֲבֹדַ֗ת מִשְׁפְּחֹ֛ת בְּנֵ֥י הַגֵּֽרְשֻׁנִּ֖י בְּאֹ֣הֶל מוֹעֵ֑ד וּמִ֨שְׁמַרְתָּ֔ם בְּיַד֙ אִֽיתָמָ֔ר בֶּֽן־אַהֲרֹ֖ן הַכֹּהֵֽן׃ {ס} דֵּין פֻּלְחַן, זַרְעֲיָת בְּנֵי גֵּרְשׁוֹן--בְּמַשְׁכַּן זִמְנָא; וּמַטַּרְתְּהוֹן--בִּידָא דְּאִיתָמָר, בַּר אַהֲרוֹן כָּהֲנָא. {ס}
ד,כט בְּנֵ֖י מְרָרִ֑י לְמִשְׁפְּחֹתָ֥ם לְבֵית־אֲבֹתָ֖ם תִּפְקֹ֥ד אֹתָֽם׃ בְּנֵי, מְרָרִי--לְזַרְעֲיָתְהוֹן לְבֵית אֲבָהָתְהוֹן, תִּמְנֵי יָתְהוֹן. 
ד,ל מִבֶּן֩ שְׁלֹשִׁ֨ים שָׁנָ֜ה וָמַ֗עְלָה וְעַ֛ד בֶּן־חֲמִשִּׁ֥ים שָׁנָ֖ה תִּפְקְדֵ֑ם כָּל־הַבָּא֙ לַצָּבָ֔א לַֽעֲבֹ֕ד אֶת־עֲבֹדַ֖ת אֹ֥הֶל מוֹעֵֽד׃ מִבַּר תְּלָתִין שְׁנִין וּלְעֵילָא, וְעַד בַּר חַמְשִׁין שְׁנִין--תִּמְנֵינוּן:  כָּל דְּאָתֵי, לְחֵילָא, לְמִפְלַח, יָת פֻּלְחַן מַשְׁכַּן זִמְנָא. 
ד,לא וְזֹאת֙ מִשְׁמֶ֣רֶת מַשָּׂאָ֔ם לְכָל־עֲבֹֽדָתָ֖ם בְּאֹ֣הֶל מוֹעֵ֑ד קַרְשֵׁי֙ הַמִּשְׁכָּ֔ן וּבְרִיחָ֖יו וְעַמּוּדָ֥יו וַֽאֲדָנָֽיו׃ וְדָא מַטְּרַת מַטוּלְהוֹן, לְכָל פֻּלְחָנְהוֹן בְּמַשְׁכַּן זִמְנָא:  דַּפֵּי, מַשְׁכְּנָא, וְעָבְרוֹהִי, וְעַמּוּדוֹהִי וְסָמְכוֹהִי. 
ד,לב וְעַמּוּדֵי֩ הֶֽחָצֵ֨ר סָבִ֜יב וְאַדְנֵיהֶ֗ם וִיתֵֽדֹתָם֙ וּמֵ֣יתְרֵיהֶ֔ם לְכָ֨ל־כְּלֵיהֶ֔ם וּלְכֹ֖ל עֲבֹֽדָתָ֑ם וּבְשֵׁמֹ֣ת תִּפְקְד֔וּ אֶת־כְּלֵ֖י מִשְׁמֶ֥רֶת מַשָּׂאָֽם׃ וְעַמּוּדֵי דָּרְתָא סְחוֹר סְחוֹר וְסָמְכֵיהוֹן, וְסִכֵּיהוֹן וַאֲטוּנֵיהוֹן, לְכָל מָנֵיהוֹן, וּלְכֹל פֻּלְחָנְהוֹן; וּבִשְׁמָהָן תִּמְנוֹן, יָת מָנֵי מַטְּרַת מַטוּלְהוֹן. 
ד,לג זֹ֣את עֲבֹדַ֗ת מִשְׁפְּחֹת֙ בְּנֵ֣י מְרָרִ֔י לְכָל־עֲבֹֽדָתָ֖ם בְּאֹ֣הֶל מוֹעֵ֑ד בְּיַד֙ אִֽיתָמָ֔ר בֶּֽן־אַהֲרֹ֖ן הַכֹּהֵֽן׃ דֵּין פֻּלְחַן, זַרְעֲיָת בְּנֵי מְרָרִי, לְכָל פֻּלְחָנְהוֹן, בְּמַשְׁכַּן זִמְנָא--בִּידָא, דְּאִיתָמָר, בַּר אַהֲרוֹן, כָּהֲנָא. 
ד,לד וַיִּפְקֹ֨ד מֹשֶׁ֧ה וְאַֽהֲרֹ֛ן וּנְשִׂיאֵ֥י הָֽעֵדָ֖ה אֶת־בְּנֵ֣י הַקְּהָתִ֑י לְמִשְׁפְּחֹתָ֖ם וּלְבֵ֥ית אֲבֹתָֽם׃ וּמְנָא מֹשֶׁה וְאַהֲרוֹן וְרַבְרְבֵי כְּנִשְׁתָּא, יָת בְּנֵי קְהָת, לְזַרְעֲיָתְהוֹן, וּלְבֵית אֲבָהָתְהוֹן. 
ד,לה מִבֶּ֨ן שְׁלֹשִׁ֤ים שָׁנָה֙ וָמַ֔עְלָה וְעַ֖ד בֶּן־חֲמִשִּׁ֣ים שָׁנָ֑ה כָּל־הַבָּא֙ לַצָּבָ֔א לַֽעֲבֹדָ֖ה בְּאֹ֥הֶל מוֹעֵֽד׃ מִבַּר תְּלָתִין שְׁנִין וּלְעֵילָא, וְעַד בַּר חַמְשִׁין שְׁנִין--כָּל דְּאָתֵי, לְחֵילָא, לְפֻלְחָנָא, בְּמַשְׁכַּן זִמְנָא. 
ד,לו וַיִּֽהְי֥וּ פְקֻֽדֵיהֶ֖ם לְמִשְׁפְּחֹתָ֑ם אַלְפַּ֕יִם שְׁבַ֥ע מֵא֖וֹת וַֽחֲמִשִּֽׁים׃ וַהֲווֹ מִנְיָנֵיהוֹן, לְזַרְעֲיָתְהוֹן--תְּרֵין אַלְפִין, שְׁבַע מְאָה וְחַמְשִׁין. 
ד,לז אֵ֤לֶּה פְקוּדֵי֙ מִשְׁפְּחֹ֣ת הַקְּהָתִ֔י כָּל־הָֽעֹבֵ֖ד בְּאֹ֣הֶל מוֹעֵ֑ד אֲשֶׁ֨ר פָּקַ֤ד מֹשֶׁה֙ וְאַֽהֲרֹ֔ן עַל־פִּ֥י יְהוָ֖ה בְּיַד־מֹשֶֽׁה׃ {ס} אִלֵּין מִנְיָנֵי זַרְעֲיָת קְהָת, כָּל דְּפָלַח בְּמַשְׁכַּן זִמְנָא, דִּמְנָא מֹשֶׁה וְאַהֲרוֹן, עַל מֵימְרָא דַּייָ בִּידָא דְּמֹשֶׁה. {ס}
ד,לח וּפְקוּדֵ֖י בְּנֵ֣י גֵֽרְשׁ֑וֹן לְמִשְׁפְּחוֹתָ֖ם וּלְבֵ֥ית אֲבֹתָֽם׃ וּמִנְיָנֵי, בְּנֵי גֵּרְשׁוֹן, לְזַרְעֲיָתְהוֹן, וּלְבֵית אֲבָהָתְהוֹן. 
ד,לט מִבֶּ֨ן שְׁלֹשִׁ֤ים שָׁנָה֙ וָמַ֔עְלָה וְעַ֖ד בֶּן־חֲמִשִּׁ֣ים שָׁנָ֑ה כָּל־הַבָּא֙ לַצָּבָ֔א לַֽעֲבֹדָ֖ה בְּאֹ֥הֶל מוֹעֵֽד׃ מִבַּר תְּלָתִין שְׁנִין וּלְעֵילָא, וְעַד בַּר חַמְשִׁין שְׁנִין--כָּל דְּאָתֵי, לְחֵילָא, לְפֻלְחָנָא, בְּמַשְׁכַּן זִמְנָא. 
ד,מ וַיִּֽהְיוּ֙ פְּקֻ֣דֵיהֶ֔ם לְמִשְׁפְּחֹתָ֖ם לְבֵ֣ית אֲבֹתָ֑ם אַלְפַּ֕יִם וְשֵׁ֥שׁ מֵא֖וֹת וּשְׁלֹשִֽׁים׃ וַהֲווֹ, מִנְיָנֵיהוֹן, לְזַרְעֲיָתְהוֹן, לְבֵית אֲבָהָתְהוֹן--תְּרֵין אַלְפִין, וְשֵׁית מְאָה וּתְלָתִין. 
ד,מא אֵ֣לֶּה פְקוּדֵ֗י מִשְׁפְּחֹת֙ בְּנֵ֣י גֵֽרְשׁ֔וֹן כָּל־הָֽעֹבֵ֖ד בְּאֹ֣הֶל מוֹעֵ֑ד אֲשֶׁ֨ר פָּקַ֥ד מֹשֶׁ֛ה וְאַֽהֲרֹ֖ן עַל־פִּ֥י יְהוָֽה׃ אִלֵּין מִנְיָנֵי, זַרְעֲיָת בְּנֵי גֵּרְשׁוֹן, כָּל דְּפָלַח, בְּמַשְׁכַּן זִמְנָא--דִּמְנָא מֹשֶׁה וְאַהֲרוֹן, עַל מֵימְרָא דַּייָ. 
ד,מב וּפְקוּדֵ֕י מִשְׁפְּחֹ֖ת בְּנֵ֣י מְרָרִ֑י לְמִשְׁפְּחֹתָ֖ם לְבֵ֥ית אֲבֹתָֽם׃ וּמִנְיָנֵי, זַרְעֲיָת בְּנֵי מְרָרִי, לְזַרְעֲיָתְהוֹן, לְבֵית אֲבָהָתְהוֹן. 
ד,מג מִבֶּ֨ן שְׁלֹשִׁ֤ים שָׁנָה֙ וָמַ֔עְלָה וְעַ֖ד בֶּן־חֲמִשִּׁ֣ים שָׁנָ֑ה כָּל־הַבָּא֙ לַצָּבָ֔א לַֽעֲבֹדָ֖ה בְּאֹ֥הֶל מוֹעֵֽד׃ מִבַּר תְּלָתִין שְׁנִין וּלְעֵילָא, וְעַד בַּר חַמְשִׁין שְׁנִין--כָּל דְּאָתֵי, לְחֵילָא, לְפֻלְחָנָא, בְּמַשְׁכַּן זִמְנָא. 
ד,מד וַיִּֽהְי֥וּ פְקֻֽדֵיהֶ֖ם לְמִשְׁפְּחֹתָ֑ם שְׁלֹ֥שֶׁת אֲלָפִ֖ים וּמָאתָֽיִם׃ וַהֲווֹ מִנְיָנֵיהוֹן, לְזַרְעֲיָתְהוֹן--תְּלָתָא אַלְפִין, וּמָאתַן. 
ד,מה אֵ֣לֶּה פְקוּדֵ֔י מִשְׁפְּחֹ֖ת בְּנֵ֣י מְרָרִ֑י אֲשֶׁ֨ר פָּקַ֤ד מֹשֶׁה֙ וְאַֽהֲרֹ֔ן עַל־פִּ֥י יְהוָ֖ה בְּיַד־מֹשֶֽׁה׃ אִלֵּין מִנְיָנֵי, זַרְעֲיָת בְּנֵי מְרָרִי, דִּמְנָא מֹשֶׁה וְאַהֲרוֹן, עַל מֵימְרָא דַּייָ בִּידָא דְּמֹשֶׁה. 
ד,מו כָּֽל־הַפְּקֻדִ֡ים אֲשֶׁר֩ פָּקַ֨ד מֹשֶׁ֧ה וְאַֽהֲרֹ֛ן וּנְשִׂיאֵ֥י יִשְׂרָאֵ֖ל אֶת־הַלְוִיִּ֑ם לְמִשְׁפְּחֹתָ֖ם וּלְבֵ֥ית אֲבֹתָֽם׃ כָּל מִנְיָנַיָּא דִּמְנָא מֹשֶׁה וְאַהֲרוֹן, וְרַבְרְבֵי יִשְׂרָאֵל--יָת לֵיוָאֵי:  לְזַרְעֲיָתְהוֹן, וּלְבֵית אֲבָהָתְהוֹן. 
ד,מז מִבֶּ֨ן שְׁלֹשִׁ֤ים שָׁנָה֙ וָמַ֔עְלָה וְעַ֖ד בֶּן־חֲמִשִּׁ֣ים שָׁנָ֑ה כָּל־הַבָּ֗א לַֽעֲבֹ֨ד עֲבֹדַ֧ת עֲבֹדָ֛ה וַֽעֲבֹדַ֥ת מַשָּׂ֖א בְּאֹ֥הֶל מוֹעֵֽד׃ מִבַּר תְּלָתִין שְׁנִין וּלְעֵילָא, וְעַד בַּר חַמְשִׁין שְׁנִין:  כָּל דְּאָתֵי, לְמִפְלַח פֻּלְחַן פֻּלְחָנָא וּפֻלְחַן מַטוּל--בְּמַשְׁכַּן זִמְנָא. 
ד,מח וַיִּֽהְי֖וּ פְּקֻֽדֵיהֶ֑ם שְׁמֹנַ֣ת אֲלָפִ֔ים וַֽחֲמֵ֥שׁ מֵא֖וֹת וּשְׁמֹנִֽים׃ וַהֲווֹ, מִנְיָנֵיהוֹן--תְּמָנְיָא אַלְפִין, וַחֲמֵישׁ מְאָה וּתְמָנַן. 
ד,מט עַל־פִּ֨י יְהוָ֜ה פָּקַ֤ד אוֹתָם֙ בְּיַד־מֹשֶׁ֔ה אִ֥ישׁ אִ֛ישׁ עַל־עֲבֹֽדָת֖וֹ וְעַל־מַשָּׂא֑וֹ וּפְקֻדָ֕יו אֲשֶׁר־צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ {פ} עַל מֵימְרָא דַּייָ מְנָא יָתְהוֹן, בִּידָא דְּמֹשֶׁה--גְּבַר גְּבַר עַל פֻּלְחָנֵיהּ, וְעַל מַטוּלֵיהּ; וּמִנְיָנוֹהִי, דְּפַקֵּיד יְיָ יָת מֹשֶׁה. {פ}
ה,א וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ה,ב צַ֚ו אֶת־בְּנֵ֣י יִשְׂרָאֵ֔ל וִֽישַׁלְּחוּ֙ מִן־הַֽמַּחֲנֶ֔ה כָּל־צָר֖וּעַ וְכָל־זָ֑ב וְכֹ֖ל טָמֵ֥א לָנָֽפֶשׁ׃ פַּקֵּיד, יָת בְּנֵי יִשְׂרָאֵל, וִישַׁלְּחוּן מִן מַשְׁרִיתָא, כָּל דִּסְגִיר וְכָל דְּדָאִיב; וְכֹל, דִּמְסָאַב לִטְמֵי נַפְשָׁא. 
ה,ג מִזָּכָ֤ר עַד־נְקֵבָה֙ תְּשַׁלֵּ֔חוּ אֶל־מִח֥וּץ לַֽמַּחֲנֶ֖ה תְּשַׁלְּח֑וּם וְלֹ֤א יְטַמְּאוּ֙ אֶת־מַ֣חֲנֵיהֶ֔ם אֲשֶׁ֥ר אֲנִ֖י שֹׁכֵ֥ן בְּתוֹכָֽם׃ מִדְּכַר עַד נֻקְבָּא תְּשַׁלְּחוּן, לְמִבַּרָא לְמַשְׁרִיתָא תְּשַׁלְּחוּנוּנוּן; וְלָא יְסָאֲבוּן יָת מַשְׁרְיָתְהוֹן, דִּשְׁכִינְתִי שָׁרְיָא בֵּינֵיהוֹן. 
ה,ד וַיַּֽעֲשׂוּ־כֵן֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וַיְשַׁלְּח֣וּ אוֹתָ֔ם אֶל־מִח֖וּץ לַֽמַּחֲנֶ֑ה כַּֽאֲשֶׁ֨ר דִּבֶּ֤ר יְהוָה֙ אֶל־מֹשֶׁ֔ה כֵּ֥ן עָשׂ֖וּ בְּנֵ֥י יִשְׂרָאֵֽל׃ {פ} וַעֲבַדוּ כֵּין, בְּנֵי יִשְׂרָאֵל, וְשַׁלַּחוּ יָתְהוֹן, לְמִבַּרָא לְמַשְׁרִיתָא:  כְּמָא דְּמַלֵּיל יְיָ עִם מֹשֶׁה, כֵּין עֲבַדוּ בְּנֵי יִשְׂרָאֵל. {פ}
ה,ה וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ה,ו דַּבֵּר֮ אֶל־בְּנֵ֣י יִשְׂרָאֵל֒ אִ֣ישׁ אֽוֹ־אִשָּׁ֗ה כִּ֤י יַֽעֲשׂוּ֙ מִכָּל־חַטֹּ֣את הָֽאָדָ֔ם לִמְעֹ֥ל מַ֖עַל בַּֽיהוָ֑ה וְאָֽשְׁמָ֖ה הַנֶּ֥פֶשׁ הַהִֽוא׃ מַלֵּיל, עִם בְּנֵי יִשְׂרָאֵל, גְּבַר אוֹ אִתָּא אֲרֵי יַעְבְּדוּן מִכָּל חוֹבֵי אֲנָשָׁא, לְשַׁקָּרָא שְׁקַר קֳדָם יְיָ; וִיחוּב, אֲנָשָׁא הַהוּא. 
ה,ז וְהִתְוַדּ֗וּ אֶֽת־חַטָּאתָם֮ אֲשֶׁ֣ר עָשׂוּ֒ וְהֵשִׁ֤יב אֶת־אֲשָׁמוֹ֙ בְּרֹאשׁ֔וֹ וַחֲמִֽישִׁת֖וֹ יֹסֵ֣ף עָלָ֑יו וְנָתַ֕ן לַֽאֲשֶׁ֖ר אָשַׁ֥ם לֽוֹ׃ וִיוַדּוֹן, יָת חוֹבֵיהוֹן דַּעֲבַדוּ, וְיָתִיב יָת חוֹבְתֵיהּ בְּרֵישֵׁיהּ, וְחֻמְשֵׁיהּ יוֹסֵיף עֲלוֹהִי; וְיִתֵּין, לִדְחַיָּב לֵיהּ. 
ה,ח וְאִם־אֵ֨ין לָאִ֜ישׁ גֹּאֵ֗ל לְהָשִׁ֤יב הָֽאָשָׁם֙ אֵלָ֔יו הָֽאָשָׁ֛ם הַמּוּשָׁ֥ב לַֽיהוָ֖ה לַכֹּהֵ֑ן מִלְּבַ֗ד אֵ֚יל הַכִּפֻּרִ֔ים אֲשֶׁ֥ר יְכַפֶּר־בּ֖וֹ עָלָֽיו׃ וְאִם לֵית לְגֻבְרָא פָּרִיק, לְאָתָבָא חוֹבְתָא לֵיהּ--חוֹבְתָא דִּמְתִיב קֳדָם יְיָ, לְכָהֲנָא:  בָּר, מִדְּכַר כִּפּוּרַיָּא, דִּיכַפַּר בֵּיהּ, עֲלוֹהִי. 
ה,ט וְכָל־תְּרוּמָ֞ה לְכָל־קָדְשֵׁ֧י בְנֵֽי־יִשְׂרָאֵ֛ל אֲשֶׁר־יַקְרִ֥יבוּ לַכֹּהֵ֖ן ל֥וֹ יִֽהְיֶֽה׃ וְכָל אַפְרָשׁוּתָא לְכָל קֻדְשַׁיָּא דִּבְנֵי יִשְׂרָאֵל, דִּיקָרְבוּן לְכָהֲנָא--דִּילֵיהּ יְהֵי. 
ה,י וְאִ֥ישׁ אֶת־קֳדָשָׁ֖יו ל֣וֹ יִֽהְי֑וּ אִ֛ישׁ אֲשֶׁר־יִתֵּ֥ן לַכֹּהֵ֖ן ל֥וֹ יִֽהְיֶֽה׃ {פ} וּגְבַר יָת מַעְסַר קֻדְשׁוֹהִי, דִּילֵיהּ יְהֵי; גְּבַר דְּיִתֵּין לְכָהֲנָא, דִּילֵיהּ יְהֵי. {פ}
ה,יא וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ה,יב דַּבֵּר֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָֽמַרְתָּ֖ אֲלֵהֶ֑ם אִ֥ישׁ אִישׁ֙ כִּֽי־תִשְׂטֶ֣ה אִשְׁתּ֔וֹ וּמָֽעֲלָ֥ה ב֖וֹ מָֽעַל׃ מַלֵּיל עִם בְּנֵי יִשְׂרָאֵל, וְתֵימַר לְהוֹן:  גְּבַר גְּבַר אֲרֵי תִּסְטֵי אִתְּתֵיהּ, וּתְשַׁקַּר בֵּיהּ שְׁקַר. 
ה,יג וְשָׁכַ֨ב אִ֣ישׁ אֹתָהּ֮ שִׁכְבַת־זֶרַע֒ וְנֶעְלַם֙ מֵֽעֵינֵ֣י אִישָׁ֔הּ וְנִסְתְּרָ֖ה וְהִ֣יא נִטְמָ֑אָה וְעֵד֙ אֵ֣ין בָּ֔הּ וְהִ֖וא לֹ֥א נִתְפָּֽשָׂה׃ וְיִשְׁכּוֹב גְּבַר יָתַהּ, שִׁכְבַת זַרְעָא, וִיהֵי מְכֻסַּא מֵעֵינֵי בַּעְלַהּ, וּמִטַּמְרָא וְהִיא מְסָאֲבָא; וְסָהִיד לֵית בַּהּ, וְהִיא לָא אִתְאֲחַדַת. 
ה,יד וְעָבַ֨ר עָלָ֧יו רֽוּחַ־קִנְאָ֛ה וְקִנֵּ֥א אֶת־אִשְׁתּ֖וֹ וְהִ֣וא נִטְמָ֑אָה אֽוֹ־עָבַ֨ר עָלָ֤יו רֽוּחַ־קִנְאָה֙ וְקִנֵּ֣א אֶת־אִשְׁתּ֔וֹ וְהִ֖יא לֹ֥א נִטְמָֽאָה׃ וְיִעְבַּר עֲלוֹהִי רוּחַ קִנְאָה וִיקַנֵּי יָת אִתְּתֵיהּ, וְהִיא מְסָאֲבָא; אוֹ עֲבַר עֲלוֹהִי רוּחַ קִנְאָה וְקַנִּי יָת אִתְּתֵיהּ, וְהִיא לָא מְסָאֲבָא. 
ה,טו וְהֵבִ֨יא הָאִ֣ישׁ אֶת־אִשְׁתּוֹ֮ אֶל־הַכֹּהֵן֒ וְהֵבִ֤יא אֶת־קָרְבָּנָהּ֙ עָלֶ֔יהָ עֲשִׂירִ֥ת הָֽאֵיפָ֖ה קֶ֣מַח שְׂעֹרִ֑ים לֹֽא־יִצֹ֨ק עָלָ֜יו שֶׁ֗מֶן וְלֹֽא־יִתֵּ֤ן עָלָיו֙ לְבֹנָ֔ה כִּֽי־מִנְחַ֤ת קְנָאֹת֙ ה֔וּא מִנְחַ֥ת זִכָּר֖וֹן מַזְכֶּ֥רֶת עָוֽ‍ֹן׃ וְיַיְתֵי גֻּבְרָא יָת אִתְּתֵיהּ, לְוָת כָּהֲנָא, וְיַיְתֵי יָת קֻרְבָּנַהּ עֲלַהּ, חַד מִן עַסְרָא בִּתְלָת סְאִין קֶמַח סְעָרִין; לָא יְרִיק עֲלוֹהִי מִשְׁחָא, וְלָא יִתֵּין עֲלוֹהִי לְבוֹנְתָא--אֲרֵי מִנְחַת קִנְאֲתָא הוּא, מִנְחַת דֻּכְרָנָא מַדְכֶרֶת חוֹבִין. 
ה,טז וְהִקְרִ֥יב אֹתָ֖הּ הַכֹּהֵ֑ן וְהֶֽעֱמִדָ֖הּ לִפְנֵ֥י יְהוָֽה׃ וִיקָרֵיב יָתַהּ, כָּהֲנָא; וִיקִימִנַּהּ, קֳדָם יְיָ. 
ה,יז וְלָקַ֧ח הַכֹּהֵ֛ן מַ֥יִם קְדֹשִׁ֖ים בִּכְלִי־חָ֑רֶשׂ וּמִן־הֶֽעָפָ֗ר אֲשֶׁ֤ר יִֽהְיֶה֙ בְּקַרְקַ֣ע הַמִּשְׁכָּ֔ן יִקַּ֥ח הַכֹּהֵ֖ן וְנָתַ֥ן אֶל־הַמָּֽיִם׃ וְיִסַּב כָּהֲנָא מֵי כִּיּוֹר, בְּמָן דַּחֲסַף; וּמִן עַפְרָא, דִּיהֵי בְּאֵיסוֹדֵי מַשְׁכְּנָא, יִסַּב כָּהֲנָא, וְיִתֵּין לְמַיָּא. 
ה,יח וְהֶֽעֱמִ֨יד הַכֹּהֵ֥ן אֶֽת־הָאִשָּׁה֮ לִפְנֵ֣י יְהוָה֒ וּפָרַע֙ אֶת־רֹ֣אשׁ הָֽאִשָּׁ֔ה וְנָתַ֣ן עַל־כַּפֶּ֗יהָ אֵ֚ת מִנְחַ֣ת הַזִּכָּר֔וֹן מִנְחַ֥ת קְנָאֹ֖ת הִ֑וא וּבְיַ֤ד הַכֹּהֵן֙ יִֽהְי֔וּ מֵ֥י הַמָּרִ֖ים הַמְאָֽרְרִֽים׃ וִיקִים כָּהֲנָא יָת אִתְּתָא, קֳדָם יְיָ, וְיִפְרַע יָת רֵישָׁא דְּאִתְּתָא, וְיִתֵּין עַל יְדַהָא יָת מִנְחַת דֻּכְרָנָא מִנְחַת קִנְאֲתָא הִיא; וּבִידָא דְּכָהֲנָא יְהוֹן, מַיָּא מָרִירַיָּא מְלָטְטַיָּא. 
ה,יט וְהִשְׁבִּ֨יעַ אֹתָ֜הּ הַכֹּהֵ֗ן וְאָמַ֤ר אֶל־הָֽאִשָּׁה֙ אִם־לֹ֨א שָׁכַ֥ב אִישׁ֙ אֹתָ֔ךְ וְאִם־לֹ֥א שָׂטִ֛ית טֻמְאָ֖ה תַּ֣חַת אִישֵׁ֑ךְ הִנָּקִ֕י מִמֵּ֛י הַמָּרִ֥ים הַֽמְאָרְרִ֖ים הָאֵֽלֶּה׃ וְיוֹמֵי יָתַהּ כָּהֲנָא, וְיֵימַר לְאִתְּתָא אִם לָא שְׁכֵיב גְּבַר יָתִיךְ, וְאִם לָא סְטֵית לְאִסְתַּאָבָא, בָּר מִבַּעְלִיךְ--הִוַי זַכָּאָה, מִמַּיָּא מָרִירַיָּא מְלָטְטַיָּא הָאִלֵּין. 
ה,כ וְאַ֗תְּ כִּ֥י שָׂטִ֛ית תַּ֥חַת אִישֵׁ֖ךְ וְכִ֣י נִטְמֵ֑את וַיִּתֵּ֨ן אִ֥ישׁ בָּךְ֙ אֶת־שְׁכָבְתּ֔וֹ מִֽבַּלְעֲדֵ֖י אִישֵֽׁךְ׃ וְאַתְּ, אֲרֵי סְטֵית בָּר מִבַּעְלִיךְ--וַאֲרֵי אִסְתָּאַבְתְּ; וִיהַב גְּבַר בִּיךְ יָת שְׁכֻבְתֵּיהּ, בָּר מִבַּעְלִיךְ. 
ה,כא וְהִשְׁבִּ֨יעַ הַכֹּהֵ֥ן אֶֽת־הָאִשָּׁה֮ בִּשְׁבֻעַ֣ת הָֽאָלָה֒ וְאָמַ֤ר הַכֹּהֵן֙ לָֽאִשָּׁ֔ה יִתֵּ֨ן יְהוָ֥ה אוֹתָ֛ךְ לְאָלָ֥ה וְלִשְׁבֻעָ֖ה בְּת֣וֹךְ עַמֵּ֑ךְ בְּתֵ֨ת יְהוָ֤ה אֶת־יְרֵכֵךְ֙ נֹפֶ֔לֶת וְאֶת־בִּטְנֵ֖ךְ צָבָֽה׃ וְיוֹמֵי כָּהֲנָא יָת אִתְּתָא, בְּמוֹמָתָא דִּלְוָטָא, וְיֵימַר כָּהֲנָא לְאִתְּתָא, יִתֵּין יְיָ יָתִיךְ לִלְוָט וּלְמוֹמֵי בְּגוֹ עַמִּיךְ--בִּדְיִתֵּין יְיָ יָת יִרְכִּיךְ מַסְיָא, וְיָת מְעַכִי נְפִיחִין. 
ה,כב וּ֠בָאוּ הַמַּ֨יִם הַמְאָֽרְרִ֤ים הָאֵ֨לֶּה֙ בְּֽמֵעַ֔יִךְ לַצְבּ֥וֹת בֶּ֖טֶן וְלַנְפִּ֣ל יָרֵ֑ךְ וְאָֽמְרָ֥ה הָֽאִשָּׁ֖ה אָמֵ֥ן ׀ אָמֵֽן׃ וְיֵיעֲלוּן מַיָּא מְלָטְטַיָּא הָאִלֵּין, בִּמְעַכִי, לְאַפָּחָא מְעִין, וּלְאַמְסָאָה יִירַךְ; וְתֵימַר אִתְּתָא, אָמֵן אָמֵן. 
ה,כג וְ֠כָתַב אֶת־הָֽאָלֹ֥ת הָאֵ֛לֶּה הַכֹּהֵ֖ן בַּסֵּ֑פֶר וּמָחָ֖ה אֶל־מֵ֥י הַמָּרִֽים׃ וְיִכְתּוֹב יָת לְוָטַיָּא הָאִלֵּין, כָּהֲנָא--בְּסִפְרָא; וְיִמְחוֹק, לְמַיָּא מְרִירַיָּא. 
ה,כד וְהִשְׁקָה֙ אֶת־הָ֣אִשָּׁ֔ה אֶת־מֵ֥י הַמָּרִ֖ים הַמְאָֽרְרִ֑ים וּבָ֥אוּ בָ֛הּ הַמַּ֥יִם הַֽמְאָרְרִ֖ים לְמָרִֽים׃ וְיַשְׁקֵי, יָת אִתְּתָא, יָת מַיָּא מָרִירַיָּא, מְלָטְטַיָּא; וְיֵיעֲלוּן בַּהּ מַיָּא מְלָטְטַיָּא, לִלְוָט. 
ה,כה וְלָקַ֤ח הַכֹּהֵן֙ מִיַּ֣ד הָֽאִשָּׁ֔ה אֵ֖ת מִנְחַ֣ת הַקְּנָאֹ֑ת וְהֵנִ֤יף אֶת־הַמִּנְחָה֙ לִפְנֵ֣י יְהוָ֔ה וְהִקְרִ֥יב אֹתָ֖הּ אֶל־הַמִּזְבֵּֽחַ׃ וְיִסַּב כָּהֲנָא מִיְּדָא דְּאִתְּתָא, יָת מִנְחַת קִנְאֲתָא; וִירִים יָת מִנְחָתָא קֳדָם יְיָ, וִיקָרֵיב יָתַהּ לְמַדְבְּחָא. 
ה,כו וְקָמַ֨ץ הַכֹּהֵ֤ן מִן־הַמִּנְחָה֙ אֶת־אַזְכָּ֣רָתָ֔הּ וְהִקְטִ֖יר הַמִּזְבֵּ֑חָה וְאַחַ֛ר יַשְׁקֶ֥ה אֶת־הָֽאִשָּׁ֖ה אֶת־הַמָּֽיִם׃ וְיִקְמוֹץ כָּהֲנָא מִן מִנְחָתָא יָת אַדְכָרְתַהּ, וְיַסֵּיק לְמַדְבְּחָא; וּבָתַר כֵּין יַשְׁקֵי יָת אִתְּתָא, יָת מַיָּא. 
ה,כז וְהִשְׁקָ֣הּ אֶת־הַמַּ֗יִם וְהָֽיְתָ֣ה אִֽם־נִטְמְאָה֮ וַתִּמְעֹ֣ל מַ֣עַל בְּאִישָׁהּ֒ וּבָ֨אוּ בָ֜הּ הַמַּ֤יִם הַמְאָֽרְרִים֙ לְמָרִ֔ים וְצָֽבְתָ֣ה בִטְנָ֔הּ וְנָֽפְלָ֖ה יְרֵכָ֑הּ וְהָֽיְתָ֧ה הָֽאִשָּׁ֛ה לְאָלָ֖ה בְּקֶ֥רֶב עַמָּֽהּ׃ וְיַשְׁקֵינַהּ יָת מַיָּא, וּתְהֵי אִם אִסְתָּאַבַת וְשַׁקַּרַת שְׁקַר בְּבַעְלַהּ--וְיֵיעֲלוּן בַּהּ מַיָּא מְלָטְטַיָּא לִלְוָט, וְיִפְּחוּן מְעַהָא וְתִתְמְסֵי יִרְכַּהּ; וּתְהֵי אִתְּתָא לִלְוָטָא, בְּגוֹ עַמַּהּ. 
ה,כח וְאִם־לֹ֤א נִטְמְאָה֙ הָֽאִשָּׁ֔ה וּטְהֹרָ֖ה הִ֑וא וְנִקְּתָ֖ה וְנִזְרְעָ֥ה זָֽרַע׃ וְאִם לָא אִסְתָּאַבַת אִתְּתָא, וְדָכְיָא הִיא--וְתִפּוֹק זַכָּאָה, וּתְעַדֵּי עִדּוּי. 
ה,כט זֹ֥את תּוֹרַ֖ת הַקְּנָאֹ֑ת אֲשֶׁ֨ר תִּשְׂטֶ֥ה אִשָּׁ֛ה תַּ֥חַת אִישָׁ֖הּ וְנִטְמָֽאָה׃ דָּא אוֹרָיְתָא, דְּקִנְאֲתָא, דְּתִסְטֵי אִתְּתָא בָּר מִבַּעְלַהּ, וְתִסְתָּאַב. 
ה,ל א֣וֹ אִ֗ישׁ אֲשֶׁ֨ר תַּֽעֲבֹ֥ר עָלָ֛יו ר֥וּחַ קִנְאָ֖ה וְקִנֵּ֣א אֶת־אִשְׁתּ֑וֹ וְהֶֽעֱמִ֤יד אֶת־הָֽאִשָּׁה֙ לִפְנֵ֣י יְהוָ֔ה וְעָ֤שָׂה לָהּ֙ הַכֹּהֵ֔ן אֵ֥ת כָּל־הַתּוֹרָ֖ה הַזֹּֽאת׃ אוֹ גְּבַר, דְּתִעְבַּר עֲלוֹהִי רוּחַ קִנְאָה--וִיקַנֵּי יָת אִתְּתֵיהּ; וִיקִים יָת אִתְּתָא, קֳדָם יְיָ, וְיַעֲבֵיד לַהּ כָּהֲנָא, יָת כָּל אוֹרָיְתָא הָדָא. 
ה,לא וְנִקָּ֥ה הָאִ֖ישׁ מֵֽעָו֑‍ֹן וְהָֽאִשָּׁ֣ה הַהִ֔וא תִּשָּׂ֖א אֶת־עֲו‍ֹנָֽהּ׃ {פ} וִיהֵי זָכָא גֻּבְרָא, מֵחוֹבִין; וְאִתְּתָא הַהִיא, תְּקַבֵּיל יָת חוֹבַהּ. {פ}
ו,א וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ו,ב דַּבֵּר֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָֽמַרְתָּ֖ אֲלֵהֶ֑ם אִ֣ישׁ אֽוֹ־אִשָּׁ֗ה כִּ֤י יַפְלִא֙ לִנְדֹּר֙ נֶ֣דֶר נָזִ֔יר לְהַזִּ֖יר לַֽיהוָֽה׃ מַלֵּיל עִם בְּנֵי יִשְׂרָאֵל, וְתֵימַר לְהוֹן:  גְּבַר אוֹ אִתָּא, אֲרֵי יְפָרֵישׁ לְמִדַּר נְדַר נְזִירוּ--לְמִזַּר, קֳדָם יְיָ. 
ו,ג מִיַּ֤יִן וְשֵׁכָר֙ יַזִּ֔יר חֹ֥מֶץ יַ֛יִן וְחֹ֥מֶץ שֵׁכָ֖ר לֹ֣א יִשְׁתֶּ֑ה וְכָל־מִשְׁרַ֤ת עֲנָבִים֙ לֹ֣א יִשְׁתֶּ֔ה וַֽעֲנָבִ֛ים לַחִ֥ים וִֽיבֵשִׁ֖ים לֹ֥א יֹאכֵֽל׃ מֵחֲמַר חֲדַת וְעַתִּיק יִזַּר, חַל דַּחֲמַר חֲדַת וְחַל דַּחֲמַר עַתִּיק לָא יִשְׁתֵּי; וְכָל מַתְרוּת עִנְבִּין לָא יִשְׁתֵּי, וְעִנְבִּין רַטִּיבִין וְיַבִּישִׁין לָא יֵיכוֹל. 
ו,ד כֹּ֖ל יְמֵ֣י נִזְר֑וֹ מִכֹּל֩ אֲשֶׁ֨ר יֵֽעָשֶׂ֜ה מִגֶּ֣פֶן הַיַּ֗יִן מֵֽחַרְצַנִּ֛ים וְעַד־זָ֖ג לֹ֥א יֹאכֵֽל׃ כֹּל, יוֹמֵי נִזְרֵיהּ:  מִכֹּל דְּיִתְעֲבֵיד מִגֻּפְנָא דְּחַמְרָא, מִפֻּרְצְנִין וְעַד עִצּוּרִין--לָא יֵיכוֹל. 
ו,ה כָּל־יְמֵי֙ נֶ֣דֶר נִזְר֔וֹ תַּ֖עַר לֹֽא־יַעֲבֹ֣ר עַל־רֹאשׁ֑וֹ עַד־מְלֹ֨את הַיָּמִ֜ם אֲשֶׁר־יַזִּ֤יר לַֽיהוָה֙ קָדֹ֣שׁ יִֽהְיֶ֔ה גַּדֵּ֥ל פֶּ֖רַע שְׂעַ֥ר רֹאשֽׁוֹ׃ כָּל יוֹמֵי נְדַר נִזְרֵיהּ, מַסְפַּר לָא יִעְבַּר עַל רֵישֵׁיהּ:  עַד מִשְׁלַם יוֹמַיָּא דִּנְזִיר קֳדָם יְיָ, קַדִּישׁ יְהֵי--יְרַבֵּי פֵּירוּעַ, סְעַר רֵישֵׁיהּ. 
ו,ו כָּל־יְמֵ֥י הַזִּיר֖וֹ לַֽיהוָ֑ה עַל־נֶ֥פֶשׁ מֵ֖ת לֹ֥א יָבֹֽא׃ כָּל יוֹמִין דִּנְזִיר, קֳדָם יְיָ, עַל נַפְשָׁת מִיתָא, לָא יֵיעוֹל. 
ו,ז לְאָבִ֣יו וּלְאִמּ֗וֹ לְאָחִיו֙ וּלְאַ֣חֹת֔וֹ לֹֽא־יִטַּמָּ֥א לָהֶ֖ם בְּמֹתָ֑ם כִּ֛י נֵ֥זֶר אֱלֹהָ֖יו עַל־רֹאשֽׁוֹ׃ לַאֲבוּהִי וּלְאִמֵּיהּ, לַאֲחוּהִי וְלַאֲחָתֵיהּ--לָא יִסְתָּאַב לְהוֹן, בְּמוֹתְהוֹן:  אֲרֵי נִזְרָא דֶּאֱלָהֵיהּ, עַל רֵישֵׁיהּ. 
ו,ח כֹּ֖ל יְמֵ֣י נִזְר֑וֹ קָדֹ֥שׁ ה֖וּא לַֽיהוָֽה׃ כֹּל, יוֹמֵי נִזְרֵיהּ, קַדִּישׁ הוּא, קֳדָם יְיָ. 
ו,ט וְכִֽי־יָמ֨וּת מֵ֤ת עָלָיו֙ בְּפֶ֣תַע פִּתְאֹ֔ם וְטִמֵּ֖א רֹ֣אשׁ נִזְר֑וֹ וְגִלַּ֤ח רֹאשׁוֹ֙ בְּי֣וֹם טָֽהֳרָת֔וֹ בַּיּ֥וֹם הַשְּׁבִיעִ֖י יְגַלְּחֶֽנּוּ׃ וַאֲרֵי יְמוּת מִיתָא עֲלוֹהִי בִּתְכֵּיף שָׁלוּ, וִיסַאֵיב רֵישׁ נִזְרֵיהּ--וִיגַלַּח רֵישֵׁיהּ בְּיוֹמָא דִּדְכוּתֵיהּ, בְּיוֹמָא שְׁבִיעָאָה יְגַלְּחִנֵּיהּ. 
ו,י וּבַיּ֣וֹם הַשְּׁמִינִ֗י יָבִא֙ שְׁתֵּ֣י תֹרִ֔ים א֥וֹ שְׁנֵ֖י בְּנֵ֣י יוֹנָ֑ה אֶ֨ל־הַכֹּהֵ֔ן אֶל־פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ וּבְיוֹמָא תְּמִינָאָה, יַיְתֵי תְּרֵין שַׁפְנִינִין, אוֹ תְּרֵין, בְּנֵי יוֹנָה:  לְוָת כָּהֲנָא--לִתְרַע, מַשְׁכַּן זִמְנָא. 
ו,יא וְעָשָׂ֣ה הַכֹּהֵ֗ן אֶחָ֤ד לְחַטָּאת֙ וְאֶחָ֣ד לְעֹלָ֔ה וְכִפֶּ֣ר עָלָ֔יו מֵֽאֲשֶׁ֥ר חָטָ֖א עַל־הַנָּ֑פֶשׁ וְקִדַּ֥שׁ אֶת־רֹאשׁ֖וֹ בַּיּ֥וֹם הַהֽוּא׃ וְיַעֲבֵיד כָּהֲנָא, חַד לְחַטָּתָא וְחַד לַעֲלָתָא, וִיכַפַּר עֲלוֹהִי, מִדְּחָב עַל מִיתָא; וִיקַדֵּישׁ יָת רֵישֵׁיהּ, בְּיוֹמָא הַהוּא. 
ו,יב וְהִזִּ֤יר לַֽיהוָה֙ אֶת־יְמֵ֣י נִזְר֔וֹ וְהֵבִ֛יא כֶּ֥בֶשׂ בֶּן־שְׁנָת֖וֹ לְאָשָׁ֑ם וְהַיָּמִ֤ים הָרִֽאשֹׁנִים֙ יִפְּל֔וּ כִּ֥י טָמֵ֖א נִזְרֽוֹ׃ וְיִזַּר קֳדָם יְיָ יָת יוֹמֵי נִזְרֵיהּ, וְיַיְתֵי אִמַּר בַּר שַׁתֵּיהּ לַאֲשָׁמָא; וְיוֹמַיָּא קַדְמָאֵי יִבְטְלוּן, אֲרֵי אִסְתָּאַב נִזְרֵיהּ. 
ו,יג וְזֹ֥את תּוֹרַ֖ת הַנָּזִ֑יר בְּי֗וֹם מְלֹאת֙ יְמֵ֣י נִזְר֔וֹ יָבִ֣יא אֹת֔וֹ אֶל־פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ וְדָא אוֹרָיְתָא, דִּנְזִירָא:  בְּיוֹם, מִשְׁלַם יוֹמֵי נִזְרֵיהּ, יַיְתֵי יָתֵיהּ, לִתְרַע מַשְׁכַּן זִמְנָא. 
ו,יד וְהִקְרִ֣יב אֶת־קָרְבָּנ֣וֹ לַֽיהוָ֡ה כֶּבֶשׂ֩ בֶּן־שְׁנָת֨וֹ תָמִ֤ים אֶחָד֙ לְעֹלָ֔ה וְכַבְשָׂ֨ה אַחַ֧ת בַּת־שְׁנָתָ֛הּ תְּמִימָ֖ה לְחַטָּ֑את וְאַֽיִל־אֶחָ֥ד תָּמִ֖ים לִשְׁלָמִֽים׃ וִיקָרֵיב יָת קֻרְבָּנֵיהּ קֳדָם יְיָ אִמַּר בַּר שַׁתֵּיהּ שְׁלִים חַד, לַעֲלָתָא, וְאִמַּרְתָּא חֲדָא בַּת שַׁתַּהּ שַׁלְמְתָא, לְחַטָּתָא; וּדְכַר חַד שְׁלִים, לְנִכְסַת קֻדְשַׁיָּא. 
ו,טו וְסַ֣ל מַצּ֗וֹת סֹ֤לֶת חַלֹּת֙ בְּלוּלֹ֣ת בַּשֶּׁ֔מֶן וּרְקִיקֵ֥י מַצּ֖וֹת מְשֻׁחִ֣ים בַּשָּׁ֑מֶן וּמִנְחָתָ֖ם וְנִסְכֵּיהֶֽם׃ וְסַל פַּטִּיר, סֹלֶת גְּרִיצָן דְּפִילָן בִּמְשַׁח, וְאֶסְפּוֹגִין פַּטִּירִין, דִּמְשִׁיחִין בִּמְשַׁח; וּמִנְחָתְהוֹן, וְנִסְכֵּיהוֹן. 
ו,טז וְהִקְרִ֥יב הַכֹּהֵ֖ן לִפְנֵ֣י יְהוָ֑ה וְעָשָׂ֥ה אֶת־חַטָּאת֖וֹ וְאֶת־עֹֽלָתֽוֹ׃ וִיקָרֵיב כָּהֲנָא, קֳדָם יְיָ; וְיַעֲבֵיד יָת חַטָּתֵיהּ, וְיָת עֲלָתֵיהּ. 
ו,יז וְאֶת־הָאַ֜יִל יַֽעֲשֶׂ֨ה זֶ֤בַח שְׁלָמִים֙ לַֽיהוָ֔ה עַ֖ל סַ֣ל הַמַּצּ֑וֹת וְעָשָׂה֙ הַכֹּהֵ֔ן אֶת־מִנְחָת֖וֹ וְאֶת־נִסְכּֽוֹ׃ וְיָת דִּכְרָא יַעֲבֵיד נִכְסַת קֻדְשַׁיָּא, קֳדָם יְיָ, עַל, סַלָּא דְּפַטִּירַיָּא; וְיַעֲבֵיד, כָּהֲנָא, יָת מִנְחָתֵיהּ, וְיָת נִסְכֵּיהּ. 
ו,יח וְגִלַּ֣ח הַנָּזִ֗יר פֶּ֛תַח אֹ֥הֶל מוֹעֵ֖ד אֶת־רֹ֣אשׁ נִזְר֑וֹ וְלָקַ֗ח אֶת־שְׂעַר֙ רֹ֣אשׁ נִזְר֔וֹ וְנָתַן֙ עַל־הָאֵ֔שׁ אֲשֶׁר־תַּ֖חַת זֶ֥בַח הַשְּׁלָמִֽים׃ וִיגַלַּח נְזִירָא, בִּתְרַע מַשְׁכַּן זִמְנָא--יָת רֵישׁ נִזְרֵיהּ; וְיִסַּב, יָת סְעַר רֵישׁ נִזְרֵיהּ, וְיִתֵּין עַל אִישָׁתָא, דִּתְחוֹת דּוּדָא דְּנִכְסַת קֻדְשַׁיָּא. 
ו,יט וְלָקַ֨ח הַכֹּהֵ֜ן אֶת־הַזְּרֹ֣עַ בְּשֵׁלָה֮ מִן־הָאַיִל֒ וְֽחַלַּ֨ת מַצָּ֤ה אַחַת֙ מִן־הַסַּ֔ל וּרְקִ֥יק מַצָּ֖ה אֶחָ֑ד וְנָתַן֙ עַל־כַּפֵּ֣י הַנָּזִ֔יר אַחַ֖ר הִֽתְגַּלְּח֥וֹ אֶת־נִזְרֽוֹ׃ וְיִסַּב כָּהֲנָא יָת דְּרָעָא בְּשֵׁילָא, מִן דִּכְרָא, וּגְרִיצְתָא פַּטִּירְתָא חֲדָא מִן סַלָּא, וְאֶסְפּוֹג פַּטִּיר חַד; וְיִתֵּין עַל יְדֵי נְזִירָא, בָּתַר דִּיגַלַּח יָת נִזְרֵיהּ. 
ו,כ וְהֵנִיף֩ אוֹתָ֨ם הַכֹּהֵ֥ן ׀ תְּנוּפָה֮ לִפְנֵ֣י יְהוָה֒ קֹ֤דֶשׁ הוּא֙ לַכֹּהֵ֔ן עַ֚ל חֲזֵ֣ה הַתְּנוּפָ֔ה וְעַ֖ל שׁ֣וֹק הַתְּרוּמָ֑ה וְאַחַ֛ר יִשְׁתֶּ֥ה הַנָּזִ֖יר יָֽיִן׃ וִירִים יָתְהוֹן כָּהֲנָא אֲרָמָא, קֳדָם יְיָ--קֻדְשָׁא הוּא לְכָהֲנָא, עַל חַדְיָא דַּאֲרָמוּתָא וְעַל שָׁקָא דְּאַפְרָשׁוּתָא; וּבָתַר כֵּין יִשְׁתֵּי נְזִירָא, חַמְרָא. 
ו,כא זֹ֣את תּוֹרַ֣ת הַנָּזִיר֮ אֲשֶׁ֣ר יִדֹּר֒ קָרְבָּנ֤וֹ לַֽיהוָה֙ עַל־נִזְר֔וֹ מִלְּבַ֖ד אֲשֶׁר־תַּשִּׂ֣יג יָד֑וֹ כְּפִ֤י נִדְרוֹ֙ אֲשֶׁ֣ר יִדֹּ֔ר כֵּ֣ן יַֽעֲשֶׂ֔ה עַ֖ל תּוֹרַ֥ת נִזְרֽוֹ׃ {פ} דָּא אוֹרָיְתָא דִּנְזִירָא, דְּיִדַּר, קֻרְבָּנֵיהּ קֳדָם יְיָ עַל נִזְרֵיהּ, בָּר מִדְּתַדְבֵּיק יְדֵיהּ; כְּפֹם נִדְרֵיהּ, דְּיִדַּר--כֵּין יַעֲבֵיד, עַל אוֹרָיְתָא דְּנִזְרֵיהּ. {פ}
ו,כב וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ו,כג דַּבֵּ֤ר אֶֽל־אַהֲרֹן֙ וְאֶל־בָּנָ֣יו לֵאמֹ֔ר כֹּ֥ה תְבָֽרְכ֖וּ אֶת־בְּנֵ֣י יִשְׂרָאֵ֑ל אָמ֖וֹר לָהֶֽם׃ {ס} מַלֵּיל עִם אַהֲרוֹן וְעִם בְּנוֹהִי לְמֵימַר, כְּדֵין תְּבָרְכוּן יָת בְּנֵי יִשְׂרָאֵל:  כַּד תֵּימְרוּן, לְהוֹן. {ס}  (בכתבי יד תימן עתיקים אין ברכת כוהנים בתרגום אונקלוס: 
ו,כד יְבָֽרֶכְךָ֥ יְהוָ֖ה וְיִשְׁמְרֶֽךָ׃ {ס} יְבָרְכִנָּךְ יְיָ, וְיִטְּרִנָּךְ. {ס}
ו,כה יָאֵ֨ר יְהוָ֧ה ׀ פָּנָ֛יו אֵלֶ֖יךָ וִֽיחֻנֶּֽךָּ׃ {ס} יַנְהַר יְיָ שְׁכִינְתֵיהּ לְוָתָךְ, וִירַחֵים יָתָךְ. {ס}
ו,כו יִשָּׂ֨א יְהוָ֤ה ׀ פָּנָיו֙ אֵלֶ֔יךָ וְיָשֵׂ֥ם לְךָ֖ שָׁלֽוֹם׃ {ס} יִסַּב יְיָ אַפּוֹהִי לְוָתָךְ, וִישַׁוֵּי לָךְ שְׁלָם. {ס} ) 
ו,כז וְשָׂמ֥וּ אֶת־שְׁמִ֖י עַל־בְּנֵ֣י יִשְׂרָאֵ֑ל וַֽאֲנִ֖י אֲבָֽרְכֵֽם׃ {ס} וִישַׁוּוֹן יָת בִּרְכַּת שְׁמִי, עַל בְּנֵי יִשְׂרָאֵל; וַאֲנָא, אֲבָרֵיכִנּוּן. {ס}
ז,א וַיְהִ֡י בְּיוֹם֩ כַּלּ֨וֹת מֹשֶׁ֜ה לְהָקִ֣ים אֶת־הַמִּשְׁכָּ֗ן וַיִּמְשַׁ֨ח אֹת֜וֹ וַיְקַדֵּ֤שׁ אֹתוֹ֙ וְאֶת־כָּל־כֵּלָ֔יו וְאֶת־הַמִּזְבֵּ֖חַ וְאֶת־כָּל־כֵּלָ֑יו וַיִּמְשָׁחֵ֖ם וַיְקַדֵּ֥שׁ אֹתָֽם׃ וַהֲוָה בְּיוֹמָא דְּשֵׁיצֵי מֹשֶׁה לְאָקָמָא יָת מַשְׁכְּנָא, וְרַבִּי יָתֵיהּ וְקַדֵּישׁ יָתֵיהּ וְיָת כָּל מָנוֹהִי, וְיָת מַדְבְּחָא, וְיָת כָּל מָנוֹהִי; וְרַבִּנּוּן, וְקַדֵּישׁ יָתְהוֹן. 
ז,ב וַיַּקְרִ֨יבוּ֙ נְשִׂיאֵ֣י יִשְׂרָאֵ֔ל רָאשֵׁ֖י בֵּ֣ית אֲבֹתָ֑ם הֵ֚ם נְשִׂיאֵ֣י הַמַּטֹּ֔ת הֵ֥ם הָעֹֽמְדִ֖ים עַל־הַפְּקֻדִֽים׃ וְקָרִיבוּ רַבְרְבֵי יִשְׂרָאֵל, רֵישֵׁי בֵּית אֲבָהָתְהוֹן:  אִנּוּן רַבְרְבֵי שִׁבְטַיָּא, אִנּוּן דְּקָיְמִין עַל מִנְיָנַיָּא. 
ז,ג וַיָּבִ֨יאוּ אֶת־קָרְבָּנָ֜ם לִפְנֵ֣י יְהוָ֗ה שֵׁשׁ־עֶגְלֹ֥ת צָב֙ וּשְׁנֵ֣י עָשָׂ֣ר בָּקָ֔ר עֲגָלָ֛ה עַל־שְׁנֵ֥י הַנְּשִׂאִ֖ים וְשׁ֣וֹר לְאֶחָ֑ד וַיַּקְרִ֥יבוּ אוֹתָ֖ם לִפְנֵ֥י הַמִּשְׁכָּֽן׃ וְאֵיתִיאוּ יָת קֻרְבָּנְהוֹן לִקְדָם יְיָ, שֵׁית עֶגְלָן כַּד מְחַפְּיָן וּתְרֵי עֲסַר תּוֹרִין--עֲגַלְתָּא עַל תְּרֵין רַבְרְבַיָּא, וְתוֹרָא לְחַד; וְקָרִיבוּ יָתְהוֹן, לִקְדָם מַשְׁכְּנָא. 
ז,ד וַיֹּ֥אמֶר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וַאֲמַר יְיָ, לְמֹשֶׁה לְמֵימַר. 
ז,ה קַ֚ח מֵֽאִתָּ֔ם וְהָי֕וּ לַֽעֲבֹ֕ד אֶת־עֲבֹדַ֖ת אֹ֣הֶל מוֹעֵ֑ד וְנָֽתַתָּ֤ה אוֹתָם֙ אֶל־הַלְוִיִּ֔ם אִ֖ישׁ כְּפִ֥י עֲבֹֽדָתֽוֹ׃ קַבֵּיל, מִנְּהוֹן, וִיהוֹן, לְמִפְלַח יָת פֻּלְחַן מַשְׁכַּן זִמְנָא; וְתִתֵּין יָתְהוֹן לְלֵיוָאֵי, גְּבַר כְּמִסַּת פֻּלְחָנֵיהּ. 
ז,ו וַיִּקַּ֣ח מֹשֶׁ֔ה אֶת־הָֽעֲגָלֹ֖ת וְאֶת־הַבָּקָ֑ר וַיִּתֵּ֥ן אוֹתָ֖ם אֶל־הַלְוִיִּֽם׃ וּנְסֵיב מֹשֶׁה, יָת עֶגְלָתָא וְיָת תּוֹרֵי; וִיהַב יָתְהוֹן, לְלֵיוָאֵי. 
ז,ז אֵ֣ת ׀ שְׁתֵּ֣י הָֽעֲגָל֗וֹת וְאֵת֙ אַרְבַּ֣עַת הַבָּקָ֔ר נָתַ֖ן לִבְנֵ֣י גֵֽרְשׁ֑וֹן כְּפִ֖י עֲבֹֽדָתָֽם׃ יָת תַּרְתֵּין עֶגְלָן, וְיָת אַרְבְּעָא תּוֹרִין--יְהַב, לִבְנֵי גֵּרְשׁוֹן:  כְּמִסַּת, פֻּלְחָנְהוֹן. 
ז,ח וְאֵ֣ת ׀ אַרְבַּ֣ע הָֽעֲגָלֹ֗ת וְאֵת֙ שְׁמֹנַ֣ת הַבָּקָ֔ר נָתַ֖ן לִבְנֵ֣י מְרָרִ֑י כְּפִי֙ עֲבֹ֣דָתָ֔ם בְּיַד֙ אִֽיתָמָ֔ר בֶּֽן־אַהֲרֹ֖ן הַכֹּהֵֽן׃ וְיָת אַרְבַּע עֶגְלָן, וְיָת תְּמָנְיָא תּוֹרִין--יְהַב, לִבְנֵי מְרָרִי:  כְּמִסַּת, פֻּלְחָנְהוֹן, בִּידָא דְּאִיתָמָר, בַּר אַהֲרוֹן כָּהֲנָא. 
ז,ט וְלִבְנֵ֥י קְהָ֖ת לֹ֣א נָתָ֑ן כִּֽי־עֲבֹדַ֤ת הַקֹּ֨דֶשׁ֙ עֲלֵהֶ֔ם בַּכָּתֵ֖ף יִשָּֽׂאוּ׃ וְלִבְנֵי קְהָת, לָא יְהַב:  אֲרֵי פֻּלְחַן קֻדְשָׁא עֲלֵיהוֹן, בְּכַתְפָּא נָטְלִין. 
ז,י וַיַּקְרִ֣יבוּ הַנְּשִׂאִ֗ים אֵ֚ת חֲנֻכַּ֣ת הַמִּזְבֵּ֔חַ בְּי֖וֹם הִמָּשַׁ֣ח אֹת֑וֹ וַיַּקְרִ֧יבוּ הַנְּשִׂיאִ֛ם אֶת־קָרְבָּנָ֖ם לִפְנֵ֥י הַמִּזְבֵּֽחַ׃ וְקָרִיבוּ רַבְרְבַיָּא, יָת חֲנֻכַּת מַדְבְּחָא, בְּיוֹמָא, דְּרַבִּיאוּ יָתֵיהּ; וְקָרִיבוּ רַבְרְבַיָּא יָת קֻרְבָּנְהוֹן, לִקְדָם מַדְבְּחָא. 
ז,יא וַיֹּ֥אמֶר יְהוָ֖ה אֶל־מֹשֶׁ֑ה נָשִׂ֨יא אֶחָ֜ד לַיּ֗וֹם נָשִׂ֤יא אֶחָד֙ לַיּ֔וֹם יַקְרִ֨יבוּ֙ אֶת־קָרְבָּנָ֔ם לַֽחֲנֻכַּ֖ת הַמִּזְבֵּֽחַ׃ {ס} וַאֲמַר יְיָ, לְמֹשֶׁה:  רַבָּא חַד לְיוֹמָא, רַבָּא חַד לְיוֹמָא, יְקָרְבוּן יָת קֻרְבָּנְהוֹן, לַחֲנֻכַּת מַדְבְּחָא. {ס}
ז,יב וַיְהִ֗י הַמַּקְרִ֛יב בַּיּ֥וֹם הָֽרִאשׁ֖וֹן אֶת־קָרְבָּנ֑וֹ נַחְשׁ֥וֹן בֶּן־עַמִּֽינָדָ֖ב לְמַטֵּ֥ה יְהוּדָֽה׃ וַהֲוָה, דִּמְקָרֵיב בְּיוֹמָא קַדְמָאָה--יָת קֻרְבָּנֵיהּ:  נַחְשׁוֹן בַּר עַמִּינָדָב, לְשִׁבְטָא דִּיהוּדָה. 
ז,יג וְקָרְבָּנ֞וֹ קַֽעֲרַת־כֶּ֣סֶף אַחַ֗ת שְׁלֹשִׁ֣ים וּמֵאָה֮ מִשְׁקָלָהּ֒ מִזְרָ֤ק אֶחָד֙ כֶּ֔סֶף שִׁבְעִ֥ים שֶׁ֖קֶל בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ שְׁנֵיהֶ֣ם ׀ מְלֵאִ֗ים סֹ֛לֶת בְּלוּלָ֥ה בַשֶּׁ֖מֶן לְמִנְחָֽה׃ וְקֻרְבָּנֵיהּ מְגִסְּתָא דִּכְסַף חֲדָא, מְאָה וּתְלָתִין סִלְעִין הָוֵי מַתְקָלַהּ, מִזְרְקָא חַד דְּכַסְפָּא, מַתְקָלֵיהּ שִׁבְעִין סִלְעִין בְּסִלְעֵי קֻדְשָׁא; תַּרְוֵיהוֹן מְלַן, סֻלְתָּא דְּפִילָא בִּמְשַׁח--לְמִנְחָתָא. 
ז,יד כַּ֥ף אַחַ֛ת עֲשָׂרָ֥ה זָהָ֖ב מְלֵאָ֥ה קְטֹֽרֶת׃ בָּזִכָּא חֲדָא מַתְקָלַהּ עֲסַר סִלְעִין הִיא דִּדְהַב, מָלְיָא קְטֹרֶת בֻּסְמַיָּא. 
ז,טו פַּ֣ר אֶחָ֞ד בֶּן־בָּקָ֗ר אַ֧יִל אֶחָ֛ד כֶּֽבֶשׂ־אֶחָ֥ד בֶּן־שְׁנָת֖וֹ לְעֹלָֽה׃ תּוֹר חַד בַּר תּוֹרֵי, דְּכַר חַד אִמַּר חַד בַּר שַׁתֵּיהּ--לַעֲלָתָא. 
ז,טז שְׂעִיר־עִזִּ֥ים אֶחָ֖ד לְחַטָּֽאת׃ צְפִיר בַּר עִזִּין חַד, לְחַטָּתָא. 
ז,יז וּלְזֶ֣בַח הַשְּׁלָמִים֮ בָּקָ֣ר שְׁנַיִם֒ אֵילִ֤ם חֲמִשָּׁה֙ עַתּוּדִ֣ים חֲמִשָּׁ֔ה כְּבָשִׂ֥ים בְּנֵֽי־שָׁנָ֖ה חֲמִשָּׁ֑ה זֶ֛ה קָרְבַּ֥ן נַחְשׁ֖וֹן בֶּן־עַמִּֽינָדָֽב׃ {פ} וּלְנִכְסַת קֻדְשַׁיָּא, תּוֹרֵי תְּרֵין, דִּכְרֵי חַמְשָׁא גְּדֵי חַמְשָׁא, אִמְּרִין בְּנֵי שְׁנָא חַמְשָׁא:  דֵּין קֻרְבָּנָא דְּנַחְשׁוֹן, בַּר עַמִּינָדָב. {פ}
ז,יח בַּיּוֹם֙ הַשֵּׁנִ֔י הִקְרִ֖יב נְתַנְאֵ֣ל בֶּן־צוּעָ֑ר נְשִׂ֖יא יִשָּׂשכָֽר׃ בְּיוֹמָא, תִּנְיָנָא, קָרֵיב, נְתַנְאֵל בַּר צוּעָר--רַב, שִׁבְטָא דְּיִשָּׂשכָר.

ז,עב בְּיוֹם֙ עַשְׁתֵּ֣י עָשָׂ֣ר י֔וֹם נָשִׂ֖יא לִבְנֵ֣י אָשֵׁ֑ר פַּגְעִיאֵ֖ל בֶּן־עָכְרָֽן׃ בְּיוֹם חַד עֲסַר יוֹמִין, רַבָּא לִבְנֵי אָשֵׁר--פַּגְעִיאֵל, בַּר עָכְרָן.

ז,עז וּלְזֶ֣בַח הַשְּׁלָמִים֮ בָּקָ֣ר שְׁנַיִם֒ אֵילִ֤ם חֲמִשָּׁה֙ עַתֻּדִ֣ים חֲמִשָּׁ֔ה כְּבָשִׂ֥ים בְּנֵֽי־שָׁנָ֖ה חֲמִשָּׁ֑ה זֶ֛ה קָרְבַּ֥ן פַּגְעִיאֵ֖ל בֶּן־עָכְרָֽן׃ {פ} וּלְנִכְסַת קֻדְשַׁיָּא, תּוֹרֵי תְּרֵין, דִּכְרֵי חַמְשָׁא גְּדֵי חַמְשָׁא, אִמְּרִין בְּנֵי שְׁנָא חַמְשָׁא:  דֵּין קֻרְבָּנָא דְּפַגְעִיאֵל, בַּר עָכְרָן. {פ}
ז,עח בְּיוֹם֙ שְׁנֵ֣ים עָשָׂ֣ר י֔וֹם נָשִׂ֖יא לִבְנֵ֣י נַפְתָּלִ֑י אֲחִירַ֖ע בֶּן־עֵינָֽן׃ בְּיוֹם תְּרֵי עֲסַר יוֹמִין, רַבָּא לִבְנֵי נַפְתָּלִי--אֲחִירַע, בַּר עֵינָן. 
ז,עט קָרְבָּנ֞וֹ קַֽעֲרַת־כֶּ֣סֶף אַחַ֗ת שְׁלֹשִׁ֣ים וּמֵאָה֮ מִשְׁקָלָהּ֒ מִזְרָ֤ק אֶחָד֙ כֶּ֔סֶף שִׁבְעִ֥ים שֶׁ֖קֶל בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ שְׁנֵיהֶ֣ם ׀ מְלֵאִ֗ים סֹ֛לֶת בְּלוּלָ֥ה בַשֶּׁ֖מֶן לְמִנְחָֽה׃ קֻרְבָּנֵיהּ מְגִסְּתָא דִּכְסַף חֲדָא, מְאָה וּתְלָתִין סִלְעִין הָוֵי מַתְקָלַהּ, מִזְרְקָא חַד דְּכַסְפָּא, מַתְקָלֵיהּ שִׁבְעִין סִלְעִין בְּסִלְעֵי קֻדְשָׁא; תַּרְוֵיהוֹן מְלַן, סֻלְתָּא דְּפִילָא בִּמְשַׁח--לְמִנְחָתָא.

ז,פג וּלְזֶ֣בַח הַשְּׁלָמִים֮ בָּקָ֣ר שְׁנַיִם֒ אֵילִ֤ם חֲמִשָּׁה֙ עַתֻּדִ֣ים חֲמִשָּׁ֔ה כְּבָשִׂ֥ים בְּנֵֽי־שָׁנָ֖ה חֲמִשָּׁ֑ה זֶ֛ה קָרְבַּ֥ן אֲחִירַ֖ע בֶּן־עֵינָֽן׃ {פ} וּלְנִכְסַת קֻדְשַׁיָּא, תּוֹרֵי תְּרֵין, דִּכְרֵי חַמְשָׁא גְּדֵי חַמְשָׁא, אִמְּרִין בְּנֵי שְׁנָא חַמְשָׁא:  דֵּין קֻרְבָּנָא דַּאֲחִירַע, בַּר עֵינָן. {פ}
ז,פד זֹ֣את ׀ חֲנֻכַּ֣ת הַמִּזְבֵּ֗חַ בְּיוֹם֙ הִמָּשַׁ֣ח אֹת֔וֹ מֵאֵ֖ת נְשִׂיאֵ֣י יִשְׂרָאֵ֑ל קַֽעֲרֹ֨ת כֶּ֜סֶף שְׁתֵּ֣ים עֶשְׂרֵ֗ה מִֽזְרְקֵי־כֶ֨סֶף֙ שְׁנֵ֣ים עָשָׂ֔ר כַּפּ֥וֹת זָהָ֖ב שְׁתֵּ֥ים עֶשְׂרֵֽה׃ דָּא חֲנֻכַּת מַדְבְּחָא, בְּיוֹמָא דְּרַבִּיאוּ יָתֵיהּ, מִן, רַבְרְבֵי יִשְׂרָאֵל:  מְגִסֵּי כַּסְפָּא תַּרְתַּא עַסְרֵי, מִזְרְקֵי כַּסְפָּא תְּרֵי עֲסַר, בָּזִכֵּי דְּדַהְבָּא, תַּרְתַּא עַסְרֵי. 
ז,פה שְׁלֹשִׁ֣ים וּמֵאָ֗ה הַקְּעָרָ֤ה הָֽאַחַת֙ כֶּ֔סֶף וְשִׁבְעִ֖ים הַמִּזְרָ֣ק הָֽאֶחָ֑ד כֹּ֚ל כֶּ֣סֶף הַכֵּלִ֔ים אַלְפַּ֥יִם וְאַרְבַּע־מֵא֖וֹת בְּשֶׁ֥קֶל הַקֹּֽדֶשׁ׃ מְאָה וּתְלָתִין סִלְעִין, הָוֵי מַתְקָלַהּ דִּמְגִסְּתָא חֲדָא דְּכַסְפָּא, וְשִׁבְעִין, דְּמִזְרְקָא חַד; כֹּל כְּסַף מָנַיָּא, תְּרֵין אַלְפִין וְאַרְבַּע מְאָה בְּסִלְעֵי קֻדְשָׁא. 
ז,פו כַּפּ֨וֹת זָהָ֤ב שְׁתֵּים־עֶשְׂרֵה֙ מְלֵאֹ֣ת קְטֹ֔רֶת עֲשָׂרָ֧ה עֲשָׂרָ֛ה הַכַּ֖ף בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ כָּל־זְהַ֥ב הַכַּפּ֖וֹת עֶשְׂרִ֥ים וּמֵאָֽה׃ בָּזִכֵּי דְּדַהְבָּא תַּרְתַּא עַסְרֵי מָלְיָן קְטֹרֶת בֻּסְמַיָּא, מַתְקַל עֲסַר סִלְעִין הָוֵי מַתְקָלַהּ דְּבָזִכָּא בְּסִלְעֵי קֻדְשָׁא; כָּל דְּהַב בָּזִכַּיָּא, מְאָה וְעַסְרִין. 
ז,פז כָּל־הַבָּקָ֨ר לָֽעֹלָ֜ה שְׁנֵ֧ים עָשָׂ֣ר פָּרִ֗ים אֵילִ֤ם שְׁנֵים־עָשָׂר֙ כְּבָשִׂ֧ים בְּנֵֽי־שָׁנָ֛ה שְׁנֵ֥ים עָשָׂ֖ר וּמִנְחָתָ֑ם וּשְׂעִירֵ֥י עִזִּ֛ים שְׁנֵ֥ים עָשָׂ֖ר לְחַטָּֽאת׃ כָּל תּוֹרֵי לַעֲלָתָא תְּרֵי עֲסַר תּוֹרִין, דִּכְרִין תְּרֵי עֲסַר אִמְּרִין בְּנֵי שְׁנָא תְּרֵי עֲסַר--וּמִנְחָתְהוֹן; וּצְפִירֵי בְּנֵי עִזֵּי תְּרֵי עֲסַר, לְחַטָּתָא. 
ז,פח וְכֹ֞ל בְּקַ֣ר ׀ זֶ֣בַח הַשְּׁלָמִ֗ים עֶשְׂרִ֣ים וְאַרְבָּעָה֮ פָּרִים֒ אֵילִ֤ם שִׁשִּׁים֙ עַתֻּדִ֣ים שִׁשִּׁ֔ים כְּבָשִׂ֥ים בְּנֵֽי־שָׁנָ֖ה שִׁשִּׁ֑ים זֹ֚את חֲנֻכַּ֣ת הַמִּזְבֵּ֔חַ אַֽחֲרֵ֖י הִמָּשַׁ֥ח אֹתֽוֹ׃ וְכֹל תּוֹרֵי לְנִכְסַת קֻדְשַׁיָּא, עַסְרִין וְאַרְבְּעָא תּוֹרִין, דִּכְרֵי שִׁתִּין גְּדֵי שִׁתִּין, אִמְּרִין בְּנֵי שְׁנָא שִׁתִּין:  דָּא חֲנֻכַּת מַדְבְּחָא, בָּתַר דְּרַבִּיאוּ יָתֵיהּ. 
ז,פט וּבְבֹ֨א מֹשֶׁ֜ה אֶל־אֹ֣הֶל מוֹעֵד֮ לְדַבֵּ֣ר אִתּוֹ֒ וַיִּשְׁמַ֨ע אֶת־הַקּ֜וֹל מִדַּבֵּ֣ר אֵלָ֗יו מֵעַ֤ל הַכַּפֹּ֨רֶת֙ אֲשֶׁר֙ עַל־אֲרֹ֣ן הָֽעֵדֻ֔ת מִבֵּ֖ין שְׁנֵ֣י הַכְּרֻבִ֑ים וַיְדַבֵּ֖ר אֵלָֽיו׃ {פ} וְכַד עָלֵיל מֹשֶׁה לְמַשְׁכַּן זִמְנָא, לְמַלָּלָא עִמֵּיהּ, וְשָׁמַע יָת קָלָא דְּמִתְמַלַּל עִמֵּיהּ מֵעִלָּוֵי כָּפֻרְתָּא דְּעַל אֲרוֹנָא דְּסָהֲדוּתָא, מִבֵּין תְּרֵין כְּרוּבַיָּא; וּמִתְמַלַּל, עִמֵּיהּ. {פ}
