שופטים

טז,יח שֹֽׁפְטִ֣ים וְשֹֽׁטְרִ֗ים תִּֽתֶּן־לְךָ֙ בְּכָל־שְׁעָרֶ֔יךָ אֲשֶׁ֨ר יְהוָ֧ה אֱלֹהֶ֛יךָ נֹתֵ֥ן לְךָ֖ לִשְׁבָטֶ֑יךָ וְשָֽׁפְט֥וּ אֶת־הָעָ֖ם מִשְׁפַּט־צֶֽדֶק׃ דַּיָּנִין וּפֻרְעָנִין, תְּמַנֵּי לָךְ בְּכָל קִרְוָךְ, דַּייָ אֱלָהָךְ יָהֵיב לָךְ, לְשִׁבְטָךְ; וִידִינוּן יָת עַמָּא, דִּין דִּקְשׁוֹט. 
טז,יט לֹֽא־תַטֶּ֣ה מִשְׁפָּ֔ט לֹ֥א תַכִּ֖יר פָּנִ֑ים וְלֹֽא־תִקַּ֣ח שֹׁ֔חַד כִּ֣י הַשֹּׁ֗חַד יְעַוֵּר֙ עֵינֵ֣י חֲכָמִ֔ים וִֽיסַלֵּ֖ף דִּבְרֵ֥י צַדִּיקִֽם׃ לָא תַּצְלֵי דִּין, לָא תִּשְׁתְּמוֹדַע אַפִּין; וְלָא תְּקַבֵּיל שֻׁחְדָּא--אֲרֵי שֻׁחְדָּא מְעַוַּר עֵינֵי חַכִּימִין, וּמְקַלְקֵיל פִּתְגָמִין תְּרִיצִין. 
טז,כ צֶ֥דֶק צֶ֖דֶק תִּרְדֹּ֑ף לְמַ֤עַן תִּֽחְיֶה֙ וְיָֽרַשְׁתָּ֣ אֶת־הָאָ֔רֶץ אֲשֶׁר־יְהוָ֥ה אֱלֹהֶ֖יךָ נֹתֵ֥ן לָֽךְ׃ {ס} קֻשְׁטָא קֻשְׁטָא, תִּרְדוֹף--בְּדִיל דְּתֵיחֵי וְתֵירַת יָת אַרְעָא, דַּייָ אֱלָהָךְ יָהֵיב לָךְ. {ס}
טז,כא לֹֽא־תִטַּ֥ע לְךָ֛ אֲשֵׁרָ֖ה כָּל־עֵ֑ץ אֵ֗צֶל מִזְבַּ֛ח יְהוָ֥ה אֱלֹהֶ֖יךָ אֲשֶׁ֥ר תַּֽעֲשֶׂה־לָּֽךְ׃ לָא תִּצּוּב לָךְ אֲשֵׁירַת, כָּל אִילָן:  בִּסְטַר, מַדְבְּחָא דַּייָ אֱלָהָךְ--דְּתַעֲבֵיד לָךְ. 
טז,כב וְלֹֽא־תָקִ֥ים לְךָ֖ מַצֵּבָ֑ה אֲשֶׁ֥ר שָׂנֵ֖א יְהוָ֥ה אֱלֹהֶֽיךָ׃ {ס} וְלָא תְּקִים לָךְ, קָמָא, דְּרַחֵיק, יְיָ אֱלָהָךְ. {ס}
יז,א לֹֽא־תִזְבַּח֩ לַֽיהוָ֨ה אֱלֹהֶ֜יךָ שׁ֣וֹר וָשֶׂ֗ה אֲשֶׁ֨ר יִֽהְיֶ֥ה בוֹ֙ מ֔וּם כֹּ֖ל דָּבָ֣ר רָ֑ע כִּ֧י תֽוֹעֲבַ֛ת יְהוָ֥ה אֱלֹהֶ֖יךָ הֽוּא׃ {ס} לָא תִּכּוֹס קֳדָם יְיָ אֱלָהָךְ תּוֹר וְאִמַּר, דִּיהֵי בֵּיהּ מוּמָא--כֹּל, מִדָּעַם בִּישׁ:  אֲרֵי מְרַחַק קֳדָם יְיָ אֱלָהָךְ, הוּא. {ס}
יז,ב כִּֽי־יִמָּצֵ֤א בְקִרְבְּךָ֙ בְּאַחַ֣ד שְׁעָרֶ֔יךָ אֲשֶׁר־יְהוָ֥ה אֱלֹהֶ֖יךָ נֹתֵ֣ן לָ֑ךְ אִ֣ישׁ אֽוֹ־אִשָּׁ֗ה אֲשֶׁ֨ר יַֽעֲשֶׂ֧ה אֶת־הָרַ֛ע בְּעֵינֵ֥י יְהוָֽה־אֱלֹהֶ֖יךָ לַֽעֲבֹ֥ר בְּרִיתֽוֹ׃ אֲרֵי יִשְׁתְּכַח בֵּינָךְ בַּחֲדָא מִן קִרְוָךְ, דַּייָ אֱלָהָךְ יָהֵיב לָךְ:  גְּבַר אוֹ אִתָּא, דְּיַעֲבֵיד יָת דְּבִישׁ קֳדָם יְיָ אֱלָהָךְ--לְמִעְבַּר עַל קְיָמֵיהּ. 
יז,ג וַיֵּ֗לֶךְ וַֽיַּעֲבֹד֙ אֱלֹהִ֣ים אֲחֵרִ֔ים וַיִּשְׁתַּ֖חוּ לָהֶ֑ם וְלַשֶּׁ֣מֶשׁ ׀ א֣וֹ לַיָּרֵ֗חַ א֛וֹ לְכָל־צְבָ֥א הַשָּׁמַ֖יִם אֲשֶׁ֥ר לֹֽא־צִוִּֽיתִי׃ וַאֲזַל, וּפְלַח לְטָעֲוָת עַמְמַיָּא, וּסְגֵיד, לְהוֹן; וּלְשִׁמְשָׁא אוֹ לְסֵיהֲרָא, אוֹ לְכָל חֵילֵי שְׁמַיָּא--דְּלָא פַּקֵּידִית. 
יז,ד וְהֻֽגַּד־לְךָ֖ וְשָׁמָ֑עְתָּ וְדָֽרַשְׁתָּ֣ הֵיטֵ֔ב וְהִנֵּ֤ה אֱמֶת֙ נָכ֣וֹן הַדָּבָ֔ר נֶֽעֶשְׂתָ֛ה הַתּֽוֹעֵבָ֥ה הַזֹּ֖את בְּיִשְׂרָאֵֽל׃ וְיִתְחַוַּא לָךְ, וְתִשְׁמַע; וְתִתְבַּע יָאוּת--וְהָא קֻשְׁטָא כֵּיוָן פִּתְגָמָא, אִתְעֲבֵידַת תּוֹעֵיבְתָא הָדָא בְּיִשְׂרָאֵל. 
יז,ה וְהֽוֹצֵאתָ֣ אֶת־הָאִ֣ישׁ הַה֡וּא אוֹ֩ אֶת־הָֽאִשָּׁ֨ה הַהִ֜וא אֲשֶׁ֣ר עָ֠שׂוּ אֶת־הַדָּבָ֨ר הָרָ֤ע הַזֶּה֙ אֶל־שְׁעָרֶ֔יךָ אֶת־הָאִ֕ישׁ א֖וֹ אֶת־הָֽאִשָּׁ֑ה וּסְקַלְתָּ֥ם בָּֽאֲבָנִ֖ים וָמֵֽתוּ׃ וְתַפֵּיק יָת גֻּבְרָא הַהוּא אוֹ יָת אִתְּתָא הַהִיא דַּעֲבַדוּ יָת פִּתְגָמָא בִּישָׁא הָדֵין, לִתְרַע בֵּית דִּינָךְ--יָת גֻּבְרָא, אוֹ יָת אִתְּתָא; וְתִרְגְּמִנּוּן בְּאַבְנַיָּא, וִימוּתוּן. 
יז,ו עַל־פִּ֣י ׀ שְׁנַ֣יִם עֵדִ֗ים א֛וֹ שְׁלֹשָׁ֥ה עֵדִ֖ים יוּמַ֣ת הַמֵּ֑ת לֹ֣א יוּמַ֔ת עַל־פִּ֖י עֵ֥ד אֶחָֽד׃ עַל מֵימַר תְּרֵין סָהֲדִין, אוֹ תְּלָתָא סָהֲדִין--יִתְקְטִיל דְּחַיָּב קְטוֹל:  לָא יִתְקְטִיל, עַל מֵימַר סָהִיד חַד. 
יז,ז יַ֣ד הָֽעֵדִ֞ים תִּֽהְיֶה־בּ֤וֹ בָרִֽאשֹׁנָה֙ לַֽהֲמִית֔וֹ וְיַ֥ד כָּל־הָעָ֖ם בָּאַֽחֲרֹנָ֑ה וּבִֽעַרְתָּ֥ הָרָ֖ע מִקִּרְבֶּֽךָ׃ {פ} יְדָא דְּסָהֲדַיָּא תְּהֵי בֵּיהּ בְּקַדְמֵיתָא, לְמִקְטְלֵיהּ, וִידָא דְּכָל עַמָּא, בְּבָתְרֵיתָא; וּתְפַלֵּי עָבֵיד דְּבִישׁ, מִבֵּינָךְ. {פ}
יז,ח כִּ֣י יִפָּלֵא֩ מִמְּךָ֨ דָבָ֜ר לַמִּשְׁפָּ֗ט בֵּֽין־דָּ֨ם ׀ לְדָ֜ם בֵּֽין־דִּ֣ין לְדִ֗ין וּבֵ֥ין נֶ֨גַע֙ לָנֶ֔גַע דִּבְרֵ֥י רִיבֹ֖ת בִּשְׁעָרֶ֑יךָ וְקַמְתָּ֣ וְעָלִ֔יתָ אֶ֨ל־הַמָּק֔וֹם אֲשֶׁ֥ר יִבְחַ֛ר יְהוָ֥ה אֱלֹהֶ֖יךָ בּֽוֹ׃ אֲרֵי יִתְפָּרַשׁ מִנָּךְ פִּתְגָמָא לְדִינָא, בֵּין דַּם לְדַם בֵּין דִּין לְדִין וּבֵין מַכְתָּשׁ סְגִירוּ לְמַכְתָּשׁ סְגִירוּ--פִּתְגָמֵי פְּלֻגַּת דִּינָא, בְּקִרְוָךְ:  וּתְקוּם וְתִסַּק--לְאַתְרָא, דְּיִתְרְעֵי יְיָ אֱלָהָךְ בֵּיהּ. 
יז,ט וּבָאתָ֗ אֶל־הַכֹּֽהֲנִים֙ הַלְוִיִּ֔ם וְאֶ֨ל־הַשֹּׁפֵ֔ט אֲשֶׁ֥ר יִֽהְיֶ֖ה בַּיָּמִ֣ים הָהֵ֑ם וְדָֽרַשְׁתָּ֙ וְהִגִּ֣ידֽוּ לְךָ֔ אֵ֖ת דְּבַ֥ר הַמִּשְׁפָּֽט׃ וְתֵיתֵי, לְוָת כָּהֲנַיָּא לֵיוָאֵי, וּלְוָת דַּיָּנָא, דִּיהֵי בְּיוֹמַיָּא הָאִנּוּן; וְתִתְבַּע וִיחַוּוֹן לָךְ, יָת פִּתְגָמָא דְּדִינָא. 
יז,י וְעָשִׂ֗יתָ עַל־פִּ֤י הַדָּבָר֙ אֲשֶׁ֣ר יַגִּ֣ידֽוּ לְךָ֔ מִן־הַמָּק֣וֹם הַה֔וּא אֲשֶׁ֖ר יִבְחַ֣ר יְהוָ֑ה וְשָֽׁמַרְתָּ֣ לַֽעֲשׂ֔וֹת כְּכֹ֖ל אֲשֶׁ֥ר יוֹרֽוּךָ׃ וְתַעֲבֵיד, עַל מֵימַר פִּתְגָמָא דִּיחַוּוֹן לָךְ, מִן אַתְרָא הַהוּא, דְּיִתְרְעֵי יְיָ; וְתִטַּר לְמֶעֱבַד, כְּכֹל דְּיַלְּפוּנָּךְ. 
יז,יא עַל־פִּ֨י הַתּוֹרָ֜ה אֲשֶׁ֣ר יוֹר֗וּךָ וְעַל־הַמִּשְׁפָּ֛ט אֲשֶׁר־יֹֽאמְר֥וּ לְךָ֖ תַּֽעֲשֶׂ֑ה לֹ֣א תָס֗וּר מִן־הַדָּבָ֛ר אֲשֶׁר־יַגִּ֥ידֽוּ לְךָ֖ יָמִ֥ין וּשְׂמֹֽאל׃ עַל מֵימַר אוֹרָיְתָא דְּיַלְּפוּנָּךְ, וְעַל דִּינָא דְּיֵימְרוּן לָךְ--תַּעֲבֵיד:  לָא תִּסְטֵי, מִן פִּתְגָמָא דִּיחַוּוֹן לָךְ--לְיַמִּינָא וְלִסְמָאלָא. 
יז,יב וְהָאִ֞ישׁ אֲשֶׁר־יַֽעֲשֶׂ֣ה בְזָד֗וֹן לְבִלְתִּ֨י שְׁמֹ֤עַ אֶל־הַכֹּהֵן֙ הָֽעֹמֵ֞ד לְשָׁ֤רֶת שָׁם֙ אֶת־יְהוָ֣ה אֱלֹהֶ֔יךָ א֖וֹ אֶל־הַשֹּׁפֵ֑ט וּמֵת֙ הָאִ֣ישׁ הַה֔וּא וּבִֽעַרְתָּ֥ הָרָ֖ע מִיִּשְׂרָאֵֽל׃ וְגֻבְרָא דְּיַעֲבֵיד בִּרְשַׁע, בְּדִיל דְּלָא לְקַבָּלָא מִן כָּהֲנָא דְּקָאֵים לְשַׁמָּשָׁא תַּמָּן קֳדָם יְיָ אֱלָהָךְ, אוֹ, מִן דַּיָּנָא--וְיִתְקְטִיל גֻּבְרָא הַהוּא, וּתְפַלֵּי עָבֵיד דְּבִישׁ מִיִּשְׂרָאֵל. 
יז,יג וְכָל־הָעָ֖ם יִשְׁמְע֣וּ וְיִרָ֑אוּ וְלֹ֥א יְזִיד֖וּן עֽוֹד׃ {ס} וְכָל עַמָּא, יִשְׁמְעוּן וְיִדְחֲלוּן; וְלָא יַרְשְׁעוּן, עוֹד. {ס}
יז,יד כִּֽי־תָבֹ֣א אֶל־הָאָ֗רֶץ אֲשֶׁ֨ר יְהוָ֤ה אֱלֹהֶ֨יךָ֙ נֹתֵ֣ן לָ֔ךְ וִֽירִשְׁתָּ֖הּ וְיָשַׁ֣בְתָּה בָּ֑הּ וְאָֽמַרְתָּ֗ אָשִׂ֤ימָה עָלַי֙ מֶ֔לֶךְ כְּכָל־הַגּוֹיִ֖ם אֲשֶׁ֥ר סְבִֽיבֹתָֽי׃ אֲרֵי תֵּיעוֹל לְאַרְעָא, דַּייָ אֱלָהָךְ יָהֵיב לָךְ, וְתֵירְתַהּ, וְתִתֵּיב בַּהּ; וְתֵימַר, אֲמַנֵּי עֲלַי מַלְכָּא, כְּכָל עַמְמַיָּא, דִּבְסַחְרָנָי. 
יז,טו שׂ֣וֹם תָּשִׂ֤ים עָלֶ֨יךָ֙ מֶ֔לֶךְ אֲשֶׁ֥ר יִבְחַ֛ר יְהוָ֥ה אֱלֹהֶ֖יךָ בּ֑וֹ מִקֶּ֣רֶב אַחֶ֗יךָ תָּשִׂ֤ים עָלֶ֨יךָ֙ מֶ֔לֶךְ לֹ֣א תוּכַ֗ל לָתֵ֤ת עָלֶ֨יךָ֙ אִ֣ישׁ נָכְרִ֔י אֲשֶׁ֥ר לֹֽא־אָחִ֖יךָ הֽוּא׃ מַנָּאָה תְּמַנֵּי עֲלָךְ מַלְכָּא, דְּיִתְרְעֵי יְיָ אֱלָהָךְ בֵּיהּ:  מִגּוֹ אֲחָךְ, תְּמַנֵּי עֲלָךְ מַלְכָּא--לֵית לָךְ רְשׁוּ לְמַנָּאָה עֲלָךְ גְּבַר נֻכְרַאי, דְּלָא אֲחוּךְ הוּא. 
יז,טז רַק֮ לֹֽא־יַרְבֶּה־לּ֣וֹ סוּסִים֒ וְלֹֽא־יָשִׁ֤יב אֶת־הָעָם֙ מִצְרַ֔יְמָה לְמַ֖עַן הַרְבּ֣וֹת ס֑וּס וַֽיהוָה֙ אָמַ֣ר לָכֶ֔ם לֹ֣א תֹֽסִפ֗וּן לָשׁ֛וּב בַּדֶּ֥רֶךְ הַזֶּ֖ה עֽוֹד׃ לְחוֹד, לָא יַסְגֵּי לֵיהּ סוּסָוָן, וְלָא יָתִיב יָת עַמָּא לְמִצְרַיִם, בְּדִיל לְאַסְגָּאָה סוּסָוָן; וַייָ, אֲמַר לְכוֹן, לָא תֵּיסְפוּן לִמְתָּב בְּאוֹרְחָא הָדָא, עוֹד. 
יז,יז וְלֹ֤א יַרְבֶּה־לּוֹ֙ נָשִׁ֔ים וְלֹ֥א יָס֖וּר לְבָב֑וֹ וְכֶ֣סֶף וְזָהָ֔ב לֹ֥א יַרְבֶּה־לּ֖וֹ מְאֹֽד׃ וְלָא יַסְגֵּי לֵיהּ נְשִׁין, וְלָא יִטְעֵי לִבֵּיהּ; וְכַסְפָּא וְדַהְבָּא, לָא יַסְגֵּי לֵיהּ לַחְדָּא. 
יז,יח וְהָיָ֣ה כְשִׁבְתּ֔וֹ עַ֖ל כִּסֵּ֣א מַמְלַכְתּ֑וֹ וְכָ֨תַב ל֜וֹ אֶת־מִשְׁנֵ֨ה הַתּוֹרָ֤ה הַזֹּאת֙ עַל־סֵ֔פֶר מִלִּפְנֵ֖י הַכֹּֽהֲנִ֥ים הַלְוִיִּֽם׃ וִיהֵי כְּמִתְּבֵיהּ, עַל כֻּרְסֵי מַלְכוּתֵיהּ--וְיִכְתּוֹב לֵיהּ יָת פַּרְשְׁגַן אוֹרָיְתָא הָדָא, עַל סִפְרָא, מִן קֳדָם, כָּהֲנַיָּא לֵיוָאֵי. 
יז,יט וְהָֽיְתָ֣ה עִמּ֔וֹ וְקָ֥רָא ב֖וֹ כָּל־יְמֵ֣י חַיָּ֑יו לְמַ֣עַן יִלְמַ֗ד לְיִרְאָה֙ אֶת־יְהוָ֣ה אֱלֹהָ֔יו לִ֠שְׁמֹר אֶֽת־כָּל־דִּבְרֵ֞י הַתּוֹרָ֥ה הַזֹּ֛את וְאֶת־הַֽחֻקִּ֥ים הָאֵ֖לֶּה לַֽעֲשֹׂתָֽם׃ וּתְהֵי עִמֵּיהּ, וִיהֵי קָרֵי בֵּיהּ כָּל יוֹמֵי חַיּוֹהִי--בְּדִיל דְּיֵילַף, לְמִדְחַל קֳדָם יְיָ אֱלָהֵיהּ, לְמִטַּר יָת כָּל פִּתְגָמֵי אוֹרָיְתָא הָדָא וְיָת קְיָמַיָּא הָאִלֵּין, לְמֶעֱבַדְהוֹן. 
יז,כ לְבִלְתִּ֤י רוּם־לְבָבוֹ֙ מֵֽאֶחָ֔יו וּלְבִלְתִּ֛י ס֥וּר מִן־הַמִּצְוָ֖ה יָמִ֣ין וּשְׂמֹ֑אול לְמַעַן֩ יַֽאֲרִ֨יךְ יָמִ֧ים עַל־מַמְלַכְתּ֛וֹ ה֥וּא וּבָנָ֖יו בְּקֶ֥רֶב יִשְׂרָאֵֽל׃ {ס} בְּדִיל דְּלָא יִירַם לִבֵּיהּ מֵאֲחוֹהִי, וּבְדִיל דְּלָא יִסְטֵי מִן תַּפְקֵידְתָּא לְיַמִּינָא וְלִסְמָאלָא--בְּדִיל דְּיוֹרֵיךְ יוֹמִין עַל מַלְכוּתֵיהּ הוּא וּבְנוֹהִי, בְּגוֹ יִשְׂרָאֵל. {ס}
יח,א לֹֽא־יִ֠הְיֶה לַכֹּֽהֲנִ֨ים הַלְוִיִּ֜ם כָּל־שֵׁ֧בֶט לֵוִ֛י חֵ֥לֶק וְנַֽחֲלָ֖ה עִם־יִשְׂרָאֵ֑ל אִשֵּׁ֧י יְהוָ֛ה וְנַֽחֲלָת֖וֹ יֹֽאכֵלֽוּן׃ לָא יְהֵי לְכָהֲנַיָּא לֵיוָאֵי כָּל שִׁבְטָא דְּלֵוִי, חוּלָק וְאַחְסָנָא--עִם יִשְׂרָאֵל; קֻרְבָּנַיָּא דַּייָ וְאַחְסָנְתֵיהּ, יֵיכְלוּן. 
יח,ב וְנַֽחֲלָ֥ה לֹא־יִֽהְיֶה־לּ֖וֹ בְּקֶ֣רֶב אֶחָ֑יו יְהוָה֙ ה֣וּא נַֽחֲלָת֔וֹ כַּֽאֲשֶׁ֖ר דִּבֶּר־לֽוֹ׃ {ס} וְאַחְסָנָא לָא יְהֵי לֵיהּ, בְּגוֹ אֲחוֹהִי:  מַתְּנָן דִּיהַב לֵיהּ יְיָ אִנִּין אַחְסָנְתֵיהּ, כְּמָא דְּמַלֵּיל לֵיהּ. {ס}
יח,ג וְזֶ֡ה יִֽהְיֶה֩ מִשְׁפַּ֨ט הַכֹּֽהֲנִ֜ים מֵאֵ֣ת הָעָ֗ם מֵאֵ֛ת זֹֽבְחֵ֥י הַזֶּ֖בַח אִם־שׁ֣וֹר אִם־שֶׂ֑ה וְנָתַן֙ לַכֹּהֵ֔ן הַזְּרֹ֥עַ וְהַלְּחָיַ֖יִם וְהַקֵּבָֽה׃ וְדֵין יְהֵי דְּחָזֵי לְכָהֲנַיָּא מִן עַמָּא, מִן נִכְסֵי נִכְסָתָא--אִם תּוֹר אִם אִמַּר:  וְיִתֵּין, לְכָהֲנָא, דְּרָעָא וְלוּעָא, וְקֵיבְתָא. 
יח,ד רֵאשִׁ֨ית דְּגָֽנְךָ֜ תִּֽירֹשְׁךָ֣ וְיִצְהָרֶ֗ךָ וְרֵאשִׁ֛ית גֵּ֥ז צֹֽאנְךָ֖ תִּתֶּן־לֽוֹ׃ רֵישׁ עֲבוּרָךְ חַמְרָךְ וּמִשְׁחָךְ, וְרֵישׁ גִּזַּת עָנָךְ--תִּתֵּין לֵיהּ. 
יח,ה כִּ֣י ב֗וֹ בָּחַ֛ר יְהוָ֥ה אֱלֹהֶ֖יךָ מִכָּל־שְׁבָטֶ֑יךָ לַֽעֲמֹ֨ד לְשָׁרֵ֧ת בְּשֵׁם־יְהוָ֛ה ה֥וּא וּבָנָ֖יו כָּל־הַיָּמִֽים׃ {ס} אֲרֵי בֵּיהּ, אִתְרְעִי יְיָ אֱלָהָךְ--מִכָּל שִׁבְטָךְ:  לִמְקָם לְשַׁמָּשָׁא בִּשְׁמָא דַּייָ הוּא וּבְנוֹהִי, כָּל יוֹמַיָּא. {ס}
יח,ו וְכִֽי־יָבֹ֨א הַלֵּוִ֜י מֵֽאַחַ֤ד שְׁעָרֶ֨יךָ֙ מִכָּל־יִשְׂרָאֵ֔ל אֲשֶׁר־ה֖וּא גָּ֣ר שָׁ֑ם וּבָא֙ בְּכָל־אַוַּ֣ת נַפְשׁ֔וֹ אֶל־הַמָּק֖וֹם אֲשֶׁר־יִבְחַ֥ר יְהוָֽה׃ וַאֲרֵי יֵיתֵי לֵיוָאָה מֵחֲדָא מִקִּרְוָךְ, מִכָּל יִשְׂרָאֵל, דְּהוּא, דָּר תַּמָּן; וְיֵיתֵי בְּכָל רְעוּת נַפְשֵׁיהּ, לְאַתְרָא דְּיִתְרְעֵי יְיָ. 
יח,ז וְשֵׁרֵ֕ת בְּשֵׁ֖ם יְהוָ֣ה אֱלֹהָ֑יו כְּכָל־אֶחָיו֙ הַלְוִיִּ֔ם הָעֹֽמְדִ֥ים שָׁ֖ם לִפְנֵ֥י יְהוָֽה׃ וִישַׁמֵּישׁ, בִּשְׁמָא דַּייָ אֱלָהֵיהּ--כְּכָל אֲחוֹהִי, לֵיוָאֵי, דִּמְשַׁמְּשִׁין תַּמָּן, קֳדָם יְיָ. 
יח,ח חֵ֥לֶק כְּחֵ֖לֶק יֹאכֵ֑לוּ לְבַ֥ד מִמְכָּרָ֖יו עַל־הָֽאָבֽוֹת׃ {ס} חוּלָק כְּחוּלָק, יֵיכְלוּן, בָּר מִמַּטַּרְתָּא דְּיָתַהּ שַׁבְּתָא, דְּכֵין אַתְקִינוּ אֲבָהָתָא. {ס}
יח,ט כִּ֤י אַתָּה֙ בָּ֣א אֶל־הָאָ֔רֶץ אֲשֶׁר־יְהוָ֥ה אֱלֹהֶ֖יךָ נֹתֵ֣ן לָ֑ךְ לֹֽא־תִלְמַ֣ד לַֽעֲשׂ֔וֹת כְּתֽוֹעֲבֹ֖ת הַגּוֹיִ֥ם הָהֵֽם׃ אֲרֵי אַתְּ עָלֵיל לְאַרְעָא, דַּייָ אֱלָהָךְ יָהֵיב לָךְ--לָא תֵּילַף לְמֶעֱבַד, כְּתוֹעֲבָת עַמְמַיָּא הָאִנּוּן. 
יח,י לֹֽא־יִמָּצֵ֣א בְךָ֔ מַֽעֲבִ֥יר בְּנֽוֹ־וּבִתּ֖וֹ בָּאֵ֑שׁ קֹסֵ֣ם קְסָמִ֔ים מְעוֹנֵ֥ן וּמְנַחֵ֖שׁ וּמְכַשֵּֽׁף׃ לָא יִשְׁתְּכַח בָּךְ, מַעְבַּר בְּרֵיהּ וּבְרַתֵּיהּ בְּנוּרָא, קָסֵים קִסְמִין, מְעָנֵין וּמְנַחֵישׁ וְחָרַשׁ. 
יח,יא וְחֹבֵ֖ר חָ֑בֶר וְשֹׁאֵ֥ל אוֹב֙ וְיִדְּעֹנִ֔י וְדֹרֵ֖שׁ אֶל־הַמֵּתִֽים׃ וְרָטֵין, רְטַן; וְשָׁאֵיל בְּבִדִּין וּבִזְכוּרוּ, וְתָבַע מִן מִיתַיָּא. 
יח,יב כִּֽי־תוֹעֲבַ֥ת יְהוָ֖ה כָּל־עֹ֣שֵׂה אֵ֑לֶּה וּבִגְלַל֙ הַתּֽוֹעֵבֹ֣ת הָאֵ֔לֶּה יְהוָ֣ה אֱלֹהֶ֔יךָ מוֹרִ֥ישׁ אוֹתָ֖ם מִפָּנֶֽיךָ׃ אֲרֵי מְרַחַק קֳדָם יְיָ, כָּל עָבֵיד אִלֵּין; וּבְדִיל, תּוֹעֵיבָתָא הָאִלֵּין, יְיָ אֱלָהָךְ, מְתָרֵיךְ יָתְהוֹן מִן קֳדָמָךְ. 
יח,יג תָּמִ֣ים תִּֽהְיֶ֔ה עִ֖ם יְהוָ֥ה אֱלֹהֶֽיךָ׃ שְׁלִים תְּהֵי, בְּדַחְלְתָא דַּייָ אֱלָהָךְ. 
יח,יד כִּ֣י ׀ הַגּוֹיִ֣ם הָאֵ֗לֶּה אֲשֶׁ֤ר אַתָּה֙ יוֹרֵ֣שׁ אוֹתָ֔ם אֶל־מְעֹֽנְנִ֥ים וְאֶל־קֹֽסְמִ֖ים יִשְׁמָ֑עוּ וְאַתָּ֕ה לֹ֣א כֵ֔ן נָ֥תַן לְךָ֖ יְהוָ֥ה אֱלֹהֶֽיךָ׃ אֲרֵי עַמְמַיָּא הָאִלֵּין, דְּאַתְּ יָרֵית יָתְהוֹן--מִן מְעָנְנַיָּא וּמִן קַסָמַיָּא, שָׁמְעִין; וְאַתְּ--לָא כֵּין, יְהַב לָךְ יְיָ אֱלָהָךְ. 
יח,טו נָבִ֨יא מִקִּרְבְּךָ֤ מֵֽאַחֶ֨יךָ֙ כָּמֹ֔נִי יָקִ֥ים לְךָ֖ יְהוָ֣ה אֱלֹהֶ֑יךָ אֵלָ֖יו תִּשְׁמָעֽוּן׃ נְבִיָּא מִבֵּינָךְ מֵאֲחָךְ כְּוָתִי, יְקִים לָךְ יְיָ אֱלָהָךְ:  מִנֵּיהּ, תְּקַבְּלוּן. 
יח,טז כְּכֹ֨ל אֲשֶׁר־שָׁאַ֜לְתָּ מֵעִ֨ם יְהוָ֤ה אֱלֹהֶ֨יךָ֙ בְּחֹרֵ֔ב בְּי֥וֹם הַקָּהָ֖ל לֵאמֹ֑ר לֹ֣א אֹסֵ֗ף לִשְׁמֹ֨עַ֙ אֶת־קוֹל֙ יְהוָ֣ה אֱלֹהָ֔י וְאֶת־הָאֵ֨שׁ הַגְּדֹלָ֥ה הַזֹּ֛את לֹֽא־אֶרְאֶ֥ה ע֖וֹד וְלֹ֥א אָמֽוּת׃ כְּכֹל דִּשְׁאֵילְתָּא מִן קֳדָם יְיָ אֱלָהָךְ, בְּחוֹרֵב, בְּיוֹמָא דִּקְהָלָא, לְמֵימַר:  לָא אוֹסֵיף, לְמִשְׁמַע יָת קָל מֵימְרָא דַּייָ אֱלָהִי, וְיָת אִישָׁתָא רַבְּתָא הָדָא לָא אֶחְזֵי עוֹד, וְלָא אֲמוּת. 
יח,יז וַיֹּ֥אמֶר יְהוָ֖ה אֵלָ֑י הֵיטִ֖יבוּ אֲשֶׁ֥ר דִּבֵּֽרוּ׃ וַאֲמַר יְיָ, לִי:  אַתְקִינוּ, דְּמַלִּילוּ. 
יח,יח נָבִ֨יא אָקִ֥ים לָהֶ֛ם מִקֶּ֥רֶב אֲחֵיהֶ֖ם כָּמ֑וֹךָ וְנָֽתַתִּ֤י דְבָרַי֙ בְּפִ֔יו וְדִבֶּ֣ר אֲלֵיהֶ֔ם אֵ֖ת כָּל־אֲשֶׁ֥ר אֲצַוֶּֽנּוּ׃ נְבִיָּא אֲקִים לְהוֹן מִגּוֹ אֲחֵיהוֹן, כְּוָתָךְ; וְאֶתֵּין פִּתְגְמֵי נְבוּאֲתִי, בְּפֻמֵּיהּ, וִימַלֵּיל עִמְּהוֹן, יָת כָּל דַּאֲפַקְּדִנֵּיהּ. 
יח,יט וְהָיָ֗ה הָאִישׁ֙ אֲשֶׁ֤ר לֹֽא־יִשְׁמַע֙ אֶל־דְּבָרַ֔י אֲשֶׁ֥ר יְדַבֵּ֖ר בִּשְׁמִ֑י אָֽנֹכִ֖י אֶדְרֹ֥שׁ מֵֽעִמּֽוֹ׃ וִיהֵי, גֻּבְרָא דְּלָא יְקַבֵּיל לְפִתְגָמַי, דִּימַלֵּיל, בִּשְׁמִי--מֵימְרִי, יִתְבַּע מִנֵּיהּ. 
יח,כ אַ֣ךְ הַנָּבִ֡יא אֲשֶׁ֣ר יָזִיד֩ לְדַבֵּ֨ר דָּבָ֜ר בִּשְׁמִ֗י אֵ֣ת אֲשֶׁ֤ר לֹֽא־צִוִּיתִיו֙ לְדַבֵּ֔ר וַֽאֲשֶׁ֣ר יְדַבֵּ֔ר בְּשֵׁ֖ם אֱלֹהִ֣ים אֲחֵרִ֑ים וּמֵ֖ת הַנָּבִ֥יא הַהֽוּא׃ בְּרַם נְבִיָּא דְּיַרְשַׁע לְמַלָּלָא פִּתְגָמָא בִּשְׁמִי, יָת דְּלָא פַּקֵּידְתֵּיהּ לְמַלָּלָא, וְדִימַלֵּיל, בְּשׁוֹם טָעֲוָת עַמְמַיָּא--וְיִתְקְטִיל, נְבִיָּא הַהוּא. 
יח,כא וְכִ֥י תֹאמַ֖ר בִּלְבָבֶ֑ךָ אֵיכָה֙ נֵדַ֣ע אֶת־הַדָּבָ֔ר אֲשֶׁ֥ר לֹֽא־דִבְּר֖וֹ יְהוָֽה׃ וַאֲרֵי תֵּימַר, בְּלִבָּךְ:  אֵיכְדֵין נִדַּע יָת פִּתְגָמָא, דְּלָא מַלְּלֵיהּ יְיָ. 
יח,כב אֲשֶׁר֩ יְדַבֵּ֨ר הַנָּבִ֜יא בְּשֵׁ֣ם יְהוָ֗ה וְלֹֽא־יִהְיֶ֤ה הַדָּבָר֙ וְלֹ֣א יָבֹ֔א ה֣וּא הַדָּבָ֔ר אֲשֶׁ֥ר לֹֽא־דִבְּר֖וֹ יְהוָ֑ה בְּזָדוֹן֙ דִּבְּר֣וֹ הַנָּבִ֔יא לֹ֥א תָג֖וּר מִמֶּֽנּוּ׃ {ס} דִּימַלֵּיל נְבִיָּא בִּשְׁמָא דַּייָ, וְלָא יְהֵי פִּתְגָמָא וְלָא יִתְקַיַּם--הוּא פִּתְגָמָא, דְּלָא מַלְּלֵיהּ יְיָ:  בִּרְשַׁע מַלְּלֵיהּ נְבִיָּא, לָא תִּדְחֲלוּן מִנֵּיהּ. {ס}
יט,א כִּֽי־יַכְרִ֞ית יְהוָ֤ה אֱלֹהֶ֨יךָ֙ אֶת־הַגּוֹיִ֔ם אֲשֶׁר֙ יְהוָ֣ה אֱלֹהֶ֔יךָ נֹתֵ֥ן לְךָ֖ אֶת־אַרְצָ֑ם וִֽירִשְׁתָּ֕ם וְיָֽשַׁבְתָּ֥ בְעָֽרֵיהֶ֖ם וּבְבָֽתֵּיהֶֽם׃ אֲרֵי יְשֵׁיצֵי יְיָ אֱלָהָךְ, יָת עַמְמַיָּא, דַּייָ אֱלָהָךְ, יָהֵיב לָךְ יָת אֲרַעְהוֹן; וְתֵירְתִנּוּן, וְתִתֵּיב בְּקִרְוֵיהוֹן וּבְבָתֵּיהוֹן. 
יט,ב שָׁל֥וֹשׁ עָרִ֖ים תַּבְדִּ֣יל לָ֑ךְ בְּת֣וֹךְ אַרְצְךָ֔ אֲשֶׁר֙ יְהוָ֣ה אֱלֹהֶ֔יךָ נֹתֵ֥ן לְךָ֖ לְרִשְׁתָּֽהּ׃ תְּלָת קִרְוִין, תַּפְרֵישׁ לָךְ:  בְּגוֹ אַרְעָךְ--דַּייָ אֱלָהָךְ, יָהֵיב לָךְ לְמֵירְתַהּ. 
יט,ג תָּכִ֣ין לְךָ֮ הַדֶּרֶךְ֒ וְשִׁלַּשְׁתָּ֙ אֶת־גְּב֣וּל אַרְצְךָ֔ אֲשֶׁ֥ר יַנְחִֽילְךָ֖ יְהוָ֣ה אֱלֹהֶ֑יךָ וְהָיָ֕ה לָנ֥וּס שָׁ֖מָּה כָּל־רֹצֵֽחַ׃ תַּתְקֵין לָךְ, אוֹרְחָא, וּתְתַלֵּית יָת תְּחוּם אַרְעָךְ, דְּיַחְסְנִנָּךְ יְיָ אֱלָהָךְ; וִיהֵי, לְמִעְרַק לְתַמָּן כָּל קָטוֹל. 
יט,ד וְזֶה֙ דְּבַ֣ר הָֽרֹצֵ֔חַ אֲשֶׁר־יָנ֥וּס שָׁ֖מָּה וָחָ֑י אֲשֶׁ֨ר יַכֶּ֤ה אֶת־רֵעֵ֨הוּ֙ בִּבְלִי־דַ֔עַת וְה֛וּא לֹֽא־שֹׂנֵ֥א ל֖וֹ מִתְּמֹ֥ל שִׁלְשֹֽׁם׃ וְדֵין פִּתְגָם קָטוֹלָא, דְּיִעְרוֹק לְתַמָּן וְיִתְקַיַּם:  דְּיִקְטוֹל יָת חַבְרֵיהּ בְּלָא מַדְּעֵיהּ, וְהוּא לָא סָנֵי לֵיהּ מֵאֶתְמָלִי וּמִדְּקַמּוֹהִי. 
יט,ה וַֽאֲשֶׁר֩ יָבֹ֨א אֶת־רֵעֵ֥הוּ בַיַּעַר֮ לַחְטֹ֣ב עֵצִים֒ וְנִדְּחָ֨ה יָד֤וֹ בַגַּרְזֶן֙ לִכְרֹ֣ת הָעֵ֔ץ וְנָשַׁ֤ל הַבַּרְזֶל֙ מִן־הָעֵ֔ץ וּמָצָ֥א אֶת־רֵעֵ֖הוּ וָמֵ֑ת ה֗וּא יָנ֛וּס אֶל־אַחַ֥ת הֶעָֽרִים־הָאֵ֖לֶּה וָחָֽי׃ וּדְיֵיעוֹל עִם חַבְרֵיהּ בְּחֻרְשָׁא, לְמִקַּץ אָעִין, וְתִתְמְרֵיג יְדֵיהּ בְּבַרְזְלָא לְמִקְּצֵיהּ אָעָא, וְיִשְׁתְּלֵיף בַּרְזְלָא מִן אָעָא וְיַשְׁכַּח יָת חַבְרֵיהּ וִימוּת:  הוּא, יִעְרוֹק לַחֲדָא מִן קִרְוַיָּא הָאִלֵּין--וְיִתְקַיַּם. 
יט,ו פֶּן־יִרְדֹּף֩ גֹּאֵ֨ל הַדָּ֜ם אַֽחֲרֵ֣י הָֽרֹצֵ֗חַ כִּֽי־יֵחַם֮ לְבָבוֹ֒ וְהִשִּׂיג֛וֹ כִּֽי־יִרְבֶּ֥ה הַדֶּ֖רֶךְ וְהִכָּ֣הוּ נָ֑פֶשׁ וְלוֹ֙ אֵ֣ין מִשְׁפַּט־מָ֔וֶת כִּ֠י לֹ֣א שֹׂנֵ֥א ה֛וּא ל֖וֹ מִתְּמ֥וֹל שִׁלְשֽׁוֹם׃ דִּלְמָא יִרְדּוֹף גָּאֵיל דְּמָא בָּתַר קָטוֹלָא, אֲרֵי יִיחַם לִבֵּיהּ, וְיִדְבְּקִנֵּיהּ אֲרֵי יִסְגֵּי אוֹרְחָא, וְיִקְטְלִנֵּיהּ נְפַשׁ; וְלֵיהּ, לֵית חוֹבַת דִּין דִּקְטוֹל, אֲרֵי לָא סָנֵי הוּא לֵיהּ, מֵאֶתְמָלִי וּמִדְּקַמּוֹהִי. 
יט,ז עַל־כֵּ֛ן אָֽנֹכִ֥י מְצַוְּךָ֖ לֵאמֹ֑ר שָׁלֹ֥שׁ עָרִ֖ים תַּבְדִּ֥יל לָֽךְ׃ עַל כֵּין אֲנָא מְפַקֵּיד לָךְ, לְמֵימַר:  תְּלָת קִרְוִין, תַּפְרֵישׁ לָךְ. 
יט,ח וְאִם־יַרְחִ֞יב יְהוָ֤ה אֱלֹהֶ֨יךָ֙ אֶת־גְּבֻ֣לְךָ֔ כַּֽאֲשֶׁ֥ר נִשְׁבַּ֖ע לַֽאֲבֹתֶ֑יךָ וְנָ֤תַן לְךָ֙ אֶת־כָּל־הָאָ֔רֶץ אֲשֶׁ֥ר דִּבֶּ֖ר לָתֵ֥ת לַֽאֲבֹתֶֽיךָ׃ וְאִם יַפְתֵּי יְיָ אֱלָהָךְ, יָת תְּחוּמָךְ, כְּמָא דְּקַיֵּים, לַאֲבָהָתָךְ; וְיִתֵּין לָךְ יָת כָּל אַרְעָא, דְּמַלֵּיל לְמִתַּן לַאֲבָהָתָךְ. 
יט,ט כִּֽי־תִשְׁמֹר֩ אֶת־כָּל־הַמִּצְוָ֨ה הַזֹּ֜את לַֽעֲשֹׂתָ֗הּ אֲשֶׁ֨ר אָֽנֹכִ֣י מְצַוְּךָ֮ הַיּוֹם֒ לְאַֽהֲבָ֞ה אֶת־יְהוָ֧ה אֱלֹהֶ֛יךָ וְלָלֶ֥כֶת בִּדְרָכָ֖יו כָּל־הַיָּמִ֑ים וְיָֽסַפְתָּ֨ לְךָ֥ עוֹד֙ שָׁלֹ֣שׁ עָרִ֔ים עַ֖ל הַשָּׁלֹ֥שׁ הָאֵֽלֶּה׃ אֲרֵי תִּטַּר יָת כָּל תַּפְקֵידְתָּא הָדָא לְמִעְבְּדַהּ, דַּאֲנָא מְפַקֵּיד לָךְ יוֹמָא דֵּין, לְמִרְחַם יָת יְיָ אֱלָהָךְ וְלִמְהָךְ בְּאוֹרְחָן דְּתָקְנָן קֳדָמוֹהִי, כָּל יוֹמַיָּא--וְתוֹסֵיף לָךְ עוֹד תְּלָת קִרְוִין, עַל תְּלָת אִלֵּין. 
יט,י וְלֹ֤א יִשָּׁפֵךְ֙ דָּ֣ם נָקִ֔י בְּקֶ֣רֶב אַרְצְךָ֔ אֲשֶׁר֙ יְהוָ֣ה אֱלֹהֶ֔יךָ נֹתֵ֥ן לְךָ֖ נַֽחֲלָ֑ה וְהָיָ֥ה עָלֶ֖יךָ דָּמִֽים׃ {פ} וְלָא יִשְׁתְּפֵיךְ, דַּם זַכַּאי, בְּגוֹ אַרְעָךְ, דַּייָ אֱלָהָךְ יָהֵיב לָךְ אַחְסָנָא; וִיהֵי עֲלָךְ, חוֹבַת דִּין דִּקְטוֹל. {פ}
יט,יא וְכִֽי־יִהְיֶ֥ה אִישׁ֙ שֹׂנֵ֣א לְרֵעֵ֔הוּ וְאָ֤רַב לוֹ֙ וְקָ֣ם עָלָ֔יו וְהִכָּ֥הוּ נֶ֖פֶשׁ וָמֵ֑ת וְנָ֕ס אֶל־אַחַ֖ת הֶֽעָרִ֥ים הָאֵֽל׃ וַאֲרֵי יְהֵי גְּבַר, סָנֵי לְחַבְרֵיהּ, וְיִכְמוֹן לֵיהּ וִיקוּם עֲלוֹהִי, וְיִקְטְלִנֵּיהּ נְפַשׁ וִימוּת; וְיִעְרוֹק, לַחֲדָא מִן קִרְוַיָּא הָאִלֵּין. 
יט,יב וְשָֽׁלְחוּ֙ זִקְנֵ֣י עִיר֔וֹ וְלָֽקְח֥וּ אֹת֖וֹ מִשָּׁ֑ם וְנָֽתְנ֣וּ אֹת֗וֹ בְּיַ֛ד גֹּאֵ֥ל הַדָּ֖ם וָמֵֽת׃ וְיִשְׁלְחוּן סָבֵי קַרְתֵּיהּ, וְיִדְבְּרוּן יָתֵיהּ מִתַּמָּן; וְיִמְסְרוּן יָתֵיהּ, בְּיַד גָּאֵיל דְּמָא--וִימוּת. 
יט,יג לֹֽא־תָח֥וֹס עֵֽינְךָ֖ עָלָ֑יו וּבִֽעַרְתָּ֧ דַֽם־הַנָּקִ֛י מִיִּשְׂרָאֵ֖ל וְט֥וֹב לָֽךְ׃ {ס} לָא תְּחוּס עֵינָךְ, עֲלוֹהִי; וּתְפַלֵּי אָשְׁדֵי דַּם זַכַּאי מִיִּשְׂרָאֵל, וְיִיטַב לָךְ. {ס}
יט,יד לֹ֤א תַסִּיג֙ גְּב֣וּל רֵֽעֲךָ֔ אֲשֶׁ֥ר גָּֽבְל֖וּ רִֽאשֹׁנִ֑ים בְּנַחֲלָֽתְךָ֙ אֲשֶׁ֣ר תִּנְחַ֔ל בָּאָ֕רֶץ אֲשֶׁר֙ יְהוָ֣ה אֱלֹהֶ֔יךָ נֹתֵ֥ן לְךָ֖ לְרִשְׁתָּֽהּ׃ {ס} לָא תְּשַׁנֵּי תְּחוּמָא דְּחַבְרָךְ, דְּתַחִימוּ קַדְמָאֵי--בְּאַחְסָנְתָךְ, דְּתַחְסֵין, בְּאַרְעָא, דַּייָ אֱלָהָךְ יָהֵיב לָךְ לְמֵירְתַהּ. {ס}
יט,טו לֹֽא־יָקוּם֩ עֵ֨ד אֶחָ֜ד בְּאִ֗ישׁ לְכָל־עָו‍ֹן֙ וּלְכָל־חַטָּ֔את בְּכָל־חֵ֖טְא אֲשֶׁ֣ר יֶֽחֱטָ֑א עַל־פִּ֣י ׀ שְׁנֵ֣י עֵדִ֗ים א֛וֹ עַל־פִּ֥י שְׁלֹשָֽׁה־עֵדִ֖ים יָק֥וּם דָּבָֽר׃ לָא יְקוּם סָהִיד חַד בִּגְבַר, לְכָל עֲוָיָן וּלְכָל חוֹבִין, בְּכָל חֵטְא, דְּיִחְטֵי:  עַל מֵימַר תְּרֵין סָהֲדִין, אוֹ עַל מֵימַר תְּלָתָא סָהֲדִין--יִתְקַיַּם פִּתְגָמָא. 
יט,טז כִּֽי־יָק֥וּם עֵד־חָמָ֖ס בְּאִ֑ישׁ לַֽעֲנ֥וֹת בּ֖וֹ סָרָֽה׃ אֲרֵי יְקוּם סָהִיד שְׁקַר, בִּגְבַר, לְאַסְהָדָא בֵּיהּ, סָטְיָא. 
יט,יז וְעָֽמְד֧וּ שְׁנֵֽי־הָאֲנָשִׁ֛ים אֲשֶׁר־לָהֶ֥ם הָרִ֖יב לִפְנֵ֣י יְהוָ֑ה לִפְנֵ֤י הַכֹּֽהֲנִים֙ וְהַשֹּׁ֣פְטִ֔ים אֲשֶׁ֥ר יִֽהְי֖וּ בַּיָּמִ֥ים הָהֵֽם׃ וִיקוּמוּן תְּרֵין גֻּבְרַיָּא דִּילְהוֹן דִּינָא, קֳדָם יְיָ, קֳדָם כָּהֲנַיָּא וְדַיָּנַיָּא, דִּיהוֹן בְּיוֹמַיָּא הָאִנּוּן. 
יט,יח וְדָֽרְשׁ֥וּ הַשֹּֽׁפְטִ֖ים הֵיטֵ֑ב וְהִנֵּ֤ה עֵֽד־שֶׁ֨קֶר֙ הָעֵ֔ד שֶׁ֖קֶר עָנָ֥ה בְאָחִֽיו׃ וְיִתְבְּעוּן דַּיָּנַיָּא, יָאוּת; וְהָא סָהִיד שִׁקְרָא סָהֲדָא, שִׁקְרָא אַסְהֵיד בַּאֲחוּהִי. 
יט,יט וַֽעֲשִׂ֣יתֶם ל֔וֹ כַּֽאֲשֶׁ֥ר זָמַ֖ם לַֽעֲשׂ֣וֹת לְאָחִ֑יו וּבִֽעַרְתָּ֥ הָרָ֖ע מִקִּרְבֶּֽךָ׃ וְתַעְבְּדוּן לֵיהּ, כְּמָא דְּחַשֵּׁיב לְמֶעֱבַד לַאֲחוּהִי; וּתְפַלֵּי עָבֵיד דְּבִישׁ, מִבֵּינָךְ. 
יט,כ וְהַנִּשְׁאָרִ֖ים יִשְׁמְע֣וּ וְיִרָ֑אוּ וְלֹֽא־יֹסִ֨פוּ לַֽעֲשׂ֜וֹת ע֗וֹד כַּדָּבָ֥ר הָרָ֛ע הַזֶּ֖ה בְּקִרְבֶּֽךָ׃ וּדְיִשְׁתְּאַרוּן, יִשְׁמְעוּן וְיִדְחֲלוּן; וְלָא יֵיסְפוּן לְמֶעֱבַד עוֹד, כְּפִתְגָמָא בִּישָׁא הָדֵין--בֵּינָךְ. 
יט,כא וְלֹ֥א תָח֖וֹס עֵינֶ֑ךָ נֶ֣פֶשׁ בְּנֶ֗פֶשׁ עַ֤יִן בְּעַ֨יִן֙ שֵׁ֣ן בְּשֵׁ֔ן יָ֥ד בְּיָ֖ד רֶ֥גֶל בְּרָֽגֶל׃ {ס} וְלָא תְּחוּס, עֵינָךְ:  נַפְשָׁא חֲלָף נַפְשָׁא, עֵינָא חֲלָף עֵינָא שִׁנָּא חֲלָף שִׁנָּא, יְדָא חֲלָף יְדָא, רִגְלָא חֲלָף רִגְלָא. {ס}
כ,א כִּֽי־תֵצֵ֨א לַמִּלְחָמָ֜ה עַל־אֹֽיְבֶ֗ךָ וְֽרָאִ֜יתָ ס֤וּס וָרֶ֨כֶב֙ עַ֚ם רַ֣ב מִמְּךָ֔ לֹ֥א תִירָ֖א מֵהֶ֑ם כִּֽי־יְהוָ֤ה אֱלֹהֶ֨יךָ֙ עִמָּ֔ךְ הַמַּֽעַלְךָ֖ מֵאֶ֥רֶץ מִצְרָֽיִם׃ אֲרֵי תִּפּוֹק לְאָגָחָא קְרָבָא עַל בַּעֲלֵי דְּבָבָךְ, וְתִחְזֵי סוּסָוָן וּרְתִכִּין עַם סַגִּי מִנָּךְ--לָא תִּדְחַל, מִנְּהוֹן:  אֲרֵי יְיָ אֱלָהָךְ מֵימְרֵיהּ בְּסַעֲדָךְ, דְּאַסְּקָךְ מֵאַרְעָא דְּמִצְרָיִם. 
כ,ב וְהָיָ֕ה כְּקָֽרָבְכֶ֖ם אֶל־הַמִּלְחָמָ֑ה וְנִגַּ֥שׁ הַכֹּהֵ֖ן וְדִבֶּ֥ר אֶל־הָעָֽם׃ וִיהֵי, כְּמִקְרַבְכוֹן לְאָגָחָא קְרָבָא; וְיִתְקָרַב כָּהֲנָא, וִימַלֵּיל עִם עַמָּא. 
כ,ג וְאָמַ֤ר אֲלֵהֶם֙ שְׁמַ֣ע יִשְׂרָאֵ֔ל אַתֶּ֨ם קְרֵבִ֥ים הַיּ֛וֹם לַמִּלְחָמָ֖ה עַל־אֹֽיְבֵיכֶ֑ם אַל־יֵרַ֣ךְ לְבַבְכֶ֗ם אַל־תִּֽירְא֧וּ וְאַֽל־תַּחְפְּז֛וּ וְאַל־תַּֽעַרְצ֖וּ מִפְּנֵיהֶֽם׃ וְיֵימַר לְהוֹן שְׁמַע יִשְׂרָאֵל, אַתּוּן מִתְקָרְבִין יוֹמָא דֵּין לְאָגָחָא קְרָבָא עַל בַּעֲלֵי דְּבָבֵיכוֹן; לָא יְזוּעַ לִבְּכוֹן, לָא תִּדְחֲלוּן וְלָא תִּתְבַּעְתוּן וְלָא תִּתַּבְרוּן--מִן קֳדָמֵיהוֹן. 
כ,ד כִּ֚י יְהוָ֣ה אֱלֹֽהֵיכֶ֔ם הַֽהֹלֵ֖ךְ עִמָּכֶ֑ם לְהִלָּחֵ֥ם לָכֶ֛ם עִם־אֹֽיְבֵיכֶ֖ם לְהוֹשִׁ֥יעַ אֶתְכֶֽם׃ אֲרֵי יְיָ אֱלָהֲכוֹן, דִּמְדַבַּר קֳדָמֵיכוֹן--לְאָגָחָא לְכוֹן קְרָב עִם בַּעֲלֵי דְּבָבֵיכוֹן, לְמִפְרַק יָתְכוֹן. 
כ,ה וְדִבְּר֣וּ הַשֹּֽׁטְרִים֮ אֶל־הָעָ֣ם לֵאמֹר֒ מִֽי־הָאִ֞ישׁ אֲשֶׁ֨ר בָּנָ֤ה בַֽיִת־חָדָשׁ֙ וְלֹ֣א חֲנָכ֔וֹ יֵלֵ֖ךְ וְיָשֹׁ֣ב לְבֵית֑וֹ פֶּן־יָמוּת֙ בַּמִּלְחָמָ֔ה וְאִ֥ישׁ אַחֵ֖ר יַחְנְכֶֽנּוּ׃ וִימַלְּלוּן סָרְכַיָּא, עִם עַמָּא לְמֵימַר, מַאן גֻּבְרָא דִּבְנָא בֵּיתָא חֲדַתָּא וְלָא חַנְכֵיהּ, יְהָךְ וִיתוּב לְבֵיתֵיהּ:  דִּלְמָא יִתְקְטִיל, בִּקְרָבָא, וּגְבַר אָחֳרָן, יַחְנְכִנֵּיהּ. 
כ,ו וּמִֽי־הָאִ֞ישׁ אֲשֶׁר־נָטַ֥ע כֶּ֨רֶם֙ וְלֹ֣א חִלְּל֔וֹ יֵלֵ֖ךְ וְיָשֹׁ֣ב לְבֵית֑וֹ פֶּן־יָמוּת֙ בַּמִּלְחָמָ֔ה וְאִ֥ישׁ אַחֵ֖ר יְחַלְּלֶֽנּוּ׃ וּמַאן גֻּבְרָא דִּנְצַב כַּרְמָא, וְלָא אַחֲלֵיהּ--יְהָךְ, וִיתוּב לְבֵיתֵיהּ:  דִּלְמָא יִתְקְטִיל, בִּקְרָבָא, וּגְבַר אָחֳרָן, יַחֲלִנֵּיהּ. 
כ,ז וּמִֽי־הָאִ֞ישׁ אֲשֶׁר־אֵרַ֤שׂ אִשָּׁה֙ וְלֹ֣א לְקָחָ֔הּ יֵלֵ֖ךְ וְיָשֹׁ֣ב לְבֵית֑וֹ פֶּן־יָמוּת֙ בַּמִּלְחָמָ֔ה וְאִ֥ישׁ אַחֵ֖ר יִקָּחֶֽנָּה׃ וּמַאן גֻּבְרָא דַּאֲרַס אִתְּתָא, וְלָא נַסְבַּהּ--יְהָךְ, וִיתוּב לְבֵיתֵיהּ:  דִּלְמָא יִתְקְטִיל, בִּקְרָבָא, וּגְבַר אָחֳרָן, יִסְּבִנַּהּ. 
כ,ח וְיָֽסְפ֣וּ הַשֹּֽׁטְרִים֮ לְדַבֵּ֣ר אֶל־הָעָם֒ וְאָֽמְר֗וּ מִֽי־הָאִ֤ישׁ הַיָּרֵא֙ וְרַ֣ךְ הַלֵּבָ֔ב יֵלֵ֖ךְ וְיָשֹׁ֣ב לְבֵית֑וֹ וְלֹ֥א יִמַּ֛ס אֶת־לְבַ֥ב אֶחָ֖יו כִּלְבָבֽוֹ׃ וְיֵיסְפוּן סָרְכַיָּא, לְמַלָּלָא עִם עַמָּא, וְיֵימְרוּן מַאן גֻּבְרָא דְּדָחֵיל וּתְבִיר לִבָּא, יְהָךְ וִיתוּב לְבֵיתֵיהּ; וְלָא יִתְבַּר יָת לִבָּא דַּאֲחוֹהִי, כְּלִבֵּיהּ. 
כ,ט וְהָיָ֛ה כְּכַלֹּ֥ת הַשֹּֽׁטְרִ֖ים לְדַבֵּ֣ר אֶל־הָעָ֑ם וּפָ֥קְד֛וּ שָׂרֵ֥י צְבָא֖וֹת בְּרֹ֥אשׁ הָעָֽם׃ {ס} וִיהֵי כַּד יְשֵׁיצוֹן סָרְכַיָּא, לְמַלָּלָא עִם עַמָּא; וִימַנּוֹן רַבָּנֵי חֵילָא, בְּרֵישׁ עַמָּא. {ס}
כ,י כִּֽי־תִקְרַ֣ב אֶל־עִ֔יר לְהִלָּחֵ֖ם עָלֶ֑יהָ וְקָרָ֥אתָ אֵלֶ֖יהָ לְשָׁלֽוֹם׃ אֲרֵי תִּקְרַב לְקַרְתָּא, לְאָגָחָא קְרָבָא עֲלַהּ--וְתִקְרֵי לַהּ, מִלִּין דִּשְׁלָם. 
כ,יא וְהָיָה֙ אִם־שָׁל֣וֹם תַּֽעַנְךָ֔ וּפָֽתְחָ֖ה לָ֑ךְ וְהָיָ֞ה כָּל־הָעָ֣ם הַנִּמְצָא־בָ֗הּ יִֽהְי֥וּ לְךָ֛ לָמַ֖ס וַֽעֲבָדֽוּךָ׃ וִיהֵי אִם שְׁלָם תַּעֲנֵינָךְ, וְתִפְתַּח לָךְ:  וִיהֵי כָּל עַמָּא דְּיִשְׁתְּכַח בַּהּ, יְהוֹן לָךְ מַסְּקֵי מִסִּין--וְיִפְלְחוּנָּךְ. 
כ,יב וְאִם־לֹ֤א תַשְׁלִים֙ עִמָּ֔ךְ וְעָֽשְׂתָ֥ה עִמְּךָ֖ מִלְחָמָ֑ה וְצַרְתָּ֖ עָלֶֽיהָ׃ וְאִם לָא תַּשְׁלֵים עִמָּךְ, וְתַעֲבֵיד עִמָּךְ קְרָב--וּתְצוּר, עֲלַהּ. 
כ,יג וּנְתָנָ֛הּ יְהוָ֥ה אֱלֹהֶ֖יךָ בְּיָדֶ֑ךָ וְהִכִּיתָ֥ אֶת־כָּל־זְכוּרָ֖הּ לְפִי־חָֽרֶב׃ וְיִמְסְרִנַּהּ יְיָ אֱלָהָךְ, בִּידָךְ; וְתִמְחֵי יָת כָּל דְּכוּרַהּ, לְפִתְגָם דְּחָרֶב. 
כ,יד רַ֣ק הַ֠נָּשִׁים וְהַטַּ֨ף וְהַבְּהֵמָ֜ה וְכֹל֩ אֲשֶׁ֨ר יִֽהְיֶ֥ה בָעִ֛יר כָּל־שְׁלָלָ֖הּ תָּבֹ֣ז לָ֑ךְ וְאָֽכַלְתָּ֙ אֶת־שְׁלַ֣ל אֹֽיְבֶ֔יךָ אֲשֶׁ֥ר נָתַ֛ן יְהוָ֥ה אֱלֹהֶ֖יךָ לָֽךְ׃ לְחוֹד נְשַׁיָּא וְטַפְלָא וּבְעִירָא וְכֹל דִּיהֵי בְּקַרְתָּא, כָּל עֲדָאַהּ--תִּבּוֹז לָךְ; וְתֵיכוֹל יָת עֲדַי סָנְאָךְ, דִּיהַב יְיָ אֱלָהָךְ לָךְ. 
כ,טו כֵּ֤ן תַּֽעֲשֶׂה֙ לְכָל־הֶ֣עָרִ֔ים הָֽרְחֹקֹ֥ת מִמְּךָ֖ מְאֹ֑ד אֲשֶׁ֛ר לֹֽא־מֵעָרֵ֥י הַגּֽוֹיִם־הָאֵ֖לֶּה הֵֽנָּה׃ כֵּין תַּעֲבֵיד לְכָל קִרְוַיָּא, דְּרַחִיקִין מִנָּךְ לַחְדָּא, דְּלָא מִקִּרְוֵי עַמְמַיָּא הָאִלֵּין, אִנִּין. 
כ,טז רַ֗ק מֵֽעָרֵ֤י הָֽעַמִּים֙ הָאֵ֔לֶּה אֲשֶׁר֙ יְהוָ֣ה אֱלֹהֶ֔יךָ נֹתֵ֥ן לְךָ֖ נַֽחֲלָ֑ה לֹ֥א תְחַיֶּ֖ה כָּל־נְשָׁמָֽה׃ לְחוֹד, מִקִּרְוֵי עַמְמַיָּא הָאִלֵּין, דַּייָ אֱלָהָךְ, יָהֵיב לָךְ אַחְסָנָא--לָא תְּקַיֵּים, כָּל נִשְׁמָא. 
כ,יז כִּֽי־הַחֲרֵ֣ם תַּֽחֲרִימֵ֗ם הַֽחִתִּ֤י וְהָֽאֱמֹרִי֙ הַכְּנַֽעֲנִ֣י וְהַפְּרִזִּ֔י הַֽחִוִּ֖י וְהַיְבוּסִ֑י כַּֽאֲשֶׁ֥ר צִוְּךָ֖ יְהוָ֥ה אֱלֹהֶֽיךָ׃ אֲרֵי גַּמָּרָא תְּגַּמְּרִנּוּן, חִתָּאֵי וֶאֱמוֹרָאֵי כְּנַעֲנָאֵי וּפְרִזָּאֵי, חִוָּאֵי, וִיבוּסָאֵי--כְּמָא דְּפַקְּדָךְ, יְיָ אֱלָהָךְ. 
כ,יח לְמַ֗עַן אֲשֶׁ֨ר לֹֽא־יְלַמְּד֤וּ אֶתְכֶם֙ לַֽעֲשׂ֔וֹת כְּכֹל֙ תּֽוֹעֲבֹתָ֔ם אֲשֶׁ֥ר עָשׂ֖וּ לֵֽאלֹהֵיהֶ֑ם וַֽחֲטָאתֶ֖ם לַֽיהוָ֥ה אֱלֹֽהֵיכֶֽם׃ {ס} בְּדִיל, דְּלָא יַלְּפוּן יָתְכוֹן לְמֶעֱבַד, כְּכֹל תּוֹעֵיבָתְהוֹן, דְּעָבְדִין לְטָעֲוָתְהוֹן; וּתְחוּבוּן, קֳדָם יְיָ אֱלָהֲכוֹן. {ס}
כ,יט כִּֽי־תָצ֣וּר אֶל־עִיר֩ יָמִ֨ים רַבִּ֜ים לְֽהִלָּחֵ֧ם עָלֶ֣יהָ לְתָפְשָׂ֗הּ לֹֽא־תַשְׁחִ֤ית אֶת־עֵצָהּ֙ לִנְדֹּ֤חַ עָלָיו֙ גַּרְזֶ֔ן כִּ֚י מִמֶּ֣נּוּ תֹאכֵ֔ל וְאֹת֖וֹ לֹ֣א תִכְרֹ֑ת כִּ֤י הָֽאָדָם֙ עֵ֣ץ הַשָּׂדֶ֔ה לָבֹ֥א מִפָּנֶ֖יךָ בַּמָּצֽוֹר׃ אֲרֵי תְּצוּר עַל קַרְתָּא יוֹמִין סַגִּיאִין לְאָגָחָא קְרָבָא עֲלַהּ לְמִכְבְּשַׁהּ, לָא תְּחַבֵּיל יָת אִילָנַהּ לְאָרָמָא עֲלוֹהִי בַּרְזְלָא--אֲרֵי מִנֵּיהּ תֵּיכוֹל, וְיָתֵיהּ לָא תְּקוּץ:  אֲרֵי לָא כַּאֲנָשָׁא אִילָן חַקְלָא, לְמֵיעַל מִן קֳדָמָךְ בִּצְיָרָא. 
כ,כ רַ֞ק עֵ֣ץ אֲשֶׁר־תֵּדַ֗ע כִּֽי־לֹא־עֵ֤ץ מַֽאֲכָל֙ ה֔וּא אֹת֥וֹ תַשְׁחִ֖ית וְכָרָ֑תָּ וּבָנִ֣יתָ מָצ֗וֹר עַל־הָעִיר֙ אֲשֶׁר־הִ֨וא עֹשָׂ֧ה עִמְּךָ֛ מִלְחָמָ֖ה עַ֥ד רִדְתָּֽהּ׃ {פ} לְחוֹד אִילָן דְּתִדַּע, אֲרֵי לָא אִילָן דְּמֵיכַל הוּא--יָתֵיהּ תְּחַבֵּיל, וּתְקוּץ; וְתִבְנֵי כַּרְקוֹמִין, עַל קַרְתָּא דְּהִיא עָבְדָא עִמָּךְ קְרָב--עַד דְּתִכְבְּשַׁהּ. {פ}
כא,א כִּֽי־יִמָּצֵ֣א חָלָ֗ל בָּֽאֲדָמָה֙ אֲשֶׁר֩ יְהוָ֨ה אֱלֹהֶ֜יךָ נֹתֵ֤ן לְךָ֙ לְרִשְׁתָּ֔הּ נֹפֵ֖ל בַּשָּׂדֶ֑ה לֹ֥א נוֹדַ֖ע מִ֥י הִכָּֽהוּ׃ אֲרֵי יִשְׁתְּכַח קְטִילָא, בְּאַרְעָא דַּייָ אֱלָהָךְ יָהֵיב לָךְ לְמֵירְתַהּ, רְמִי, בְּחַקְלָא:  לָא יְדִיעַ, מַאן קַטְלֵיהּ. 
כא,ב וְיָֽצְא֥וּ זְקֵנֶ֖יךָ וְשֹֽׁפְטֶ֑יךָ וּמָֽדְדוּ֙ אֶל־הֶ֣עָרִ֔ים אֲשֶׁ֖ר סְבִיבֹ֥ת הֶֽחָלָֽל׃ וְיִפְּקוּן סָבָךְ, וְדַיָּנָךְ; וְיִמְשְׁחוּן, לְקִרְוַיָּא, דִּבְסַחְרָנוּת, קְטִילָא. 
כא,ג וְהָיָ֣ה הָעִ֔יר הַקְּרֹבָ֖ה אֶל־הֶֽחָלָ֑ל וְלָֽקְח֡וּ זִקְנֵי֩ הָעִ֨יר הַהִ֜וא עֶגְלַ֣ת בָּקָ֗ר אֲשֶׁ֤ר לֹֽא־עֻבַּד֙ בָּ֔הּ אֲשֶׁ֥ר לֹֽא־מָשְׁכָ֖ה בְּעֹֽל׃ וּתְהֵי קַרְתָּא, דְּקָרִיבָא לִקְטִילָא--וְיִסְּבוּן סָבֵי קַרְתָּא הַהִיא עֶגְלַת תּוֹרִין, דְּלָא אִתְפְּלַח בַּהּ, דְּלָא נְגַדַת, בְּנִיר. 
כא,ד וְהוֹרִ֡דוּ זִקְנֵי֩ הָעִ֨יר הַהִ֤וא אֶת־הָֽעֶגְלָה֙ אֶל־נַ֣חַל אֵיתָ֔ן אֲשֶׁ֛ר לֹֽא־יֵעָבֵ֥ד בּ֖וֹ וְלֹ֣א יִזָּרֵ֑עַ וְעָֽרְפוּ־שָׁ֥ם אֶת־הָֽעֶגְלָ֖ה בַּנָּֽחַל׃ וְיַחֲתוּן סָבֵי קַרְתָּא הַהִיא יָת עֶגְלְתָא, לִנְחַל בְּיָר, דְּלָא אִתְפְּלַח בֵּיהּ, וְלָא יִזְדְּרַע; וְיִקְפוֹן תַּמָּן יָת עֶגְלְתָא, בְּנַחְלָא. 
כא,ה וְנִגְּשׁ֣וּ הַכֹּֽהֲנִים֮ בְּנֵ֣י לֵוִי֒ כִּ֣י בָ֗ם בָּחַ֞ר יְהוָ֤ה אֱלֹהֶ֨יךָ֙ לְשָׁ֣רְת֔וֹ וּלְבָרֵ֖ךְ בְּשֵׁ֣ם יְהוָ֑ה וְעַל־פִּיהֶ֥ם יִֽהְיֶ֖ה כָּל־רִ֥יב וְכָל־נָֽגַע׃ וְיִתְקָרְבוּן כָּהֲנַיָּא, בְּנֵי לֵוִי--אֲרֵי בְּהוֹן אִתְרְעִי יְיָ אֱלָהָךְ לְשַׁמָּשׁוּתֵיהּ, וּלְבָרָכָא בִּשְׁמָא דַּייָ; וְעַל מֵימַרְהוֹן יְהֵי, כָּל דִּין וְכָל מַכְתָּשׁ סְגִירוּ. 
כא,ו וְכֹ֗ל זִקְנֵי֙ הָעִ֣יר הַהִ֔וא הַקְּרֹבִ֖ים אֶל־הֶֽחָלָ֑ל יִרְחֲצוּ֙ אֶת־יְדֵיהֶ֔ם עַל־הָֽעֶגְלָ֖ה הָֽעֲרוּפָ֥ה בַנָּֽחַל׃ וְכֹל, סָבֵי קַרְתָּא הַהִיא, דְּקָרִיבִין, לִקְטִילָא--יַסְחוֹן, יָת יְדֵיהוֹן, עַל עֶגְלְתָא, דִּנְקִיפָא בְּנַחְלָא. 
כא,ז וְעָנ֖וּ וְאָֽמְר֑וּ יָדֵ֗ינוּ לֹ֤א שפכה (שָֽׁפְכוּ֙) אֶת־הַדָּ֣ם הַזֶּ֔ה וְעֵינֵ֖ינוּ לֹ֥א רָאֽוּ׃ וְיָתִיבוּן, וְיֵימְרוּן:  יְדַנָא, לָא אֲשַׁדָא יָת דְּמָא הָדֵין, וְעֵינַנָא, לָא חֲזַאָה. 
כא,ח כַּפֵּר֩ לְעַמְּךָ֨ יִשְׂרָאֵ֤ל אֲשֶׁר־פָּדִ֨יתָ֙ יְהוָ֔ה וְאַל־תִּתֵּן֙ דָּ֣ם נָקִ֔י בְּקֶ֖רֶב עַמְּךָ֣ יִשְׂרָאֵ֑ל וְנִכַּפֵּ֥ר לָהֶ֖ם הַדָּֽם׃ כָּהֲנַיָּא יֵימְרוּן כַּפַּר לְעַמָּךְ יִשְׂרָאֵל דִּפְרַקְתָּא, יְיָ, וְלָא תִּתֵּין חוֹבַת דַּם זַכַּאי, בְּגוֹ עַמָּךְ יִשְׂרָאֵל; וְיִתְכַּפַּר לְהוֹן, עַל דְּמָא. 
כא,ט וְאַתָּ֗ה תְּבַעֵ֛ר הַדָּ֥ם הַנָּקִ֖י מִקִּרְבֶּ֑ךָ כִּֽי־תַעֲשֶׂ֥ה הַיָּשָׁ֖ר בְּעֵינֵ֥י יְהוָֽה׃ {ס} וְאַתְּ, תְּפַלֵּי אָשְׁדֵי דַּם זַכַּאי--מִבֵּינָךְ:  אֲרֵי תַּעֲבֵיד דְּכָשַׁר, קֳדָם יְיָ. {ס}
