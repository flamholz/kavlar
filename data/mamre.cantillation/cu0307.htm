קדושים

יט,א וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
יט,ב דַּבֵּ֞ר אֶל־כָּל־עֲדַ֧ת בְּנֵֽי־יִשְׂרָאֵ֛ל וְאָֽמַרְתָּ֥ אֲלֵהֶ֖ם קְדֹשִׁ֣ים תִּֽהְי֑וּ כִּ֣י קָד֔וֹשׁ אֲנִ֖י יְהוָ֥ה אֱלֹֽהֵיכֶֽם׃ מַלֵּיל עִם כָּל כְּנִשְׁתָּא דִּבְנֵי יִשְׂרָאֵל, וְתֵימַר לְהוֹן--קַדִּישִׁין תְּהוֹן:  אֲרֵי קַדִּישׁ, אֲנָא יְיָ אֱלָהֲכוֹן. 
יט,ג אִ֣ישׁ אִמּ֤וֹ וְאָבִיו֙ תִּירָ֔אוּ וְאֶת־שַׁבְּתֹתַ֖י תִּשְׁמֹ֑רוּ אֲנִ֖י יְהוָ֥ה אֱלֹֽהֵיכֶֽם׃ גְּבַר מִן אִמֵּיהּ וּמִן אֲבוּהִי תְּהוֹן דָּחֲלִין, וְיָת יוֹמֵי שַׁבַּיָּא דִּילִי תִּטְּרוּן:  אֲנָא, יְיָ אֱלָהֲכוֹן. 
יט,ד אַל־תִּפְנוּ֙ אֶל־הָ֣אֱלִילִ֔ם וֵֽאלֹהֵי֙ מַסֵּכָ֔ה לֹ֥א תַֽעֲשׂ֖וּ לָכֶ֑ם אֲנִ֖י יְהוָ֥ה אֱלֹֽהֵיכֶֽם׃ לָא תִּתְפְּנוֹן, בָּתַר טָעֲוָן, וְדַחְלָן דְּמַתְּכָא, לָא תַּעְבְּדוּן לְכוֹן:  אֲנָא, יְיָ אֱלָהֲכוֹן. 
יט,ה וְכִ֧י תִזְבְּח֛וּ זֶ֥בַח שְׁלָמִ֖ים לַֽיהוָ֑ה לִֽרְצֹנְכֶ֖ם תִּזְבָּחֻֽהוּ׃ וַאֲרֵי תִּכְּסוּן נִכְסַת קֻדְשִׁין, קֳדָם יְיָ--לְרַעֲוָא לְכוֹן, תִּכְּסוּנֵּיהּ. 
יט,ו בְּי֧וֹם זִבְחֲכֶ֛ם יֵֽאָכֵ֖ל וּמִֽמָּחֳרָ֑ת וְהַנּוֹתָר֙ עַד־י֣וֹם הַשְּׁלִישִׁ֔י בָּאֵ֖שׁ יִשָּׂרֵֽף׃ בְּיוֹמָא דְּיִתְנְכֵיס יִתְאֲכִיל, וּבְיוֹמָא דְּבָתְרוֹהִי; וּדְיִשְׁתְּאַר עַד יוֹמָא תְּלִיתָאָה, בְּנוּרָא יִתּוֹקַד. 
יט,ז וְאִ֛ם הֵֽאָכֹ֥ל יֵֽאָכֵ֖ל בַּיּ֣וֹם הַשְּׁלִישִׁ֑י פִּגּ֥וּל ה֖וּא לֹ֥א יֵֽרָצֶֽה׃ וְאִם אִתְאֲכָלָא יִתְאֲכִיל, בְּיוֹמָא תְּלִיתָאָה--מְרַחַק הוּא, לָא יְהֵי לְרַעֲוָא. 
יט,ח וְאֹֽכְלָיו֙ עֲו‍ֹנ֣וֹ יִשָּׂ֔א כִּֽי־אֶת־קֹ֥דֶשׁ יְהוָ֖ה חִלֵּ֑ל וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מֵֽעַמֶּֽיהָ׃ וּדְיֵיכְלִנֵּיהּ חוֹבֵיהּ יְקַבֵּיל, אֲרֵי יָת קֻדְשָׁא דַּייָ אַחֵיל; וְיִשְׁתֵּיצֵי אֲנָשָׁא הַהוּא, מֵעַמֵּיהּ. 
יט,ט וּֽבְקֻצְרְכֶם֙ אֶת־קְצִ֣יר אַרְצְכֶ֔ם לֹ֧א תְכַלֶּ֛ה פְּאַ֥ת שָֽׂדְךָ֖ לִקְצֹ֑ר וְלֶ֥קֶט קְצִֽירְךָ֖ לֹ֥א תְלַקֵּֽט׃ וּבְמִחְצַדְכוֹן יָת חֲצָדָא דַּאֲרַעְכוֹן, לָא תְּשֵׁיצֵי פָּתָא דְּחַקְלָךְ לְמִחְצַד; וּלְקָטָא דַּחֲצָדָךְ, לָא תְּלַקֵּיט. 
יט,י וְכַרְמְךָ֙ לֹ֣א תְעוֹלֵ֔ל וּפֶ֥רֶט כַּרְמְךָ֖ לֹ֣א תְלַקֵּ֑ט לֶֽעָנִ֤י וְלַגֵּר֙ תַּֽעֲזֹ֣ב אֹתָ֔ם אֲנִ֖י יְהוָ֥ה אֱלֹֽהֵיכֶֽם׃ וְכַרְמָךְ לָא תְּעָלֵיל, וְנִתְרָא דְּכַרְמָךְ לָא תְּלַקֵּיט:  לְעַנְיֵי וּלְגִיּוֹרֵי תִּשְׁבּוֹק יָתְהוֹן, אֲנָא יְיָ אֱלָהֲכוֹן. 
יט,יא לֹ֖א תִּגְנֹ֑בוּ וְלֹֽא־תְכַחֲשׁ֥וּ וְלֹֽא־תְשַׁקְּר֖וּ אִ֥ישׁ בַּֽעֲמִיתֽוֹ׃ לָא, תִּגְנְבוּן; וְלָא תְּכַדְּבוּן וְלָא תְּשַׁקְּרוּן, אֲנָשׁ בְּחַבְרֵיהּ. 
יט,יב וְלֹֽא־תִשָּׁבְע֥וּ בִשְׁמִ֖י לַשָּׁ֑קֶר וְחִלַּלְתָּ֛ אֶת־שֵׁ֥ם אֱלֹהֶ֖יךָ אֲנִ֥י יְהוָֽה׃ וְלָא תִּשְׁתַּבְעוּן בִּשְׁמִי, לְשִׁקְרָא:  וְתַחֵיל יָת שְׁמָא דֶּאֱלָהָךְ, אֲנָא יְיָ. 
יט,יג לֹֽא־תַעֲשֹׁ֥ק אֶת־רֵֽעֲךָ֖ וְלֹ֣א תִגְזֹ֑ל לֹֽא־תָלִ֞ין פְּעֻלַּ֥ת שָׂכִ֛יר אִתְּךָ֖ עַד־בֹּֽקֶר׃ לָא תַּעְשׁוֹק יָת חַבְרָךְ, וְלָא תֵּינוֹס; לָא תְּבִית אַגְרָא דַּאֲגִירָא, לְוָתָךְ--עַד צַפְרָא. 
יט,יד לֹֽא־תְקַלֵּ֣ל חֵרֵ֔שׁ וְלִפְנֵ֣י עִוֵּ֔ר לֹ֥א תִתֵּ֖ן מִכְשֹׁ֑ל וְיָרֵ֥אתָ מֵּֽאֱלֹהֶ֖יךָ אֲנִ֥י יְהוָֽה׃ לָא תְּלוּט דְּלָא שָׁמַע--וּקְדָם דְּלָא חָזֵי, לָא תְּשִׂים תַּקְלָא; וְתִדְחַל מֵאֱלָהָךְ, אֲנָא יְיָ. 
יט,טו לֹֽא־תַעֲשׂ֥וּ עָ֨וֶל֙ בַּמִּשְׁפָּ֔ט לֹֽא־תִשָּׂ֣א פְנֵי־דָ֔ל וְלֹ֥א תֶהְדַּ֖ר פְּנֵ֣י גָד֑וֹל בְּצֶ֖דֶק תִּשְׁפֹּ֥ט עֲמִיתֶֽךָ׃ לָא תַּעְבְּדוּן שְׁקַר, בְּדִין--לָא תִּסַּב אַפֵּי מִסְכֵּינָא, וְלָא תֶּהְדַּר אַפֵּי רַבָּא:  בְּקֻשְׁטָא, תְּדִינֵיהּ לְחַבְרָךְ. 
יט,טז לֹֽא־תֵלֵ֤ךְ רָכִיל֙ בְּעַמֶּ֔יךָ לֹ֥א תַֽעֲמֹ֖ד עַל־דַּ֣ם רֵעֶ֑ךָ אֲנִ֖י יְהוָֽה׃ לָא תֵּיכוֹל קֻרְצִין בְּעַמָּךְ, לָא תְּקוּם עַל דְּמָא דְּחַבְרָךְ:  אֲנָא, יְיָ. 
יט,יז לֹֽא־תִשְׂנָ֥א אֶת־אָחִ֖יךָ בִּלְבָבֶ֑ךָ הוֹכֵ֤חַ תּוֹכִ֨יחַ֙ אֶת־עֲמִיתֶ֔ךָ וְלֹֽא־תִשָּׂ֥א עָלָ֖יו חֵֽטְא׃ לָא תִּסְנֵי יָת אֲחוּךְ, בְּלִבָּךְ; אוֹכָחָא תּוֹכַח יָת חַבְרָךְ, וְלָא תְּקַבֵּיל עַל דִּילֵיהּ חוֹבָא. 
יט,יח לֹֽא־תִקֹּ֤ם וְלֹֽא־תִטֹּר֙ אֶת־בְּנֵ֣י עַמֶּ֔ךָ וְאָֽהַבְתָּ֥ לְרֵֽעֲךָ֖ כָּמ֑וֹךָ אֲנִ֖י יְהוָֽה׃ לָא תִּקּוֹם וְלָא תִּטַּר דְּבָבוּ לִבְנֵי עַמָּךְ, וְתִרְחֲמֵיהּ לְחַבְרָךְ כְּוָתָךְ:  אֲנָא, יְיָ. 
יט,יט אֶֽת־חֻקֹּתַי֮ תִּשְׁמֹרוּ֒ בְּהֶמְתְּךָ֙ לֹֽא־תַרְבִּ֣יעַ כִּלְאַ֔יִם שָֽׂדְךָ֖ לֹֽא־תִזְרַ֣ע כִּלְאָ֑יִם וּבֶ֤גֶד כִּלְאַ֨יִם֙ שַֽׁעַטְנֵ֔ז לֹ֥א יַֽעֲלֶ֖ה עָלֶֽיךָ׃ יָת קְיָמַי, תִּטְּרוּן--בְּעִירָךְ לָא תַּרְכֵּיב עֵירוּבִין, חַקְלָךְ לָא תִּזְרַע עֵירוּבִין; וּלְבוּשׁ עֵירוּבִין שַׁעַטְנֵיזָא, לָא יִסַּק עֲלָךְ. 
יט,כ וְ֠אִישׁ כִּֽי־יִשְׁכַּ֨ב אֶת־אִשָּׁ֜ה שִׁכְבַת־זֶ֗רַע וְהִ֤וא שִׁפְחָה֙ נֶֽחֱרֶ֣פֶת לְאִ֔ישׁ וְהָפְדֵּה֙ לֹ֣א נִפְדָּ֔תָה א֥וֹ חֻפְשָׁ֖ה לֹ֣א נִתַּן־לָ֑הּ בִּקֹּ֧רֶת תִּֽהְיֶ֛ה לֹ֥א יֽוּמְת֖וּ כִּי־לֹ֥א חֻפָּֽשָׁה׃ וּגְבַר אֲרֵי יִשְׁכּוֹב יָת אִתְּתָא שִׁכְבַת זַרְעָא, וְהִיא אָמָא אֲחִידָא לִגְבַר, וְאִתְפְּרָקָא לָא אִתְפְּרֵיקַת בְּכַסְפָּא, אוֹ חֵירוּתָא לָא אִתְיְהֵיבַת לַהּ בִּשְׁטָר--בִּקֻּרְתָּא תְּהֵי בַּהּ לָא יְמוּתוּן, אֲרֵי לָא אִתְחָרַרַת. 
יט,כא וְהֵבִ֤יא אֶת־אֲשָׁמוֹ֙ לַֽיהוָ֔ה אֶל־פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד אֵ֖יל אָשָֽׁם׃ וְיַיְתֵי יָת אֲשָׁמֵיהּ לִקְדָם יְיָ, לִתְרַע מַשְׁכַּן זִמְנָא--דִּכְרָא, לַאֲשָׁמָא. 
יט,כב וְכִפֶּר֩ עָלָ֨יו הַכֹּהֵ֜ן בְּאֵ֤יל הָֽאָשָׁם֙ לִפְנֵ֣י יְהוָ֔ה עַל־חַטָּאת֖וֹ אֲשֶׁ֣ר חָטָ֑א וְנִסְלַ֣ח ל֔וֹ מֵֽחַטָּאת֖וֹ אֲשֶׁ֥ר חָטָֽא׃ {פ} וִיכַפַּר עֲלוֹהִי כָּהֲנָא בְּדִכְרָא דַּאֲשָׁמָא, קֳדָם יְיָ, עַל חוֹבְתֵיהּ, דְּחָב; וְיִשְׁתְּבֵיק לֵיהּ, מֵחוֹבְתֵיהּ דְּחָב. {פ}
יט,כג וְכִֽי־תָבֹ֣אוּ אֶל־הָאָ֗רֶץ וּנְטַעְתֶּם֙ כָּל־עֵ֣ץ מַֽאֲכָ֔ל וַֽעֲרַלְתֶּ֥ם עָרְלָת֖וֹ אֶת־פִּרְי֑וֹ שָׁלֹ֣שׁ שָׁנִ֗ים יִֽהְיֶ֥ה לָכֶ֛ם עֲרֵלִ֖ים לֹ֥א יֵֽאָכֵֽל׃ וַאֲרֵי תֵּיעֲלוּן לְאַרְעָא, וְתִצְּבוּן כָּל אִילָן דְּמֵיכַל--וּתְרַחֲקוּן רַחָקָא, יָת אִבֵּיהּ; תְּלָת שְׁנִין, יְהֵי לְכוֹן מְרַחַק לְאַבָּדָא--לָא יִתְאֲכִיל. 
יט,כד וּבַשָּׁנָה֙ הָֽרְבִיעִ֔ת יִֽהְיֶ֖ה כָּל־פִּרְי֑וֹ קֹ֥דֶשׁ הִלּוּלִ֖ים לַֽיהוָֽה׃ וּבְשַׁתָּא, רְבִיעֵיתָא, יְהֵי, כָּל אִבֵּיהּ--קֹדֶשׁ תֻּשְׁבְּחָן, קֳדָם יְיָ. 
יט,כה וּבַשָּׁנָ֣ה הַֽחֲמִישִׁ֗ת תֹּֽאכְלוּ֙ אֶת־פִּרְי֔וֹ לְהוֹסִ֥יף לָכֶ֖ם תְּבֽוּאָת֑וֹ אֲנִ֖י יְהוָ֥ה אֱלֹֽהֵיכֶֽם׃ וּבְשַׁתָּא חֲמִישֵׁיתָא, תֵּיכְלוּן יָת אִבֵּיהּ, לְאוֹסָפָא לְכוֹן, עֲלַלְתֵּיהּ:  אֲנָא, יְיָ אֱלָהֲכוֹן. 
יט,כו לֹ֥א תֹֽאכְל֖וּ עַל־הַדָּ֑ם לֹ֥א תְנַֽחֲשׁ֖וּ וְלֹ֥א תְעוֹנֵֽנוּ׃ לָא תֵּיכְלוּן, עַל דְּמָא; לָא תְּנַחֲשׁוּן, וְלָא תְּעָנוֹן. 
יט,כז לֹ֣א תַקִּ֔פוּ פְּאַ֖ת רֹֽאשְׁכֶ֑ם וְלֹ֣א תַשְׁחִ֔ית אֵ֖ת פְּאַ֥ת זְקָנֶֽךָ׃ לָא תַּקְּפוּן, פָּתָא דְּרֵישְׁכוֹן; וְלָא תְּחַבֵּיל, יָת פָּתָא דְּדִקְנָךְ. 
יט,כח וְשֶׂ֣רֶט לָנֶ֗פֶשׁ לֹ֤א תִתְּנוּ֙ בִּבְשַׂרְכֶ֔ם וּכְתֹ֣בֶת קַֽעֲקַ֔ע לֹ֥א תִתְּנ֖וּ בָּכֶ֑ם אֲנִ֖י יְהוָֽה׃ וְחִבּוּל עַל מִית, לָא תִּתְּנוּן בִּבְסַרְכוֹן, וְרֻשְׁמִין חֲרִיתִין, לָא תִּתְּנוּן בְּכוֹן:  אֲנָא, יְיָ. 
יט,כט אַל־תְּחַלֵּ֥ל אֶֽת־בִּתְּךָ֖ לְהַזְנוֹתָ֑הּ וְלֹֽא־תִזְנֶ֣ה הָאָ֔רֶץ וּמָֽלְאָ֥ה הָאָ֖רֶץ זִמָּֽה׃ לָא תַּחֵיל יָת בְּרַתָּךְ, לְאַטְעָיוּתַהּ; וְלָא תִּטְעֵי אַרְעָא, וְתִתְמְלֵי אַרְעָא עֵיצַת חֲטִין. 
יט,ל אֶת־שַׁבְּתֹתַ֣י תִּשְׁמֹ֔רוּ וּמִקְדָּשִׁ֖י תִּירָ֑אוּ אֲנִ֖י יְהוָֽה׃ יָת יוֹמֵי שַׁבַּיָּא דִּילִי תִּטְּרוּן, וּלְבֵית מַקְדְּשִׁי תְּהוֹן דָּחֲלִין:  אֲנָא, יְיָ. 
יט,לא אַל־תִּפְנ֤וּ אֶל־הָֽאֹבֹת֙ וְאֶל־הַיִּדְּעֹנִ֔ים אַל־תְּבַקְשׁ֖וּ לְטָמְאָ֣ה בָהֶ֑ם אֲנִ֖י יְהוָ֥ה אֱלֹֽהֵיכֶֽם׃ לָא תִּתְפְּנוֹן בָּתַר בִּדִּין וּזְכוּרוּ, לָא תִּתְבְּעוּן לְאִסְתַּאָבָא בְּהוֹן:  אֲנָא, יְיָ אֱלָהֲכוֹן. 
יט,לב מִפְּנֵ֤י שֵׂיבָה֙ תָּק֔וּם וְהָֽדַרְתָּ֖ פְּנֵ֣י זָקֵ֑ן וְיָרֵ֥אתָ מֵּֽאֱלֹהֶ֖יךָ אֲנִ֥י יְהוָֽה׃ {ס} מִן קֳדָם דְּסָבַר בְּאוֹרָיְתָא תְּקוּם, וְתֶהְדַּר אַפֵּי סָבָא; וְתִדְחַל מֵאֱלָהָךְ, אֲנָא יְיָ. {ס}
יט,לג וְכִֽי־יָג֧וּר אִתְּךָ֛ גֵּ֖ר בְּאַרְצְכֶ֑ם לֹ֥א תוֹנ֖וּ אֹתֽוֹ׃ וַאֲרֵי יִתְגַּיַּר עִמְּכוֹן גִּיּוֹרָא, בַּאֲרַעְכוֹן--לָא תּוֹנוֹן, יָתֵיהּ. 
יט,לד כְּאֶזְרָ֣ח מִכֶּם֩ יִֽהְיֶ֨ה לָכֶ֜ם הַגֵּ֣ר ׀ הַגָּ֣ר אִתְּכֶ֗ם וְאָֽהַבְתָּ֥ לוֹ֙ כָּמ֔וֹךָ כִּֽי־גֵרִ֥ים הֱיִיתֶ֖ם בְּאֶ֣רֶץ מִצְרָ֑יִם אֲנִ֖י יְהוָ֥ה אֱלֹֽהֵיכֶֽם׃ כְּיַצִּיבָא מִנְּכוֹן יְהֵי לְכוֹן גִּיּוֹרָא דְּיִתְגַּיַּר עִמְּכוֹן, וְתִרְחַם לֵיהּ כְּוָתָךְ--אֲרֵי דַּיָּרִין הֲוֵיתוֹן, בְּאַרְעָא דְּמִצְרָיִם:  אֲנָא, יְיָ אֱלָהֲכוֹן. 
יט,לה לֹֽא־תַעֲשׂ֥וּ עָ֖וֶל בַּמִּשְׁפָּ֑ט בַּמִּדָּ֕ה בַּמִּשְׁקָ֖ל וּבַמְּשׂוּרָֽה׃ לָא תַּעְבְּדוּן שְׁקַר, בְּדִין, בְּמִשְׁחֲתָא, בְּמַתְקָלָא וּבִמְכִילְתָא. 
יט,לו מֹ֧אזְנֵי צֶ֣דֶק אַבְנֵי־צֶ֗דֶק אֵ֥יפַת צֶ֛דֶק וְהִ֥ין צֶ֖דֶק יִֽהְיֶ֣ה לָכֶ֑ם אֲנִי֙ יְהוָ֣ה אֱלֹֽהֵיכֶ֔ם אֲשֶׁר־הוֹצֵ֥אתִי אֶתְכֶ֖ם מֵאֶ֥רֶץ מִצְרָֽיִם׃ מוֹזְנָוָן דִּקְשׁוֹט מַתְקָלִין דִּקְשׁוֹט, מְכִילָן דִּקְשׁוֹט וְהִינִין דִּקְשׁוֹט--יְהוֹן לְכוֹן:  אֲנָא יְיָ אֱלָהֲכוֹן, דְּאַפֵּיקִית יָתְכוֹן מֵאַרְעָא דְּמִצְרָיִם. 
יט,לז וּשְׁמַרְתֶּ֤ם אֶת־כָּל־חֻקֹּתַי֙ וְאֶת־כָּל־מִשְׁפָּטַ֔י וַֽעֲשִׂיתֶ֖ם אֹתָ֑ם אֲנִ֖י יְהוָֽה׃ {פ} וְתִטְּרוּן יָת כָּל קְיָמַי וְיָת כָּל דִּינַי, וְתַעְבְּדוּן יָתְהוֹן:  אֲנָא, יְיָ. {פ}
כ,א וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
כ,ב וְאֶל־בְּנֵ֣י יִשְׂרָאֵל֮ תֹּאמַר֒ אִ֣ישׁ אִישׁ֩ מִבְּנֵ֨י יִשְׂרָאֵ֜ל וּמִן־הַגֵּ֣ר ׀ הַגָּ֣ר בְּיִשְׂרָאֵ֗ל אֲשֶׁ֨ר יִתֵּ֧ן מִזַּרְע֛וֹ לַמֹּ֖לֶךְ מ֣וֹת יוּמָ֑ת עַ֥ם הָאָ֖רֶץ יִרְגְּמֻ֥הוּ בָאָֽבֶן׃ וְעִם בְּנֵי יִשְׂרָאֵל, תֵּימַר, גְּבַר גְּבַר מִבְּנֵי יִשְׂרָאֵל וּמִן גִּיּוֹרַיָּא דְּיִתְגַּיְּרוּן בְּיִשְׂרָאֵל דְּיִתֵּין מִזַּרְעֵיהּ לְמֹלֶךְ, אִתְקְטָלָא יִתְקְטִיל; עַמָּא בֵּית יִשְׂרָאֵל, יִרְגְּמוּנֵּיהּ בְּאַבְנָא. 
כ,ג וַֽאֲנִ֞י אֶתֵּ֤ן אֶת־פָּנַי֙ בָּאִ֣ישׁ הַה֔וּא וְהִכְרַתִּ֥י אֹת֖וֹ מִקֶּ֣רֶב עַמּ֑וֹ כִּ֤י מִזַּרְעוֹ֙ נָתַ֣ן לַמֹּ֔לֶךְ לְמַ֗עַן טַמֵּא֙ אֶת־מִקְדָּשִׁ֔י וּלְחַלֵּ֖ל אֶת־שֵׁ֥ם קָדְשִֽׁי׃ וַאֲנָא אֶתֵּין יָת רֻגְזִי, בְּגֻבְרָא הַהוּא, וַאֲשֵׁיצֵי יָתֵיהּ, מִגּוֹ עַמֵּיהּ:  אֲרֵי מִזַּרְעֵיהּ, יְהַב לְמֹלֶךְ--בְּדִיל לְסַאָבָא יָת מַקְדְּשִׁי, וּלְאַחָלָא יָת שְׁמָא דְּקֻדְשִׁי. 
כ,ד וְאִ֡ם הַעְלֵ֣ם יַעְלִימֽוּ֩ עַ֨ם הָאָ֜רֶץ אֶת־עֵֽינֵיהֶם֙ מִן־הָאִ֣ישׁ הַה֔וּא בְּתִתּ֥וֹ מִזַּרְע֖וֹ לַמֹּ֑לֶךְ לְבִלְתִּ֖י הָמִ֥ית אֹתֽוֹ׃ וְאִם מִכְבָּשׁ יִכְבְּשׁוּן עַמָּא בֵּית יִשְׂרָאֵל יָת עֵינֵיהוֹן, מִן גֻּבְרָא הַהוּא, בְּדִיהַב מִזַּרְעֵיהּ, לְמֹלֶךְ--בְּדִיל דְּלָא, לְמִקְטַל יָתֵיהּ. 
כ,ה וְשַׂמְתִּ֨י אֲנִ֧י אֶת־פָּנַ֛י בָּאִ֥ישׁ הַה֖וּא וּבְמִשְׁפַּחְתּ֑וֹ וְהִכְרַתִּ֨י אֹת֜וֹ וְאֵ֣ת ׀ כָּל־הַזֹּנִ֣ים אַֽחֲרָ֗יו לִזְנ֛וֹת אַֽחֲרֵ֥י הַמֹּ֖לֶךְ מִקֶּ֥רֶב עַמָּֽם׃ וַאֲשַׁוֵּי אֲנָא יָת רֻגְזִי בְּגֻבְרָא הַהוּא, וּבְסָעֲדוֹהִי; וַאֲשֵׁיצֵי יָתֵיהּ וְיָת כָּל דְּטָעַן בָּתְרוֹהִי, לְמִטְעֵי בָּתַר מֹלֶךְ--מִגּוֹ עַמְּהוֹן. 
כ,ו וְהַנֶּ֗פֶשׁ אֲשֶׁ֨ר תִּפְנֶ֤ה אֶל־הָֽאֹבֹת֙ וְאֶל־הַיִּדְּעֹנִ֔ים לִזְנֹ֖ת אַֽחֲרֵיהֶ֑ם וְנָֽתַתִּ֤י אֶת־פָּנַי֙ בַּנֶּ֣פֶשׁ הַהִ֔וא וְהִכְרַתִּ֥י אֹת֖וֹ מִקֶּ֥רֶב עַמּֽוֹ׃ וַאֲנָשׁ, דְּיִתְפְּנֵי בָּתַר בִּדִּין וּזְכוּרוּ, לְמִטְעֵי, בָּתְרֵיהוֹן--וְאֶתֵּין יָת רֻגְזִי בַּאֲנָשָׁא הַהוּא, וַאֲשֵׁיצֵי יָתֵיהּ מִגּוֹ עַמֵּיהּ. 
כ,ז וְהִ֨תְקַדִּשְׁתֶּ֔ם וִֽהְיִיתֶ֖ם קְדֹשִׁ֑ים כִּ֛י אֲנִ֥י יְהוָ֖ה אֱלֹֽהֵיכֶֽם׃ וְתִתְקַדְּשׁוּן--וּתְהוֹן, קַדִּישִׁין:  אֲרֵי אֲנָא יְיָ, אֱלָהֲכוֹן. 
כ,ח וּשְׁמַרְתֶּם֙ אֶת־חֻקֹּתַ֔י וַֽעֲשִׂיתֶ֖ם אֹתָ֑ם אֲנִ֥י יְהוָ֖ה מְקַדִּשְׁכֶֽם׃ וְתִטְּרוּן, יָת קְיָמַי, וְתַעְבְּדוּן, יָתְהוֹן:  אֲנָא יְיָ, מְקַדִּשְׁכוֹן. 
כ,ט כִּֽי־אִ֣ישׁ אִ֗ישׁ אֲשֶׁ֨ר יְקַלֵּ֧ל אֶת־אָבִ֛יו וְאֶת־אִמּ֖וֹ מ֣וֹת יוּמָ֑ת אָבִ֧יו וְאִמּ֛וֹ קִלֵּ֖ל דָּמָ֥יו בּֽוֹ׃ אֲרֵי גְּבַר גְּבַר, דִּילוּט יָת אֲבוּהִי וְיָת אִמֵּיהּ--אִתְקְטָלָא יִתְקְטִיל:  אֲבוּהִי וְאִמֵּיהּ לָט, קַטְלָא חַיָּב. 
כ,י וְאִ֗ישׁ אֲשֶׁ֤ר יִנְאַף֙ אֶת־אֵ֣שֶׁת אִ֔ישׁ אֲשֶׁ֥ר יִנְאַ֖ף אֶת־אֵ֣שֶׁת רֵעֵ֑הוּ מֽוֹת־יוּמַ֥ת הַנֹּאֵ֖ף וְהַנֹּאָֽפֶת׃ וּגְבַר, דִּיגוּף יָת אִתַּת גְּבַר, דִּיגוּף, יָת אִתַּת חַבְרֵיהּ--אִתְקְטָלָא יִתְקְטִיל גַּיָּפָא, וְגַיָּפְתָא. 
כ,יא וְאִ֗ישׁ אֲשֶׁ֤ר יִשְׁכַּב֙ אֶת־אֵ֣שֶׁת אָבִ֔יו עֶרְוַ֥ת אָבִ֖יו גִּלָּ֑ה מֽוֹת־יוּמְת֥וּ שְׁנֵיהֶ֖ם דְּמֵיהֶ֥ם בָּֽם׃ וּגְבַר, דְּיִשְׁכּוֹב יָת אִתַּת אֲבוּהִי--עֶרְיְתָא דַּאֲבוּהִי, גַּלִּי; אִתְקְטָלָא יִתְקַטְלוּן תַּרְוֵיהוֹן, קַטְלָא חַיָּבִין. 
כ,יב וְאִ֗ישׁ אֲשֶׁ֤ר יִשְׁכַּב֙ אֶת־כַּלָּת֔וֹ מ֥וֹת יֽוּמְת֖וּ שְׁנֵיהֶ֑ם תֶּ֥בֶל עָשׂ֖וּ דְּמֵיהֶ֥ם בָּֽם׃ וּגְבַר, דְּיִשְׁכּוֹב יָת כַּלְּתֵיהּ--אִתְקְטָלָא יִתְקַטְלוּן, תַּרְוֵיהוֹן:  תָּבְלָא עֲבַדוּ, קַטְלָא חַיָּבִין. 
כ,יג וְאִ֗ישׁ אֲשֶׁ֨ר יִשְׁכַּ֤ב אֶת־זָכָר֙ מִשְׁכְּבֵ֣י אִשָּׁ֔ה תּֽוֹעֵבָ֥ה עָשׂ֖וּ שְׁנֵיהֶ֑ם מ֥וֹת יוּמָ֖תוּ דְּמֵיהֶ֥ם בָּֽם׃ וּגְבַר, דְּיִשְׁכּוֹב יָת דְּכוּרָא מִשְׁכְּבֵי אִתָּא--תּוֹעֵיבָא עֲבַדוּ, תַּרְוֵיהוֹן; אִתְקְטָלָא יִתְקַטְלוּן, קַטְלָא חַיָּבִין. 
כ,יד וְאִ֗ישׁ אֲשֶׁ֨ר יִקַּ֧ח אֶת־אִשָּׁ֛ה וְאֶת־אִמָּ֖הּ זִמָּ֣ה הִ֑וא בָּאֵ֞שׁ יִשְׂרְפ֤וּ אֹתוֹ֙ וְאֶתְהֶ֔ן וְלֹֽא־תִהְיֶ֥ה זִמָּ֖ה בְּתֽוֹכְכֶֽם׃ וּגְבַר, דְּיִסַּב יָת אִתְּתָא וְיָת אִמַּהּ--עֵיצַת חֲטִין הִיא; בְּנוּרָא יוֹקְדוּן יָתֵיהּ, וְיָתְהוֹן, וְלָא תְּהֵי עֵיצַת חֲטִין, בֵּינֵיכוֹן. 
כ,טו וְאִ֗ישׁ אֲשֶׁ֨ר יִתֵּ֧ן שְׁכָבְתּ֛וֹ בִּבְהֵמָ֖ה מ֣וֹת יוּמָ֑ת וְאֶת־הַבְּהֵמָ֖ה תַּֽהֲרֹֽגוּ׃ וּגְבַר, דְּיִתֵּין שְׁכֻבְתֵּיהּ בִּבְעִירָא--אִתְקְטָלָא יִתְקְטִיל; וְיָת בְּעִירָא, תִּקְטְלוּן. 
כ,טז וְאִשָּׁ֗ה אֲשֶׁ֨ר תִּקְרַ֤ב אֶל־כָּל־בְּהֵמָה֙ לְרִבְעָ֣ה אֹתָ֔הּ וְהָֽרַגְתָּ֥ אֶת־הָֽאִשָּׁ֖ה וְאֶת־הַבְּהֵמָ֑ה מ֥וֹת יוּמָ֖תוּ דְּמֵיהֶ֥ם בָּֽם׃ וְאִתְּתָא, דְּתִקְרַב לְוָת כָּל בְּעִירָא לְמִשְׁלַט בַּהּ--וְתִקְטוֹל יָת אִתְּתָא, וְיָת בְּעִירָא; אִתְקְטָלָא יִתְקַטְלוּן, קַטְלָא חַיָּבִין. 
כ,יז וְאִ֣ישׁ אֲשֶׁר־יִקַּ֣ח אֶת־אֲחֹת֡וֹ בַּת־אָבִ֣יו א֣וֹ בַת־אִ֠מּוֹ וְרָאָ֨ה אֶת־עֶרְוָתָ֜הּ וְהִֽיא־תִרְאֶ֤ה אֶת־עֶרְוָתוֹ֙ חֶ֣סֶד ה֔וּא וְנִ֨כְרְת֔וּ לְעֵינֵ֖י בְּנֵ֣י עַמָּ֑ם עֶרְוַ֧ת אֲחֹת֛וֹ גִּלָּ֖ה עֲו‍ֹנ֥וֹ יִשָּֽׂא׃ וּגְבַר דְּיִסַּב יָת אֲחָתֵיהּ בַּת אֲבוּהִי אוֹ בַּת אִמֵּיהּ וְיִחְזֵי יָת עֶרְיְתַהּ וְהִיא תִּחְזֵי יָת עֶרְיְתֵיהּ, קְלָנָא הוּא--וְיִשְׁתֵּיצוֹן, לְעֵינֵי בְּנֵי עַמְּהוֹן; עֶרְיַת אֲחָתֵיהּ גַּלִּי, חוֹבֵיהּ יְקַבֵּיל. 
כ,יח וְ֠אִישׁ אֲשֶׁר־יִשְׁכַּ֨ב אֶת־אִשָּׁ֜ה דָּוָ֗ה וְגִלָּ֤ה אֶת־עֶרְוָתָהּ֙ אֶת־מְקֹרָ֣הּ הֶֽעֱרָ֔ה וְהִ֕וא גִּלְּתָ֖ה אֶת־מְק֣וֹר דָּמֶ֑יהָ וְנִכְרְת֥וּ שְׁנֵיהֶ֖ם מִקֶּ֥רֶב עַמָּֽם׃ וּגְבַר דְּיִשְׁכּוֹב יָת אִתָּא טֻמְאָה, וִיגַלֵּי יָת עֶרְיְתַהּ יָת קְלָנַהּ גַּלִּי, וְהִיא, תְּגַלֵּי יָת סוֹאֲבָת דְּמַהָא--וְיִשְׁתֵּיצוֹן תַּרְוֵיהוֹן, מִגּוֹ עַמְּהוֹן. 
כ,יט וְעֶרְוַ֨ת אֲח֧וֹת אִמְּךָ֛ וַֽאֲח֥וֹת אָבִ֖יךָ לֹ֣א תְגַלֵּ֑ה כִּ֧י אֶת־שְׁאֵר֛וֹ הֶֽעֱרָ֖ה עֲו‍ֹנָ֥ם יִשָּֽׂאוּ׃ וְעֶרְיַת אֲחָת אִמָּךְ וַאֲחָת אֲבוּךְ, לָא תְּגַלֵּי:  אֲרֵי יָת קָרִיבְתֵיהּ גַּלִּי, חוֹבְהוֹן יְקַבְּלוּן. 
כ,כ וְאִ֗ישׁ אֲשֶׁ֤ר יִשְׁכַּב֙ אֶת־דֹּ֣דָת֔וֹ עֶרְוַ֥ת דֹּד֖וֹ גִּלָּ֑ה חֶטְאָ֥ם יִשָּׂ֖אוּ עֲרִירִ֥ים יָמֻֽתוּ׃ וּגְבַר, דְּיִשְׁכּוֹב יָת אִתַּת אַחְבּוּהִי--עֶרְיְתָא דְּאַחְבּוּהִי, גַּלִּי; חוֹבְהוֹן יְקַבְּלוּן, דְּלָא וְלַד יְמוּתוּן. 
כ,כא וְאִ֗ישׁ אֲשֶׁ֥ר יִקַּ֛ח אֶת־אֵ֥שֶׁת אָחִ֖יו נִדָּ֣ה הִ֑וא עֶרְוַ֥ת אָחִ֛יו גִּלָּ֖ה עֲרִירִ֥ים יִֽהְיֽוּ׃ וּגְבַר, דְּיִסַּב יָת אִתַּת אֲחוּהִי--מְרַחֲקָא הִיא; עֶרְיְתָא דַּאֲחוּהִי גַּלִּי, דְּלָא וְלַד יְהוֹן. 
כ,כב וּשְׁמַרְתֶּ֤ם אֶת־כָּל־חֻקֹּתַי֙ וְאֶת־כָּל־מִשְׁפָּטַ֔י וַֽעֲשִׂיתֶ֖ם אֹתָ֑ם וְלֹֽא־תָקִ֤יא אֶתְכֶם֙ הָאָ֔רֶץ אֲשֶׁ֨ר אֲנִ֜י מֵבִ֥יא אֶתְכֶ֛ם שָׁ֖מָּה לָשֶׁ֥בֶת בָּֽהּ׃ וְתִטְּרוּן יָת כָּל קְיָמַי וְיָת כָּל דִּינַי, וְתַעְבְּדוּן יָתְהוֹן; וְלָא תְּרוֹקֵין יָתְכוֹן, אַרְעָא, דַּאֲנָא מַעֵיל יָתְכוֹן לְתַמָּן, לְמִתַּב בַּהּ. 
כ,כג וְלֹ֤א תֵֽלְכוּ֙ בְּחֻקֹּ֣ת הַגּ֔וֹי אֲשֶׁר־אֲנִ֥י מְשַׁלֵּ֖חַ מִפְּנֵיכֶ֑ם כִּ֤י אֶת־כָּל־אֵ֨לֶּה֙ עָשׂ֔וּ וָֽאָקֻ֖ץ בָּֽם׃ וְלָא תְּהָכוּן בְּנִמּוֹסֵי עַמְמַיָּא, דַּאֲנָא מַגְלֵי מִן קֳדָמֵיכוֹן:  אֲרֵי יָת כָּל אִלֵּין עֲבַדוּ, וְרַחֵיק מֵימְרִי יָתְהוֹן. 
כ,כד וָֽאֹמַ֣ר לָכֶ֗ם אַתֶּם֮ תִּֽירְשׁ֣וּ אֶת־אַדְמָתָם֒ וַֽאֲנִ֞י אֶתְּנֶ֤נָּה לָכֶם֙ לָרֶ֣שֶׁת אֹתָ֔הּ אֶ֛רֶץ זָבַ֥ת חָלָ֖ב וּדְבָ֑שׁ אֲנִי֙ יְהוָ֣ה אֱלֹֽהֵיכֶ֔ם אֲשֶׁר־הִבְדַּ֥לְתִּי אֶתְכֶ֖ם מִן־הָֽעַמִּֽים׃ וַאֲמַרִית לְכוֹן, אַתּוּן תֵּירְתוּן יָת אֲרַעְהוֹן, וַאֲנָא אֶתְּנִנַּהּ לְכוֹן לְמֵירַת יָתַהּ, אֲרַע עָבְדָא חֲלָב וּדְבָשׁ:  אֲנָא יְיָ אֱלָהֲכוֹן, דְּאַפְרֵישִׁית יָתְכוֹן מִן עַמְמַיָּא. 
כ,כה וְהִבְדַּלְתֶּ֞ם בֵּֽין־הַבְּהֵמָ֤ה הַטְּהֹרָה֙ לַטְּמֵאָ֔ה וּבֵין־הָע֥וֹף הַטָּמֵ֖א לַטָּהֹ֑ר וְלֹֽא־תְשַׁקְּצ֨וּ אֶת־נַפְשֹֽׁתֵיכֶ֜ם בַּבְּהֵמָ֣ה וּבָע֗וֹף וּבְכֹל֙ אֲשֶׁ֣ר תִּרְמֹ֣שׂ הָֽאֲדָמָ֔ה אֲשֶׁר־הִבְדַּ֥לְתִּי לָכֶ֖ם לְטַמֵּֽא׃ וְתַפְרְשׁוּן בֵּין בְּעִירָא דָּכְיָא, לִמְסָאֲבָא, וּבֵין עוֹפָא מְסָאֲבָא, לְדָכְיָא; וְלָא תְּשַׁקְּצוּן יָת נַפְשָׁתְכוֹן בִּבְעִירָא וּבְעוֹפָא, וּבְכֹל דְּתַרְחֵישׁ אַרְעָא, דְּאַפְרֵישִׁית לְכוֹן, לְסַאָבָא. 
כ,כו וִֽהְיִ֤יתֶם לִי֙ קְדֹשִׁ֔ים כִּ֥י קָד֖וֹשׁ אֲנִ֣י יְהוָ֑ה וָֽאַבְדִּ֥ל אֶתְכֶ֛ם מִן־הָֽעַמִּ֖ים לִֽהְי֥וֹת לִֽי׃ וּתְהוֹן קֳדָמַי קַדִּישִׁין, אֲרֵי קַדִּישׁ אֲנָא יְיָ; וְאַפְרֵישִׁית יָתְכוֹן מִן עַמְמַיָּא, לְמִהְוֵי פָּלְחִין קֳדָמָי. 
כ,כז וְאִ֣ישׁ אֽוֹ־אִשָּׁ֗ה כִּֽי־יִהְיֶ֨ה בָהֶ֥ם א֛וֹב א֥וֹ יִדְּעֹנִ֖י מ֣וֹת יוּמָ֑תוּ בָּאֶ֛בֶן יִרְגְּמ֥וּ אֹתָ֖ם דְּמֵיהֶ֥ם בָּֽם׃ {פ} וּגְבַר אוֹ אִתָּא, אֲרֵי יְהֵי בְּהוֹן בִּדִּין אוֹ זְכוּרוּ--אִתְקְטָלָא יִתְקַטְלוּן; בְּאַבְנָא יִרְגְּמוּן יָתְהוֹן, קַטְלָא חַיָּבִין. {פ}
