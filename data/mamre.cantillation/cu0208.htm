תצווה

כז,כ וְאַתָּ֞ה תְּצַוֶּ֣ה ׀ אֶת־בְּנֵ֣י יִשְׂרָאֵ֗ל וְיִקְח֨וּ אֵלֶ֜יךָ שֶׁ֣מֶן זַ֥יִת זָ֛ךְ כָּתִ֖ית לַמָּא֑וֹר לְהַֽעֲלֹ֥ת נֵ֖ר תָּמִֽיד׃ וְאַתְּ תְּפַקֵּיד יָת בְּנֵי יִשְׂרָאֵל, וְיִסְּבוּן לָךְ מְשַׁח זֵיתָא דָּכְיָא כָּתִישָׁא--לְאַנְהָרָא:  לְאַדְלָקָא בּוֹצִינַיָּא, תְּדִירָא. 
כז,כא בְּאֹ֣הֶל מוֹעֵד֩ מִח֨וּץ לַפָּרֹ֜כֶת אֲשֶׁ֣ר עַל־הָֽעֵדֻ֗ת יַֽעֲרֹךְ֩ אֹת֨וֹ אַֽהֲרֹ֧ן וּבָנָ֛יו מֵעֶ֥רֶב עַד־בֹּ֖קֶר לִפְנֵ֣י יְהוָ֑ה חֻקַּ֤ת עוֹלָם֙ לְדֹ֣רֹתָ֔ם מֵאֵ֖ת בְּנֵ֥י יִשְׂרָאֵֽל׃ {ס} בְּמַשְׁכַּן זִמְנָא מִבַּרָא לְפָרֻכְתָּא דְּעַל סָהֲדוּתָא, יַסְדַּר יָתֵיהּ אַהֲרוֹן וּבְנוֹהִי מֵרַמְשָׁא עַד צַפְרָא--קֳדָם יְיָ:  קְיָם עָלַם לְדָרֵיהוֹן, מִן בְּנֵי יִשְׂרָאֵל. {ס}
כח,א וְאַתָּ֡ה הַקְרֵ֣ב אֵלֶיךָ֩ אֶת־אַֽהֲרֹ֨ן אָחִ֜יךָ וְאֶת־בָּנָ֣יו אִתּ֗וֹ מִתּ֛וֹךְ בְּנֵ֥י יִשְׂרָאֵ֖ל לְכַֽהֲנוֹ־לִ֑י אַֽהֲרֹ֕ן נָדָ֧ב וַֽאֲבִיה֛וּא אֶלְעָזָ֥ר וְאִֽיתָמָ֖ר בְּנֵ֥י אַֽהֲרֹֽן׃ וְאַתְּ קָרֵיב לְוָתָךְ יָת אַהֲרוֹן אֲחוּךְ וְיָת בְּנוֹהִי עִמֵּיהּ, מִגּוֹ בְּנֵי יִשְׂרָאֵל--לְשַׁמָּשָׁא קֳדָמָי:  אַהֲרוֹן--נָדָב וַאֲבִיהוּא אֶלְעָזָר וְאִיתָמָר, בְּנֵי אַהֲרוֹן. 
כח,ב וְעָשִׂ֥יתָ בִגְדֵי־קֹ֖דֶשׁ לְאַֽהֲרֹ֣ן אָחִ֑יךָ לְכָב֖וֹד וּלְתִפְאָֽרֶת׃ וְתַעֲבֵיד לְבוּשֵׁי קֻדְשָׁא, לְאַהֲרוֹן אֲחוּךְ, לִיקָר, וּלְתֻשְׁבְּחָא. 
כח,ג וְאַתָּ֗ה תְּדַבֵּר֙ אֶל־כָּל־חַכְמֵי־לֵ֔ב אֲשֶׁ֥ר מִלֵּאתִ֖יו ר֣וּחַ חָכְמָ֑ה וְעָשׂ֞וּ אֶת־בִּגְדֵ֧י אַֽהֲרֹ֛ן לְקַדְּשׁ֖וֹ לְכַֽהֲנוֹ־לִֽי׃ וְאַתְּ, תְּמַלֵּיל עִם כָּל חַכִּימֵי לִבָּא, דְּאַשְׁלֵימִית עִמְּהוֹן, רוּחַ חָכְמָא; וְיַעְבְּדוּן יָת לְבוּשֵׁי אַהֲרוֹן, לְקַדָּשׁוּתֵיהּ--לְשַׁמָּשָׁא קֳדָמָי. 
כח,ד וְאֵ֨לֶּה הַבְּגָדִ֜ים אֲשֶׁ֣ר יַֽעֲשׂ֗וּ חֹ֤שֶׁן וְאֵפוֹד֙ וּמְעִ֔יל וּכְתֹ֥נֶת תַּשְׁבֵּ֖ץ מִצְנֶ֣פֶת וְאַבְנֵ֑ט וְעָשׂ֨וּ בִגְדֵי־קֹ֜דֶשׁ לְאַֽהֲרֹ֥ן אָחִ֛יךָ וּלְבָנָ֖יו לְכַֽהֲנוֹ־לִֽי׃ וְאִלֵּין לְבוּשַׁיָּא דְּיַעְבְּדוּן, חֻשְׁנָא וְאֵיפוֹדָא וּמְעִילָא, וְכִתּוּנִין מְרַמְּצָן, מַצְנְפָן וְהִמְיָנִין; וְיַעְבְּדוּן לְבוּשֵׁי קֻדְשָׁא לְאַהֲרוֹן אֲחוּךְ, וְלִבְנוֹהִי--לְשַׁמָּשָׁא קֳדָמָי. 
כח,ה וְהֵם֙ יִקְח֣וּ אֶת־הַזָּהָ֔ב וְאֶת־הַתְּכֵ֖לֶת וְאֶת־הָֽאַרְגָּמָ֑ן וְאֶת־תּוֹלַ֥עַת הַשָּׁנִ֖י וְאֶת־הַשֵּֽׁשׁ׃ {פ} וְאִנּוּן יִסְּבוּן יָת דַּהְבָּא, וְיָת תַּכְלָא וְיָת אַרְגְּוָנָא, וְיָת צְבַע זְהוֹרִי, וְיָת בּוּצָא. {פ}
כח,ו וְעָשׂ֖וּ אֶת־הָֽאֵפֹ֑ד זָ֠הָב תְּכֵ֨לֶת וְאַרְגָּמָ֜ן תּוֹלַ֧עַת שָׁנִ֛י וְשֵׁ֥שׁ מָשְׁזָ֖ר מַֽעֲשֵׂ֥ה חֹשֵֽׁב׃ וְיַעְבְּדוּן, יָת אֵיפוֹדָא:  דַּהְבָּא תַּכְלָא וְאַרְגְּוָנָא צְבַע זְהוֹרִי, וּבוּץ שְׁזִיר--עוֹבָד אֻמָּן. 
כח,ז שְׁתֵּ֧י כְתֵפֹ֣ת חֹֽבְרֹ֗ת יִֽהְיֶה־לּ֛וֹ אֶל־שְׁנֵ֥י קְצוֹתָ֖יו וְחֻבָּֽר׃ תְּרֵין כִּתְפִין מְלָפְפָן, יְהוֹן לֵיהּ עַל תְּרֵין סִטְרוֹהִי--וְיִתְלָפַף. 
כח,ח וְחֵ֤שֶׁב אֲפֻדָּתוֹ֙ אֲשֶׁ֣ר עָלָ֔יו כְּמַֽעֲשֵׂ֖הוּ מִמֶּ֣נּוּ יִֽהְיֶ֑ה זָהָ֗ב תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֥שׁ מָשְׁזָֽר׃ וְהִמְיַן תִּקּוּנֵיהּ דַּעֲלוֹהִי, כְּעוֹבָדֵיהּ מִנֵּיהּ יְהֵי:  דַּהְבָּא, תַּכְלָא וְאַרְגְּוָנָא וּצְבַע זְהוֹרִי--וּבוּץ שְׁזִיר. 
כח,ט וְלָ֣קַחְתָּ֔ אֶת־שְׁתֵּ֖י אַבְנֵי־שֹׁ֑הַם וּפִתַּחְתָּ֣ עֲלֵיהֶ֔ם שְׁמ֖וֹת בְּנֵ֥י יִשְׂרָאֵֽל׃ וְתִסַּב, יָת תַּרְתֵּין אַבְנֵי בֻּרְלָא; וְתִגְלוֹף עֲלֵיהוֹן, שְׁמָהָת בְּנֵי יִשְׂרָאֵל. 
כח,י שִׁשָּׁה֙ מִשְּׁמֹתָ֔ם עַ֖ל הָאֶ֣בֶן הָֽאֶחָ֑ת וְאֶת־שְׁמ֞וֹת הַשִּׁשָּׁ֧ה הַנּֽוֹתָרִ֛ים עַל־הָאֶ֥בֶן הַשֵּׁנִ֖ית כְּתֽוֹלְדֹתָֽם׃ שִׁתָּא, מִשְּׁמָהָתְהוֹן, עַל, אַבְנָא חֲדָא; וְיָת שְׁמָהָת שִׁתָּא דְּאִשְׁתְּאַרוּ, עַל אַבְנָא תִּנְיֵיתָא--כְּתוֹלְדָתְהוֹן. 
כח,יא מַֽעֲשֵׂ֣ה חָרַשׁ֮ אֶבֶן֒ פִּתּוּחֵ֣י חֹתָ֗ם תְּפַתַּח֙ אֶת־שְׁתֵּ֣י הָֽאֲבָנִ֔ים עַל־שְׁמֹ֖ת בְּנֵ֣י יִשְׂרָאֵ֑ל מֻֽסַבֹּ֛ת מִשְׁבְּצ֥וֹת זָהָ֖ב תַּֽעֲשֶׂ֥ה אֹתָֽם׃ עוֹבָד אֻמָּן, אֶבֶן טָבָא--כְּתָב מְפָרַשׁ כִּגְלָף דְּעִזְקָא תִּגְלוֹף יָת תַּרְתֵּין אַבְנַיָּא, עַל שְׁמָהָת בְּנֵי יִשְׂרָאֵל; מְשַׁקְּעָן מְרַמְּצָן דִּדְהַב, תַּעֲבֵיד יָתְהוֹן. 
כח,יב וְשַׂמְתָּ֞ אֶת־שְׁתֵּ֣י הָֽאֲבָנִ֗ים עַ֚ל כִּתְפֹ֣ת הָֽאֵפֹ֔ד אַבְנֵ֥י זִכָּרֹ֖ן לִבְנֵ֣י יִשְׂרָאֵ֑ל וְנָשָׂא֩ אַֽהֲרֹ֨ן אֶת־שְׁמוֹתָ֜ם לִפְנֵ֧י יְהוָ֛ה עַל־שְׁתֵּ֥י כְתֵפָ֖יו לְזִכָּרֹֽן׃ {ס} וּתְשַׁוֵּי יָת תַּרְתֵּין אַבְנַיָּא, עַל כִּתְפֵי אֵיפוֹדָא, אַבְנֵי דֻּכְרָנָא, לִבְנֵי יִשְׂרָאֵל; וְיִטּוֹל אַהֲרוֹן יָת שְׁמָהָתְהוֹן קֳדָם יְיָ, עַל תְּרֵין כִּתְפוֹהִי--לְדֻכְרָנָא. {ס}
כח,יג וְעָשִׂ֥יתָ מִשְׁבְּצֹ֖ת זָהָֽב׃ וְתַעֲבֵיד מְרַמְּצָן, דִּדְהַב. 
כח,יד וּשְׁתֵּ֤י שַׁרְשְׁרֹת֙ זָהָ֣ב טָה֔וֹר מִגְבָּלֹ֛ת תַּֽעֲשֶׂ֥ה אֹתָ֖ם מַֽעֲשֵׂ֣ה עֲבֹ֑ת וְנָֽתַתָּ֛ה אֶת־שַׁרְשְׁרֹ֥ת הָֽעֲבֹתֹ֖ת עַל־הַֽמִּשְׁבְּצֹֽת׃ {ס} וְתַרְתֵּין תִּכִּין דִּדְהַב דְּכֵי, מְתַחֲמָן תַּעֲבֵיד יָתְהוֹן עוֹבָד גְּדִילוּ; וְתִתֵּין יָת תִּכַּיָּא גְּדִילָתָא, עַל מְרַמְּצָתָא. {ס}
כח,טו וְעָשִׂ֜יתָ חֹ֤שֶׁן מִשְׁפָּט֙ מַֽעֲשֵׂ֣ה חֹשֵׁ֔ב כְּמַֽעֲשֵׂ֥ה אֵפֹ֖ד תַּֽעֲשֶׂ֑נּוּ זָ֠הָב תְּכֵ֨לֶת וְאַרְגָּמָ֜ן וְתוֹלַ֧עַת שָׁנִ֛י וְשֵׁ֥שׁ מָשְׁזָ֖ר תַּֽעֲשֶׂ֥ה אֹתֽוֹ׃ וְתַעֲבֵיד חֹשֶׁן דִּינָא, עוֹבָד אֻמָּן--כְּעוֹבָד אֵיפוֹדָא, תַּעְבְּדִנֵּיהּ; דַּהְבָּא תַּכְלָא וְאַרְגְּוָנָא וּצְבַע זְהוֹרִי, וּבוּץ שְׁזִיר--תַּעֲבֵיד יָתֵיהּ. 
כח,טז רָב֥וּעַ יִֽהְיֶ֖ה כָּפ֑וּל זֶ֥רֶת אָרְכּ֖וֹ וְזֶ֥רֶת רָחְבּֽוֹ׃ מְרֻבַּע יְהֵי, עִיף; זַרְתָּא אֻרְכֵּיהּ, וְזַרְתָּא פֻּתְיֵיהּ. 
כח,יז וּמִלֵּאתָ֥ בוֹ֙ מִלֻּ֣אַת אֶ֔בֶן אַרְבָּעָ֖ה טוּרִ֣ים אָ֑בֶן ט֗וּר אֹ֤דֶם פִּטְדָה֙ וּבָרֶ֔קֶת הַטּ֖וּר הָֽאֶחָֽד׃ וְתַשְׁלֵים בֵּיהּ אַשְׁלָמוּת אַבְנָא, אַרְבְּעָא סִדְרִין דְּאֶבֶן טָבָא:  סִדְרָא קַדְמָאָה, סָמְקָן יָרְקָן וּבָרְקָן--סִדְרָא, חַד. 
כח,יח וְהַטּ֖וּר הַשֵּׁנִ֑י נֹ֥פֶךְ סַפִּ֖יר וְיָֽהֲלֹֽם׃ וְסִדְרָא, תִּנְיָנָא--אִזְמַרַגְדִּין שַׁבְזֵיז, וְסַבְהֲלוֹם. 
כח,יט וְהַטּ֖וּר הַשְּׁלִישִׁ֑י לֶ֥שֶׁם שְׁב֖וֹ וְאַחְלָֽמָה׃ וְסִדְרָא, תְּלִיתָאָה--קַנְכֵּירִי טְרַקְיָא, וְעֵין עִגְלָא. 
כח,כ וְהַטּוּר֙ הָֽרְבִיעִ֔י תַּרְשִׁ֥ישׁ וְשֹׁ֖הַם וְיָֽשְׁפֵ֑ה מְשֻׁבָּצִ֥ים זָהָ֛ב יִֽהְי֖וּ בְּמִלּֽוּאֹתָֽם׃ וְסִדְרָא, רְבִיעָאָה--כְּרוּם יַמָּא וּבֻרְלָא, וּפַנְתֵּירִי; מְרַמְּצָן בִּדְהַב יְהוֹן, בְּאַשְׁלָמוּתְהוֹן. 
כח,כא וְ֠הָֽאֲבָנִים תִּֽהְיֶ֜יןָ עַל־שְׁמֹ֧ת בְּנֵֽי־יִשְׂרָאֵ֛ל שְׁתֵּ֥ים עֶשְׂרֵ֖ה עַל־שְׁמֹתָ֑ם פִּתּוּחֵ֤י חוֹתָם֙ אִ֣ישׁ עַל־שְׁמ֔וֹ תִּֽהְיֶ֕יןָ לִשְׁנֵ֥י עָשָׂ֖ר שָֽׁבֶט׃ וְאַבְנַיָּא יִהְוְיָן עַל שְׁמָהָת בְּנֵי יִשְׂרָאֵל, תַּרְתַּא עַסְרֵי--עַל שְׁמָהָתְהוֹן; כְּתָב מְפָרַשׁ כִּגְלָף דְּעִזְקָא, גְּבַר עַל שְׁמֵיהּ, יִהְוְיָן, לִתְרֵי עֲסַר שִׁבְטִין. 
כח,כב וְעָשִׂ֧יתָ עַל־הַחֹ֛שֶׁן שַֽׁרְשֹׁ֥ת גַּבְלֻ֖ת מַֽעֲשֵׂ֣ה עֲבֹ֑ת זָהָ֖ב טָהֽוֹר׃ וְתַעֲבֵיד עַל חֻשְׁנָא תִּכִּין מְתַחֲמָן, עוֹבָד גְּדִילוּ, דִּדְהַב, דְּכֵי. 
כח,כג וְעָשִׂ֨יתָ֙ עַל־הַחֹ֔שֶׁן שְׁתֵּ֖י טַבְּע֣וֹת זָהָ֑ב וְנָֽתַתָּ֗ אֶת־שְׁתֵּי֙ הַטַּבָּע֔וֹת עַל־שְׁנֵ֖י קְצ֥וֹת הַחֹֽשֶׁן׃ וְתַעֲבֵיד, עַל חֻשְׁנָא, תַּרְתֵּין, עִזְקָן דִּדְהַב; וְתִתֵּין, יָת תַּרְתֵּין עִזְקָתָא, עַל תְּרֵין, סִטְרֵי חֻשְׁנָא. 
כח,כד וְנָֽתַתָּ֗ה אֶת־שְׁתֵּי֙ עֲבֹתֹ֣ת הַזָּהָ֔ב עַל־שְׁתֵּ֖י הַטַּבָּעֹ֑ת אֶל־קְצ֖וֹת הַחֹֽשֶׁן׃ וְתִתֵּין, יָת תַּרְתֵּין גְּדִילָן דִּדְהַב, עַל תַּרְתֵּין, עִזְקָתָא--בְּסִטְרֵי, חֻשְׁנָא. 
כח,כה וְאֵ֨ת שְׁתֵּ֤י קְצוֹת֙ שְׁתֵּ֣י הָֽעֲבֹתֹ֔ת תִּתֵּ֖ן עַל־שְׁתֵּ֣י הַֽמִּשְׁבְּצ֑וֹת וְנָֽתַתָּ֛ה עַל־כִּתְפ֥וֹת הָֽאֵפֹ֖ד אֶל־מ֥וּל פָּנָֽיו׃ וְיָת תַּרְתֵּין גְּדִילָן דְּעַל תְּרֵין סִטְרוֹהִי, תִּתֵּין עַל תַּרְתֵּין מְרַמְּצָתָא; וְתִתֵּין עַל כִּתְפֵי אֵיפוֹדָא, לָקֳבֵיל אַפּוֹהִי. 
כח,כו וְעָשִׂ֗יתָ שְׁתֵּי֙ טַבְּע֣וֹת זָהָ֔ב וְשַׂמְתָּ֣ אֹתָ֔ם עַל־שְׁנֵ֖י קְצ֣וֹת הַחֹ֑שֶׁן עַל־שְׂפָת֕וֹ אֲשֶׁ֛ר אֶל־עֵ֥בֶר הָֽאֵפֹ֖ד בָּֽיְתָה׃ וְתַעֲבֵיד, תַּרְתֵּין עִזְקָן דִּדְהַב, וּתְשַׁוֵּי יָתְהוֹן, עַל תְּרֵין סִטְרֵי חֻשְׁנָא--עַל סִפְתֵּיהּ, דִּלְעִבְרָא דְּאֵיפוֹדָא לְגָיו. 
כח,כז וְעָשִׂיתָ֮ שְׁתֵּ֣י טַבְּע֣וֹת זָהָב֒ וְנָֽתַתָּ֣ה אֹתָ֡ם עַל־שְׁתֵּי֩ כִתְפ֨וֹת הָֽאֵפ֤וֹד מִלְּמַ֨טָּה֙ מִמּ֣וּל פָּנָ֔יו לְעֻמַּ֖ת מַחְבַּרְתּ֑וֹ מִמַּ֕עַל לְחֵ֖שֶׁב הָֽאֵפֽוֹד׃ וְתַעֲבֵיד, תַּרְתֵּין עִזְקָן דִּדְהַב, וְתִתֵּין יָתְהוֹן עַל תְּרֵין כִּתְפֵי אֵיפוֹדָא מִלְּרַע מִלָּקֳבֵיל אַפּוֹהִי, לָקֳבֵיל בֵּית לוֹפִי--מֵעִלָּוֵי, לְהִמְיַן אֵיפוֹדָא. 
כח,כח וְיִרְכְּס֣וּ אֶת־הַ֠חֹשֶׁן מִֽטַּבְּעֹתָ֞ו אֶל־טַבְּעֹ֤ת הָֽאֵפוֹד֙ בִּפְתִ֣יל תְּכֵ֔לֶת לִֽהְי֖וֹת עַל־חֵ֣שֶׁב הָֽאֵפ֑וֹד וְלֹֽא־יִזַּ֣ח הַחֹ֔שֶׁן מֵעַ֖ל הָֽאֵפֽוֹד׃ וְיַחֲדוּן יָת חֻשְׁנָא מֵעִזְקָתֵיהּ לְעִזְקָת אֵיפוֹדָא, בְּחוּטָא דִּתְכִילְתָא, לְמִהְוֵי, עַל הִמְיַן אֵיפוֹדָא; וְלָא יִתְפָּרַק חֻשְׁנָא, מֵעִלָּוֵי אֵיפוֹדָא. 
כח,כט וְנָשָׂ֣א אַֽ֠הֲרֹן אֶת־שְׁמ֨וֹת בְּנֵֽי־יִשְׂרָאֵ֜ל בְּחֹ֧שֶׁן הַמִּשְׁפָּ֛ט עַל־לִבּ֖וֹ בְּבֹא֣וֹ אֶל־הַקֹּ֑דֶשׁ לְזִכָּרֹ֥ן לִפְנֵֽי־יְהוָ֖ה תָּמִֽיד׃ וְיִטּוֹל אַהֲרוֹן יָת שְׁמָהָת בְּנֵי יִשְׂרָאֵל בְּחֹשֶׁן דִּינָא, עַל לִבֵּיהּ--בְּמֵיעֲלֵיהּ לְקֻדְשָׁא:  לְדֻכְרָנָא קֳדָם יְיָ, תְּדִירָא. 
כח,ל וְנָֽתַתָּ֞ אֶל־חֹ֣שֶׁן הַמִּשְׁפָּ֗ט אֶת־הָֽאוּרִים֙ וְאֶת־הַתֻּמִּ֔ים וְהָיוּ֙ עַל־לֵ֣ב אַֽהֲרֹ֔ן בְּבֹא֖וֹ לִפְנֵ֣י יְהוָ֑ה וְנָשָׂ֣א אַֽ֠הֲרֹן אֶת־מִשְׁפַּ֨ט בְּנֵֽי־יִשְׂרָאֵ֧ל עַל־לִבּ֛וֹ לִפְנֵ֥י יְהוָ֖ה תָּמִֽיד׃ {ס} וְתִתֵּין בְּחֹשֶׁן דִּינָא, יָת אוּרַיָּא וְיָת תֻּמַּיָּא, וִיהוֹן עַל לִבָּא דְּאַהֲרוֹן, בְּמֵיעֲלֵיהּ לִקְדָם יְיָ; וְיִטּוֹל אַהֲרוֹן יָת דִּין בְּנֵי יִשְׂרָאֵל עַל לִבֵּיהּ, קֳדָם יְיָ--תְּדִירָא. {ס}
כח,לא וְעָשִׂ֛יתָ אֶת־מְעִ֥יל הָֽאֵפ֖וֹד כְּלִ֥יל תְּכֵֽלֶת׃ וְתַעֲבֵיד יָת מְעִיל אֵיפוֹדָא, גְּמִיר תַּכְלָא. 
כח,לב וְהָיָ֥ה פִֽי־רֹאשׁ֖וֹ בְּתוֹכ֑וֹ שָׂפָ֡ה יִֽהְיֶה֩ לְפִ֨יו סָבִ֜יב מַֽעֲשֵׂ֣ה אֹרֵ֗ג כְּפִ֥י תַחְרָ֛א יִֽהְיֶה־לּ֖וֹ לֹ֥א יִקָּרֵֽעַ׃ וִיהֵי פֻּמֵּיהּ, כְּפִיל לְגַוֵּיהּ; תּוּרָא יְהֵי מַקַּף לְפֻמֵּיהּ סְחוֹר סְחוֹר עוֹבָד מָחֵי, כְּפֹם שִׁרְיָן יְהֵי לֵיהּ--דְּלָא יִתְבְּזַע. 
כח,לג וְעָשִׂ֣יתָ עַל־שׁוּלָ֗יו רִמֹּנֵי֙ תְּכֵ֤לֶת וְאַרְגָּמָן֙ וְתוֹלַ֣עַת שָׁנִ֔י עַל־שׁוּלָ֖יו סָבִ֑יב וּפַֽעֲמֹנֵ֥י זָהָ֛ב בְּתוֹכָ֖ם סָבִֽיב׃ וְתַעֲבֵיד עַל שִׁפּוֹלוֹהִי, רִמּוֹנֵי תַּכְלָא וְאַרְגְּוָנָא וּצְבַע זְהוֹרִי--עַל שִׁפּוֹלוֹהִי, סְחוֹר סְחוֹר; וְזַגִּין דִּדְהַב בֵּינֵיהוֹן, סְחוֹר סְחוֹר. 
כח,לד פַּֽעֲמֹ֤ן זָהָב֙ וְרִמּ֔וֹן פַּֽעֲמֹ֥ן זָהָ֖ב וְרִמּ֑וֹן עַל־שׁוּלֵ֥י הַמְּעִ֖יל סָבִֽיב׃ זַגָּא דְּדַהְבָּא וְרִמּוֹנָא, זַגָּא דְּדַהְבָּא וְרִמּוֹנָא, עַל שִׁפּוֹלֵי מְעִילָא, סְחוֹר סְחוֹר. 
כח,לה וְהָיָ֥ה עַֽל־אַהֲרֹ֖ן לְשָׁרֵ֑ת וְנִשְׁמַ֣ע ק֠וֹלוֹ בְּבֹא֨וֹ אֶל־הַקֹּ֜דֶשׁ לִפְנֵ֧י יְהוָ֛ה וּבְצֵאת֖וֹ וְלֹ֥א יָמֽוּת׃ {ס} וִיהֵי עַל אַהֲרוֹן, לְשַׁמָּשָׁא; וְיִשְׁתְּמַע קָלֵיהּ בְּמֵיעֲלֵיהּ לְקֻדְשָׁא לִקְדָם יְיָ, וּבְמִפְּקֵיהּ--וְלָא יְמוּת. {ס}
כח,לו וְעָשִׂ֥יתָ צִּ֖יץ זָהָ֣ב טָה֑וֹר וּפִתַּחְתָּ֤ עָלָיו֙ פִּתּוּחֵ֣י חֹתָ֔ם קֹ֖דֶשׁ לַֽיהוָֽה׃ וְתַעֲבֵיד צִיצָא, דִּדְהַב דְּכֵי; וְתִגְלוֹף עֲלוֹהִי כְּתָב מְפָרַשׁ, קֹדֶשׁ לַייָ. 
כח,לז וְשַׂמְתָּ֤ אֹתוֹ֙ עַל־פְּתִ֣יל תְּכֵ֔לֶת וְהָיָ֖ה עַל־הַמִּצְנָ֑פֶת אֶל־מ֥וּל פְּנֵֽי־הַמִּצְנֶ֖פֶת יִֽהְיֶֽה׃ וּתְשַׁוֵּי יָתֵיהּ עַל חוּטָא דִּתְכִילְתָא, וִיהֵי עַל מַצְנַפְתָּא; לָקֳבֵיל אַפֵּי מַצְנַפְתָּא, יְהֵי. 
כח,לח וְהָיָה֮ עַל־מֵ֣צַח אַֽהֲרֹן֒ וְנָשָׂ֨א אַֽהֲרֹ֜ן אֶת־עֲו֣‍ֹן הַקֳּדָשִׁ֗ים אֲשֶׁ֤ר יַקְדִּ֨ישׁוּ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל לְכָֽל־מַתְּנֹ֖ת קָדְשֵׁיהֶ֑ם וְהָיָ֤ה עַל־מִצְחוֹ֙ תָּמִ֔יד לְרָצ֥וֹן לָהֶ֖ם לִפְנֵ֥י יְהוָֽה׃ וִיהֵי, עַל בֵּית עֵינוֹהִי דְּאַהֲרוֹן, וְיִטּוֹל אַהֲרוֹן יָת עֲוָיָת קֻדְשַׁיָּא דְּיַקְדְּשׁוּן בְּנֵי יִשְׂרָאֵל, לְכל מַתְּנָת קֻדְשֵׁיהוֹן; וִיהֵי עַל בֵּית עֵינוֹהִי תְּדִירָא, לְרַעֲוָא לְהוֹן קֳדָם יְיָ. 
כח,לט וְשִׁבַּצְתָּ֙ הַכְּתֹ֣נֶת שֵׁ֔שׁ וְעָשִׂ֖יתָ מִצְנֶ֣פֶת שֵׁ֑שׁ וְאַבְנֵ֥ט תַּֽעֲשֶׂ֖ה מַֽעֲשֵׂ֥ה רֹקֵֽם׃ וּתְרַמֵּיץ כִּתּוּנָא דְּבוּצָא, וְתַעֲבֵיד מַצְנַפְתָּא דְּבוּצָא; וְהִמְיָן תַּעֲבֵיד, עוֹבָד צַיָּר. 
כח,מ וְלִבְנֵ֤י אַֽהֲרֹן֙ תַּֽעֲשֶׂ֣ה כֻתֳּנֹ֔ת וְעָשִׂ֥יתָ לָהֶ֖ם אַבְנֵטִ֑ים וּמִגְבָּעוֹת֙ תַּֽעֲשֶׂ֣ה לָהֶ֔ם לְכָב֖וֹד וּלְתִפְאָֽרֶת׃ וְלִבְנֵי אַהֲרוֹן תַּעֲבֵיד כִּתּוּנִין, וְתַעֲבֵיד לְהוֹן הִמְיָנִין; וְקוֹבְעִין תַּעֲבֵיד לְהוֹן, לִיקָר וּלְתֻשְׁבְּחָא. 
כח,מא וְהִלְבַּשְׁתָּ֤ אֹתָם֙ אֶת־אַֽהֲרֹ֣ן אָחִ֔יךָ וְאֶת־בָּנָ֖יו אִתּ֑וֹ וּמָֽשַׁחְתָּ֨ אֹתָ֜ם וּמִלֵּאתָ֧ אֶת־יָדָ֛ם וְקִדַּשְׁתָּ֥ אֹתָ֖ם וְכִֽהֲנ֥וּ לִֽי׃ וְתַלְבֵּישׁ יָתְהוֹן יָת אַהֲרוֹן אֲחוּךְ, וְיָת בְּנוֹהִי עִמֵּיהּ; וּתְרַבֵּי יָתְהוֹן וּתְקָרֵיב יָת קֻרְבָּנְהוֹן, וּתְקַדֵּישׁ יָתְהוֹן--וִישַׁמְּשׁוּן קֳדָמָי. 
כח,מב וַֽעֲשֵׂ֤ה לָהֶם֙ מִכְנְסֵי־בָ֔ד לְכַסּ֖וֹת בְּשַׂ֣ר עֶרְוָ֑ה מִמָּתְנַ֥יִם וְעַד־יְרֵכַ֖יִם יִֽהְיֽוּ׃ וַעֲבֵיד לְהוֹן מִכְנְסִין דְּבוּץ, לְכַסָּאָה בְּסַר עֶרְיָא; מֵחַרְצִין וְעַד יִרְכָן, יְהוֹן. 
כח,מג וְהָיוּ֩ עַל־אַֽהֲרֹ֨ן וְעַל־בָּנָ֜יו בְּבֹאָ֣ם ׀ אֶל־אֹ֣הֶל מוֹעֵ֗ד א֣וֹ בְגִשְׁתָּ֤ם אֶל־הַמִּזְבֵּ֨חַ֙ לְשָׁרֵ֣ת בַּקֹּ֔דֶשׁ וְלֹֽא־יִשְׂא֥וּ עָו֖‍ֹן וָמֵ֑תוּ חֻקַּ֥ת עוֹלָ֛ם ל֖וֹ וּלְזַרְע֥וֹ אַֽחֲרָֽיו׃ {ס} וִיהוֹן עַל אַהֲרוֹן וְעַל בְּנוֹהִי בְּמֵיעַלְהוֹן לְמַשְׁכַּן זִמְנָא, אוֹ בְּמִקְרַבְהוֹן לְמַדְבְּחָא לְשַׁמָּשָׁא בְּקֻדְשָׁא, וְלָא יְקַבְּלוּן חוֹבָא, וְלָא יְמוּתוּן:  קְיָם עָלַם לֵיהּ, וְלִבְנוֹהִי בָּתְרוֹהִי. {ס}
כט,א וְזֶ֨ה הַדָּבָ֜ר אֲשֶֽׁר־תַּעֲשֶׂ֥ה לָהֶ֛ם לְקַדֵּ֥שׁ אֹתָ֖ם לְכַהֵ֣ן לִ֑י לְ֠קַח פַּ֣ר אֶחָ֧ד בֶּן־בָּקָ֛ר וְאֵילִ֥ם שְׁנַ֖יִם תְּמִימִֽם׃ וְדֵין פִּתְגָמָא דְּתַעֲבֵיד לְהוֹן, לְקַדָּשָׁא יָתְהוֹן--לְשַׁמָּשָׁא קֳדָמָי:  סַב תּוֹר חַד בַּר תּוֹרֵי, וְדִכְרִין תְּרֵין--שַׁלְמִין. 
כט,ב וְלֶ֣חֶם מַצּ֗וֹת וְחַלֹּ֤ת מַצֹּת֙ בְּלוּלֹ֣ת בַּשֶּׁ֔מֶן וּרְקִיקֵ֥י מַצּ֖וֹת מְשֻׁחִ֣ים בַּשָּׁ֑מֶן סֹ֥לֶת חִטִּ֖ים תַּֽעֲשֶׂ֥ה אֹתָֽם׃ וּלְחֵים פַּטִּיר, וּגְרִיצָן פַּטִּירָן דְּפִילָן בִּמְשַׁח, וְאֶסְפּוֹגִין פַּטִּירִין, דִּמְשִׁיחִין בִּמְשַׁח; סֹלֶת דְּחִטִּין, תַּעֲבֵיד יָתְהוֹן. 
כט,ג וְנָֽתַתָּ֤ אוֹתָם֙ עַל־סַ֣ל אֶחָ֔ד וְהִקְרַבְתָּ֥ אֹתָ֖ם בַּסָּ֑ל וְאֶ֨ת־הַפָּ֔ר וְאֵ֖ת שְׁנֵ֥י הָֽאֵילִֽם׃ וְתִתֵּין יָתְהוֹן עַל סַלָּא חַד, וּתְקָרֵיב יָתְהוֹן בְּסַלָּא; וְיָת תּוֹרָא--וְיָת, תְּרֵין דִּכְרִין. 
כט,ד וְאֶת־אַֽהֲרֹ֤ן וְאֶת־בָּנָיו֙ תַּקְרִ֔יב אֶל־פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד וְרָֽחַצְתָּ֥ אֹתָ֖ם בַּמָּֽיִם׃ וְיָת אַהֲרוֹן וְיָת בְּנוֹהִי תְּקָרֵיב, לִתְרַע מַשְׁכַּן זִמְנָא; וְתַסְחֵי יָתְהוֹן, בְּמַיָּא. 
כט,ה וְלָֽקַחְתָּ֣ אֶת־הַבְּגָדִ֗ים וְהִלְבַּשְׁתָּ֤ אֶֽת־אַהֲרֹן֙ אֶת־הַכֻּתֹּ֔נֶת וְאֵת֙ מְעִ֣יל הָֽאֵפֹ֔ד וְאֶת־הָֽאֵפֹ֖ד וְאֶת־הַחֹ֑שֶׁן וְאָֽפַדְתָּ֣ ל֔וֹ בְּחֵ֖שֶׁב הָֽאֵפֹֽד׃ וְתִסַּב יָת לְבוּשַׁיָּא, וְתַלְבֵּישׁ יָת אַהֲרוֹן יָת כִּתּוּנָא, וְיָת מְעִיל אֵיפוֹדָא, וְיָת אֵיפוֹדָא וְיָת חֻשְׁנָא; וְתַתְקֵין לֵיהּ, בְּהִמְיַן אֵיפוֹדָא. 
כט,ו וְשַׂמְתָּ֥ הַמִּצְנֶ֖פֶת עַל־רֹאשׁ֑וֹ וְנָֽתַתָּ֛ אֶת־נֵ֥זֶר הַקֹּ֖דֶשׁ עַל־הַמִּצְנָֽפֶת׃ וּתְשַׁוֵּי מַצְנַפְתָּא, עַל רֵישֵׁיהּ; וְתִתֵּין יָת כְּלִילָא דְּקֻדְשָׁא, עַל מַצְנַפְתָּא. 
כט,ז וְלָֽקַחְתָּ֙ אֶת־שֶׁ֣מֶן הַמִּשְׁחָ֔ה וְיָֽצַקְתָּ֖ עַל־רֹאשׁ֑וֹ וּמָֽשַׁחְתָּ֖ אֹתֽוֹ׃ וְתִסַּב יָת מִשְׁחָא דִּרְבוּתָא, וּתְרִיק עַל רֵישֵׁיהּ; וּתְרַבִּי, יָתֵיהּ. 
כט,ח וְאֶת־בָּנָ֖יו תַּקְרִ֑יב וְהִלְבַּשְׁתָּ֖ם כֻּתֳּנֹֽת׃ וְיָת בְּנוֹהִי, תְּקָרֵיב; וְתַלְבֵּישִׁנּוּן, כִּתּוּנִין. 
כט,ט וְחָֽגַרְתָּ֩ אֹתָ֨ם אַבְנֵ֜ט אַֽהֲרֹ֣ן וּבָנָ֗יו וְחָֽבַשְׁתָּ֤ לָהֶם֙ מִגְבָּעֹ֔ת וְהָֽיְתָ֥ה לָהֶ֛ם כְּהֻנָּ֖ה לְחֻקַּ֣ת עוֹלָ֑ם וּמִלֵּאתָ֥ יַֽד־אַהֲרֹ֖ן וְיַד־בָּנָֽיו׃ וּתְזָרֵיז יָתְהוֹן הִמְיָנִין אַהֲרוֹן וּבְנוֹהִי, וְתַתְקֵין לְהוֹן קוֹבְעִין, וּתְהֵי לְהוֹן כְּהֻנְּתָא, לִקְיָם עָלַם; וּתְקָרֵיב קֻרְבָּנָא דְּאַהֲרוֹן, וְקֻרְבָּנָא דִּבְנוֹהִי. 
כט,י וְהִקְרַבְתָּ֙ אֶת־הַפָּ֔ר לִפְנֵ֖י אֹ֣הֶל מוֹעֵ֑ד וְסָמַ֨ךְ אַֽהֲרֹ֧ן וּבָנָ֛יו אֶת־יְדֵיהֶ֖ם עַל־רֹ֥אשׁ הַפָּֽר׃ וּתְקָרֵיב, יָת תּוֹרָא, לִקְדָם, מַשְׁכַּן זִמְנָא; וְיִסְמוֹךְ אַהֲרוֹן וּבְנוֹהִי יָת יְדֵיהוֹן, עַל רֵישׁ תּוֹרָא. 
כט,יא וְשָֽׁחַטְתָּ֥ אֶת־הַפָּ֖ר לִפְנֵ֣י יְהוָ֑ה פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ וְתִכּוֹס יָת תּוֹרָא, קֳדָם יְיָ, בִּתְרַע, מַשְׁכַּן זִמְנָא. 
כט,יב וְלָֽקַחְתָּ֙ מִדַּ֣ם הַפָּ֔ר וְנָֽתַתָּ֛ה עַל־קַרְנֹ֥ת הַמִּזְבֵּ֖חַ בְּאֶצְבָּעֶ֑ךָ וְאֶת־כָּל־הַדָּ֣ם תִּשְׁפֹּ֔ךְ אֶל־יְס֖וֹד הַמִּזְבֵּֽחַ׃ וְתִסַּב מִדְּמָא דְּתוֹרָא, וְתִתֵּין עַל קַרְנָת מַדְבְּחָא בְּאֶצְבְּעָךְ; וְיָת כָּל דְּמָא תִּשְׁפּוֹךְ, לִיסוֹדָא דְּמַדְבְּחָא. 
כט,יג וְלָֽקַחְתָּ֗ אֶֽת־כָּל־הַחֵלֶב֮ הַֽמְכַסֶּ֣ה אֶת־הַקֶּרֶב֒ וְאֵ֗ת הַיֹּתֶ֨רֶת֙ עַל־הַכָּבֵ֔ד וְאֵת֙ שְׁתֵּ֣י הַכְּלָיֹ֔ת וְאֶת־הַחֵ֖לֶב אֲשֶׁ֣ר עֲלֵיהֶ֑ן וְהִקְטַרְתָּ֖ הַמִּזְבֵּֽחָה׃ וְתִסַּב, יָת כָּל תַּרְבָּא דְּחָפֵי יָת גַּוָּא, וְיָת חַצְרָא דְּעַל כַּבְדָּא, וְיָת תַּרְתֵּין כּוֹלְיָן וְיָת תַּרְבָּא דַּעֲלֵיהוֹן; וְתַסֵּיק, לְמַדְבְּחָא. 
כט,יד וְאֶת־בְּשַׂ֤ר הַפָּר֙ וְאֶת־עֹר֣וֹ וְאֶת־פִּרְשׁ֔וֹ תִּשְׂרֹ֣ף בָּאֵ֔שׁ מִח֖וּץ לַֽמַּחֲנֶ֑ה חַטָּ֖את הֽוּא׃ וְיָת בְּסַר תּוֹרָא, וְיָת מַשְׁכֵּיהּ וְיָת אֻכְלֵיהּ, תּוֹקֵיד בְּנוּרָא, מִבַּרָא לְמַשְׁרִיתָא:  חַטָּתָא, הוּא. 
כט,טו וְאֶת־הָאַ֥יִל הָֽאֶחָ֖ד תִּקָּ֑ח וְסָ֨מְכ֜וּ אַֽהֲרֹ֧ן וּבָנָ֛יו אֶת־יְדֵיהֶ֖ם עַל־רֹ֥אשׁ הָאָֽיִל׃ וְיָת דִּכְרָא חַד, תִּסַּב; וְיִסְמְכוּן אַהֲרוֹן וּבְנוֹהִי, יָת יְדֵיהוֹן--עַל רֵישׁ דִּכְרָא. 
כט,טז וְשָֽׁחַטְתָּ֖ אֶת־הָאָ֑יִל וְלָֽקַחְתָּ֙ אֶת־דָּמ֔וֹ וְזָֽרַקְתָּ֥ עַל־הַמִּזְבֵּ֖חַ סָבִֽיב׃ וְתִכּוֹס, יָת דִּכְרָא; וְתִסַּב, יָת דְּמֵיהּ, וְתִזְרוֹק עַל מַדְבְּחָא, סְחוֹר סְחוֹר. 
כט,יז וְאֶ֨ת־הָאַ֔יִל תְּנַתֵּ֖חַ לִנְתָחָ֑יו וְרָֽחַצְתָּ֤ קִרְבּוֹ֙ וּכְרָעָ֔יו וְנָֽתַתָּ֥ עַל־נְתָחָ֖יו וְעַל־רֹאשֽׁוֹ׃ וְיָת דִּכְרָא--תְּפַלֵּיג, לְאֶבְרוֹהִי; וּתְחַלֵּיל גַּוֵּיהּ וּכְרָעוֹהִי, וְתִתֵּין עַל אֶבְרוֹהִי וְעַל רֵישֵׁיהּ. 
כט,יח וְהִקְטַרְתָּ֤ אֶת־כָּל־הָאַ֨יִל֙ הַמִּזְבֵּ֔חָה עֹלָ֥ה ה֖וּא לַֽיהוָ֑ה רֵ֣יחַ נִיח֔וֹחַ אִשֶּׁ֥ה לַֽיהוָ֖ה הֽוּא׃ וְתַסֵּיק יָת כָּל דִּכְרָא לְמַדְבְּחָא, עֲלָתָא הוּא קֳדָם יְיָ; לְאִתְקַבָּלָא בְּרַעֲוָא, קֻרְבָּנָא קֳדָם יְיָ הוּא. 
כט,יט וְלָ֣קַחְתָּ֔ אֵ֖ת הָאַ֣יִל הַשֵּׁנִ֑י וְסָמַ֨ךְ אַֽהֲרֹ֧ן וּבָנָ֛יו אֶת־יְדֵיהֶ֖ם עַל־רֹ֥אשׁ הָאָֽיִל׃ וְתִסַּב, יָת דִּכְרָא תִּנְיָנָא; וְיִסְמוֹךְ אַהֲרוֹן וּבְנוֹהִי יָת יְדֵיהוֹן, עַל רֵישׁ דִּכְרָא. 
כט,כ וְשָֽׁחַטְתָּ֣ אֶת־הָאַ֗יִל וְלָֽקַחְתָּ֤ מִדָּמוֹ֙ וְנָֽתַתָּ֡ה עַל־תְּנוּךְ֩ אֹ֨זֶן אַֽהֲרֹ֜ן וְעַל־תְּנ֨וּךְ אֹ֤זֶן בָּנָיו֙ הַיְמָנִ֔ית וְעַל־בֹּ֤הֶן יָדָם֙ הַיְמָנִ֔ית וְעַל־בֹּ֥הֶן רַגְלָ֖ם הַיְמָנִ֑ית וְזָֽרַקְתָּ֧ אֶת־הַדָּ֛ם עַל־הַמִּזְבֵּ֖חַ סָבִֽיב׃ וְתִכּוֹס יָת דִּכְרָא, וְתִסַּב מִדְּמֵיהּ וְתִתֵּין עַל רוּם אֻדְנָא דְּאַהֲרוֹן וְעַל רוּם אֻדְנָא דִּבְנוֹהִי דְּיַמִּינָא, וְעַל אִלְיוֹן יַדְהוֹן דְּיַמִּינָא, וְעַל אִלְיוֹן רַגְלְהוֹן דְּיַמִּינָא; וְתִזְרוֹק יָת דְּמָא עַל מַדְבְּחָא, סְחוֹר סְחוֹר. 
כט,כא וְלָֽקַחְתָּ֞ מִן־הַדָּ֨ם אֲשֶׁ֥ר עַֽל־הַמִּזְבֵּחַ֮ וּמִשֶּׁ֣מֶן הַמִּשְׁחָה֒ וְהִזֵּיתָ֤ עַֽל־אַהֲרֹן֙ וְעַל־בְּגָדָ֔יו וְעַל־בָּנָ֛יו וְעַל־בִּגְדֵ֥י בָנָ֖יו אִתּ֑וֹ וְקָדַ֥שׁ הוּא֙ וּבְגָדָ֔יו וּבָנָ֛יו וּבִגְדֵ֥י בָנָ֖יו אִתּֽוֹ׃ וְתִסַּב מִן דְּמָא דְּעַל מַדְבְּחָא, וּמִמִּשְׁחָא דִּרְבוּתָא, וְתַדֵּי עַל אַהֲרוֹן וְעַל לְבוּשׁוֹהִי, וְעַל בְּנוֹהִי וְעַל לְבוּשֵׁי בְּנוֹהִי עִמֵּיהּ; וְיִתְקַדַּשׁ הוּא וּלְבוּשׁוֹהִי, וּבְנוֹהִי וּלְבוּשֵׁי בְּנוֹהִי עִמֵּיהּ. 
כט,כב וְלָֽקַחְתָּ֣ מִן־הָ֠אַיִל הַחֵ֨לֶב וְהָֽאַלְיָ֜ה וְאֶת־הַחֵ֣לֶב ׀ הַֽמְכַסֶּ֣ה אֶת־הַקֶּ֗רֶב וְאֵ֨ת יֹתֶ֤רֶת הַכָּבֵד֙ וְאֵ֣ת ׀ שְׁתֵּ֣י הַכְּלָיֹ֗ת וְאֶת־הַחֵ֨לֶב֙ אֲשֶׁ֣ר עֲלֵיהֶ֔ן וְאֵ֖ת שׁ֣וֹק הַיָּמִ֑ין כִּ֛י אֵ֥יל מִלֻּאִ֖ים הֽוּא׃ וְתִסַּב מִן דִּכְרָא תַּרְבָּא וְאַלְיְתָא וְיָת תַּרְבָּא דְּחָפֵי יָת גַּוָּא, וְיָת חֲצַר כַּבְדָּא וְיָת תַּרְתֵּין כּוֹלְיָן וְיָת תַּרְבָּא דַּעֲלֵיהוֹן, וְיָת, שָׁקָא דְּיַמִּינָא:  אֲרֵי דְּכַר קֻרְבָּנַיָּא, הוּא. 
כט,כג וְכִכַּ֨ר לֶ֜חֶם אַחַ֗ת וְֽחַלַּ֨ת לֶ֥חֶם שֶׁ֛מֶן אַחַ֖ת וְרָקִ֣יק אֶחָ֑ד מִסַּל֙ הַמַּצּ֔וֹת אֲשֶׁ֖ר לִפְנֵ֥י יְהוָֽה׃ וּפִתָּא דִּלְחֵים חֲדָא, וּגְרִיצְתָא דִּלְחֵים מְשַׁח חֲדָא--וְאֶסְפּוֹג חַד:  מִסַּלָּא, דְּפַטִּירַיָּא--דִּקְדָם יְיָ. 
כט,כד וְשַׂמְתָּ֣ הַכֹּ֔ל עַ֚ל כַּפֵּ֣י אַֽהֲרֹ֔ן וְעַ֖ל כַּפֵּ֣י בָנָ֑יו וְהֵֽנַפְתָּ֥ אֹתָ֛ם תְּנוּפָ֖ה לִפְנֵ֥י יְהוָֽה׃ וּתְשַׁוֵּי כּוֹלָא--עַל יְדֵי אַהֲרוֹן, וְעַל יְדֵי בְּנוֹהִי; וּתְרִים יָתְהוֹן אֲרָמָא, קֳדָם יְיָ. 
כט,כה וְלָֽקַחְתָּ֤ אֹתָם֙ מִיָּדָ֔ם וְהִקְטַרְתָּ֥ הַמִּזְבֵּ֖חָה עַל־הָֽעֹלָ֑ה לְרֵ֤יחַ נִיח֨וֹחַ֙ לִפְנֵ֣י יְהוָ֔ה אִשֶּׁ֥ה ה֖וּא לַֽיהוָֽה׃ וְתִסַּב יָתְהוֹן מִיַּדְהוֹן, וְתַסֵּיק לְמַדְבְּחָא עַל עֲלָתָא; לְאִתְקַבָּלָא בְּרַעֲוָא קֳדָם יְיָ, קֻרְבָּנָא הוּא קֳדָם יְיָ. 
כט,כו וְלָֽקַחְתָּ֣ אֶת־הֶֽחָזֶ֗ה מֵאֵ֤יל הַמִּלֻּאִים֙ אֲשֶׁ֣ר לְאַֽהֲרֹ֔ן וְהֵֽנַפְתָּ֥ אֹת֛וֹ תְּנוּפָ֖ה לִפְנֵ֣י יְהוָ֑ה וְהָיָ֥ה לְךָ֖ לְמָנָֽה׃ וְתִסַּב יָת חַדְיָא, מִדְּכַר קֻרְבָּנַיָּא דִּלְאַהֲרוֹן, וּתְרִים יָתֵיהּ אֲרָמָא, קֳדָם יְיָ; וִיהֵי לָךְ, לֻחְלָק. 
כט,כז וְקִדַּשְׁתָּ֞ אֵ֣ת ׀ חֲזֵ֣ה הַתְּנוּפָ֗ה וְאֵת֙ שׁ֣וֹק הַתְּרוּמָ֔ה אֲשֶׁ֥ר הוּנַ֖ף וַֽאֲשֶׁ֣ר הוּרָ֑ם מֵאֵיל֙ הַמִּלֻּאִ֔ים מֵֽאֲשֶׁ֥ר לְאַֽהֲרֹ֖ן וּמֵֽאֲשֶׁ֥ר לְבָנָֽיו׃ וּתְקַדֵּישׁ יָת חַדְיָא דַּאֲרָמוּתָא, וְיָת שָׁקָא דְּאַפְרָשׁוּתָא, דְּאִתָּרַם, וּדְאִתַּפְרַשׁ:  מִדְּכַר, קֻרְבָּנַיָּא--מִדִּלְאַהֲרוֹן, וּמִדִּלְבְּנוֹהִי. 
כט,כח וְהָיָה֩ לְאַֽהֲרֹ֨ן וּלְבָנָ֜יו לְחָק־עוֹלָ֗ם מֵאֵת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל כִּ֥י תְרוּמָ֖ה ה֑וּא וּתְרוּמָ֞ה יִֽהְיֶ֨ה מֵאֵ֤ת בְּנֵֽי־יִשְׂרָאֵל֙ מִזִּבְחֵ֣י שַׁלְמֵיהֶ֔ם תְּרֽוּמָתָ֖ם לַֽיהוָֽה׃ וִיהֵי לְאַהֲרוֹן וְלִבְנוֹהִי לִקְיָם עָלַם, מִן בְּנֵי יִשְׂרָאֵל, אֲרֵי אַפְרָשׁוּתָא, הוּא; וְאַפְרָשׁוּתָא יְהֵי מִן בְּנֵי יִשְׂרָאֵל, מִנִּכְסַת קֻדְשֵׁיהוֹן--אַפְרָשׁוּתְהוֹן, קֳדָם יְיָ. 
כט,כט וּבִגְדֵ֤י הַקֹּ֨דֶשׁ֙ אֲשֶׁ֣ר לְאַֽהֲרֹ֔ן יִֽהְי֥וּ לְבָנָ֖יו אַֽחֲרָ֑יו לְמָשְׁחָ֣ה בָהֶ֔ם וּלְמַלֵּא־בָ֖ם אֶת־יָדָֽם׃ וּלְבוּשֵׁי קֻדְשָׁא דִּלְאַהֲרוֹן, יְהוֹן לִבְנוֹהִי בָּתְרוֹהִי, לְרַבָּאָה בְּהוֹן, וּלְקָרָבָא בְּהוֹן יָת קֻרְבָּנְהוֹן. 
כט,ל שִׁבְעַ֣ת יָמִ֗ים יִלְבָּשָׁ֧ם הַכֹּהֵ֛ן תַּחְתָּ֖יו מִבָּנָ֑יו אֲשֶׁ֥ר יָבֹ֛א אֶל־אֹ֥הֶל מוֹעֵ֖ד לְשָׁרֵ֥ת בַּקֹּֽדֶשׁ׃ שִׁבְעָא יוֹמִין, יִלְבְּשִׁנּוּן כָּהֲנָא תְּחוֹתוֹהִי--מִבְּנוֹהִי:  דְּיֵיעוֹל לְמַשְׁכַּן זִמְנָא, לְשַׁמָּשָׁא בְּקֻדְשָׁא. 
כט,לא וְאֵ֛ת אֵ֥יל הַמִּלֻּאִ֖ים תִּקָּ֑ח וּבִשַּׁלְתָּ֥ אֶת־בְּשָׂר֖וֹ בְּמָקֹ֥ם קָדֹֽשׁ׃ וְיָת דְּכַר קֻרְבָּנַיָּא, תִּסַּב; וּתְבַשֵּׁיל יָת בִּסְרֵיהּ, בַּאֲתַר קַדִּישׁ. 
כט,לב וְאָכַ֨ל אַֽהֲרֹ֤ן וּבָנָיו֙ אֶת־בְּשַׂ֣ר הָאַ֔יִל וְאֶת־הַלֶּ֖חֶם אֲשֶׁ֣ר בַּסָּ֑ל פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ וְיֵיכוֹל אַהֲרוֹן וּבְנוֹהִי יָת בְּסַר דִּכְרָא, וְיָת לַחְמָא דִּבְסַלָּא, בִּתְרַע, מַשְׁכַּן זִמְנָא. 
כט,לג וְאָֽכְל֤וּ אֹתָם֙ אֲשֶׁ֣ר כֻּפַּ֣ר בָּהֶ֔ם לְמַלֵּ֥א אֶת־יָדָ֖ם לְקַדֵּ֣שׁ אֹתָ֑ם וְזָ֥ר לֹֽא־יֹאכַ֖ל כִּי־קֹ֥דֶשׁ הֵֽם׃ וְיֵיכְלוּן יָתְהוֹן דְּאִתְכַּפַּר בְּהוֹן, לְקָרָבָא יָת קֻרְבָּנְהוֹן לְקַדָּשָׁא יָתְהוֹן; וְחִילוֹנַי לָא יֵיכוֹל, אֲרֵי קֻדְשָׁא אִנּוּן. 
כט,לד וְֽאִם־יִוָּתֵ֞ר מִבְּשַׂ֧ר הַמִּלֻּאִ֛ים וּמִן־הַלֶּ֖חֶם עַד־הַבֹּ֑קֶר וְשָֽׂרַפְתָּ֤ אֶת־הַנּוֹתָר֙ בָּאֵ֔שׁ לֹ֥א יֵֽאָכֵ֖ל כִּי־קֹ֥דֶשׁ הֽוּא׃ וְאִם יִשְׁתְּאַר מִבְּסַר קֻרְבָּנַיָּא, וּמִן לַחְמָא--עַד צַפְרָא:  וְתוֹקֵיד יָת דְּאִשְׁתְּאַר בְּנוּרָא, לָא יִתְאֲכִיל אֲרֵי קֻדְשָׁא הוּא. 
כט,לה וְעָשִׂ֜יתָ לְאַֽהֲרֹ֤ן וּלְבָנָיו֙ כָּ֔כָה כְּכֹ֥ל אֲשֶׁר־צִוִּ֖יתִי אֹתָ֑כָה שִׁבְעַ֥ת יָמִ֖ים תְּמַלֵּ֥א יָדָֽם׃ וְתַעֲבֵיד לְאַהֲרוֹן וְלִבְנוֹהִי, כְּדֵין, כְּכֹל דְּפַקֵּידִית, יָתָךְ; שִׁבְעָא יוֹמִין, תְּקָרֵיב קֻרְבָּנְהוֹן. 
כט,לו וּפַ֨ר חַטָּ֜את תַּֽעֲשֶׂ֤ה לַיּוֹם֙ עַל־הַכִּפֻּרִ֔ים וְחִטֵּאתָ֙ עַל־הַמִּזְבֵּ֔חַ בְּכַפֶּרְךָ֖ עָלָ֑יו וּמָֽשַׁחְתָּ֥ אֹת֖וֹ לְקַדְּשֽׁוֹ׃ וְתוֹרָא דְּחַטָּתָא תַּעֲבֵיד לְיוֹמָא, עַל כִּפּוּרַיָּא, וּתְדַכֵּי עַל מַדְבְּחָא, בְּכַפָּרוּתָךְ עֲלוֹהִי; וּתְרַבֵּי יָתֵיהּ, לְקַדָּשׁוּתֵיהּ. 
כט,לז שִׁבְעַ֣ת יָמִ֗ים תְּכַפֵּר֙ עַל־הַמִּזְבֵּ֔חַ וְקִדַּשְׁתָּ֖ אֹת֑וֹ וְהָיָ֤ה הַמִּזְבֵּ֨חַ֙ קֹ֣דֶשׁ קָֽדָשִׁ֔ים כָּל־הַנֹּגֵ֥עַ בַּמִּזְבֵּ֖חַ יִקְדָּֽשׁ׃ {ס} שִׁבְעָא יוֹמִין, תְּכַפַּר עַל מַדְבְּחָא, וּתְקַדֵּישׁ, יָתֵיהּ; וִיהֵי מַדְבְּחָא קֹדֶשׁ קֻדְשִׁין, כָּל דְּיִקְרַב בְּמַדְבְּחָא יִתְקַדַּשׁ. {ס}
כט,לח וְזֶ֕ה אֲשֶׁ֥ר תַּֽעֲשֶׂ֖ה עַל־הַמִּזְבֵּ֑חַ כְּבָשִׂ֧ים בְּנֵֽי־שָׁנָ֛ה שְׁנַ֥יִם לַיּ֖וֹם תָּמִֽיד׃ וְדֵין, דְּתַעֲבֵיד עַל מַדְבְּחָא:  אִמְּרִין בְּנֵי שְׁנָא תְּרֵין לְיוֹמָא, תְּדִירָא. 
כט,לט אֶת־הַכֶּ֥בֶשׂ הָֽאֶחָ֖ד תַּֽעֲשֶׂ֣ה בַבֹּ֑קֶר וְאֵת֙ הַכֶּ֣בֶשׂ הַשֵּׁנִ֔י תַּֽעֲשֶׂ֖ה בֵּ֥ין הָֽעַרְבָּֽיִם׃ יָת אִמְּרָא חַד, תַּעֲבֵיד בְּצַפְרָא; וְיָת אִמְּרָא תִּנְיָנָא, תַּעֲבֵיד בֵּין שִׁמְשַׁיָּא. 
כט,מ וְעִשָּׂרֹ֨ן סֹ֜לֶת בָּל֨וּל בְּשֶׁ֤מֶן כָּתִית֙ רֶ֣בַע הַהִ֔ין וְנֵ֕סֶךְ רְבִיעִ֥ת הַהִ֖ין יָ֑יִן לַכֶּ֖בֶשׂ הָֽאֶחָֽד׃ וְעֶסְרוֹנָא סֻלְתָּא דְּפִיל בִּמְשַׁח כָּתִישָׁא, רַבְעוּת הִינָא, וְנִסְכָּא, רַבְעוּת הִינָא חַמְרָא--לְאִמְּרָא, חַד. 
כט,מא וְאֵת֙ הַכֶּ֣בֶשׂ הַשֵּׁנִ֔י תַּֽעֲשֶׂ֖ה בֵּ֣ין הָֽעַרְבָּ֑יִם כְּמִנְחַ֨ת הַבֹּ֤קֶר וּכְנִסְכָּהּ֙ תַּֽעֲשֶׂה־לָּ֔הּ לְרֵ֣יחַ נִיחֹ֔חַ אִשֶּׁ֖ה לַֽיהוָֽה׃ וְיָת אִמְּרָא תִּנְיָנָא, תַּעֲבֵיד בֵּין שִׁמְשַׁיָּא:  כְּמִנְחַת צַפְרָא וּכְנִסְכַּהּ, תַּעֲבֵיד לַהּ, לְאִתְקַבָּלָא בְּרַעֲוָא, קֻרְבָּנָא קֳדָם יְיָ. 
כט,מב עֹלַ֤ת תָּמִיד֙ לְדֹרֹ֣תֵיכֶ֔ם פֶּ֥תַח אֹֽהֶל־מוֹעֵ֖ד לִפְנֵ֣י יְהוָ֑ה אֲשֶׁ֨ר אִוָּעֵ֤ד לָכֶם֙ שָׁ֔מָּה לְדַבֵּ֥ר אֵלֶ֖יךָ שָֽׁם׃ עֲלַת תְּדִירָא לְדָרֵיכוֹן, בִּתְרַע מַשְׁכַּן זִמְנָא קֳדָם יְיָ, דַּאֲזָמֵין מֵימְרִי לְכוֹן תַּמָּן, לְמַלָּלָא עִמָּךְ תַּמָּן. 
כט,מג וְנֹֽעַדְתִּ֥י שָׁ֖מָּה לִבְנֵ֣י יִשְׂרָאֵ֑ל וְנִקְדַּ֖שׁ בִּכְבֹדִֽי׃ וַאֲזָמֵין מֵימְרִי תַּמָּן, לִבְנֵי יִשְׂרָאֵל; וְאֶתְקַדַּשׁ, בִּיקָרִי. 
כט,מד וְקִדַּשְׁתִּ֛י אֶת־אֹ֥הֶל מוֹעֵ֖ד וְאֶת־הַמִּזְבֵּ֑חַ וְאֶת־אַֽהֲרֹ֧ן וְאֶת־בָּנָ֛יו אֲקַדֵּ֖שׁ לְכַהֵ֥ן לִֽי׃ וַאֲקַדֵּישׁ יָת מַשְׁכַּן זִמְנָא, וְיָת מַדְבְּחָא; וְיָת אַהֲרוֹן וְיָת בְּנוֹהִי אֲקַדֵּישׁ, לְשַׁמָּשָׁא קֳדָמָי. 
כט,מה וְשָׁ֣כַנְתִּ֔י בְּת֖וֹךְ בְּנֵ֣י יִשְׂרָאֵ֑ל וְהָיִ֥יתִי לָהֶ֖ם לֵֽאלֹהִֽים׃ וְאַשְׁרֵי שְׁכִינְתִי, בְּגוֹ בְּנֵי יִשְׂרָאֵל; וְאֶהְוֵי לְהוֹן, לֶאֱלָהּ. 
כט,מו וְיָֽדְע֗וּ כִּ֣י אֲנִ֤י יְהוָה֙ אֱלֹ֣הֵיהֶ֔ם אֲשֶׁ֨ר הוֹצֵ֧אתִי אֹתָ֛ם מֵאֶ֥רֶץ מִצְרַ֖יִם לְשָׁכְנִ֣י בְתוֹכָ֑ם אֲנִ֖י יְהוָ֥ה אֱלֹֽהֵיהֶֽם׃ {פ} וְיִדְּעוּן, אֲרֵי אֲנָא יְיָ אֱלָהֲהוֹן, דְּאַפֵּיקִית יָתְהוֹן מֵאַרְעָא דְּמִצְרַיִם, לְאַשְׁרָאָה שְׁכִינְתִי בֵּינֵיהוֹן:  אֲנָא, יְיָ אֱלָהֲהוֹן. {פ}
ל,א וְעָשִׂ֥יתָ מִזְבֵּ֖חַ מִקְטַ֣ר קְטֹ֑רֶת עֲצֵ֥י שִׁטִּ֖ים תַּֽעֲשֶׂ֥ה אֹתֽוֹ׃ וְתַעֲבֵיד מַדְבְּחָא, לְאַקְטָרָא עֲלוֹהִי קְטֹרֶת בֻּסְמִין; דְּאָעֵי שִׁטִּין, תַּעֲבֵיד יָתֵיהּ. 
ל,ב אַמָּ֨ה אָרְכּ֜וֹ וְאַמָּ֤ה רָחְבּוֹ֙ רָב֣וּעַ יִֽהְיֶ֔ה וְאַמָּתַ֖יִם קֹֽמָת֑וֹ מִמֶּ֖נּוּ קַרְנֹתָֽיו׃ אַמְּתָא אֻרְכֵּיהּ וְאַמְּתָא פֻּתְיֵיהּ, מְרֻבַּע יְהֵי, וְתַרְתֵּין אַמִּין, רוּמֵיהּ; מִנֵּיהּ, קַרְנוֹהִי. 
ל,ג וְצִפִּיתָ֨ אֹת֜וֹ זָהָ֣ב טָה֗וֹר אֶת־גַּגּ֧וֹ וְאֶת־קִֽירֹתָ֛יו סָבִ֖יב וְאֶת־קַרְנֹתָ֑יו וְעָשִׂ֥יתָ לּ֛וֹ זֵ֥ר זָהָ֖ב סָבִֽיב׃ וְתִחְפֵי יָתֵיהּ דְּהַב דְּכֵי, יָת אִגָּרֵיהּ וְיָת כֻּתְלוֹהִי סְחוֹר סְחוֹר--וְיָת קַרְנוֹהִי; וְתַעֲבֵיד לֵיהּ זִיר דִּדְהַב, סְחוֹר סְחוֹר. 
ל,ד וּשְׁתֵּי֩ טַבְּעֹ֨ת זָהָ֜ב תַּֽעֲשֶׂה־לּ֣וֹ ׀ מִתַּ֣חַת לְזֵר֗וֹ עַ֚ל שְׁתֵּ֣י צַלְעֹתָ֔יו תַּֽעֲשֶׂ֖ה עַל־שְׁנֵ֣י צִדָּ֑יו וְהָיָה֙ לְבָתִּ֣ים לְבַדִּ֔ים לָשֵׂ֥את אֹת֖וֹ בָּהֵֽמָּה׃ וְתַרְתֵּין עִזְקָן דִּדְהַב תַּעֲבֵיד לֵיהּ מִלְּרַע לְזֵירֵיהּ, עַל תַּרְתֵּין זָוְיָתֵיהּ--תַּעֲבֵיד, עַל תְּרֵין סִטְרוֹהִי; וִיהֵי לְאַתְרָא לַאֲרִיחַיָּא, לְמִטַּל יָתֵיהּ בְּהוֹן. 
ל,ה וְעָשִׂ֥יתָ אֶת־הַבַּדִּ֖ים עֲצֵ֣י שִׁטִּ֑ים וְצִפִּיתָ֥ אֹתָ֖ם זָהָֽב׃ וְתַעֲבֵיד יָת אֲרִיחַיָּא, דְּאָעֵי שִׁטִּין; וְתִחְפֵי יָתְהוֹן, דַּהְבָּא. 
ל,ו וְנָֽתַתָּ֤ה אֹתוֹ֙ לִפְנֵ֣י הַפָּרֹ֔כֶת אֲשֶׁ֖ר עַל־אֲרֹ֣ן הָֽעֵדֻ֑ת לִפְנֵ֣י הַכַּפֹּ֗רֶת אֲשֶׁר֙ עַל־הָ֣עֵדֻ֔ת אֲשֶׁ֛ר אִוָּעֵ֥ד לְךָ֖ שָֽׁמָּה׃ וְתִתֵּין יָתֵיהּ קֳדָם פָּרֻכְתָּא, דְּעַל אֲרוֹנָא דְּסָהֲדוּתָא--קֳדָם כָּפֻרְתָּא, דְּעַל סָהֲדוּתָא, דַּאֲזָמֵין מֵימְרִי לָךְ, תַּמָּן. 
ל,ז וְהִקְטִ֥יר עָלָ֛יו אַֽהֲרֹ֖ן קְטֹ֣רֶת סַמִּ֑ים בַּבֹּ֣קֶר בַּבֹּ֗קֶר בְּהֵֽיטִיב֛וֹ אֶת־הַנֵּרֹ֖ת יַקְטִירֶֽנָּה׃ וְיַקְטַר עֲלוֹהִי אַהֲרוֹן, קְטֹרֶת בֻּסְמִין; בִּצְפַר בִּצְפַר, בְּאַתְקָנוּתֵיהּ יָת בּוֹצִינַיָּא--יַקְטְרִנַּהּ. 
ל,ח וּבְהַֽעֲלֹ֨ת אַֽהֲרֹ֧ן אֶת־הַנֵּרֹ֛ת בֵּ֥ין הָֽעַרְבַּ֖יִם יַקְטִירֶ֑נָּה קְטֹ֧רֶת תָּמִ֛יד לִפְנֵ֥י יְהוָ֖ה לְדֹרֹֽתֵיכֶֽם׃ וּבְאַדְלָקוּת אַהֲרוֹן יָת בּוֹצִינַיָּא בֵּין שִׁמְשַׁיָּא, יַקְטְרִנַּהּ--קְטֹרֶת בֻּסְמִין תְּדִירָא קֳדָם יְיָ, לְדָרֵיכוֹן. 
ל,ט לֹֽא־תַעֲל֥וּ עָלָ֛יו קְטֹ֥רֶת זָרָ֖ה וְעֹלָ֣ה וּמִנְחָ֑ה וְנֵ֕סֶךְ לֹ֥א תִסְּכ֖וּ עָלָֽיו׃ לָא תַּסְּקוּן עֲלוֹהִי קְטֹרֶת בֻּסְמִין נֻכְרָאִין, וַעֲלָתָא וּמִנְחָתָא; וְנִסְכָּא, לָא תְּנַסְּכוּן עֲלוֹהִי. 
ל,י וְכִפֶּ֤ר אַֽהֲרֹן֙ עַל־קַרְנֹתָ֔יו אַחַ֖ת בַּשָּׁנָ֑ה מִדַּ֞ם חַטַּ֣את הַכִּפֻּרִ֗ים אַחַ֤ת בַּשָּׁנָה֙ יְכַפֵּ֤ר עָלָיו֙ לְדֹרֹ֣תֵיכֶ֔ם קֹֽדֶשׁ־קָֽדָשִׁ֥ים ה֖וּא לַֽיהוָֽה׃ {פ} וִיכַפַּר אַהֲרוֹן עַל קַרְנָתֵיהּ, חֲדָא בְּשַׁתָּא:  מִדַּם חַטַּאת כִּפּוּרַיָּא, חֲדָא בְּשַׁתָּא יְכַפַּר עֲלוֹהִי לְדָרֵיכוֹן--קֹדֶשׁ קֻדְשִׁין הוּא, קֳדָם יְיָ. {פ}
