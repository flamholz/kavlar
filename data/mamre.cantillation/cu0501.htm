דברים

א,א אֵ֣לֶּה הַדְּבָרִ֗ים אֲשֶׁ֨ר דִּבֶּ֤ר מֹשֶׁה֙ אֶל־כָּל־יִשְׂרָאֵ֔ל בְּעֵ֖בֶר הַיַּרְדֵּ֑ן בַּמִּדְבָּ֡ר בָּֽעֲרָבָה֩ מ֨וֹל ס֜וּף בֵּֽין־פָּארָ֧ן וּבֵֽין־תֹּ֛פֶל וְלָבָ֥ן וַֽחֲצֵרֹ֖ת וְדִ֥י זָהָֽב׃ אִלֵּין פִּתְגָמַיָּא, דְּמַלֵּיל מֹשֶׁה עִם כָּל יִשְׂרָאֵל, בְּעִבְרָא, דְּיַרְדְּנָא:  אוֹכַח יָתְהוֹן עַל דְּחָבוּ בְּמַדְבְּרָא וְעַל דְּאַרְגִּיזוּ בְּמֵישְׁרָא לָקֳבֵיל יַם סוּף בְּפָארָן אִתַּפַּלוּ עַל מַנָּא, וּבַחֲצֵירוֹת אַרְגִּיזוּ עַל בִּסְרָא--וְעַל דַּעֲבַדוּ עֵיגַל דִּדְהַב. 
א,ב אַחַ֨ד עָשָׂ֥ר יוֹם֙ מֵֽחֹרֵ֔ב דֶּ֖רֶךְ הַר־שֵׂעִ֑יר עַ֖ד קָדֵ֥שׁ בַּרְנֵֽעַ׃ מַהְלַךְ חַד עֲסַר יוֹמִין מֵחוֹרֵב, אוֹרַח טוּרָא דְּשֵׂעִיר, עַד, רְקַם גֵּיאָה. 
א,ג וַֽיְהִי֙ בְּאַרְבָּעִ֣ים שָׁנָ֔ה בְּעַשְׁתֵּֽי־עָשָׂ֥ר חֹ֖דֶשׁ בְּאֶחָ֣ד לַחֹ֑דֶשׁ דִּבֶּ֤ר מֹשֶׁה֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל כְּ֠כֹל אֲשֶׁ֨ר צִוָּ֧ה יְהוָ֛ה אֹת֖וֹ אֲלֵהֶֽם׃ וַהֲוָה בְּאַרְבְּעִין שְׁנִין, בְּחַד עֲסַר יַרְחִין בְּחַד לְיַרְחָא; מַלֵּיל מֹשֶׁה, עִם בְּנֵי יִשְׂרָאֵל, כְּכֹל דְּפַקֵּיד יְיָ יָתֵיהּ, לְוָתְהוֹן. 
א,ד אַֽחֲרֵ֣י הַכֹּת֗וֹ אֵ֚ת סִיחֹן֙ מֶ֣לֶךְ הָֽאֱמֹרִ֔י אֲשֶׁ֥ר יוֹשֵׁ֖ב בְּחֶשְׁבּ֑וֹן וְאֵ֗ת ע֚וֹג מֶ֣לֶךְ הַבָּשָׁ֔ן אֲשֶׁר־יוֹשֵׁ֥ב בְּעַשְׁתָּרֹ֖ת בְּאֶדְרֶֽעִי׃ בָּתַר דִּמְחָא, יָת סִיחוֹן מַלְכָּא אֱמוֹרָאָה, דְּיָתֵיב, בְּחֶשְׁבּוֹן--וְיָת, עוֹג מַלְכָּא דְּמַתְנַן, דְּיָתֵיב בְּעַשְׁתָּרוֹת, בְּאֶדְרֶעִי. 
א,ה בְּעֵ֥בֶר הַיַּרְדֵּ֖ן בְּאֶ֣רֶץ מוֹאָ֑ב הוֹאִ֣יל מֹשֶׁ֔ה בֵּאֵ֛ר אֶת־הַתּוֹרָ֥ה הַזֹּ֖את לֵאמֹֽר׃ בְּעִבְרָא דְּיַרְדְּנָא, בְּאַרְעָא דְּמוֹאָב, שָׁרִי מֹשֶׁה, פָּרֵישׁ יָת אֻלְפַן אוֹרָיְתָא הָדָא לְמֵימַר. 
א,ו יְהוָ֧ה אֱלֹהֵ֛ינוּ דִּבֶּ֥ר אֵלֵ֖ינוּ בְּחֹרֵ֣ב לֵאמֹ֑ר רַב־לָכֶ֥ם שֶׁ֖בֶת בָּהָ֥ר הַזֶּֽה׃ יְיָ אֱלָהַנָא מַלֵּיל עִמַּנָא, בְּחוֹרֵב לְמֵימַר:  סַגִּי לְכוֹן דִּיתֵיבְתּוּן, בְּטוּרָא הָדֵין. 
א,ז פְּנ֣וּ ׀ וּסְע֣וּ לָכֶ֗ם וּבֹ֨אוּ הַ֥ר הָֽאֱמֹרִי֮ וְאֶל־כָּל־שְׁכֵנָיו֒ בָּֽעֲרָבָ֥ה בָהָ֛ר וּבַשְּׁפֵלָ֥ה וּבַנֶּ֖גֶב וּבְח֣וֹף הַיָּ֑ם אֶ֤רֶץ הַֽכְּנַעֲנִי֙ וְהַלְּבָנ֔וֹן עַד־הַנָּהָ֥ר הַגָּדֹ֖ל נְהַר־פְּרָֽת׃ אִתְפְּנוּ וְטוּלוּ לְכוֹן, וְעוּלוּ לְטוּרָא דֶּאֱמוֹרָאָה וּלְכָל מְגִירוֹהִי, בְּמֵישְׁרָא בְּטוּרָא וּבְשָׁפֵילְתָּא וּבְדָרוֹמָא, וּבִסְפַר יַמָּא--אֲרַע כְּנַעֲנָאָה וְלִבְנָן, עַד נַהְרָא רַבָּא נַהְרָא פְּרָת. 
א,ח רְאֵ֛ה נָתַ֥תִּי לִפְנֵיכֶ֖ם אֶת־הָאָ֑רֶץ בֹּ֚אוּ וּרְשׁ֣וּ אֶת־הָאָ֔רֶץ אֲשֶׁ֣ר נִשְׁבַּ֣ע יְ֠הוָה לַאֲבֹ֨תֵיכֶ֜ם לְאַבְרָהָ֨ם לְיִצְחָ֤ק וּֽלְיַעֲקֹב֙ לָתֵ֣ת לָהֶ֔ם וּלְזַרְעָ֖ם אַֽחֲרֵיהֶֽם׃ חֲזוֹ דִּיהַבִית קֳדָמֵיכוֹן, יָת אַרְעָא; עוּלוּ, וְאַחְסִינוּ יָת אַרְעָא, דְּקַיֵּים יְיָ לַאֲבָהָתְכוֹן לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקוֹב לְמִתַּן לְהוֹן, וְלִבְנֵיהוֹן בָּתְרֵיהוֹן. 
א,ט וָֽאֹמַ֣ר אֲלֵכֶ֔ם בָּעֵ֥ת הַהִ֖וא לֵאמֹ֑ר לֹֽא־אוּכַ֥ל לְבַדִּ֖י שְׂאֵ֥ת אֶתְכֶֽם׃ וַאֲמַרִית לְכוֹן, בְּעִדָּנָא הַהוּא לְמֵימַר:  לֵית אֲנָא יָכֵיל בִּלְחוֹדַי, לְסוֹבָרָא יָתְכוֹן. 
א,י יְהוָ֥ה אֱלֹֽהֵיכֶ֖ם הִרְבָּ֣ה אֶתְכֶ֑ם וְהִנְּכֶ֣ם הַיּ֔וֹם כְּכֽוֹכְבֵ֥י הַשָּׁמַ֖יִם לָרֹֽב׃ יְיָ אֱלָהֲכוֹן, אַסְגִּי יָתְכוֹן; וְהָא אִיתֵיכוֹן יוֹמָא דֵּין, כְּכוֹכְבֵי שְׁמַיָּא לִסְגֵּי. 
א,יא יְהוָ֞ה אֱלֹהֵ֣י אֲבֽוֹתֵכֶ֗ם יֹסֵ֧ף עֲלֵיכֶ֛ם כָּכֶ֖ם אֶ֣לֶף פְּעָמִ֑ים וִֽיבָרֵ֣ךְ אֶתְכֶ֔ם כַּֽאֲשֶׁ֖ר דִּבֶּ֥ר לָכֶֽם׃ יְיָ אֱלָהָא דַּאֲבָהָתְכוֹן, יוֹסֵיף עֲלֵיכוֹן כְּוָתְכוֹן--אֶלֶף זִמְנִין; וִיבָרֵיךְ יָתְכוֹן, כְּמָא דְּמַלֵּיל לְכוֹן. 
א,יב אֵיכָ֥ה אֶשָּׂ֖א לְבַדִּ֑י טָרְחֲכֶ֥ם וּמַֽשַּׂאֲכֶ֖ם וְרִֽיבְכֶֽם׃ אֵיכְדֵין אֲסוֹבַר, בִּלְחוֹדַי, טֻרְחֵיכוֹן וְעִסְקֵיכוֹן, וְדִינְכוֹן. 
א,יג הָב֣וּ לָ֠כֶם אֲנָשִׁ֨ים חֲכָמִ֧ים וּנְבֹנִ֛ים וִֽידֻעִ֖ים לְשִׁבְטֵיכֶ֑ם וַֽאֲשִׂימֵ֖ם בְּרָֽאשֵׁיכֶֽם׃ הַבוּ לְכוֹן גֻּבְרִין חַכִּימִין וְסֻכְלְתָנִין, וּמַדְּעַן--לְשִׁבְטֵיכוֹן; וַאֲמַנֵּינוּן, רֵישִׁין עֲלֵיכוֹן. 
א,יד וַֽתַּעֲנ֖וּ אֹתִ֑י וַתֹּ֣אמְר֔וּ טֽוֹב־הַדָּבָ֥ר אֲשֶׁר־דִּבַּ֖רְתָּ לַֽעֲשֽׂוֹת׃ וַאֲתֵיבְתּוּן, יָתִי; וַאֲמַרְתּוּן, תָּקֵין פִּתְגָמָא דְּמַלֵּילְתָּא לְמֶעֱבַד. 
א,טו וָֽאֶקַּ֞ח אֶת־רָאשֵׁ֣י שִׁבְטֵיכֶ֗ם אֲנָשִׁ֤ים חֲכָמִים֙ וִֽידֻעִ֔ים וָֽאֶתֵּ֥ן אוֹתָ֛ם רָאשִׁ֖ים עֲלֵיכֶ֑ם שָׂרֵ֨י אֲלָפִ֜ים וְשָׂרֵ֣י מֵא֗וֹת וְשָׂרֵ֤י חֲמִשִּׁים֙ וְשָׂרֵ֣י עֲשָׂרֹ֔ת וְשֹֽׁטְרִ֖ים לְשִׁבְטֵיכֶֽם׃ וּדְבַרִית יָת רֵישֵׁי שִׁבְטֵיכוֹן, גֻּבְרִין חַכִּימִין וּמַדְּעַן, וּמַנִּיתִי יָתְהוֹן רֵישִׁין, עֲלֵיכוֹן:  רַבָּנֵי אַלְפֵי וְרַבָּנֵי מָאוָתָא, וְרַבָּנֵי חַמְשִׁין וְרַבָּנֵי עֲסוֹרְיָתָא, וְסָרְכִין, לְשִׁבְטֵיכוֹן. 
א,טז וָֽאֲצַוֶּה֙ אֶת־שֹׁ֣פְטֵיכֶ֔ם בָּעֵ֥ת הַהִ֖וא לֵאמֹ֑ר שָׁמֹ֤עַ בֵּין־אֲחֵיכֶם֙ וּשְׁפַטְתֶּ֣ם צֶ֔דֶק בֵּֽין־אִ֥ישׁ וּבֵין־אָחִ֖יו וּבֵ֥ין גֵּרֽוֹ׃ וּפַקֵּידִית, יָת דַּיָּנֵיכוֹן, בְּעִדָּנָא הַהוּא, לְמֵימַר:  שְׁמַעוּ בֵּין אֲחֵיכוֹן וּתְדִינוּן קֻשְׁטָא, בֵּין גֻּבְרָא וּבֵין אֲחוּהִי וּבֵין גִּיּוֹרֵיהּ. 
א,יז לֹֽא־תַכִּ֨ירוּ פָנִ֜ים בַּמִּשְׁפָּ֗ט כַּקָּטֹ֤ן כַּגָּדֹל֙ תִּשְׁמָע֔וּן לֹ֤א תָג֨וּרוּ֙ מִפְּנֵי־אִ֔ישׁ כִּ֥י הַמִּשְׁפָּ֖ט לֵֽאלֹהִ֣ים ה֑וּא וְהַדָּבָר֙ אֲשֶׁ֣ר יִקְשֶׁ֣ה מִכֶּ֔ם תַּקְרִב֥וּן אֵלַ֖י וּשְׁמַעְתִּֽיו׃ לָא תִּשְׁתְּמוֹדְעוּן אַפֵּי בְּדִינָא, מִלֵּי זְעֵירָא כְּרַבָּא תִּשְׁמְעוּן--לָא תִּדְחֲלוּן מִן קֳדָם גֻּבְרָא, אֲרֵי דִּינָא דַּייָ הוּא; וּפִתְגָמָא דְּיִקְשֵׁי מִנְּכוֹן, תְּקָרְבוּן לְוָתִי וְאֶשְׁמְעִנֵּיהּ. 
א,יח וָֽאֲצַוֶּ֥ה אֶתְכֶ֖ם בָּעֵ֣ת הַהִ֑וא אֵ֥ת כָּל־הַדְּבָרִ֖ים אֲשֶׁ֥ר תַּֽעֲשֽׂוּן׃ וּפַקֵּידִית יָתְכוֹן, בְּעִדָּנָא הַהוּא, יָת כָּל פִּתְגָמַיָּא, דְּתַעְבְּדוּן. 
א,יט וַנִּסַּ֣ע מֵֽחֹרֵ֗ב וַנֵּ֡לֶךְ אֵ֣ת כָּל־הַמִּדְבָּ֣ר הַגָּדוֹל֩ וְהַנּוֹרָ֨א הַה֜וּא אֲשֶׁ֣ר רְאִיתֶ֗ם דֶּ֚רֶךְ הַ֣ר הָֽאֱמֹרִ֔י כַּֽאֲשֶׁ֥ר צִוָּ֛ה יְהוָ֥ה אֱלֹהֵ֖ינוּ אֹתָ֑נוּ וַנָּבֹ֕א עַ֖ד קָדֵ֥שׁ בַּרְנֵֽעַ׃ וּנְטַלְנָא מֵחוֹרֵב, וְהַלֵּיכְנָא יָת כָּל מַדְבְּרָא רַבָּא וּדְחִילָא הַהוּא דַּחֲזֵיתוֹן אוֹרַח טוּרָא דֶּאֱמוֹרָאָה, כְּמָא דְּפַקֵּיד יְיָ אֱלָהַנָא, יָתַנָא; וַאֲתֵינָא, עַד רְקַם גֵּיאָה. 
א,כ וָֽאֹמַ֖ר אֲלֵכֶ֑ם בָּאתֶם֙ עַד־הַ֣ר הָֽאֱמֹרִ֔י אֲשֶׁר־יְהוָ֥ה אֱלֹהֵ֖ינוּ נֹתֵ֥ן לָֽנוּ׃ וַאֲמַרִית, לְכוֹן:  אֲתֵיתוּן עַד טוּרָא דֶּאֱמוֹרָאָה, דַּייָ אֱלָהַנָא יָהֵיב לַנָא. 
א,כא רְ֠אֵה נָתַ֨ן יְהוָ֧ה אֱלֹהֶ֛יךָ לְפָנֶ֖יךָ אֶת־הָאָ֑רֶץ עֲלֵ֣ה רֵ֗שׁ כַּֽאֲשֶׁר֩ דִּבֶּ֨ר יְהוָ֜ה אֱלֹהֵ֤י אֲבֹתֶ֨יךָ֙ לָ֔ךְ אַל־תִּירָ֖א וְאַל־תֵּחָֽת׃ חֲזִי דִּיהַב יְיָ אֱלָהָךְ, קֳדָמָךְ--יָת אַרְעָא:  סַק אַחְסֵין, כְּמָא דְּמַלֵּיל יְיָ אֱלָהָא דַּאֲבָהָתָךְ לָךְ--לָא תִּדְחַל, וְלָא תִּתְּבַר. 
א,כב וַתִּקְרְב֣וּן אֵלַי֮ כֻּלְּכֶם֒ וַתֹּֽאמְר֗וּ נִשְׁלְחָ֤ה אֲנָשִׁים֙ לְפָנֵ֔ינוּ וְיַחְפְּרוּ־לָ֖נוּ אֶת־הָאָ֑רֶץ וְיָשִׁ֤בוּ אֹתָ֨נוּ֙ דָּבָ֔ר אֶת־הַדֶּ֨רֶךְ֙ אֲשֶׁ֣ר נַֽעֲלֶה־בָּ֔הּ וְאֵת֙ הֶֽעָרִ֔ים אֲשֶׁ֥ר נָבֹ֖א אֲלֵיהֶֽן׃ וּקְרֵיבְתּוּן לְוָתִי, כֻּלְּכוֹן, וַאֲמַרְתּוּן נִשְׁלַח גֻּבְרִין קֳדָמַנָא, וִיאַלְּלוּן לַנָא יָת אַרְעָא; וְיָתִיבוּן יָתַנָא, פִּתְגָמָא--יָת אוֹרְחָא דְּנִסַּק בַּהּ, וְיָת קִרְוַיָּא דְּנֵיעוֹל לְהוֹן. 
א,כג וַיִּיטַ֥ב בְּעֵינַ֖י הַדָּבָ֑ר וָֽאֶקַּ֤ח מִכֶּם֙ שְׁנֵ֣ים עָשָׂ֣ר אֲנָשִׁ֔ים אִ֥ישׁ אֶחָ֖ד לַשָּֽׁבֶט׃ וּשְׁפַר בְּעֵינַי, פִּתְגָמָא; וּדְבַרִית מִנְּכוֹן תְּרֵי עֲסַר גֻּבְרִין, גֻּבְרָא חַד לְשִׁבְטָא. 
א,כד וַיִּפְנוּ֙ וַיַּֽעֲל֣וּ הָהָ֔רָה וַיָּבֹ֖אוּ עַד־נַ֣חַל אֶשְׁכֹּ֑ל וַֽיְרַגְּל֖וּ אֹתָֽהּ׃ וְאִתְפְּנִיאוּ וּסְלִיקוּ לְטוּרָא, וַאֲתוֹ עַד נַחְלָא דְּאֶתְכָּלָא; וְאַלִּילוּ, יָתַהּ. 
א,כה וַיִּקְח֤וּ בְיָדָם֙ מִפְּרִ֣י הָאָ֔רֶץ וַיּוֹרִ֖דוּ אֵלֵ֑ינוּ וַיָּשִׁ֨בוּ אֹתָ֤נוּ דָבָר֙ וַיֹּ֣אמְר֔וּ טוֹבָ֣ה הָאָ֔רֶץ אֲשֶׁר־יְהוָ֥ה אֱלֹהֵ֖ינוּ נֹתֵ֥ן לָֽנוּ׃ וּנְסִיבוּ בְּיַדְהוֹן מֵאִבָּא דְּאַרְעָא, וַאֲחִיתוּ לַנָא; וַאֲתִיבוּ יָתַנָא פִּתְגָמָא, וַאֲמַרוּ, טָבָא אַרְעָא, דַּייָ אֱלָהַנָא יָהֵיב לַנָא. 
א,כו וְלֹ֥א אֲבִיתֶ֖ם לַֽעֲלֹ֑ת וַתַּמְר֕וּ אֶת־פִּ֥י יְהוָ֖ה אֱלֹֽהֵיכֶֽם׃ וְלָא אֲבֵיתוּן, לְמִסַּק; וְסָרֵיבְתּוּן, עַל מֵימְרָא דַּייָ אֱלָהֲכוֹן. 
א,כז וַתֵּרָֽגְנ֤וּ בְאָֽהֳלֵיכֶם֙ וַתֹּ֣אמְר֔וּ בְּשִׂנְאַ֤ת יְהוָה֙ אֹתָ֔נוּ הֽוֹצִיאָ֖נוּ מֵאֶ֣רֶץ מִצְרָ֑יִם לָתֵ֥ת אֹתָ֛נוּ בְּיַ֥ד הָֽאֱמֹרִ֖י לְהַשְׁמִידֵֽנוּ׃ וְאִתְרָעַמְתּוּן בְּמַשְׁכְּנֵיכוֹן, וַאֲמַרְתּוּן, בִּדְסָנֵי יְיָ יָתַנָא, אַפְּקַנָא מֵאַרְעָא דְּמִצְרָיִם--לְמִמְסַר יָתַנָא בִּידָא דֶּאֱמוֹרָאָה, לְשֵׁיצָיוּתַנָא. 
א,כח אָנָ֣ה ׀ אֲנַ֣חְנוּ עֹלִ֗ים אַחֵינוּ֩ הֵמַ֨סּוּ אֶת־לְבָבֵ֜נוּ לֵאמֹ֗ר עַ֣ם גָּד֤וֹל וָרָם֙ מִמֶּ֔נּוּ עָרִ֛ים גְּדֹלֹ֥ת וּבְצוּרֹ֖ת בַּשָּׁמָ֑יִם וְגַם־בְּנֵ֥י עֲנָקִ֖ים רָאִ֥ינוּ שָֽׁם׃ לְאָן אֲנַחְנָא סָלְקִין, אֲחַנָא תְּבַרוּ יָת לִבַּנָא לְמֵימַר עַם רַב וְתַקִּיף מִנַּנָא, קִרְוִין רַבְרְבָן וּכְּרִיכָן, עַד צֵית שְׁמַיָּא; וְאַף בְּנֵי גִּבָּרָא, חֲזֵינָא תַּמָּן. 
א,כט וָֽאֹמַ֖ר אֲלֵכֶ֑ם לֹא־תַֽעַרְצ֥וּן וְֽלֹא־תִֽירְא֖וּן מֵהֶֽם׃ וַאֲמַרִית, לְכוֹן:  לָא תִּתַּבְרוּן וְלָא תִּדְחֲלוּן, מִנְּהוֹן. 
א,ל יְהוָ֤ה אֱלֹֽהֵיכֶם֙ הַֽהֹלֵ֣ךְ לִפְנֵיכֶ֔ם ה֖וּא יִלָּחֵ֣ם לָכֶ֑ם כְּ֠כֹל אֲשֶׁ֨ר עָשָׂ֧ה אִתְּכֶ֛ם בְּמִצְרַ֖יִם לְעֵֽינֵיכֶֽם׃ יְיָ אֱלָהֲכוֹן דִּמְדַבַּר קֳדָמֵיכוֹן, מֵימְרֵיהּ יְגִיחַ לְכוֹן:  כְּכֹל דַּעֲבַד עִמְּכוֹן, בְּמִצְרַיִם--לְעֵינֵיכוֹן. 
א,לא וּבַמִּדְבָּר֙ אֲשֶׁ֣ר רָאִ֔יתָ אֲשֶׁ֤ר נְשָֽׂאֲךָ֙ יְהוָ֣ה אֱלֹהֶ֔יךָ כַּֽאֲשֶׁ֥ר יִשָּׂא־אִ֖ישׁ אֶת־בְּנ֑וֹ בְּכָל־הַדֶּ֨רֶךְ֙ אֲשֶׁ֣ר הֲלַכְתֶּ֔ם עַד־בֹּֽאֲכֶ֖ם עַד־הַמָּק֥וֹם הַזֶּֽה׃ וּבְמַדְבְּרָא, דַּחֲזֵיתָא, דְּסוֹבְרָךְ יְיָ אֱלָהָךְ, כְּמָא דִּמְסוֹבַר גֻּבְרָא יָת בְּרֵיהּ--בְּכָל אוֹרְחָא דְּהַלֵּיכְתּוּן, עַד מֵיתֵיכוֹן עַד אַתְרָא הָדֵין. 
א,לב וּבַדָּבָ֖ר הַזֶּ֑ה אֵֽינְכֶם֙ מַֽאֲמִינִ֔ם בַּֽיהוָ֖ה אֱלֹֽהֵיכֶֽם׃ וּבְפִתְגָמָא, הָדֵין--לֵיתֵיכוֹן, מְהֵימְנִין, בְּמֵימְרָא דַּייָ, אֱלָהֲכוֹן. 
א,לג הַֽהֹלֵ֨ךְ לִפְנֵיכֶ֜ם בַּדֶּ֗רֶךְ לָת֥וּר לָכֶ֛ם מָק֖וֹם לַחֲנֹֽתְכֶ֑ם בָּאֵ֣שׁ ׀ לַ֗יְלָה לַרְאֹֽתְכֶם֙ בַּדֶּ֨רֶךְ֙ אֲשֶׁ֣ר תֵּֽלְכוּ־בָ֔הּ וּבֶֽעָנָ֖ן יוֹמָֽם׃ דִּמְדַבַּר קֳדָמֵיכוֹן בְּאוֹרְחָא, לְאַתְקָנָא לְכוֹן אֲתַר בֵּית מִשְׁרֵי--לְאַשְׁרָיוּתְכוֹן:  בְּעַמּוּדָא דְּאִישָׁתָא בְּלֵילְיָא, לְאַחְזָיוּתְכוֹן בְּאוֹרְחָא דִּתְהָכוּן בַּהּ, וּבְעַמּוּדָא דַּעֲנָנָא, בִּימָמָא. 
א,לד וַיִּשְׁמַ֥ע יְהוָ֖ה אֶת־ק֣וֹל דִּבְרֵיכֶ֑ם וַיִּקְצֹ֖ף וַיִּשָּׁבַ֥ע לֵאמֹֽר׃ וּשְׁמִיעַ קֳדָם יְיָ, יָת קָל פִּתְגָמֵיכוֹן; וּרְגֵיז, וְקַיֵּים לְמֵימַר. 
א,לה אִם־יִרְאֶ֥ה אִישׁ֙ בָּֽאֲנָשִׁ֣ים הָאֵ֔לֶּה הַדּ֥וֹר הָרָ֖ע הַזֶּ֑ה אֵ֚ת הָאָ֣רֶץ הַטּוֹבָ֔ה אֲשֶׁ֣ר נִשְׁבַּ֔עְתִּי לָתֵ֖ת לַאֲבֹֽתֵיכֶֽם׃ אִם יִחְזֵי גְּבַר בְּגֻבְרַיָּא הָאִלֵּין, דָּרָא בִּישָׁא הָדֵין--יָת, אַרְעָא טָבְתָא, דְּקַיֵּימִית, לְמִתַּן לַאֲבָהָתְכוֹן. 
א,לו זֽוּלָתִ֞י כָּלֵ֤ב בֶּן־יְפֻנֶּה֙ ה֣וּא יִרְאֶ֔נָּה וְלֽוֹ־אֶתֵּ֧ן אֶת־הָאָ֛רֶץ אֲשֶׁ֥ר דָּֽרַךְ־בָּ֖הּ וּלְבָנָ֑יו יַ֕עַן אֲשֶׁ֥ר מִלֵּ֖א אַֽחֲרֵ֥י יְהוָֽה׃ אֱלָהֵין כָּלֵב בַּר יְפֻנֶּה, הוּא יִחְזֵינַהּ, וְלֵיהּ אֶתֵּין יָת אַרְעָא דִּדְרַךְ בַּהּ, וְלִבְנוֹהִי--חֲלָף, דְּאַשְׁלֵים בָּתַר דַּחְלְתָא דַּייָ. 
א,לז גַּם־בִּי֙ הִתְאַנַּ֣ף יְהוָ֔ה בִּגְלַלְכֶ֖ם לֵאמֹ֑ר גַּם־אַתָּ֖ה לֹֽא־תָבֹ֥א שָֽׁם׃ אַף עֲלַי הֲוָה רְגַז מִן קֳדָם יְיָ, בְּדִילְכוֹן לְמֵימַר:  אַף אַתְּ, לָא תֵּיעוֹל לְתַמָּן. 
א,לח יְהוֹשֻׁ֤עַ בִּן־נוּן֙ הָֽעֹמֵ֣ד לְפָנֶ֔יךָ ה֖וּא יָ֣בֹא שָׁ֑מָּה אֹת֣וֹ חַזֵּ֔ק כִּי־ה֖וּא יַנְחִלֶ֥נָּה אֶת־יִשְׂרָאֵֽל׃ יְהוֹשׁוּעַ בַּר נוּן דְּקָאֵים קֳדָמָךְ, הוּא יֵיעוֹל לְתַמָּן; יָתֵיהּ תַּקֵּיף, אֲרֵי הוּא יַחְסְנִנַּהּ לְיִשְׂרָאֵל. 
א,לט וְטַפְּכֶם֩ אֲשֶׁ֨ר אֲמַרְתֶּ֜ם לָבַ֣ז יִֽהְיֶ֗ה וּ֠בְנֵיכֶם אֲשֶׁ֨ר לֹֽא־יָדְע֤וּ הַיּוֹם֙ ט֣וֹב וָרָ֔ע הֵ֖מָּה יָבֹ֣אוּ שָׁ֑מָּה וְלָהֶ֣ם אֶתְּנֶ֔נָּה וְהֵ֖ם יִֽירָשֽׁוּהָ׃ וְטַפְלְכוֹן דַּאֲמַרְתּוּן לְבִזָּא יְהוֹן, וּבְנֵיכוֹן דְּלָא יְדַעוּ יוֹמָא דֵּין טָב וּבִישׁ--אִנּוּן, יֵיעֲלוּן לְתַמָּן; וּלְהוֹן אֶתְּנִנַּהּ, וְאִנּוּן יֵירְתוּנַּהּ. 
א,מ וְאַתֶּ֖ם פְּנ֣וּ לָכֶ֑ם וּסְע֥וּ הַמִּדְבָּ֖רָה דֶּ֥רֶךְ יַם־סֽוּף׃ וְאַתּוּן, אִתְפְּנוֹ לְכוֹן; וְטוּלוּ לְמַדְבְּרָא, אוֹרַח יַמָּא דְּסוּף. 
א,מא וַֽתַּעֲנ֣וּ ׀ וַתֹּֽאמְר֣וּ אֵלַ֗י חָטָאנוּ֮ לַֽיהוָה֒ אֲנַ֤חְנוּ נַֽעֲלֶה֙ וְנִלְחַ֔מְנוּ כְּכֹ֥ל אֲשֶׁר־צִוָּ֖נוּ יְהוָ֣ה אֱלֹהֵ֑ינוּ וַֽתַּחְגְּר֗וּ אִ֚ישׁ אֶת־כְּלֵ֣י מִלְחַמְתּ֔וֹ וַתָּהִ֖ינוּ לַֽעֲלֹ֥ת הָהָֽרָה׃ וַאֲתֵיבְתּוּן וַאֲמַרְתּוּן לִי, חַבְנָא קֳדָם יְיָ--אֲנַחְנָא נִסַּק וּנְגִיחַ קְרָב, כְּכֹל דְּפַקְּדַנָא יְיָ אֱלָהַנָא; וְזָרֵיזְתּוּן, גְּבַר יָת מָנֵי קְרָבֵיהּ, וְשָׁרִיתוּן, לְמִסַּק לְטוּרָא. 
א,מב וַיֹּ֨אמֶר יְהוָ֜ה אֵלַ֗י אֱמֹ֤ר לָהֶם֙ לֹ֤א תַֽעֲלוּ֙ וְלֹא־תִלָּ֣חֲמ֔וּ כִּ֥י אֵינֶ֖נִּי בְּקִרְבְּכֶ֑ם וְלֹא֙ תִּנָּ֣גְפ֔וּ לִפְנֵ֖י אֹֽיְבֵיכֶֽם׃ וַאֲמַר יְיָ לִי, אֵימַר לְהוֹן לָא תִּסְּקוּן וְלָא תְּגִיחוּן קְרָב--אֲרֵי לֵית שְׁכִינְתִי, בֵּינֵיכוֹן; וְלָא, תִּתַּבְרוּן, קֳדָם, בַּעֲלֵי דְּבָבֵיכוֹן. 
א,מג וָֽאֲדַבֵּ֥ר אֲלֵיכֶ֖ם וְלֹ֣א שְׁמַעְתֶּ֑ם וַתַּמְרוּ֙ אֶת־פִּ֣י יְהוָ֔ה וַתָּזִ֖דוּ וַתַּֽעֲל֥וּ הָהָֽרָה׃ וּמַלֵּילִית עִמְּכוֹן, וְלָא קַבֵּילְתּוּן; וְסָרֵיבְתּוּן עַל מֵימְרָא דַּייָ, וְאַרְשַׁעְתּוּן וּסְלֵיקְתּוּן לְטוּרָא. 
א,מד וַיֵּצֵ֨א הָֽאֱמֹרִ֜י הַיֹּשֵׁ֨ב בָּהָ֤ר הַהוּא֙ לִקְרַאתְכֶ֔ם וַיִּרְדְּפ֣וּ אֶתְכֶ֔ם כַּֽאֲשֶׁ֥ר תַּֽעֲשֶׂ֖ינָה הַדְּבֹרִ֑ים וַֽיַּכְּת֥וּ אֶתְכֶ֛ם בְּשֵׂעִ֖יר עַד־חָרְמָֽה׃ וּנְפַק אֱמוֹרָאָה דְּיָתֵיב בְּטוּרָא הַהוּא, לְקַדָּמוּתְכוֹן, וּרְדַפוּ יָתְכוֹן, כְּמָא דְּנָתְזָן דַּבְרַיָתָא; וּטְרַדוּ יָתְכוֹן בְּשֵׂעִיר, עַד חָרְמָה. 
א,מה וַתָּשֻׁ֥בוּ וַתִּבְכּ֖וּ לִפְנֵ֣י יְהוָ֑ה וְלֹֽא־שָׁמַ֤ע יְהוָה֙ בְּקֹ֣לְכֶ֔ם וְלֹ֥א הֶֽאֱזִ֖ין אֲלֵיכֶֽם׃ וְתַבְתּוּן וּבְכֵיתוֹן, קֳדָם יְיָ; וְלָא קַבֵּיל יְיָ צְלוֹתְכוֹן, וְלָא אַצֵּית לְמִלֵּיכוֹן. 
א,מו וַתֵּֽשְׁב֥וּ בְקָדֵ֖שׁ יָמִ֣ים רַבִּ֑ים כַּיָּמִ֖ים אֲשֶׁ֥ר יְשַׁבְתֶּֽם׃ וִיתֵיבְתּוּן בִּרְקַם, יוֹמִין סַגִּיאִין, כְּיוֹמַיָּא, דִּיתֵיבְתּוּן. 
ב,א וַנֵּ֜פֶן וַנִּסַּ֤ע הַמִּדְבָּ֨רָה֙ דֶּ֣רֶךְ יַם־ס֔וּף כַּֽאֲשֶׁ֛ר דִּבֶּ֥ר יְהוָ֖ה אֵלָ֑י וַנָּ֥סָב אֶת־הַר־שֵׂעִ֖יר יָמִ֥ים רַבִּֽים׃ {ס} וְאִתְפְּנִינָא וּנְטַלְנָא לְמַדְבְּרָא, אוֹרַח יַמָּא דְּסוּף, כְּמָא דְּמַלֵּיל יְיָ, עִמִּי; וְאַקֵּיפְנָא יָת טוּרָא דְּשֵׂעִיר, יוֹמִין סַגִּיאִין. {ס}
ב,ב וַיֹּ֥אמֶר יְהוָ֖ה אֵלַ֥י לֵאמֹֽר׃ וַאֲמַר יְיָ, לִי לְמֵימַר. 
ב,ג רַב־לָכֶ֕ם סֹ֖ב אֶת־הָהָ֣ר הַזֶּ֑ה פְּנ֥וּ לָכֶ֖ם צָפֹֽנָה׃ סַגִּי לְכוֹן, דְּאַקֵּיפְתּוּן יָת טוּרָא הָדֵין; אִתְפְּנוֹ לְכוֹן, לְצִפּוּנָא. 
ב,ד וְאֶת־הָעָם֮ צַ֣ו לֵאמֹר֒ אַתֶּ֣ם עֹֽבְרִ֗ים בִּגְבוּל֙ אֲחֵיכֶ֣ם בְּנֵֽי־עֵשָׂ֔ו הַיֹּֽשְׁבִ֖ים בְּשֵׂעִ֑יר וְיִֽירְא֣וּ מִכֶּ֔ם וְנִשְׁמַרְתֶּ֖ם מְאֹֽד׃ וְיָת עַמָּא, פַּקֵּיד לְמֵימַר, אַתּוּן עָבְרִין בִּתְחוּם אֲחֵיכוֹן בְּנֵי עֵשָׂו, דְּיָתְבִין בְּשֵׂעִיר; וְיִדְחֲלוּן מִנְּכוֹן, וְתִסְתַּמְרוּן לַחְדָּא. 
ב,ה אַל־תִּתְגָּר֣וּ בָ֔ם כִּ֠י לֹֽא־אֶתֵּ֤ן לָכֶם֙ מֵֽאַרְצָ֔ם עַ֖ד מִדְרַ֣ךְ כַּף־רָ֑גֶל כִּֽי־יְרֻשָּׁ֣ה לְעֵשָׂ֔ו נָתַ֖תִּי אֶת־הַ֥ר שֵׂעִֽיר׃ לָא תִּתְגָּרַן בְּהוֹן--אֲרֵי לָא אֶתֵּין לְכוֹן מֵאֲרַעְהוֹן, עַד מִדְרַךְ פַּרְסַת רְגַל:  אֲרֵי יְרֻתָּא לְעֵשָׂו, יְהַבִית יָת טוּרָא דְּשֵׂעִיר. 
ב,ו אֹ֣כֶל תִּשְׁבְּר֧וּ מֵֽאִתָּ֛ם בַּכֶּ֖סֶף וַֽאֲכַלְתֶּ֑ם וְגַם־מַ֜יִם תִּכְר֧וּ מֵֽאִתָּ֛ם בַּכֶּ֖סֶף וּשְׁתִיתֶֽם׃ עֲבוּרָא תִּזְבְּנוּן מִנְּהוֹן בְּכַסְפָּא, וְתֵיכְלוּן; וְאַף מַיָּא תִּזְבְּנוּן מִנְּהוֹן, בְּכַסְפָּא--וְתִשְׁתּוֹן. 
ב,ז כִּי֩ יְהוָ֨ה אֱלֹהֶ֜יךָ בֵּֽרַכְךָ֗ בְּכֹל֙ מַֽעֲשֵׂ֣ה יָדֶ֔ךָ יָדַ֣ע לֶכְתְּךָ֔ אֶת־הַמִּדְבָּ֥ר הַגָּדֹ֖ל הַזֶּ֑ה זֶ֣ה ׀ אַרְבָּעִ֣ים שָׁנָ֗ה יְהוָ֤ה אֱלֹהֶ֨יךָ֙ עִמָּ֔ךְ לֹ֥א חָסַ֖רְתָּ דָּבָֽר׃ אֲרֵי יְיָ אֱלָהָךְ בָּרְכָךְ, בְּכֹל עוּבָדֵי יְדָךְ--סוֹפֵיק לָךְ צֻרְכָּךְ בִּמְהָכָךְ, יָת מַדְבְּרָא רַבָּא הָדֵין:  דְּנָן אַרְבְּעִין שְׁנִין, מֵימְרָא דַּייָ אֱלָהָךְ בְּסַעֲדָךְ--לָא חֲסַרְתָּא, מִדָּעַם. 
ב,ח וַֽנַּעֲבֹ֞ר מֵאֵ֧ת אַחֵ֣ינוּ בְנֵֽי־עֵשָׂ֗ו הַיֹּֽשְׁבִים֙ בְּשֵׂעִ֔יר מִדֶּ֨רֶךְ֙ הָֽעֲרָבָ֔ה מֵֽאֵילַ֖ת וּמֵֽעֶצְיֹ֣ן גָּ֑בֶר {ס} וַנֵּ֨פֶן֙ וַֽנַּעֲבֹ֔ר דֶּ֖רֶךְ מִדְבַּ֥ר מוֹאָֽב׃ וַעֲבַרְנָא מִלְּוָת אֲחַנָא בְּנֵי עֵשָׂו, דְּיָתְבִין בְּשֵׂעִיר, מֵאוֹרַח מֵישְׁרָא, מֵאֵילַת וּמֵעֶצְיוֹן גָּבֶר;  {ס}  וְאִתְפְּנִינָא, וַעֲבַרְנָא, אוֹרַח, מַדְבְּרָא דְּמוֹאָב. 
ב,ט וַיֹּ֨אמֶר יְהוָ֜ה אֵלַ֗י אַל־תָּ֨צַר֙ אֶת־מוֹאָ֔ב וְאַל־תִּתְגָּ֥ר בָּ֖ם מִלְחָמָ֑ה כִּ֠י לֹֽא־אֶתֵּ֨ן לְךָ֤ מֵֽאַרְצוֹ֙ יְרֻשָּׁ֔ה כִּ֣י לִבְנֵי־ל֔וֹט נָתַ֥תִּי אֶת־עָ֖ר יְרֻשָּֽׁה׃ וַאֲמַר יְיָ לִי, לָא תְּצוּר עַל מוֹאֲבָאֵי, וְלָא תִּתְגָּרֵי לְמֶעֱבַד עִמְּהוֹן, קְרָב:  אֲרֵי לָא אֶתֵּין לָךְ מֵאַרְעֲהוֹן, יְרֻתָּא--אֲרֵי לִבְנֵי לוֹט, יְהַבִית יָת לְחָיַת יְרֻתָּא. 
ב,י הָֽאֵמִ֥ים לְפָנִ֖ים יָ֣שְׁבוּ בָ֑הּ עַ֣ם גָּד֥וֹל וְרַ֛ב וָרָ֖ם כָּֽעֲנָקִֽים׃ אֵימְתָנֵי מִלְּקַדְמִין, יְתִיבוּ בַּהּ--עַם רַב וְסַגִּי וְתַקִּיף, כְּגִבָּרַיָּא. 
ב,יא רְפָאִ֛ים יֵחָֽשְׁב֥וּ אַף־הֵ֖ם כָּֽעֲנָקִ֑ים וְהַמֹּ֣אָבִ֔ים יִקְרְא֥וּ לָהֶ֖ם אֵמִֽים׃ גִּבָּרַיָּא מִתְחַשְׁבִין אַף אִנּוּן, כְּגִבָּרַיָּא; וּמוֹאֲבָאֵי, קָרַן לְהוֹן אֵימְתָנֵי. 
ב,יב וּבְשֵׂעִ֞יר יָֽשְׁב֣וּ הַֽחֹרִים֮ לְפָנִים֒ וּבְנֵ֧י עֵשָׂ֣ו יִֽירָשׁ֗וּם וַיַּשְׁמִידוּם֙ מִפְּנֵיהֶ֔ם וַיֵּֽשְׁב֖וּ תַּחְתָּ֑ם כַּֽאֲשֶׁ֧ר עָשָׂ֣ה יִשְׂרָאֵ֗ל לְאֶ֨רֶץ֙ יְרֻשָּׁת֔וֹ אֲשֶׁר־נָתַ֥ן יְהוָ֖ה לָהֶֽם׃ וּבְשֵׂעִיר יְתִיבוּ חוֹרָאֵי, מִלְּקַדְמִין, וּבְנֵי עֵשָׂו תָּרִיכוּנוּן וְשֵׁיצִיאוּנוּן מִן קֳדָמֵיהוֹן, וִיתִיבוּ בַּאֲתַרְהוֹן:  כְּמָא דַּעֲבַד יִשְׂרָאֵל, לַאֲרַע יְרֻתְּתֵיהּ, דִּיהַב יְיָ, לְהוֹן. 
ב,יג עַתָּ֗ה קֻ֛מוּ וְעִבְר֥וּ לָכֶ֖ם אֶת־נַ֣חַל זָ֑רֶד וַֽנַּעֲבֹ֖ר אֶת־נַ֥חַל זָֽרֶד׃ כְּעַן, קוּמוּ וְעֵיבַרוּ לְכוֹן--יָת נַחְלָא דְּזָרֶד; וַעֲבַרְנָא, יָת נַחְלָא דְּזָרֶד. 
ב,יד וְהַיָּמִ֞ים אֲשֶׁר־הָלַ֣כְנוּ ׀ מִקָּדֵ֣שׁ בַּרְנֵ֗עַ עַ֤ד אֲשֶׁר־עָבַ֨רְנוּ֙ אֶת־נַ֣חַל זֶ֔רֶד שְׁלֹשִׁ֥ים וּשְׁמֹנֶ֖ה שָׁנָ֑ה עַד־תֹּ֨ם כָּל־הַדּ֜וֹר אַנְשֵׁ֤י הַמִּלְחָמָה֙ מִקֶּ֣רֶב הַֽמַּחֲנֶ֔ה כַּֽאֲשֶׁ֛ר נִשְׁבַּ֥ע יְהוָ֖ה לָהֶֽם׃ וְיוֹמַיָּא דְּהַלֵּיכְנָא מֵרְקַם גֵּיאָה, עַד דַּעֲבַרְנָא יָת נַחְלָא דְּזֶרֶד, תְּלָתִין וְתַמְנֵי, שְׁנִין--עַד דְּסָף כָּל דָּרָא גֻּבְרֵי מְגִיחֵי קְרָבָא, מִגּוֹ מַשְׁרִיתָא, כְּמָא דְּקַיֵּים יְיָ, לְהוֹן. 
ב,טו וְגַ֤ם יַד־יְהוָה֙ הָ֣יְתָה בָּ֔ם לְהֻמָּ֖ם מִקֶּ֣רֶב הַֽמַּחֲנֶ֑ה עַ֖ד תֻּמָּֽם׃ וְאַף מַחָא מִן קֳדָם יְיָ הֲוָת בְּהוֹן, לְשֵׁיצָיוּתְהוֹן מִגּוֹ מַשְׁרִיתָא, עַד, דִּשְׁלִימוּ. 
ב,טז וַיְהִ֨י כַֽאֲשֶׁר־תַּ֜מּוּ כָּל־אַנְשֵׁ֧י הַמִּלְחָמָ֛ה לָמ֖וּת מִקֶּ֥רֶב הָעָֽם׃ {ס} וַהֲוָה כַּד שְׁלִימוּ כָּל גֻּבְרֵי מְגִיחֵי קְרָבָא, לִמְמָת--מִגּוֹ עַמָּא. {ס}
ב,יז וַיְדַבֵּ֥ר יְהוָ֖ה אֵלַ֥י לֵאמֹֽר׃ וּמַלֵּיל יְיָ, לִי לְמֵימַר. 
ב,יח אַתָּ֨ה עֹבֵ֥ר הַיּ֛וֹם אֶת־גְּב֥וּל מוֹאָ֖ב אֶת־עָֽר׃ אַתְּ עָבַר יוֹמָא דֵּין יָת תְּחוּם מוֹאָב, יָת לְחָיַת. 
ב,יט וְקָֽרַבְתָּ֗ מ֚וּל בְּנֵ֣י עַמּ֔וֹן אַל־תְּצֻרֵ֖ם וְאַל־תִּתְגָּ֣ר בָּ֑ם כִּ֣י לֹֽא־אֶ֠תֵּן מֵאֶ֨רֶץ בְּנֵֽי־עַמּ֤וֹן לְךָ֙ יְרֻשָּׁ֔ה כִּ֥י לִבְנֵי־ל֖וֹט נְתַתִּ֥יהָ יְרֻשָּֽׁה׃ וְתִתְקָרַב, לָקֳבֵיל בְּנֵי עַמּוֹן--לָא תְּצוּר עֲלֵיהוֹן, וְלָא תִּתְגָּרֵי לְמֶעֱבַד עִמְּהוֹן קְרָב:  אֲרֵי לָא אֶתֵּין מֵאֲרַע בְּנֵי עַמּוֹן לָךְ, יְרֻתָּא--אֲרֵי לִבְנֵי לוֹט, יְהַבְתַּהּ יְרֻתָּא. 
ב,כ אֶֽרֶץ־רְפָאִ֥ים תֵּֽחָשֵׁ֖ב אַף־הִ֑וא רְפָאִ֤ים יָֽשְׁבוּ־בָהּ֙ לְפָנִ֔ים וְהָֽעַמֹּנִ֔ים יִקְרְא֥וּ לָהֶ֖ם זַמְזֻמִּֽים׃ אֲרַע גִּבָּרַיָּא מִתְחַשְׁבָא, אַף הִיא:  גִּבָּרִין יְתִיבוּ בַּהּ, מִלְּקַדְמִין, וְעַמּוֹנָאֵי, קָרַן לְהוֹן חַשְׁבָּנֵי. 
ב,כא עַ֣ם גָּד֥וֹל וְרַ֛ב וָרָ֖ם כָּֽעֲנָקִ֑ים וַיַּשְׁמִידֵ֤ם יְהוָה֙ מִפְּנֵיהֶ֔ם וַיִּֽירָשֻׁ֖ם וַיֵּֽשְׁב֥וּ תַחְתָּֽם׃ עַם רַב וְסַגִּי וְתַקִּיף, כְּגִבָּרַיָּא; וְשֵׁיצִינוּן יְיָ מִן קֳדָמֵיהוֹן, וְתָרִיכוּנוּן וִיתִיבוּ בַּאֲתַרְהוֹן. 
ב,כב כַּֽאֲשֶׁ֤ר עָשָׂה֙ לִבְנֵ֣י עֵשָׂ֔ו הַיֹּֽשְׁבִ֖ים בְּשֵׂעִ֑יר אֲשֶׁ֨ר הִשְׁמִ֤יד אֶת־הַֽחֹרִי֙ מִפְּנֵיהֶ֔ם וַיִּֽירָשֻׁם֙ וַיֵּֽשְׁב֣וּ תַחְתָּ֔ם עַ֖ד הַיּ֥וֹם הַזֶּֽה׃ כְּמָא דַּעֲבַד לִבְנֵי עֵשָׂו, דְּיָתְבִין בְּשֵׂעִיר--דְּשֵׁיצִי יָת חוֹרָאֵי, מִן קֳדָמֵיהוֹן, וְתָרִיכוּנוּן וִיתִיבוּ בַּאֲתַרְהוֹן, עַד יוֹמָא הָדֵין. 
ב,כג וְהָֽעַוִּ֛ים הַיֹּֽשְׁבִ֥ים בַּֽחֲצֵרִ֖ים עַד־עַזָּ֑ה כַּפְתֹּרִים֙ הַיֹּֽצְאִ֣ים מִכַּפְתֹּ֔ר הִשְׁמִידֻ֖ם וַיֵּֽשְׁב֥וּ תַחְתָּֽם׃ וְעַוָּאֵי דְּיָתְבִין בִּרְפִיחַ, עַד עַזָּה--קְפֻטְקָאֵי דִּנְפַקוּ מִקְּפֻטְקְיָא, שֵׁיצִיאוּנוּן וִיתִיבוּ בַּאֲתַרְהוֹן. 
ב,כד ק֣וּמוּ סְּע֗וּ וְעִבְרוּ֮ אֶת־נַ֣חַל אַרְנֹן֒ רְאֵ֣ה נָתַ֣תִּי בְ֠יָֽדְךָ אֶת־סִיחֹ֨ן מֶֽלֶךְ־חֶשְׁבּ֧וֹן הָֽאֱמֹרִ֛י וְאֶת־אַרְצ֖וֹ הָחֵ֣ל רָ֑שׁ וְהִתְגָּ֥ר בּ֖וֹ מִלְחָמָֽה׃ קוּמוּ טוּלוּ, וְעֵיבַרוּ יָת נַחְלָא דְּאַרְנוֹן--חֲזִי דִּמְסַרִית בִּידָךְ יָת סִיחוֹן מַלְכָּא דְּחֶשְׁבּוֹן אֱמוֹרָאָה וְיָת אַרְעֵיהּ, שָׁרִי לְתָרָכוּתֵיהּ; וְתִתְגְּרֵי לְמֶעֱבַד עִמֵּיהּ, קְרָב. 
ב,כה הַיּ֣וֹם הַזֶּ֗ה אָחֵל֙ תֵּ֤ת פַּחְדְּךָ֙ וְיִרְאָ֣תְךָ֔ עַל־פְּנֵי֙ הָֽעַמִּ֔ים תַּ֖חַת כָּל־הַשָּׁמָ֑יִם אֲשֶׁ֤ר יִשְׁמְעוּן֙ שִׁמְעֲךָ֔ וְרָֽגְז֥וּ וְחָל֖וּ מִפָּנֶֽיךָ׃ יוֹמָא הָדֵין, אֲשָׁרֵי לְמִתַּן זֵיעֲתָךְ וְדַחְלְתָךְ, עַל אַפֵּי עַמְמַיָּא, דִּתְחוֹת כָּל שְׁמַיָּא--דְּיִשְׁמְעוּן שִׁמְעָךְ, וִיזוּעוּן וְיִתַּבְרוּן מִן קֳדָמָךְ. 
ב,כו וָֽאֶשְׁלַ֤ח מַלְאָכִים֙ מִמִּדְבַּ֣ר קְדֵמ֔וֹת אֶל־סִיח֖וֹן מֶ֣לֶךְ חֶשְׁבּ֑וֹן דִּבְרֵ֥י שָׁל֖וֹם לֵאמֹֽר׃ וּשְׁלַחִית אִזְגַּדִּין מִמִּדְבַּר קְדֵימוֹת, לְוָת סִיחוֹן מַלְכָּא דְּחֶשְׁבּוֹן, פִּתְגָמֵי שְׁלָמָא, לְמֵימַר. 
ב,כז אֶעְבְּרָ֣ה בְאַרְצֶ֔ךָ בַּדֶּ֥רֶךְ בַּדֶּ֖רֶךְ אֵלֵ֑ךְ לֹ֥א אָס֖וּר יָמִ֥ין וּשְׂמֹֽאול׃ אֶעְבַּר בְּאַרְעָךְ, בְּאוֹרְחָא בְּאוֹרְחָא אֵיזֵיל:  לָא אֶסְטֵי, לְיַמִּינָא וְלִסְמָאלָא. 
ב,כח אֹ֣כֶל בַּכֶּ֤סֶף תַּשְׁבִּרֵ֨נִי֙ וְאָכַ֔לְתִּי וּמַ֛יִם בַּכֶּ֥סֶף תִּתֶּן־לִ֖י וְשָׁתִ֑יתִי רַ֖ק אֶעְבְּרָ֥ה בְרַגְלָֽי׃ עֲבוּרָא בְּכַסְפָּא תְּזַבֵּין לִי וְאֵיכוֹל, וּמַיָּא בְּכַסְפָּא תִּתֵּין לִי וְאֶשְׁתֵּי; לְחוֹד, אֶעְבַּר בְּרַגְלָי. 
ב,כט כַּֽאֲשֶׁ֨ר עָֽשׂוּ־לִ֜י בְּנֵ֣י עֵשָׂ֗ו הַיֹּֽשְׁבִים֙ בְּשֵׂעִ֔יר וְהַמּ֣וֹאָבִ֔ים הַיֹּֽשְׁבִ֖ים בְּעָ֑ר עַ֤ד אֲשֶֽׁר־אֶעֱבֹר֙ אֶת־הַיַּרְדֵּ֔ן אֶל־הָאָ֕רֶץ אֲשֶׁר־יְהוָ֥ה אֱלֹהֵ֖ינוּ נֹתֵ֥ן לָֽנוּ׃ כְּמָא דַּעֲבַדוּ לִי בְּנֵי עֵשָׂו, דְּיָתְבִין בְּשֵׂעִיר, וּמוֹאֲבָאֵי, דְּיָתְבִין בִּלְחָיַת--עַד דְּאֶעְבַּר, יָת יַרְדְּנָא, לְאַרְעָא, דַּייָ אֱלָהַנָא יָהֵיב לַנָא. 
ב,ל וְלֹ֣א אָבָ֗ה סִיחֹן֙ מֶ֣לֶךְ חֶשְׁבּ֔וֹן הַֽעֲבִרֵ֖נוּ בּ֑וֹ כִּֽי־הִקְשָׁה֩ יְהוָ֨ה אֱלֹהֶ֜יךָ אֶת־רוּח֗וֹ וְאִמֵּץ֙ אֶת־לְבָב֔וֹ לְמַ֛עַן תִּתּ֥וֹ בְיָֽדְךָ֖ כַּיּ֥וֹם הַזֶּֽה׃ {ס} וְלָא אֲבָא, סִיחוֹן מַלְכָּא דְּחֶשְׁבּוֹן, לְמִשְׁבְּקַנָא לְמִעְבַּר, בִּתְחוּמֵיהּ:  אֲרֵי אַקְשִׁי יְיָ אֱלָהָךְ יָת רוּחֵיהּ, וְתַקֵּיף יָת לִבֵּיהּ, בְּדִיל לְמִמְסְרֵיהּ בִּידָךְ, כְּיוֹמָא הָדֵין. {ס}
ב,לא וַיֹּ֤אמֶר יְהוָה֙ אֵלַ֔י רְאֵ֗ה הַֽחִלֹּ֨תִי֙ תֵּ֣ת לְפָנֶ֔יךָ אֶת־סִיחֹ֖ן וְאֶת־אַרְצ֑וֹ הָחֵ֣ל רָ֔שׁ לָרֶ֖שֶׁת אֶת־אַרְצֽוֹ׃ וַאֲמַר יְיָ, לִי, חֲזִי דְּשָׁרִיתִי לְמִמְסַר קֳדָמָךְ, יָת סִיחוֹן וְיָת אַרְעֵיהּ; שָׁרִי לְתָרָכוּתֵיהּ, לְמֵירַת יָת אַרְעֵיהּ. 
ב,לב וַיֵּצֵא֩ סִיחֹ֨ן לִקְרָאתֵ֜נוּ ה֧וּא וְכָל־עַמּ֛וֹ לַמִּלְחָמָ֖ה יָֽהְצָה׃ וּנְפַק סִיחוֹן לְקַדָּמוּתַנָא הוּא וְכָל עַמֵּיהּ, לְאָגָחָא קְרָבָא--לְיָהָץ. 
ב,לג וַֽיִּתְּנֵ֛הוּ יְהוָ֥ה אֱלֹהֵ֖ינוּ לְפָנֵ֑ינוּ וַנַּ֥ךְ אֹת֛וֹ וְאֶת־בָּנָ֖ו וְאֶת־כָּל־עַמּֽוֹ׃ וּמַסְרֵיהּ יְיָ אֱלָהַנָא, קֳדָמַנָא; וּמְחֵינָא יָתֵיהּ וְיָת בְּנוֹהִי, וְיָת כָּל עַמֵּיהּ. 
ב,לד וַנִּלְכֹּ֤ד אֶת־כָּל־עָרָיו֙ בָּעֵ֣ת הַהִ֔וא וַֽנַּחֲרֵם֙ אֶת־כָּל־עִ֣יר מְתִ֔ם וְהַנָּשִׁ֖ים וְהַטָּ֑ף לֹ֥א הִשְׁאַ֖רְנוּ שָׂרִֽיד׃ וּכְבַשְׁנָא יָת כָּל קִרְווֹהִי, בְּעִדָּנָא הַהוּא, וְגַּמַּרְנָא יָת כָּל קִרְוַיָּא גֻּבְרַיָּא, וּנְשַׁיָּא וְטַפְלָא:  לָא אַשְׁאַרְנָא, מְשֵׁיזֵיב. 
ב,לה רַ֥ק הַבְּהֵמָ֖ה בָּזַ֣זְנוּ לָ֑נוּ וּשְׁלַ֥ל הֶֽעָרִ֖ים אֲשֶׁ֥ר לָכָֽדְנוּ׃ לְחוֹד בְּעִירָא, בַּזְנָא לַנָא, וַעֲדַי קִרְוַיָּא, דִּכְבַשְׁנָא. 
ב,לו מֵֽעֲרֹעֵ֡ר אֲשֶׁר֩ עַל־שְׂפַת־נַ֨חַל אַרְנֹ֜ן וְהָעִ֨יר אֲשֶׁ֤ר בַּנַּ֨חַל֙ וְעַד־הַגִּלְעָ֔ד לֹ֤א הָֽיְתָה֙ קִרְיָ֔ה אֲשֶׁ֥ר שָֽׂגְבָ֖ה מִמֶּ֑נּוּ אֶת־הַכֹּ֕ל נָתַ֛ן יְהוָ֥ה אֱלֹהֵ֖ינוּ לְפָנֵֽינוּ׃ מֵעֲרוֹעֵר דְּעַל כֵּיף נַחְלָא דְּאַרְנוֹן וְקַרְתָּא דִּבְנַחְלָא, וְעַד גִּלְעָד, לָא הֲוָת קַרְתָּא, דִּתְקֵיפַת מִנַּנָא:  יָת כּוֹלָא, מְסַר יְיָ אֱלָהַנָא קֳדָמַנָא. 
ב,לז רַ֛ק אֶל־אֶ֥רֶץ בְּנֵֽי־עַמּ֖וֹן לֹ֣א קָרָ֑בְתָּ כָּל־יַ֞ד נַ֤חַל יַבֹּק֙ וְעָרֵ֣י הָהָ֔ר וְכֹ֥ל אֲשֶׁר־צִוָּ֖ה יְהוָ֥ה אֱלֹהֵֽינוּ׃ לְחוֹד לַאֲרַע בְּנֵי עַמּוֹן, לָא קָרֵיבְתָּא:  כָּל כֵּיף נְחַל יֻבְקָא, וְקִרְוֵי טוּרָא, וְכֹל דְּפַקֵּיד, יְיָ אֱלָהַנָא. 
ג,א וַנֵּ֣פֶן וַנַּ֔עַל דֶּ֖רֶךְ הַבָּשָׁ֑ן וַיֵּצֵ֣א עוֹג֩ מֶֽלֶךְ־הַבָּשָׁ֨ן לִקְרָאתֵ֜נוּ ה֧וּא וְכָל־עַמּ֛וֹ לַמִּלְחָמָ֖ה אֶדְרֶֽעִי׃ וְאִתְפְּנִינָא וּסְלֵיקְנָא, לְאוֹרַח מַתְנַן; וּנְפַק עוֹג מַלְכָּא דְּמַתְנַן לְקַדָּמוּתַנָא הוּא וְכָל עַמֵּיהּ, לְאָגָחָא קְרָבָא--לְאֶדְרֶעִי. 
ג,ב וַיֹּ֨אמֶר יְהוָ֤ה אֵלַי֙ אַל־תִּירָ֣א אֹת֔וֹ כִּ֣י בְיָֽדְךָ֞ נָתַ֧תִּי אֹת֛וֹ וְאֶת־כָּל־עַמּ֖וֹ וְאֶת־אַרְצ֑וֹ וְעָשִׂ֣יתָ לּ֔וֹ כַּֽאֲשֶׁ֣ר עָשִׂ֗יתָ לְסִיחֹן֙ מֶ֣לֶךְ הָֽאֱמֹרִ֔י אֲשֶׁ֥ר יוֹשֵׁ֖ב בְּחֶשְׁבּֽוֹן׃ וַאֲמַר יְיָ לִי, לָא תִּדְחַל מִנֵּיהּ--אֲרֵי בִּידָךְ מְסַרִית יָתֵיהּ וְיָת כָּל עַמֵּיהּ, וְיָת אַרְעֵיהּ; וְתַעֲבֵיד לֵיהּ--כְּמָא דַּעֲבַדְתָּא לְסִיחוֹן מַלְכָּא אֱמוֹרָאָה, דְּיָתֵיב בְּחֶשְׁבּוֹן. 
ג,ג וַיִּתֵּן֩ יְהוָ֨ה אֱלֹהֵ֜ינוּ בְּיָדֵ֗נוּ גַּ֛ם אֶת־ע֥וֹג מֶֽלֶךְ־הַבָּשָׁ֖ן וְאֶת־כָּל־עַמּ֑וֹ וַנַּכֵּ֕הוּ עַד־בִּלְתִּ֥י הִשְׁאִֽיר־ל֖וֹ שָׂרִֽיד׃ וּמְסַר יְיָ אֱלָהַנָא בִּידַנָא, אַף יָת עוֹג מַלְכָּא דְּמַתְנַן--וְיָת כָּל עַמֵּיהּ; וּמְחֵינָהִי, עַד דְּלָא אִשְׁתְּאַר לֵיהּ מְשֵׁיזֵיב. 
ג,ד וַנִּלְכֹּ֤ד אֶת־כָּל־עָרָיו֙ בָּעֵ֣ת הַהִ֔וא לֹ֤א הָֽיְתָה֙ קִרְיָ֔ה אֲשֶׁ֥ר לֹֽא־לָקַ֖חְנוּ מֵֽאִתָּ֑ם שִׁשִּׁ֥ים עִיר֙ כָּל־חֶ֣בֶל אַרְגֹּ֔ב מַמְלֶ֥כֶת ע֖וֹג בַּבָּשָֽׁן׃ וּכְבַשְׁנָא יָת כָּל קִרְווֹהִי, בְּעִדָּנָא הַהוּא--לָא הֲוָת קַרְתָּא, דְּלָא נְסֵיבְנָא מִנְּהוֹן:  שִׁתִּין קִרְוִין כָּל בֵּית פֶּלֶךְ טְרָכוֹנָא, מַלְכוּתֵיהּ דְּעוֹג בְּמַתְנַן. 
ג,ה כָּל־אֵ֜לֶּה עָרִ֧ים בְּצֻרֹ֛ת חוֹמָ֥ה גְבֹהָ֖ה דְּלָתַ֣יִם וּבְרִ֑יחַ לְבַ֛ד מֵֽעָרֵ֥י הַפְּרָזִ֖י הַרְבֵּ֥ה מְאֹֽד׃ כָּל אִלֵּין קִרְוִין כְּרִיכָן, מַקְּפָן שׁוּר רָם--דִּילְהוֹן דַּשִּׁין וְעָבְרִין:  בָּר מִקִּרְוֵי פַּצְחַיָּא, דְּסַגִּיאָן לַחְדָּא. 
ג,ו וַנַּֽחֲרֵ֣ם אוֹתָ֔ם כַּֽאֲשֶׁ֣ר עָשִׂ֔ינוּ לְסִיחֹ֖ן מֶ֣לֶךְ חֶשְׁבּ֑וֹן הַֽחֲרֵם֙ כָּל־עִ֣יר מְתִ֔ם הַנָּשִׁ֖ים וְהַטָּֽף׃ וְגַּמַּרְנָא יָתְהוֹן--כְּמָא דַּעֲבַדְנָא, לְסִיחוֹן מַלְכָּא דְּחֶשְׁבּוֹן:  גַּמַּרְנָא כָּל קִרְוַיָּא גֻּבְרַיָּא, נְשַׁיָּא וְטַפְלָא. 
ג,ז וְכָל־הַבְּהֵמָ֛ה וּשְׁלַ֥ל הֶֽעָרִ֖ים בַּזּ֥וֹנוּ לָֽנוּ׃ וְכָל בְּעִירָא וַעֲדַי קִרְוַיָּא, בַּזְנָא לַנָא. 
ג,ח וַנִּקַּ֞ח בָּעֵ֤ת הַהִוא֙ אֶת־הָאָ֔רֶץ מִיַּ֗ד שְׁנֵי֙ מַלְכֵ֣י הָֽאֱמֹרִ֔י אֲשֶׁ֖ר בְּעֵ֣בֶר הַיַּרְדֵּ֑ן מִנַּ֥חַל אַרְנֹ֖ן עַד־הַ֥ר חֶרְמֽוֹן׃ וּנְסֵיבְנָא בְּעִדָּנָא הַהוּא, יָת אַרְעָא, מִיַּד תְּרֵין מַלְכֵי אֱמוֹרָאָה, דִּבְעִבְרָא דְּיַרְדְּנָא--מִנַּחְלָא דְּאַרְנוֹן, עַד טוּרָא דְּחֶרְמוֹן. 
ג,ט צִֽידֹנִ֛ים יִקְרְא֥וּ לְחֶרְמ֖וֹן שִׂרְיֹ֑ן וְהָ֣אֱמֹרִ֔י יִקְרְאוּ־ל֖וֹ שְׂנִֽיר׃ צִידוֹנָאֵי קָרַן לְחֶרְמוֹן, שִׂרְיוֹן; וֶאֱמוֹרָאֵי, קָרַן לֵיהּ טוּר תַּלְגָּא. 
ג,י כֹּ֣ל ׀ עָרֵ֣י הַמִּישֹׁ֗ר וְכָל־הַגִּלְעָד֙ וְכָל־הַבָּשָׁ֔ן עַד־סַלְכָ֖ה וְאֶדְרֶ֑עִי עָרֵ֛י מַמְלֶ֥כֶת ע֖וֹג בַּבָּשָֽׁן׃ כֹּל קִרְוֵי מֵישְׁרָא, וְכָל גִּלְעָד וְכָל מַתְנַן, עַד סַלְכָה, וְאֶדְרֶעִי--קִרְוֵי מַלְכוּתֵיהּ דְּעוֹג, בְּמַתְנַן. 
ג,יא כִּ֣י רַק־ע֞וֹג מֶ֣לֶךְ הַבָּשָׁ֗ן נִשְׁאַר֮ מִיֶּ֣תֶר הָֽרְפָאִים֒ הִנֵּ֤ה עַרְשׂוֹ֙ עֶ֣רֶשׂ בַּרְזֶ֔ל הֲלֹ֣ה הִ֔וא בְּרַבַּ֖ת בְּנֵ֣י עַמּ֑וֹן תֵּ֧שַׁע אַמּ֣וֹת אָרְכָּ֗הּ וְאַרְבַּ֥ע אַמּ֛וֹת רָחְבָּ֖הּ בְּאַמַּת־אִֽישׁ׃ אֲרֵי לְחוֹד עוֹג מַלְכָּא דְּמַתְנַן, אִשְׁתְּאַר מִשְּׁאָר גִּבָּרַיָּא--הָא עַרְסֵיהּ עַרְסָא דְּבַרְזְלָא, הֲלָא הִיא בְּרַבַּת בְּנֵי עַמּוֹן:  תְּשַׁע אַמִּין אֻרְכַּהּ, וְאַרְבַּע אַמִּין פֻּתְיַהּ--בְּאַמַּת מְלַךְ. 
ג,יב וְאֶת־הָאָ֧רֶץ הַזֹּ֛את יָרַ֖שְׁנוּ בָּעֵ֣ת הַהִ֑וא מֵֽעֲרֹעֵ֞ר אֲשֶׁר־עַל־נַ֣חַל אַרְנֹ֗ן וַֽחֲצִ֤י הַֽר־הַגִּלְעָד֙ וְעָרָ֔יו נָתַ֕תִּי לָרֽאוּבֵנִ֖י וְלַגָּדִֽי׃ וְיָת אַרְעָא הָדָא יְרֵיתְנָא, בְּעִדָּנָא הַהוּא; מֵעֲרוֹעֵר דְּעַל נַחְלָא דְּאַרְנוֹן, וּפַלְגוּת טוּרָא דְּגִלְעָד וְקִרְווֹהִי--יְהַבִית, לְשֵׁיבַט רְאוּבֵן וּלְשֵׁיבַט גָּד. 
ג,יג וְיֶ֨תֶר הַגִּלְעָ֤ד וְכָל־הַבָּשָׁן֙ מַמְלֶ֣כֶת ע֔וֹג נָתַ֕תִּי לַֽחֲצִ֖י שֵׁ֣בֶט הַֽמְנַשֶּׁ֑ה כֹּ֣ל חֶ֤בֶל הָֽאַרְגֹּב֙ לְכָל־הַבָּשָׁ֔ן הַה֥וּא יִקָּרֵ֖א אֶ֥רֶץ רְפָאִֽים׃ וּשְׁאָר גִּלְעָד וְכָל מַתְנַן, מַלְכוּתֵיהּ דְּעוֹג--יְהַבִית, לְפַלְגוּת שִׁבְטָא דִּמְנַשֶּׁה:  כֹּל בֵּית פֶּלֶךְ טְרָכוֹנָא לְכָל מַתְנַן, הַהוּא מִתְקְרֵי אֲרַע גִּבָּרַיָּא. 
ג,יד יָאִ֣יר בֶּן־מְנַשֶּׁ֗ה לָקַח֙ אֶת־כָּל־חֶ֣בֶל אַרְגֹּ֔ב עַד־גְּב֥וּל הַגְּשׁוּרִ֖י וְהַמַּֽעֲכָתִ֑י וַיִּקְרָא֩ אֹתָ֨ם עַל־שְׁמ֤וֹ אֶת־הַבָּשָׁן֙ חַוּ֣‍ֹת יָאִ֔יר עַ֖ד הַיּ֥וֹם הַזֶּֽה׃ יָאִיר בַּר מְנַשֶּׁה, נְסֵיב יָת כָּל בֵּית פֶּלֶךְ טְרָכוֹנָא, עַד תְּחוּם גְּשׁוּרָאָה, וַאֲפֵיקֵירוֹס; וּקְרָא יָתְהוֹן עַל שְׁמֵיהּ יָת מַתְנַן כַּפְרָנֵי יָאִיר, עַד יוֹמָא הָדֵין. 
ג,טו וּלְמָכִ֖יר נָתַ֥תִּי אֶת־הַגִּלְעָֽד׃ וּלְמָכִיר, יְהַבִית יָת גִּלְעָד. 
ג,טז וְלָרֽאוּבֵנִ֨י וְלַגָּדִ֜י נָתַ֤תִּי מִן־הַגִּלְעָד֙ וְעַד־נַ֣חַל אַרְנֹ֔ן תּ֥וֹךְ הַנַּ֖חַל וּגְבֻ֑ל וְעַד֙ יַבֹּ֣ק הַנַּ֔חַל גְּב֖וּל בְּנֵ֥י עַמּֽוֹן׃ וּלְשֵׁיבַט רְאוּבֵן וּלְשֵׁיבַט גָּד יְהַבִית מִן גִּלְעָד, וְעַד נַחְלָא דְּאַרְנוֹן, גּוֹ נַחְלָא, וּתְחוּמֵיהּ--וְעַד יֻבְקָא דְּנַחְלָא, תְּחוּמָא דִּבְנֵי עַמּוֹן. 
ג,יז וְהָֽעֲרָבָ֖ה וְהַיַּרְדֵּ֣ן וּגְבֻ֑ל מִכִּנֶּ֗רֶת וְעַ֨ד יָ֤ם הָֽעֲרָבָה֙ יָ֣ם הַמֶּ֔לַח תַּ֛חַת אַשְׁדֹּ֥ת הַפִּסְגָּ֖ה מִזְרָֽחָה׃ וּמֵישְׁרָא, וְיַרְדְּנָא וּתְחוּמֵיהּ--מִגִּנֵּיסַר, וְעַד יַמָּא דְּמֵישְׁרָא יַמָּא דְּמִלְחָא, תְּחוֹת מַשְׁפַּךְ מֵי רָמְתָא, מַדְנְחָא. 
ג,יח וָֽאֲצַ֣ו אֶתְכֶ֔ם בָּעֵ֥ת הַהִ֖וא לֵאמֹ֑ר יְהוָ֣ה אֱלֹֽהֵיכֶ֗ם נָתַ֨ן לָכֶ֜ם אֶת־הָאָ֤רֶץ הַזֹּאת֙ לְרִשְׁתָּ֔הּ חֲלוּצִ֣ים תַּֽעַבְר֗וּ לִפְנֵ֛י אֲחֵיכֶ֥ם בְּנֵֽי־יִשְׂרָאֵ֖ל כָּל־בְּנֵי־חָֽיִל׃ וּפַקֵּידִית יָתְכוֹן, בְּעִדָּנָא הַהוּא לְמֵימַר:  יְיָ אֱלָהֲכוֹן, יְהַב לְכוֹן יָת אַרְעָא הָדָא לְמֵירְתַהּ--מְזָרְזִין תִּעְבְּרוּן קֳדָם אֲחֵיכוֹן בְּנֵי יִשְׂרָאֵל, כָּל מְזָרַז חֵילָא. 
ג,יט רַ֠ק נְשֵׁיכֶ֣ם וְטַפְּכֶם֮ וּמִקְנֵכֶם֒ יָדַ֕עְתִּי כִּֽי־מִקְנֶ֥ה רַ֖ב לָכֶ֑ם יֵֽשְׁבוּ֙ בְּעָ֣רֵיכֶ֔ם אֲשֶׁ֥ר נָתַ֖תִּי לָכֶֽם׃ לְחוֹד נְשֵׁיכוֹן וְטַפְלְכוֹן, וּבְעִירְכוֹן, יָדַעְנָא, אֲרֵי בְּעִיר סַגִּי לְכוֹן--יִתְּבוּן, בְּקִרְוֵיכוֹן, דִּיהַבִית, לְכוֹן. 
ג,כ עַ֠ד אֲשֶׁר־יָנִ֨יחַ יְהוָ֥ה ׀ לַֽאֲחֵיכֶם֮ כָּכֶם֒ וְיָֽרְשׁ֣וּ גַם־הֵ֔ם אֶת־הָאָ֕רֶץ אֲשֶׁ֨ר יְהוָ֧ה אֱלֹֽהֵיכֶ֛ם נֹתֵ֥ן לָהֶ֖ם בְּעֵ֣בֶר הַיַּרְדֵּ֑ן וְשַׁבְתֶּ֗ם אִ֚ישׁ לִֽירֻשָּׁת֔וֹ אֲשֶׁ֥ר נָתַ֖תִּי לָכֶֽם׃ עַד דִּינִיחַ יְיָ לַאֲחֵיכוֹן, כְּוָתְכוֹן, וְיֵירְתוּן אַף אִנּוּן, יָת אַרְעָא דַּייָ אֱלָהֲכוֹן יָהֵיב לְהוֹן בְּעִבְרָא דְּיַרְדְּנָא; וּתְתוּבוּן, גְּבַר לִירֻתְּתֵיהּ, דִּיהַבִית, לְכוֹן. 
ג,כא וְאֶת־יְהוֹשׁ֣וּעַ צִוֵּ֔יתִי בָּעֵ֥ת הַהִ֖וא לֵאמֹ֑ר עֵינֶ֣יךָ הָֽרֹאֹ֗ת אֵת֩ כָּל־אֲשֶׁ֨ר עָשָׂ֜ה יְהוָ֤ה אֱלֹֽהֵיכֶם֙ לִשְׁנֵי֙ הַמְּלָכִ֣ים הָאֵ֔לֶּה כֵּֽן־יַעֲשֶׂ֤ה יְהוָה֙ לְכָל־הַמַּמְלָכ֔וֹת אֲשֶׁ֥ר אַתָּ֖ה עֹבֵ֥ר שָֽׁמָּה׃ וְיָת יְהוֹשׁוּעַ פַּקֵּידִית, בְּעִדָּנָא הַהוּא לְמֵימַר:  עֵינָךְ חֲזָאָה, יָת כָּל דַּעֲבַד יְיָ אֱלָהֲכוֹן לִתְרֵין מַלְכַיָּא הָאִלֵּין--כֵּין יַעֲבֵיד יְיָ לְכָל מַלְכְוָתָא, דְּאַתְּ עָבַר לְתַמָּן. 
ג,כב לֹ֖א תִּֽירָא֑וּם כִּ֚י יְהוָ֣ה אֱלֹֽהֵיכֶ֔ם ה֖וּא הַנִּלְחָ֥ם לָכֶֽם׃ {ס} לָא, תִּדְחֲלוּן מִנְּהוֹן:  אֲרֵי יְיָ אֱלָהֲכוֹן, מֵימְרֵיהּ מְגִיחַ לְכוֹן. {ס}
