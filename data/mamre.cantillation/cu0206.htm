משפטים

כא,א וְאֵ֨לֶּה֙ הַמִּשְׁפָּטִ֔ים אֲשֶׁ֥ר תָּשִׂ֖ים לִפְנֵיהֶֽם׃ וְאִלֵּין, דִּינַיָּא, דְּתַסְדַּר, קֳדָמֵיהוֹן. 
כא,ב כִּ֤י תִקְנֶה֙ עֶ֣בֶד עִבְרִ֔י שֵׁ֥שׁ שָׁנִ֖ים יַֽעֲבֹ֑ד וּבַ֨שְּׁבִעִ֔ת יֵצֵ֥א לַֽחָפְשִׁ֖י חִנָּֽם׃ אֲרֵי תִּזְבּוֹן עַבְדָּא בַּר יִשְׂרָאֵל, שֵׁית שְׁנִין יִפְלַח; וּבִשְׁבִיעֵיתָא--יִפּוֹק לְבַר חוֹרִין, מַגָּן. 
כא,ג אִם־בְּגַפּ֥וֹ יָבֹ֖א בְּגַפּ֣וֹ יֵצֵ֑א אִם־בַּ֤עַל אִשָּׁה֙ ה֔וּא וְיָֽצְאָ֥ה אִשְׁתּ֖וֹ עִמּֽוֹ׃ אִם בִּלְחוֹדוֹהִי יֵיעוֹל, בִּלְחוֹדוֹהִי יִפּוֹק; אִם בְּעֵיל אִתְּתָא הוּא, וְתִפּוֹק אִתְּתֵיהּ עִמֵּיהּ. 
כא,ד אִם־אֲדֹנָיו֙ יִתֶּן־ל֣וֹ אִשָּׁ֔ה וְיָֽלְדָה־ל֥וֹ בָנִ֖ים א֣וֹ בָנ֑וֹת הָֽאִשָּׁ֣ה וִֽילָדֶ֗יהָ תִּֽהְיֶה֙ לַֽאדֹנֶ֔יהָ וְה֖וּא יֵצֵ֥א בְגַפּֽוֹ׃ אִם רִבּוֹנֵיהּ יִתֵּין לֵיהּ אִתְּתָא, וּתְלִיד לֵיהּ בְּנִין אוֹ בְּנָן--אִתְּתָא וּבְנַהָא, תְּהֵי לְרִבּוֹנַהּ, וְהוּא, יִפּוֹק בִּלְחוֹדוֹהִי. 
כא,ה וְאִם־אָמֹ֤ר יֹאמַר֙ הָעֶ֔בֶד אָהַ֨בְתִּי֙ אֶת־אֲדֹנִ֔י אֶת־אִשְׁתִּ֖י וְאֶת־בָּנָ֑י לֹ֥א אֵצֵ֖א חָפְשִֽׁי׃ וְאִם מֵימָר יֵימַר, עַבְדָּא, רָחֵימְנָא יָת רִבּוֹנִי, יָת אִתְּתִי וְיָת בְּנָי; לָא אֶפּוֹק, בַּר חוֹרִין. 
כא,ו וְהִגִּישׁ֤וֹ אֲדֹנָיו֙ אֶל־הָ֣אֱלֹהִ֔ים וְהִגִּישׁוֹ֙ אֶל־הַדֶּ֔לֶת א֖וֹ אֶל־הַמְּזוּזָ֑ה וְרָצַ֨ע אֲדֹנָ֤יו אֶת־אָזְנוֹ֙ בַּמַּרְצֵ֔עַ וַֽעֲבָד֖וֹ לְעֹלָֽם׃ {ס} וִיקָרְבִנֵּיהּ רִבּוֹנֵיהּ, לִקְדָם דַּיָּנַיָּא, וִיקָרְבִנֵּיהּ לְוָת דַּשָּׁא, אוֹ דִּלְוָת מְזוּזְתָא; וְיַרְצַע רִבּוֹנֵיהּ יָת אֻדְנֵיהּ בְּמַרְצְעָא, וִיהֵי לֵיהּ עֶבֶד פָּלַח לְעָלַם. {ס}
כא,ז וְכִֽי־יִמְכֹּ֥ר אִ֛ישׁ אֶת־בִּתּ֖וֹ לְאָמָ֑ה לֹ֥א תֵצֵ֖א כְּצֵ֥את הָֽעֲבָדִֽים׃ וַאֲרֵי יְזַבֵּין גְּבַר יָת בְּרַתֵּיהּ, לְאַמְהוּ--לָא תִּפּוֹק, כְּמַפְּקָנוּת עַבְדַיָּא. 
כא,ח אִם־רָעָ֞ה בְּעֵינֵ֧י אֲדֹנֶ֛יהָ אֲשֶׁר־לא (ל֥וֹ) יְעָדָ֖הּ וְהֶפְדָּ֑הּ לְעַ֥ם נָכְרִ֛י לֹֽא־יִמְשֹׁ֥ל לְמָכְרָ֖הּ בְּבִגְדוֹ־בָֽהּ׃ אִם בִּישָׁת בְּעֵינֵי רִבּוֹנַהּ, דִּיקַיְּמַהּ לֵיהּ--וְיִפְרְקִנַּהּ:  לִגְבַר אָחֳרָן לֵית לֵיהּ רְשׁוּ לְזַבּוֹנַהּ, בְּמִשְׁלְטֵיהּ בַּהּ. 
כא,ט וְאִם־לִבְנ֖וֹ יִֽיעָדֶ֑נָּה כְּמִשְׁפַּ֥ט הַבָּנ֖וֹת יַֽעֲשֶׂה־לָּֽהּ׃ וְאִם לִבְרֵיהּ, יְקַיְּמִנַּהּ--כְּהִלְכָּת בְּנָת יִשְׂרָאֵל, יַעֲבֵיד לַהּ. 
כא,י אִם־אַחֶ֖רֶת יִֽקַּֽח־ל֑וֹ שְׁאֵרָ֛הּ כְּסוּתָ֥הּ וְעֹֽנָתָ֖הּ לֹ֥א יִגְרָֽע׃ אִם אֻחְרָנְתָא, יִסַּב לֵיהּ--זִיּוּנַהּ כְּסוּתַהּ וְעָנְתַהּ, לָא יִמְנַע. 
כא,יא וְאִ֨ם־שְׁלָשׁ־אֵ֔לֶּה לֹ֥א יַֽעֲשֶׂ֖ה לָ֑הּ וְיָֽצְאָ֥ה חִנָּ֖ם אֵ֥ין כָּֽסֶף׃ {ס} וְאִם תְּלָת אִלֵּין--לָא יַעֲבֵיד, לַהּ:  וְתִפּוֹק מַגָּן, דְּלָא כְּסַף. {ס}
כא,יב מַכֵּ֥ה אִ֛ישׁ וָמֵ֖ת מ֥וֹת יוּמָֽת׃ דְּיִמְחֵי לַאֲנָשׁ וְיִקְטְלִנֵּיהּ, אִתְקְטָלָא יִתְקְטִיל. 
כא,יג וַֽאֲשֶׁר֙ לֹ֣א צָדָ֔ה וְהָֽאֱלֹהִ֖ים אִנָּ֣ה לְיָד֑וֹ וְשַׂמְתִּ֤י לְךָ֙ מָק֔וֹם אֲשֶׁ֥ר יָנ֖וּס שָֽׁמָּה׃ {ס} וּדְלָא כְּמַן לֵיהּ, וּמִן קֳדָם יְיָ אִתְמְסַר לִידֵיהּ--וַאֲשַׁוֵּי לָךְ אֲתַר, דְּיִעְרוֹק לְתַמָּן. {ס}
כא,יד וְכִֽי־יָזִ֥ד אִ֛ישׁ עַל־רֵעֵ֖הוּ לְהָרְג֣וֹ בְעָרְמָ֑ה מֵעִ֣ם מִזְבְּחִ֔י תִּקָּחֶ֖נּוּ לָמֽוּת׃ {ס} וַאֲרֵי יַרְשַׁע גְּבַר עַל חַבְרֵיהּ, לְמִקְטְלֵיהּ בִּנְכִילוּ--מִן מַדְבְּחִי, תִּדְבְּרִנֵּיהּ לְמִקְטַל. {ס}
כא,טו וּמַכֵּ֥ה אָבִ֛יו וְאִמּ֖וֹ מ֥וֹת יוּמָֽת׃ {ס} וּדְיִמְחֵי לַאֲבוּהִי וּלְאִמֵּיהּ, אִתְקְטָלָא יִתְקְטִיל. {ס}
כא,טז וְגֹנֵ֨ב אִ֧ישׁ וּמְכָר֛וֹ וְנִמְצָ֥א בְיָד֖וֹ מ֥וֹת יוּמָֽת׃ {ס} וּדְיִגְנוֹב נַפְשָׁא מִבְּנֵי יִשְׂרָאֵל וִיזַבְּנִנֵּיהּ וְיִשְׁתְּכַח בִּידֵיהּ, אִתְקְטָלָא יִתְקְטִיל. {ס}
כא,יז וּמְקַלֵּ֥ל אָבִ֛יו וְאִמּ֖וֹ מ֥וֹת יוּמָֽת׃ {ס} וְדִילוּט לַאֲבוּהִי וּלְאִמֵּיהּ, אִתְקְטָלָא יִתְקְטִיל. {ס}
כא,יח וְכִֽי־יְרִיבֻ֣ן אֲנָשִׁ֔ים וְהִכָּה־אִישׁ֙ אֶת־רֵעֵ֔הוּ בְּאֶ֖בֶן א֣וֹ בְאֶגְרֹ֑ף וְלֹ֥א יָמ֖וּת וְנָפַ֥ל לְמִשְׁכָּֽב׃ וַאֲרֵי יִנְצוֹן גֻּבְרִין--וְיִמְחֵי גְּבַר יָת חַבְרֵיהּ, בְּאַבְנָא אוֹ בְּכֻרְמֵיזָא; וְלָא יְמוּת, וְיִפּוֹל לְבֻטְלָן. 
כא,יט אִם־יָק֞וּם וְהִתְהַלֵּ֥ךְ בַּח֛וּץ עַל־מִשְׁעַנְתּ֖וֹ וְנִקָּ֣ה הַמַּכֶּ֑ה רַ֥ק שִׁבְתּ֛וֹ יִתֵּ֖ן וְרַפֹּ֥א יְרַפֵּֽא׃ {ס} אִם יְקוּם וִיהַלֵּיךְ בְּבָרָא, עַל בָּרְיֵיהּ--וִיהֵי זָכָא מָחְיָא:  לְחוֹד בֻּטְלָנֵיהּ יִתֵּין, וַאֲגַר אַסְיָא יְשַׁלֵּים. {ס}
כא,כ וְכִֽי־יַכֶּה֩ אִ֨ישׁ אֶת־עַבְדּ֜וֹ א֤וֹ אֶת־אֲמָתוֹ֙ בַּשֵּׁ֔בֶט וּמֵ֖ת תַּ֣חַת יָד֑וֹ נָקֹ֖ם יִנָּקֵֽם׃ וַאֲרֵי יִמְחֵי גְּבַר יָת עַבְדֵּיהּ אוֹ יָת אַמְתֵּיהּ, בְּשֻׁלְטָן, וִימוּת, תְּחוֹת יְדֵיהּ--אִתְּדָנָא, יִתְּדָן. 
כא,כא אַ֥ךְ אִם־י֛וֹם א֥וֹ יוֹמַ֖יִם יַֽעֲמֹ֑ד לֹ֣א יֻקַּ֔ם כִּ֥י כַסְפּ֖וֹ הֽוּא׃ {ס} בְּרַם אִם יוֹמָא אוֹ תְּרֵין יוֹמִין, יִתְקַיַּם--לָא יִתְּדָן, אֲרֵי כַּסְפֵּיהּ הוּא. {ס}
כא,כב וְכִֽי־יִנָּצ֣וּ אֲנָשִׁ֗ים וְנָ֨גְפ֜וּ אִשָּׁ֤ה הָרָה֙ וְיָֽצְא֣וּ יְלָדֶ֔יהָ וְלֹ֥א יִֽהְיֶ֖ה אָס֑וֹן עָנ֣וֹשׁ יֵֽעָנֵ֗שׁ כַּֽאֲשֶׁ֨ר יָשִׁ֤ית עָלָיו֙ בַּ֣עַל הָֽאִשָּׁ֔ה וְנָתַ֖ן בִּפְלִלִֽים׃ וַאֲרֵי יִנְצוֹן גֻּבְרִין, וְיִמְחוֹן אִתְּתָא מְעַדְּיָא וְיִפְּקוּן וַלְדַּהָא, וְלָא יְהֵי, מוֹתָא--אִתְגְּבָאָה יִתְגְּבֵי, כְּמָא דִּישַׁוֵּי עֲלוֹהִי בַּעְלַהּ דְּאִתְּתָא, וְיִתֵּין, מִמֵּימַר דַּיָּנַיָּא. 
כא,כג וְאִם־אָס֖וֹן יִֽהְיֶ֑ה וְנָֽתַתָּ֥ה נֶ֖פֶשׁ תַּ֥חַת נָֽפֶשׁ׃ וְאִם מוֹתָא, יְהֵי--וְתִתֵּין נַפְשָׁא, חֲלָף נַפְשָׁא. 
כא,כד עַ֚יִן תַּ֣חַת עַ֔יִן שֵׁ֖ן תַּ֣חַת שֵׁ֑ן יָ֚ד תַּ֣חַת יָ֔ד רֶ֖גֶל תַּ֥חַת רָֽגֶל׃ עֵינָא חֲלָף עֵינָא, שִׁנָּא חֲלָף שִׁנָּא, יְדָא חֲלָף יְדָא, רִגְלָא חֲלָף רִגְלָא. 
כא,כה כְּוִיָּה֙ תַּ֣חַת כְּוִיָּ֔ה פֶּ֖צַע תַּ֣חַת פָּ֑צַע חַבּוּרָ֕ה תַּ֖חַת חַבּוּרָֽה׃ {ס} כְּוַאָה חֲלָף כְּוַאָה, פִּדְעָא חֲלָף פִּדְעָא, מַשְׁקוֹפִי, חֲלָף מַשְׁקוֹפִי. {ס}
כא,כו וְכִֽי־יַכֶּ֨ה אִ֜ישׁ אֶת־עֵ֥ין עַבְדּ֛וֹ אֽוֹ־אֶת־עֵ֥ין אֲמָת֖וֹ וְשִֽׁחֲתָ֑הּ לַֽחָפְשִׁ֥י יְשַׁלְּחֶ֖נּוּ תַּ֥חַת עֵינֽוֹ׃ וַאֲרֵי יִמְחֵי גְּבַר יָת עֵינָא דְּעַבְדֵּיהּ, אוֹ יָת עֵינָא דְּאַמְתֵּיהּ--וִיחַבְּלִנַּהּ:  לְבַר חוֹרִין יִפְטְרִנֵּיהּ, חֲלָף עֵינֵיהּ. 
כא,כז וְאִם־שֵׁ֥ן עַבְדּ֛וֹ אֽוֹ־שֵׁ֥ן אֲמָת֖וֹ יַפִּ֑יל לַֽחָפְשִׁ֥י יְשַׁלְּחֶ֖נּוּ תַּ֥חַת שִׁנּֽוֹ׃ {פ} וְאִם שִׁנָּא דְּעַבְדֵּיהּ אוֹ שִׁנָּא דְּאַמְתֵּיהּ, יַפִּיל--לְבַר חוֹרִין יִפְטְרִנֵּיהּ, חֲלָף שִׁנֵּיהּ. {פ}
כא,כח וְכִֽי־יִגַּ֨ח שׁ֥וֹר אֶת־אִ֛ישׁ א֥וֹ אֶת־אִשָּׁ֖ה וָמֵ֑ת סָק֨וֹל יִסָּקֵ֜ל הַשּׁ֗וֹר וְלֹ֤א יֵֽאָכֵל֙ אֶת־בְּשָׂר֔וֹ וּבַ֥עַל הַשּׁ֖וֹר נָקִֽי׃ וַאֲרֵי יִגַּח תּוֹרָא יָת גֻּבְרָא אוֹ יָת אִתְּתָא, וִימוּת--אִתְרְגָמָא יִתְרְגֵים תּוֹרָא, וְלָא יִתְאֲכִיל יָת בִּסְרֵיהּ, וּמָרֵיהּ דְּתוֹרָא, יְהֵי זָכָא. 
כא,כט וְאִ֡ם שׁוֹר֩ נַגָּ֨ח ה֜וּא מִתְּמֹ֣ל שִׁלְשֹׁ֗ם וְהוּעַ֤ד בִּבְעָלָיו֙ וְלֹ֣א יִשְׁמְרֶ֔נּוּ וְהֵמִ֥ית אִ֖ישׁ א֣וֹ אִשָּׁ֑ה הַשּׁוֹר֙ יִסָּקֵ֔ל וְגַם־בְּעָלָ֖יו יוּמָֽת׃ וְאִם תּוֹר נַגָּח הוּא מֵאֶתְמָלִי וּמִדְּקַמּוֹהִי, וְאִתַּסְהַד בְּמָרֵיהּ וְלָא נַטְרֵיהּ, וְיִקְטוֹל גְּבַר, אוֹ אִתָּא--תּוֹרָא, יִתְרְגֵים, וְאַף מָרֵיהּ, יִתְקְטִיל. 
כא,ל אִם־כֹּ֖פֶר יוּשַׁ֣ת עָלָ֑יו וְנָתַן֙ פִּדְיֹ֣ן נַפְשׁ֔וֹ כְּכֹ֥ל אֲשֶׁר־יוּשַׁ֖ת עָלָֽיו׃ אִם מָמוֹן, יְשַׁוּוֹן עֲלוֹהִי--וְיִתֵּין פֻּרְקַן נַפְשֵׁיהּ, כְּכֹל דִּישַׁוּוֹן עֲלוֹהִי. 
כא,לא אוֹ־בֵ֥ן יִגָּ֖ח אוֹ־בַ֣ת יִגָּ֑ח כַּמִּשְׁפָּ֥ט הַזֶּ֖ה יֵעָ֥שֶׂה לּֽוֹ׃ אוֹ לְבַר יִשְׂרָאֵל יִגַּח תּוֹרָא, אוֹ לְבַת יִשְׂרָאֵל יִגַּח--כְּדִינָא הָדֵין, יִתְעֲבֵיד לֵיהּ. 
כא,לב אִם־עֶ֛בֶד יִגַּ֥ח הַשּׁ֖וֹר א֣וֹ אָמָ֑ה כֶּ֣סֶף ׀ שְׁלֹשִׁ֣ים שְׁקָלִ֗ים יִתֵּן֙ לַֽאדֹנָ֔יו וְהַשּׁ֖וֹר יִסָּקֵֽל׃ {ס} אִם לְעַבְדָּא יִגַּח תּוֹרָא, אוֹ לְאַמְתָּא--כַּסְפָּא תְּלָתִין סִלְעִין, יִתֵּין לְרִבּוֹנֵיהּ, וְתוֹרָא, יִתְרְגֵים. {ס}
כא,לג וְכִֽי־יִפְתַּ֨ח אִ֜ישׁ בּ֗וֹר א֠וֹ כִּֽי־יִכְרֶ֥ה אִ֛ישׁ בֹּ֖ר וְלֹ֣א יְכַסֶּ֑נּוּ וְנָֽפַל־שָׁ֥מָּה שּׁ֖וֹר א֥וֹ חֲמֽוֹר׃ וַאֲרֵי יִפְתַּח גְּבַר גּוּב, אוֹ אֲרֵי יִכְרֵי גְּבַר גּוּב--וְלָא יְכַסֵּינֵיהּ; וְיִפּוֹל תַּמָּן תּוֹרָא, אוֹ חֲמָרָא. 
כא,לד בַּ֤עַל הַבּוֹר֙ יְשַׁלֵּ֔ם כֶּ֖סֶף יָשִׁ֣יב לִבְעָלָ֑יו וְהַמֵּ֖ת יִֽהְיֶה־לּֽוֹ׃ {ס} מָרֵיהּ דְּגֻבָּא יְשַׁלֵּים, כַּסְפָּא יָתִיב לְמָרוֹהִי; וּמִיתָא, יְהֵי דִּילֵיהּ. {ס}
כא,לה וְכִֽי־יִגֹּ֧ף שֽׁוֹר־אִ֛ישׁ אֶת־שׁ֥וֹר רֵעֵ֖הוּ וָמֵ֑ת וּמָ֨כְר֜וּ אֶת־הַשּׁ֤וֹר הַחַי֙ וְחָצ֣וּ אֶת־כַּסְפּ֔וֹ וְגַ֥ם אֶת־הַמֵּ֖ת יֶֽחֱצֽוּן׃ וַאֲרֵי יִגּוֹף תּוֹר דִּגְבַר יָת תּוֹרָא דְּחַבְרֵיהּ, וִימוּת--וִיזַבְּנוּן יָת תּוֹרָא חַיָא, וְיִפְלְגוּן יָת כַּסְפֵּיהּ, וְאַף יָת דְּמֵי מִיתָא, יִפְלְגוּן. 
כא,לו א֣וֹ נוֹדַ֗ע כִּ֠י שׁ֣וֹר נַגָּ֥ח הוּא֙ מִתְּמ֣וֹל שִׁלְשֹׁ֔ם וְלֹ֥א יִשְׁמְרֶ֖נּוּ בְּעָלָ֑יו שַׁלֵּ֨ם יְשַׁלֵּ֥ם שׁוֹר֙ תַּ֣חַת הַשּׁ֔וֹר וְהַמֵּ֖ת יִֽהְיֶה־לּֽוֹ׃ {ס} אוֹ אִתְיְדַע, אֲרֵי תּוֹר נַגָּח הוּא מֵאֶתְמָלִי וּמִדְּקַמּוֹהִי, וְלָא נַטְרֵיהּ, מָרֵיהּ--שַׁלָּמָא יְשַׁלֵּים תּוֹרָא חֲלָף תּוֹרָא, וּמִיתָא יְהֵי דִּילֵיהּ. {ס}
כא,לז כִּ֤י יִגְנֹֽב־אִישׁ֙ שׁ֣וֹר אוֹ־שֶׂ֔ה וּטְבָח֖וֹ א֣וֹ מְכָר֑וֹ חֲמִשָּׁ֣ה בָקָ֗ר יְשַׁלֵּם֙ תַּ֣חַת הַשּׁ֔וֹר וְאַרְבַּע־צֹ֖אן תַּ֥חַת הַשֶּֽׂה׃ אֲרֵי יִגְנוֹב גְּבַר תּוֹר אוֹ אִמַּר, וְיִכְּסִנֵּיהּ אוֹ יְזַבְּנִנֵּיהּ--חַמְשָׁא תּוֹרִין, יְשַׁלֵּים חֲלָף תּוֹרָא, וְאַרְבַּע עָנָא, חֲלָף אִמְּרָא. 
כב,א אִם־בַּמַּחְתֶּ֛רֶת יִמָּצֵ֥א הַגַּנָּ֖ב וְהֻכָּ֣ה וָמֵ֑ת אֵ֥ין ל֖וֹ דָּמִֽים׃ אִם בְּמַחְתַּרְתָּא יִשְׁתְּכַח גַּנָּבָא, וְיִתְמְחֵי וִימוּת--לֵית לֵיהּ, דַּם. 
כב,ב אִם־זָֽרְחָ֥ה הַשֶּׁ֛מֶשׁ עָלָ֖יו דָּמִ֣ים ל֑וֹ שַׁלֵּ֣ם יְשַׁלֵּ֔ם אִם־אֵ֣ין ל֔וֹ וְנִמְכַּ֖ר בִּגְנֵֽבָתֽוֹ׃ אִם עֵינָא דְּסָהֲדַיָּא נְפַלַת עֲלוֹהִי, דְּמָא לֵיהּ:  שַׁלָּמָא יְשַׁלֵּים--אִם לֵית לֵיהּ, וְיִזְדַּבַּן בִּגְנֻבְתֵּיהּ. 
כב,ג אִֽם־הִמָּצֵא֩ תִמָּצֵ֨א בְיָד֜וֹ הַגְּנֵבָ֗ה מִשּׁ֧וֹר עַד־חֲמ֛וֹר עַד־שֶׂ֖ה חַיִּ֑ים שְׁנַ֖יִם יְשַׁלֵּֽם׃ {ס} אִם אִשְׁתְּכָחָא תִּשְׁתְּכַח בִּידֵיהּ גְּנֻבְתָּא, מִתּוֹר עַד חֲמָר עַד אִמַּר--אִנּוּן חַיִין:  עַל חַד תְּרֵין, יְשַׁלֵּים. {ס}
כב,ד כִּ֤י יַבְעֶר־אִישׁ֙ שָׂדֶ֣ה אוֹ־כֶ֔רֶם וְשִׁלַּח֙ אֶת־בְּעִירֹ֔ה וּבִעֵ֖ר בִּשְׂדֵ֣ה אַחֵ֑ר מֵיטַ֥ב שָׂדֵ֛הוּ וּמֵיטַ֥ב כַּרְמ֖וֹ יְשַׁלֵּֽם׃ {ס} אֲרֵי יוֹכֵיל גְּבַר, חֲקַל אוֹ כְּרַם, וִישַׁלַּח יָת בְּעִירֵיהּ, וְיֵיכוֹל בַּחֲקַל אָחֳרָן--שְׁפַר חַקְלֵיהּ וּשְׁפַר כַּרְמֵיהּ, יְשַׁלֵּים. {ס}
כב,ה כִּֽי־תֵצֵ֨א אֵ֜שׁ וּמָֽצְאָ֤ה קֹצִים֙ וְנֶֽאֱכַ֣ל גָּדִ֔ישׁ א֥וֹ הַקָּמָ֖ה א֣וֹ הַשָּׂדֶ֑ה שַׁלֵּ֣ם יְשַׁלֵּ֔ם הַמַּבְעִ֖ר אֶת־הַבְּעֵרָֽה׃ {ס} אֲרֵי יִתַּפַּק נוּר וְיַשְׁכַּח כֻּבִּין, וְיֵיכוֹל גְּדִישִׁין, אוֹ קָמָא, אוֹ חֲקַל--שַׁלָּמָא יְשַׁלֵּים, דְּאַדְלֵיק יָת דְּלֵיקְתָא. {ס}
כב,ו כִּֽי־יִתֵּן֩ אִ֨ישׁ אֶל־רֵעֵ֜הוּ כֶּ֤סֶף אֽוֹ־כֵלִים֙ לִשְׁמֹ֔ר וְגֻנַּ֖ב מִבֵּ֣ית הָאִ֑ישׁ אִם־יִמָּצֵ֥א הַגַּנָּ֖ב יְשַׁלֵּ֥ם שְׁנָֽיִם׃ אֲרֵי יִתֵּין גְּבַר לְחַבְרֵיהּ כְּסַף אוֹ מָנִין, לְמִטַּר, וְיִתְגַּנְבוּן, מִבֵּית גֻּבְרָא--אִם יִשְׁתְּכַח גַּנָּבָא, יְשַׁלֵּים עַל חַד תְּרֵין. 
כב,ז אִם־לֹ֤א יִמָּצֵא֙ הַגַּנָּ֔ב וְנִקְרַ֥ב בַּֽעַל־הַבַּ֖יִת אֶל־הָֽאֱלֹהִ֑ים אִם־לֹ֥א שָׁלַ֛ח יָד֖וֹ בִּמְלֶ֥אכֶת רֵעֵֽהוּ׃ אִם לָא יִשְׁתְּכַח גַּנָּבָא, וְיִתְקָרַב מָרֵיהּ דְּבֵיתָא לִקְדָם דַּיָּנַיָּא:  אִם לָא אוֹשֵׁיט יְדֵיהּ, בְּמָא דִּמְסַר לֵיהּ חַבְרֵיהּ. 
כב,ח עַֽל־כָּל־דְּבַר־פֶּ֡שַׁע עַל־שׁ֡וֹר עַל־חֲ֠מוֹר עַל־שֶׂ֨ה עַל־שַׂלְמָ֜ה עַל־כָּל־אֲבֵדָ֗ה אֲשֶׁ֤ר יֹאמַר֙ כִּי־ה֣וּא זֶ֔ה עַ֚ד הָֽאֱלֹהִ֔ים יָבֹ֖א דְּבַר־שְׁנֵיהֶ֑ם אֲשֶׁ֤ר יַרְשִׁיעֻן֙ אֱלֹהִ֔ים יְשַׁלֵּ֥ם שְׁנַ֖יִם לְרֵעֵֽהוּ׃ {ס} עַל כָּל פִּתְגָם דְּחוֹב עַל תּוֹר עַל חֲמָר עַל אִמַּר עַל כְּסוּ עַל כָּל אֲבֵידְתָא, דְּיֵימַר אֲרֵי הוּא דֵּין--לִקְדָם דַּיָּנַיָּא, יֵיעוֹל דִּין תַּרְוֵיהוֹן:  דִּיחַיְּבוּן דַּיָּנַיָּא, יְשַׁלֵּים עַל חַד תְּרֵין לְחַבְרֵיהּ. {ס}
כב,ט כִּֽי־יִתֵּן֩ אִ֨ישׁ אֶל־רֵעֵ֜הוּ חֲמ֨וֹר אוֹ־שׁ֥וֹר אוֹ־שֶׂ֛ה וְכָל־בְּהֵמָ֖ה לִשְׁמֹ֑ר וּמֵ֛ת אֽוֹ־נִשְׁבַּ֥ר אֽוֹ־נִשְׁבָּ֖ה אֵ֥ין רֹאֶֽה׃ אֲרֵי יִתֵּין גְּבַר לְחַבְרֵיהּ חֲמָר אוֹ תּוֹר אוֹ אִמַּר, וְכָל בְּעִירָא--לְמִטַּר; וּמִית אוֹ אִתְּבַר אוֹ אִשְׁתְּבִי, לֵית דְּחָזֵי. 
כב,י שְׁבֻעַ֣ת יְהוָ֗ה תִּֽהְיֶה֙ בֵּ֣ין שְׁנֵיהֶ֔ם אִם־לֹ֥א שָׁלַ֛ח יָד֖וֹ בִּמְלֶ֣אכֶת רֵעֵ֑הוּ וְלָקַ֥ח בְּעָלָ֖יו וְלֹ֥א יְשַׁלֵּֽם׃ מוֹמָתָא דַּייָ, תְּהֵי בֵּין תַּרְוֵיהוֹן--אִם לָא אוֹשֵׁיט יְדֵיהּ, בְּמָא דִּמְסַר לֵיהּ חַבְרֵיהּ; וִיקַבֵּיל מָרֵיהּ מִנֵּיהּ מוֹמָתָא, וְלָא יְשַׁלֵּים. 
כב,יא וְאִם־גָּנֹ֥ב יִגָּנֵ֖ב מֵֽעִמּ֑וֹ יְשַׁלֵּ֖ם לִבְעָלָֽיו׃ וְאִם אִתְגְּנָבָא יִתְגְּנֵיב, מֵעִמֵּיהּ--יְשַׁלֵּים, לְמָרוֹהִי. 
כב,יב אִם־טָרֹ֥ף יִטָּרֵ֖ף יְבִאֵ֣הוּ עֵ֑ד הַטְּרֵפָ֖ה לֹ֥א יְשַׁלֵּֽם׃ {פ} אִם אִתְּבָרָא יִתְּבַר, יַיְתֵי סָהֲדִין:  דִּתְבִיר, לָא יְשַׁלֵּים. {פ}
כב,יג וְכִֽי־יִשְׁאַ֥ל אִ֛ישׁ מֵעִ֥ם רֵעֵ֖הוּ וְנִשְׁבַּ֣ר אוֹ־מֵ֑ת בְּעָלָ֥יו אֵין־עִמּ֖וֹ שַׁלֵּ֥ם יְשַׁלֵּֽם׃ וַאֲרֵי יִשְׁאַל גְּבַר מִן חַבְרֵיהּ, וְיִתְּבַר אוֹ יְמוּת; מָרֵיהּ לֵית עִמֵּיהּ, שַׁלָּמָא יְשַׁלֵּים. 
כב,יד אִם־בְּעָלָ֥יו עִמּ֖וֹ לֹ֣א יְשַׁלֵּ֑ם אִם־שָׂכִ֣יר ה֔וּא בָּ֖א בִּשְׂכָרֽוֹ׃ {ס} אִם מָרֵיהּ עִמֵּיהּ, לָא יְשַׁלֵּים:  אִם אֲגִירָא הוּא, עָאל בְּאַגְרֵיהּ. {ס}
כב,טו וְכִֽי־יְפַתֶּ֣ה אִ֗ישׁ בְּתוּלָ֛ה אֲשֶׁ֥ר לֹֽא־אֹרָ֖שָׂה וְשָׁכַ֣ב עִמָּ֑הּ מָהֹ֛ר יִמְהָרֶ֥נָּה לּ֖וֹ לְאִשָּֽׁה׃ וַאֲרֵי יְשַׁדֵּיל גְּבַר, בְּתוּלְתָא דְּלָא מְאָרְסָא--וְיִשְׁכּוֹב עִמַּהּ:  קַיָּמָא יְקַיְּמִנַּהּ לֵיהּ, לְאִתּוּ. 
כב,טז אִם־מָאֵ֧ן יְמָאֵ֛ן אָבִ֖יהָ לְתִתָּ֣הּ ל֑וֹ כֶּ֣סֶף יִשְׁקֹ֔ל כְּמֹ֖הַר הַבְּתוּלֹֽת׃ {ס} אִם מִצְבָּא לָא יִצְבֵּי אֲבוּהָא, לְמִתְּנַהּ לֵיהּ--כַּסְפָּא יִתְקוּל, כְּמֻהְרֵי בְּתוּלָתָא. {ס}
כב,יז מְכַשֵּׁפָ֖ה לֹ֥א תְחַיֶּֽה׃ חָרָשָׁא, לָא תַּחֵי. 
כב,יח כָּל־שֹׁכֵ֥ב עִם־בְּהֵמָ֖ה מ֥וֹת יוּמָֽת׃ {ס} כָּל דְּיִשְׁכּוֹב עִם בְּעִירָא, אִתְקְטָלָא יִתְקְטִיל. {ס}
כב,יט זֹבֵ֥חַ לָֽאֱלֹהִ֖ים יָֽחֳרָ֑ם בִּלְתִּ֥י לַֽיהוָ֖ה לְבַדּֽוֹ׃ דִּידַבַּח לְטָעֲוָת עַמְמַיָּא, יִתְקְטִיל--אֱלָהֵין לִשְׁמָא דַּייָ, בִּלְחוֹדוֹהִי. 
כב,כ וְגֵ֥ר לֹֽא־תוֹנֶ֖ה וְלֹ֣א תִלְחָצֶ֑נּוּ כִּֽי־גֵרִ֥ים הֱיִיתֶ֖ם בְּאֶ֥רֶץ מִצְרָֽיִם׃ וּלְגִיּוֹרָא לָא תּוֹנוֹן, וְלָא תָּעִיקוּן:  אֲרֵי דַּיָּרִין הֲוֵיתוֹן, בְּאַרְעָא דְּמִצְרָיִם. 
כב,כא כָּל־אַלְמָנָ֥ה וְיָת֖וֹם לֹ֥א תְעַנּֽוּן׃ כָּל אַרְמְלָא וְיִיתַם, לָא תְּעַנּוֹן. 
כב,כב אִם־עַנֵּ֥ה תְעַנֶּ֖ה אֹת֑וֹ כִּ֣י אִם־צָעֹ֤ק יִצְעַק֙ אֵלַ֔י שָׁמֹ֥עַ אֶשְׁמַ֖ע צַֽעֲקָתֽוֹ׃ אִם עַנָּאָה תְּעַנֵּי, יָתֵיהּ--אֲרֵי אִם מִקְבָּל יִקְבַּל קֳדָמַי, קַבָּלָא אֲקַבֵּיל קְבִילְתֵיהּ. 
כב,כג וְחָרָ֣ה אַפִּ֔י וְהָֽרַגְתִּ֥י אֶתְכֶ֖ם בֶּחָ֑רֶב וְהָי֤וּ נְשֵׁיכֶם֙ אַלְמָנ֔וֹת וּבְנֵיכֶ֖ם יְתֹמִֽים׃ {פ} וְיִתְקַף רֻגְזִי, וְאֶקְטוֹל יָתְכוֹן בְּחַרְבָּא; וְיִהְוְיָן נְשֵׁיכוֹן אַרְמְלָן, וּבְנֵיכוֹן יַתְמִין. {פ}
כב,כד אִם־כֶּ֣סֶף ׀ תַּלְוֶ֣ה אֶת־עַמִּ֗י אֶת־הֶֽעָנִי֙ עִמָּ֔ךְ לֹֽא־תִהְיֶ֥ה ל֖וֹ כְּנֹשֶׁ֑ה לֹֽא־תְשִׂימ֥וּן עָלָ֖יו נֶֽשֶׁךְ׃ אִם כַּסְפָּא תּוֹזֵיף בְּעַמִּי, לְעַנְיָא דְּעִמָּךְ--לָא תְּהֵי לֵיהּ, כְּרָשְׁיָא; לָא תְּשַׁוּוֹן עֲלוֹהִי, חִיבֻלְיָא. 
כב,כה אִם־חָבֹ֥ל תַּחְבֹּ֖ל שַׂלְמַ֣ת רֵעֶ֑ךָ עַד־בֹּ֥א הַשֶּׁ֖מֶשׁ תְּשִׁיבֶ֥נּוּ לֽוֹ׃ אִם מִשְׁכּוֹנָא תִּסַּב, כְּסוּתָא דְּחַבְרָךְ--עַד מֵיעַל שִׁמְשָׁא, תָּתִיבִנֵּיהּ לֵיהּ. 
כב,כו כִּ֣י הִ֤וא כְסוּתֹה֙ לְבַדָּ֔הּ הִ֥וא שִׂמְלָת֖וֹ לְעֹר֑וֹ בַּמֶּ֣ה יִשְׁכָּ֔ב וְהָיָה֙ כִּֽי־יִצְעַ֣ק אֵלַ֔י וְשָֽׁמַעְתִּ֖י כִּֽי־חַנּ֥וּן אָֽנִי׃ {ס} אֲרֵי הִיא כְּסוּתֵיהּ בִּלְחוֹדַהּ, הִיא תֻּתְבֵּיהּ לְמַשְׁכֵּיהּ; בְּמָא יִשְׁכּוֹב--וִיהֵי אֲרֵי יִקְבַּל קֳדָמַי, וַאֲקַבֵּיל קְבִילְתֵיהּ אֲרֵי חַנָּנָא אֲנָא. {ס}
כב,כז אֱלֹהִ֖ים לֹ֣א תְקַלֵּ֑ל וְנָשִׂ֥יא בְעַמְּךָ֖ לֹ֥א תָאֹֽר׃ דַּיָּנָא, לָא תַּקִיל; וְרַבָּא בְּעַמָּךְ, לָא תְּלוּט. 
כב,כח מְלֵאָֽתְךָ֥ וְדִמְעֲךָ֖ לֹ֣א תְאַחֵ֑ר בְּכ֥וֹר בָּנֶ֖יךָ תִּתֶּן־לִֽי׃ בִּכּוּרָךְ וְדִמְעָךְ, לָא תְּאַחַר; בֻּכְרָא דִּבְנָךְ, תַּפְרֵישׁ קֳדָמָי. 
כב,כט כֵּֽן־תַּעֲשֶׂ֥ה לְשֹֽׁרְךָ֖ לְצֹאנֶ֑ךָ שִׁבְעַ֤ת יָמִים֙ יִֽהְיֶ֣ה עִם־אִמּ֔וֹ בַּיּ֥וֹם הַשְּׁמִינִ֖י תִּתְּנוֹ־לִֽי׃ כֵּין תַּעֲבֵיד לְתוֹרָךְ, לְעָנָךְ:  שִׁבְעָא יוֹמִין יְהֵי עִם אִמֵּיהּ, בְּיוֹמָא תְּמִינָאָה תַּפְרְשִׁנֵּיהּ קֳדָמָי. 
כב,ל וְאַנְשֵׁי־קֹ֖דֶשׁ תִּֽהְי֣וּן לִ֑י וּבָשָׂ֨ר בַּשָּׂדֶ֤ה טְרֵפָה֙ לֹ֣א תֹאכֵ֔לוּ לַכֶּ֖לֶב תַּשְׁלִכ֥וּן אֹתֽוֹ׃ {ס} וַאֲנָשִׁין קַדִּישִׁין, תְּהוֹן קֳדָמָי; וּבְסַר תְּלִישׁ מִן חֵיוָא חַיָא לָא תֵּיכְלוּן, לְכַלְבָּא תִּרְמוֹן יָתֵיהּ. {ס}
כג,א לֹ֥א תִשָּׂ֖א שֵׁ֣מַע שָׁ֑וְא אַל־תָּ֤שֶׁת יָֽדְךָ֙ עִם־רָשָׁ֔ע לִֽהְיֹ֖ת עֵ֥ד חָמָֽס׃ לָא תְּקַבֵּיל, שֵׁימַע דִּשְׁקַר; לָא תְּשַׁוֵּי יְדָךְ עִם חַיָּבָא, לְמִהְוֵי לֵיהּ סָהִיד שַׁקָּר. 
כג,ב לֹֽא־תִהְיֶ֥ה אַחֲרֵֽי־רַבִּ֖ים לְרָעֹ֑ת וְלֹֽא־תַעֲנֶ֣ה עַל־רִ֗ב לִנְטֹ֛ת אַֽחֲרֵ֥י רַבִּ֖ים לְהַטֹּֽת׃ לָא תְּהֵי בָּתַר סַגִּיאֵי, לְאַבְאָשָׁא; וְלָא תִּתְמְנַע מִלְּאַלָּפָא מָא דִּבְעֵינָךְ עַל דִּינָא, בָּתַר סַגִּיאֵי--שַׁלֵּים דִּינָא. 
כג,ג וְדָ֕ל לֹ֥א תֶהְדַּ֖ר בְּרִיבֽוֹ׃ {ס} וְעַל מִסְכֵּינָא, לָא תְּרַחֵים בְּדִינֵיהּ. {ס}
כג,ד כִּ֣י תִפְגַּ֞ע שׁ֧וֹר אֹֽיִבְךָ֛ א֥וֹ חֲמֹר֖וֹ תֹּעֶ֑ה הָשֵׁ֥ב תְּשִׁיבֶ֖נּוּ לֽוֹ׃ {ס} אֲרֵי תְּעָרַע תּוֹרָא דְּסָנְאָךְ, אוֹ חֲמָרֵיהּ--דְּטָעֵי:  אָתָבָא תָּתִיבִנֵּיהּ, לֵיהּ. {ס}
כג,ה כִּֽי־תִרְאֶ֞ה חֲמ֣וֹר שֹׂנַֽאֲךָ֗ רֹבֵץ֙ תַּ֣חַת מַשָּׂא֔וֹ וְחָֽדַלְתָּ֖ מֵֽעֲזֹ֣ב ל֑וֹ עָזֹ֥ב תַּֽעֲזֹ֖ב עִמּֽוֹ׃ {ס} אֲרֵי תִּחְזֵי חֲמָרָא דְּסָנְאָךְ, רְבִיעַ תְּחוֹת טֻעְנֵיהּ, וְתִתְמְנַע, מִלְּמִשְׁקַל לֵיהּ--מִשְׁבָּק תִּשְׁבּוֹק מָא דִּבְלִבָּךְ עֲלוֹהִי, וּתְפָרֵיק עִמֵּיהּ. {ס}
כג,ו לֹ֥א תַטֶּ֛ה מִשְׁפַּ֥ט אֶבְיֹֽנְךָ֖ בְּרִיבֽוֹ׃ לָא תַּצְלֵי דִּין מִסְכֵּינָךְ, בְּדִינֵיהּ. 
כג,ז מִדְּבַר־שֶׁ֖קֶר תִּרְחָ֑ק וְנָקִ֤י וְצַדִּיק֙ אַֽל־תַּהֲרֹ֔ג כִּ֥י לֹֽא־אַצְדִּ֖יק רָשָֽׁע׃ מִפִּתְגָמָא דְּשִׁקְרָא, הֱוִי רַחִיק; וְדִזְכֵּי וְדִנְפַק דְּכֵי מִן דִּינָא לָא תִּקְטוּל, אֲרֵי לָא אֲזַכֵּי חַיָּבָא. 
כג,ח וְשֹׁ֖חַד לֹ֣א תִקָּ֑ח כִּ֤י הַשֹּׁ֨חַד֙ יְעַוֵּ֣ר פִּקְחִ֔ים וִֽיסַלֵּ֖ף דִּבְרֵ֥י צַדִּיקִֽים׃ וְשֻׁחְדָּא, לָא תְּקַבֵּיל:  אֲרֵי שֻׁחְדָּא מְעַוַּר עֵינֵי חַכִּימִין, וּמְקַלְקֵיל פִּתְגָמִין תְּרִיצִין. 
כג,ט וְגֵ֖ר לֹ֣א תִלְחָ֑ץ וְאַתֶּ֗ם יְדַעְתֶּם֙ אֶת־נֶ֣פֶשׁ הַגֵּ֔ר כִּֽי־גֵרִ֥ים הֱיִיתֶ֖ם בְּאֶ֥רֶץ מִצְרָֽיִם׃ וּלְגִיּוֹרָא, לָא תָּעִיקוּן; וְאַתּוּן, יְדַעְתּוּן יָת נַפְשָׁא דְּגִיּוֹרָא--אֲרֵי דַּיָּרִין הֲוֵיתוֹן, בְּאַרְעָא דְּמִצְרָיִם. 
כג,י וְשֵׁ֥שׁ שָׁנִ֖ים תִּזְרַ֣ע אֶת־אַרְצֶ֑ךָ וְאָֽסַפְתָּ֖ אֶת־תְּבֽוּאָתָֽהּ׃ וְשֵׁית שְׁנִין, תִּזְרַע יָת אַרְעָךְ; וְתִכְנוֹשׁ, יָת עֲלַלְתַּהּ. 
כג,יא וְהַשְּׁבִיעִ֞ת תִּשְׁמְטֶ֣נָּה וּנְטַשְׁתָּ֗הּ וְאָֽכְלוּ֙ אֶבְיֹנֵ֣י עַמֶּ֔ךָ וְיִתְרָ֕ם תֹּאכַ֖ל חַיַּ֣ת הַשָּׂדֶ֑ה כֵּֽן־תַּעֲשֶׂ֥ה לְכַרְמְךָ֖ לְזֵיתֶֽךָ׃ וּשְׁבִיעֵיתָא תִּשְׁמְטִנַּהּ וְתִרְטְשִׁנַּהּ, וְיֵיכְלוּן מִסְכֵּינֵי עַמָּךְ, וּשְׁאָרְהוֹן, תֵּיכוֹל חַיַּת בָּרָא; כֵּין תַּעֲבֵיד לְכַרְמָךְ, לְזֵיתָךְ. 
כג,יב שֵׁ֤שֶׁת יָמִים֙ תַּֽעֲשֶׂ֣ה מַֽעֲשֶׂ֔יךָ וּבַיּ֥וֹם הַשְּׁבִיעִ֖י תִּשְׁבֹּ֑ת לְמַ֣עַן יָנ֗וּחַ שֽׁוֹרְךָ֙ וַֽחֲמֹרֶ֔ךָ וְיִנָּפֵ֥שׁ בֶּן־אֲמָֽתְךָ֖ וְהַגֵּֽר׃ שִׁתָּא יוֹמִין תַּעֲבֵיד עוּבָדָךְ, וּבְיוֹמָא שְׁבִיעָאָה תְּנוּחַ--בְּדִיל דִּינוּחַ, תּוֹרָךְ וּחְמָרָךְ, וְיִשְׁקוֹט בַּר אַמְתָּךְ, וְגִיּוֹרָא. 
כג,יג וּבְכֹ֛ל אֲשֶׁר־אָמַ֥רְתִּי אֲלֵיכֶ֖ם תִּשָּׁמֵ֑רוּ וְשֵׁ֨ם אֱלֹהִ֤ים אֲחֵרִים֙ לֹ֣א תַזְכִּ֔ירוּ לֹ֥א יִשָּׁמַ֖ע עַל־פִּֽיךָ׃ וּבְכֹל דַּאֲמַרִית לְכוֹן, תִּסְתַּמְרוּן; וְשׁוֹם טָעֲוָת עַמְמַיָּא לָא תִּדְכְרוּן, לָא יִשְׁתְּמַע עַל פֻּמְּכוֹן. 
כג,יד שָׁלֹ֣שׁ רְגָלִ֔ים תָּחֹ֥ג לִ֖י בַּשָּׁנָֽה׃ תְּלָת זִמְנִין, תֵּיחֲגוּן קֳדָמַי בְּשַׁתָּא. 
כג,טו אֶת־חַ֣ג הַמַּצּוֹת֮ תִּשְׁמֹר֒ שִׁבְעַ֣ת יָמִים֩ תֹּאכַ֨ל מַצּ֜וֹת כַּֽאֲשֶׁ֣ר צִוִּיתִ֗ךָ לְמוֹעֵד֙ חֹ֣דֶשׁ הָֽאָבִ֔יב כִּי־ב֖וֹ יָצָ֣אתָ מִמִּצְרָ֑יִם וְלֹֽא־יֵרָא֥וּ פָנַ֖י רֵיקָֽם׃ יָת חַגָּא דְּפַטִּירַיָּא, תִּטַּר--שִׁבְעָא יוֹמִין תֵּיכוֹל פַּטִּירָא כְּמָא דְּפַקֵּידְתָּךְ לִזְמַן יַרְחָא דַּאֲבִיבָא, אֲרֵי בֵּיהּ נְפַקְתָּא מִמִּצְרָיִם; וְלָא יִתַּחְזוֹן קֳדָמַי, רֵיקָנִין. 
כג,טז וְחַ֤ג הַקָּצִיר֙ בִּכּוּרֵ֣י מַֽעֲשֶׂ֔יךָ אֲשֶׁ֥ר תִּזְרַ֖ע בַּשָּׂדֶ֑ה וְחַ֤ג הָֽאָסִף֙ בְּצֵ֣את הַשָּׁנָ֔ה בְּאָסְפְּךָ֥ אֶֽת־מַעֲשֶׂ֖יךָ מִן־הַשָּׂדֶֽה׃ וְחַגָּא דִּחְצָדָא בִּכּוּרֵי עוּבָדָךְ, דְּתִזְרַע בְּחַקְלָא; וְחַגָּא דִּכְנָשָׁא בְּמִפְּקַהּ דְּשַׁתָּא, בְּמִכְנְשָׁךְ יָת עוּבָדָךְ מִן חַקְלָא. 
כג,יז שָׁלֹ֥שׁ פְּעָמִ֖ים בַּשָּׁנָ֑ה יֵֽרָאֶה֙ כָּל־זְכ֣וּרְךָ֔ אֶל־פְּנֵ֖י הָֽאָדֹ֥ן ׀ יְהוָֽה׃ תְּלָת זִמְנִין, בְּשַׁתָּא--יִתַּחְזוֹן, כָּל דְּכוּרָךְ, קֳדָם, רִבּוֹן עָלְמָא יְיָ. 
כג,יח לֹֽא־תִזְבַּ֥ח עַל־חָמֵ֖ץ דַּם־זִבְחִ֑י וְלֹֽא־יָלִ֥ין חֵֽלֶב־חַגִּ֖י עַד־בֹּֽקֶר׃ לָא תִּכּוֹס עַל חֲמִיעַ, דַּם פִּסְחִי; וְלָא יְבִיתוּן בָּר מִמַּדְבְּחָא תַּרְבֵּי נִכְסַת חַגָּא, עַד צַפְרָא. 
כג,יט רֵאשִׁ֗ית בִּכּוּרֵי֙ אַדְמָ֣תְךָ֔ תָּבִ֕יא בֵּ֖ית יְהוָ֣ה אֱלֹהֶ֑יךָ לֹֽא־תְבַשֵּׁ֥ל גְּדִ֖י בַּֽחֲלֵ֥ב אִמּֽוֹ׃ {פ} רֵישׁ, בִּכּוּרֵי אַרְעָךְ, תַּיְתֵי, לְבֵית מַקְדְּשָׁא דַּייָ אֱלָהָךְ; לָא תֵּיכְלוּן בְּסַר, בַּחֲלַב. {פ}
כג,כ הִנֵּ֨ה אָֽנֹכִ֜י שֹׁלֵ֤חַ מַלְאָךְ֙ לְפָנֶ֔יךָ לִשְׁמָרְךָ֖ בַּדָּ֑רֶךְ וְלַהֲבִ֣יאֲךָ֔ אֶל־הַמָּק֖וֹם אֲשֶׁ֥ר הֲכִנֹֽתִי׃ הָא אֲנָא שָׁלַח מַלְאֲכָא, קֳדָמָךְ, לְמִטְּרָךְ, בְּאוֹרְחָא; וּלְאַעָלוּתָךְ, לְאַתְרָא דְּאַתְקֵינִית. 
כג,כא הִשָּׁ֧מֶר מִפָּנָ֛יו וּשְׁמַ֥ע בְּקֹל֖וֹ אַל־תַּמֵּ֣ר בּ֑וֹ כִּ֣י לֹ֤א יִשָּׂא֙ לְפִשְׁעֲכֶ֔ם כִּ֥י שְׁמִ֖י בְּקִרְבּֽוֹ׃ אִסְתְּמַר מִן קֳדָמוֹהִי וְקַבֵּיל לְמֵימְרֵיהּ, לָא תְּסָרֵיב לְקִבְלֵיהּ:  אֲרֵי לָא יִשְׁבּוֹק לְחוֹבֵיכוֹן, אֲרֵי בִּשְׁמִי מֵימְרֵיהּ. 
כג,כב כִּ֣י אִם־שָׁמ֤וֹעַ תִּשְׁמַע֙ בְּקֹל֔וֹ וְעָשִׂ֕יתָ כֹּ֖ל אֲשֶׁ֣ר אֲדַבֵּ֑ר וְאָֽיַבְתִּי֙ אֶת־אֹ֣יְבֶ֔יךָ וְצַרְתִּ֖י אֶת־צֹֽרְרֶֽיךָ׃ אֲרֵי אִם קַבָּלָא תְּקַבֵּיל, לְמֵימְרֵיהּ, וְתַעֲבֵיד, כֹּל דַּאֲמַלֵּיל--וְאֶסְנֵי, יָת סָנְאָךְ, וְאַעֵיק, לְדִמְעִיקִין לָךְ. 
כג,כג כִּֽי־יֵלֵ֣ךְ מַלְאָכִי֮ לְפָנֶיךָ֒ וֶהֱבִֽיאֲךָ֗ אֶל־הָֽאֱמֹרִי֙ וְהַ֣חִתִּ֔י וְהַפְּרִזִּי֙ וְהַֽכְּנַעֲנִ֔י הַֽחִוִּ֖י וְהַיְבוּסִ֑י וְהִכְחַדְתִּֽיו׃ אֲרֵי יְהָךְ מַלְאֲכִי, קֳדָמָךְ, וְיַעֵילִנָּךְ לְוָת אֱמוֹרָאֵי וְחִתָּאֵי, וּפְרִזָּאֵי וּכְנַעֲנָאֵי חִוָּאֵי וִיבוּסָאֵי; וַאֲשֵׁיצֵינוּן. 
כג,כד לֹֽא־תִשְׁתַּחֲוֶ֤ה לֵאלֹֽהֵיהֶם֙ וְלֹ֣א תָֽעָבְדֵ֔ם וְלֹ֥א תַֽעֲשֶׂ֖ה כְּמַֽעֲשֵׂיהֶ֑ם כִּ֤י הָרֵס֙ תְּהָ֣רְסֵ֔ם וְשַׁבֵּ֥ר תְּשַׁבֵּ֖ר מַצֵּבֹֽתֵיהֶֽם׃ לָא תִּסְגּוֹד לְטָעֲוָתְהוֹן וְלָא תִּפְלְחִנִּין, וְלָא תַּעֲבֵיד כְּעוּבָדֵיהוֹן:  אֲרֵי פַּגָּרָא תְּפַגְּרִנּוּן, וְתַבָּרָא תְּתַבַּר קָמָתְהוֹן. 
כג,כה וַֽעֲבַדְתֶּ֗ם אֵ֚ת יְהוָ֣ה אֱלֹֽהֵיכֶ֔ם וּבֵרַ֥ךְ אֶֽת־לַחְמְךָ֖ וְאֶת־מֵימֶ֑יךָ וַהֲסִֽרֹתִ֥י מַֽחֲלָ֖ה מִקִּרְבֶּֽךָ׃ {ס} וְתִפְלְחוּן, קֳדָם יְיָ אֱלָהֲכוֹן, וִיבָרֵיךְ יָת מֵיכְלָךְ, וְיָת מִשְׁתְּיָךְ; וְאַעְדֵּי מַרְעִין בִּישִׁין, מִבֵּינָךְ. {ס}
כג,כו לֹ֥א תִֽהְיֶ֛ה מְשַׁכֵּלָ֥ה וַֽעֲקָרָ֖ה בְּאַרְצֶ֑ךָ אֶת־מִסְפַּ֥ר יָמֶ֖יךָ אֲמַלֵּֽא׃ לָא תְּהֵי תָּכְלָא וְעַקְרָא, בְּאַרְעָךְ; יָת מִנְיַן יוֹמָךְ, אַשְׁלֵים. 
כג,כז אֶת־אֵֽימָתִי֙ אֲשַׁלַּ֣ח לְפָנֶ֔יךָ וְהַמֹּתִי֙ אֶת־כָּל־הָעָ֔ם אֲשֶׁ֥ר תָּבֹ֖א בָּהֶ֑ם וְנָֽתַתִּ֧י אֶת־כָּל־אֹֽיְבֶ֛יךָ אֵלֶ֖יךָ עֹֽרֶף׃ יָת אֵימְתִי, אֲשַׁלַּח קֳדָמָךְ, וַאֲשַׁגֵּישׁ יָת כָּל עַמָּא, דְּאַתְּ אָתֵי לְאָגָחָא קְרָבָא בְּהוֹן; וְאֶמְסַר יָת כָּל בַּעֲלֵי דְּבָבָךְ קֳדָמָךְ, מַחְזְרֵי קְדָל. 
כג,כח וְשָֽׁלַחְתִּ֥י אֶת־הַצִּרְעָ֖ה לְפָנֶ֑יךָ וְגֵֽרְשָׁ֗ה אֶת־הַֽחִוִּ֧י אֶת־הַֽכְּנַעֲנִ֛י וְאֶת־הַֽחִתִּ֖י מִלְּפָנֶֽיךָ׃ וְאֶשְׁלַח יָת עָרָעִיתָא, קֳדָמָךְ; וּתְתָרֵיךְ, יָת חִוָּאֵי יָת כְּנַעֲנָאֵי וְיָת חִתָּאֵי--מִן קֳדָמָךְ. 
כג,כט לֹ֧א אֲגָֽרְשֶׁ֛נּוּ מִפָּנֶ֖יךָ בְּשָׁנָ֣ה אֶחָ֑ת פֶּן־תִּֽהְיֶ֤ה הָאָ֨רֶץ֙ שְׁמָמָ֔ה וְרַבָּ֥ה עָלֶ֖יךָ חַיַּ֥ת הַשָּׂדֶֽה׃ לָא אֲתָרֵיכִנּוּן מִן קֳדָמָךְ, בְּשַׁתָּא חֲדָא:  דִּלְמָא תְּהֵי אַרְעָא צָדְיָא, וְתִסְגֵּי עֲלָךְ חַיַּת בָּרָא. 
כג,ל מְעַ֥ט מְעַ֛ט אֲגָֽרְשֶׁ֖נּוּ מִפָּנֶ֑יךָ עַ֚ד אֲשֶׁ֣ר תִּפְרֶ֔ה וְנָֽחַלְתָּ֖ אֶת־הָאָֽרֶץ׃ זְעֵיר זְעֵיר אֲתָרֵיכִנּוּן, מִן קֳדָמָךְ, עַד דְּתִסְגֵּי, וְתַחְסֵין יָת אַרְעָא. 
כג,לא וְשַׁתִּ֣י אֶת־גְּבֻֽלְךָ֗ מִיַּם־סוּף֙ וְעַד־יָ֣ם פְּלִשְׁתִּ֔ים וּמִמִּדְבָּ֖ר עַד־הַנָּהָ֑ר כִּ֣י ׀ אֶתֵּ֣ן בְּיֶדְכֶ֗ם אֵ֚ת יֹֽשְׁבֵ֣י הָאָ֔רֶץ וְגֵֽרַשְׁתָּ֖מוֹ מִפָּנֶֽיךָ׃ וַאֲשַׁוֵּי יָת תְּחוּמָךְ, מִיַּמָּא דְּסוּף וְעַד יַמָּא דִּפְלִשְׁתָּאֵי, וּמִמַּדְבְּרָא, עַד פְּרָת:  אֲרֵי אֶמְסַר בְּיַדְכוֹן, יָת יָתְבֵי אַרְעָא, וּתְתָרֵיכִנּוּן, מִן קֳדָמָךְ. 
כג,לב לֹֽא־תִכְרֹ֥ת לָהֶ֛ם וְלֵאלֹֽהֵיהֶ֖ם בְּרִֽית׃ לָא תִּגְזַר לְהוֹן וּלְטָעֲוָתְהוֹן, קְיָם. 
כג,לג לֹ֤א יֵֽשְׁבוּ֙ בְּאַרְצְךָ֔ פֶּן־יַֽחֲטִ֥יאוּ אֹֽתְךָ֖ לִ֑י כִּ֤י תַֽעֲבֹד֙ אֶת־אֱלֹ֣הֵיהֶ֔ם כִּֽי־יִהְיֶ֥ה לְךָ֖ לְמוֹקֵֽשׁ׃ {פ} לָא יִתְּבוּן בְּאַרְעָךְ, דִּלְמָא יְחַיְּבוּן יָתָךְ קֳדָמָי:  אֲרֵי תִּפְלַח יָת טָעֲוָתְהוֹן, אֲרֵי יְהוֹן לָךְ לְתַקְלָא. {פ}
כד,א וְאֶל־מֹשֶׁ֨ה אָמַ֜ר עֲלֵ֣ה אֶל־יְהוָ֗ה אַתָּה֙ וְאַֽהֲרֹן֙ נָדָ֣ב וַֽאֲבִיה֔וּא וְשִׁבְעִ֖ים מִזִּקְנֵ֣י יִשְׂרָאֵ֑ל וְהִשְׁתַּֽחֲוִיתֶ֖ם מֵֽרָחֹֽק׃ וּלְמֹשֶׁה אֲמַר סַק לִקְדָם יְיָ, אַתְּ וְאַהֲרוֹן נָדָב וַאֲבִיהוּא, וְשִׁבְעִין, מִסָּבֵי יִשְׂרָאֵל; וְתִסְגְּדוּן, מֵרַחִיק. 
כד,ב וְנִגַּ֨שׁ מֹשֶׁ֤ה לְבַדּוֹ֙ אֶל־יְהוָ֔ה וְהֵ֖ם לֹ֣א יִגָּ֑שׁוּ וְהָעָ֕ם לֹ֥א יַֽעֲל֖וּ עִמּֽוֹ׃ וְיִתְקָרַב מֹשֶׁה בִּלְחוֹדוֹהִי לִקְדָם יְיָ, וְאִנּוּן לָא יִתְקָרְבוּן; וְעַמָּא, לָא יִסְּקוּן עִמֵּיהּ. 
כד,ג וַיָּבֹ֣א מֹשֶׁ֗ה וַיְסַפֵּ֤ר לָעָם֙ אֵ֚ת כָּל־דִּבְרֵ֣י יְהוָ֔ה וְאֵ֖ת כָּל־הַמִּשְׁפָּטִ֑ים וַיַּ֨עַן כָּל־הָעָ֜ם ק֤וֹל אֶחָד֙ וַיֹּ֣אמְר֔וּ כָּל־הַדְּבָרִ֛ים אֲשֶׁר־דִּבֶּ֥ר יְהוָ֖ה נַֽעֲשֶֽׂה׃ וַאֲתָא מֹשֶׁה, וְאִשְׁתַּעִי לְעַמָּא יָת כָּל פִּתְגָמַיָּא דַּייָ, וְיָת, כָּל דִּינַיָּא; וַאֲתֵיב כָּל עַמָּא קָלָא חַד, וַאֲמַרוּ, כָּל פִּתְגָמַיָּא דְּמַלֵּיל יְיָ, נַעֲבֵיד. 
כד,ד וַיִּכְתֹּ֣ב מֹשֶׁ֗ה אֵ֚ת כָּל־דִּבְרֵ֣י יְהוָ֔ה וַיַּשְׁכֵּ֣ם בַּבֹּ֔קֶר וַיִּ֥בֶן מִזְבֵּ֖חַ תַּ֣חַת הָהָ֑ר וּשְׁתֵּ֤ים עֶשְׂרֵה֙ מַצֵּבָ֔ה לִשְׁנֵ֥ים עָשָׂ֖ר שִׁבְטֵ֥י יִשְׂרָאֵֽל׃ וּכְתַב מֹשֶׁה, יָת כָּל פִּתְגָמַיָּא דַּייָ, וְאַקְדֵּים בְּצַפְרָא, וּבְנָא מַדְבְּחָא בְּשִׁפּוֹלֵי טוּרָא; וְתַרְתַּא עַסְרֵי קָמָא, לִתְרֵי עֲסַר שִׁבְטַיָּא דְּיִשְׂרָאֵל. 
כד,ה וַיִּשְׁלַ֗ח אֶֽת־נַעֲרֵי֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וַֽיַּעֲל֖וּ עֹלֹ֑ת וַֽיִּזְבְּח֞וּ זְבָחִ֧ים שְׁלָמִ֛ים לַֽיהוָ֖ה פָּרִֽים׃ וּשְׁלַח, יָת בְּכוֹרֵי בְּנֵי יִשְׂרָאֵל, וְאַסִּיקוּ, עֲלָוָן; וְנַכִּיסוּ נִכְסַת קֻדְשִׁין, קֳדָם יְיָ--תּוֹרִין. 
כד,ו וַיִּקַּ֤ח מֹשֶׁה֙ חֲצִ֣י הַדָּ֔ם וַיָּ֖שֶׂם בָּֽאַגָּנֹ֑ת וַֽחֲצִ֣י הַדָּ֔ם זָרַ֖ק עַל־הַמִּזְבֵּֽחַ׃ וּנְסֵיב מֹשֶׁה פַּלְגוּת דְּמָא, וְשַׁוִּי בְּמִזְרְקַיָּא; וּפַלְגוּת דְּמָא, זְרַק עַל מַדְבְּחָא. 
כד,ז וַיִּקַּח֙ סֵ֣פֶר הַבְּרִ֔ית וַיִּקְרָ֖א בְּאָזְנֵ֣י הָעָ֑ם וַיֹּ֣אמְר֔וּ כֹּ֛ל אֲשֶׁר־דִּבֶּ֥ר יְהוָ֖ה נַֽעֲשֶׂ֥ה וְנִשְׁמָֽע׃ וּנְסֵיב סִפְרָא דִּקְיָמָא, וּקְרָא קֳדָם עַמָּא; וַאֲמַרוּ, כֹּל דְּמַלֵּיל יְיָ נַעֲבֵיד וּנְקַבֵּיל. 
כד,ח וַיִּקַּ֤ח מֹשֶׁה֙ אֶת־הַדָּ֔ם וַיִּזְרֹ֖ק עַל־הָעָ֑ם וַיֹּ֗אמֶר הִנֵּ֤ה דַֽם־הַבְּרִית֙ אֲשֶׁ֨ר כָּרַ֤ת יְהוָה֙ עִמָּכֶ֔ם עַ֥ל כָּל־הַדְּבָרִ֖ים הָאֵֽלֶּה׃ וּנְסֵיב מֹשֶׁה יָת דְּמָא, וּזְרַק עַל מַדְבְּחָא לְכַפָּרָא עַל עַמָּא; וַאֲמַר, הָא דֵּין דַּם קְיָמָא דִּגְזַר יְיָ עִמְּכוֹן, עַל כָּל פִּתְגָמַיָּא, הָאִלֵּין. 
כד,ט וַיַּ֥עַל מֹשֶׁ֖ה וְאַֽהֲרֹ֑ן נָדָב֙ וַֽאֲבִיה֔וּא וְשִׁבְעִ֖ים מִזִּקְנֵ֥י יִשְׂרָאֵֽל׃ וּסְלֵיק מֹשֶׁה, וְאַהֲרוֹן--נָדָב, וַאֲבִיהוּא, וְשִׁבְעִין, מִסָּבֵי יִשְׂרָאֵל. 
כד,י וַיִּרְא֕וּ אֵ֖ת אֱלֹהֵ֣י יִשְׂרָאֵ֑ל וְתַ֣חַת רַגְלָ֗יו כְּמַֽעֲשֵׂה֙ לִבְנַ֣ת הַסַּפִּ֔יר וּכְעֶ֥צֶם הַשָּׁמַ֖יִם לָטֹֽהַר׃ וַחֲזוֹ, יָת יְקָר אֱלָהָא דְּיִשְׂרָאֵל; וּתְחוֹת כֻּרְסֵי יְקָרֵיהּ, כְּעוֹבָד אֶבֶן טָבָא, וּכְמִחְזֵי שְׁמַיָּא, לְבָרִירוּ. 
כד,יא וְאֶל־אֲצִילֵי֙ בְּנֵ֣י יִשְׂרָאֵ֔ל לֹ֥א שָׁלַ֖ח יָד֑וֹ וַֽיֶּחֱזוּ֙ אֶת־הָ֣אֱלֹהִ֔ים וַיֹּֽאכְל֖וּ וַיִּשְׁתּֽוּ׃ {ס} וּלְרַבְרְבֵי בְּנֵי יִשְׂרָאֵל, לָא הֲוָה נִזְקָא; וַחֲזוֹ, יָת יְקָרָא דַּייָ, וַהֲווֹ חָדַן בְּקֻרְבָּנֵיהוֹן דְּאִתְקַבַּלוּ כְּאִלּוּ אָכְלִין, וְשָׁתַן. {ס}
כד,יב וַיֹּ֨אמֶר יְהוָ֜ה אֶל־מֹשֶׁ֗ה עֲלֵ֥ה אֵלַ֛י הָהָ֖רָה וֶֽהְיֵה־שָׁ֑ם וְאֶתְּנָ֨ה לְךָ֜ אֶת־לֻחֹ֣ת הָאֶ֗בֶן וְהַתּוֹרָה֙ וְהַמִּצְוָ֔ה אֲשֶׁ֥ר כָּתַ֖בְתִּי לְהֽוֹרֹתָֽם׃ וַאֲמַר יְיָ לְמֹשֶׁה, סַק לִקְדָמַי לְטוּרָא--וִהְוִי תַּמָּן; וְאֶתֵּין לָךְ יָת לוּחֵי אַבְנָא, וְאוֹרָיְתָא וְתַפְקֵידְתָּא, דִּכְתַבִית, לְאַלּוֹפֵיהוֹן. 
כד,יג וַיָּ֣קָם מֹשֶׁ֔ה וִֽיהוֹשֻׁ֖עַ מְשָֽׁרְת֑וֹ וַיַּ֥עַל מֹשֶׁ֖ה אֶל־הַ֥ר הָֽאֱלֹהִֽים׃ וְקָם מֹשֶׁה, וִיהוֹשׁוּעַ מְשֻׁמְשָׁנֵיהּ; וּסְלֵיק מֹשֶׁה, לְטוּרָא דְּאִתְגְּלִי עֲלוֹהִי יְקָרָא דַּייָ. 
כד,יד וְאֶל־הַזְּקֵנִ֤ים אָמַר֙ שְׁבוּ־לָ֣נוּ בָזֶ֔ה עַ֥ד אֲשֶׁר־נָשׁ֖וּב אֲלֵיכֶ֑ם וְהִנֵּ֨ה אַֽהֲרֹ֤ן וְחוּר֙ עִמָּכֶ֔ם מִי־בַ֥עַל דְּבָרִ֖ים יִגַּ֥שׁ אֲלֵהֶֽם׃ וּלְסָבַיָּא אֲמַר אוֹרִיכוּ לַנָא הָכָא, עַד דִּנְתוּב לְוָתְכוֹן; וְהָא אַהֲרוֹן וְחוּר עִמְּכוֹן, מַאן דְּאִית לֵיהּ דִּינָא יִתְקָרַב לִקְדָמֵיהוֹן. 
כד,טו וַיַּ֥עַל מֹשֶׁ֖ה אֶל־הָהָ֑ר וַיְכַ֥ס הֶֽעָנָ֖ן אֶת־הָהָֽר׃ וּסְלֵיק מֹשֶׁה, לְטוּרָא; וַחֲפָא עֲנָנָא, יָת טוּרָא. 
כד,טז וַיִּשְׁכֹּ֤ן כְּבוֹד־יְהוָה֙ עַל־הַ֣ר סִינַ֔י וַיְכַסֵּ֥הוּ הֶֽעָנָ֖ן שֵׁ֣שֶׁת יָמִ֑ים וַיִּקְרָ֧א אֶל־מֹשֶׁ֛ה בַּיּ֥וֹם הַשְּׁבִיעִ֖י מִתּ֥וֹךְ הֶֽעָנָֽן׃ וּשְׁרָא יְקָרָא דַּייָ עַל טוּרָא דְּסִינַי, וַחֲפָהִי עֲנָנָא שִׁתָּא יוֹמִין; וּקְרָא לְמֹשֶׁה בְּיוֹמָא שְׁבִיעָאָה, מִגּוֹ עֲנָנָא. 
כד,יז וּמַרְאֵה֙ כְּב֣וֹד יְהוָ֔ה כְּאֵ֥שׁ אֹכֶ֖לֶת בְּרֹ֣אשׁ הָהָ֑ר לְעֵינֵ֖י בְּנֵ֥י יִשְׂרָאֵֽל׃ וְחֵיזוּ יְקָרָא דַּייָ, כְּאִישָׁא אָכְלָא בְּרֵישׁ טוּרָא, לְעֵינֵי, בְּנֵי יִשְׂרָאֵל. 
כד,יח וַיָּבֹ֥א מֹשֶׁ֛ה בְּת֥וֹךְ הֶֽעָנָ֖ן וַיַּ֣עַל אֶל־הָהָ֑ר וַיְהִ֤י מֹשֶׁה֙ בָּהָ֔ר אַרְבָּעִ֣ים י֔וֹם וְאַרְבָּעִ֖ים לָֽיְלָה׃ {פ} וְעָאל מֹשֶׁה בְּגוֹ עֲנָנָא, וּסְלֵיק לְטוּרָא; וַהֲוָה מֹשֶׁה, בְּטוּרָא, אַרְבְּעִין יְמָמִין, וְאַרְבְּעִין לֵילָוָן. {פ}
