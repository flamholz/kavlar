צו

ו,א וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ו,ב צַ֤ו אֶֽת־אַהֲרֹן֙ וְאֶת־בָּנָ֣יו לֵאמֹ֔ר זֹ֥את תּוֹרַ֖ת הָֽעֹלָ֑ה הִ֣וא הָֽעֹלָ֡ה עַל֩ מֽוֹקְדָ֨ה עַל־הַמִּזְבֵּ֤חַ כָּל־הַלַּ֨יְלָה֙ עַד־הַבֹּ֔קֶר וְאֵ֥שׁ הַמִּזְבֵּ֖חַ תּ֥וּקַד בּֽוֹ׃ פַּקֵּיד יָת אַהֲרוֹן וְיָת בְּנוֹהִי לְמֵימַר, דָּא אוֹרָיְתָא דַּעֲלָתָא:  הִיא עֲלָתָא דְּמִתּוֹקְדָא עַל מַדְבְּחָא כָּל לֵילְיָא, עַד צַפְרָא, וְאִישָׁתָא דְּמַדְבְּחָא, תְּהֵי יָקְדָא בֵּיהּ. 
ו,ג וְלָבַ֨שׁ הַכֹּהֵ֜ן מִדּ֣וֹ בַ֗ד וּמִֽכְנְסֵי־בַד֮ יִלְבַּ֣שׁ עַל־בְּשָׂרוֹ֒ וְהֵרִ֣ים אֶת־הַדֶּ֗שֶׁן אֲשֶׁ֨ר תֹּאכַ֥ל הָאֵ֛שׁ אֶת־הָֽעֹלָ֖ה עַל־הַמִּזְבֵּ֑חַ וְשָׂמ֕וֹ אֵ֖צֶל הַמִּזְבֵּֽחַ׃ וְיִלְבַּשׁ כָּהֲנָא לְבוּשִׁין דְּבוּץ, וּמִכְנְסִין דְּבוּץ יִלְבַּשׁ עַל בִּסְרֵיהּ, וְיַפְרֵישׁ יָת דִּשְׁנָא דְּתֵיכוֹל אִישָׁתָא יָת עֲלָתָא, עַל מַדְבְּחָא; וִישַׁוֵּינֵיהּ, בִּסְטַר מַדְבְּחָא. 
ו,ד וּפָשַׁט֙ אֶת־בְּגָדָ֔יו וְלָבַ֖שׁ בְּגָדִ֣ים אֲחֵרִ֑ים וְהוֹצִ֤יא אֶת־הַדֶּ֨שֶׁן֙ אֶל־מִח֣וּץ לַֽמַּחֲנֶ֔ה אֶל־מָק֖וֹם טָהֽוֹר׃ וְיַשְׁלַח, יָת לְבוּשׁוֹהִי, וְיִלְבַּשׁ, לְבוּשִׁין אָחֳרָנִין; וְיַפֵּיק יָת דִּשְׁנָא לְמִבַּרָא לְמַשְׁרִיתָא, לַאֲתַר דְּכֵי. 
ו,ה וְהָאֵ֨שׁ עַל־הַמִּזְבֵּ֤חַ תּֽוּקַד־בּוֹ֙ לֹ֣א תִכְבֶּ֔ה וּבִעֵ֨ר עָלֶ֧יהָ הַכֹּהֵ֛ן עֵצִ֖ים בַּבֹּ֣קֶר בַּבֹּ֑קֶר וְעָרַ֤ךְ עָלֶ֨יהָ֙ הָֽעֹלָ֔ה וְהִקְטִ֥יר עָלֶ֖יהָ חֶלְבֵ֥י הַשְּׁלָמִֽים׃ וְאִישָׁתָא עַל מַדְבְּחָא תְּהֵי יָקְדָא בֵּיהּ לָא תִּטְפֵי, וְיַבְעַר עֲלַהּ כָּהֲנָא אָעִין בִּצְפַר בִּצְפַר; וְיַסְדַּר עֲלַהּ עֲלָתָא, וְיַסֵּיק עֲלַהּ תַּרְבֵּי נִכְסַת קֻדְשַׁיָּא. 
ו,ו אֵ֗שׁ תָּמִ֛יד תּוּקַ֥ד עַל־הַמִּזְבֵּ֖חַ לֹ֥א תִכְבֶּֽה׃ {ס} אִישָׁתָא, תְּדִירָא תְּהֵי יָקְדָא עַל מַדְבְּחָא--לָא תִּטְפֵי. {ס}
ו,ז וְזֹ֥את תּוֹרַ֖ת הַמִּנְחָ֑ה הַקְרֵ֨ב אֹתָ֤הּ בְּנֵֽי־אַהֲרֹן֙ לִפְנֵ֣י יְהוָ֔ה אֶל־פְּנֵ֖י הַמִּזְבֵּֽחַ׃ וְדָא אוֹרָיְתָא, דְּמִנְחָתָא:  דִּיקָרְבוּן יָתַהּ בְּנֵי אַהֲרוֹן קֳדָם יְיָ, לִקְדָם מַדְבְּחָא. 
ו,ח וְהֵרִ֨ים מִמֶּ֜נּוּ בְּקֻמְצ֗וֹ מִסֹּ֤לֶת הַמִּנְחָה֙ וּמִשַּׁמְנָ֔הּ וְאֵת֙ כָּל־הַלְּבֹנָ֔ה אֲשֶׁ֖ר עַל־הַמִּנְחָ֑ה וְהִקְטִ֣יר הַמִּזְבֵּ֗חַ רֵ֧יחַ נִיחֹ֛חַ אַזְכָּֽרָתָ֖הּ לַֽיהוָֽה׃ וְיַפְרֵישׁ מִנַּהּ בְּקֻמְצֵיהּ, מִסֻּלְתָּא דְּמִנְחָתָא וּמִמִּשְׁחַהּ, וְיָת כָּל לְבוֹנְתָא, דְּעַל מִנְחָתָא; וְיַסֵּיק לְמַדְבְּחָא, לְאִתְקַבָּלָא בְּרַעֲוָא אַדְכָרְתַהּ--קֳדָם יְיָ. 
ו,ט וְהַנּוֹתֶ֣רֶת מִמֶּ֔נָּה יֹֽאכְל֖וּ אַֽהֲרֹ֣ן וּבָנָ֑יו מַצּ֤וֹת תֵּֽאָכֵל֙ בְּמָק֣וֹם קָדֹ֔שׁ בַּֽחֲצַ֥ר אֹֽהֶל־מוֹעֵ֖ד יֹֽאכְלֽוּהָ׃ וּדְיִשְׁתְּאַר מִנַּהּ, יֵיכְלוּן אַהֲרוֹן וּבְנוֹהִי; פַּטִּיר תִּתְאֲכִיל בַּאֲתַר קַדִּישׁ, בְּדָרַת מַשְׁכַּן זִמְנָא יֵיכְלוּנַּהּ. 
ו,י לֹ֤א תֵֽאָפֶה֙ חָמֵ֔ץ חֶלְקָ֛ם נָתַ֥תִּי אֹתָ֖הּ מֵֽאִשָּׁ֑י קֹ֤דֶשׁ קָֽדָשִׁים֙ הִ֔וא כַּֽחַטָּ֖את וְכָֽאָשָֽׁם׃ לָא תִּתְאֲפֵי חֲמִיעַ, חוּלָקְהוֹן יְהַבִית יָתַהּ מִקֻּרְבָּנָי; קֹדֶשׁ קֻדְשִׁין הִיא, כְּחַטָּתָא וְכַאֲשָׁמָא. 
ו,יא כָּל־זָכָ֞ר בִּבְנֵ֤י אַֽהֲרֹן֙ יֹֽאכְלֶ֔נָּה חָק־עוֹלָם֙ לְדֹרֹ֣תֵיכֶ֔ם מֵֽאִשֵּׁ֖י יְהוָ֑ה כֹּ֛ל אֲשֶׁר־יִגַּ֥ע בָּהֶ֖ם יִקְדָּֽשׁ׃ {פ} כָּל דְּכוּרָא בִּבְנֵי אַהֲרוֹן, יֵיכְלִנַּהּ--קְיָם עָלַם לְדָרֵיכוֹן, מִקֻּרְבָּנַיָּא דַּייָ; כֹּל דְּיִקְרַב בְּהוֹן, יִתְקַדַּשׁ. {פ}
ו,יב וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ו,יג זֶ֡ה קָרְבַּן֩ אַֽהֲרֹ֨ן וּבָנָ֜יו אֲשֶׁר־יַקְרִ֣יבוּ לַֽיהוָ֗ה בְּיוֹם֙ הִמָּשַׁ֣ח אֹת֔וֹ עֲשִׂירִ֨ת הָֽאֵפָ֥ה סֹ֛לֶת מִנְחָ֖ה תָּמִ֑יד מַֽחֲצִיתָ֣הּ בַּבֹּ֔קֶר וּמַֽחֲצִיתָ֖הּ בָּעָֽרֶב׃ דֵּין קֻרְבָּנָא דְּאַהֲרוֹן וְדִבְנוֹהִי דִּיקָרְבוּן קֳדָם יְיָ, בְּיוֹמָא דִּירַבּוֹן יָתֵיהּ--חַד מִן עַסְרָא בִּתְלָת סְאִין סֻלְתָּא מִנְחָתָא, תְּדִירָא:  פַּלְגוּתַהּ בְּצַפְרָא, וּפַלְגוּתַהּ בְּרַמְשָׁא. 
ו,יד עַֽל־מַחֲבַ֗ת בַּשֶּׁ֛מֶן תֵּֽעָשֶׂ֖ה מֻרְבֶּ֣כֶת תְּבִיאֶ֑נָּה תֻּֽפִינֵי֙ מִנְחַ֣ת פִּתִּ֔ים תַּקְרִ֥יב רֵֽיחַ־נִיחֹ֖חַ לַֽיהוָֽה׃ עַל מַסְרֵיתָא, בִּמְשַׁח תִּתְעֲבֵיד--רְבִיכָא תַּיְתֵינַהּ; תּוּפִינֵי מִנְחַת בִּצּוּעִין, תְּקָרֵיב לְאִתְקַבָּלָא בְּרַעֲוָא קֳדָם יְיָ. 
ו,טו וְהַכֹּהֵ֨ן הַמָּשִׁ֧יחַ תַּחְתָּ֛יו מִבָּנָ֖יו יַֽעֲשֶׂ֣ה אֹתָ֑הּ חָק־עוֹלָ֕ם לַֽיהוָ֖ה כָּלִ֥יל תָּקְטָֽר׃ וְכָהֲנָא דְּיִתְרַבָּא תְּחוֹתוֹהִי מִבְּנוֹהִי, יַעֲבֵיד יָתַהּ; קְיָם עָלַם, קֳדָם יְיָ גְּמִיר תִּתַּסַּק. 
ו,טז וְכָל־מִנְחַ֥ת כֹּהֵ֛ן כָּלִ֥יל תִּֽהְיֶ֖ה לֹ֥א תֵֽאָכֵֽל׃ {פ} וְכָל מִנְחָתָא דְּכָהֲנָא גְּמִיר תְּהֵי, לָא תִּתְאֲכִיל. {פ}
ו,יז וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ו,יח דַּבֵּ֤ר אֶֽל־אַהֲרֹן֙ וְאֶל־בָּנָ֣יו לֵאמֹ֔ר זֹ֥את תּוֹרַ֖ת הַֽחַטָּ֑את בִּמְק֡וֹם אֲשֶׁר֩ תִּשָּׁחֵ֨ט הָֽעֹלָ֜ה תִּשָּׁחֵ֤ט הַֽחַטָּאת֙ לִפְנֵ֣י יְהוָ֔ה קֹ֥דֶשׁ קָֽדָשִׁ֖ים הִֽוא׃ מַלֵּיל עִם אַהֲרוֹן וְעִם בְּנוֹהִי לְמֵימַר, דָּא אוֹרָיְתָא דְּחַטָּתָא:  בְּאַתְרָא דְּתִתְנְכֵיס עֲלָתָא תִּתְנְכֵיס חַטָּתָא, קֳדָם יְיָ--קֹדֶשׁ קֻדְשִׁין, הִיא. 
ו,יט הַכֹּהֵ֛ן הַֽמְחַטֵּ֥א אֹתָ֖הּ יֹֽאכְלֶ֑נָּה בְּמָק֤וֹם קָדֹשׁ֙ תֵּֽאָכֵ֔ל בַּֽחֲצַ֖ר אֹ֥הֶל מוֹעֵֽד׃ כָּהֲנָא דִּמְכַפַּר בִּדְמַהּ, יֵיכְלִנַּהּ; בַּאֲתַר קַדִּישׁ תִּתְאֲכִיל, בְּדָרַת מַשְׁכַּן זִמְנָא. 
ו,כ כֹּ֛ל אֲשֶׁר־יִגַּ֥ע בִּבְשָׂרָ֖הּ יִקְדָּ֑שׁ וַֽאֲשֶׁ֨ר יִזֶּ֤ה מִדָּמָהּ֙ עַל־הַבֶּ֔גֶד אֲשֶׁר֙ יִזֶּ֣ה עָלֶ֔יהָ תְּכַבֵּ֖ס בְּמָק֥וֹם קָדֹֽשׁ׃ כֹּל דְּיִקְרַב בְּבִסְרַהּ, יִתְקַדַּשׁ; וּדְיַדֵּי מִדְּמַהּ, עַל לְבוּשָׁא--דְּיַדֵּי עֲלַהּ, תְּחַוַּר בַּאֲתַר קַדִּישׁ. 
ו,כא וּכְלִי־חֶ֛רֶשׂ אֲשֶׁ֥ר תְּבֻשַּׁל־בּ֖וֹ יִשָּׁבֵ֑ר וְאִם־בִּכְלִ֤י נְחֹ֨שֶׁת֙ בֻּשָּׁ֔לָה וּמֹרַ֥ק וְשֻׁטַּ֖ף בַּמָּֽיִם׃ וּמָן דַּחֲסַף דְּתִתְבַּשַּׁל בֵּיהּ, יִתְּבַר; וְאִם בְּמָנָא דִּנְחָשָׁא תִּתְבַּשַּׁל, וְיִתְמְרֵיק וְיִשְׁתְּטֵיף בְּמַיָּא. 
ו,כב כָּל־זָכָ֥ר בַּכֹּֽהֲנִ֖ים יֹאכַ֣ל אֹתָ֑הּ קֹ֥דֶשׁ קָֽדָשִׁ֖ים הִֽוא׃ כָּל דְּכוּרָא בְּכָהֲנַיָּא, יֵיכוֹל יָתַהּ; קֹדֶשׁ קֻדְשִׁין, הִיא. 
ו,כג וְכָל־חַטָּ֡את אֲשֶׁר֩ יוּבָ֨א מִדָּמָ֜הּ אֶל־אֹ֧הֶל מוֹעֵ֛ד לְכַפֵּ֥ר בַּקֹּ֖דֶשׁ לֹ֣א תֵֽאָכֵ֑ל בָּאֵ֖שׁ תִּשָּׂרֵֽף׃ {פ} וְכָל חַטָּתָא דְּיִתָּעַל מִדְּמַהּ לְמַשְׁכַּן זִמְנָא, לְכַפָּרָא בְּקֻדְשָׁא--לָא תִּתְאֲכִיל; בְּנוּרָא, תִּתּוֹקַד. {פ}
ז,א וְזֹ֥את תּוֹרַ֖ת הָֽאָשָׁ֑ם קֹ֥דֶשׁ קָֽדָשִׁ֖ים הֽוּא׃ וְדָא אוֹרָיְתָא, דַּאֲשָׁמָא:  קֹדֶשׁ קֻדְשִׁין, הוּא. 
ז,ב בִּמְק֗וֹם אֲשֶׁ֤ר יִשְׁחֲטוּ֙ אֶת־הָ֣עֹלָ֔ה יִשְׁחֲט֖וּ אֶת־הָֽאָשָׁ֑ם וְאֶת־דָּמ֛וֹ יִזְרֹ֥ק עַל־הַמִּזְבֵּ֖חַ סָבִֽיב׃ בְּאַתְרָא, דְּיִכְּסוּן יָת עֲלָתָא, יִכְּסוּן, יָת אֲשָׁמָא; וְיָת דְּמֵיהּ יִזְרוֹק עַל מַדְבְּחָא, סְחוֹר סְחוֹר. 
ז,ג וְאֵ֥ת כָּל־חֶלְבּ֖וֹ יַקְרִ֣יב מִמֶּ֑נּוּ אֵ֚ת הָֽאַלְיָ֔ה וְאֶת־הַחֵ֖לֶב הַֽמְכַסֶּ֥ה אֶת־הַקֶּֽרֶב׃ וְיָת כָּל תַּרְבֵּיהּ, יְקָרֵיב מִנֵּיהּ--יָת, אַלְיְתָא, וְיָת תַּרְבָּא, דְּחָפֵי יָת גַּוָּא. 
ז,ד וְאֵת֙ שְׁתֵּ֣י הַכְּלָיֹ֔ת וְאֶת־הַחֵ֨לֶב֙ אֲשֶׁ֣ר עֲלֵיהֶ֔ן אֲשֶׁ֖ר עַל־הַכְּסָלִ֑ים וְאֶת־הַיֹּתֶ֨רֶת֙ עַל־הַכָּבֵ֔ד עַל־הַכְּלָיֹ֖ת יְסִירֶֽנָּה׃ וְיָת, תַּרְתֵּין כּוֹלְיָן, וְיָת תַּרְבָּא דַּעֲלֵיהוֹן, דְּעַל גִּסְסַיָּא; וְיָת חַצְרָא, דְּעַל כַּבְדָּא, עַל כּוֹלְיָתָא, יַעְדֵּינַהּ. 
ז,ה וְהִקְטִ֨יר אֹתָ֤ם הַכֹּהֵן֙ הַמִּזְבֵּ֔חָה אִשֶּׁ֖ה לַֽיהוָ֑ה אָשָׁ֖ם הֽוּא׃ וְיַסֵּיק יָתְהוֹן כָּהֲנָא לְמַדְבְּחָא, קֻרְבָּנָא קֳדָם יְיָ; אֲשָׁמָא, הוּא. 
ז,ו כָּל־זָכָ֥ר בַּכֹּֽהֲנִ֖ים יֹֽאכְלֶ֑נּוּ בְּמָק֤וֹם קָדוֹשׁ֙ יֵֽאָכֵ֔ל קֹ֥דֶשׁ קָֽדָשִׁ֖ים הֽוּא׃ כָּל דְּכוּרָא בְּכָהֲנַיָּא, יֵיכְלִנֵּיהּ; בַּאֲתַר קַדִּישׁ יִתְאֲכִיל, קֹדֶשׁ קֻדְשִׁין הוּא. 
ז,ז כַּֽחַטָּאת֙ כָּֽאָשָׁ֔ם תּוֹרָ֥ה אַחַ֖ת לָהֶ֑ם הַכֹּהֵ֛ן אֲשֶׁ֥ר יְכַפֶּר־בּ֖וֹ ל֥וֹ יִֽהְיֶֽה׃ כְּחַטָּתָא, כֵּין אֲשָׁמָא--אוֹרָיְתָא חֲדָא, לְהוֹן; כָּהֲנָא דִּיכַפַּר בֵּיהּ, דִּילֵיהּ יְהֵי. 
ז,ח וְהַ֨כֹּהֵ֔ן הַמַּקְרִ֖יב אֶת־עֹ֣לַת אִ֑ישׁ ע֤וֹר הָֽעֹלָה֙ אֲשֶׁ֣ר הִקְרִ֔יב לַכֹּהֵ֖ן ל֥וֹ יִֽהְיֶֽה׃ וְכָהֲנָא--דִּמְקָרֵיב, יָת עֲלַת גְּבַר:  מְשַׁךְ עֲלָתָא דִּיקָרֵיב, לְכָהֲנָא דִּילֵיהּ יְהֵי. 
ז,ט וְכָל־מִנְחָ֗ה אֲשֶׁ֤ר תֵּֽאָפֶה֙ בַּתַּנּ֔וּר וְכָל־נַֽעֲשָׂ֥ה בַמַּרְחֶ֖שֶׁת וְעַֽל־מַחֲבַ֑ת לַכֹּהֵ֛ן הַמַּקְרִ֥יב אֹתָ֖הּ ל֥וֹ תִֽהְיֶֽה׃ וְכָל מִנְחָתָא, דְּתִתְאֲפֵי בְּתַנּוּרָא, וְכָל דְּתִתְעֲבֵיד בְּרָדְתָא, וְעַל מַסְרֵיתָא--לְכָהֲנָא דִּמְקָרֵיב יָתַהּ, דִּילֵיהּ תְּהֵי. 
ז,י וְכָל־מִנְחָ֥ה בְלוּלָֽה־בַשֶּׁ֖מֶן וַֽחֲרֵבָ֑ה לְכָל־בְּנֵ֧י אַֽהֲרֹ֛ן תִּֽהְיֶ֖ה אִ֥ישׁ כְּאָחִֽיו׃ {פ} וְכָל מִנְחָתָא דְּפִילָא בִּמְשַׁח, וּדְלָא פִּילָא--לְכָל בְּנֵי אַהֲרוֹן תְּהֵי, גְּבַר כַּאֲחוּהִי. {פ}
ז,יא וְזֹ֥את תּוֹרַ֖ת זֶ֣בַח הַשְּׁלָמִ֑ים אֲשֶׁ֥ר יַקְרִ֖יב לַֽיהוָֽה׃ וְדָא אוֹרָיְתָא, דְּנִכְסַת קֻדְשַׁיָּא, דִּיקָרֵיב, קֳדָם יְיָ. 
ז,יב אִ֣ם עַל־תּוֹדָה֮ יַקְרִיבֶנּוּ֒ וְהִקְרִ֣יב ׀ עַל־זֶ֣בַח הַתּוֹדָ֗ה חַלּ֤וֹת מַצּוֹת֙ בְּלוּלֹ֣ת בַּשֶּׁ֔מֶן וּרְקִיקֵ֥י מַצּ֖וֹת מְשֻׁחִ֣ים בַּשָּׁ֑מֶן וְסֹ֣לֶת מֻרְבֶּ֔כֶת חַלֹּ֖ת בְּלוּלֹ֥ת בַּשָּֽׁמֶן׃ אִם עַל תּוֹדְתָא, יְקָרְבִנֵּיהּ--וִיקָרֵיב עַל נִכְסַת תּוֹדְתָא גְּרִיצָן פַּטִּירָן דְּפִילָן בִּמְשַׁח, וְאֶסְפּוֹגִין פַּטִּירִין דִּמְשִׁיחִין בִּמְשַׁח; וְסֹלֶת רְבִיכָא, גְּרִיצָן דְּפִילָן בִּמְשַׁח. 
ז,יג עַל־חַלֹּת֙ לֶ֣חֶם חָמֵ֔ץ יַקְרִ֖יב קָרְבָּנ֑וֹ עַל־זֶ֖בַח תּוֹדַ֥ת שְׁלָמָֽיו׃ עַל גְּרִיצָן דִּלְחֵים חֲמִיעַ, יְקָרֵיב קֻרְבָּנֵיהּ, עַל נִכְסַת, תּוֹדַת קֻדְשׁוֹהִי. 
ז,יד וְהִקְרִ֨יב מִמֶּ֤נּוּ אֶחָד֙ מִכָּל־קָרְבָּ֔ן תְּרוּמָ֖ה לַֽיהוָ֑ה לַכֹּהֵ֗ן הַזֹּרֵ֛ק אֶת־דַּ֥ם הַשְּׁלָמִ֖ים ל֥וֹ יִֽהְיֶֽה׃ וִיקָרֵיב מִנֵּיהּ חַד מִכָּל קֻרְבָּנָא, אַפְרָשׁוּתָא קֳדָם יְיָ; לְכָהֲנָא, דְּיִזְרוֹק יָת דַּם נִכְסַת קֻדְשַׁיָּא--דִּילֵיהּ יְהֵי. 
ז,טו וּבְשַׂ֗ר זֶ֚בַח תּוֹדַ֣ת שְׁלָמָ֔יו בְּי֥וֹם קָרְבָּנ֖וֹ יֵֽאָכֵ֑ל לֹֽא־יַנִּ֥יחַ מִמֶּ֖נּוּ עַד־בֹּֽקֶר׃ וּבְסַר, נִכְסַת תּוֹדַת קֻדְשׁוֹהִי--בְּיוֹם קֻרְבָּנֵיהּ, יִתְאֲכִיל:  לָא יַצְנַע מִנֵּיהּ, עַד צַפְרָא. 
ז,טז וְאִם־נֶ֣דֶר ׀ א֣וֹ נְדָבָ֗ה זֶ֚בַח קָרְבָּנ֔וֹ בְּי֛וֹם הַקְרִיב֥וֹ אֶת־זִבְח֖וֹ יֵֽאָכֵ֑ל וּמִֽמָּחֳרָ֔ת וְהַנּוֹתָ֥ר מִמֶּ֖נּוּ יֵֽאָכֵֽל׃ וְאִם נִדְרָא אוֹ נְדַבְתָּא, נִכְסַת קֻרְבָּנֵיהּ--בְּיוֹמָא דִּיקָרֵיב יָת נִכְסְתֵיהּ, יִתְאֲכִיל; וּבְיוֹמָא דְּבָתְרוֹהִי, וּדְיִשְׁתְּאַר מִנֵּיהּ יִתְאֲכִיל. 
ז,יז וְהַנּוֹתָ֖ר מִבְּשַׂ֣ר הַזָּ֑בַח בַּיּוֹם֙ הַשְּׁלִישִׁ֔י בָּאֵ֖שׁ יִשָּׂרֵֽף׃ וּדְיִשְׁתְּאַר, מִבְּסַר נִכְסְתָא--בְּיוֹמָא, תְּלִיתָאָה, בְּנוּרָא, יִתּוֹקַד. 
ז,יח וְאִ֣ם הֵֽאָכֹ֣ל יֵֽ֠אָכֵל מִבְּשַׂר־זֶ֨בַח שְׁלָמָ֜יו בַּיּ֣וֹם הַשְּׁלִישִׁי֮ לֹ֣א יֵֽרָצֶה֒ הַמַּקְרִ֣יב אֹת֗וֹ לֹ֧א יֵֽחָשֵׁ֛ב ל֖וֹ פִּגּ֣וּל יִֽהְיֶ֑ה וְהַנֶּ֛פֶשׁ הָֽאֹכֶ֥לֶת מִמֶּ֖נּוּ עֲו‍ֹנָ֥הּ תִּשָּֽׂא׃ וְאִם אִתְאֲכָלָא יִתְאֲכִיל מִבְּסַר נִכְסַת קֻדְשׁוֹהִי בְּיוֹמָא תְּלִיתָאָה, לָא יְהֵי לְרַעֲוָא--דִּמְקָרֵיב יָתֵיהּ לָא יִתְחֲשֵׁיב לֵיהּ, מְרַחַק יְהֵי; וַאֲנָשׁ דְּיֵיכוֹל מִנֵּיהּ, חוֹבֵיהּ יְקַבֵּיל. 
ז,יט וְהַבָּשָׂ֞ר אֲשֶׁר־יִגַּ֤ע בְּכָל־טָמֵא֙ לֹ֣א יֵֽאָכֵ֔ל בָּאֵ֖שׁ יִשָּׂרֵ֑ף וְהַ֨בָּשָׂ֔ר כָּל־טָה֖וֹר יֹאכַ֥ל בָּשָֽׂר׃ וּבְסַר קֻדְשָׁא דְּיִקְרַב בְּכָל מְסָאַב, לָא יִתְאֲכִיל--בְּנוּרָא, יִתּוֹקַד; וּבְסַר קֻדְשָׁא--כָּל דְּיִדְכֵּי לְקֻדְשָׁא, יֵיכוֹל בְּסַר קֻדְשָׁא. 
ז,כ וְהַנֶּ֜פֶשׁ אֲשֶׁר־תֹּאכַ֣ל בָּשָׂ֗ר מִזֶּ֤בַח הַשְּׁלָמִים֙ אֲשֶׁ֣ר לַֽיהוָ֔ה וְטֻמְאָת֖וֹ עָלָ֑יו וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מֵֽעַמֶּֽיהָ׃ וַאֲנָשׁ דְּיֵיכוֹל בִּסְרָא, מִנִּכְסַת קֻדְשַׁיָּא דִּקְדָם יְיָ, וּסְאוֹבְתֵיהּ, עֲלוֹהִי--וְיִשְׁתֵּיצֵי אֲנָשָׁא הַהוּא, מֵעַמֵּיהּ. 
ז,כא וְנֶ֜פֶשׁ כִּֽי־תִגַּ֣ע בְּכָל־טָמֵ֗א בְּטֻמְאַ֤ת אָדָם֙ א֣וֹ ׀ בִּבְהֵמָ֣ה טְמֵאָ֗ה א֚וֹ בְּכָל־שֶׁ֣קֶץ טָמֵ֔א וְאָכַ֛ל מִבְּשַׂר־זֶ֥בַח הַשְּׁלָמִ֖ים אֲשֶׁ֣ר לַֽיהוָ֑ה וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מֵֽעַמֶּֽיהָ׃ {פ} וַאֲנָשׁ אֲרֵי יִקְרַב בְּכָל מְסָאַב, בְּסוֹאֲבָת אֲנָשָׁא אוֹ בִּבְעִירָא מְסָאֲבָא אוֹ בְּכָל שְׁקֵיץ מְסָאַב, וְיֵיכוֹל מִבְּסַר נִכְסַת קֻדְשַׁיָּא, דִּקְדָם יְיָ--וְיִשְׁתֵּיצֵי אֲנָשָׁא הַהוּא, מֵעַמֵּיהּ. {פ}
ז,כב וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ז,כג דַּבֵּ֛ר אֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר כָּל־חֵ֜לֶב שׁ֥וֹר וְכֶ֛שֶׂב וָעֵ֖ז לֹ֥א תֹאכֵֽלוּ׃ מַלֵּיל עִם בְּנֵי יִשְׂרָאֵל, לְמֵימַר:  כָּל תְּרַב תּוֹר וְאִמַּר, וְעֵז--לָא תֵּיכְלוּן. 
ז,כד וְחֵ֤לֶב נְבֵלָה֙ וְחֵ֣לֶב טְרֵפָ֔ה יֵֽעָשֶׂ֖ה לְכָל־מְלָאכָ֑ה וְאָכֹ֖ל לֹ֥א תֹֽאכְלֻֽהוּ׃ וּתְרַב נְבִילָא וּתְרַב תְּבִירָא, יִתְעֲבֵיד לְכָל עֲבִידָא; וּמֵיכָל, לָא תֵּיכְלוּנֵּיהּ. 
ז,כה כִּ֚י כָּל־אֹכֵ֣ל חֵ֔לֶב מִ֨ן־הַבְּהֵמָ֔ה אֲשֶׁ֨ר יַקְרִ֥יב מִמֶּ֛נָּה אִשֶּׁ֖ה לַֽיהוָ֑ה וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הָֽאֹכֶ֖לֶת מֵֽעַמֶּֽיהָ׃ אֲרֵי, כָּל דְּיֵיכוֹל תַּרְבָּא, מִן בְּעִירָא, דִּיקָרֵיב מִנַּהּ קֻרְבָּנָא קֳדָם יְיָ--וְיִשְׁתֵּיצֵי אֲנָשָׁא דְּיֵיכוֹל, מֵעַמֵּיהּ. 
ז,כו וְכָל־דָּם֙ לֹ֣א תֹֽאכְל֔וּ בְּכֹ֖ל מוֹשְׁבֹֽתֵיכֶ֑ם לָע֖וֹף וְלַבְּהֵמָֽה׃ וְכָל דְּמָא לָא תֵּיכְלוּן, בְּכֹל מוֹתְבָנֵיכוֹן, דְּעוֹפָא, וְדִבְעִירָא. 
ז,כז כָּל־נֶ֖פֶשׁ אֲשֶׁר־תֹּאכַ֣ל כָּל־דָּ֑ם וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מֵֽעַמֶּֽיהָ׃ כָּל אֲנָשׁ, דְּיֵיכוֹל כָּל דַּם--וְיִשְׁתֵּיצֵי אֲנָשָׁא הַהוּא, מֵעַמֵּיהּ. 
ז,כח וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ז,כט דַּבֵּ֛ר אֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר הַמַּקְרִ֞יב אֶת־זֶ֤בַח שְׁלָמָיו֙ לַֽיהוָ֔ה יָבִ֧יא אֶת־קָרְבָּנ֛וֹ לַֽיהוָ֖ה מִזֶּ֥בַח שְׁלָמָֽיו׃ מַלֵּיל עִם בְּנֵי יִשְׂרָאֵל, לְמֵימַר:  דִּמְקָרֵיב יָת נִכְסַת קֻדְשׁוֹהִי, קֳדָם יְיָ--יַיְתֵי יָת קֻרְבָּנֵיהּ לִקְדָם יְיָ, מִנִּכְסַת קֻדְשׁוֹהִי. 
ז,ל יָדָ֣יו תְּבִיאֶ֔ינָה אֵ֖ת אִשֵּׁ֣י יְהוָ֑ה אֶת־הַחֵ֤לֶב עַל־הֶֽחָזֶה֙ יְבִיאֶ֔נּוּ אֵ֣ת הֶֽחָזֶ֗ה לְהָנִ֥יף אֹת֛וֹ תְּנוּפָ֖ה לִפְנֵ֥י יְהוָֽה׃ יְדוֹהִי יֵיתְיָן, יָת קֻרְבְּנַיָּא דַּייָ; יָת תַּרְבָּא עַל חַדְיָא, יַיְתֵינֵיהּ--יָת חַדְיָא לְאָרָמָא יָתֵיהּ אֲרָמָא, קֳדָם יְיָ. 
ז,לא וְהִקְטִ֧יר הַכֹּהֵ֛ן אֶת־הַחֵ֖לֶב הַמִּזְבֵּ֑חָה וְהָיָה֙ הֶֽחָזֶ֔ה לְאַֽהֲרֹ֖ן וּלְבָנָֽיו׃ וְיַסֵּיק כָּהֲנָא יָת תַּרְבָּא, לְמַדְבְּחָא; וִיהֵי, חַדְיָא, לְאַהֲרוֹן, וְלִבְנוֹהִי. 
ז,לב וְאֵת֙ שׁ֣וֹק הַיָּמִ֔ין תִּתְּנ֥וּ תְרוּמָ֖ה לַכֹּהֵ֑ן מִזִּבְחֵ֖י שַׁלְמֵיכֶֽם׃ וְיָת שָׁקָא דְּיַמִּינָא, תִּתְּנוּן אַפְרָשׁוּתָא לְכָהֲנָא, מִנִּכְסַת, קֻדְשֵׁיכוֹן. 
ז,לג הַמַּקְרִ֞יב אֶת־דַּ֧ם הַשְּׁלָמִ֛ים וְאֶת־הַחֵ֖לֶב מִבְּנֵ֣י אַֽהֲרֹ֑ן ל֧וֹ תִֽהְיֶ֛ה שׁ֥וֹק הַיָּמִ֖ין לְמָנָֽה׃ דִּמְקָרֵיב יָת דַּם נִכְסַת קֻדְשַׁיָּא, וְיָת תַּרְבָּא--מִבְּנֵי אַהֲרוֹן:  דִּילֵיהּ תְּהֵי שָׁקָא דְּיַמִּינָא, לֻחְלָק. 
ז,לד כִּי֩ אֶת־חֲזֵ֨ה הַתְּנוּפָ֜ה וְאֵ֣ת ׀ שׁ֣וֹק הַתְּרוּמָ֗ה לָקַ֨חְתִּי֙ מֵאֵ֣ת בְּנֵֽי־יִשְׂרָאֵ֔ל מִזִּבְחֵ֖י שַׁלְמֵיהֶ֑ם וָֽאֶתֵּ֣ן אֹ֠תָם לְאַֽהֲרֹ֨ן הַכֹּהֵ֤ן וּלְבָנָיו֙ לְחָק־עוֹלָ֔ם מֵאֵ֖ת בְּנֵ֥י יִשְׂרָאֵֽל׃ אֲרֵי יָת חַדְיָא דַּאֲרָמוּתָא וְיָת שָׁקָא דְּאַפְרָשׁוּתָא, נְסֵיבִית מִן בְּנֵי יִשְׂרָאֵל, מִנִּכְסַת, קֻדְשֵׁיהוֹן; וִיהַבִית יָתְהוֹן לְאַהֲרוֹן כָּהֲנָא וְלִבְנוֹהִי, לִקְיָם עָלַם, מִן, בְּנֵי יִשְׂרָאֵל. 
ז,לה זֹ֣את מִשְׁחַ֤ת אַֽהֲרֹן֙ וּמִשְׁחַ֣ת בָּנָ֔יו מֵֽאִשֵּׁ֖י יְהוָ֑ה בְּיוֹם֙ הִקְרִ֣יב אֹתָ֔ם לְכַהֵ֖ן לַֽיהוָֽה׃ דָּא רְבוּת אַהֲרוֹן וּרְבוּת בְּנוֹהִי, מִקֻּרְבָּנַיָּא דַּייָ, בְּיוֹמָא דִּיקָרְבוּן יָתְהוֹן, לְשַׁמָּשָׁא קֳדָם יְיָ. 
ז,לו אֲשֶׁר֩ צִוָּ֨ה יְהוָ֜ה לָתֵ֣ת לָהֶ֗ם בְּיוֹם֙ מָשְׁח֣וֹ אֹתָ֔ם מֵאֵ֖ת בְּנֵ֣י יִשְׂרָאֵ֑ל חֻקַּ֥ת עוֹלָ֖ם לְדֹֽרֹתָֽם׃ דְּפַקֵּיד יְיָ לְמִתַּן לְהוֹן, בְּיוֹמָא דִּירַבּוֹן יָתְהוֹן, מִן, בְּנֵי יִשְׂרָאֵל--קְיָם עָלַם, לְדָרֵיהוֹן. 
ז,לז זֹ֣את הַתּוֹרָ֗ה לָֽעֹלָה֙ לַמִּנְחָ֔ה וְלַֽחַטָּ֖את וְלָֽאָשָׁ֑ם וְלַ֨מִּלּוּאִ֔ים וּלְזֶ֖בַח הַשְּׁלָמִֽים׃ דָּא אוֹרָיְתָא, לַעֲלָתָא לְמִנְחָתָא, וּלְחַטָּתָא, וְלַאֲשָׁמָא; וּלְקֻרְבָּנַיָּא--וּלְנִכְסַת, קֻדְשַׁיָּא. 
ז,לח אֲשֶׁ֨ר צִוָּ֧ה יְהוָ֛ה אֶת־מֹשֶׁ֖ה בְּהַ֣ר סִינָ֑י בְּי֨וֹם צַוּ‍ֹת֜וֹ אֶת־בְּנֵ֣י יִשְׂרָאֵ֗ל לְהַקְרִ֧יב אֶת־קָרְבְּנֵיהֶ֛ם לַֽיהוָ֖ה בְּמִדְבַּ֥ר סִינָֽי׃ {פ} דְּפַקֵּיד יְיָ יָת מֹשֶׁה, בְּטוּרָא דְּסִינָי:  בְּיוֹמָא דְּפַקֵּיד יָת בְּנֵי יִשְׂרָאֵל, לְקָרָבָא יָת קֻרְבָּנְהוֹן קֳדָם יְיָ--בְּמַדְבְּרָא דְּסִינָי. {פ}
ח,א וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ח,ב קַ֤ח אֶֽת־אַהֲרֹן֙ וְאֶת־בָּנָ֣יו אִתּ֔וֹ וְאֵת֙ הַבְּגָדִ֔ים וְאֵ֖ת שֶׁ֣מֶן הַמִּשְׁחָ֑ה וְאֵ֣ת ׀ פַּ֣ר הַֽחַטָּ֗את וְאֵת֙ שְׁנֵ֣י הָֽאֵילִ֔ים וְאֵ֖ת סַ֥ל הַמַּצּֽוֹת׃ קָרֵיב יָת אַהֲרוֹן, וְיָת בְּנוֹהִי עִמֵּיהּ, וְיָת לְבוּשַׁיָּא, וְיָת מִשְׁחָא דִּרְבוּתָא; וְיָת תּוֹרָא דְּחַטָּתָא, וְיָת תְּרֵין דִּכְרִין, וְיָת, סַלָּא דְּפַטִּירַיָּא. 
ח,ג וְאֵ֥ת כָּל־הָֽעֵדָ֖ה הַקְהֵ֑ל אֶל־פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ וְיָת כָּל כְּנִשְׁתָּא, כְּנוֹשׁ, לִתְרַע, מַשְׁכַּן זִמְנָא. 
ח,ד וַיַּ֣עַשׂ מֹשֶׁ֔ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֹת֑וֹ וַתִּקָּהֵל֙ הָֽעֵדָ֔ה אֶל־פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ וַעֲבַד מֹשֶׁה, כְּמָא דְּפַקֵּיד יְיָ יָתֵיהּ; וְאִתְכְּנֵישַׁת, כְּנִשְׁתָּא, לִתְרַע, מַשְׁכַּן זִמְנָא. 
ח,ה וַיֹּ֥אמֶר מֹשֶׁ֖ה אֶל־הָֽעֵדָ֑ה זֶ֣ה הַדָּבָ֔ר אֲשֶׁר־צִוָּ֥ה יְהוָ֖ה לַֽעֲשֽׂוֹת׃ וַאֲמַר מֹשֶׁה, לִכְנִשְׁתָּא:  דֵּין פִּתְגָמָא, דְּפַקֵּיד יְיָ לְמֶעֱבַד. 
ח,ו וַיַּקְרֵ֣ב מֹשֶׁ֔ה אֶֽת־אַהֲרֹ֖ן וְאֶת־בָּנָ֑יו וַיִּרְחַ֥ץ אֹתָ֖ם בַּמָּֽיִם׃ וְקָרֵיב מֹשֶׁה, יָת אַהֲרוֹן וְיָת בְּנוֹהִי; וְאַסְחִי יָתְהוֹן, בְּמַיָּא. 
ח,ז וַיִּתֵּ֨ן עָלָ֜יו אֶת־הַכֻּתֹּ֗נֶת וַיַּחְגֹּ֤ר אֹתוֹ֙ בָּֽאַבְנֵ֔ט וַיַּלְבֵּ֤שׁ אֹתוֹ֙ אֶֽת־הַמְּעִ֔יל וַיִּתֵּ֥ן עָלָ֖יו אֶת־הָֽאֵפֹ֑ד וַיַּחְגֹּ֣ר אֹת֗וֹ בְּחֵ֨שֶׁב֙ הָֽאֵפֹ֔ד וַיֶּאְפֹּ֥ד ל֖וֹ בּֽוֹ׃ וִיהַב עֲלוֹהִי יָת כִּתּוּנָא, וְזָרֵיז יָתֵיהּ בְּהִמְיָנָא, וְאַלְבֵּישׁ יָתֵיהּ יָת מְעִילָא, וִיהַב עֲלוֹהִי יָת אֵיפוֹדָא; וְזָרֵיז יָתֵיהּ, בְּהִמְיַן אֵיפוֹדָא, וְאַתְקֵין לֵיהּ, בֵּיהּ. 
ח,ח וַיָּ֥שֶׂם עָלָ֖יו אֶת־הַחֹ֑שֶׁן וַיִּתֵּן֙ אֶל־הַחֹ֔שֶׁן אֶת־הָֽאוּרִ֖ים וְאֶת־הַתֻּמִּֽים׃ וְשַׁוִּי עֲלוֹהִי, יָת חֻשְׁנָא; וִיהַב, בְּחֻשְׁנָא, יָת אוּרַיָּא, וְיָת תֻּמַּיָּא. 
ח,ט וַיָּ֥שֶׂם אֶת־הַמִּצְנֶ֖פֶת עַל־רֹאשׁ֑וֹ וַיָּ֨שֶׂם עַֽל־הַמִּצְנֶ֜פֶת אֶל־מ֣וּל פָּנָ֗יו אֵ֣ת צִ֤יץ הַזָּהָב֙ נֵ֣זֶר הַקֹּ֔דֶשׁ כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ וְשַׁוִּי יָת מַצְנַפְתָּא, עַל רֵישֵׁיהּ; וְשַׁוִּי עַל מַצְנַפְתָּא לָקֳבֵיל אַפּוֹהִי, יָת צִיצָא דְּדַהְבָּא כְּלִילָא דְּקֻדְשָׁא, כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. 
ח,י וַיִּקַּ֤ח מֹשֶׁה֙ אֶת־שֶׁ֣מֶן הַמִּשְׁחָ֔ה וַיִּמְשַׁ֥ח אֶת־הַמִּשְׁכָּ֖ן וְאֶת־כָּל־אֲשֶׁר־בּ֑וֹ וַיְקַדֵּ֖שׁ אֹתָֽם׃ וּנְסֵיב מֹשֶׁה יָת מִשְׁחָא דִּרְבוּתָא, וְרַבִּי יָת מַשְׁכְּנָא וְיָת כָּל דְּבֵיהּ; וְקַדֵּישׁ, יָתְהוֹן. 
ח,יא וַיַּ֥ז מִמֶּ֛נּוּ עַל־הַמִּזְבֵּ֖חַ שֶׁ֣בַע פְּעָמִ֑ים וַיִּמְשַׁ֨ח אֶת־הַמִּזְבֵּ֜חַ וְאֶת־כָּל־כֵּלָ֗יו וְאֶת־הַכִּיֹּ֛ר וְאֶת־כַּנּ֖וֹ לְקַדְּשָֽׁם׃ וְאַדִּי מִנֵּיהּ עַל מַדְבְּחָא, שְׁבַע זִמְנִין; וְרַבִּי יָת מַדְבְּחָא וְיָת כָּל מָנוֹהִי, וְיָת כִּיּוֹרָא וְיָת בְּסִיסֵיהּ--לְקַדָּשׁוּתְהוֹן. 
ח,יב וַיִּצֹק֙ מִשֶּׁ֣מֶן הַמִּשְׁחָ֔ה עַ֖ל רֹ֣אשׁ אַֽהֲרֹ֑ן וַיִּמְשַׁ֥ח אֹת֖וֹ לְקַדְּשֽׁוֹ׃ וַאֲרֵיק מִמִּשְׁחָא דִּרְבוּתָא, עַל רֵישָׁא דְּאַהֲרוֹן; וְרַבִּי יָתֵיהּ, לְקַדָּשׁוּתֵיהּ. 
ח,יג וַיַּקְרֵ֨ב מֹשֶׁ֜ה אֶת־בְּנֵ֣י אַֽהֲרֹ֗ן וַיַּלְבִּשֵׁ֤ם כֻּתֳּנֹת֙ וַיַּחְגֹּ֤ר אֹתָם֙ אַבְנֵ֔ט וַיַּֽחֲבֹ֥שׁ לָהֶ֖ם מִגְבָּע֑וֹת כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ וְקָרֵיב מֹשֶׁה יָת בְּנֵי אַהֲרוֹן, וְאַלְבֵּישִׁנּוּן כִּתּוּנִין וְזָרֵיז יָתְהוֹן הִמְיָנִין, וְאַתְקֵין לְהוֹן, קוֹבְעִין--כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. 
ח,יד וַיַּגֵּ֕שׁ אֵ֖ת פַּ֣ר הַֽחַטָּ֑את וַיִּסְמֹ֨ךְ אַֽהֲרֹ֤ן וּבָנָיו֙ אֶת־יְדֵיהֶ֔ם עַל־רֹ֖אשׁ פַּ֥ר הַֽחַטָּֽאת׃ וְקָרֵיב, יָת תּוֹרָא דְּחַטָּתָא; וּסְמַךְ אַהֲרוֹן וּבְנוֹהִי יָת יְדֵיהוֹן, עַל רֵישׁ תּוֹרָא דְּחַטָּתָא. 
ח,טו וַיִּשְׁחָ֗ט וַיִּקַּ֨ח מֹשֶׁ֤ה אֶת־הַדָּם֙ וַ֠יִּתֵּן עַל־קַרְנ֨וֹת הַמִּזְבֵּ֤חַ סָבִיב֙ בְּאֶצְבָּע֔וֹ וַיְחַטֵּ֖א אֶת־הַמִּזְבֵּ֑חַ וְאֶת־הַדָּ֗ם יָצַק֙ אֶל־יְס֣וֹד הַמִּזְבֵּ֔חַ וַֽיְקַדְּשֵׁ֖הוּ לְכַפֵּ֥ר עָלָֽיו׃ וּנְכַס, וּנְסֵיב מֹשֶׁה יָת דְּמָא וִיהַב עַל קַרְנָת מַדְבְּחָא סְחוֹר סְחוֹר בְּאֶצְבְּעֵיהּ, וְדַכִּי, יָת מַדְבְּחָא; וְיָת דְּמָא, אֲרֵיק לִיסוֹדָא דְּמַדְבְּחָא, וְקַדְּשֵׁיהּ, לְכַפָּרָא עֲלוֹהִי. 
ח,טז וַיִּקַּ֗ח אֶֽת־כָּל־הַחֵלֶב֮ אֲשֶׁ֣ר עַל־הַקֶּרֶב֒ וְאֵת֙ יֹתֶ֣רֶת הַכָּבֵ֔ד וְאֶת־שְׁתֵּ֥י הַכְּלָיֹ֖ת וְאֶֽת־חֶלְבְּהֶ֑ן וַיַּקְטֵ֥ר מֹשֶׁ֖ה הַמִּזְבֵּֽחָה׃ וּנְסֵיב, יָת כָּל תַּרְבָּא דְּעַל גַּוָּא, וְיָת חֲצַר כַּבְדָּא, וְיָת תַּרְתֵּין כּוֹלְיָן וְיָת תַּרְבְּהוֹן; וְאַסֵּיק מֹשֶׁה, לְמַדְבְּחָא. 
ח,יז וְאֶת־הַפָּ֤ר וְאֶת־עֹרוֹ֙ וְאֶת־בְּשָׂר֣וֹ וְאֶת־פִּרְשׁ֔וֹ שָׂרַ֣ף בָּאֵ֔שׁ מִח֖וּץ לַֽמַּחֲנֶ֑ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ וְיָת תּוֹרָא וְיָת מַשְׁכֵּיהּ, וְיָת בִּסְרֵיהּ וְיָת אֻכְלֵיהּ--אוֹקֵיד בְּנוּרָא, מִבַּרָא לְמַשְׁרִיתָא:  כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. 
ח,יח וַיַּקְרֵ֕ב אֵ֖ת אֵ֣יל הָֽעֹלָ֑ה וַֽיִּסְמְכ֞וּ אַֽהֲרֹ֧ן וּבָנָ֛יו אֶת־יְדֵיהֶ֖ם עַל־רֹ֥אשׁ הָאָֽיִל׃ וְקָרֵיב, יָת דִּכְרָא דַּעֲלָתָא; וּסְמַכוּ אַהֲרוֹן וּבְנוֹהִי, יָת יְדֵיהוֹן--עַל רֵישׁ דִּכְרָא. 
ח,יט וַיִּשְׁחָ֑ט וַיִּזְרֹ֨ק מֹשֶׁ֧ה אֶת־הַדָּ֛ם עַל־הַמִּזְבֵּ֖חַ סָבִֽיב׃ וּנְכַס; וּזְרַק מֹשֶׁה יָת דְּמָא עַל מַדְבְּחָא, סְחוֹר סְחוֹר. 
ח,כ וְאֶ֨ת־הָאַ֔יִל נִתַּ֖ח לִנְתָחָ֑יו וַיַּקְטֵ֤ר מֹשֶׁה֙ אֶת־הָרֹ֔אשׁ וְאֶת־הַנְּתָחִ֖ים וְאֶת־הַפָּֽדֶר׃ וְיָת דִּכְרָא--פַּלֵּיג, לְאֶבְרוֹהִי; וְאַסֵּיק מֹשֶׁה יָת רֵישָׁא, וְיָת אֶבְרַיָּא וְיָת תַּרְבָּא. 
ח,כא וְאֶת־הַקֶּ֥רֶב וְאֶת־הַכְּרָעַ֖יִם רָחַ֣ץ בַּמָּ֑יִם וַיַּקְטֵר֩ מֹשֶׁ֨ה אֶת־כָּל־הָאַ֜יִל הַמִּזְבֵּ֗חָה עֹלָ֨ה ה֤וּא לְרֵֽיחַ־נִיחֹ֨חַ֙ אִשֶּׁ֥ה הוּא֙ לַֽיהוָ֔ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ וְיָת גַּוָּא וְיָת כְּרָעַיָּא, חַלֵּיל בְּמַיָּא; וְאַסֵּיק מֹשֶׁה יָת כָּל דִּכְרָא לְמַדְבְּחָא, עֲלָתָא הוּא לְאִתְקַבָּלָא בְּרַעֲוָא קֻרְבָּנָא הוּא קֳדָם יְיָ, כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. 
ח,כב וַיַּקְרֵב֙ אֶת־הָאַ֣יִל הַשֵּׁנִ֔י אֵ֖יל הַמִּלֻּאִ֑ים וַֽיִּסְמְכ֞וּ אַֽהֲרֹ֧ן וּבָנָ֛יו אֶת־יְדֵיהֶ֖ם עַל־רֹ֥אשׁ הָאָֽיִל׃ וְקָרֵיב יָת דִּכְרָא תִּנְיָנָא, דְּכַר קֻרְבָּנַיָּא; וּסְמַכוּ אַהֲרוֹן וּבְנוֹהִי, יָת יְדֵיהוֹן--עַל רֵישׁ דִּכְרָא. 
ח,כג וַיִּשְׁחָ֓ט ׀ וַיִּקַּ֤ח מֹשֶׁה֙ מִדָּמ֔וֹ וַיִּתֵּ֛ן עַל־תְּנ֥וּךְ אֹֽזֶן־אַהֲרֹ֖ן הַיְמָנִ֑ית וְעַל־בֹּ֤הֶן יָדוֹ֙ הַיְמָנִ֔ית וְעַל־בֹּ֥הֶן רַגְל֖וֹ הַיְמָנִֽית׃ וּנְכַס--וּנְסֵיב מֹשֶׁה מִדְּמֵיהּ, וִיהַב עַל רוּם אֻדְנָא דְּאַהֲרוֹן דְּיַמִּינָא; וְעַל אִלְיוֹן יְדֵיהּ דְּיַמִּינָא, וְעַל אִלְיוֹן רַגְלֵיהּ דְּיַמִּינָא. 
ח,כד וַיַּקְרֵ֞ב אֶת־בְּנֵ֣י אַֽהֲרֹ֗ן וַיִּתֵּ֨ן מֹשֶׁ֤ה מִן־הַדָּם֙ עַל־תְּנ֤וּךְ אָזְנָם֙ הַיְמָנִ֔ית וְעַל־בֹּ֤הֶן יָדָם֙ הַיְמָנִ֔ית וְעַל־בֹּ֥הֶן רַגְלָ֖ם הַיְמָנִ֑ית וַיִּזְרֹ֨ק מֹשֶׁ֧ה אֶת־הַדָּ֛ם עַל־הַמִּזְבֵּ֖חַ סָבִֽיב׃ וְקָרֵיב יָת בְּנֵי אַהֲרוֹן, וִיהַב מֹשֶׁה מִן דְּמָא עַל רוּם אֻדְנְהוֹן דְּיַמִּינָא, וְעַל אִלְיוֹן יַדְהוֹן דְּיַמִּינָא, וְעַל אִלְיוֹן רַגְלְהוֹן דְּיַמִּינָא; וּזְרַק מֹשֶׁה יָת דְּמָא עַל מַדְבְּחָא, סְחוֹר סְחוֹר. 
ח,כה וַיִּקַּ֞ח אֶת־הַחֵ֣לֶב וְאֶת־הָֽאַלְיָ֗ה וְאֶֽת־כָּל־הַחֵלֶב֮ אֲשֶׁ֣ר עַל־הַקֶּרֶב֒ וְאֵת֙ יֹתֶ֣רֶת הַכָּבֵ֔ד וְאֶת־שְׁתֵּ֥י הַכְּלָיֹ֖ת וְאֶֽת־חֶלְבְּהֶ֑ן וְאֵ֖ת שׁ֥וֹק הַיָּמִֽין׃ וּנְסֵיב יָת תַּרְבָּא וְיָת אַלְיְתָא, וְיָת כָּל תַּרְבָּא דְּעַל גַּוָּא, וְיָת חֲצַר כַּבְדָּא, וְיָת תַּרְתֵּין כּוֹלְיָן וְיָת תַּרְבְּהוֹן; וְיָת, שָׁקָא דְּיַמִּינָא. 
ח,כו וּמִסַּ֨ל הַמַּצּ֜וֹת אֲשֶׁ֣ר ׀ לִפְנֵ֣י יְהוָ֗ה לָ֠קַח חַלַּ֨ת מַצָּ֤ה אַחַת֙ וְֽחַלַּ֨ת לֶ֥חֶם שֶׁ֛מֶן אַחַ֖ת וְרָקִ֣יק אֶחָ֑ד וַיָּ֨שֶׂם֙ עַל־הַ֣חֲלָבִ֔ים וְעַ֖ל שׁ֥וֹק הַיָּמִֽין׃ וּמִסַּלָּא דְּפַטִּירַיָּא דִּקְדָם יְיָ, נְסֵיב גְּרִיצְתָא פַּטִּירְתָא חֲדָא וּגְרִיצְתָא דִּלְחֵים מְשַׁח חֲדָא--וְאֶסְפּוֹג חַד; וְשַׁוִּי, עַל תַּרְבַּיָּא, וְעַל, שָׁקָא דְּיַמִּינָא. 
ח,כז וַיִּתֵּ֣ן אֶת־הַכֹּ֔ל עַ֚ל כַּפֵּ֣י אַֽהֲרֹ֔ן וְעַ֖ל כַּפֵּ֣י בָנָ֑יו וַיָּ֧נֶף אֹתָ֛ם תְּנוּפָ֖ה לִפְנֵ֥י יְהוָֽה׃ וִיהַב יָת כּוֹלָא--עַל יְדֵי אַהֲרוֹן, וְעַל יְדֵי בְּנוֹהִי; וַאֲרֵים יָתְהוֹן אֲרָמָא, קֳדָם יְיָ. 
ח,כח וַיִּקַּ֨ח מֹשֶׁ֤ה אֹתָם֙ מֵעַ֣ל כַּפֵּיהֶ֔ם וַיַּקְטֵ֥ר הַמִּזְבֵּ֖חָה עַל־הָֽעֹלָ֑ה מִלֻּאִ֥ים הֵם֙ לְרֵ֣יחַ נִיחֹ֔חַ אִשֶּׁ֥ה ה֖וּא לַֽיהוָֽה׃ וּנְסֵיב מֹשֶׁה יָתְהוֹן מֵעַל יְדֵיהוֹן, וְאַסֵּיק מַדְבְּחָא עַל עֲלָתָא; קֻרְבָּנַיָּא אִנּוּן לְאִתְקַבָּלָא בְּרַעֲוָא, קֻרְבָּנָא הוּא קֳדָם יְיָ. 
ח,כט וַיִּקַּ֤ח מֹשֶׁה֙ אֶת־הֶ֣חָזֶ֔ה וַיְנִיפֵ֥הוּ תְנוּפָ֖ה לִפְנֵ֣י יְהוָ֑ה מֵאֵ֣יל הַמִּלֻּאִ֗ים לְמֹשֶׁ֤ה הָיָה֙ לְמָנָ֔ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ וּנְסֵיב מֹשֶׁה יָת חַדְיָא, וַאֲרִימֵיהּ אֲרָמָא קֳדָם יְיָ:  מִדְּכַר קֻרְבָּנַיָּא, לְמֹשֶׁה הֲוָה לֻחְלָק, כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. 
ח,ל וַיִּקַּ֨ח מֹשֶׁ֜ה מִשֶּׁ֣מֶן הַמִּשְׁחָ֗ה וּמִן־הַדָּם֮ אֲשֶׁ֣ר עַל־הַמִּזְבֵּחַ֒ וַיַּ֤ז עַֽל־אַהֲרֹן֙ עַל־בְּגָדָ֔יו וְעַל־בָּנָ֛יו וְעַל־בִּגְדֵ֥י בָנָ֖יו אִתּ֑וֹ וַיְקַדֵּ֤שׁ אֶֽת־אַהֲרֹן֙ אֶת־בְּגָדָ֔יו וְאֶת־בָּנָ֛יו וְאֶת־בִּגְדֵ֥י בָנָ֖יו אִתּֽוֹ׃ וּנְסֵיב מֹשֶׁה מִמִּשְׁחָא דִּרְבוּתָא, וּמִן דְּמָא דְּעַל מַדְבְּחָא, וְאַדִּי עַל אַהֲרוֹן עַל לְבוּשׁוֹהִי, וְעַל בְּנוֹהִי וְעַל לְבוּשֵׁי בְּנוֹהִי עִמֵּיהּ; וְקַדֵּישׁ יָת אַהֲרוֹן יָת לְבוּשׁוֹהִי, וְיָת בְּנוֹהי וְיָת לְבוּשֵׁי בְּנוֹהִי עִמֵּיהּ. 
ח,לא וַיֹּ֨אמֶר מֹשֶׁ֜ה אֶל־אַֽהֲרֹ֣ן וְאֶל־בָּנָ֗יו בַּשְּׁל֣וּ אֶת־הַבָּשָׂר֮ פֶּ֣תַח אֹ֣הֶל מוֹעֵד֒ וְשָׁם֙ תֹּֽאכְל֣וּ אֹת֔וֹ וְאֶ֨ת־הַלֶּ֔חֶם אֲשֶׁ֖ר בְּסַ֣ל הַמִּלֻּאִ֑ים כַּֽאֲשֶׁ֤ר צִוֵּ֨יתִי֙ לֵאמֹ֔ר אַֽהֲרֹ֥ן וּבָנָ֖יו יֹֽאכְלֻֽהוּ׃ וַאֲמַר מֹשֶׁה לְאַהֲרוֹן וְלִבְנוֹהִי, בַּשִּׁילוּ יָת בִּסְרָא בִּתְרַע מַשְׁכַּן זִמְנָא, וְתַמָּן תֵּיכְלוּן יָתֵיהּ, וְיָת לַחְמָא דִּבְסַל קֻרְבָּנַיָּא:  כְּמָא דְּפַקֵּידִית לְמֵימַר, אַהֲרוֹן וּבְנוֹהִי יֵיכְלוּנֵּיהּ. 
ח,לב וְהַנּוֹתָ֥ר בַּבָּשָׂ֖ר וּבַלָּ֑חֶם בָּאֵ֖שׁ תִּשְׂרֹֽפוּ׃ וּדְיִשְׁתְּאַר בְּבִסְרָא, וּבְלַחְמָא--בְּנוּרָא, תֵּיקְדוּן. 
ח,לג וּמִפֶּתַח֩ אֹ֨הֶל מוֹעֵ֜ד לֹ֤א תֵֽצְאוּ֙ שִׁבְעַ֣ת יָמִ֔ים עַ֚ד י֣וֹם מְלֹ֔את יְמֵ֖י מִלֻּֽאֵיכֶ֑ם כִּ֚י שִׁבְעַ֣ת יָמִ֔ים יְמַלֵּ֖א אֶת־יֶדְכֶֽם׃ וּמִתְּרַע מַשְׁכַּן זִמְנָא לָא תִּפְּקוּן, שִׁבְעָא יוֹמִין--עַד יוֹם מִשְׁלַם, יוֹמֵי קֻרְבָּנְכוֹן:  אֲרֵי שִׁבְעָא יוֹמִין, יִתְקָרַב קֻרְבָּנְכוֹן. 
ח,לד כַּֽאֲשֶׁ֥ר עָשָׂ֖ה בַּיּ֣וֹם הַזֶּ֑ה צִוָּ֧ה יְהוָ֛ה לַֽעֲשֹׂ֖ת לְכַפֵּ֥ר עֲלֵיכֶֽם׃ כְּמָא דַּעֲבַד, בְּיוֹמָא הָדֵין--פַּקֵּיד יְיָ לְמֶעֱבַד, לְכַפָּרָא עֲלֵיכוֹן. 
ח,לה וּפֶתַח֩ אֹ֨הֶל מוֹעֵ֜ד תֵּֽשְׁב֨וּ יוֹמָ֤ם וָלַ֨יְלָה֙ שִׁבְעַ֣ת יָמִ֔ים וּשְׁמַרְתֶּ֛ם אֶת־מִשְׁמֶ֥רֶת יְהוָ֖ה וְלֹ֣א תָמ֑וּתוּ כִּי־כֵ֖ן צֻוֵּֽיתִי׃ וּבִתְרַע מַשְׁכַּן זִמְנָא תִּתְּבוּן יֵימָם וְלֵילֵי, שִׁבְעָא יוֹמִין, וְתִטְּרוּן יָת מַטְּרַת מֵימְרָא דַּייָ, וְלָא תְּמוּתוּן:  אֲרֵי כֵּין, אִתְפַּקַּדִית. 
ח,לו וַיַּ֥עַשׂ אַֽהֲרֹ֖ן וּבָנָ֑יו אֵ֚ת כָּל־הַדְּבָרִ֔ים אֲשֶׁר־צִוָּ֥ה יְהוָ֖ה בְּיַד־מֹשֶֽׁה׃ {ס} וַעֲבַד אַהֲרוֹן, וּבְנוֹהִי--יָת, כָּל פִּתְגָמַיָּא, דְּפַקֵּיד יְיָ, בִּידָא דְּמֹשֶׁה. {ס}
