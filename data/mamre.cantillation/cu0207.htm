תרומה

כה,א וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
כה,ב דַּבֵּר֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל וְיִקְחוּ־לִ֖י תְּרוּמָ֑ה מֵאֵ֤ת כָּל־אִישׁ֙ אֲשֶׁ֣ר יִדְּבֶ֣נּוּ לִבּ֔וֹ תִּקְח֖וּ אֶת־תְּרֽוּמָתִֽי׃ מַלֵּיל עִם בְּנֵי יִשְׂרָאֵל, וְיַפְרְשׁוּן קֳדָמַי אַפְרָשׁוּתָא:  מִן כָּל גְּבַר דְּיִתְרְעֵי לִבֵּיהּ, תִּסְּבוּן יָת אַפְרָשׁוּתִי. 
כה,ג וְזֹאת֙ הַתְּרוּמָ֔ה אֲשֶׁ֥ר תִּקְח֖וּ מֵֽאִתָּ֑ם זָהָ֥ב וָכֶ֖סֶף וּנְחֹֽשֶׁת׃ וְדָא, אַפְרָשׁוּתָא, דְּתִסְּבְוּן, מִנְּהוֹן:  דַּהְבָּא וְכַסְפָּא, וּנְחָשָׁא. 
כה,ד וּתְכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֥שׁ וְעִזִּֽים׃ וְתַכְלָא וְאַרְגְּוָנָא וּצְבַע זְהוֹרִי, וּבוּץ וּמַעְזֵי. 
כה,ה וְעֹרֹ֨ת אֵילִ֧ם מְאָדָּמִ֛ים וְעֹרֹ֥ת תְּחָשִׁ֖ים וַֽעֲצֵ֥י שִׁטִּֽים׃ וּמַשְׁכֵּי דְּדִכְרֵי מְסֻמְּקֵי וּמַשְׁכֵּי סָסְגוֹנָא, וְאָעֵי שִׁטִּין. 
כה,ו שֶׁ֖מֶן לַמָּאֹ֑ר בְּשָׂמִים֙ לְשֶׁ֣מֶן הַמִּשְׁחָ֔ה וְלִקְטֹ֖רֶת הַסַּמִּֽים׃ מִשְׁחָא, לְאַנְהָרוּתָא; בֻּסְמַיָּא לִמְשַׁח רְבוּתָא, וְלִקְטֹרֶת בֻּסְמַיָּא. 
כה,ז אַבְנֵי־שֹׁ֕הַם וְאַבְנֵ֖י מִלֻּאִ֑ים לָֽאֵפֹ֖ד וְלַחֹֽשֶׁן׃ אַבְנֵי בֻּרְלָא, וְאַבְנֵי אַשְׁלָמוּתָא, לְשַׁקָּעָא בְּאֵיפוֹדָא, וּבְחֻשְׁנָא. 
כה,ח וְעָ֥שׂוּ לִ֖י מִקְדָּ֑שׁ וְשָֽׁכַנְתִּ֖י בְּתוֹכָֽם׃ וְיַעְבְּדוּן קֳדָמַי, מַקְדַּשׁ; וְאַשְׁרֵי שְׁכִינְתִי, בֵּינֵיהוֹן. 
כה,ט כְּכֹ֗ל אֲשֶׁ֤ר אֲנִי֙ מַרְאֶ֣ה אֽוֹתְךָ֔ אֵ֚ת תַּבְנִ֣ית הַמִּשְׁכָּ֔ן וְאֵ֖ת תַּבְנִ֣ית כָּל־כֵּלָ֑יו וְכֵ֖ן תַּֽעֲשֽׂוּ׃ {ס} כְּכֹל, דַּאֲנָא מַחְזֵי יָתָךְ, יָת דְּמוּת מַשְׁכְּנָא, וְיָת דְּמוּת כָּל מָנוֹהִי; וְכֵין, תַּעְבְּדוּן. {ס}
כה,י וְעָשׂ֥וּ אֲר֖וֹן עֲצֵ֣י שִׁטִּ֑ים אַמָּתַ֨יִם וָחֵ֜צִי אָרְכּ֗וֹ וְאַמָּ֤ה וָחֵ֨צִי֙ רָחְבּ֔וֹ וְאַמָּ֥ה וָחֵ֖צִי קֹֽמָתֽוֹ׃ וְיַעְבְּדוּן אֲרוֹנָא, דְּאָעֵי שִׁטִּין:  תַּרְתֵּין אַמִּין וּפַלְגָּא אֻרְכֵּיהּ, וְאַמְּתָא וּפַלְגָּא פֻּתְיֵיהּ, וְאַמְּתָא וּפַלְגָּא, רוּמֵיהּ. 
כה,יא וְצִפִּיתָ֤ אֹתוֹ֙ זָהָ֣ב טָה֔וֹר מִבַּ֥יִת וּמִח֖וּץ תְּצַפֶּ֑נּוּ וְעָשִׂ֧יתָ עָלָ֛יו זֵ֥ר זָהָ֖ב סָבִֽיב׃ וְתִחְפֵי יָתֵיהּ דְּהַב דְּכֵי, מִגָּיו וּמִבַּרָא תִּחְפֵינֵיהּ; וְתַעֲבֵיד עֲלוֹהִי זִיר דִּדְהַב, סְחוֹר סְחוֹר. 
כה,יב וְיָצַ֣קְתָּ לּ֗וֹ אַרְבַּע֙ טַבְּעֹ֣ת זָהָ֔ב וְנָ֣תַתָּ֔ה עַ֖ל אַרְבַּ֣ע פַּֽעֲמֹתָ֑יו וּשְׁתֵּ֣י טַבָּעֹ֗ת עַל־צַלְעוֹ֙ הָֽאֶחָ֔ת וּשְׁתֵּי֙ טַבָּעֹ֔ת עַל־צַלְע֖וֹ הַשֵּׁנִֽית׃ וְתַתֵּיךְ לֵיהּ, אַרְבַּע עִזְקָן דִּדְהַב, וְתִתֵּין, עַל אַרְבַּע זָוְיָתֵיהּ; וְתַרְתֵּין עִזְקָן, עַל סִטְרֵיהּ חַד, וְתַרְתֵּין עִזְקָן, עַל סִטְרֵיהּ תִּנְיָנָא. 
כה,יג וְעָשִׂ֥יתָ בַדֵּ֖י עֲצֵ֣י שִׁטִּ֑ים וְצִפִּיתָ֥ אֹתָ֖ם זָהָֽב׃ וְתַעֲבֵיד אֲרִיחֵי, דְּאָעֵי שִׁטִּין; וְתִחְפֵי יָתְהוֹן, דַּהְבָּא. 
כה,יד וְהֵֽבֵאתָ֤ אֶת־הַבַּדִּים֙ בַּטַּבָּעֹ֔ת עַ֖ל צַלְעֹ֣ת הָֽאָרֹ֑ן לָשֵׂ֥את אֶת־הָֽאָרֹ֖ן בָּהֶֽם׃ וְתַעֵיל יָת אֲרִיחַיָּא בְּעִזְקָתָא, עַל סִטְרֵי אֲרוֹנָא, לְמִטַּל יָת אֲרוֹנָא, בְּהוֹן. 
כה,טו בְּטַבְּעֹת֙ הָֽאָרֹ֔ן יִֽהְי֖וּ הַבַּדִּ֑ים לֹ֥א יָסֻ֖רוּ מִמֶּֽנּוּ׃ בְּעִזְקָת, אֲרוֹנָא, יְהוֹן, אֲרִיחַיָּא:  לָא יִעְדּוֹן, מִנֵּיהּ. 
כה,טז וְנָֽתַתָּ֖ אֶל־הָֽאָרֹ֑ן אֵ֚ת הָֽעֵדֻ֔ת אֲשֶׁ֥ר אֶתֵּ֖ן אֵלֶֽיךָ׃ וְתִתֵּין, בַּאֲרוֹנָא--יָת, סָהֲדוּתָא, דְּאֶתֵּין, לָךְ. 
כה,יז וְעָשִׂ֥יתָ כַפֹּ֖רֶת זָהָ֣ב טָה֑וֹר אַמָּתַ֤יִם וָחֵ֨צִי֙ אָרְכָּ֔הּ וְאַמָּ֥ה וָחֵ֖צִי רָחְבָּֽהּ׃ וְתַעֲבֵיד כָּפֻרְתָּא, דִּדְהַב דְּכֵי:  תַּרְתֵּין אַמִּין וּפַלְגָּא אֻרְכַּהּ, וְאַמְּתָא וּפַלְגָּא פֻּתְיַהּ. 
כה,יח וְעָשִׂ֛יתָ שְׁנַ֥יִם כְּרֻבִ֖ים זָהָ֑ב מִקְשָׁה֙ תַּֽעֲשֶׂ֣ה אֹתָ֔ם מִשְּׁנֵ֖י קְצ֥וֹת הַכַּפֹּֽרֶת׃ וְתַעֲבֵיד תְּרֵין כְּרוּבִין, דִּדְהַב; נְגִיד תַּעֲבֵיד יָתְהוֹן, מִתְּרֵין סִטְרֵי כָּפֻרְתָּא. 
כה,יט וַֽ֠עֲשֵׂה כְּר֨וּב אֶחָ֤ד מִקָּצָה֙ מִזֶּ֔ה וּכְרוּב־אֶחָ֥ד מִקָּצָ֖ה מִזֶּ֑ה מִן־הַכַּפֹּ֛רֶת תַּֽעֲשׂ֥וּ אֶת־הַכְּרֻבִ֖ים עַל־שְׁנֵ֥י קְצוֹתָֽיו׃ וַעֲבֵיד כְּרוּבָא חַד מִסִּטְרָא, מִכָּא, וּכְרוּבָא חַד מִסִּטְרָא, מִכָּא; מִן כָּפֻרְתָּא תַּעְבְּדוּן יָת כְּרוּבַיָּא, עַל תְּרֵין סִטְרוֹהִי. 
כה,כ וְהָי֣וּ הַכְּרֻבִים֩ פֹּֽרְשֵׂ֨י כְנָפַ֜יִם לְמַ֗עְלָה סֹֽכְכִ֤ים בְּכַנְפֵיהֶם֙ עַל־הַכַּפֹּ֔רֶת וּפְנֵיהֶ֖ם אִ֣ישׁ אֶל־אָחִ֑יו אֶ֨ל־הַכַּפֹּ֔רֶת יִֽהְי֖וּ פְּנֵ֥י הַכְּרֻבִֽים׃ וִיהוֹן כְּרוּבַיָּא פְּרִיסִין גַּדְפֵּיהוֹן לְעֵילָא, מְטַלַּן בְּגַדְפֵּיהוֹן עַל כָּפֻרְתָּא, וְאַפֵּיהוֹן, חַד לָקֳבֵיל חַד; לָקֳבֵיל כָּפֻרְתָּא--יְהוֹן, אַפֵּי כְּרוּבַיָּא. 
כה,כא וְנָֽתַתָּ֧ אֶת־הַכַּפֹּ֛רֶת עַל־הָֽאָרֹ֖ן מִלְמָ֑עְלָה וְאֶל־הָ֣אָרֹ֔ן תִּתֵּן֙ אֶת־הָ֣עֵדֻ֔ת אֲשֶׁ֥ר אֶתֵּ֖ן אֵלֶֽיךָ׃ וְתִתֵּין יָת כָּפֻרְתָּא עַל אֲרוֹנָא, מִלְּעֵילָא; וּבַאֲרוֹנָא--תִּתֵּין יָת סָהֲדוּתָא, דְּאֶתֵּין לָךְ. 
כה,כב וְנֽוֹעַדְתִּ֣י לְךָ֮ שָׁם֒ וְדִבַּרְתִּ֨י אִתְּךָ֜ מֵעַ֣ל הַכַּפֹּ֗רֶת מִבֵּין֙ שְׁנֵ֣י הַכְּרֻבִ֔ים אֲשֶׁ֖ר עַל־אֲר֣וֹן הָֽעֵדֻ֑ת אֵ֣ת כָּל־אֲשֶׁ֧ר אֲצַוֶּ֛ה אֽוֹתְךָ֖ אֶל־בְּנֵ֥י יִשְׂרָאֵֽל׃ {פ} וַאֲזָמֵין מֵימְרִי לָךְ, תַּמָּן, וַאֲמַלֵּיל עִמָּךְ מֵעִלָּוֵי כָּפֻרְתָּא מִבֵּין תְּרֵין כְּרוּבַיָּא, דְּעַל אֲרוֹנָא דְּסָהֲדוּתָא--יָת כָּל דַּאֲפַקֵּיד יָתָךְ, לְוָת בְּנֵי יִשְׂרָאֵל. {פ}
כה,כג וְעָשִׂ֥יתָ שֻׁלְחָ֖ן עֲצֵ֣י שִׁטִּ֑ים אַמָּתַ֤יִם אָרְכּוֹ֙ וְאַמָּ֣ה רָחְבּ֔וֹ וְאַמָּ֥ה וָחֵ֖צִי קֹֽמָתֽוֹ׃ וְתַעֲבֵיד פָּתוּרָא, דְּאָעֵי שִׁטִּין:  תַּרְתֵּין אַמִּין אֻרְכֵּיהּ וְאַמְּתָא פֻּתְיֵיהּ, וְאַמְּתָא וּפַלְגָּא רוּמֵיהּ. 
כה,כד וְצִפִּיתָ֥ אֹת֖וֹ זָהָ֣ב טָה֑וֹר וְעָשִׂ֥יתָ לּ֛וֹ זֵ֥ר זָהָ֖ב סָבִֽיב׃ וְתִחְפֵי יָתֵיהּ, דְּהַב דְּכֵי; וְתַעֲבֵיד לֵיהּ זִיר דִּדְהַב, סְחוֹר סְחוֹר. 
כה,כה וְעָשִׂ֨יתָ לּ֥וֹ מִסְגֶּ֛רֶת טֹ֖פַח סָבִ֑יב וְעָשִׂ֧יתָ זֵר־זָהָ֛ב לְמִסְגַּרְתּ֖וֹ סָבִֽיב׃ וְתַעֲבֵיד לֵיהּ גְּדָנְפָא רוּמֵיהּ פֻּשְׁכָּא, סְחוֹר סְחוֹר; וְתַעֲבֵיד זִיר דִּדְהַב לִגְדָנְפֵיהּ, סְחוֹר סְחוֹר. 
כה,כו וְעָשִׂ֣יתָ לּ֔וֹ אַרְבַּ֖ע טַבְּעֹ֣ת זָהָ֑ב וְנָֽתַתָּ֙ אֶת־הַטַּבָּעֹ֔ת עַ֚ל אַרְבַּ֣ע הַפֵּאֹ֔ת אֲשֶׁ֖ר לְאַרְבַּ֥ע רַגְלָֽיו׃ וְתַעֲבֵיד לֵיהּ, אַרְבַּע עִזְקָן דִּדְהַב; וְתִתֵּין, יָת עִזְקָתָא, עַל אַרְבַּע זָוְיָתָא, דִּלְאַרְבַּע רַגְלוֹהִי. 
כה,כז לְעֻמַּת֙ הַמִּסְגֶּ֔רֶת תִּֽהְיֶ֖יןָ הַטַּבָּעֹ֑ת לְבָתִּ֣ים לְבַדִּ֔ים לָשֵׂ֖את אֶת־הַשֻּׁלְחָֽן׃ לָקֳבֵיל, גְּדָנְפָא, יִהְוְיָן, עִזְקָתָא--לְאַתְרָא לַאֲרִיחַיָּא, לְמִטַּל יָת פָּתוּרָא. 
כה,כח וְעָשִׂ֤יתָ אֶת־הַבַּדִּים֙ עֲצֵ֣י שִׁטִּ֔ים וְצִפִּיתָ֥ אֹתָ֖ם זָהָ֑ב וְנִשָּׂא־בָ֖ם אֶת־הַשֻּׁלְחָֽן׃ וְתַעֲבֵיד יָת אֲרִיחַיָּא דְּאָעֵי שִׁטִּין, וְתִחְפֵי יָתְהוֹן דַּהְבָּא; וִיהוֹן נָטְלִין בְּהוֹן, יָת פָּתוּרָא. 
כה,כט וְעָשִׂ֨יתָ קְּעָֽרֹתָ֜יו וְכַפֹּתָ֗יו וּקְשׂוֹתָיו֙ וּמְנַקִּיֹּתָ֔יו אֲשֶׁ֥ר יֻסַּ֖ךְ בָּהֵ֑ן זָהָ֥ב טָה֖וֹר תַּֽעֲשֶׂ֥ה אֹתָֽם׃ וְתַעֲבֵיד מְגִסּוֹהִי וּבָזִכּוֹהִי, וְקָסְוָתֵיהּ וּמְכִילָתֵיהּ, דְּיִתְנַסַּךְ, בְּהוֹן; דְּהַב דְּכֵי, תַּעֲבֵיד יָתְהוֹן. 
כה,ל וְנָֽתַתָּ֧ עַֽל־הַשֻּׁלְחָ֛ן לֶ֥חֶם פָּנִ֖ים לְפָנַ֥י תָּמִֽיד׃ {פ} וְתִתֵּין עַל פָּתוּרָא לְחֵים אַפַּיָּא, קֳדָמַי תְּדִירָא. {פ}
כה,לא וְעָשִׂ֥יתָ מְנֹרַ֖ת זָהָ֣ב טָה֑וֹר מִקְשָׁ֞ה תֵּֽעָשֶׂ֤ה הַמְּנוֹרָה֙ יְרֵכָ֣הּ וְקָנָ֔הּ גְּבִיעֶ֛יהָ כַּפְתֹּרֶ֥יהָ וּפְרָחֶ֖יהָ מִמֶּ֥נָּה יִֽהְיֽוּ׃ וְתַעֲבֵיד מְנָרְתָא, דִּדְהַב דְּכֵי; נְגִיד תִּתְעֲבֵיד מְנָרְתָא, שִׁדַּהּ וּקְנַהּ, כַּלִּידַהָא חַזּוּרַהָא וְשׁוֹשַׁנַּהָא, מִנַּהּ יְהוֹן. 
כה,לב וְשִׁשָּׁ֣ה קָנִ֔ים יֹֽצְאִ֖ים מִצִּדֶּ֑יהָ שְׁלֹשָׁ֣ה ׀ קְנֵ֣י מְנֹרָ֗ה מִצִּדָּהּ֙ הָֽאֶחָ֔ד וּשְׁלֹשָׁה֙ קְנֵ֣י מְנֹרָ֔ה מִצִּדָּ֖הּ הַשֵּׁנִֽי׃ וְשִׁתָּא קְנִין, נָפְקִין מִסִּטְרַהָא:  תְּלָתָא קְנֵי מְנָרְתָא, מִסִּטְרַהּ חַד, וּתְלָתָא קְנֵי מְנָרְתָא, מִסִּטְרַהּ תִּנְיָנָא. 
כה,לג שְׁלֹשָׁ֣ה גְ֠בִעִים מְֽשֻׁקָּדִ֞ים בַּקָּנֶ֣ה הָֽאֶחָד֮ כַּפְתֹּ֣ר וָפֶרַח֒ וּשְׁלֹשָׁ֣ה גְבִעִ֗ים מְשֻׁקָּדִ֛ים בַּקָּנֶ֥ה הָֽאֶחָ֖ד כַּפְתֹּ֣ר וָפָ֑רַח כֵּ֚ן לְשֵׁ֣שֶׁת הַקָּנִ֔ים הַיֹּֽצְאִ֖ים מִן־הַמְּנֹרָֽה׃ תְּלָתָא כַּלִּידִין מְצָיְרִין בְּקַנְיָא חַד, חַזּוּר וְשׁוֹשָׁן, וּתְלָתָא כַּלִּידִין מְצָיְרִין בְּקַנְיָא חַד, חַזּוּר וְשׁוֹשָׁן; כֵּין לְשִׁתָּא קְנִין, דְּנָפְקִין מִן מְנָרְתָא. 
כה,לד וּבַמְּנֹרָ֖ה אַרְבָּעָ֣ה גְבִעִ֑ים מְשֻׁ֨קָּדִ֔ים כַּפְתֹּרֶ֖יהָ וּפְרָחֶֽיהָ׃ וּבִמְנָרְתָא, אַרְבְּעָא כַּלִּידִין:  מְצָיְרִין--חַזּוּרַהָא, וְשׁוֹשַׁנַּהָא. 
כה,לה וְכַפְתֹּ֡ר תַּחַת֩ שְׁנֵ֨י הַקָּנִ֜ים מִמֶּ֗נָּה וְכַפְתֹּר֙ תַּ֣חַת שְׁנֵ֤י הַקָּנִים֙ מִמֶּ֔נָּה וְכַפְתֹּ֕ר תַּֽחַת־שְׁנֵ֥י הַקָּנִ֖ים מִמֶּ֑נָּה לְשֵׁ֨שֶׁת֙ הַקָּנִ֔ים הַיֹּֽצְאִ֖ים מִן־הַמְּנֹרָֽה׃ וְחַזּוּר תְּחוֹת תְּרֵין קְנִין דְּמִנַּהּ, וְחַזּוּר תְּחוֹת תְּרֵין קְנִין דְּמִנַּהּ, וְחַזּוּר, תְּחוֹת תְּרֵין קְנִין דְּמִנַּהּ--לְשִׁתָּא, קְנִין, דְּנָפְקִין, מִן מְנָרְתָא. 
כה,לו כַּפְתֹּֽרֵיהֶ֥ם וּקְנֹתָ֖ם מִמֶּ֣נָּה יִֽהְי֑וּ כֻּלָּ֛הּ מִקְשָׁ֥ה אַחַ֖ת זָהָ֥ב טָהֽוֹר׃ חַזּוּרֵיהוֹן וּקְנֵיהוֹן, מִנַּהּ יְהוֹן; כֻּלַּהּ נְגִידָא חֲדָא, דִּדְהַב דְּכֵי. 
כה,לז וְעָשִׂ֥יתָ אֶת־נֵֽרֹתֶ֖יהָ שִׁבְעָ֑ה וְהֶֽעֱלָה֙ אֶת־נֵ֣רֹתֶ֔יהָ וְהֵאִ֖יר עַל־עֵ֥בֶר פָּנֶֽיהָ׃ וְתַעֲבֵיד יָת בּוֹצִינַהָא, שִׁבְעָא; וְתַדְלֵיק, יָת בּוֹצִינַהָא, וִיהוֹן מְנָהֲרִין, לָקֳבֵיל אַפַּהָא. 
כה,לח וּמַלְקָחֶ֥יהָ וּמַחְתֹּתֶ֖יהָ זָהָ֥ב טָהֽוֹר׃ וְצֵיבְתַהָא וּמַחְתְּיָתַהָא, דִּדְהַב דְּכֵי. 
כה,לט כִּכָּ֛ר זָהָ֥ב טָה֖וֹר יַֽעֲשֶׂ֣ה אֹתָ֑הּ אֵ֥ת כָּל־הַכֵּלִ֖ים הָאֵֽלֶּה׃ כַּכְּרָא דְּדַהְבָּא דָּכְיָא, יַעֲבֵיד יָתַהּ--יָת כָּל מָנַיָּא, הָאִלֵּין. 
כה,מ וּרְאֵ֖ה וַֽעֲשֵׂ֑ה בְּתַ֨בְנִיתָ֔ם אֲשֶׁר־אַתָּ֥ה מָרְאֶ֖ה בָּהָֽר׃ {ס} וַחְזִי, וַעֲבֵיד:  בִּדְמוּתְהוֹן--דְּאַתְּ מִתַּחְזֵי, בְּטוּרָא. {ס}
כו,א וְאֶת־הַמִּשְׁכָּ֥ן תַּֽעֲשֶׂ֖ה עֶ֣שֶׂר יְרִיעֹ֑ת שֵׁ֣שׁ מָשְׁזָ֗ר וּתְכֵ֤לֶת וְאַרְגָּמָן֙ וְתֹלַ֣עַת שָׁנִ֔י כְּרֻבִ֛ים מַֽעֲשֵׂ֥ה חֹשֵׁ֖ב תַּֽעֲשֶׂ֥ה אֹתָֽם׃ וְיָת מַשְׁכְּנָא תַּעֲבֵיד, עֲסַר יְרִיעָן:  דְּבוּץ שְׁזִיר, וְתַכְלָא וְאַרְגְּוָנָא וּצְבַע זְהוֹרִי--צוּרַת כְּרוּבִין עוֹבָד אֻמָּן, תַּעֲבֵיד יָתְהוֹן. 
כו,ב אֹ֣רֶךְ ׀ הַיְרִיעָ֣ה הָֽאַחַ֗ת שְׁמֹנֶ֤ה וְעֶשְׂרִים֙ בָּֽאַמָּ֔ה וְרֹ֨חַב֙ אַרְבַּ֣ע בָּֽאַמָּ֔ה הַיְרִיעָ֖ה הָֽאֶחָ֑ת מִדָּ֥ה אַחַ֖ת לְכָל־הַיְרִיעֹֽת׃ אֻרְכָּא דִּירִיעֲתָא חֲדָא, עַסְרִין וְתַמְנֵי אַמִּין, וּפֻתְיָא אַרְבַּע אַמִּין, דִּירִיעֲתָא חֲדָא; מִשְׁחֲתָא חֲדָא, לְכָל יְרִיעָתָא. 
כו,ג חֲמֵ֣שׁ הַיְרִיעֹ֗ת תִּֽהְיֶ֨יןָ֙ חֹֽבְרֹ֔ת אִשָּׁ֖ה אֶל־אֲחֹתָ֑הּ וְחָמֵ֤שׁ יְרִיעֹת֙ חֹֽבְרֹ֔ת אִשָּׁ֖ה אֶל־אֲחֹתָֽהּ׃ חֲמֵישׁ יְרִיעָן, יִהְוְיָן מְלָפְפָן, חֲדָא, עִם חֲדָא; וַחֲמֵישׁ יְרִיעָן מְלָפְפָן, חֲדָא עִם חֲדָא. 
כו,ד וְעָשִׂ֜יתָ לֻֽלְאֹ֣ת תְּכֵ֗לֶת עַ֣ל שְׂפַ֤ת הַיְרִיעָה֙ הָֽאֶחָ֔ת מִקָּצָ֖ה בַּֽחֹבָ֑רֶת וְכֵ֤ן תַּֽעֲשֶׂה֙ בִּשְׂפַ֣ת הַיְרִיעָ֔ה הַקִּ֣יצוֹנָ֔ה בַּמַּחְבֶּ֖רֶת הַשֵּׁנִֽית׃ וְתַעֲבֵיד עֲנֻבִּין דְּתַכְלָא, עַל סִפְתָּא דִּירִיעֲתָא חֲדָא, מִסִּטְרָא, בֵּית לוֹפֵי; וְכֵין תַּעֲבֵיד, בְּסִפְתָּא דִּירִיעֲתָא, בְּסִטְרָא, בֵּית לוֹפֵי תִּנְיָנָא. 
כו,ה חֲמִשִּׁ֣ים לֻֽלָאֹ֗ת תַּֽעֲשֶׂה֮ בַּיְרִיעָ֣ה הָֽאֶחָת֒ וַֽחֲמִשִּׁ֣ים לֻֽלָאֹ֗ת תַּֽעֲשֶׂה֙ בִּקְצֵ֣ה הַיְרִיעָ֔ה אֲשֶׁ֖ר בַּמַּחְבֶּ֣רֶת הַשֵּׁנִ֑ית מַקְבִּילֹת֙ הַלֻּ֣לָאֹ֔ת אִשָּׁ֖ה אֶל־אֲחֹתָֽהּ׃ חַמְשִׁין עֲנֻבִּין, תַּעֲבֵיד בִּירִיעֲתָא חֲדָא, וְחַמְשִׁין עֲנֻבִּין תַּעֲבֵיד בְּסִטְרָא דִּירִיעֲתָא, דְּבֵית לוֹפֵי תִּנְיָנָא:  מַכְוְנָן, עֲנֻבַּיָּא, חֲדָא, לָקֳבֵיל חֲדָא. 
כו,ו וְעָשִׂ֕יתָ חֲמִשִּׁ֖ים קַרְסֵ֣י זָהָ֑ב וְחִבַּרְתָּ֨ אֶת־הַיְרִיעֹ֜ת אִשָּׁ֤ה אֶל־אֲחֹתָהּ֙ בַּקְּרָסִ֔ים וְהָיָ֥ה הַמִּשְׁכָּ֖ן אֶחָֽד׃ וְתַעֲבֵיד, חַמְשִׁין פֻּרְפִין דִּדְהַב; וּתְלָפֵיף יָת יְרִיעָתָא חֲדָא עִם חֲדָא, בְּפֻרְפַיָּא, וִיהֵי מַשְׁכְּנָא, חַד. 
כו,ז וְעָשִׂ֨יתָ֙ יְרִיעֹ֣ת עִזִּ֔ים לְאֹ֖הֶל עַל־הַמִּשְׁכָּ֑ן עַשְׁתֵּי־עֶשְׂרֵ֥ה יְרִיעֹ֖ת תַּֽעֲשֶׂ֥ה אֹתָֽם׃ וְתַעֲבֵיד יְרִיעָן דְּמַעְזֵי, לִפְרָסָא עַל מַשְׁכְּנָא; חֲדָא עַסְרֵי יְרִיעָן, תַּעֲבֵיד יָתְהוֹן. 
כו,ח אֹ֣רֶךְ ׀ הַיְרִיעָ֣ה הָֽאַחַ֗ת שְׁלֹשִׁים֙ בָּֽאַמָּ֔ה וְרֹ֨חַב֙ אַרְבַּ֣ע בָּֽאַמָּ֔ה הַיְרִיעָ֖ה הָֽאֶחָ֑ת מִדָּ֣ה אַחַ֔ת לְעַשְׁתֵּ֥י עֶשְׂרֵ֖ה יְרִיעֹֽת׃ אֻרְכָּא דִּירִיעֲתָא חֲדָא, תְּלָתִין אַמִּין, וּפֻתְיָא אַרְבַּע אַמִּין, דִּירִיעֲתָא חֲדָא; מִשְׁחֲתָא חֲדָא, לַחֲדָא עַסְרֵי יְרִיעָן. 
כו,ט וְחִבַּרְתָּ֞ אֶת־חֲמֵ֤שׁ הַיְרִיעֹת֙ לְבָ֔ד וְאֶת־שֵׁ֥שׁ הַיְרִיעֹ֖ת לְבָ֑ד וְכָֽפַלְתָּ֙ אֶת־הַיְרִיעָ֣ה הַשִּׁשִּׁ֔ית אֶל־מ֖וּל פְּנֵ֥י הָאֹֽהֶל׃ וּתְלָפֵיף יָת חֲמֵישׁ יְרִיעָן, לְחוֹד, וְיָת שֵׁית יְרִיעָן, לְחוֹד; וְתַעֵיף יָת יְרִיעֲתָא שְׁתִיתֵיתָא, לָקֳבֵיל אַפֵּי מַשְׁכְּנָא. 
כו,י וְעָשִׂ֜יתָ חֲמִשִּׁ֣ים לֻֽלָאֹ֗ת עַ֣ל שְׂפַ֤ת הַיְרִיעָה֙ הָֽאֶחָ֔ת הַקִּֽיצֹנָ֖ה בַּֽחֹבָ֑רֶת וַֽחֲמִשִּׁ֣ים לֻֽלָאֹ֗ת עַ֚ל שְׂפַ֣ת הַיְרִיעָ֔ה הַֽחֹבֶ֖רֶת הַשֵּׁנִֽית׃ וְתַעֲבֵיד חַמְשִׁין עֲנֻבִּין, עַל סִפְתָּא דִּירִיעֲתָא חֲדָא, בְּסִטְרָא, בֵּית לוֹפֵי; וְחַמְשִׁין עֲנֻבִּין, עַל סִפְתָּא דִּירִיעֲתָא, דְּבֵית לוֹפֵי, תִּנְיָנָא. 
כו,יא וְעָשִׂ֛יתָ קַרְסֵ֥י נְחֹ֖שֶׁת חֲמִשִּׁ֑ים וְהֵֽבֵאתָ֤ אֶת־הַקְּרָסִים֙ בַּלֻּ֣לָאֹ֔ת וְחִבַּרְתָּ֥ אֶת־הָאֹ֖הֶל וְהָיָ֥ה אֶחָֽד׃ וְתַעֲבֵיד פֻּרְפִין דִּנְחָשׁ, חַמְשִׁין; וְתַעֵיל יָת פֻּרְפַיָּא בַּעֲנֻבַּיָּא, וּתְלָפֵיף יָת מַשְׁכְּנָא וִיהֵי חַד. 
כו,יב וְסֶ֨רַח֙ הָֽעֹדֵ֔ף בִּֽירִיעֹ֖ת הָאֹ֑הֶל חֲצִ֤י הַיְרִיעָה֙ הָֽעֹדֶ֔פֶת תִּסְרַ֕ח עַ֖ל אֲחֹרֵ֥י הַמִּשְׁכָּֽן׃ וְסִרְחָא, דְּיַתִּיר, בִּירִיעָת, מַשְׁכְּנָא--פַּלְגוּת יְרִיעֲתָא, דְּיָתְרָא, תִּסְרַח, עַל אֲחוֹרֵי מַשְׁכְּנָא. 
כו,יג וְהָֽאַמָּ֨ה מִזֶּ֜ה וְהָֽאַמָּ֤ה מִזֶּה֙ בָּֽעֹדֵ֔ף בְּאֹ֖רֶךְ יְרִיעֹ֣ת הָאֹ֑הֶל יִֽהְיֶ֨ה סָר֜וּחַ עַל־צִדֵּ֧י הַמִּשְׁכָּ֛ן מִזֶּ֥ה וּמִזֶּ֖ה לְכַסֹּתֽוֹ׃ וְאַמְּתָא מִכָּא וְאַמְּתָא מִכָּא, בִּדְיַתִּיר, בְּאֹרֶךְ, יְרִיעָת מַשְׁכְּנָא:  יְהֵי סְרִיחַ עַל סִטְרֵי מַשְׁכְּנָא, מִכָּא וּמִכָּא--לְכַסָּיוּתֵיהּ. 
כו,יד וְעָשִׂ֤יתָ מִכְסֶה֙ לָאֹ֔הֶל עֹרֹ֥ת אֵילִ֖ם מְאָדָּמִ֑ים וּמִכְסֵ֛ה עֹרֹ֥ת תְּחָשִׁ֖ים מִלְמָֽעְלָה׃ {פ} וְתַעֲבֵיד חוּפָאָה לְמַשְׁכְּנָא, דְּמַשְׁכֵּי דִּכְרֵי מְסֻמְּקֵי, וְחוּפָאָה דְּמַשְׁכֵּי סָסְגוֹנָא, מִלְּעֵילָא. {פ}
כו,טו וְעָשִׂ֥יתָ אֶת־הַקְּרָשִׁ֖ים לַמִּשְׁכָּ֑ן עֲצֵ֥י שִׁטִּ֖ים עֹֽמְדִֽים׃ וְתַעֲבֵיד יָת דַּפַּיָּא, לְמַשְׁכְּנָא, דְּאָעֵי שִׁטִּין, קָיְמִין. 
כו,טז עֶ֥שֶׂר אַמּ֖וֹת אֹ֣רֶךְ הַקָּ֑רֶשׁ וְאַמָּה֙ וַֽחֲצִ֣י הָֽאַמָּ֔ה רֹ֖חַב הַקֶּ֥רֶשׁ הָֽאֶחָֽד׃ עֲסַר אַמִּין, אֻרְכָּא דְּדַפָּא; וְאַמְּתָא וּפַלְגוּת אַמְּתָא, פֻּתְיָא דְּדַפָּא חַד. 
כו,יז שְׁתֵּ֣י יָד֗וֹת לַקֶּ֨רֶשׁ֙ הָֽאֶחָ֔ד מְשֻׁ֨לָּבֹ֔ת אִשָּׁ֖ה אֶל־אֲחֹתָ֑הּ כֵּ֣ן תַּֽעֲשֶׂ֔ה לְכֹ֖ל קַרְשֵׁ֥י הַמִּשְׁכָּֽן׃ תְּרֵין צִירִין, לְדַפָּא חַד--מְשֻׁלְּבִין, חַד לָקֳבֵיל חַד; כֵּין תַּעֲבֵיד, לְכֹל דַּפֵּי מַשְׁכְּנָא. 
כו,יח וְעָשִׂ֥יתָ אֶת־הַקְּרָשִׁ֖ים לַמִּשְׁכָּ֑ן עֶשְׂרִ֣ים קֶ֔רֶשׁ לִפְאַ֖ת נֶ֥גְבָּה תֵימָֽנָה׃ וְתַעֲבֵיד יָת דַּפַּיָּא, לְמַשְׁכְּנָא, עַסְרִין דַּפִּין, לְרוּחַ עֵיבַר דָּרוֹמָא. 
כו,יט וְאַרְבָּעִים֙ אַדְנֵי־כֶ֔סֶף תַּֽעֲשֶׂ֕ה תַּ֖חַת עֶשְׂרִ֣ים הַקָּ֑רֶשׁ שְׁנֵ֨י אֲדָנִ֜ים תַּֽחַת־הַקֶּ֤רֶשׁ הָֽאֶחָד֙ לִשְׁתֵּ֣י יְדֹתָ֔יו וּשְׁנֵ֧י אֲדָנִ֛ים תַּֽחַת־הַקֶּ֥רֶשׁ הָֽאֶחָ֖ד לִשְׁתֵּ֥י יְדֹתָֽיו׃ וְאַרְבְּעִין, סָמְכִין דִּכְסַף, תַּעֲבֵיד, תְּחוֹת עַסְרִין דַּפִּין:  תְּרֵין סָמְכִין תְּחוֹת דַּפָּא חַד, לִתְרֵין צִירוֹהִי, וּתְרֵין סָמְכִין תְּחוֹת דַּפָּא חַד, לִתְרֵין צִירוֹהִי. 
כו,כ וּלְצֶ֧לַע הַמִּשְׁכָּ֛ן הַשֵּׁנִ֖ית לִפְאַ֣ת צָפ֑וֹן עֶשְׂרִ֖ים קָֽרֶשׁ׃ וְלִסְטַר מַשְׁכְּנָא תִּנְיָנָא, לְרוּחַ צִפּוּנָא, עַסְרִין, דַּפִּין. 
כו,כא וְאַרְבָּעִ֥ים אַדְנֵיהֶ֖ם כָּ֑סֶף שְׁנֵ֣י אֲדָנִ֗ים תַּ֚חַת הַקֶּ֣רֶשׁ הָֽאֶחָ֔ד וּשְׁנֵ֣י אֲדָנִ֔ים תַּ֖חַת הַקֶּ֥רֶשׁ הָֽאֶחָֽד׃ וְאַרְבְּעִין סָמְכֵיהוֹן, דִּכְסַף:  תְּרֵין סָמְכִין, תְּחוֹת דַּפָּא חַד, וּתְרֵין סָמְכִין, תְּחוֹת דַּפָּא חַד. 
כו,כב וּֽלְיַרְכְּתֵ֥י הַמִּשְׁכָּ֖ן יָ֑מָּה תַּֽעֲשֶׂ֖ה שִׁשָּׁ֥ה קְרָשִֽׁים׃ וְלִסְיָפֵי מַשְׁכְּנָא, מַעְרְבָא, תַּעֲבֵיד, שִׁתָּא דַּפִּין. 
כו,כג וּשְׁנֵ֤י קְרָשִׁים֙ תַּֽעֲשֶׂ֔ה לִמְקֻצְעֹ֖ת הַמִּשְׁכָּ֑ן בַּיַּרְכָתָֽיִם׃ וּתְרֵין דַּפִּין תַּעֲבֵיד, לְזָוְיָת מַשְׁכְּנָא--בְּסוֹפְהוֹן. 
כו,כד וְיִֽהְי֣וּ תֹֽאֲמִם֮ מִלְּמַטָּה֒ וְיַחְדָּ֗ו יִֽהְי֤וּ תַמִּים֙ עַל־רֹאשׁ֔וֹ אֶל־הַטַּבַּ֖עַת הָֽאֶחָ֑ת כֵּ֚ן יִֽהְיֶ֣ה לִשְׁנֵיהֶ֔ם לִשְׁנֵ֥י הַמִּקְצֹעֹ֖ת יִֽהְיֽוּ׃ וִיהוֹן מַכְוְנִין, מִלְּרַע, וְכַחְדָּא יְהוֹן מַכְוְנִין עַל רֵישֵׁיהוֹן, לְעִזְקְתָא חֲדָא; כֵּין יְהֵי לְתַרְוֵיהוֹן, לְתַרְתֵּין זָוְיָן יְהוֹן. 
כו,כה וְהָיוּ֙ שְׁמֹנָ֣ה קְרָשִׁ֔ים וְאַדְנֵיהֶ֣ם כֶּ֔סֶף שִׁשָּׁ֥ה עָשָׂ֖ר אֲדָנִ֑ים שְׁנֵ֣י אֲדָנִ֗ים תַּ֚חַת הַקֶּ֣רֶשׁ הָֽאֶחָ֔ד וּשְׁנֵ֣י אֲדָנִ֔ים תַּ֖חַת הַקֶּ֥רֶשׁ הָֽאֶחָֽד׃ וִיהוֹן, תְּמָנְיָא דַּפִּין, וְסָמְכֵיהוֹן דִּכְסַף, שִׁתַּת עֲסַר סָמְכִין:  תְּרֵין סָמְכִין, תְּחוֹת דַּפָּא חַד, וּתְרֵין סָמְכִין, תְּחוֹת דַּפָּא חַד. 
כו,כו וְעָשִׂ֥יתָ בְרִיחִ֖ם עֲצֵ֣י שִׁטִּ֑ים חֲמִשָּׁ֕ה לְקַרְשֵׁ֥י צֶֽלַע־הַמִּשְׁכָּ֖ן הָֽאֶחָֽד׃ וְתַעֲבֵיד עָבְרֵי, דְּאָעֵי שִׁטִּין, חַמְשָׁא, לְדַפֵּי סְטַר מַשְׁכְּנָא חַד. 
כו,כז וַֽחֲמִשָּׁ֣ה בְרִיחִ֔ם לְקַרְשֵׁ֥י צֶֽלַע־הַמִּשְׁכָּ֖ן הַשֵּׁנִ֑ית וַֽחֲמִשָּׁ֣ה בְרִיחִ֗ם לְקַרְשֵׁי֙ צֶ֣לַע הַמִּשְׁכָּ֔ן לַיַּרְכָתַ֖יִם יָֽמָּה׃ וְחַמְשָׁא עָבְרִין, לְדַפֵּי סְטַר מַשְׁכְּנָא תִּנְיָנָא; וְחַמְשָׁא עָבְרִין, לְדַפֵּי סְטַר מַשְׁכְּנָא, לְסוֹפְהוֹן, מַעְרְבָא. 
כו,כח וְהַבְּרִ֥יחַ הַתִּיכֹ֖ן בְּת֣וֹךְ הַקְּרָשִׁ֑ים מַבְרִ֕חַ מִן־הַקָּצֶ֖ה אֶל־הַקָּצֶֽה׃ וְעָבְרָא מְצִיעָאָה, בְּגוֹ דַּפַּיָּא, מַעְבַּר, מִן סְיָפֵי לִסְיָפֵי. 
כו,כט וְֽאֶת־הַקְּרָשִׁ֞ים תְּצַפֶּ֣ה זָהָ֗ב וְאֶת־טַבְּעֹֽתֵיהֶם֙ תַּֽעֲשֶׂ֣ה זָהָ֔ב בָּתִּ֖ים לַבְּרִיחִ֑ם וְצִפִּיתָ֥ אֶת־הַבְּרִיחִ֖ם זָהָֽב׃ וְיָת דַּפַּיָּא תִּחְפֵי דַּהְבָּא, וְיָת עִזְקָתְהוֹן תַּעֲבֵיד דַּהְבָּא--אַתְרָא, לְעָבְרַיָּא; וְתִחְפֵי יָת עָבְרַיָּא, דַּהְבָּא. 
כו,ל וַהֲקֵֽמֹתָ֖ אֶת־הַמִּשְׁכָּ֑ן כְּמִ֨שְׁפָּט֔וֹ אֲשֶׁ֥ר הָרְאֵ֖יתָ בָּהָֽר׃ {ס} וּתְקִים, יָת מַשְׁכְּנָא:  כְּהִלְכְּתֵיהּ--דְּאִתַּחְזִיתָא, בְּטוּרָא. {ס}
כו,לא וְעָשִׂ֣יתָ פָרֹ֗כֶת תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֣שׁ מָשְׁזָ֑ר מַֽעֲשֵׂ֥ה חֹשֵׁ֛ב יַֽעֲשֶׂ֥ה אֹתָ֖הּ כְּרֻבִֽים׃ וְתַעֲבֵיד פָּרֻכְתָּא, דְּתַכְלָא וְאַרְגְּוָנָא וּצְבַע זְהוֹרִי--וּבוּץ שְׁזִיר; עוֹבָד אֻמָּן יַעֲבֵיד יָתַהּ, צוּרַת כְּרוּבִין. 
כו,לב וְנָֽתַתָּ֣ה אֹתָ֗הּ עַל־אַרְבָּעָה֙ עַמּוּדֵ֣י שִׁטִּ֔ים מְצֻפִּ֣ים זָהָ֔ב וָֽוֵיהֶ֖ם זָהָ֑ב עַל־אַרְבָּעָ֖ה אַדְנֵי־כָֽסֶף׃ וְתִתֵּין יָתַהּ, עַל אַרְבְּעָא עַמּוּדֵי שִׁטִּין, מְחֻפַּן דַּהְבָּא, וָוֵיהוֹן דַּהְבָּא--עַל אַרְבְּעָא, סָמְכִין דִּכְסַף. 
כו,לג וְנָֽתַתָּ֣ה אֶת־הַפָּרֹכֶת֮ תַּ֣חַת הַקְּרָסִים֒ וְהֵֽבֵאתָ֥ שָׁ֨מָּה֙ מִבֵּ֣ית לַפָּרֹ֔כֶת אֵ֖ת אֲר֣וֹן הָֽעֵד֑וּת וְהִבְדִּילָ֤ה הַפָּרֹ֨כֶת֙ לָכֶ֔ם בֵּ֣ין הַקֹּ֔דֶשׁ וּבֵ֖ין קֹ֥דֶשׁ הַקֳּדָשִֽׁים׃ וְתִתֵּין יָת פָּרֻכְתָּא, תְּחוֹת פֻּרְפַיָּא, וְתַעֵיל לְתַמָּן מִגָּיו לְפָרֻכְתָּא, יָת אֲרוֹנָא דְּסָהֲדוּתָא; וְתַפְרֵישׁ פָּרֻכְתָּא, לְכוֹן, בֵּין קֻדְשָׁא, וּבֵין קֹדֶשׁ קֻדְשַׁיָּא. 
כו,לד וְנָֽתַתָּ֙ אֶת־הַכַּפֹּ֔רֶת עַ֖ל אֲר֣וֹן הָֽעֵדֻ֑ת בְּקֹ֖דֶשׁ הַקֳּדָשִֽׁים׃ וְתִתֵּין, יָת כָּפֻרְתָּא, עַל, אֲרוֹנָא דְּסָהֲדוּתָא--בְּקֹדֶשׁ, קֻדְשַׁיָּא. 
כו,לה וְשַׂמְתָּ֤ אֶת־הַשֻּׁלְחָן֙ מִח֣וּץ לַפָּרֹ֔כֶת וְאֶת־הַמְּנֹרָה֙ נֹ֣כַח הַשֻּׁלְחָ֔ן עַ֛ל צֶ֥לַע הַמִּשְׁכָּ֖ן תֵּימָ֑נָה וְהַ֨שֻּׁלְחָ֔ן תִּתֵּ֖ן עַל־צֶ֥לַע צָפֽוֹן׃ וּתְשַׁוֵּי יָת פָּתוּרָא, מִבַּרָא לְפָרֻכְתָּא, וְיָת מְנָרְתָא לָקֳבֵיל פָּתוּרָא, עַל סְטַר מַשְׁכְּנָא דָּרוֹמָא; וּפָתוּרָא--תִּתֵּין, עַל סְטַר צִפּוּנָא. 
כו,לו וְעָשִׂ֤יתָ מָסָךְ֙ לְפֶ֣תַח הָאֹ֔הֶל תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֣שׁ מָשְׁזָ֑ר מַֽעֲשֵׂ֖ה רֹקֵֽם׃ וְתַעֲבֵיד פְּרָסָא לִתְרַע מַשְׁכְּנָא, תַּכְלָא וְאַרְגְּוָנָא וּצְבַע זְהוֹרִי וּבוּץ שְׁזִיר, עוֹבָד, צַיָּר. 
כו,לז וְעָשִׂ֣יתָ לַמָּסָ֗ךְ חֲמִשָּׁה֙ עַמּוּדֵ֣י שִׁטִּ֔ים וְצִפִּיתָ֤ אֹתָם֙ זָהָ֔ב וָֽוֵיהֶ֖ם זָהָ֑ב וְיָֽצַקְתָּ֣ לָהֶ֔ם חֲמִשָּׁ֖ה אַדְנֵ֥י נְחֹֽשֶׁת׃ {ס} וְתַעֲבֵיד לִפְרָסָא, חַמְשָׁא עַמּוּדֵי שִׁטִּין, וְתִחְפֵי יָתְהוֹן דַּהְבָּא, וָוֵיהוֹן דַּהְבָּא; וְתַתֵּיךְ לְהוֹן, חַמְשָׁא סָמְכִין דִּנְחָשָׁא. {ס}
כז,א וְעָשִׂ֥יתָ אֶת־הַמִּזְבֵּ֖חַ עֲצֵ֣י שִׁטִּ֑ים חָמֵשׁ֩ אַמּ֨וֹת אֹ֜רֶךְ וְחָמֵ֧שׁ אַמּ֣וֹת רֹ֗חַב רָב֤וּעַ יִֽהְיֶה֙ הַמִּזְבֵּ֔חַ וְשָׁלֹ֥שׁ אַמּ֖וֹת קֹֽמָתֽוֹ׃ וְתַעֲבֵיד יָת מַדְבְּחָא, דְּאָעֵי שִׁטִּין:  חֲמֵישׁ אַמִּין אֻרְכָּא וַחֲמֵישׁ אַמִּין פֻּתְיָא, מְרֻבַּע יְהֵי מַדְבְּחָא, וּתְלָת אַמִּין, רוּמֵיהּ. 
כז,ב וְעָשִׂ֣יתָ קַרְנֹתָ֗יו עַ֚ל אַרְבַּ֣ע פִּנֹּתָ֔יו מִמֶּ֖נּוּ תִּֽהְיֶ֣יןָ קַרְנֹתָ֑יו וְצִפִּיתָ֥ אֹת֖וֹ נְחֹֽשֶׁת׃ וְתַעֲבֵיד קַרְנוֹהִי, עַל אַרְבַּע זָוְיָתֵיהּ--מִנֵּיהּ, יִהְוְיָן קַרְנוֹהִי; וְתִחְפֵי יָתֵיהּ, נְחָשָׁא. 
כז,ג וְעָשִׂ֤יתָ סִּֽירֹתָיו֙ לְדַשְּׁנ֔וֹ וְיָעָיו֙ וּמִזְרְקֹתָ֔יו וּמִזְלְגֹתָ֖יו וּמַחְתֹּתָ֑יו לְכָל־כֵּלָ֖יו תַּֽעֲשֶׂ֥ה נְחֹֽשֶׁת׃ וְתַעֲבֵיד פְּסַכְתֵּירְוָתֵיהּ, לְמִסְפֵּי קִטְמֵיהּ, וּמַגְרוֹפְיָתֵיהּ וּמִזְרְקוֹהִי, וְצִנּוֹרְיָתֵיהּ וּמַחְתְּיָתֵיהּ; לְכָל מָנוֹהִי, תַּעֲבֵיד נְחָשָׁא. 
כז,ד וְעָשִׂ֤יתָ לּוֹ֙ מִכְבָּ֔ר מַֽעֲשֵׂ֖ה רֶ֣שֶׁת נְחֹ֑שֶׁת וְעָשִׂ֣יתָ עַל־הָרֶ֗שֶׁת אַרְבַּע֙ טַבְּעֹ֣ת נְחֹ֔שֶׁת עַ֖ל אַרְבַּ֥ע קְצוֹתָֽיו׃ וְתַעֲבֵיד לֵיהּ סְרָדָא, עוֹבָד מְצָדְתָא דִּנְחָשָׁא; וְתַעֲבֵיד עַל מְצָדְתָא, אַרְבַּע עִזְקָן דִּנְחָשָׁא, עַל, אַרְבְּעָא סִטְרוֹהִי. 
כז,ה וְנָֽתַתָּ֣ה אֹתָ֗הּ תַּ֛חַת כַּרְכֹּ֥ב הַמִּזְבֵּ֖חַ מִלְּמָ֑טָּה וְהָֽיְתָ֣ה הָרֶ֔שֶׁת עַ֖ד חֲצִ֥י הַמִּזְבֵּֽחַ׃ וְתִתֵּין יָתַהּ, תְּחוֹת סוֹבֵיבָא דְּמַדְבְּחָא--מִלְּרַע; וּתְהֵי מְצָדְתָא, עַד פַּלְגוּת מַדְבְּחָא. 
כז,ו וְעָשִׂ֤יתָ בַדִּים֙ לַמִּזְבֵּ֔חַ בַּדֵּ֖י עֲצֵ֣י שִׁטִּ֑ים וְצִפִּיתָ֥ אֹתָ֖ם נְחֹֽשֶׁת׃ וְתַעֲבֵיד אֲרִיחַיָּא לְמַדְבְּחָא, אֲרִיחֵי דְּאָעֵי שִׁטִּין; וְתִחְפֵי יָתְהוֹן, נְחָשָׁא. 
כז,ז וְהוּבָ֥א אֶת־בַּדָּ֖יו בַּטַּבָּעֹ֑ת וְהָי֣וּ הַבַּדִּ֗ים עַל־שְׁתֵּ֛י צַלְעֹ֥ת הַמִּזְבֵּ֖חַ בִּשְׂאֵ֥ת אֹתֽוֹ׃ וְיַעֵיל יָת אֲרִיחוֹהִי, בְּעִזְקָתָא; וִיהוֹן אֲרִיחַיָּא, עַל תְּרֵין סִטְרֵי מַדְבְּחָא--בְּמִטַּל יָתֵיהּ. 
כז,ח נְב֥וּב לֻחֹ֖ת תַּֽעֲשֶׂ֣ה אֹת֑וֹ כַּֽאֲשֶׁ֨ר הֶרְאָ֥ה אֹֽתְךָ֛ בָּהָ֖ר כֵּ֥ן יַֽעֲשֽׂוּ׃ {ס} חֲלִיל לוּחִין, תַּעֲבֵיד יָתֵיהּ:  כְּמָא דְּאַחְזִי יָתָךְ בְּטוּרָא, כֵּין יַעְבְּדוּן. {ס}
כז,ט וְעָשִׂ֕יתָ אֵ֖ת חֲצַ֣ר הַמִּשְׁכָּ֑ן לִפְאַ֣ת נֶֽגֶב־תֵּ֠ימָנָה קְלָעִ֨ים לֶֽחָצֵ֜ר שֵׁ֣שׁ מָשְׁזָ֗ר מֵאָ֤ה בָֽאַמָּה֙ אֹ֔רֶךְ לַפֵּאָ֖ה הָֽאֶחָֽת׃ וְתַעֲבֵיד, יָת דָּרַת מַשְׁכְּנָא--לְרוּחַ עֵיבַר דָּרוֹמָא סְרָדֵי לְדָרְתָא דְּבוּץ שְׁזִיר, מְאָה אַמִּין אֻרְכָּא, לְרוּחָא, חֲדָא. 
כז,י וְעַמֻּדָ֣יו עֶשְׂרִ֔ים וְאַדְנֵיהֶ֥ם עֶשְׂרִ֖ים נְחֹ֑שֶׁת וָוֵ֧י הָֽעַמֻּדִ֛ים וַחֲשֻֽׁקֵיהֶ֖ם כָּֽסֶף׃ וְעַמּוּדוֹהִי עַסְרִין, וְסָמְכֵיהוֹן עַסְרִין דִּנְחָשָׁא; וָוֵי עַמּוּדַיָּא וְכִבּוּשֵׁיהוֹן, כְּסַף. 
כז,יא וְכֵ֨ן לִפְאַ֤ת צָפוֹן֙ בָּאֹ֔רֶךְ קְלָעִ֖ים מֵ֣אָה אֹ֑רֶךְ וְעַמֻּדָ֣ו עֶשְׂרִ֗ים וְאַדְנֵיהֶ֤ם עֶשְׂרִים֙ נְחֹ֔שֶׁת וָוֵ֧י הָֽעַמֻּדִ֛ים וַחֲשֻֽׁקֵיהֶ֖ם כָּֽסֶף׃ וְכֵין לְרוּחַ צִפּוּנָא בְּאֻרְכָּא, סְרָדֵי מְאָה אֻרְכָּא; וְעַמּוּדוֹהִי עַסְרִין, וְסָמְכֵיהוֹן עַסְרִין דִּנְחָשָׁא, וָוֵי עַמּוּדַיָּא וְכִבּוּשֵׁיהוֹן, כְּסַף. 
כז,יב וְרֹ֤חַב הֶֽחָצֵר֙ לִפְאַת־יָ֔ם קְלָעִ֖ים חֲמִשִּׁ֣ים אַמָּ֑ה עַמֻּֽדֵיהֶ֣ם עֲשָׂרָ֔ה וְאַדְנֵיהֶ֖ם עֲשָׂרָֽה׃ וּפֻתְיָא דְּדָרְתָא לְרוּחַ מַעְרְבָא, סְרָדֵי חַמְשִׁין אַמִּין; עַמּוּדֵיהוֹן עַסְרָא, וְסָמְכֵיהוֹן עַסְרָא. 
כז,יג וְרֹ֣חַב הֶֽחָצֵ֗ר לִפְאַ֛ת קֵ֥דְמָה מִזְרָ֖חָה חֲמִשִּׁ֥ים אַמָּֽה׃ וּפֻתְיָא דְּדָרְתָא, לְרוּחַ קִדּוּמָא מַדְנְחָא--חַמְשִׁין אַמִּין. 
כז,יד וַֽחֲמֵ֨שׁ עֶשְׂרֵ֥ה אַמָּ֛ה קְלָעִ֖ים לַכָּתֵ֑ף עַמֻּֽדֵיהֶ֣ם שְׁלֹשָׁ֔ה וְאַדְנֵיהֶ֖ם שְׁלֹשָֽׁה׃ וַחֲמֵישׁ עַסְרֵי אַמִּין סְרָדֵי, לְעִבְרָא; עַמּוּדֵיהוֹן תְּלָתָא, וְסָמְכֵיהוֹן תְּלָתָא. 
כז,טו וְלַכָּתֵף֙ הַשֵּׁנִ֔ית חֲמֵ֥שׁ עֶשְׂרֵ֖ה קְלָעִ֑ים עַמֻּֽדֵיהֶ֣ם שְׁלֹשָׁ֔ה וְאַדְנֵיהֶ֖ם שְׁלֹשָֽׁה׃ וּלְעִבְרָא, תִּנְיָנָא--חֲמֵישׁ עַסְרֵי, סְרָדִין; עַמּוּדֵיהוֹן תְּלָתָא, וְסָמְכֵיהוֹן תְּלָתָא. 
כז,טז וּלְשַׁ֨עַר הֶֽחָצֵ֜ר מָסָ֣ךְ ׀ עֶשְׂרִ֣ים אַמָּ֗ה תְּכֵ֨לֶת וְאַרְגָּמָ֜ן וְתוֹלַ֧עַת שָׁנִ֛י וְשֵׁ֥שׁ מָשְׁזָ֖ר מַֽעֲשֵׂ֣ה רֹקֵ֑ם עַמֻּֽדֵיהֶם֙ אַרְבָּעָ֔ה וְאַדְנֵיהֶ֖ם אַרְבָּעָֽה׃ וְלִתְרַע דָּרְתָא פְּרָסָא עַסְרִין אַמִּין, דְּתַכְלָא וְאַרְגְּוָנָא וּצְבַע זְהוֹרִי וּבוּץ שְׁזִיר--עוֹבָד צַיָּר; עַמּוּדֵיהוֹן, אַרְבְּעָא, וְסָמְכֵיהוֹן, אַרְבְּעָא. 
כז,יז כָּל־עַמּוּדֵ֨י הֶֽחָצֵ֤ר סָבִיב֙ מְחֻשָּׁקִ֣ים כֶּ֔סֶף וָֽוֵיהֶ֖ם כָּ֑סֶף וְאַדְנֵיהֶ֖ם נְחֹֽשֶׁת׃ כָּל עַמּוּדֵי דָּרְתָא סְחוֹר סְחוֹר מְכֻבְּשִׁין כְּסַף, וָוֵיהוֹן כְּסַף; וְסָמְכֵיהוֹן, דִּנְחָשָׁא. 
כז,יח אֹ֣רֶךְ הֶֽחָצֵר֩ מֵאָ֨ה בָֽאַמָּ֜ה וְרֹ֣חַב ׀ חֲמִשִּׁ֣ים בַּֽחֲמִשִּׁ֗ים וְקֹמָ֛ה חָמֵ֥שׁ אַמּ֖וֹת שֵׁ֣שׁ מָשְׁזָ֑ר וְאַדְנֵיהֶ֖ם נְחֹֽשֶׁת׃ אֻרְכָּא דְּדָרְתָא מְאָה אַמִּין וּפֻתְיָא חַמְשִׁין בְּחַמְשִׁין, וְרוּמָא חֲמֵישׁ אַמִּין--דְּבוּץ שְׁזִיר; וְסָמְכֵיהוֹן, דִּנְחָשָׁא. 
כז,יט לְכֹל֙ כְּלֵ֣י הַמִּשְׁכָּ֔ן בְּכֹ֖ל עֲבֹֽדָת֑וֹ וְכָל־יְתֵֽדֹתָ֛יו וְכָל־יִתְדֹ֥ת הֶֽחָצֵ֖ר נְחֹֽשֶׁת׃ {ס} לְכֹל מָנֵי מַשְׁכְּנָא, בְּכֹל פֻּלְחָנֵיהּ, וְכָל סִכּוֹהִי וְכָל סִכֵּי דָּרְתָא, דִּנְחָשָׁא. {ס}
