מקץ

מא,א וַיְהִ֕י מִקֵּ֖ץ שְׁנָתַ֣יִם יָמִ֑ים וּפַרְעֹ֣ה חֹלֵ֔ם וְהִנֵּ֖ה עֹמֵ֥ד עַל־הַיְאֹֽר׃ וַהֲוָה, מִסּוֹף תַּרְתֵּין שְׁנִין; וּפַרְעֹה חָלֵים, וְהָא קָאֵים עַל נַהְרָא. 
מא,ב וְהִנֵּ֣ה מִן־הַיְאֹ֗ר עֹלֹת֙ שֶׁ֣בַע פָּר֔וֹת יְפ֥וֹת מַרְאֶ֖ה וּבְרִיאֹ֣ת בָּשָׂ֑ר וַתִּרְעֶ֖ינָה בָּאָֽחוּ׃ וְהָא מִן נַהְרָא, סָלְקָן שְׁבַע תּוֹרָן, שַׁפִּירָן לְמִחְזֵי, וּפַטִּימָן בְּסַר; וְרָעֲיָן, בְּאַחְוָא. 
מא,ג וְהִנֵּ֞ה שֶׁ֧בַע פָּר֣וֹת אֲחֵר֗וֹת עֹל֤וֹת אַֽחֲרֵיהֶן֙ מִן־הַיְאֹ֔ר רָע֥וֹת מַרְאֶ֖ה וְדַקּ֣וֹת בָּשָׂ֑ר וַֽתַּעֲמֹ֛דְנָה אֵ֥צֶל הַפָּר֖וֹת עַל־שְׂפַ֥ת הַיְאֹֽר׃ וְהָא שְׁבַע תּוֹרָן אָחֳרָנְיָן, סְלִיקָא בָּתְרֵיהוֹן מִן נַהְרָא, בִּישָׁן לְמִחְזֵי, וְחַסִּירָן בְּסַר; וְקַמָא לְקִבְלְהוֹן דְּתוֹרָתָא, עַל כֵּיף נַהְרָא. 
מא,ד וַתֹּאכַ֣לְנָה הַפָּר֗וֹת רָע֤וֹת הַמַּרְאֶה֙ וְדַקֹּ֣ת הַבָּשָׂ֔ר אֵ֚ת שֶׁ֣בַע הַפָּר֔וֹת יְפֹ֥ת הַמַּרְאֶ֖ה וְהַבְּרִיאֹ֑ת וַיִּיקַ֖ץ פַּרְעֹֽה׃ וַאֲכַלָא תּוֹרָתָא, דְּבִישָׁן לְמִחְזֵי וְחַסִּירָן בְּסַר, יָת שְׁבַע תּוֹרָתָא, דְּשַׁפִּירָן לְמִחְזֵי וּפַטִּימָתָא; וְאִתְּעַר, פַּרְעֹה. 
מא,ה וַיִּישָׁ֕ן וַֽיַּחֲלֹ֖ם שֵׁנִ֑ית וְהִנֵּ֣ה ׀ שֶׁ֣בַע שִׁבֳּלִ֗ים עֹל֛וֹת בְּקָנֶ֥ה אֶחָ֖ד בְּרִיא֥וֹת וְטֹבֽוֹת׃ וּדְמוּךְ, וַחֲלַם תִּנְיָנוּת; וְהָא שְׁבַע שֻׁבְלִין, סָלְקָן בְּקַנְיָא חַד--פַּטִּימָן וְטָבָן. 
מא,ו וְהִנֵּה֙ שֶׁ֣בַע שִׁבֳּלִ֔ים דַּקּ֖וֹת וּשְׁדוּפֹ֣ת קָדִ֑ים צֹֽמְח֖וֹת אַֽחֲרֵיהֶֽן׃ וְהָא שְׁבַע שֻׁבְלִין, לָקְיָן וּשְׁקִיפָן קִדּוּם--צָמְחָן, בָּתְרֵיהוֹן. 
מא,ז וַתִּבְלַ֨עְנָה֙ הַשִּׁבֳּלִ֣ים הַדַּקּ֔וֹת אֵ֚ת שֶׁ֣בַע הַֽשִּׁבֳּלִ֔ים הַבְּרִיא֖וֹת וְהַמְּלֵא֑וֹת וַיִּיקַ֥ץ פַּרְעֹ֖ה וְהִנֵּ֥ה חֲלֽוֹם׃ וּבְלַעָא, שֻׁבְלַיָּא לָקְיָתָא, יָת שְׁבַע שֻׁבְלַיָּא, פַּטִּימָתָא וּמַלְיָתָא; וְאִתְּעַר פַּרְעֹה, וְהָא חֶלְמָא. 
מא,ח וַיְהִ֤י בַבֹּ֨קֶר֙ וַתִּפָּ֣עֶם רוּח֔וֹ וַיִּשְׁלַ֗ח וַיִּקְרָ֛א אֶת־כָּל־חַרְטֻמֵּ֥י מִצְרַ֖יִם וְאֶת־כָּל־חֲכָמֶ֑יהָ וַיְסַפֵּ֨ר פַּרְעֹ֤ה לָהֶם֙ אֶת־חֲלֹמ֔וֹ וְאֵין־פּוֹתֵ֥ר אוֹתָ֖ם לְפַרְעֹֽה׃ וַהֲוָה בְּצַפְרָא, וּמִטָּרְפָא רוּחֵיהּ, וּשְׁלַח וּקְרָא יָת כָּל חָרָשֵׁי מִצְרַיִם, וְיָת כָּל חַכִּימַהָא; וְאִשְׁתַּעִי פַּרְעֹה לְהוֹן יָת חֶלְמֵיהּ, וְלֵית דְּפָשַׁר יָתְהוֹן לְפַרְעֹה. 
מא,ט וַיְדַבֵּר֙ שַׂ֣ר הַמַּשְׁקִ֔ים אֶת־פַּרְעֹ֖ה לֵאמֹ֑ר אֶת־חֲטָאַ֕י אֲנִ֖י מַזְכִּ֥יר הַיּֽוֹם׃ וּמַלֵּיל רַב שָׁקֵי, עִם פַּרְעֹה לְמֵימַר:  יָת סֻרְחָנִי, אֲנָא מַדְכַּר יוֹמָא דֵּין. 
מא,י פַּרְעֹ֖ה קָצַ֣ף עַל־עֲבָדָ֑יו וַיִּתֵּ֨ן אֹתִ֜י בְּמִשְׁמַ֗ר בֵּ֚ית שַׂ֣ר הַטַּבָּחִ֔ים אֹתִ֕י וְאֵ֖ת שַׂ֥ר הָֽאֹפִֽים׃ פַּרְעֹה, רְגֵיז עַל עַבְדּוֹהִי; וִיהַב יָתִי בְּמַטְּרַת, בֵּית רַב קָטוֹלַיָּא--יָתִי, וְיָת רַב נַחְתּוֹמֵי. 
מא,יא וַנַּֽחַלְמָ֥ה חֲל֛וֹם בְּלַ֥יְלָה אֶחָ֖ד אֲנִ֣י וָה֑וּא אִ֛ישׁ כְּפִתְר֥וֹן חֲלֹמ֖וֹ חָלָֽמְנוּ׃ וַחֲלַמְנָא חֶלְמָא בְּלֵילְיָא חַד, אֲנָא וְהוּא:  גְּבַר כְּפֻשְׁרַן חֶלְמֵיהּ, חֲלַמְנָא. 
מא,יב וְשָׁ֨ם אִתָּ֜נוּ נַ֣עַר עִבְרִ֗י עֶ֚בֶד לְשַׂ֣ר הַטַּבָּחִ֔ים וַ֨נְּסַפֶּר־ל֔וֹ וַיִּפְתָּר־לָ֖נוּ אֶת־חֲלֹֽמֹתֵ֑ינוּ אִ֥ישׁ כַּֽחֲלֹמ֖וֹ פָּתָֽר׃ וְתַמָּן עִמַּנָא עוֹלֵים עִבְרַאי, עַבְדָּא לְרַב קָטוֹלַיָּא, וְאִשְׁתַּעִינָא לֵיהּ, וּפַשַּׁר לַנָא יָת חֶלְמַנָא:  גְּבַר כְּחֶלְמֵיהּ, פַּשַּׁר. 
מא,יג וַיְהִ֛י כַּֽאֲשֶׁ֥ר פָּֽתַר־לָ֖נוּ כֵּ֣ן הָיָ֑ה אֹתִ֛י הֵשִׁ֥יב עַל־כַּנִּ֖י וְאֹת֥וֹ תָלָֽה׃ וַהֲוָה כְּמָא דְּפַשַּׁר לַנָא, כֵּין הֲוָה:  יָתִי אֲתֵיב עַל שִׁמּוּשִׁי, וְיָתֵיהּ צְלַב. 
מא,יד וַיִּשְׁלַ֤ח פַּרְעֹה֙ וַיִּקְרָ֣א אֶת־יוֹסֵ֔ף וַיְרִיצֻ֖הוּ מִן־הַבּ֑וֹר וַיְגַלַּח֙ וַיְחַלֵּ֣ף שִׂמְלֹתָ֔יו וַיָּבֹ֖א אֶל־פַּרְעֹֽה׃ וּשְׁלַח פַּרְעֹה וּקְרָא יָת יוֹסֵף, וְאַרְהֲטוּהִי מִן בֵּית אֲסִירֵי; וְסַפַּר וְשַׁנִּי כְּסוּתֵיהּ, וְעָאל לְוָת פַּרְעֹה. 
מא,טו וַיֹּ֤אמֶר פַּרְעֹה֙ אֶל־יוֹסֵ֔ף חֲל֣וֹם חָלַ֔מְתִּי וּפֹתֵ֖ר אֵ֣ין אֹת֑וֹ וַֽאֲנִ֗י שָׁמַ֤עְתִּי עָלֶ֨יךָ֙ לֵאמֹ֔ר תִּשְׁמַ֥ע חֲל֖וֹם לִפְתֹּ֥ר אֹתֽוֹ׃ וַאֲמַר פַּרְעֹה, לְיוֹסֵף, חֶלְמָא חֲלַמִית, וּפָשַׁר לֵית לֵיהּ; וַאֲנָא, שְׁמַעִית עֲלָךְ לְמֵימַר, דְּאַתְּ שָׁמַע חֶלְמָא, וּמְפַשַּׁר לֵיהּ. 
מא,טז וַיַּ֨עַן יוֹסֵ֧ף אֶת־פַּרְעֹ֛ה לֵאמֹ֖ר בִּלְעָדָ֑י אֱלֹהִ֕ים יַֽעֲנֶ֖ה אֶת־שְׁל֥וֹם פַּרְעֹֽה׃ וַאֲתֵיב יוֹסֵף יָת פַּרְעֹה לְמֵימַר, לָא מִן חָכְמְתִי:  אֱלָהֵין מִן קֳדָם יְיָ, יִתָּתַב שְׁלָמָא דְּפַרְעֹה. 
מא,יז וַיְדַבֵּ֥ר פַּרְעֹ֖ה אֶל־יוֹסֵ֑ף בַּֽחֲלֹמִ֕י הִנְנִ֥י עֹמֵ֖ד עַל־שְׂפַ֥ת הַיְאֹֽר׃ וּמַלֵּיל פַּרְעֹה, עִם יוֹסֵף:  בְּחֶלְמִי, הָאֲנָא קָאֵים עַל כֵּיף נַהְרָא. 
מא,יח וְהִנֵּ֣ה מִן־הַיְאֹ֗ר עֹלֹת֙ שֶׁ֣בַע פָּר֔וֹת בְּרִיא֥וֹת בָּשָׂ֖ר וִיפֹ֣ת תֹּ֑אַר וַתִּרְעֶ֖ינָה בָּאָֽחוּ׃ וְהָא מִן נַהְרָא, סָלְקָן שְׁבַע תּוֹרָן, פַּטִּימָן בְּסַר, וְשַׁפִּירָן לְמִחְזֵי; וְרָעֲיָן, בְּאַחְוָא. 
מא,יט וְהִנֵּ֞ה שֶֽׁבַע־פָּר֤וֹת אֲחֵרוֹת֙ עֹל֣וֹת אַֽחֲרֵיהֶ֔ן דַּלּ֨וֹת וְרָע֥וֹת תֹּ֛אַר מְאֹ֖ד וְרַקּ֣וֹת בָּשָׂ֑ר לֹֽא־רָאִ֧יתִי כָהֵ֛נָּה בְּכָל־אֶ֥רֶץ מִצְרַ֖יִם לָרֹֽעַ׃ וְהָא שְׁבַע תּוֹרָן אָחֳרָנְיָן, סְלִיקָא בָּתְרֵיהוֹן, חַסִּיכָן וּבִישָׁן לְמִחְזֵי לַחְדָּא, וְחַסִּירָן בְּסַר:  לָא חֲזֵיתִי דִּכְוָתְהוֹן בְּכָל אַרְעָא דְּמִצְרַיִם, לְבִישׁוּ. 
מא,כ וַתֹּאכַ֨לְנָה֙ הַפָּר֔וֹת הָֽרַקּ֖וֹת וְהָֽרָע֑וֹת אֵ֣ת שֶׁ֧בַע הַפָּר֛וֹת הָרִֽאשֹׁנ֖וֹת הַבְּרִיאֹֽת׃ וַאֲכַלָא, תּוֹרָתָא, חַסִּיכָתָא, וּבִישָׁתָא--יָת שְׁבַע תּוֹרָתָא קַדְמַיָתָא, פַּטִּימָתָא. 
מא,כא וַתָּבֹ֣אנָה אֶל־קִרְבֶּ֗נָה וְלֹ֤א נוֹדַע֙ כִּי־בָ֣אוּ אֶל־קִרְבֶּ֔נָה וּמַרְאֵיהֶ֣ן רַ֔ע כַּֽאֲשֶׁ֖ר בַּתְּחִלָּ֑ה וָֽאִיקָֽץ׃ וְעַאלָא לִמְעֵיהוֹן, וְלָא אִתְיְדַע אֲרֵי עַאלָא לִמְעֵיהוֹן, וּמִחְזֵיהוֹן בִּישׁ, כִּד בְּקַדְמֵיתָא; וְאִתְּעַרִית. 
מא,כב וָאֵ֖רֶא בַּֽחֲלֹמִ֑י וְהִנֵּ֣ה ׀ שֶׁ֣בַע שִׁבֳּלִ֗ים עֹלֹ֛ת בְּקָנֶ֥ה אֶחָ֖ד מְלֵאֹ֥ת וְטֹבֽוֹת׃ וַחֲזֵית, בְּחֶלְמִי; וְהָא שְׁבַע שֻׁבְלִין, סָלְקָן בְּקַנְיָא חַד--מַלְיָן וְטָבָן. 
מא,כג וְהִנֵּה֙ שֶׁ֣בַע שִׁבֳּלִ֔ים צְנֻמ֥וֹת דַּקּ֖וֹת שְׁדֻפ֣וֹת קָדִ֑ים צֹֽמְח֖וֹת אַֽחֲרֵיהֶֽם׃ וְהָא שְׁבַע שֻׁבְלִין, נָצַן לָקְיָן שְׁקִיפָן קִדּוּם--צָמְחָן, בָּתְרֵיהוֹן. 
מא,כד וַתִּבְלַ֨עְןָ֙ הַשִּׁבֳּלִ֣ים הַדַּקֹּ֔ת אֵ֛ת שֶׁ֥בַע הַֽשִּׁבֳּלִ֖ים הַטֹּב֑וֹת וָֽאֹמַר֙ אֶל־הַֽחַרְטֻמִּ֔ים וְאֵ֥ין מַגִּ֖יד לִֽי׃ וּבְלַעָא שֻׁבְלַיָּא לָקְיָתָא, יָת שְׁבַע שֻׁבְלַיָּא טָבָתָא; וַאֲמַרִית, לְחָרָשַׁיָּא, וְלֵית דִּמְחַוֵּי, לִי. 
מא,כה וַיֹּ֤אמֶר יוֹסֵף֙ אֶל־פַּרְעֹ֔ה חֲל֥וֹם פַּרְעֹ֖ה אֶחָ֣ד ה֑וּא אֵ֣ת אֲשֶׁ֧ר הָֽאֱלֹהִ֛ים עֹשֶׂ֖ה הִגִּ֥יד לְפַרְעֹֽה׃ וַאֲמַר יוֹסֵף לְפַרְעֹה, חֶלְמָא דְּפַרְעֹה חַד הוּא:  יָת דַּייָ עֲתִיד לְמֶעֱבַד, חַוִּי לְפַרְעֹה. 
מא,כו שֶׁ֧בַע פָּרֹ֣ת הַטֹּבֹ֗ת שֶׁ֤בַע שָׁנִים֙ הֵ֔נָּה וְשֶׁ֤בַע הַֽשִּׁבֳּלִים֙ הַטֹּבֹ֔ת שֶׁ֥בַע שָׁנִ֖ים הֵ֑נָּה חֲל֖וֹם אֶחָ֥ד הֽוּא׃ שְׁבַע תּוֹרָתָא טָבָתָא, שְׁבַע שְׁנַיָּא אִנִּין, וּשְׁבַע שֻׁבְלַיָּא טָבָתָא, שְׁבַע שְׁנַיָּא אִנִּין:  חֶלְמָא, חַד הוּא. 
מא,כז וְשֶׁ֣בַע הַ֠פָּרוֹת הָֽרַקּ֨וֹת וְהָֽרָעֹ֜ת הָֽעֹלֹ֣ת אַֽחֲרֵיהֶ֗ן שֶׁ֤בַע שָׁנִים֙ הֵ֔נָּה וְשֶׁ֤בַע הַֽשִּׁבֳּלִים֙ הָֽרֵק֔וֹת שְׁדֻפ֖וֹת הַקָּדִ֑ים יִֽהְי֕וּ שֶׁ֖בַע שְׁנֵ֥י רָעָֽב׃ וּשְׁבַע תּוֹרָתָא חַסִּיכָתָא וּבִישָׁתָא דִּסְלִיקָא בָּתְרֵיהוֹן, שְׁבַע שְׁנַיָּא אִנִּין, וּשְׁבַע שֻׁבְלַיָּא לָקְיָתָא, דִּשְׁקִיפָן קִדּוּם--יִהְוְיָן, שְׁבַע שְׁנֵי כַּפְנָא. 
מא,כח ה֣וּא הַדָּבָ֔ר אֲשֶׁ֥ר דִּבַּ֖רְתִּי אֶל־פַּרְעֹ֑ה אֲשֶׁ֧ר הָֽאֱלֹהִ֛ים עֹשֶׂ֖ה הֶרְאָ֥ה אֶת־פַּרְעֹֽה׃ הוּא פִּתְגָמָא, דְּמַלֵּילִית עִם פַּרְעֹה:  דַּייָ עֲתִיד לְמֶעֱבַד, אַחְזִי יָת פַּרְעֹה. 
מא,כט הִנֵּ֛ה שֶׁ֥בַע שָׁנִ֖ים בָּא֑וֹת שָׂבָ֥ע גָּד֖וֹל בְּכָל־אֶ֥רֶץ מִצְרָֽיִם׃ הָא שְׁבַע שְׁנַיָּא, אָתְיָן--סִבְעָא רַבָּא, בְּכָל אַרְעָא דְּמִצְרָיִם. 
מא,ל וְ֠קָמוּ שֶׁ֜בַע שְׁנֵ֤י רָעָב֙ אַֽחֲרֵיהֶ֔ן וְנִשְׁכַּ֥ח כָּל־הַשָּׂבָ֖ע בְּאֶ֣רֶץ מִצְרָ֑יִם וְכִלָּ֥ה הָֽרָעָ֖ב אֶת־הָאָֽרֶץ׃ וִיקוּמָן שְׁבַע שְׁנֵי כַּפְנָא, בָּתְרֵיהוֹן, וְיִתְנְשֵׁי כָּל סִבְעָא, בְּאַרְעָא דְּמִצְרָיִם; וִישֵׁיצֵי כַּפְנָא, יָת עַמָּא דְּאַרְעָא. 
מא,לא וְלֹֽא־יִוָּדַ֤ע הַשָּׂבָע֙ בָּאָ֔רֶץ מִפְּנֵ֛י הָֽרָעָ֥ב הַה֖וּא אַֽחֲרֵי־כֵ֑ן כִּֽי־כָבֵ֥ד ה֖וּא מְאֹֽד׃ וְלָא יִתְיְדַע סִבְעָא בְּאַרְעָא, מִן קֳדָם כַּפְנָא הַהוּא דִּיהֵי בָּתַר כֵּין:  אֲרֵי תַּקִּיף הוּא, לַחְדָּא. 
מא,לב וְעַ֨ל הִשָּׁנ֧וֹת הַֽחֲל֛וֹם אֶל־פַּרְעֹ֖ה פַּֽעֲמָ֑יִם כִּֽי־נָכ֤וֹן הַדָּבָר֙ מֵעִ֣ם הָֽאֱלֹהִ֔ים וּמְמַהֵ֥ר הָֽאֱלֹהִ֖ים לַֽעֲשֹׂתֽוֹ׃ וְעַל דְּאִתַּנִּי חֶלְמָא לְפַרְעֹה, תַּרְתֵּין זִמְנִין--אֲרֵי תָּקֵין פִּתְגָמָא מִן קֳדָם יְיָ, וּמוֹחֵי יְיָ לְמִעְבְּדֵיהּ. 
מא,לג וְעַתָּה֙ יֵרֶ֣א פַרְעֹ֔ה אִ֖ישׁ נָב֣וֹן וְחָכָ֑ם וִֽישִׁיתֵ֖הוּ עַל־אֶ֥רֶץ מִצְרָֽיִם׃ וּכְעַן יִחְזֵי פַּרְעֹה, גְּבַר סֻכְלְתָן וְחַכִּים; וִימַנֵּינֵיהּ, עַל אַרְעָא דְּמִצְרָיִם. 
מא,לד יַֽעֲשֶׂ֣ה פַרְעֹ֔ה וְיַפְקֵ֥ד פְּקִדִ֖ים עַל־הָאָ֑רֶץ וְחִמֵּשׁ֙ אֶת־אֶ֣רֶץ מִצְרַ֔יִם בְּשֶׁ֖בַע שְׁנֵ֥י הַשָּׂבָֽע׃ יַעֲבֵיד פַּרְעֹה, וִימַנֵּי מְהֵימְנִין עַל אַרְעָא; וִיזָרְזוּן יָת אַרְעָא דְּמִצְרַיִם, בִּשְׁבַע שְׁנֵי סִבְעָא. 
מא,לה וְיִקְבְּצ֗וּ אֶת־כָּל־אֹ֨כֶל֙ הַשָּׁנִ֣ים הַטֹּב֔וֹת הַבָּאֹ֖ת הָאֵ֑לֶּה וְיִצְבְּרוּ־בָ֞ר תַּ֧חַת יַד־פַּרְעֹ֛ה אֹ֥כֶל בֶּֽעָרִ֖ים וְשָׁמָֽרוּ׃ וְיִכְנְשׁוּן, יָת כָּל עֲבוּר שְׁנַיָּא טָבָתָא, דְּאָתְיָן, אִלֵּין; וְיֵיצְרוּן עֲבוּרָא תְּחוֹת יַד מְהֵימְנֵי פַּרְעֹה, עֲבוּרָא בְּקִרְוַיָּא--וְיִטְּרוּן. 
מא,לו וְהָיָ֨ה הָאֹ֤כֶל לְפִקָּדוֹן֙ לָאָ֔רֶץ לְשֶׁ֨בַע֙ שְׁנֵ֣י הָֽרָעָ֔ב אֲשֶׁ֥ר תִּֽהְיֶ֖יןָ בְּאֶ֣רֶץ מִצְרָ֑יִם וְלֹֽא־תִכָּרֵ֥ת הָאָ֖רֶץ בָּֽרָעָֽב׃ וִיהֵי עֲבוּרָא גְּנִיז, לְעַמָּא דְּאַרְעָא, לִשְׁבַע שְׁנֵי כַּפְנָא, דְּיִהְוְיָן בְּאַרְעָא דְּמִצְרָיִם; וְלָא יִשְׁתֵּיצֵי עַמָּא דְּאַרְעָא, בְּכַפְנָא. 
מא,לז וַיִּיטַ֥ב הַדָּבָ֖ר בְּעֵינֵ֣י פַרְעֹ֑ה וּבְעֵינֵ֖י כָּל־עֲבָדָֽיו׃ וּשְׁפַר פִּתְגָמָא, בְּעֵינֵי פַּרְעֹה, וּבְעֵינֵי, כָּל עַבְדּוֹהִי. 
מא,לח וַיֹּ֥אמֶר פַּרְעֹ֖ה אֶל־עֲבָדָ֑יו הֲנִמְצָ֣א כָזֶ֔ה אִ֕ישׁ אֲשֶׁ֛ר ר֥וּחַ אֱלֹהִ֖ים בּֽוֹ׃ וַאֲמַר פַּרְעֹה, לְעַבְדּוֹהִי:  הֲנַשְׁכַּח כְּדֵין--גְּבַר, דְּרוּחַ נְבוּאָה מִן קֳדָם יְיָ בֵּיהּ. 
מא,לט וַיֹּ֤אמֶר פַּרְעֹה֙ אֶל־יוֹסֵ֔ף אַֽחֲרֵ֨י הוֹדִ֧יעַ אֱלֹהִ֛ים אֽוֹתְךָ֖ אֶת־כָּל־זֹ֑את אֵין־נָב֥וֹן וְחָכָ֖ם כָּמֽוֹךָ׃ וַאֲמַר פַּרְעֹה לְיוֹסֵף, בָּתַר דְּהוֹדַע יְיָ יָתָךְ יָת כָּל דָּא, לֵית סֻכְלְתָן וְחַכִּים, כְּוָתָךְ. 
מא,מ אַתָּה֙ תִּֽהְיֶ֣ה עַל־בֵּיתִ֔י וְעַל־פִּ֖יךָ יִשַּׁ֣ק כָּל־עַמִּ֑י רַ֥ק הַכִּסֵּ֖א אֶגְדַּ֥ל מִמֶּֽךָּ׃ אַתְּ תְּהֵי מְמֻנַּא עַל בֵּיתִי, וְעַל מֵימְרָךְ יִתְּזָן כָּל עַמִּי; לְחוֹד כֻּרְסֵי מַלְכוּתָא הָדֵין, אֵיהֵי יַקִּיר מִנָּךְ. 
מא,מא וַיֹּ֥אמֶר פַּרְעֹ֖ה אֶל־יוֹסֵ֑ף רְאֵה֙ נָתַ֣תִּי אֹֽתְךָ֔ עַ֖ל כָּל־אֶ֥רֶץ מִצְרָֽיִם׃ וַאֲמַר פַּרְעֹה, לְיוֹסֵף:  חֲזִי דְּמַנִּיתִי יָתָךְ, עַל כָּל אַרְעָא דְּמִצְרָיִם. 
מא,מב וַיָּ֨סַר פַּרְעֹ֤ה אֶת־טַבַּעְתּוֹ֙ מֵעַ֣ל יָד֔וֹ וַיִּתֵּ֥ן אֹתָ֖הּ עַל־יַ֣ד יוֹסֵ֑ף וַיַּלְבֵּ֤שׁ אֹתוֹ֙ בִּגְדֵי־שֵׁ֔שׁ וַיָּ֛שֶׂם רְבִ֥ד הַזָּהָ֖ב עַל־צַוָּארֽוֹ׃ וְאַעְדִּי פַּרְעֹה יָת עִזְקְתֵיהּ מֵעַל יְדֵיהּ, וִיהַב יָתַהּ עַל יְדָא דְּיוֹסֵף; וְאַלְבֵּישׁ יָתֵיהּ לְבוּשִׁין דְּבוּץ, וְשַׁוִּי מְנִיכָא דְּדַהְבָּא עַל צַוְרֵיהּ. 
מא,מג וַיַּרְכֵּ֣ב אֹת֗וֹ בְּמִרְכֶּ֤בֶת הַמִּשְׁנֶה֙ אֲשֶׁר־ל֔וֹ וַיִּקְרְא֥וּ לְפָנָ֖יו אַבְרֵ֑ךְ וְנָת֣וֹן אֹת֔וֹ עַ֖ל כָּל־אֶ֥רֶץ מִצְרָֽיִם׃ וְאַרְכֵּיב יָתֵיהּ, בִּרְתִכָּא תִּנְיֵיתָא דִּילֵיהּ, וְאַכְרִיזוּ קֳדָמוֹהִי, דֵּין אַבָּא לְמַלְכָּא; וּמַנִּי יָתֵיהּ, עַל כָּל אַרְעָא דְּמִצְרָיִם. 
מא,מד וַיֹּ֧אמֶר פַּרְעֹ֛ה אֶל־יוֹסֵ֖ף אֲנִ֣י פַרְעֹ֑ה וּבִלְעָדֶ֗יךָ לֹֽא־יָרִ֨ים אִ֧ישׁ אֶת־יָד֛וֹ וְאֶת־רַגְל֖וֹ בְּכָל־אֶ֥רֶץ מִצְרָֽיִם׃ וַאֲמַר פַּרְעֹה לְיוֹסֵף, אֲנָא פַּרְעֹה; וּבָר מִמֵּימְרָךְ, לָא יְרִים גְּבַר יָת יְדֵיהּ לְמֵיחַד זֵין וְיָת רַגְלֵיהּ לְמִרְכַּב עַל סוּסְיָא--בְּכָל אַרְעָא דְּמִצְרָיִם. 
מא,מה וַיִּקְרָ֨א פַרְעֹ֣ה שֵׁם־יוֹסֵף֮ צָֽפְנַ֣ת פַּעְנֵחַ֒ וַיִּתֶּן־ל֣וֹ אֶת־אָֽסְנַ֗ת בַּת־פּ֥וֹטִי פֶ֛רַע כֹּהֵ֥ן אֹ֖ן לְאִשָּׁ֑ה וַיֵּצֵ֥א יוֹסֵ֖ף עַל־אֶ֥רֶץ מִצְרָֽיִם׃ וּקְרָא פַּרְעֹה שׁוֹם יוֹסֵף, גֻּבְרָא דְּמִטַּמְרָן גָּלְיָן לֵיהּ, וִיהַב לֵיהּ יָת אָסְנַת בַּת פּוֹטִי פֶרַע רַבָּא דְּאוֹן, לְאִתּוּ; וּנְפַק יוֹסֵף, שַׁלִּיט עַל אַרְעָא דְּמִצְרָיִם. 
מא,מו וְיוֹסֵף֙ בֶּן־שְׁלֹשִׁ֣ים שָׁנָ֔ה בְּעָמְד֕וֹ לִפְנֵ֖י פַּרְעֹ֣ה מֶֽלֶךְ־מִצְרָ֑יִם וַיֵּצֵ֤א יוֹסֵף֙ מִלִּפְנֵ֣י פַרְעֹ֔ה וַֽיַּעֲבֹ֖ר בְּכָל־אֶ֥רֶץ מִצְרָֽיִם׃ וְיוֹסֵף, בַּר תְּלָתִין שְׁנִין, כַּד קָם, קֳדָם פַּרְעֹה מַלְכָּא דְּמִצְרָיִם; וּנְפַק יוֹסֵף מִן קֳדָם פַּרְעֹה, וַעֲבַר בְּכָל אַרְעָא דְּמִצְרָיִם. 
מא,מז וַתַּ֣עַשׂ הָאָ֔רֶץ בְּשֶׁ֖בַע שְׁנֵ֣י הַשָּׂבָ֑ע לִקְמָצִֽים׃ וּכְנַשׁוּ דָּיְרֵי אַרְעָא, בִּשְׁבַע שְׁנֵי סִבְעָא--עֲבוּרָא לְאוֹצְרִין. 
מא,מח וַיִּקְבֹּ֞ץ אֶת־כָּל־אֹ֣כֶל ׀ שֶׁ֣בַע שָׁנִ֗ים אֲשֶׁ֤ר הָיוּ֙ בְּאֶ֣רֶץ מִצְרַ֔יִם וַיִּתֶּן־אֹ֖כֶל בֶּֽעָרִ֑ים אֹ֧כֶל שְׂדֵה־הָעִ֛יר אֲשֶׁ֥ר סְבִֽיבֹתֶ֖יהָ נָתַ֥ן בְּתוֹכָֽהּ׃ וּכְנַשׁ יָת כָּל עֲבוּר שְׁבַע שְׁנַיָּא, דַּהֲוַאָה בְּאַרְעָא דְּמִצְרַיִם, וִיהַב עֲבוּרָא, בְּקִרְוַיָּא:  עֲבוּר חֲקַל קַרְתָּא דִּבְסַחְרָנַהָא, יְהַב בְּגַוַּהּ. 
מא,מט וַיִּצְבֹּ֨ר יוֹסֵ֥ף בָּ֛ר כְּח֥וֹל הַיָּ֖ם הַרְבֵּ֣ה מְאֹ֑ד עַ֛ד כִּֽי־חָדַ֥ל לִסְפֹּ֖ר כִּי־אֵ֥ין מִסְפָּֽר׃ וּכְנַשׁ יוֹסֵף עֲבוּרָא כְּחָלָא דְּיַמָּא, סַגִּי לַחְדָּא--עַד דִּפְסַק מִלְּמִמְנֵי, אֲרֵי לֵית מִנְיָן. 
מא,נ וּלְיוֹסֵ֤ף יֻלַּד֙ שְׁנֵ֣י בָנִ֔ים בְּטֶ֥רֶם תָּב֖וֹא שְׁנַ֣ת הָֽרָעָ֑ב אֲשֶׁ֤ר יָֽלְדָה־לּוֹ֙ אָֽסְנַ֔ת בַּת־פּ֥וֹטִי פֶ֖רַע כֹּהֵ֥ן אֽוֹן׃ וּלְיוֹסֵף אִתְיְלִידוּ תְּרֵין בְּנִין, עַד לָא עַאלַת שַׁתָּא דְּכַפְנָא, דִּילֵידַת לֵיהּ אָסְנַת, בַּת פּוֹטִי פֶרַע רַבָּא דְּאוֹן. 
מא,נא וַיִּקְרָ֥א יוֹסֵ֛ף אֶת־שֵׁ֥ם הַבְּכ֖וֹר מְנַשֶּׁ֑ה כִּֽי־נַשַּׁ֤נִי אֱלֹהִים֙ אֶת־כָּל־עֲמָלִ֔י וְאֵ֖ת כָּל־בֵּ֥ית אָבִֽי׃ וּקְרָא יוֹסֵף יָת שׁוֹם בֻּכְרָא, מְנַשֶּׁה:  אֲרֵי אַנְשְׁיַנִי יְיָ יָת כָּל עַמְלִי, וְיָת כָּל בֵּית אַבָּא. 
מא,נב וְאֵ֛ת שֵׁ֥ם הַשֵּׁנִ֖י קָרָ֣א אֶפְרָ֑יִם כִּֽי־הִפְרַ֥נִי אֱלֹהִ֖ים בְּאֶ֥רֶץ עָנְיִֽי׃ וְיָת שׁוֹם תִּנְיָנָא, קְרָא אֶפְרָיִם:  אֲרֵי אַפְּשַׁנִי יְיָ, בַּאֲרַע שִׁעְבּוּדִי. 
מא,נג וַתִּכְלֶ֕ינָה שֶׁ֖בַע שְׁנֵ֣י הַשָּׂבָ֑ע אֲשֶׁ֥ר הָיָ֖ה בְּאֶ֥רֶץ מִצְרָֽיִם׃ וּשְׁלִימָא, שְׁבַע שְׁנֵי סִבְעָא, דַּהֲוַאָה, בְּאַרְעָא דְּמִצְרָיִם. 
מא,נד וַתְּחִלֶּ֜ינָה שֶׁ֣בַע שְׁנֵ֤י הָֽרָעָב֙ לָב֔וֹא כַּֽאֲשֶׁ֖ר אָמַ֣ר יוֹסֵ֑ף וַיְהִ֤י רָעָב֙ בְּכָל־הָ֣אֲרָצ֔וֹת וּבְכָל־אֶ֥רֶץ מִצְרַ֖יִם הָ֥יָה לָֽחֶם׃ וְשָׁרִיאָה שְׁבַע שְׁנֵי כַּפְנָא, לְמֵיתֵי--כְּמָא דַּאֲמַר יוֹסֵף; וַהֲוָה כַּפְנָא בְּכָל אַרְעָתָא, וּבְכָל אַרְעָא דְּמִצְרַיִם הֲוָה לַחְמָא. 
מא,נה וַתִּרְעַב֙ כָּל־אֶ֣רֶץ מִצְרַ֔יִם וַיִּצְעַ֥ק הָעָ֛ם אֶל־פַּרְעֹ֖ה לַלָּ֑חֶם וַיֹּ֨אמֶר פַּרְעֹ֤ה לְכָל־מִצְרַ֨יִם֙ לְכ֣וּ אֶל־יוֹסֵ֔ף אֲשֶׁר־יֹאמַ֥ר לָכֶ֖ם תַּֽעֲשֽׂוּ׃ וּכְפֵינַת כָּל אַרְעָא דְּמִצְרַיִם, וּצְוַח עַמָּא קֳדָם פַּרְעֹה עַל לַחְמָא; וַאֲמַר פַּרְעֹה לְכָל מִצְרָאֵי אִיזִילוּ לְוָת יוֹסֵף, דְּיֵימַר לְכוֹן תַּעְבְּדוּן. 
מא,נו וְהָֽרָעָ֣ב הָיָ֔ה עַ֖ל כָּל־פְּנֵ֣י הָאָ֑רֶץ וַיִּפְתַּ֨ח יוֹסֵ֜ף אֶֽת־כָּל־אֲשֶׁ֤ר בָּהֶם֙ וַיִּשְׁבֹּ֣ר לְמִצְרַ֔יִם וַיֶּֽחֱזַ֥ק הָֽרָעָ֖ב בְּאֶ֥רֶץ מִצְרָֽיִם׃ וְכַפְנָא הֲוָה, עַל כָּל אַפֵּי אַרְעָא; וּפְתַח יוֹסֵף יָת כָּל אוֹצְרַיָּא דִּבְהוֹן עֲבוּרָא, וְזַבֵּין לְמִצְרָאֵי, וּתְקֵיף כַּפְנָא, בְּאַרְעָא דְּמִצְרָיִם. 
מא,נז וְכָל־הָאָ֨רֶץ֙ בָּ֣אוּ מִצְרַ֔יְמָה לִשְׁבֹּ֖ר אֶל־יוֹסֵ֑ף כִּֽי־חָזַ֥ק הָֽרָעָ֖ב בְּכָל־הָאָֽרֶץ׃ וְכָל דָּיְרֵי אַרְעָא אֲתוֹ לְמִצְרַיִם, לְמִזְבַּן עֲבוּרָא מִן יוֹסֵף:  אֲרֵי תְּקֵיף כַּפְנָא, בְּכָל אַרְעָא. 
מב,א וַיַּ֣רְא יַֽעֲקֹ֔ב כִּ֥י יֶשׁ־שֶׁ֖בֶר בְּמִצְרָ֑יִם וַיֹּ֤אמֶר יַֽעֲקֹב֙ לְבָנָ֔יו לָ֖מָּה תִּתְרָאֽוּ׃ וַחֲזָא יַעֲקוֹב, אֲרֵי אִית עֲבוּר מִזְדַּבַּן בְּמִצְרָיִם; וַאֲמַר יַעֲקוֹב לִבְנוֹהִי, לְמָא תִּתַּחְזוֹן. 
מב,ב וַיֹּ֕אמֶר הִנֵּ֣ה שָׁמַ֔עְתִּי כִּ֥י יֶשׁ־שֶׁ֖בֶר בְּמִצְרָ֑יִם רְדוּ־שָׁ֨מָּה֙ וְשִׁבְרוּ־לָ֣נוּ מִשָּׁ֔ם וְנִֽחְיֶ֖ה וְלֹ֥א נָמֽוּת׃ וַאֲמַר--הָא שְׁמַעִית, אֲרֵי אִית עֲבוּר מִזְדַּבַּן בְּמִצְרָיִם; חוּתוּ לְתַמָּן וּזְבוּנוּ לַנָא מִתַּמָּן, וְנֵיחֵי וְלָא נְמוּת. 
מב,ג וַיֵּֽרְד֥וּ אֲחֵֽי־יוֹסֵ֖ף עֲשָׂרָ֑ה לִשְׁבֹּ֥ר בָּ֖ר מִמִּצְרָֽיִם׃ וּנְחַתוּ אֲחֵי יוֹסֵף, עַסְרָא, לְמִזְבַּן עֲבוּרָא, מִמִּצְרָיִם. 
מב,ד וְאֶת־בִּנְיָמִין֙ אֲחִ֣י יוֹסֵ֔ף לֹֽא־שָׁלַ֥ח יַֽעֲקֹ֖ב אֶת־אֶחָ֑יו כִּ֣י אָמַ֔ר פֶּן־יִקְרָאֶ֖נּוּ אָסֽוֹן׃ וְיָת בִּנְיָמִין אֲחוּהִי דְּיוֹסֵף, לָא שְׁלַח יַעֲקוֹב עִם אֲחוֹהִי:  אֲרֵי אֲמַר, דִּלְמָא יְעָרְעִנֵּיהּ מוֹתָא. 
מב,ה וַיָּבֹ֨אוּ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל לִשְׁבֹּ֖ר בְּת֣וֹךְ הַבָּאִ֑ים כִּֽי־הָיָ֥ה הָֽרָעָ֖ב בְּאֶ֥רֶץ כְּנָֽעַן׃ וַאֲתוֹ בְּנֵי יִשְׂרָאֵל, לְמִזְבַּן עֲבוּרָא בְּגוֹ עָאלַיָּא:  אֲרֵי הֲוָה כַּפְנָא, בְּאַרְעָא דִּכְנָעַן. 
מב,ו וְיוֹסֵ֗ף ה֚וּא הַשַּׁלִּ֣יט עַל־הָאָ֔רֶץ ה֥וּא הַמַּשְׁבִּ֖יר לְכָל־עַ֣ם הָאָ֑רֶץ וַיָּבֹ֨אוּ֙ אֲחֵ֣י יוֹסֵ֔ף וַיִּשְׁתַּֽחֲווּ־ל֥וֹ אַפַּ֖יִם אָֽרְצָה׃ וְיוֹסֵף, הוּא דְּשַׁלִּיט עַל אַרְעָא--הוּא מְזַבֵּין עֲבוּרָא, לְכָל עַמָּא דְּאַרְעָא; וַאֲתוֹ אֲחֵי יוֹסֵף, וּסְגִידוּ לֵיהּ עַל אַפֵּיהוֹן עַל אַרְעָא. 
מב,ז וַיַּ֥רְא יוֹסֵ֛ף אֶת־אֶחָ֖יו וַיַּכִּרֵ֑ם וַיִּתְנַכֵּ֨ר אֲלֵיהֶ֜ם וַיְדַבֵּ֧ר אִתָּ֣ם קָשׁ֗וֹת וַיֹּ֤אמֶר אֲלֵהֶם֙ מֵאַ֣יִן בָּאתֶ֔ם וַיֹּ֣אמְר֔וּ מֵאֶ֥רֶץ כְּנַ֖עַן לִשְׁבָּר־אֹֽכֶל׃ וַחֲזָא יוֹסֵף יָת אֲחוֹהִי, וְאִשְׁתְּמוֹדְעִנּוּן; וְחַשֵּׁיב מָא דִּימַלֵּיל עִמְּהוֹן וּמַלֵּיל עִמְּהוֹן קַשְׁיָן, וַאֲמַר לְהוֹן מְנָן אֲתֵיתוֹן, וַאֲמַרוּ, מֵאַרְעָא דִּכְנַעַן לְמִזְבַּן עֲבוּרָא. 
מב,ח וַיַּכֵּ֥ר יוֹסֵ֖ף אֶת־אֶחָ֑יו וְהֵ֖ם לֹ֥א הִכִּרֻֽהוּ׃ וְאִשְׁתְּמוֹדַע יוֹסֵף, יָת אֲחוֹהִי; וְאִנּוּן, לָא אִשְׁתְּמוֹדְעוּהִי. 
מב,ט וַיִּזְכֹּ֣ר יוֹסֵ֔ף אֵ֚ת הַֽחֲלֹמ֔וֹת אֲשֶׁ֥ר חָלַ֖ם לָהֶ֑ם וַיֹּ֤אמֶר אֲלֵהֶם֙ מְרַגְּלִ֣ים אַתֶּ֔ם לִרְא֛וֹת אֶת־עֶרְוַ֥ת הָאָ֖רֶץ בָּאתֶֽם׃ וְאִדְּכַר יוֹסֵף--יָת חֶלְמַיָּא, דַּהֲוָה חָלֵים לְהוֹן; וַאֲמַר לְהוֹן אַלִּילֵי אַתּוּן, לְמִחְזֵי יָת בִּדְקַהּ דְּאַרְעָא אֲתֵיתוֹן. 
מב,י וַיֹּֽאמְר֥וּ אֵלָ֖יו לֹ֣א אֲדֹנִ֑י וַֽעֲבָדֶ֥יךָ בָּ֖אוּ לִשְׁבָּר־אֹֽכֶל׃ וַאֲמַרוּ לֵיהּ, לָא רִבּוֹנִי; וְעַבְדָּךְ אֲתוֹ, לְמִזְבַּן עֲבוּרָא. 
מב,יא כֻּלָּ֕נוּ בְּנֵ֥י אִישׁ־אֶחָ֖ד נָ֑חְנוּ כֵּנִ֣ים אֲנַ֔חְנוּ לֹֽא־הָי֥וּ עֲבָדֶ֖יךָ מְרַגְּלִֽים׃ כֻּלַּנָא, בְּנֵי גֻּבְרָא חַד נַחְנָא; כֵּיוָנֵי אֲנַחְנָא, לָא הֲווֹ עַבְדָּךְ אַלִּילֵי. 
מב,יב וַיֹּ֖אמֶר אֲלֵהֶ֑ם לֹ֕א כִּֽי־עֶרְוַ֥ת הָאָ֖רֶץ בָּאתֶ֥ם לִרְאֽוֹת׃ וַאֲמַר, לְהוֹן:  לָא, אֱלָהֵין בִּדְקַהּ דְּאַרְעָא אֲתֵיתוֹן לְמִחְזֵי. 
מב,יג וַיֹּֽאמְר֗וּ שְׁנֵ֣ים עָשָׂר֩ עֲבָדֶ֨יךָ אַחִ֧ים ׀ אֲנַ֛חְנוּ בְּנֵ֥י אִישׁ־אֶחָ֖ד בְּאֶ֣רֶץ כְּנָ֑עַן וְהִנֵּ֨ה הַקָּטֹ֤ן אֶת־אָבִ֨ינוּ֙ הַיּ֔וֹם וְהָֽאֶחָ֖ד אֵינֶֽנּוּ׃ וַאֲמַרוּ, תְּרֵי עֲסַר עַבְדָּךְ אַחִין אֲנַחְנָא בְּנֵי גֻּבְרָא חַד--בְּאַרְעָא דִּכְנָעַן; וְהָא זְעֵירָא עִם אֲבוּנָא יוֹמָא דֵּין, וְחַד לָיְתוֹהִי. 
מב,יד וַיֹּ֥אמֶר אֲלֵהֶ֖ם יוֹסֵ֑ף ה֗וּא אֲשֶׁ֨ר דִּבַּ֧רְתִּי אֲלֵכֶ֛ם לֵאמֹ֖ר מְרַגְּלִ֥ים אַתֶּֽם׃ וַאֲמַר לְהוֹן, יוֹסֵף:  הוּא, דְּמַלֵּילִית עִמְּכוֹן לְמֵימַר--אַלִּילֵי אַתּוּן. 
מב,טו בְּזֹ֖את תִּבָּחֵ֑נוּ חֵ֤י פַרְעֹה֙ אִם־תֵּֽצְא֣וּ מִזֶּ֔ה כִּ֧י אִם־בְּב֛וֹא אֲחִיכֶ֥ם הַקָּטֹ֖ן הֵֽנָּה׃ בְּדָא, תִּתְבַּחְרוּן:  חַיֵּי פַּרְעֹה אִם תִּפְּקוּן מִכָּא, אֱלָהֵין בְּמֵיתֵי אֲחוּכוֹן זְעֵירָא הָלְכָא. 
מב,טז שִׁלְח֨וּ מִכֶּ֣ם אֶחָד֮ וְיִקַּ֣ח אֶת־אֲחִיכֶם֒ וְאַתֶּם֙ הֵאָ֣סְר֔וּ וְיִבָּֽחֲנוּ֙ דִּבְרֵיכֶ֔ם הַֽאֱמֶ֖ת אִתְּכֶ֑ם וְאִם־לֹ֕א חֵ֣י פַרְעֹ֔ה כִּ֥י מְרַגְּלִ֖ים אַתֶּֽם׃ שְׁלַחוּ מִנְּכוֹן חַד, וְיִדְבַּר יָת אֲחוּכוֹן, וְאַתּוּן תִּתְאַסְרוּן, וְיִתְבַּחְרוּן פִּתְגָמֵיכוֹן הַקְשׁוֹט אַתּוּן אָמְרִין; וְאִם לָא--חַיֵּי פַּרְעֹה, אֲרֵי אַלִּילֵי אַתּוּן. 
מב,יז וַיֶּֽאֱסֹ֥ף אֹתָ֛ם אֶל־מִשְׁמָ֖ר שְׁלֹ֥שֶׁת יָמִֽים׃ וּכְנַשׁ יָתְהוֹן לְבֵית מַטְּרָא, תְּלָתָא יוֹמִין. 
מב,יח וַיֹּ֨אמֶר אֲלֵהֶ֤ם יוֹסֵף֙ בַּיּ֣וֹם הַשְּׁלִישִׁ֔י זֹ֥את עֲשׂ֖וּ וִֽחְי֑וּ אֶת־הָֽאֱלֹהִ֖ים אֲנִ֥י יָרֵֽא׃ וַאֲמַר לְהוֹן יוֹסֵף בְּיוֹמָא תְּלִיתָאָה, דָּא עֲבִידוּ וְאִתְקַיַּמוּ; מִן קֳדָם יְיָ, אֲנָא דָּחֵיל. 
מב,יט אִם־כֵּנִ֣ים אַתֶּ֔ם אֲחִיכֶ֣ם אֶחָ֔ד יֵֽאָסֵ֖ר בְּבֵ֣ית מִשְׁמַרְכֶ֑ם וְאַתֶּם֙ לְכ֣וּ הָבִ֔יאוּ שֶׁ֖בֶר רַֽעֲב֥וֹן בָּֽתֵּיכֶֽם׃ אִם כֵּיוָנֵי אַתּוּן--אֲחוּכוֹן חַד, יִתְאֲסַר בְּבֵית מַטַּרְתְּכוֹן; וְאַתּוּן אִיזִילוּ אוֹבִילוּ, עֲבוּרָא דְּחַסִּיר בְּבָתֵּיכוֹן. 
מב,כ וְאֶת־אֲחִיכֶ֤ם הַקָּטֹן֙ תָּבִ֣יאוּ אֵלַ֔י וְיֵאָֽמְנ֥וּ דִבְרֵיכֶ֖ם וְלֹ֣א תָמ֑וּתוּ וַיַּֽעֲשׂוּ־כֵֽן׃ וְיָת אֲחוּכוֹן זְעֵירָא תַּיְתוֹן לְוָתִי, וְיִתְהָמְנוּן פִּתְגָמֵיכוֹן וְלָא תְּמוּתוּן; וַעֲבַדוּ כֵּין. 
מב,כא וַיֹּֽאמְר֞וּ אִ֣ישׁ אֶל־אָחִ֗יו אֲבָל֮ אֲשֵׁמִ֣ים ׀ אֲנַחְנוּ֮ עַל־אָחִינוּ֒ אֲשֶׁ֨ר רָאִ֜ינוּ צָרַ֥ת נַפְשׁ֛וֹ בְּהִתְחַֽנְנ֥וֹ אֵלֵ֖ינוּ וְלֹ֣א שָׁמָ֑עְנוּ עַל־כֵּן֙ בָּ֣אָה אֵלֵ֔ינוּ הַצָּרָ֖ה הַזֹּֽאת׃ וַאֲמַרוּ גְּבַר לַאֲחוּהִי, בְּקֻשְׁטָא חַיָּבִין אֲנַחְנָא עַל אֲחוּנָא, דַּחֲזֵינָא בְּעָקַת נַפְשֵׁיהּ כַּד הֲוָה מִתְחַנַּן לַנָא, וְלָא קַבֵּילְנָא מִנֵּיהּ; עַל כֵּין אֲתָת לַנָא, עָקְתָא הָדָא. 
מב,כב וַיַּעַן֩ רְאוּבֵ֨ן אֹתָ֜ם לֵאמֹ֗ר הֲלוֹא֩ אָמַ֨רְתִּי אֲלֵיכֶ֧ם ׀ לֵאמֹ֛ר אַל־תֶּֽחֶטְא֥וּ בַיֶּ֖לֶד וְלֹ֣א שְׁמַעְתֶּ֑ם וְגַם־דָּמ֖וֹ הִנֵּ֥ה נִדְרָֽשׁ׃ וַאֲתֵיב רְאוּבֵן יָתְהוֹן לְמֵימַר, הֲלָא אֲמַרִית לְכוֹן לְמֵימַר לָא תִּחְטוֹן בְּעוּלֵימָא--וְלָא קַבֵּילְתּוּן; וְאַף דְּמֵיהּ, הָא מִתְבְּעֵי. 
מב,כג וְהֵם֙ לֹ֣א יָֽדְע֔וּ כִּ֥י שֹׁמֵ֖עַ יוֹסֵ֑ף כִּ֥י הַמֵּלִ֖יץ בֵּֽינֹתָֽם׃ וְאִנּוּן לָא יָדְעִין, אֲרֵי שָׁמִיעַ יוֹסֵף:  אֲרֵי מְתֻרְגְּמָן, הֲוָה בֵּינֵיהוֹן. 
מב,כד וַיִּסֹּ֥ב מֵֽעֲלֵיהֶ֖ם וַיֵּ֑בְךְּ וַיָּ֤שָׁב אֲלֵהֶם֙ וַיְדַבֵּ֣ר אֲלֵהֶ֔ם וַיִּקַּ֤ח מֵֽאִתָּם֙ אֶת־שִׁמְע֔וֹן וַיֶּֽאֱסֹ֥ר אֹת֖וֹ לְעֵֽינֵיהֶֽם׃ וְאִסְתְּחַר מִלְּוָתְהוֹן, וּבְכָא; וְתָב לְוָתְהוֹן, וּמַלֵּיל עִמְּהוֹן, וּדְבַר מִלְּוָתְהוֹן יָת שִׁמְעוֹן, וַאֲסַר יָתֵיהּ לְעֵינֵיהוֹן. 
מב,כה וַיְצַ֣ו יוֹסֵ֗ף וַיְמַלְא֣וּ אֶת־כְּלֵיהֶם֮ בָּר֒ וּלְהָשִׁ֤יב כַּסְפֵּיהֶם֙ אִ֣ישׁ אֶל־שַׂקּ֔וֹ וְלָתֵ֥ת לָהֶ֛ם צֵדָ֖ה לַדָּ֑רֶךְ וַיַּ֥עַשׂ לָהֶ֖ם כֵּֽן׃ וּפַקֵּיד יוֹסֵף, וּמְלוֹ יָת מָנֵיהוֹן עֲבוּרָא, וּלְאָתָבָא כַּסְפֵּיהוֹן גְּבַר לְסַקֵּיהּ, וּלְמִתַּן לְהוֹן זְוָדִין לְאוֹרְחָא; וַעֲבַד לְהוֹן, כֵּין. 
מב,כו וַיִּשְׂא֥וּ אֶת־שִׁבְרָ֖ם עַל־חֲמֹֽרֵיהֶ֑ם וַיֵּֽלְכ֖וּ מִשָּֽׁם׃ וּנְטַלוּ יָת עֲבוּרְהוֹן, עַל חֲמָרֵיהוֹן; וַאֲזַלוּ, מִתַּמָּן. 
מב,כז וַיִּפְתַּ֨ח הָֽאֶחָ֜ד אֶת־שַׂקּ֗וֹ לָתֵ֥ת מִסְפּ֛וֹא לַֽחֲמֹר֖וֹ בַּמָּל֑וֹן וַיַּרְא֙ אֶת־כַּסְפּ֔וֹ וְהִנֵּה־ה֖וּא בְּפִ֥י אַמְתַּחְתּֽוֹ׃ וּפְתַח חַד יָת סַקֵּיהּ, לְמִתַּן כִּסְּתָא לִחְמָרֵיהּ--בְּבֵית מְבָתָא; וַחֲזָא, יָת כַּסְפֵּיהּ, וְהָא הוּא, בְּפֹם טֻעְנֵיהּ. 
מב,כח וַיֹּ֤אמֶר אֶל־אֶחָיו֙ הוּשַׁ֣ב כַּסְפִּ֔י וְגַ֖ם הִנֵּ֣ה בְאַמְתַּחְתִּ֑י וַיֵּצֵ֣א לִבָּ֗ם וַיֶּֽחֶרְד֞וּ אִ֤ישׁ אֶל־אָחִיו֙ לֵאמֹ֔ר מַה־זֹּ֛את עָשָׂ֥ה אֱלֹהִ֖ים לָֽנוּ׃ וַאֲמַר לַאֲחוֹהִי אִתָּתַב כַּסְפִּי, וְאַף הָא בְּטֻעְנִי; וּנְפַק מַדַּע לִבְּהוֹן, וּתְוַהוּ גְּבַר בַּאֲחוּהִי לְמֵימַר, מָא דָּא עֲבַד יְיָ, לַנָא. 
מב,כט וַיָּבֹ֛אוּ אֶל־יַֽעֲקֹ֥ב אֲבִיהֶ֖ם אַ֣רְצָה כְּנָ֑עַן וַיַּגִּ֣ידוּ ל֔וֹ אֵ֛ת כָּל־הַקֹּרֹ֥ת אֹתָ֖ם לֵאמֹֽר׃ וַאֲתוֹ לְוָת יַעֲקוֹב אֲבוּהוֹן, לְאַרְעָא דִּכְנָעַן; וְחַוִּיאוּ לֵיהּ, יָת כָּל דְּעָרַעָא יָתְהוֹן לְמֵימַר. 
מב,ל דִּ֠בֶּר הָאִ֨ישׁ אֲדֹנֵ֥י הָאָ֛רֶץ אִתָּ֖נוּ קָשׁ֑וֹת וַיִּתֵּ֣ן אֹתָ֔נוּ כִּֽמְרַגְּלִ֖ים אֶת־הָאָֽרֶץ׃ מַלֵּיל גֻּבְרָא רִבּוֹנַהּ דְּאַרְעָא, עִמַּנָא--קַשְׁיָן; וִיהַב יָתַנָא, כִּמְאַלְּלֵי אַרְעָא. 
מב,לא וַנֹּ֥אמֶר אֵלָ֖יו כֵּנִ֣ים אֲנָ֑חְנוּ לֹ֥א הָיִ֖ינוּ מְרַגְּלִֽים׃ וַאֲמַרְנָא לֵיהּ, כֵּיוָנֵי אֲנַחְנָא:  לָא הֲוֵינָא, אַלִּילֵי. 
מב,לב שְׁנֵים־עָשָׂ֥ר אֲנַ֛חְנוּ אַחִ֖ים בְּנֵ֣י אָבִ֑ינוּ הָֽאֶחָ֣ד אֵינֶ֔נּוּ וְהַקָּטֹ֥ן הַיּ֛וֹם אֶת־אָבִ֖ינוּ בְּאֶ֥רֶץ כְּנָֽעַן׃ תְּרֵי עֲסַר אֲנַחְנָא אַחִין, בְּנֵי אֲבוּנָא; חַד לָיְתוֹהִי, וּזְעֵירָא יוֹמָא דֵּין עִם אֲבוּנָא בְּאַרְעָא דִּכְנָעַן. 
מב,לג וַיֹּ֣אמֶר אֵלֵ֗ינוּ הָאִישׁ֙ אֲדֹנֵ֣י הָאָ֔רֶץ בְּזֹ֣את אֵדַ֔ע כִּ֥י כֵנִ֖ים אַתֶּ֑ם אֲחִיכֶ֤ם הָֽאֶחָד֙ הַנִּ֣יחוּ אִתִּ֔י וְאֶת־רַֽעֲב֥וֹן בָּֽתֵּיכֶ֖ם קְח֥וּ וָלֵֽכוּ׃ וַאֲמַר לַנָא, גֻּבְרָא רִבּוֹנַהּ דְּאַרְעָא, בְּדָא אֶדַּע, אֲרֵי כֵּיוָנֵי אַתּוּן:  אֲחוּכוֹן חַד שְׁבוּקוּ לְוָתִי, וְיָת עֲבוּרָא דְּחַסִּיר בְּבָתֵּיכוֹן סַבוּ וְאִיזִילוּ. 
מב,לד וְ֠הָבִיאוּ אֶת־אֲחִיכֶ֣ם הַקָּטֹן֮ אֵלַי֒ וְאֵֽדְעָ֗ה כִּ֣י לֹ֤א מְרַגְּלִים֙ אַתֶּ֔ם כִּ֥י כֵנִ֖ים אַתֶּ֑ם אֶת־אֲחִיכֶם֙ אֶתֵּ֣ן לָכֶ֔ם וְאֶת־הָאָ֖רֶץ תִּסְחָֽרוּ׃ וְאֵיתוֹ יָת אֲחוּכוֹן זְעֵירָא, לְוָתִי, וְאֶדַּע אֲרֵי לָא אַלִּילֵי אַתּוּן, אֲרֵי כֵּיוָנֵי אַתּוּן; יָת אֲחוּכוֹן אֶתֵּין לְכוֹן, וְיָת אַרְעָא תַּעְבְּדוּן בַּהּ סְחוֹרְתָא. 
מב,לה וַיְהִ֗י הֵ֚ם מְרִיקִ֣ים שַׂקֵּיהֶ֔ם וְהִנֵּה־אִ֥ישׁ צְרוֹר־כַּסְפּ֖וֹ בְּשַׂקּ֑וֹ וַיִּרְא֞וּ אֶת־צְרֹר֧וֹת כַּסְפֵּיהֶ֛ם הֵ֥מָּה וַֽאֲבִיהֶ֖ם וַיִּירָֽאוּ׃ וַהֲוָה, אִנּוּן מְרִיקִין סַקֵּיהוֹן, וְהָא גְּבַר צְרָר כַּסְפֵּיהּ, בְּסַקֵּיהּ; וַחֲזוֹ יָת צְרָרֵי כַּסְפֵּיהוֹן, אִנּוּן וַאֲבוּהוֹן--וּדְחִילוּ. 
מב,לו וַיֹּ֤אמֶר אֲלֵהֶם֙ יַֽעֲקֹ֣ב אֲבִיהֶ֔ם אֹתִ֖י שִׁכַּלְתֶּ֑ם יוֹסֵ֤ף אֵינֶ֨נּוּ֙ וְשִׁמְע֣וֹן אֵינֶ֔נּוּ וְאֶת־בִּנְיָמִ֣ן תִּקָּ֔חוּ עָלַ֖י הָי֥וּ כֻלָּֽנָה׃ וַאֲמַר לְהוֹן יַעֲקוֹב אֲבוּהוֹן, יָתִי אַתְכֵּילְתּוּן:  יוֹסֵף לָיְתוֹהִי, וְשִׁמְעוֹן לָא הֲוָה כָּא, וְיָת בִּנְיָמִין תִּדְבְּרוּן, עָלַי הֲוַאָה כֻּלְּהוֹן. 
מב,לז וַיֹּ֤אמֶר רְאוּבֵן֙ אֶל־אָבִ֣יו לֵאמֹ֔ר אֶת־שְׁנֵ֤י בָנַי֙ תָּמִ֔ית אִם־לֹ֥א אֲבִיאֶ֖נּוּ אֵלֶ֑יךָ תְּנָ֤ה אֹתוֹ֙ עַל־יָדִ֔י וַֽאֲנִ֖י אֲשִׁיבֶ֥נּוּ אֵלֶֽיךָ׃ וַאֲמַר רְאוּבֵן, לַאֲבוּהִי לְמֵימַר, יָת תְּרֵין בְּנַי תְּמִית, אִם לָא אַיְתֵינֵיהּ לָךְ; הַב יָתֵיהּ עַל יְדִי, וַאֲנָא אֲתִיבִנֵּיהּ לָךְ. 
מב,לח וַיֹּ֕אמֶר לֹֽא־יֵרֵ֥ד בְּנִ֖י עִמָּכֶ֑ם כִּֽי־אָחִ֨יו מֵ֜ת וְה֧וּא לְבַדּ֣וֹ נִשְׁאָ֗ר וּקְרָאָ֤הוּ אָסוֹן֙ בַּדֶּ֨רֶךְ֙ אֲשֶׁ֣ר תֵּֽלְכוּ־בָ֔הּ וְהֽוֹרַדְתֶּ֧ם אֶת־שֵֽׂיבָתִ֛י בְּיָג֖וֹן שְׁאֽוֹלָה׃ וַאֲמַר, לָא יֵיחוֹת בְּרִי עִמְּכוֹן:  אֲרֵי אֲחוּהִי מִית וְהוּא בִּלְחוֹדוֹהִי אִשְׁתְּאַר, וִיעָרְעִנֵּיהּ מוֹתָא בְּאוֹרְחָא דִּתְהָכוּן בַּהּ, וְתַחֲתוּן יָת סֵיבְתִי בְּדָווֹנָא, לִשְׁאוֹל. 
מג,א וְהָֽרָעָ֖ב כָּבֵ֥ד בָּאָֽרֶץ׃ וְכַפְנָא, תְּקֵיף בְּאַרְעָא. 
מג,ב וַיְהִ֗י כַּֽאֲשֶׁ֤ר כִּלּוּ֙ לֶֽאֱכֹ֣ל אֶת־הַשֶּׁ֔בֶר אֲשֶׁ֥ר הֵבִ֖יאוּ מִמִּצְרָ֑יִם וַיֹּ֤אמֶר אֲלֵיהֶם֙ אֲבִיהֶ֔ם שֻׁ֖בוּ שִׁבְרוּ־לָ֥נוּ מְעַט־אֹֽכֶל׃ וַהֲוָה, כַּד סְפִיקוּ לְמֵיכַל יָת עֲבוּרָא, דְּאֵיתִיאוּ, מִמִּצְרָיִם; וַאֲמַר לְהוֹן אֲבוּהוֹן, תּוּבוּ זְבוּנוּ לַנָא זְעֵיר עֲבוּרָא. 
מג,ג וַיֹּ֧אמֶר אֵלָ֛יו יְהוּדָ֖ה לֵאמֹ֑ר הָעֵ֣ד הֵעִד֩ בָּ֨נוּ הָאִ֤ישׁ לֵאמֹר֙ לֹֽא־תִרְא֣וּ פָנַ֔י בִּלְתִּ֖י אֲחִיכֶ֥ם אִתְּכֶֽם׃ וַאֲמַר לֵיהּ יְהוּדָה, לְמֵימַר:  אַסְהָדָא אַסְהֵיד בַּנָא גֻּבְרָא לְמֵימַר לָא תִּחְזוֹן אַפַּי, אֱלָהֵין כַּד אֲחוּכוֹן עִמְּכוֹן. 
מג,ד אִם־יֶשְׁךָ֛ מְשַׁלֵּ֥חַ אֶת־אָחִ֖ינוּ אִתָּ֑נוּ נֵֽרְדָ֕ה וְנִשְׁבְּרָ֥ה לְךָ֖ אֹֽכֶל׃ אִם אִיתָךְ מְשַׁלַּח יָת אֲחוּנָא, עִמַּנָא--נֵיחוֹת, וְנִזְבּוֹן לָךְ עֲבוּרָא. 
מג,ה וְאִם־אֵֽינְךָ֥ מְשַׁלֵּ֖חַ לֹ֣א נֵרֵ֑ד כִּֽי־הָאִ֞ישׁ אָמַ֤ר אֵלֵ֨ינוּ֙ לֹֽא־תִרְא֣וּ פָנַ֔י בִּלְתִּ֖י אֲחִיכֶ֥ם אִתְּכֶֽם׃ וְאִם לָיְתָךְ מְשַׁלַּח, לָא נֵיחוֹת:  אֲרֵי גֻּבְרָא אֲמַר לַנָא, לָא תִּחְזוֹן אַפַּי, אֱלָהֵין, כַּד אֲחוּכוֹן עִמְּכוֹן. 
מג,ו וַיֹּ֨אמֶר֙ יִשְׂרָאֵ֔ל לָמָ֥ה הֲרֵֽעֹתֶ֖ם לִ֑י לְהַגִּ֣יד לָאִ֔ישׁ הַע֥וֹד לָכֶ֖ם אָֽח׃ וַאֲמַר, יִשְׂרָאֵל, לְמָא אַבְאֵישְׁתּוּן, לִי--לְחַוָּאָה לְגֻבְרָא, הַעוֹד לְכוֹן אֲחָא. 
מג,ז וַיֹּֽאמְר֡וּ שָׁא֣וֹל שָֽׁאַל־הָ֠אִישׁ לָ֣נוּ וּלְמֽוֹלַדְתֵּ֜נוּ לֵאמֹ֗ר הַע֨וֹד אֲבִיכֶ֥ם חַי֙ הֲיֵ֣שׁ לָכֶ֣ם אָ֔ח וַנַּ֨גֶּד־ל֔וֹ עַל־פִּ֖י הַדְּבָרִ֣ים הָאֵ֑לֶּה הֲיָד֣וֹעַ נֵדַ֔ע כִּ֣י יֹאמַ֔ר הוֹרִ֖ידוּ אֶת־אֲחִיכֶֽם׃ וַאֲמַרוּ שַׁאָלָא שַׁאֵיל גֻּבְרָא לַנָא וּלְיַלָּדוּתַנָא לְמֵימַר, הַעַד כְּעַן אֲבוּכוֹן קַיָּם הַאִית לְכוֹן אֲחָא, וְחַוִּינָא לֵיהּ, עַל מֵימַר פִּתְגָמַיָּא הָאִלֵּין; הֲמֵידָע הֲוֵינָא יָדְעִין--אֲרֵי יֵימַר, אַחִיתוּ יָת אֲחוּכוֹן. 
מג,ח וַיֹּ֨אמֶר יְהוּדָ֜ה אֶל־יִשְׂרָאֵ֣ל אָבִ֗יו שִׁלְחָ֥ה הַנַּ֛עַר אִתִּ֖י וְנָק֣וּמָה וְנֵלֵ֑כָה וְנִֽחְיֶה֙ וְלֹ֣א נָמ֔וּת גַּם־אֲנַ֥חְנוּ גַם־אַתָּ֖ה גַּם־טַפֵּֽנוּ׃ וַאֲמַר יְהוּדָה לְיִשְׂרָאֵל אֲבוּהִי, שְׁלַח עוּלֵימָא עִמִּי--וּנְקוּם וְנֵיזֵיל; וְנֵיחֵי וְלָא נְמוּת, אַף אֲנַחְנָא אַף אַתְּ אַף טַפְלַנָא. 
מג,ט אָֽנֹכִי֙ אֶֽעֶרְבֶ֔נּוּ מִיָּדִ֖י תְּבַקְשֶׁ֑נּוּ אִם־לֹ֨א הֲבִֽיאֹתִ֤יו אֵלֶ֨יךָ֙ וְהִצַּגְתִּ֣יו לְפָנֶ֔יךָ וְחָטָ֥אתִֽי לְךָ֖ כָּל־הַיָּמִֽים׃ אֲנָא, מְעָרַבְנָא בֵּיהּ--מִן יְדִי, תִּבְעֵינֵיהּ:  אִם לָא אַיְתֵינֵיהּ לָךְ וַאֲקִימִנֵּיהּ קֳדָמָךְ, וְאֵיהֵי חָטֵי לָךְ כָּל יוֹמַיָּא. 
מג,י כִּ֖י לוּלֵ֣א הִתְמַהְמָ֑הְנוּ כִּֽי־עַתָּ֥ה שַׁ֖בְנוּ זֶ֥ה פַֽעֲמָֽיִם׃ אֲרֵי, אִלּוּ לָא פּוֹן בְּדָא אִתְעַכַּבְנָא--אֲרֵי כְּעַן תַּבְנָא, דְּנָן תַּרְתֵּין זִמְנִין. 
מג,יא וַיֹּ֨אמֶר אֲלֵהֶ֜ם יִשְׂרָאֵ֣ל אֲבִיהֶ֗ם אִם־כֵּ֣ן ׀ אֵפוֹא֮ זֹ֣את עֲשׂוּ֒ קְח֞וּ מִזִּמְרַ֤ת הָאָ֨רֶץ֙ בִּכְלֵיכֶ֔ם וְהוֹרִ֥ידוּ לָאִ֖ישׁ מִנְחָ֑ה מְעַ֤ט צֳרִי֙ וּמְעַ֣ט דְּבַ֔שׁ נְכֹ֣את וָלֹ֔ט בָּטְנִ֖ים וּשְׁקֵדִֽים׃ וַאֲמַר לְהוֹן יִשְׂרָאֵל אֲבוּהוֹן, אִם כֵּין הוּא דָּא עֲבִידוּ--סַבוּ מִדִּמְשֻׁבַּח בְּאַרְעָא בְּמָנֵיכוֹן, וַאֲחִיתוּ לְגֻבְרָא תִּקְרֻבְתָּא:  זְעֵיר קְטַף, וּזְעֵיר דְּבַשׁ, שְׁעַף וּלְטוֹם, בֻּטְמִין וְשִׁגְדִּין. 
מג,יב וְכֶ֥סֶף מִשְׁנֶ֖ה קְח֣וּ בְיֶדְכֶ֑ם וְאֶת־הַכֶּ֜סֶף הַמּוּשָׁ֨ב בְּפִ֤י אַמְתְּחֹֽתֵיכֶם֙ תָּשִׁ֣יבוּ בְיֶדְכֶ֔ם אוּלַ֥י מִשְׁגֶּ֖ה הֽוּא׃ וְכַסְפָּא עַל חַד תְּרֵין, סַבוּ בְּיַדְכוֹן; וְיָת כַּסְפָּא דְּאִתָּתַב בְּפֹם טֻעְנֵיכוֹן, תָּתִיבוּן בְּיַדְכוֹן--דִּלְמָא שָׁלוּ, הֲוָת. 
מג,יג וְאֶת־אֲחִיכֶ֖ם קָ֑חוּ וְק֖וּמוּ שׁ֥וּבוּ אֶל־הָאִֽישׁ׃ וְיָת אֲחוּכוֹן, דְּבַרוּ; וְקוּמוּ, תּוּבוּ לְוָת גֻּבְרָא. 
מג,יד וְאֵ֣ל שַׁדַּ֗י יִתֵּ֨ן לָכֶ֤ם רַֽחֲמִים֙ לִפְנֵ֣י הָאִ֔ישׁ וְשִׁלַּ֥ח לָכֶ֛ם אֶת־אֲחִיכֶ֥ם אַחֵ֖ר וְאֶת־בִּנְיָמִ֑ין וַֽאֲנִ֕י כַּֽאֲשֶׁ֥ר שָׁכֹ֖לְתִּי שָׁכָֽלְתִּי׃ וְאֵל שַׁדַּי, יִתֵּין לְכוֹן רַחֲמִין קֳדָם גֻּבְרָא, וְיִפְטַר לְכוֹן יָת אֲחוּכוֹן אֻחְרָנָא, וְיָת בִּנְיָמִין; וַאֲנָא, כְּמָא דִּתְכֵילִית תְּכֵילִית. 
מג,טו וַיִּקְח֤וּ הָֽאֲנָשִׁים֙ אֶת־הַמִּנְחָ֣ה הַזֹּ֔את וּמִשְׁנֶה־כֶּ֛סֶף לָֽקְח֥וּ בְיָדָ֖ם וְאֶת־בִּנְיָמִ֑ן וַיָּקֻ֨מוּ֙ וַיֵּֽרְד֣וּ מִצְרַ֔יִם וַיַּֽעַמְד֖וּ לִפְנֵ֥י יוֹסֵֽף׃ וּנְסִיבוּ גֻּבְרַיָּא יָת תִּקְרֻבְתָּא הָדָא, וְעַל חַד תְּרֵין כַּסְפָּא נְסִיבוּ בְּיַדְהוֹן וּדְבַרוּ יָת בִּנְיָמִין; וְקָמוּ וּנְחַתוּ לְמִצְרַיִם, וְקָמוּ קֳדָם יוֹסֵף. 
מג,טז וַיַּ֨רְא יוֹסֵ֣ף אִתָּם֮ אֶת־בִּנְיָמִין֒ וַיֹּ֨אמֶר֙ לַֽאֲשֶׁ֣ר עַל־בֵּית֔וֹ הָבֵ֥א אֶת־הָֽאֲנָשִׁ֖ים הַבָּ֑יְתָה וּטְבֹ֤חַ טֶ֨בַח֙ וְהָכֵ֔ן כִּ֥י אִתִּ֛י יֹֽאכְל֥וּ הָֽאֲנָשִׁ֖ים בַּֽצָּהֳרָֽיִם׃ וַחֲזָא יוֹסֵף עִמְּהוֹן, יָת בִּנְיָמִין, וַאֲמַר לְדִמְמֻנַּא עַל בֵּיתֵיהּ, אַעֵיל יָת גֻּבְרַיָּא לְבֵיתָא; וְכוֹס נִכְסְתָא וְאַתְקֵין, אֲרֵי עִמִּי אָכְלִין גֻּבְרַיָּא בְּשֵׁירוּתָא. 
מג,יז וַיַּ֣עַשׂ הָאִ֔ישׁ כַּֽאֲשֶׁ֖ר אָמַ֣ר יוֹסֵ֑ף וַיָּבֵ֥א הָאִ֛ישׁ אֶת־הָֽאֲנָשִׁ֖ים בֵּ֥יתָה יוֹסֵֽף׃ וַעֲבַד גֻּבְרָא, כְּמָא דַּאֲמַר יוֹסֵף; וְאַעֵיל גֻּבְרָא יָת גֻּבְרַיָּא, לְבֵית יוֹסֵף. 
מג,יח וַיִּֽירְא֣וּ הָֽאֲנָשִׁ֗ים כִּ֣י הֽוּבְאוּ֮ בֵּ֣ית יוֹסֵף֒ וַיֹּֽאמְר֗וּ עַל־דְּבַ֤ר הַכֶּ֨סֶף֙ הַשָּׁ֤ב בְּאַמְתְּחֹתֵ֨ינוּ֙ בַּתְּחִלָּ֔ה אֲנַ֖חְנוּ מֽוּבָאִ֑ים לְהִתְגֹּלֵ֤ל עָלֵ֨ינוּ֙ וּלְהִתְנַפֵּ֣ל עָלֵ֔ינוּ וְלָקַ֧חַת אֹתָ֛נוּ לַֽעֲבָדִ֖ים וְאֶת־חֲמֹרֵֽינוּ׃ וּדְחִילוּ גֻּבְרַיָּא, אֲרֵי אִתָּעֲלוּ לְבֵית יוֹסֵף, וַאֲמַרוּ עַל עֵיסַק כַּסְפָּא דְּאִתָּתַב בְּטֻעְנַנָא בְּקַדְמֵיתָא, אֲנַחְנָא מִתָּעֲלִין--לְאִתְרַבְרָבָא עֲלַנָא וּלְאִסְתַּקָּפָא עֲלַנָא, וּלְמִקְנֵי יָתַנָא לְעַבְדִּין וּלְמִדְבַּר יָת חֲמָרַנָא. 
מג,יט וַֽיִּגְּשׁוּ֙ אֶל־הָאִ֔ישׁ אֲשֶׁ֖ר עַל־בֵּ֣ית יוֹסֵ֑ף וַיְדַבְּר֥וּ אֵלָ֖יו פֶּ֥תַח הַבָּֽיִת׃ וּקְרִיבוּ, לְוָת גֻּבְרָא, דִּמְמֻנַּא, עַל בֵּית יוֹסֵף; וּמַלִּילוּ עִמֵּיהּ, בִּתְרַע בֵּיתָא. 
מג,כ וַיֹּֽאמְר֖וּ בִּ֣י אֲדֹנִ֑י יָרֹ֥ד יָרַ֛דְנוּ בַּתְּחִלָּ֖ה לִשְׁבָּר־אֹֽכֶל׃ וַאֲמַרוּ, בְּבָעוּ רִבּוֹנִי; מֵיחָת נְחַתְנָא בְּקַדְמֵיתָא, לְמִזְבַּן עֲבוּרָא. 
מג,כא וַיְהִ֞י כִּי־בָ֣אנוּ אֶל־הַמָּל֗וֹן וַֽנִּפְתְּחָה֙ אֶת־אַמְתְּחֹתֵ֔ינוּ וְהִנֵּ֤ה כֶֽסֶף־אִישׁ֙ בְּפִ֣י אַמְתַּחְתּ֔וֹ כַּסְפֵּ֖נוּ בְּמִשְׁקָל֑וֹ וַנָּ֥שֶׁב אֹת֖וֹ בְּיָדֵֽנוּ׃ וַהֲוָה כַּד אֲתֵינָא לְבֵית מְבָתָא, וּפְתַחְנָא יָת טֻעְנַנָא, וְהָא כְּסַף גְּבַר בְּפֹם טֻעְנֵיהּ, כַּסְפַּנָא בְּמַתְקָלֵיהּ; וַאֲתֵיבְנָא יָתֵיהּ, בִּידַנָא. 
מג,כב וְכֶ֧סֶף אַחֵ֛ר הוֹרַ֥דְנוּ בְיָדֵ֖נוּ לִשְׁבָּר־אֹ֑כֶל לֹ֣א יָדַ֔עְנוּ מִי־שָׂ֥ם כַּסְפֵּ֖נוּ בְּאַמְתְּחֹתֵֽינוּ׃ וְכַסְפָּא אֻחְרָנָא אַחֵיתְנָא בִּידַנָא, לְמִזְבַּן עֲבוּרָא; לָא יְדַעְנָא, מַאן שַׁוִּי כַּסְפַּנָא בְּטֻעְנַנָא. 
מג,כג וַיֹּאמֶר֩ שָׁל֨וֹם לָכֶ֜ם אַל־תִּירָ֗אוּ אֱלֹ֨הֵיכֶ֜ם וֵֽאלֹהֵ֤י אֲבִיכֶם֙ נָתַ֨ן לָכֶ֤ם מַטְמוֹן֙ בְּאַמְתְּחֹ֣תֵיכֶ֔ם כַּסְפְּכֶ֖ם בָּ֣א אֵלָ֑י וַיּוֹצֵ֥א אֲלֵהֶ֖ם אֶת־שִׁמְעֽוֹן׃ וַאֲמַר שְׁלָם לְכוֹן לָא תִּדְחֲלוּן, אֱלָהֲכוֹן וֵאלָהָא דַּאֲבוּכוֹן יְהַב לְכוֹן סִימָן בְּטֻעְנֵיכוֹן--כַּסְפְּכוֹן, אֲתָא לְוָתִי; וְאַפֵּיק לְוָתְהוֹן, יָת שִׁמְעוֹן. 
מג,כד וַיָּבֵ֥א הָאִ֛ישׁ אֶת־הָֽאֲנָשִׁ֖ים בֵּ֣יתָה יוֹסֵ֑ף וַיִּתֶּן־מַ֨יִם֙ וַיִּרְחֲצ֣וּ רַגְלֵיהֶ֔ם וַיִּתֵּ֥ן מִסְפּ֖וֹא לַחֲמֹֽרֵיהֶֽם׃ וְאַעֵיל גֻּבְרָא יָת גֻּבְרַיָּא, לְבֵית יוֹסֵף; וִיהַב מַיָּא וְאַסְחוֹ רַגְלֵיהוֹן, וִיהַב כִּסְּתָא לִחְמָרֵיהוֹן. 
מג,כה וַיָּכִ֨ינוּ֙ אֶת־הַמִּנְחָ֔ה עַד־בּ֥וֹא יוֹסֵ֖ף בַּֽצָּהֳרָ֑יִם כִּ֣י שָֽׁמְע֔וּ כִּי־שָׁ֖ם יֹ֥אכְלוּ לָֽחֶם׃ וְאַתְקִינוּ, יָת תִּקְרֻבְתָּא, עַד דְּעָאל יוֹסֵף, בְּשֵׁירוּתָא:  אֲרֵי שְׁמַעוּ, אֲרֵי תַּמָּן אָכְלִין לַחְמָא. 
מג,כו וַיָּבֹ֤א יוֹסֵף֙ הַבַּ֔יְתָה וַיָּבִ֥יאּוּ ל֛וֹ אֶת־הַמִּנְחָ֥ה אֲשֶׁר־בְּיָדָ֖ם הַבָּ֑יְתָה וַיִּשְׁתַּֽחֲווּ־ל֖וֹ אָֽרְצָה׃ וְעָאל יוֹסֵף לְבֵיתָא, וְאַעִילוּ לֵיהּ יָת תִּקְרֻבְתָּא דִּבְיַדְהוֹן לְבֵיתָא; וּסְגִידוּ לֵיהּ, עַל אַרְעָא. 
מג,כז וַיִּשְׁאַ֤ל לָהֶם֙ לְשָׁל֔וֹם וַיֹּ֗אמֶר הֲשָׁל֛וֹם אֲבִיכֶ֥ם הַזָּקֵ֖ן אֲשֶׁ֣ר אֲמַרְתֶּ֑ם הַֽעוֹדֶ֖נּוּ חָֽי׃ וּשְׁאֵיל לְהוֹן, לִשְׁלָם, וַאֲמַר הַשְׁלָם אֲבוּכוֹן סָבָא, דַּאֲמַרְתּוּן--הַעַד כְּעַן, קַיָּם. 
מג,כח וַיֹּֽאמְר֗וּ שָׁל֛וֹם לְעַבְדְּךָ֥ לְאָבִ֖ינוּ עוֹדֶ֣נּוּ חָ֑י וַֽיִּקְּד֖וּ וישתחו (וַיִּֽשְׁתַּחֲוֽוּ)׃ וַאֲמַרוּ, שְׁלָם לְעַבְדָּךְ לַאֲבוּנָא--עַד כְּעַן קַיָּם; וּכְרַעוּ, וּסְגִידוּ. 
מג,כט וַיִּשָּׂ֣א עֵינָ֗יו וַיַּ֞רְא אֶת־בִּנְיָמִ֣ין אָחִיו֮ בֶּן־אִמּוֹ֒ וַיֹּ֗אמֶר הֲזֶה֙ אֲחִיכֶ֣ם הַקָּטֹ֔ן אֲשֶׁ֥ר אֲמַרְתֶּ֖ם אֵלָ֑י וַיֹּאמַ֕ר אֱלֹהִ֥ים יָחְנְךָ֖ בְּנִֽי׃ וּזְקַף עֵינוֹהִי, וַחֲזָא יָת בִּנְיָמִין אֲחוּהִי בַּר אִמֵּיהּ, וַאֲמַר הֲדֵין אֲחוּכוֹן זְעֵירָא, דַּאֲמַרְתּוּן לִי; וַאֲמַר, מִן קֳדָם יְיָ יִתְרַחַם עֲלָךְ בְּרִי. 
מג,ל וַיְמַהֵ֣ר יוֹסֵ֗ף כִּֽי־נִכְמְר֤וּ רַֽחֲמָיו֙ אֶל־אָחִ֔יו וַיְבַקֵּ֖שׁ לִבְכּ֑וֹת וַיָּבֹ֥א הַחַ֖דְרָה וַיֵּ֥בְךְּ שָֽׁמָּה׃ וְאוֹחִי יוֹסֵף, אֲרֵי אִתְגּוֹלַלוּ רַחֲמוֹהִי עַל אֲחוּהִי, וּבְעָא, לְמִבְכֵּי; וְעָאל לְאִדְּרוֹן בֵּית מִשְׁכְּבָא, וּבְכָא תַּמָּן. 
מג,לא וַיִּרְחַ֥ץ פָּנָ֖יו וַיֵּצֵ֑א וַיִּ֨תְאַפַּ֔ק וַיֹּ֖אמֶר שִׂ֥ימוּ לָֽחֶם׃ וְאַסְחִי אַפּוֹהִי, וּנְפַק; וְאִתְחַסַּן--וַאֲמַר, שַׁוּוֹ לַחְמָא. 
מג,לב וַיָּשִׂ֥ימוּ ל֛וֹ לְבַדּ֖וֹ וְלָהֶ֣ם לְבַדָּ֑ם וְלַמִּצְרִ֞ים הָאֹֽכְלִ֤ים אִתּוֹ֙ לְבַדָּ֔ם כִּי֩ לֹ֨א יֽוּכְל֜וּן הַמִּצְרִ֗ים לֶֽאֱכֹ֤ל אֶת־הָֽעִבְרִים֙ לֶ֔חֶם כִּֽי־תוֹעֵבָ֥ה הִ֖וא לְמִצְרָֽיִם׃ וְשַׁוִּיאוּ לֵיהּ בִּלְחוֹדוֹהִי, וּלְהוֹן בִּלְחוֹדֵיהוֹן; וּלְמִצְרָאֵי דְּאָכְלִין עִמֵּיהּ, בִּלְחוֹדֵיהוֹן--אֲרֵי לָא יָכְלִין מִצְרָאֵי לְמֵיכַל עִם עִבְרָאֵי לַחְמָא, אֲרֵי בְּעִירָא דְּמִצְרָאֵי דָּחֲלִין לֵיהּ עִבְרָאֵי אָכְלִין. 
מג,לג וַיֵּֽשְׁב֣וּ לְפָנָ֔יו הַבְּכֹר֙ כִּבְכֹ֣רָת֔וֹ וְהַצָּעִ֖יר כִּצְעִֽרָת֑וֹ וַיִּתְמְה֥וּ הָֽאֲנָשִׁ֖ים אִ֥ישׁ אֶל־רֵעֵֽהוּ׃ וְאַסְחַרוּ קֳדָמוֹהִי--רַבָּא כְּרַבְיוּתֵיהּ, וּזְעֵירָא כִּזְעֵירוּתֵיהּ; וּתְוַהוּ גֻּבְרַיָּא, גְּבַר בְּחַבְרֵיהּ. 
מג,לד וַיִּשָּׂ֨א מַשְׂאֹ֜ת מֵאֵ֣ת פָּנָיו֮ אֲלֵהֶם֒ וַתֵּ֜רֶב מַשְׂאַ֧ת בִּנְיָמִ֛ן מִמַּשְׂאֹ֥ת כֻּלָּ֖ם חָמֵ֣שׁ יָד֑וֹת וַיִּשְׁתּ֥וּ וַֽיִּשְׁכְּר֖וּ עִמּֽוֹ׃ וּנְטַל חוּלָקִין מִן קֳדָמוֹהִי, לִקְדָמֵיהוֹן, וּסְגִי חוּלָקָא דְּבִנְיָמִין מֵחוּלָקֵי כֻּלְּהוֹן, חַמְשָׁא חוּלָקִין; וּשְׁתִיאוּ וּרְוִיאוּ, עִמֵּיהּ. 
מד,א וַיְצַ֞ו אֶת־אֲשֶׁ֣ר עַל־בֵּיתוֹ֮ לֵאמֹר֒ מַלֵּ֞א אֶת־אַמְתְּחֹ֤ת הָֽאֲנָשִׁים֙ אֹ֔כֶל כַּֽאֲשֶׁ֥ר יֽוּכְל֖וּן שְׂאֵ֑ת וְשִׂ֥ים כֶּֽסֶף־אִ֖ישׁ בְּפִ֥י אַמְתַּחְתּֽוֹ׃ וּפַקֵּיד יָת דִּמְמֻנַּא עַל בֵּיתֵיהּ, לְמֵימַר, מַלִּי יָת טֻעְנֵי גֻּבְרַיָּא עֲבוּרָא, כְּמָא דְּיָכְלִין לְמִטְעַן; וְשַׁו כְּסַף גְּבַר, בְּפֹם טֻעְנֵיהּ. 
מד,ב וְאֶת־גְּבִיעִ֞י גְּבִ֣יעַ הַכֶּ֗סֶף תָּשִׂים֙ בְּפִי֙ אַמְתַּ֣חַת הַקָּטֹ֔ן וְאֵ֖ת כֶּ֣סֶף שִׁבְר֑וֹ וַיַּ֕עַשׂ כִּדְבַ֥ר יוֹסֵ֖ף אֲשֶׁ֥ר דִּבֵּֽר׃ וְיָת כַּלִּידִי כַּלִּידָא דְּכַסְפָּא, תְּשַׁוֵּי בְּפֹם טֻעְנָא דִּזְעֵירָא, וְיָת, כְּסַף זְבִינוֹהִי; וַעֲבַד, כְּפִתְגָמָא דְּיוֹסֵף דְּמַלֵּיל. 
מד,ג הַבֹּ֖קֶר א֑וֹר וְהָֽאֲנָשִׁ֣ים שֻׁלְּח֔וּ הֵ֖מָּה וַחֲמֹֽרֵיהֶֽם׃ צַפְרָא, נְהַר; וְגֻבְרַיָּא אִתְפְּטַרוּ, אִנּוּן וּחְמָרֵיהוֹן. 
מד,ד הֵ֠ם יָֽצְא֣וּ אֶת־הָעִיר֮ לֹ֣א הִרְחִיקוּ֒ וְיוֹסֵ֤ף אָמַר֙ לַֽאֲשֶׁ֣ר עַל־בֵּית֔וֹ ק֥וּם רְדֹ֖ף אַֽחֲרֵ֣י הָֽאֲנָשִׁ֑ים וְהִשַּׂגְתָּם֙ וְאָֽמַרְתָּ֣ אֲלֵהֶ֔ם לָ֛מָּה שִׁלַּמְתֶּ֥ם רָעָ֖ה תַּ֥חַת טוֹבָֽה׃ אִנּוּן נְפַקוּ מִן קַרְתָּא, לָא אַרְחִיקוּ, וְיוֹסֵף אֲמַר לְדִמְמֻנַּא עַל בֵּיתֵיהּ, קוּם רְדוֹף בָּתַר גֻּבְרַיָּא; וְתַדְבֵּיקִנּוּן וְתֵימַר לְהוֹן, לְמָא שַׁלֵּימְתּוּן בִּשְׁתָּא חֲלָף טָבְתָא. 
מד,ה הֲל֣וֹא זֶ֗ה אֲשֶׁ֨ר יִשְׁתֶּ֤ה אֲדֹנִי֙ בּ֔וֹ וְה֕וּא נַחֵ֥שׁ יְנַחֵ֖שׁ בּ֑וֹ הֲרֵֽעֹתֶ֖ם אֲשֶׁ֥ר עֲשִׂיתֶֽם׃ הֲלָא דֵּין, דְּשָׁתֵי רִבּוֹנִי בֵּיהּ, וְהוּא, בַּדָּקָא מְבַדֵּיק בֵּיהּ; אַבְאֵישְׁתּוּן, דַּעֲבַדְתּוּן. 
מד,ו וַֽיַּשִּׂגֵ֑ם וַיְדַבֵּ֣ר אֲלֵהֶ֔ם אֶת־הַדְּבָרִ֖ים הָאֵֽלֶּה׃ וְאַדְבֵּיקִנּוּן; וּמַלֵּיל עִמְּהוֹן, יָת פִּתְגָמַיָּא הָאִלֵּין. 
מד,ז וַיֹּֽאמְר֣וּ אֵלָ֔יו לָ֚מָּה יְדַבֵּ֣ר אֲדֹנִ֔י כַּדְּבָרִ֖ים הָאֵ֑לֶּה חָלִ֨ילָה֙ לַֽעֲבָדֶ֔יךָ מֵֽעֲשׂ֖וֹת כַּדָּבָ֥ר הַזֶּֽה׃ וַאֲמַרוּ לֵיהּ--לְמָא יְמַלֵּיל רִבּוֹנִי, כְּפִתְגָמַיָּא הָאִלֵּין; חַס, לְעַבְדָּךְ, מִלְּמֶעֱבַד, כְּפִתְגָמָא הָדֵין. 
מד,ח הֵ֣ן כֶּ֗סֶף אֲשֶׁ֤ר מָצָ֨אנוּ֙ בְּפִ֣י אַמְתְּחֹתֵ֔ינוּ הֱשִׁיבֹ֥נוּ אֵלֶ֖יךָ מֵאֶ֣רֶץ כְּנָ֑עַן וְאֵ֗יךְ נִגְנֹב֙ מִבֵּ֣ית אֲדֹנֶ֔יךָ כֶּ֖סֶף א֥וֹ זָהָֽב׃ הָא כַּסְפָּא, דְּאַשְׁכַּחְנָא בְּפֹם טֻעְנַנָא--אֲתֵיבְנָהִי לָךְ, מֵאַרְעָא דִּכְנָעַן; וְאֵיכְדֵין, נִגְנוֹב מִבֵּית רִבּוֹנָךְ, מָנִין דִּכְסַף, אוֹ מָנִין דִּדְהַב. 
מד,ט אֲשֶׁ֨ר יִמָּצֵ֥א אִתּ֛וֹ מֵֽעֲבָדֶ֖יךָ וָמֵ֑ת וְגַם־אֲנַ֕חְנוּ נִֽהְיֶ֥ה לַֽאדֹנִ֖י לַֽעֲבָדִֽים׃ דְּיִשְׁתְּכַח עִמֵּיהּ מֵעַבְדָּךְ, יִתְקְטִיל; וְאַף אֲנַחְנָא, נְהֵי לְרִבּוֹנִי לְעַבְדִּין. 
מד,י וַיֹּ֕אמֶר גַּם־עַתָּ֥ה כְדִבְרֵיכֶ֖ם כֶּן־ה֑וּא אֲשֶׁ֨ר יִמָּצֵ֤א אִתּוֹ֙ יִֽהְיֶה־לִּ֣י עָ֔בֶד וְאַתֶּ֖ם תִּֽהְי֥וּ נְקִיִּֽם׃ וַאֲמַר, אַף כְּעַן כְּפִתְגָמֵיכוֹן כֵּין הוּא:  דְּיִשְׁתְּכַח עִמֵּיהּ יְהֵי לִי עַבְדָּא, וְאַתּוּן תְּהוֹן זַכָּאִין. 
מד,יא וַֽיְמַהֲר֗וּ וַיּוֹרִ֛דוּ אִ֥ישׁ אֶת־אַמְתַּחְתּ֖וֹ אָ֑רְצָה וַֽיִּפְתְּח֖וּ אִ֥ישׁ אַמְתַּחְתּֽוֹ׃ וְאוֹחִיאוּ, וְאַחִיתוּ גְּבַר יָת טֻעְנֵיהּ--לְאַרְעָא; וּפְתַחוּ, גְּבַר טֻעְנֵיהּ. 
מד,יב וַיְחַפֵּ֕שׂ בַּגָּד֣וֹל הֵחֵ֔ל וּבַקָּטֹ֖ן כִּלָּ֑ה וַיִּמָּצֵא֙ הַגָּבִ֔יעַ בְּאַמְתַּ֖חַת בִּנְיָמִֽן׃ וּבְלַשׁ--בְּרַבָּא שָׁרִי, וּבִזְעֵירָא שֵׁיצִי; וְאִשְׁתְּכַח, כַּלִּידָא, בְּטֻעְנָא, דְּבִנְיָמִין. 
מד,יג וַֽיִּקְרְע֖וּ שִׂמְלֹתָ֑ם וַֽיַּעֲמֹס֙ אִ֣ישׁ עַל־חֲמֹר֔וֹ וַיָּשֻׁ֖בוּ הָעִֽירָה׃ וּבַזַּעוּ, לְבוּשֵׁיהוֹן; וּרְמוֹ גְּבַר עַל חֲמָרֵיהּ, וְתָבוּ לְקַרְתָּא. 
מד,יד וַיָּבֹ֨א יְהוּדָ֤ה וְאֶחָיו֙ בֵּ֣יתָה יוֹסֵ֔ף וְה֖וּא עוֹדֶ֣נּוּ שָׁ֑ם וַיִּפְּל֥וּ לְפָנָ֖יו אָֽרְצָה׃ וְעָאל יְהוּדָה וַאֲחוֹהִי לְבֵית יוֹסֵף, וְהוּא עַד כְּעַן תַּמָּן; וּנְפַלוּ קֳדָמוֹהִי, עַל אַרְעָא. 
מד,טו וַיֹּ֤אמֶר לָהֶם֙ יוֹסֵ֔ף מָֽה־הַמַּעֲשֶׂ֥ה הַזֶּ֖ה אֲשֶׁ֣ר עֲשִׂיתֶ֑ם הֲל֣וֹא יְדַעְתֶּ֔ם כִּֽי־נַחֵ֧שׁ יְנַחֵ֛שׁ אִ֖ישׁ אֲשֶׁ֥ר כָּמֹֽנִי׃ וַאֲמַר לְהוֹן יוֹסֵף, מָא עוּבָדָא הָדֵין דַּעֲבַדְתּוּן; הֲלָא יְדַעְתּוּן, אֲרֵי בַּדָּקָא מְבַדֵּיק גֻּבְרָא דִּכְוָתִי. 
מד,טז וַיֹּ֣אמֶר יְהוּדָ֗ה מַה־נֹּאמַר֙ לַֽאדֹנִ֔י מַה־נְּדַבֵּ֖ר וּמַה־נִּצְטַדָּ֑ק הָֽאֱלֹהִ֗ים מָצָא֙ אֶת־עֲו֣‍ֹן עֲבָדֶ֔יךָ הִנֶּ֤נּוּ עֲבָדִים֙ לַֽאדֹנִ֔י גַּם־אֲנַ֕חְנוּ גַּ֛ם אֲשֶׁר־נִמְצָ֥א הַגָּבִ֖יעַ בְּיָדֽוֹ׃ וַאֲמַר יְהוּדָה, מָא נֵימַר לְרִבּוֹנִי, מָא נְמַלֵּיל, וּמָא נִזְכֵּי; מִן קֳדָם יְיָ, אִשְׁתְּכַח חוֹבָא בְּעַבְדָּךְ--הָא אֲנַחְנָא עַבְדִּין לְרִבּוֹנִי, אַף אֲנַחְנָא אַף דְּאִשְׁתְּכַח כַּלִּידָא בִּידֵיהּ. 
מד,יז וַיֹּ֕אמֶר חָלִ֣ילָה לִּ֔י מֵֽעֲשׂ֖וֹת זֹ֑את הָאִ֡ישׁ אֲשֶׁר֩ נִמְצָ֨א הַגָּבִ֜יעַ בְּיָד֗וֹ ה֚וּא יִֽהְיֶה־לִּ֣י עָ֔בֶד וְאַתֶּ֕ם עֲל֥וּ לְשָׁל֖וֹם אֶל־אֲבִיכֶֽם׃ {ס} וַאֲמַר--חַס לִי, מִלְּמֶעֱבַד דָּא; גֻּבְרָא דְּאִשְׁתְּכַח כַּלִּידָא בִּידֵיהּ, הוּא יְהֵי לִי עַבְדָּא, וְאַתּוּן, סַקוּ לִשְׁלָם לְוָת אֲבוּכוֹן. {ס}
