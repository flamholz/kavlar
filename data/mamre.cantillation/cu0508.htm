ניצבים

כט,ט אַתֶּ֨ם נִצָּבִ֤ים הַיּוֹם֙ כֻּלְּכֶ֔ם לִפְנֵ֖י יְהוָ֣ה אֱלֹֽהֵיכֶ֑ם רָֽאשֵׁיכֶ֣ם שִׁבְטֵיכֶ֗ם זִקְנֵיכֶם֙ וְשֹׁ֣טְרֵיכֶ֔ם כֹּ֖ל אִ֥ישׁ יִשְׂרָאֵֽל׃ אַתּוּן קָיְמִין יוֹמָא דֵּין כֻּלְּכוֹן, קֳדָם יְיָ אֱלָהָכוֹן:  רֵישֵׁיכוֹן שִׁבְטֵיכוֹן, סָבֵיכוֹן וְסָרְכֵיכוֹן, כֹּל, אֲנָשׁ יִשְׂרָאֵל. 
כט,י טַפְּכֶ֣ם נְשֵׁיכֶ֔ם וְגֵ֣רְךָ֔ אֲשֶׁ֖ר בְּקֶ֣רֶב מַֽחֲנֶ֑יךָ מֵֽחֹטֵ֣ב עֵצֶ֔יךָ עַ֖ד שֹׁאֵ֥ב מֵימֶֽיךָ׃ טַפְלְכוֹן נְשֵׁיכוֹן--וְגִיּוֹרָךְ, דִּבְגוֹ מַשְׁרִיתָךְ:  מִלָּקֵיט אָעָךְ, עַד מָלֵי מַיָּךְ. 
כט,יא לְעָבְרְךָ֗ בִּבְרִ֛ית יְהוָ֥ה אֱלֹהֶ֖יךָ וּבְאָֽלָת֑וֹ אֲשֶׁר֙ יְהוָ֣ה אֱלֹהֶ֔יךָ כֹּרֵ֥ת עִמְּךָ֖ הַיּֽוֹם׃ לְאַעָלוּתָךְ, בִּקְיָמָא דַּייָ אֱלָהָךְ--וּבְמוֹמָתֵיהּ:  דַּייָ אֱלָהָךְ, גָּזַר עִמָּךְ יוֹמָא דֵּין. 
כט,יב לְמַ֣עַן הָקִֽים־אֹתְךָ֩ הַיּ֨וֹם ׀ ל֜וֹ לְעָ֗ם וְה֤וּא יִֽהְיֶה־לְּךָ֙ לֵֽאלֹהִ֔ים כַּֽאֲשֶׁ֖ר דִּבֶּר־לָ֑ךְ וְכַֽאֲשֶׁ֤ר נִשְׁבַּע֙ לַֽאֲבֹתֶ֔יךָ לְאַבְרָהָ֥ם לְיִצְחָ֖ק וּֽלְיַעֲקֹֽב׃ בְּדִיל לְקַיָּמָא יָתָךְ יוֹמָא דֵּין קֳדָמוֹהִי לְעַם, וְהוּא יִהְוֵי לָךְ לֶאֱלָהּ--כְּמָא דְּמַלֵּיל לָךְ; וּכְמָא דְּקַיֵּים לַאֲבָהָתָךְ, לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקוֹב. 
כט,יג וְלֹ֥א אִתְּכֶ֖ם לְבַדְּכֶ֑ם אָֽנֹכִ֗י כֹּרֵת֙ אֶת־הַבְּרִ֣ית הַזֹּ֔את וְאֶת־הָֽאָלָ֖ה הַזֹּֽאת׃ וְלָא עִמְּכוֹן, בִּלְחוֹדֵיכוֹן--אֲנָא, גָּזַר יָת קְיָמָא הָדֵין, וְיָת מוֹמָתָא, הָדָא. 
כט,יד כִּי֩ אֶת־אֲשֶׁ֨ר יֶשְׁנ֜וֹ פֹּ֗ה עִמָּ֨נוּ֙ עֹמֵ֣ד הַיּ֔וֹם לִפְנֵ֖י יְהוָ֣ה אֱלֹהֵ֑ינוּ וְאֵ֨ת אֲשֶׁ֥ר אֵינֶ֛נּוּ פֹּ֖ה עִמָּ֥נוּ הַיּֽוֹם׃ אֲרֵי יָת מַאן דְּאִיתוֹהִי הָכָא, עִמַּנָא קָאֵים יוֹמָא דֵּין, קֳדָם, יְיָ אֱלָהַנָא; וְיָת מַאן דְּלָיְתוֹהִי הָכָא, עִמַּנָא יוֹמָא דֵּין. 
כט,טו כִּֽי־אַתֶּ֣ם יְדַעְתֶּ֔ם אֵ֥ת אֲשֶׁר־יָשַׁ֖בְנוּ בְּאֶ֣רֶץ מִצְרָ֑יִם וְאֵ֧ת אֲשֶׁר־עָבַ֛רְנוּ בְּקֶ֥רֶב הַגּוֹיִ֖ם אֲשֶׁ֥ר עֲבַרְתֶּֽם׃ אֲרֵי אַתּוּן יְדַעְתּוּן, יָת דִּיתֵיבְנָא בְּאַרְעָא דְּמִצְרָיִם, וְיָת דַּעֲבַרְנָא בֵּינֵי עַמְמַיָּא, דַּעֲבַרְתּוּן. 
כט,טז וַתִּרְאוּ֙ אֶת־שִׁקּ֣וּצֵיהֶ֔ם וְאֵ֖ת גִּלֻּֽלֵיהֶ֑ם עֵ֣ץ וָאֶ֔בֶן כֶּ֥סֶף וְזָהָ֖ב אֲשֶׁ֥ר עִמָּהֶֽם׃ וַחֲזֵיתוֹן, יָת שִׁקּוּצֵיהוֹן, וְיָת, טָעֲוָתְהוֹן--אָעָא וְאַבְנָא, כַּסְפָּא וְדַהְבָּא דְּעִמְּהוֹן. 
כט,יז פֶּן־יֵ֣שׁ בָּ֠כֶם אִ֣ישׁ אֽוֹ־אִשָּׁ֞ה א֧וֹ מִשְׁפָּחָ֣ה אוֹ־שֵׁ֗בֶט אֲשֶׁר֩ לְבָב֨וֹ פֹנֶ֤ה הַיּוֹם֙ מֵעִם֙ יְהוָ֣ה אֱלֹהֵ֔ינוּ לָלֶ֣כֶת לַֽעֲבֹ֔ד אֶת־אֱלֹהֵ֖י הַגּוֹיִ֣ם הָהֵ֑ם פֶּן־יֵ֣שׁ בָּכֶ֗ם שֹׁ֛רֶשׁ פֹּרֶ֥ה רֹ֖אשׁ וְלַֽעֲנָֽה׃ דִּלְמָא אִית בְּכוֹן גְּבַר אוֹ אִתָּא אוֹ זַרְעִי אוֹ שִׁבְטָא, דְּלִבֵּיהּ פְּנִי יוֹמָא דֵּין מִדַּחְלְתָא דַּייָ אֱלָהַנָא, לִמְהָךְ לְמִפְלַח, יָת טָעֲוָת עַמְמַיָּא הָאִנּוּן:  דִּלְמָא אִית בְּכוֹן, גְּבַר מְהַרְהֵיר חֲטִין--אוֹ זָדוֹן. 
כט,יח וְהָיָ֡ה בְּשָׁמְעוֹ֩ אֶת־דִּבְרֵ֨י הָֽאָלָ֜ה הַזֹּ֗את וְהִתְבָּרֵ֨ךְ בִּלְבָב֤וֹ לֵאמֹר֙ שָׁל֣וֹם יִֽהְיֶה־לִּ֔י כִּ֛י בִּשְׁרִר֥וּת לִבִּ֖י אֵלֵ֑ךְ לְמַ֛עַן סְפ֥וֹת הָֽרָוָ֖ה אֶת־הַצְּמֵאָֽה׃ וִיהֵי בְּמִשְׁמְעֵיהּ יָת פִּתְגָמֵי מוֹמָתָא הָדָא, וִיחַשֵּׁיב בְּלִבֵּיהּ לְמֵימַר שְׁלָמָא יְהֵי לִי--אֲרֵי בְּהַרְהוּר לִבִּי, אֲנָא אָזֵיל:  בְּדִיל לְאוֹסָפָא לֵיהּ חֲטָאֵי שָׁלוּתָא, עַל זֵידָנוּתָא. 
כט,יט לֹֽא־יֹאבֶ֣ה יְהוָה֮ סְלֹ֣חַֽ לוֹ֒ כִּ֣י אָ֠ז יֶעְשַׁ֨ן אַף־יְהוָ֤ה וְקִנְאָתוֹ֙ בָּאִ֣ישׁ הַה֔וּא וְרָ֤בְצָה בּוֹ֙ כָּל־הָ֣אָלָ֔ה הַכְּתוּבָ֖ה בַּסֵּ֣פֶר הַזֶּ֑ה וּמָחָ֤ה יְהוָה֙ אֶת־שְׁמ֔וֹ מִתַּ֖חַת הַשָּׁמָֽיִם׃ לָא יֵיבֵי יְיָ, לְמִשְׁבַּק לֵיהּ--אֲרֵי בְּכֵין יִתְקַף רֻגְזָא דַּייָ וְחִמְתֵּיהּ בְּגֻבְרָא הַהוּא, וְיִדְבְּקוּן בֵּיהּ כָּל לְוָטַיָּא דִּכְתִיבִין בְּסִפְרָא הָדֵין; וְיִמְחֵי יְיָ יָת שְׁמֵיהּ, מִתְּחוֹת שְׁמַיָּא. 
כט,כ וְהִבְדִּיל֤וֹ יְהוָה֙ לְרָעָ֔ה מִכֹּ֖ל שִׁבְטֵ֣י יִשְׂרָאֵ֑ל כְּכֹל֙ אָל֣וֹת הַבְּרִ֔ית הַכְּתוּבָ֕ה בְּסֵ֥פֶר הַתּוֹרָ֖ה הַזֶּֽה׃ וְיַפְרְשִׁנֵּיהּ יְיָ לְבִישָׁא, מִכֹּל שִׁבְטַיָּא דְּיִשְׂרָאֵל--כְּכֹל, לְוָטֵי קְיָמָא, דִּכְתִיבִין, בִּסְפַר אוֹרָיְתָא הָדֵין. 
כט,כא וְאָמַ֞ר הַדּ֣וֹר הָאַֽחֲר֗וֹן בְּנֵיכֶם֙ אֲשֶׁ֤ר יָק֨וּמוּ֙ מֵאַ֣חֲרֵיכֶ֔ם וְהַ֨נָּכְרִ֔י אֲשֶׁ֥ר יָבֹ֖א מֵאֶ֣רֶץ רְחוֹקָ֑ה וְ֠רָאוּ אֶת־מַכּ֞וֹת הָאָ֤רֶץ הַהִוא֙ וְאֶת־תַּ֣חֲלֻאֶ֔יהָ אֲשֶׁר־חִלָּ֥ה יְהוָ֖ה בָּֽהּ׃ וְיֵימַר דָּרָא בָּתְרָאָה, בְּנֵיכוֹן דִּיקוּמוּן מִבָּתְרֵיכוֹן, וּבַר עַמְמִין, דְּיֵיתֵי מֵאֲרַע רַחִיקָא; וְיִחְזוֹן יָת מַחָתַהּ דְּאַרְעָא הַהִיא, וְיָת מַרְעַהָא, דְּאַמְרַע יְיָ, בַּהּ. 
כט,כב גָּפְרִ֣ית וָמֶלַח֮ שְׂרֵפָ֣ה כָל־אַרְצָהּ֒ לֹ֤א תִזָּרַע֙ וְלֹ֣א תַצְמִ֔חַ וְלֹֽא־יַעֲלֶ֥ה בָ֖הּ כָּל־עֵ֑שֶׂב כְּֽמַהְפֵּכַ֞ת סְדֹ֤ם וַֽעֲמֹרָה֙ אַדְמָ֣ה וּצְבֹיִ֔ים אֲשֶׁר֙ הָפַ֣ךְ יְהוָ֔ה בְּאַפּ֖וֹ וּבַֽחֲמָתֽוֹ׃ גֻּפְרֵיתָא וּמִלְחָא, תְּהֵי יָקְדָא כָּל אַרְעָהּ--לָא תִּזְדְּרַע וְלָא תְּצַמַּח, וְלָא יִסַּק בַּהּ כָּל עֵיסַב:  כַּהֲפֵיכְתָא דִּסְדוֹם וַעֲמוֹרָה, אַדְמָה וּצְבוֹיִים, דַּהֲפַךְ יְיָ, בְּרֻגְזֵיהּ וּבְחִמְתֵּיהּ. 
כט,כג וְאָֽמְרוּ֙ כָּל־הַגּוֹיִ֔ם עַל־מֶ֨ה עָשָׂ֧ה יְהוָ֛ה כָּ֖כָה לָאָ֣רֶץ הַזֹּ֑את מֶ֥ה חֳרִ֛י הָאַ֥ף הַגָּד֖וֹל הַזֶּֽה׃ וְיֵימְרוּן, כָּל עַמְמַיָּא, עַל מָא עֲבַד יְיָ כְּדֵין, לְאַרְעָא הָדָא; מָא תְּקוֹף רֻגְזָא רַבָּא, הָדֵין. 
כט,כד וְאָ֣מְר֔וּ עַ֚ל אֲשֶׁ֣ר עָֽזְב֔וּ אֶת־בְּרִ֥ית יְהוָ֖ה אֱלֹהֵ֣י אֲבֹתָ֑ם אֲשֶׁר֙ כָּרַ֣ת עִמָּ֔ם בְּהֽוֹצִיא֥וֹ אֹתָ֖ם מֵאֶ֥רֶץ מִצְרָֽיִם׃ וְיֵימְרוּן--עַל דִּשְׁבַקוּ, יָת קְיָמָא דַּייָ אֱלָהָא דַּאֲבָהָתְהוֹן:  דִּגְזַר עִמְּהוֹן, בְּאַפָּקוּתֵיהּ יָתְהוֹן מֵאַרְעָא דְּמִצְרָיִם. 
כט,כה וַיֵּֽלְכ֗וּ וַיַּֽעַבְדוּ֙ אֱלֹהִ֣ים אֲחֵרִ֔ים וַיִּֽשְׁתַּחֲו֖וּ לָהֶ֑ם אֱלֹהִים֙ אֲשֶׁ֣ר לֹֽא־יְדָע֔וּם וְלֹ֥א חָלַ֖ק לָהֶֽם׃ וַאֲזַלוּ, וּפְלַחוּ לְטָעֲוָת עַמְמַיָּא, וּסְגִידוּ, לְהוֹן:  דַּחְלָן דְּלָא יְדַעוּנִין, וְלָא אוֹטִיבָא לְהוֹן. 
כט,כו וַיִּֽחַר־אַ֥ף יְהוָ֖ה בָּאָ֣רֶץ הַהִ֑וא לְהָבִ֤יא עָלֶ֨יהָ֙ אֶת־כָּל־הַקְּלָלָ֔ה הַכְּתוּבָ֖ה בַּסֵּ֥פֶר הַזֶּֽה׃ וּתְקֵיף רֻגְזָא דַּייָ, בְּאַרְעָא הַהִיא, לְאֵיתָאָה עֲלַהּ יָת כָּל לְוָטַיָּא, דִּכְתִיבִין בְּסִפְרָא הָדֵין. 
כט,כז וַיִּתְּשֵׁ֤ם יְהוָה֙ מֵעַ֣ל אַדְמָתָ֔ם בְּאַ֥ף וּבְחֵמָ֖ה וּבְקֶ֣צֶף גָּד֑וֹל וַיַּשְׁלִכֵ֛ם אֶל־אֶ֥רֶץ אַחֶ֖רֶת כַּיּ֥וֹם הַזֶּֽה׃ וְטַלְטֵילִנּוּן יְיָ מֵעַל אֲרַעְהוֹן, בִּרְגַז וּבִחְמָא וּבִתְקוֹף רַב; וְאַגְלִינוּן לַאֲרַע אֻחְרִי, כְּיוֹמָא הָדֵין. 
כט,כח הַנִּ֨סְתָּרֹ֔ת לַֽיהוָ֖ה אֱלֹהֵ֑ינוּ וְהַנִּגְלֹ֞ת לָ֤ׄנׄוּׄ וּׄלְׄבָׄנֵ֨ׄיׄנׄוּ֙ׄ עַׄד־עוֹלָ֔ם לַֽעֲשׂ֕וֹת אֶת־כָּל־דִּבְרֵ֖י הַתּוֹרָ֥ה הַזֹּֽאת׃ {ס} דְּמִטַּמְרָן--קֳדָם יְיָ, אֱלָהַנָא; וּדְגַלְיָן לַנָא וְלִבְנַנָא, עַד עָלְמָא--לְמֶעֱבַד, יָת כָּל פִּתְגָמֵי אוֹרָיְתָא הָדָא. {ס}
ל,א וְהָיָה֩ כִֽי־יָבֹ֨אוּ עָלֶ֜יךָ כָּל־הַדְּבָרִ֣ים הָאֵ֗לֶּה הַבְּרָכָה֙ וְהַקְּלָלָ֔ה אֲשֶׁ֥ר נָתַ֖תִּי לְפָנֶ֑יךָ וַהֲשֵֽׁבֹתָ֙ אֶל־לְבָבֶ֔ךָ בְּכָ֨ל־הַגּוֹיִ֔ם אֲשֶׁ֧ר הִדִּֽיחֲךָ֛ יְהוָ֥ה אֱלֹהֶ֖יךָ שָֽׁמָּה׃ וִיהֵי אֲרֵי יֵיתוֹן עֲלָךְ כָּל פִּתְגָמַיָּא הָאִלֵּין, בִּרְכָן וּלְוָטִין, דִּיהַבִית, קֳדָמָךְ; וְתָתִיב, לְלִבָּךְ, בְּכָל עַמְמַיָּא, דְּאַגְלְיָךְ יְיָ אֱלָהָךְ לְתַמָּן. 
ל,ב וְשַׁבְתָּ֞ עַד־יְהוָ֤ה אֱלֹהֶ֨יךָ֙ וְשָֽׁמַעְתָּ֣ בְקֹל֔וֹ כְּכֹ֛ל אֲשֶׁר־אָֽנֹכִ֥י מְצַוְּךָ֖ הַיּ֑וֹם אַתָּ֣ה וּבָנֶ֔יךָ בְּכָל־לְבָֽבְךָ֖ וּבְכָל־נַפְשֶֽׁךָ׃ וּתְתוּב לְדַחְלְתָא דַּייָ אֱלָהָךְ, וּתְקַבֵּיל לְמֵימְרֵיהּ, כְּכֹל דַּאֲנָא מְפַקֵּיד לָךְ, יוֹמָא דֵּין:  אַתְּ וּבְנָךְ, בְּכָל לִבָּךְ וּבְכָל נַפְשָׁךְ. 
ל,ג וְשָׁ֨ב יְהוָ֧ה אֱלֹהֶ֛יךָ אֶת־שְׁבֽוּתְךָ֖ וְרִֽחֲמֶ֑ךָ וְשָׁ֗ב וְקִבֶּצְךָ֙ מִכָּל־הָ֣עַמִּ֔ים אֲשֶׁ֧ר הֱפִֽיצְךָ֛ יְהוָ֥ה אֱלֹהֶ֖יךָ שָֽׁמָּה׃ וְיָתִיב יְיָ אֱלָהָךְ יָת שָׁבֵי גָּלְוָתָךְ, וִירַחֵים עֲלָךְ; וִיתוּב, וְיִכְנְשִׁנָּךְ מִכָּל עַמְמַיָּא, דְּבַדְּרָךְ יְיָ אֱלָהָךְ, לְתַמָּן. 
ל,ד אִם־יִֽהְיֶ֥ה נִֽדַּחֲךָ֖ בִּקְצֵ֣ה הַשָּׁמָ֑יִם מִשָּׁ֗ם יְקַבֶּצְךָ֙ יְהוָ֣ה אֱלֹהֶ֔יךָ וּמִשָּׁ֖ם יִקָּחֶֽךָ׃ אִם יְהוֹן גָּלְוָתָךְ, בִּסְיָפֵי שְׁמַיָּא--מִתַּמָּן, יִכְנְשִׁנָּךְ יְיָ אֱלָהָךְ, וּמִתַּמָּן, יְקָרְבִנָּךְ. 
ל,ה וֶהֱבִֽיאֲךָ֞ יְהוָ֣ה אֱלֹהֶ֗יךָ אֶל־הָאָ֛רֶץ אֲשֶׁר־יָֽרְשׁ֥וּ אֲבֹתֶ֖יךָ וִֽירִשְׁתָּ֑הּ וְהֵיטִֽבְךָ֥ וְהִרְבְּךָ֖ מֵֽאֲבֹתֶֽיךָ׃ וְיַעֵילִנָּךְ יְיָ אֱלָהָךְ, לְאַרְעָא דִּירִיתוּ אֲבָהָתָךְ--וְתֵירְתַהּ; וְיוֹטֵיב לָךְ וְיַסְגֵּינָךְ, מֵאֲבָהָתָךְ. 
ל,ו וּמָ֨ל יְהוָ֧ה אֱלֹהֶ֛יךָ אֶת־לְבָֽבְךָ֖ וְאֶת־לְבַ֣ב זַרְעֶ֑ךָ לְאַֽהֲבָ֞ה אֶת־יְהוָ֧ה אֱלֹהֶ֛יךָ בְּכָל־לְבָֽבְךָ֥ וּבְכָל־נַפְשְׁךָ֖ לְמַ֥עַן חַיֶּֽיךָ׃ וְיַעְדֵּי יְיָ אֱלָהָךְ יָת טַפְשׁוּת לִבָּךְ, וְיָת טַפְשׁוּת לִבָּא דִּבְנָךְ:  לְמִרְחַם יָת יְיָ אֱלָהָךְ, בְּכָל לִבָּךְ וּבְכָל נַפְשָׁךְ--בְּדִיל חַיָּךְ. 
ל,ז וְנָתַן֙ יְהוָ֣ה אֱלֹהֶ֔יךָ אֵ֥ת כָּל־הָֽאָל֖וֹת הָאֵ֑לֶּה עַל־אֹֽיְבֶ֥יךָ וְעַל־שֹֽׂנְאֶ֖יךָ אֲשֶׁ֥ר רְדָפֽוּךָ׃ וְיִתֵּין יְיָ אֱלָהָךְ, יָת כָּל לְוָטַיָּא הָאִלֵּין, עַל בַּעֲלֵי דְּבָבָךְ וְעַל סָנְאָךְ, דְּרַדְפוּךְ. 
ל,ח וְאַתָּ֣ה תָשׁ֔וּב וְשָֽׁמַעְתָּ֖ בְּק֣וֹל יְהוָ֑ה וְעָשִׂ֨יתָ֙ אֶת־כָּל־מִצְו‍ֹתָ֔יו אֲשֶׁ֛ר אָֽנֹכִ֥י מְצַוְּךָ֖ הַיּֽוֹם׃ וְאַתְּ תְּתוּב, וּתְקַבֵּיל לְמֵימְרָא דַּייָ; וְתַעֲבֵיד, יָת כָּל פִּקּוֹדוֹהִי, דַּאֲנָא מְפַקֵּיד לָךְ, יוֹמָא דֵּין. 
ל,ט וְהוֹתִֽירְךָ֩ יְהוָ֨ה אֱלֹהֶ֜יךָ בְּכֹ֣ל ׀ מַֽעֲשֵׂ֣ה יָדֶ֗ךָ בִּפְרִ֨י בִטְנְךָ֜ וּבִפְרִ֧י בְהֶמְתְּךָ֛ וּבִפְרִ֥י אַדְמָֽתְךָ֖ לְטֹבָ֑ה כִּ֣י ׀ יָשׁ֣וּב יְהוָ֗ה לָשׂ֤וּשׂ עָלֶ֨יךָ֙ לְט֔וֹב כַּֽאֲשֶׁר־שָׂ֖שׂ עַל־אֲבֹתֶֽיךָ׃ וְיוֹתְרִנָּךְ יְיָ אֱלָהָךְ בְּכֹל עוּבָדֵי יְדָךְ, בְּוַלְדָּא דִּמְעָךְ וּבְוַלְדָּא דִּבְעִירָךְ וּבְאִבָּא דְּאַרְעָךְ--לְטָבָא:  אֲרֵי יְתוּב יְיָ, לְמִחְדֵּי עֲלָךְ לְטָב, כְּמָא דַּחְדִּי, עַל אֲבָהָתָךְ. 
ל,י כִּ֣י תִשְׁמַ֗ע בְּקוֹל֙ יְהוָ֣ה אֱלֹהֶ֔יךָ לִשְׁמֹ֤ר מִצְו‍ֹתָיו֙ וְחֻקֹּתָ֔יו הַכְּתוּבָ֕ה בְּסֵ֥פֶר הַתּוֹרָ֖ה הַזֶּ֑ה כִּ֤י תָשׁוּב֙ אֶל־יְהוָ֣ה אֱלֹהֶ֔יךָ בְּכָל־לְבָֽבְךָ֖ וּבְכָל־נַפְשֶֽׁךָ׃ {ס} אֲרֵי תְּקַבֵּיל, לְמֵימְרָא דַּייָ אֱלָהָךְ, לְמִטַּר פִּקּוֹדוֹהִי וּקְיָמוֹהִי, דִּכְתִיבִין בִּסְפַר אוֹרָיְתָא הָדֵין:  אֲרֵי תְּתוּב לְדַחְלְתָא דַּייָ אֱלָהָךְ, בְּכָל לִבָּךְ וּבְכָל נַפְשָׁךְ. {ס}
ל,יא כִּ֚י הַמִּצְוָ֣ה הַזֹּ֔את אֲשֶׁ֛ר אָֽנֹכִ֥י מְצַוְּךָ֖ הַיּ֑וֹם לֹֽא־נִפְלֵ֥את הִוא֙ מִמְּךָ֔ וְלֹ֥א רְחֹקָ֖ה הִֽוא׃ אֲרֵי תַּפְקֵידְתָּא הָדָא, דַּאֲנָא מְפַקֵּיד לָךְ יוֹמָא דֵּין--לָא מְפָרְשָׁא הִיא מִנָּךְ, וְלָא רַחִיקָא הִיא. 
ל,יב לֹ֥א בַשָּׁמַ֖יִם הִ֑וא לֵאמֹ֗ר מִ֣י יַֽעֲלֶה־לָּ֤נוּ הַשָּׁמַ֨יְמָה֙ וְיִקָּחֶ֣הָ לָּ֔נוּ וְיַשְׁמִעֵ֥נוּ אֹתָ֖הּ וְנַֽעֲשֶֽׂנָּה׃ לָא בִּשְׁמַיָּא, הִיא:  לְמֵימַר, מַאן יִסַּק לַנָא לִשְׁמַיָּא וְיִסְּבַהּ לַנָא, וְיַשְׁמְעִנַּנָא יָתַהּ, וְנַעְבְּדִנַּהּ. 
ל,יג וְלֹֽא־מֵעֵ֥בֶר לַיָּ֖ם הִ֑וא לֵאמֹ֗ר מִ֣י יַֽעֲבָר־לָ֜נוּ אֶל־עֵ֤בֶר הַיָּם֙ וְיִקָּחֶ֣הָ לָּ֔נוּ וְיַשְׁמִעֵ֥נוּ אֹתָ֖הּ וְנַֽעֲשֶֽׂנָּה׃ וְלָא מֵעִבְרָא לְיַמָּא, הִיא:  לְמֵימַר, מַאן יִעְבַּר לַנָא לְעֵיבַר יַמָּא וְיִסְּבַהּ לַנָא, וְיַשְׁמְעִנַּנָא יָתַהּ, וְנַעְבְּדִנַּהּ. 
ל,יד כִּֽי־קָר֥וֹב אֵלֶ֛יךָ הַדָּבָ֖ר מְאֹ֑ד בְּפִ֥יךָ וּבִֽלְבָבְךָ֖ לַֽעֲשֹׂתֽוֹ׃ {ס} אֲרֵי קָרִיב לָךְ פִּתְגָמָא, לַחְדָּא:  בְּפֻמָּךְ וּבְלִבָּךְ, לְמִעְבְּדֵיהּ. {ס}
ל,טו רְאֵ֨ה נָתַ֤תִּי לְפָנֶ֨יךָ֙ הַיּ֔וֹם אֶת־הַֽחַיִּ֖ים וְאֶת־הַטּ֑וֹב וְאֶת־הַמָּ֖וֶת וְאֶת־הָרָֽע׃ חֲזִי דִּיהַבִית קֳדָמָךְ יוֹמָא דֵּין, יָת חַיֵּי וְיָת טָבְתָא, וְיָת מוֹתָא, וְיָת בִּשְׁתָּא. 
ל,טז אֲשֶׁ֨ר אָֽנֹכִ֣י מְצַוְּךָ֮ הַיּוֹם֒ לְאַֽהֲבָ֞ה אֶת־יְהוָ֤ה אֱלֹהֶ֨יךָ֙ לָלֶ֣כֶת בִּדְרָכָ֔יו וְלִשְׁמֹ֛ר מִצְו‍ֹתָ֥יו וְחֻקֹּתָ֖יו וּמִשְׁפָּטָ֑יו וְחָיִ֣יתָ וְרָבִ֔יתָ וּבֵֽרַכְךָ֙ יְהוָ֣ה אֱלֹהֶ֔יךָ בָּאָ֕רֶץ אֲשֶׁר־אַתָּ֥ה בָא־שָׁ֖מָּה לְרִשְׁתָּֽהּ׃ דַּאֲנָא מְפַקֵּיד לָךְ, יוֹמָא דֵּין, לְמִרְחַם יָת יְיָ אֱלָהָךְ לִמְהָךְ בְּאוֹרְחָן דְּתָקְנָן קֳדָמוֹהִי, וּלְמִטַּר פִּקּוֹדוֹהִי וּקְיָמוֹהִי וְדִינוֹהִי; וְתֵיחֵי וְתִסְגֵּי--וִיבָרְכִנָּךְ יְיָ אֱלָהָךְ, בְּאַרְעָא דְּאַתְּ עָלֵיל לְתַמָּן לְמֵירְתַהּ. 
ל,יז וְאִם־יִפְנֶ֥ה לְבָֽבְךָ֖ וְלֹ֣א תִשְׁמָ֑ע וְנִדַּחְתָּ֗ וְהִֽשְׁתַּחֲוִ֛יתָ לֵֽאלֹהִ֥ים אֲחֵרִ֖ים וַֽעֲבַדְתָּֽם׃ וְאִם יִתְפְּנֵי לִבָּךְ, וְלָא תְּקַבֵּיל; וְתִטְעֵי, וְתִסְגּוֹד לְטָעֲוָת עַמְמַיָּא--וְתִפְלְחִנִּין. 
ל,יח הִגַּ֤דְתִּי לָכֶם֙ הַיּ֔וֹם כִּ֥י אָבֹ֖ד תֹּֽאבֵד֑וּן לֹֽא־תַאֲרִיכֻ֤ן יָמִים֙ עַל־הָ֣אֲדָמָ֔ה אֲשֶׁ֨ר אַתָּ֤ה עֹבֵר֙ אֶת־הַיַּרְדֵּ֔ן לָב֥וֹא שָׁ֖מָּה לְרִשְׁתָּֽהּ׃ חַוִּיתִי לְכוֹן יוֹמָא דֵּין, אֲרֵי מֵיבַד תֵּיבְדוּן:  לָא תֵּירְכוּן יוֹמִין, עַל אַרְעָא, דְּאַתְּ עָבַר יָת יַרְדְּנָא, לְמֵיעַל לְתַמָּן לְמֵירְתַהּ. 
ל,יט הַֽעִדֹ֨תִי בָכֶ֣ם הַיּוֹם֮ אֶת־הַשָּׁמַ֣יִם וְאֶת־הָאָרֶץ֒ הַֽחַיִּ֤ים וְהַמָּ֨וֶת֙ נָתַ֣תִּי לְפָנֶ֔יךָ הַבְּרָכָ֖ה וְהַקְּלָלָ֑ה וּבָֽחַרְתָּ֙ בַּֽחַיִּ֔ים לְמַ֥עַן תִּֽחְיֶ֖ה אַתָּ֥ה וְזַרְעֶֽךָ׃ אַסְהֵידִית בְּכוֹן יוֹמָא דֵּין, יָת שְׁמַיָּא וְיָת אַרְעָא--חַיֵּי וּמוֹתָא יְהַבִית קֳדָמָךְ, בִּרְכָן וּלְוָטִין; וְתִתְרְעֵי, בְּחַיֵּי--בְּדִיל דְּתֵיחֵי, אַתְּ וּבְנָךְ. 
ל,כ לְאַֽהֲבָה֙ אֶת־יְהוָ֣ה אֱלֹהֶ֔יךָ לִשְׁמֹ֥עַ בְּקֹל֖וֹ וּלְדָבְקָה־ב֑וֹ כִּ֣י ה֤וּא חַיֶּ֨יךָ֙ וְאֹ֣רֶךְ יָמֶ֔יךָ לָשֶׁ֣בֶת עַל־הָֽאֲדָמָ֗ה אֲשֶׁר֩ נִשְׁבַּ֨ע יְהוָ֧ה לַֽאֲבֹתֶ֛יךָ לְאַבְרָהָ֛ם לְיִצְחָ֥ק וּֽלְיַעֲקֹ֖ב לָתֵ֥ת לָהֶֽם׃ {פ} לְמִרְחַם יָת יְיָ אֱלָהָךְ, לְקַבָּלָא לְמֵימְרֵיהּ וּלְאִתְקָרָבָא לְדַחְלְתֵיהּ:  אֲרֵי הוּא חַיָּךְ, וְאוֹרָכוּת יוֹמָךְ--לְמִתַּב עַל אַרְעָא דְּקַיֵּים יְיָ לַאֲבָהָתָךְ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקוֹב, לְמִתַּן לְהוֹן. {פ}
