בהר סיניי

כה,א וַיְדַבֵּ֤ר יְהוָה֙ אֶל־מֹשֶׁ֔ה בְּהַ֥ר סִינַ֖י לֵאמֹֽר׃ וּמַלֵּיל יְיָ עִם מֹשֶׁה, בְּטוּרָא דְּסִינַי לְמֵימַר. 
כה,ב דַּבֵּ֞ר אֶל־בְּנֵ֤י יִשְׂרָאֵל֙ וְאָֽמַרְתָּ֣ אֲלֵהֶ֔ם כִּ֤י תָבֹ֨אוּ֙ אֶל־הָאָ֔רֶץ אֲשֶׁ֥ר אֲנִ֖י נֹתֵ֣ן לָכֶ֑ם וְשָֽׁבְתָ֣ה הָאָ֔רֶץ שַׁבָּ֖ת לַֽיהוָֽה׃ מַלֵּיל עִם בְּנֵי יִשְׂרָאֵל, וְתֵימַר לְהוֹן, אֲרֵי תֵּיעֲלוּן לְאַרְעָא, דַּאֲנָא יָהֵיב לְכוֹן--וְתַשְׁמֵיט אַרְעָא, שְׁמִטְּתָא קֳדָם יְיָ. 
כה,ג שֵׁ֤שׁ שָׁנִים֙ תִּזְרַ֣ע שָׂדֶ֔ךָ וְשֵׁ֥שׁ שָׁנִ֖ים תִּזְמֹ֣ר כַּרְמֶ֑ךָ וְאָֽסַפְתָּ֖ אֶת־תְּבֽוּאָתָֽהּ׃ שֵׁית שְׁנִין תִּזְרַע חַקְלָךְ, וְשֵׁית שְׁנִין תִּכְסַח כַּרְמָךְ; וְתִכְנוֹשׁ, יָת עֲלַלְתַּהּ. 
כה,ד וּבַשָּׁנָ֣ה הַשְּׁבִיעִ֗ת שַׁבַּ֤ת שַׁבָּתוֹן֙ יִֽהְיֶ֣ה לָאָ֔רֶץ שַׁבָּ֖ת לַֽיהוָ֑ה שָֽׂדְךָ֙ לֹ֣א תִזְרָ֔ע וְכַרְמְךָ֖ לֹ֥א תִזְמֹֽר׃ וּבְשַׁתָּא שְׁבִיעֵיתָא, נְיָח שְׁמִטְּתָא יְהֵי לְאַרְעָא--דְּתַשְׁמֵיט, קֳדָם יְיָ:  חַקְלָךְ לָא תִּזְרַע, וְכַרְמָךְ לָא תִּכְסַח. 
כה,ה אֵ֣ת סְפִ֤יחַ קְצִֽירְךָ֙ לֹ֣א תִקְצ֔וֹר וְאֶת־עִנְּבֵ֥י נְזִירֶ֖ךָ לֹ֣א תִבְצֹ֑ר שְׁנַ֥ת שַׁבָּת֖וֹן יִֽהְיֶ֥ה לָאָֽרֶץ׃ יָת כָּתֵי חֲצָדָךְ לָא תִּחְצוּד, וְיָת עִנְּבֵי שִׁבְקָךְ לָא תִּקְטוּף:  שְׁנַת שְׁמִטְּתָא, יְהֵי לְאַרְעָא. 
כה,ו וְ֠הָֽיְתָה שַׁבַּ֨ת הָאָ֤רֶץ לָכֶם֙ לְאָכְלָ֔ה לְךָ֖ וּלְעַבְדְּךָ֣ וְלַֽאֲמָתֶ֑ךָ וְלִשְׂכִֽירְךָ֙ וּלְתוֹשָׁ֣בְךָ֔ הַגָּרִ֖ים עִמָּֽךְ׃ וּתְהֵי שְׁמִטַּת אַרְעָא לְכוֹן, לְמֵיכַל--לָךְ, וּלְעַבְדָךְ וּלְאַמְתָּךְ; וְלַאֲגִירָךְ, וּלְתוֹתָבָךְ, דְּדָיְרִין, עִמָּךְ. 
כה,ז וְלִ֨בְהֶמְתְּךָ֔ וְלַֽחַיָּ֖ה אֲשֶׁ֣ר בְּאַרְצֶ֑ךָ תִּֽהְיֶ֥ה כָל־תְּבֽוּאָתָ֖הּ לֶֽאֱכֹֽל׃ {ס} וְלִבְעִירָךְ--וּלְחַיְתָא, דִּבְאַרְעָךְ:  תְּהֵי כָּל עֲלַלְתַּהּ, לְמֵיכַל. {ס}
כה,ח וְסָֽפַרְתָּ֣ לְךָ֗ שֶׁ֚בַע שַׁבְּתֹ֣ת שָׁנִ֔ים שֶׁ֥בַע שָׁנִ֖ים שֶׁ֣בַע פְּעָמִ֑ים וְהָי֣וּ לְךָ֗ יְמֵי֙ שֶׁ֚בַע שַׁבְּתֹ֣ת הַשָּׁנִ֔ים תֵּ֥שַׁע וְאַרְבָּעִ֖ים שָׁנָֽה׃ וְתִמְנֵי לָךְ, שְׁבַע שְׁמִטָּן דִּשְׁנִין--שְׁבַע שְׁנִין, שְׁבַע זִמְנִין; וִיהוֹן לָךְ, יוֹמֵי שְׁבַע שְׁמִטָּן דִּשְׁנִין, אַרְבְּעִין וּתְשַׁע, שְׁנִין. 
כה,ט וְהַֽעֲבַרְתָּ֞ שׁוֹפַ֤ר תְּרוּעָה֙ בַּחֹ֣דֶשׁ הַשְּׁבִעִ֔י בֶּֽעָשׂ֖וֹר לַחֹ֑דֶשׁ בְּיוֹם֙ הַכִּפֻּרִ֔ים תַּֽעֲבִ֥ירוּ שׁוֹפָ֖ר בְּכָל־אַרְצְכֶֽם׃ וְתַעְבַּר שׁוֹפַר יַבָּבָא, בְּיַרְחָא שְׁבִיעָאָה, בְּעַסְרָא, לְיַרְחָא; בְּיוֹמָא, דְּכִפּוּרַיָּא, תַּעְבְּרוּן שׁוֹפָרָא, בְּכָל אֲרַעְכוֹן. 
כה,י וְקִדַּשְׁתֶּ֗ם אֵ֣ת שְׁנַ֤ת הַֽחֲמִשִּׁים֙ שָׁנָ֔ה וּקְרָאתֶ֥ם דְּר֛וֹר בָּאָ֖רֶץ לְכָל־יֹֽשְׁבֶ֑יהָ יוֹבֵ֥ל הִוא֙ תִּֽהְיֶ֣ה לָכֶ֔ם וְשַׁבְתֶּ֗ם אִ֚ישׁ אֶל־אֲחֻזָּת֔וֹ וְאִ֥ישׁ אֶל־מִשְׁפַּחְתּ֖וֹ תָּשֻֽׁבוּ׃ וּתְקַדְּשׁוּן, יָת שְׁנַת חַמְשִׁין שְׁנִין, וְתִקְרוֹן חֵירוּתָא בְּאַרְעָא, לְכָל יָתְבַהָא; יוֹבֵילָא הִיא, תְּהֵי לְכוֹן, וּתְתוּבוּן גְּבַר לְאַחְסָנְתֵיהּ, וּגְבַר לְזַרְעִיתֵיהּ תְּתוּבוּן. 
כה,יא יוֹבֵ֣ל הִ֗וא שְׁנַ֛ת הַֽחֲמִשִּׁ֥ים שָׁנָ֖ה תִּֽהְיֶ֣ה לָכֶ֑ם לֹ֣א תִזְרָ֔עוּ וְלֹ֤א תִקְצְרוּ֙ אֶת־סְפִיחֶ֔יהָ וְלֹ֥א תִבְצְר֖וּ אֶת־נְזִרֶֽיהָ׃ יוֹבֵילָא הִיא, שְׁנַת חַמְשִׁין שְׁנִין--תְּהֵי לְכוֹן; לָא תִּזְרְעוּן--וְלָא תִּחְצְדוּן יָת כָּתַהָא, וְלָא תִּקְטְפוּן יָת שִׁבְקַהָא. 
כה,יב כִּ֚י יוֹבֵ֣ל הִ֔וא קֹ֖דֶשׁ תִּֽהְיֶ֣ה לָכֶ֑ם מִ֨ן־הַשָּׂדֶ֔ה תֹּֽאכְל֖וּ אֶת־תְּבֽוּאָתָֽהּ׃ אֲרֵי יוֹבֵילָא הִיא, קֻדְשָׁא תְּהֵי לְכוֹן; מִן חַקְלָא--תֵּיכְלוּן, יָת עֲלַלְתַּהּ. 
כה,יג בִּשְׁנַ֥ת הַיּוֹבֵ֖ל הַזֹּ֑את תָּשֻׁ֕בוּ אִ֖ישׁ אֶל־אֲחֻזָּתֽוֹ׃ בְּשַׁתָּא דְּיוֹבֵילָא, הָדָא, תְּתוּבוּן, גְּבַר לְאַחְסָנְתֵיהּ. 
כה,יד וְכִֽי־תִמְכְּר֤וּ מִמְכָּר֙ לַֽעֲמִיתֶ֔ךָ א֥וֹ קָנֹ֖ה מִיַּ֣ד עֲמִיתֶ֑ךָ אַל־תּוֹנ֖וּ אִ֥ישׁ אֶת־אָחִֽיו׃ וַאֲרֵי תְּזַבֵּין זְבִינִין לְחַבְרָךְ, אוֹ תִּזְבּוֹן מִיַּד חַבְרָךְ--לָא תּוֹנוֹן, גְּבַר יָת אֲחוּהִי. 
כה,טו בְּמִסְפַּ֤ר שָׁנִים֙ אַחַ֣ר הַיּוֹבֵ֔ל תִּקְנֶ֖ה מֵאֵ֣ת עֲמִיתֶ֑ךָ בְּמִסְפַּ֥ר שְׁנֵֽי־תְבוּאֹ֖ת יִמְכָּר־לָֽךְ׃ בְּמִנְיַן שְׁנַיָּא בָּתַר יוֹבֵילָא, תִּזְבּוֹן מִן חַבְרָךְ; בְּמִנְיַן שְׁנֵי עֲלַלְתָּא, יְזַבֵּין לָךְ. 
כה,טז לְפִ֣י ׀ רֹ֣ב הַשָּׁנִ֗ים תַּרְבֶּה֙ מִקְנָת֔וֹ וּלְפִי֙ מְעֹ֣ט הַשָּׁנִ֔ים תַּמְעִ֖יט מִקְנָת֑וֹ כִּ֚י מִסְפַּ֣ר תְּבוּאֹ֔ת ה֥וּא מֹכֵ֖ר לָֽךְ׃ לְפֹם סַגְיוּת שְׁנַיָּא, תַּסְגֵּי זְבִינוֹהִי, וּלְפֹם זְעֵירוּת שְׁנַיָּא, תַּזְעַר זְבִינוֹהִי:  אֲרֵי מִנְיַן עֲלַלְתָּא, הוּא מְזַבֵּין לָךְ. 
כה,יז וְלֹ֤א תוֹנוּ֙ אִ֣ישׁ אֶת־עֲמִית֔וֹ וְיָרֵ֖אתָ מֵֽאֱלֹהֶ֑יךָ כִּ֛י אֲנִ֥י יְהוָ֖ה אֱלֹֽהֵיכֶֽם׃ וְלָא תּוֹנוֹן גְּבַר יָת חַבְרֵיהּ, וְתִדְחַל מֵאֱלָהָךְ:  אֲרֵי אֲנָא יְיָ, אֱלָהֲכוֹן. 
כה,יח וַֽעֲשִׂיתֶם֙ אֶת־חֻקֹּתַ֔י וְאֶת־מִשְׁפָּטַ֥י תִּשְׁמְר֖וּ וַֽעֲשִׂיתֶ֣ם אֹתָ֑ם וִֽישַׁבְתֶּ֥ם עַל־הָאָ֖רֶץ לָבֶֽטַח׃ וְתַעְבְּדוּן, יָת קְיָמַי, וְיָת דִּינַי תִּטְּרוּן, וְתַעְבְּדוּן יָתְהוֹן--וְתִתְּבוּן עַל אַרְעָא, לְרֻחְצָן. 
כה,יט וְנָֽתְנָ֤ה הָאָ֨רֶץ֙ פִּרְיָ֔הּ וַֽאֲכַלְתֶּ֖ם לָשֹׂ֑בַע וִֽישַׁבְתֶּ֥ם לָבֶ֖טַח עָלֶֽיהָ׃ וְתִתֵּין אַרְעָא אִבַּהּ, וְתֵיכְלוּן לְמִסְבַּע; וְתִתְּבוּן לְרֻחְצָן, עֲלַהּ. 
כה,כ וְכִ֣י תֹֽאמְר֔וּ מַה־נֹּאכַ֖ל בַּשָּׁנָ֣ה הַשְּׁבִיעִ֑ת הֵ֚ן לֹ֣א נִזְרָ֔ע וְלֹ֥א נֶֽאֱסֹ֖ף אֶת־תְּבֽוּאָתֵֽנוּ׃ וַאֲרֵי תֵּימְרוּן, מָא נֵיכוֹל בְּשַׁתָּא שְׁבִיעֵיתָא:  הָא לָא נִזְרַע, וְלָא נִכְנוֹשׁ יָת עֲלַלְתַּנָא. 
כה,כא וְצִוִּ֤יתִי אֶת־בִּרְכָתִי֙ לָכֶ֔ם בַּשָּׁנָ֖ה הַשִּׁשִּׁ֑ית וְעָשָׂת֙ אֶת־הַתְּבוּאָ֔ה לִשְׁלֹ֖שׁ הַשָּׁנִֽים׃ וַאֲפַקֵּיד יָת בִּרְכְתִי לְכוֹן, בְּשַׁתָּא שְׁתִיתֵיתָא; וְתַעֲבֵיד, יָת עֲלַלְתָּא, לִתְלָת, שְׁנִין. 
כה,כב וּזְרַעְתֶּ֗ם אֵ֚ת הַשָּׁנָ֣ה הַשְּׁמִינִ֔ת וַֽאֲכַלְתֶּ֖ם מִן־הַתְּבוּאָ֣ה יָשָׁ֑ן עַ֣ד ׀ הַשָּׁנָ֣ה הַתְּשִׁיעִ֗ת עַד־בּוֹא֙ תְּב֣וּאָתָ֔הּ תֹּֽאכְל֖וּ יָשָֽׁן׃ וְתִזְרְעוּן, יָת שַׁתָּא תְּמִינֵיתָא, וְתֵיכְלוּן, מִן עֲלַלְתָּא עַתִּיקָא; עַד שַׁתָּא תְּשִׁיעֵיתָא, עַד מֵיעַל עֲלַלְתַּהּ--תֵּיכְלוּן, עַתִּיקָא. 
כה,כג וְהָאָ֗רֶץ לֹ֤א תִמָּכֵר֙ לִצְמִתֻ֔ת כִּי־לִ֖י הָאָ֑רֶץ כִּֽי־גֵרִ֧ים וְתֽוֹשָׁבִ֛ים אַתֶּ֖ם עִמָּדִֽי׃ וְאַרְעָא, לָא תִּזְדַּבַּן לַחְלוּטִין--אֲרֵי דִּילִי, אַרְעָא:  אֲרֵי דַּיָּרִין וְתוֹתָבִין אַתּוּן, קֳדָמָי. 
כה,כד וּבְכֹ֖ל אֶ֣רֶץ אֲחֻזַּתְכֶ֑ם גְּאֻלָּ֖ה תִּתְּנ֥וּ לָאָֽרֶץ׃ {ס} וּבְכֹל, אֲרַע אַחְסָנַתְכוֹן, פֻּרְקָנָא, תִּתְּנוּן לְאַרְעָא. {ס}
כה,כה כִּֽי־יָמ֣וּךְ אָחִ֔יךָ וּמָכַ֖ר מֵֽאֲחֻזָּת֑וֹ וּבָ֤א גֹֽאֲלוֹ֙ הַקָּרֹ֣ב אֵלָ֔יו וְגָאַ֕ל אֵ֖ת מִמְכַּ֥ר אָחִֽיו׃ אֲרֵי יִתְמַסְכַּן אֲחוּךְ, וִיזַבֵּין מֵאַחְסָנְתֵיהּ--וְיֵיתֵי פָּרִיקֵיהּ, דְּקָרִיב לֵיהּ, וְיִפְרוֹק, יָת זְבִינֵי אֲחוּהִי. 
כה,כו וְאִ֕ישׁ כִּ֛י לֹ֥א יִֽהְיֶה־לּ֖וֹ גֹּאֵ֑ל וְהִשִּׂ֣יגָה יָד֔וֹ וּמָצָ֖א כְּדֵ֥י גְאֻלָּתֽוֹ׃ וּגְבַר, אֲרֵי לָא יְהֵי לֵיהּ פָּרִיק, וְתַדְבֵּיק יְדֵיהּ, וְיַשְׁכַּח כְּמִסַּת פֻּרְקָנֵיהּ. 
כה,כז וְחִשַּׁב֙ אֶת־שְׁנֵ֣י מִמְכָּר֔וֹ וְהֵשִׁיב֙ אֶת־הָ֣עֹדֵ֔ף לָאִ֖ישׁ אֲשֶׁ֣ר מָֽכַר־ל֑וֹ וְשָׁ֖ב לַֽאֲחֻזָּתֽוֹ׃ וִיחַשֵּׁיב, יָת שְׁנֵי זְבִינוֹהִי, וְיָתִיב יָת מוֹתָרָא, לִגְבַר דְּזַבֵּין לֵיהּ; וִיתוּב, לְאַחְסָנְתֵיהּ. 
כה,כח וְאִ֨ם לֹֽא־מָצְאָ֜ה יָד֗וֹ דֵּי֮ הָשִׁ֣יב לוֹ֒ וְהָיָ֣ה מִמְכָּר֗וֹ בְּיַד֙ הַקֹּנֶ֣ה אֹת֔וֹ עַ֖ד שְׁנַ֣ת הַיּוֹבֵ֑ל וְיָצָא֙ בַּיֹּבֵ֔ל וְשָׁ֖ב לַֽאֲחֻזָּתֽוֹ׃ {ס} וְאִם לָא אַשְׁכַּחַת יְדֵיהּ, כְּמִסַּת דְּיָתִיב לֵיהּ--וִיהֵי זְבִינוֹהִי בְּיַד דִּזְבַן יָתֵיהּ, עַד שַׁתָּא דְּיוֹבֵילָא; וְיִפּוֹק, בְּיוֹבֵילָא, וִיתוּב, לְאַחְסָנְתֵיהּ. {ס}
כה,כט וְאִ֗ישׁ כִּֽי־יִמְכֹּ֤ר בֵּית־מוֹשַׁב֙ עִ֣יר חוֹמָ֔ה וְהָֽיְתָה֙ גְּאֻלָּת֔וֹ עַד־תֹּ֖ם שְׁנַ֣ת מִמְכָּר֑וֹ יָמִ֖ים תִּֽהְיֶ֥ה גְאֻלָּתֽוֹ׃ וּגְבַר, אֲרֵי יְזַבֵּין בֵּית מוֹתַב קַרְתָּא מַקְּפָא שׁוּר--וִיהֵי פֻּרְקָנֵיהּ, עַד מִשְׁלַם שַׁתָּא דִּזְבִינוֹהִי:  עִדָּן בְּעִדָּן, יְהֵי פֻּרְקָנֵיהּ. 
כה,ל וְאִ֣ם לֹֽא־יִגָּאֵ֗ל עַד־מְלֹ֣את לוֹ֮ שָׁנָ֣ה תְמִימָה֒ וְ֠קָם הַבַּ֨יִת אֲשֶׁר־בָּעִ֜יר אֲשֶׁר־לא (ל֣וֹ) חֹמָ֗ה לַצְּמִיתֻ֛ת לַקֹּנֶ֥ה אֹת֖וֹ לְדֹֽרֹתָ֑יו לֹ֥א יֵצֵ֖א בַּיֹּבֵֽל׃ וְאִם לָא יִתְפְּרֵיק, עַד מִשְׁלַם לֵיהּ שַׁתָּא שַׁלְמְתָא--וִיקוּם בֵּיתָא דִּבְקַרְתָּא דְּלֵיהּ שׁוּרָא לַחְלוּטִין לְדִזְבַן יָתֵיהּ, לְדָרוֹהִי:  לָא יִפּוֹק, בְּיוֹבֵילָא. 
כה,לא וּבָתֵּ֣י הַֽחֲצֵרִ֗ים אֲשֶׁ֨ר אֵין־לָהֶ֤ם חֹמָה֙ סָבִ֔יב עַל־שְׂדֵ֥ה הָאָ֖רֶץ יֵֽחָשֵׁ֑ב גְּאֻלָּה֙ תִּֽהְיֶה־לּ֔וֹ וּבַיֹּבֵ֖ל יֵצֵֽא׃ וּבָתֵּי פַּצְחַיָּא, דְּלֵית לְהוֹן שׁוּר מַקַּף סְחוֹר סְחוֹר--עַל חֲקַל אַרְעָא, יִתְחַשְׁבוּן:  פֻּרְקָנָא, יְהֵי לְהוֹן, וּבְיוֹבֵילָא, יִפְּקוּן. 
כה,לב וְעָרֵי֙ הַלְוִיִּ֔ם בָּתֵּ֖י עָרֵ֣י אֲחֻזָּתָ֑ם גְּאֻלַּ֥ת עוֹלָ֖ם תִּֽהְיֶ֥ה לַלְוִיִּֽם׃ וְקִרְוֵי, לֵיוָאֵי--בָּתֵּי, קִרְוֵי אַחְסָנַתְהוֹן:  פֻּרְקַן עָלַם, יְהֵי לְלֵיוָאֵי. 
כה,לג וַֽאֲשֶׁ֤ר יִגְאַל֙ מִן־הַלְוִיִּ֔ם וְיָצָ֧א מִמְכַּר־בַּ֛יִת וְעִ֥יר אֲחֻזָּת֖וֹ בַּיֹּבֵ֑ל כִּ֣י בָתֵּ֞י עָרֵ֣י הַלְוִיִּ֗ם הִ֚וא אֲחֻזָּתָ֔ם בְּת֖וֹךְ בְּנֵ֥י יִשְׂרָאֵֽל׃ וּדְיִפְרוֹק מִן לֵיוָאֵי, וְיִפְּקוּן זְבִינֵי בֵּיתָא וְקִרְוֵי אַחְסָנְתֵיהּ בְּיוֹבֵילָא:  אֲרֵי בָּתֵּי קִרְוֵי לֵיוָאֵי, אִנִּין אַחְסָנַתְהוֹן, בְּגוֹ, בְּנֵי יִשְׂרָאֵל. 
כה,לד וּֽשְׂדֵ֛ה מִגְרַ֥שׁ עָֽרֵיהֶ֖ם לֹ֣א יִמָּכֵ֑ר כִּֽי־אֲחֻזַּ֥ת עוֹלָ֛ם ה֖וּא לָהֶֽם׃ {ס} וַחֲקַל רְוַח קִרְוֵיהוֹן, לָא יִזְדַּבַּן:  אֲרֵי אַחְסָנַת עָלַם הוּא, לְהוֹן. {ס}
כה,לה וְכִֽי־יָמ֣וּךְ אָחִ֔יךָ וּמָ֥טָה יָד֖וֹ עִמָּ֑ךְ וְהֶֽחֱזַ֣קְתָּ בּ֔וֹ גֵּ֧ר וְתוֹשָׁ֛ב וָחַ֖י עִמָּֽךְ׃ וַאֲרֵי יִתְמַסְכַּן אֲחוּךְ, וּתְמוּט יְדֵיהּ עִמָּךְ--וְתַתְקֵיף בֵּיהּ, יְדוּר וְיִתּוֹתַב וְיֵיחֵי עִמָּךְ. 
כה,לו אַל־תִּקַּ֤ח מֵֽאִתּוֹ֙ נֶ֣שֶׁךְ וְתַרְבִּ֔ית וְיָרֵ֖אתָ מֵֽאֱלֹהֶ֑יךָ וְחֵ֥י אָחִ֖יךָ עִמָּֽךְ׃ לָא תִּסַּב מִנֵּיהּ חִיבֻלְיָא וְרִבִּיתָא, וְתִדְחַל מֵאֱלָהָךְ; וְיֵיחֵי אֲחוּךְ, עִמָּךְ. 
כה,לז אֶ֨ת־כַּסְפְּךָ֔ לֹֽא־תִתֵּ֥ן ל֖וֹ בְּנֶ֑שֶׁךְ וּבְמַרְבִּ֖ית לֹֽא־תִתֵּ֥ן אָכְלֶֽךָ׃ יָת כַּסְפָּךְ--לָא תִּתֵּין לֵיהּ, בְּחִיבֻלְיָא; וּבְרִבִּיתָא, לָא תִּתֵּין מֵיכְלָךְ. 
כה,לח אֲנִ֗י יְהוָה֙ אֱלֹ֣הֵיכֶ֔ם אֲשֶׁר־הוֹצֵ֥אתִי אֶתְכֶ֖ם מֵאֶ֣רֶץ מִצְרָ֑יִם לָתֵ֤ת לָכֶם֙ אֶת־אֶ֣רֶץ כְּנַ֔עַן לִֽהְי֥וֹת לָכֶ֖ם לֵֽאלֹהִֽים׃ {ס} אֲנָא, יְיָ אֱלָהֲכוֹן, דְּאַפֵּיקִית יָתְכוֹן, מֵאַרְעָא דְּמִצְרָיִם--לְמִתַּן לְכוֹן יָת אַרְעָא דִּכְנַעַן, לְמִהְוֵי לְכוֹן לֶאֱלָהּ. {ס}
כה,לט וְכִֽי־יָמ֥וּךְ אָחִ֛יךָ עִמָּ֖ךְ וְנִמְכַּר־לָ֑ךְ לֹֽא־תַעֲבֹ֥ד בּ֖וֹ עֲבֹ֥דַת עָֽבֶד׃ וַאֲרֵי יִתְמַסְכַּן אֲחוּךְ עִמָּךְ, וְיִזְדַּבַּן לָךְ--לָא תִּפְלַח בֵּיהּ, פֻּלְחַן עַבְדִּין. 
כה,מ כְּשָׂכִ֥יר כְּתוֹשָׁ֖ב יִֽהְיֶ֣ה עִמָּ֑ךְ עַד־שְׁנַ֥ת הַיֹּבֵ֖ל יַֽעֲבֹ֥ד עִמָּֽךְ׃ כַּאֲגִירָא כְּתוֹתָבָא, יְהֵי עִמָּךְ; עַד שַׁתָּא דְּיוֹבֵילָא, יִפְלַח עִמָּךְ. 
כה,מא וְיָצָא֙ מֵֽעִמָּ֔ךְ ה֖וּא וּבָנָ֣יו עִמּ֑וֹ וְשָׁב֙ אֶל־מִשְׁפַּחְתּ֔וֹ וְאֶל־אֲחֻזַּ֥ת אֲבֹתָ֖יו יָשֽׁוּב׃ וְיִפּוֹק, מֵעִמָּךְ--הוּא, וּבְנוֹהִי עִמֵּיהּ; וִיתוּב, לְזַרְעִיתֵיהּ, וּלְאַחְסָנַת אֲבָהָתוֹהִי, יְתוּב. 
כה,מב כִּֽי־עֲבָדַ֣י הֵ֔ם אֲשֶׁר־הוֹצֵ֥אתִי אֹתָ֖ם מֵאֶ֣רֶץ מִצְרָ֑יִם לֹ֥א יִמָּֽכְר֖וּ מִמְכֶּ֥רֶת עָֽבֶד׃ אֲרֵי עַבְדַּי אִנּוּן, דְּאַפֵּיקִית יָתְהוֹן מֵאַרְעָא דְּמִצְרָיִם; לָא יִזְדַּבְּנוּן, זִבּוּן עַבְדִּין. 
כה,מג לֹֽא־תִרְדֶּ֥ה ב֖וֹ בְּפָ֑רֶךְ וְיָרֵ֖אתָ מֵֽאֱלֹהֶֽיךָ׃ לָא תִּפְלַח בֵּיהּ, בְּקַשְׁיוּ; וְתִדְחַל, מֵאֱלָהָךְ. 
כה,מד וְעַבְדְּךָ֥ וַאֲמָֽתְךָ֖ אֲשֶׁ֣ר יִֽהְיוּ־לָ֑ךְ מֵאֵ֣ת הַגּוֹיִ֗ם אֲשֶׁר֙ סְבִיבֹ֣תֵיכֶ֔ם מֵהֶ֥ם תִּקְנ֖וּ עֶ֥בֶד וְאָמָֽה׃ וְעַבְדָּךְ וְאַמְתָּךְ, דִּיהוֹן לָךְ:  מִן עַמְמַיָּא, דִּבְסַחְרָנֵיכוֹן--מִנְּהוֹן תִּקְנוֹן, עַבְדִּין וְאַמְהָן. 
כה,מה וְ֠גַם מִבְּנֵ֨י הַתּֽוֹשָׁבִ֜ים הַגָּרִ֤ים עִמָּכֶם֙ מֵהֶ֣ם תִּקְנ֔וּ וּמִמִּשְׁפַּחְתָּם֙ אֲשֶׁ֣ר עִמָּכֶ֔ם אֲשֶׁ֥ר הוֹלִ֖ידוּ בְּאַרְצְכֶ֑ם וְהָי֥וּ לָכֶ֖ם לַֽאֲחֻזָּֽה׃ וְאַף מִבְּנֵי תּוֹתָבַיָּא עַרְלַיָּא דְּדָיְרִין עִמְּכוֹן, מִנְּהוֹן תִּקְנוֹן, וּמִזַּרְעִיתְהוֹן דְּעִמְּכוֹן, דְּאִתְיַלַּדוּ בַּאֲרַעְכוֹן; וִיהוֹן לְכוֹן, לְאַחְסָנָא. 
כה,מו וְהִתְנַֽחַלְתֶּ֨ם אֹתָ֜ם לִבְנֵיכֶ֤ם אַֽחֲרֵיכֶם֙ לָרֶ֣שֶׁת אֲחֻזָּ֔ה לְעֹלָ֖ם בָּהֶ֣ם תַּֽעֲבֹ֑דוּ וּבְאַ֨חֵיכֶ֤ם בְּנֵֽי־יִשְׂרָאֵל֙ אִ֣ישׁ בְּאָחִ֔יו לֹֽא־תִרְדֶּ֥ה ב֖וֹ בְּפָֽרֶךְ׃ {ס} וְתַחְסְנוּן יָתְהוֹן לִבְנֵיכוֹן בָּתְרֵיכוֹן, לִירֻתַּת אַחְסָנָא--לְעָלַם, בְּהוֹן תִּפְלְחוּן; וּבַאֲחֵיכוֹן בְּנֵי יִשְׂרָאֵל גְּבַר בַּאֲחוּהִי, לָא תִּפְלַח בֵּיהּ בְּקַשְׁיוּ. {ס}
כה,מז וְכִ֣י תַשִּׂ֗יג יַ֣ד גֵּ֤ר וְתוֹשָׁב֙ עִמָּ֔ךְ וּמָ֥ךְ אָחִ֖יךָ עִמּ֑וֹ וְנִמְכַּ֗ר לְגֵ֤ר תּוֹשָׁב֙ עִמָּ֔ךְ א֥וֹ לְעֵ֖קֶר מִשְׁפַּ֥חַת גֵּֽר׃ וַאֲרֵי תַּדְבֵּיק, יַד עֲרַל וְתוֹתָב עִמָּךְ, וְיִתְמַסְכַּן אֲחוּךְ, עִמֵּיהּ; וְיִזְדַּבַּן, לַעֲרַל תּוֹתָב דְּעִמָּךְ, אוֹ לְאַרְמַאי, מִזַּרְעִית גִּיּוֹרָא. 
כה,מח אַֽחֲרֵ֣י נִמְכַּ֔ר גְּאֻלָּ֖ה תִּֽהְיֶה־לּ֑וֹ אֶחָ֥ד מֵֽאֶחָ֖יו יִגְאָלֶֽנּוּ׃ בָּתַר דְּאִזְדַּבַּן, פֻּרְקָנָא יְהֵי לֵיהּ:  חַד מֵאֲחוֹהִי, יִפְרְקִנֵּיהּ. 
כה,מט אֽוֹ־דֹד֞וֹ א֤וֹ בֶן־דֹּדוֹ֙ יִגְאָלֶ֔נּוּ אֽוֹ־מִשְּׁאֵ֧ר בְּשָׂר֛וֹ מִמִּשְׁפַּחְתּ֖וֹ יִגְאָלֶ֑נּוּ אֽוֹ־הִשִּׂ֥יגָה יָד֖וֹ וְנִגְאָֽל׃ אוֹ אַחְבּוּהִי אוֹ בַּר אַחְבּוּהִי, יִפְרְקִנֵּיהּ, אוֹ מִקָּרִיב בִּסְרֵיהּ מִזַּרְעִיתֵיהּ, יִפְרְקִנֵּיהּ; אוֹ דְּתַדְבֵּיק יְדֵיהּ, וְיִתְפְּרֵיק. 
כה,נ וְחִשַּׁב֙ עִם־קֹנֵ֔הוּ מִשְּׁנַת֙ הִמָּ֣כְרוֹ ל֔וֹ עַ֖ד שְׁנַ֣ת הַיֹּבֵ֑ל וְהָיָ֞ה כֶּ֤סֶף מִמְכָּרוֹ֙ בְּמִסְפַּ֣ר שָׁנִ֔ים כִּימֵ֥י שָׂכִ֖יר יִֽהְיֶ֥ה עִמּֽוֹ׃ וִיחַשֵּׁיב, עִם זָבְנֵיהּ, מִשַּׁתָּא דְּאִזְדַּבַּן לֵיהּ, עַד שַׁתָּא דְּיוֹבֵילָא; וִיהֵי כְּסַף זְבִינוֹהִי, בְּמִנְיַן שְׁנַיָּא, כְּיוֹמֵי אֲגִירָא, יְהֵי עִמֵּיהּ. 
כה,נא אִם־ע֥וֹד רַבּ֖וֹת בַּשָּׁנִ֑ים לְפִיהֶן֙ יָשִׁ֣יב גְּאֻלָּת֔וֹ מִכֶּ֖סֶף מִקְנָתֽוֹ׃ אִם עוֹד סַגְיוּת, בִּשְׁנַיָּא--לְפוֹמְהוֹן יָתִיב פֻּרְקָנֵיהּ, מִכְּסַף זְבִינוֹהִי. 
כה,נב וְאִם־מְעַ֞ט נִשְׁאַ֧ר בַּשָּׁנִ֛ים עַד־שְׁנַ֥ת הַיֹּבֵ֖ל וְחִשַּׁב־ל֑וֹ כְּפִ֣י שָׁנָ֔יו יָשִׁ֖יב אֶת־גְּאֻלָּתֽוֹ׃ וְאִם זְעֵיר אִשְׁתְּאַר בִּשְׁנַיָּא, עַד שַׁתָּא דְּיוֹבֵילָא--וִיחַשֵּׁיב לֵיהּ; כְּפֹם שְׁנוֹהִי, יָתִיב יָת פֻּרְקָנֵיהּ. 
כה,נג כִּשְׂכִ֥יר שָׁנָ֛ה בְּשָׁנָ֖ה יִֽהְיֶ֣ה עִמּ֑וֹ לֹֽא־יִרְדֶּ֥נּֽוּ בְּפֶ֖רֶךְ לְעֵינֶֽיךָ׃ כַּאֲגִיר שְׁנָא בִּשְׁנָא, יְהֵי עִמֵּיהּ; לָא יִפְלַח בֵּיהּ בְּקַשְׁיוּ, לְעֵינָךְ. 
כה,נד וְאִם־לֹ֥א יִגָּאֵ֖ל בְּאֵ֑לֶּה וְיָצָא֙ בִּשְׁנַ֣ת הַיֹּבֵ֔ל ה֖וּא וּבָנָ֥יו עִמּֽוֹ׃ וְאִם לָא יִתְפְּרֵיק, בְּאִלֵּין--וְיִפּוֹק בְּשַׁתָּא דְּיוֹבֵילָא, הוּא וּבְנוֹהִי עִמֵּיהּ. 
כה,נה כִּֽי־לִ֤י בְנֵֽי־יִשְׂרָאֵל֙ עֲבָדִ֔ים עֲבָדַ֣י הֵ֔ם אֲשֶׁר־הוֹצֵ֥אתִי אוֹתָ֖ם מֵאֶ֣רֶץ מִצְרָ֑יִם אֲנִ֖י יְהוָ֥ה אֱלֹֽהֵיכֶֽם׃ אֲרֵי דִּילִי בְּנֵי יִשְׂרָאֵל, עַבְדִּין--עַבְדַּי אִנּוּן, דְּאַפֵּיקִית יָתְהוֹן מֵאַרְעָא דְּמִצְרָיִם:  אֲנָא, יְיָ אֱלָהֲכוֹן. 
כו,א לֹֽא־תַעֲשׂ֨וּ לָכֶ֜ם אֱלִילִ֗ם וּפֶ֤סֶל וּמַצֵּבָה֙ לֹֽא־תָקִ֣ימוּ לָכֶ֔ם וְאֶ֣בֶן מַשְׂכִּ֗ית לֹ֤א תִתְּנוּ֙ בְּאַרְצְכֶ֔ם לְהִֽשְׁתַּחֲו֖‍ֹת עָלֶ֑יהָ כִּ֛י אֲנִ֥י יְהוָ֖ה אֱלֹֽהֵיכֶֽם׃ לָא תַּעְבְּדוּן לְכוֹן טָעֲוָן, וְצֵילַם וְקָמָא לָא תְּקִימוּן לְכוֹן, וְאֶבֶן סִגְדָּא לָא תִּתְּנוּן בַּאֲרַעְכוֹן, לְמִסְגַּד עֲלַהּ:  אֲרֵי אֲנָא יְיָ, אֱלָהֲכוֹן. 
כו,ב אֶת־שַׁבְּתֹתַ֣י תִּשְׁמֹ֔רוּ וּמִקְדָּשִׁ֖י תִּירָ֑אוּ אֲנִ֖י יְהוָֽה׃ {פ} יָת יוֹמֵי שַׁבַּיָּא דִּילִי תִּטְּרוּן, וּלְבֵית מַקְדְּשִׁי תְּהוֹן דָּחֲלִין:  אֲנָא, יְיָ. {פ}
