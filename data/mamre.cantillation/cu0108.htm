וישלח

לב,ד וַיִּשְׁלַ֨ח יַֽעֲקֹ֤ב מַלְאָכִים֙ לְפָנָ֔יו אֶל־עֵשָׂ֖ו אָחִ֑יו אַ֥רְצָה שֵׂעִ֖יר שְׂדֵ֥ה אֱדֽוֹם׃ וּשְׁלַח יַעֲקוֹב אִזְגַּדִּין קֳדָמוֹהִי, לְוָת עֵשָׂו אֲחוּהִי, לְאַרְעָא דְּשֵׂעִיר, לְחַקְלֵי אֱדוֹם. 
לב,ה וַיְצַ֤ו אֹתָם֙ לֵאמֹ֔ר כֹּ֣ה תֹֽאמְר֔וּן לַֽאדֹנִ֖י לְעֵשָׂ֑ו כֹּ֤ה אָמַר֙ עַבְדְּךָ֣ יַֽעֲקֹ֔ב עִם־לָבָ֣ן גַּ֔רְתִּי וָֽאֵחַ֖ר עַד־עָֽתָּה׃ וּפַקֵּיד יָתְהוֹן, לְמֵימַר, כְּדֵין תֵּימְרוּן, לְרִבּוֹנִי לְעֵשָׂו:  כִּדְנָן אֲמַר, עַבְדָּךְ יַעֲקוֹב, עִם לָבָן דַּרִית, וְאוֹחַרִית עַד כְּעַן. 
לב,ו וַֽיְהִי־לִי֙ שׁ֣וֹר וַֽחֲמ֔וֹר צֹ֖אן וְעֶ֣בֶד וְשִׁפְחָ֑ה וָֽאֶשְׁלְחָה֙ לְהַגִּ֣יד לַֽאדֹנִ֔י לִמְצֹא־חֵ֖ן בְּעֵינֶֽיךָ׃ וַהֲווֹ לִי תּוֹרִין וּחְמָרִין, עָן וְעַבְדִּין וְאַמְהָן; וּשְׁלַחִית לְחַוָּאָה לְרִבּוֹנִי, לְאַשְׁכָּחָא רַחֲמִין בְּעֵינָךְ. 
לב,ז וַיָּשֻׁ֨בוּ֙ הַמַּלְאָכִ֔ים אֶֽל־יַעֲקֹ֖ב לֵאמֹ֑ר בָּ֤אנוּ אֶל־אָחִ֨יךָ֙ אֶל־עֵשָׂ֔ו וְגַם֙ הֹלֵ֣ךְ לִקְרָֽאתְךָ֔ וְאַרְבַּע־מֵא֥וֹת אִ֖ישׁ עִמּֽוֹ׃ וְתָבוּ, אִזְגַּדַּיָּא, לְוָת יַעֲקוֹב, לְמֵימַר:  אֲתֵינָא לְוָת אֲחוּךְ, לְוָת עֵשָׂו, וְאַף אָתֵי לְקַדָּמוּתָךְ, וְאַרְבַּע מְאָה גֻּבְרָא עִמֵּיהּ. 
לב,ח וַיִּירָ֧א יַֽעֲקֹ֛ב מְאֹ֖ד וַיֵּ֣צֶר ל֑וֹ וַיַּ֜חַץ אֶת־הָעָ֣ם אֲשֶׁר־אִתּ֗וֹ וְאֶת־הַצֹּ֧אן וְאֶת־הַבָּקָ֛ר וְהַגְּמַלִּ֖ים לִשְׁנֵ֥י מַֽחֲנֽוֹת׃ וּדְחֵיל יַעֲקוֹב לַחְדָּא, וְעַקַת לֵיהּ; וּפַלֵּיג יָת עַמָּא דְּעִמֵּיהּ, וְיָת עָנָא וְיָת תּוֹרֵי וְגַמְלַיָּא--לְתַרְתֵּין מַשְׁרְיָן. 
לב,ט וַיֹּ֕אמֶר אִם־יָב֥וֹא עֵשָׂ֛ו אֶל־הַמַּֽחֲנֶ֥ה הָֽאַחַ֖ת וְהִכָּ֑הוּ וְהָיָ֛ה הַמַּֽחֲנֶ֥ה הַנִּשְׁאָ֖ר לִפְלֵיטָֽה׃ וַאֲמַר, אִם יֵיתֵי עֵשָׂו לְמַשְׁרִיתָא חֲדָא וְיִמְחֵינַהּ--וּתְהֵי מַשְׁרִיתָא דְּתִשְׁתְּאַר, לְשֵׁיזָבָא. 
לב,י וַיֹּאמֶר֮ יַֽעֲקֹב֒ אֱלֹהֵי֙ אָבִ֣י אַבְרָהָ֔ם וֵֽאלֹהֵ֖י אָבִ֣י יִצְחָ֑ק יְהוָ֞ה הָֽאֹמֵ֣ר אֵלַ֗י שׁ֧וּב לְאַרְצְךָ֛ וּלְמֽוֹלַדְתְּךָ֖ וְאֵיטִ֥יבָה עִמָּֽךְ׃ וַאֲמַר, יַעֲקוֹב, אֱלָהֵיהּ דְּאַבָּא אַבְרָהָם, וֵאלָהֵיהּ דְּאַבָּא יִצְחָק:  יְיָ דַּאֲמַר לִי, תּוּב לְאַרְעָךְ וּלְיַלָּדוּתָךְ--וְאוֹטֵיב עִמָּךְ. 
לב,יא קָטֹ֜נְתִּי מִכֹּ֤ל הַֽחֲסָדִים֙ וּמִכָּל־הָ֣אֱמֶ֔ת אֲשֶׁ֥ר עָשִׂ֖יתָ אֶת־עַבְדֶּ֑ךָ כִּ֣י בְמַקְלִ֗י עָבַ֨רְתִּי֙ אֶת־הַיַּרְדֵּ֣ן הַזֶּ֔ה וְעַתָּ֥ה הָיִ֖יתִי לִשְׁנֵ֥י מַֽחֲנֽוֹת׃ זְעֵירָן זָכְוָתִי מִכֹּל חִסְדִּין, וּמִכָּל טָבְוָן, דַּעֲבַדְתְּ, עִם עַבְדָּךְ:  אֲרֵי יְחִידַאי, עֲבַרִית יָת יַרְדְּנָא הָדֵין, וּכְעַן הֲוֵיתִי, לְתַרְתֵּין מַשְׁרְיָן. 
לב,יב הַצִּילֵ֥נִי נָ֛א מִיַּ֥ד אָחִ֖י מִיַּ֣ד עֵשָׂ֑ו כִּֽי־יָרֵ֤א אָֽנֹכִי֙ אֹת֔וֹ פֶּן־יָב֣וֹא וְהִכַּ֔נִי אֵ֖ם עַל־בָּנִֽים׃ שֵׁיזֵיבְנִי כְּעַן מִיְּדָא דַּאֲחִי, מִיְּדָא דְּעֵשָׂו:  אֲרֵי דָּחֵיל אֲנָא, מִנֵּיהּ--דִּלְמָא יֵיתֵי וְיִמְחֵינַנִי, אִמָּא עַל בְּנַיָּא. 
לב,יג וְאַתָּ֣ה אָמַ֔רְתָּ הֵיטֵ֥ב אֵיטִ֖יב עִמָּ֑ךְ וְשַׂמְתִּ֤י אֶֽת־זַרְעֲךָ֙ כְּח֣וֹל הַיָּ֔ם אֲשֶׁ֥ר לֹֽא־יִסָּפֵ֖ר מֵרֹֽב׃ וְאַתְּ אֲמַרְתְּ, אוֹטָבָא אוֹטֵיב עִמָּךְ; וַאֲשַׁוֵּי יָת בְּנָךְ סַגִּיאִין כְּחָלָא דְּיַמָּא, דְּלָא יִתְמְנוֹן מִסְּגֵי. 
לב,יד וַיָּ֥לֶן שָׁ֖ם בַּלַּ֣יְלָה הַה֑וּא וַיִּקַּ֞ח מִן־הַבָּ֧א בְיָד֛וֹ מִנְחָ֖ה לְעֵשָׂ֥ו אָחִֽיו׃ וּבָת תַּמָּן, בְּלֵילְיָא הַהוּא; וּנְסֵיב מִן דְּאַיְתִי בִּידֵיהּ, תִּקְרֻבְתָּא--לְעֵשָׂו אֲחוּהִי. 
לב,טו עִזִּ֣ים מָאתַ֔יִם וּתְיָשִׁ֖ים עֶשְׂרִ֑ים רְחֵלִ֥ים מָאתַ֖יִם וְאֵילִ֥ים עֶשְׂרִֽים׃ עִזֵּי מָאתַן, וְתֵישַׁיָּא עַסְרִין, רַחְלֵי מָאתַן, וְדִכְרֵי עַסְרִין. 
לב,טז גְּמַלִּ֧ים מֵֽינִיק֛וֹת וּבְנֵיהֶ֖ם שְׁלֹשִׁ֑ים פָּר֤וֹת אַרְבָּעִים֙ וּפָרִ֣ים עֲשָׂרָ֔ה אֲתֹנֹ֣ת עֶשְׂרִ֔ים וַעְיָרִ֖ם עֲשָׂרָֽה׃ גַּמְלֵי מֵינְקָתָא וּבְנֵיהוֹן, תְּלָתִין; תּוֹרָתָא אַרְבְּעִין, וְתוֹרֵי עַסְרָא, אֲתָנָן עַסְרִין, וְעִלֵּי עַסְרָא. 
לב,יז וַיִּתֵּן֙ בְּיַד־עֲבָדָ֔יו עֵ֥דֶר עֵ֖דֶר לְבַדּ֑וֹ וַ֤יֹּאמֶר אֶל־עֲבָדָיו֙ עִבְר֣וּ לְפָנַ֔י וְרֶ֣וַח תָּשִׂ֔ימוּ בֵּ֥ין עֵ֖דֶר וּבֵ֥ין עֵֽדֶר׃ וִיהַב, בְּיַד עַבְדּוֹהִי, עֶדְרָא עֶדְרָא, בִּלְחוֹדוֹהִי; וַאֲמַר לְעַבְדּוֹהִי, עֵיבַרוּ קֳדָמַי, וְרַוְחָא תְּשַׁוּוֹן, בֵּין עֶדְרָא וּבֵין עֶדְרָא. 
לב,יח וַיְצַ֥ו אֶת־הָֽרִאשׁ֖וֹן לֵאמֹ֑ר כִּ֣י יִֽפְגָשְׁךָ֞ עֵשָׂ֣ו אָחִ֗י וּשְׁאֵֽלְךָ֙ לֵאמֹ֔ר לְמִי־אַ֨תָּה֙ וְאָ֣נָה תֵלֵ֔ךְ וּלְמִ֖י אֵ֥לֶּה לְפָנֶֽיךָ׃ וּפַקֵּיד יָת קַדְמָאָה, לְמֵימַר:  אֲרֵי יְעָרְעִנָּךְ עֵשָׂו אֲחִי, וְיִשְׁאֲלִנָּךְ לְמֵימַר, דְּמַאן אַתְּ וּלְאָן אַתְּ אָזֵיל, וּדְמַאן אִלֵּין דִּקְדָמָךְ. 
לב,יט וְאָֽמַרְתָּ֙ לְעַבְדְּךָ֣ לְיַֽעֲקֹ֔ב מִנְחָ֥ה הִוא֙ שְׁלוּחָ֔ה לַֽאדֹנִ֖י לְעֵשָׂ֑ו וְהִנֵּ֥ה גַם־ה֖וּא אַֽחֲרֵֽינוּ׃ וְתֵימַר, דְּעַבְדָּךְ דְּיַעֲקוֹב--תִּקְרֻבְתָּא הִיא דִּמְשַׁלְחָא, לְרִבּוֹנִי לְעֵשָׂו; וְהָא אַף הוּא, אָתֵי בָּתְרַנָא. 
לב,כ וַיְצַ֞ו גַּ֣ם אֶת־הַשֵּׁנִ֗י גַּ֚ם אֶת־הַשְּׁלִישִׁ֔י גַּ֚ם אֶת־כָּל־הַהֹ֣לְכִ֔ים אַֽחֲרֵ֥י הָֽעֲדָרִ֖ים לֵאמֹ֑ר כַּדָּבָ֤ר הַזֶּה֙ תְּדַבְּר֣וּן אֶל־עֵשָׂ֔ו בְּמֹצַֽאֲכֶ֖ם אֹתֽוֹ׃ וּפַקֵּיד אַף יָת תִּנְיָנָא, אַף יָת תְּלִיתָאָה, אַף יָת כָּל דְּאָזְלִין, בָּתַר עֶדְרַיָּא לְמֵימַר:  כְּפִתְגָמָא הָדֵין תְּמַלְּלוּן עִם עֵשָׂו, כַּד תַּשְׁכְּחוּן יָתֵיהּ. 
לב,כא וַֽאֲמַרְתֶּ֕ם גַּ֗ם הִנֵּ֛ה עַבְדְּךָ֥ יַֽעֲקֹ֖ב אַֽחֲרֵ֑ינוּ כִּֽי־אָמַ֞ר אֲכַפְּרָ֣ה פָנָ֗יו בַּמִּנְחָה֙ הַֽהֹלֶ֣כֶת לְפָנָ֔י וְאַֽחֲרֵי־כֵן֙ אֶרְאֶ֣ה פָנָ֔יו אוּלַ֖י יִשָּׂ֥א פָנָֽי׃ וְתֵימְרוּן--אַף הָא עַבְדָּךְ יַעֲקוֹב, אָתֵי בָּתְרַנָא:  אֲרֵי אֲמַר אֲנִיחִנֵּיהּ לְרֻגְזֵיהּ, בְּתִקְרֻבְתָּא דְּאָזְלָא קֳדָמַי, וּבָתַר כֵּין אֶחְזֵי אַפּוֹהִי, מָאִם יִסַּב אַפָּי. 
לב,כב וַתַּֽעֲבֹ֥ר הַמִּנְחָ֖ה עַל־פָּנָ֑יו וְה֛וּא לָ֥ן בַּלַּֽיְלָה־הַה֖וּא בַּֽמַּחֲנֶֽה׃ וַעֲבַרַת תִּקְרֻבְתָּא, עַל אַפּוֹהִי; וְהוּא בָּת בְּלֵילְיָא הַהוּא, בְּמַשְׁרִיתָא. 
לב,כג וַיָּ֣קָם ׀ בַּלַּ֣יְלָה ה֗וּא וַיִּקַּ֞ח אֶת־שְׁתֵּ֤י נָשָׁיו֙ וְאֶת־שְׁתֵּ֣י שִׁפְחֹתָ֔יו וְאֶת־אַחַ֥ד עָשָׂ֖ר יְלָדָ֑יו וַֽיַּעֲבֹ֔ר אֵ֖ת מַֽעֲבַ֥ר יַבֹּֽק׃ וְקָם בְּלֵילְיָא הוּא, וּדְבַר יָת תַּרְתֵּין נְשׁוֹהִי וְיָת תַּרְתֵּין לְחֵינָתֵיהּ, וְיָת חַד עֲסַר, בְּנוֹהִי; וַעֲבַר, יָת מַעֲבַר יֻבְקָא. 
לב,כד וַיִּ֨קָּחֵ֔ם וַיַּֽעֲבִרֵ֖ם אֶת־הַנָּ֑חַל וַֽיַּעֲבֵ֖ר אֶת־אֲשֶׁר־לֽוֹ׃ וּדְבַרִנּוּן--וְאַעְבְּרִנּוּן, יָת נַחְלָא; וְאַעְבַּר, יָת דְּלֵיהּ. 
לב,כה וַיִּוָּתֵ֥ר יַֽעֲקֹ֖ב לְבַדּ֑וֹ וַיֵּֽאָבֵ֥ק אִישׁ֙ עִמּ֔וֹ עַ֖ד עֲל֥וֹת הַשָּֽׁחַר׃ וְאִשְׁתְּאַר יַעֲקוֹב, בִּלְחוֹדוֹהִי; וְאִשְׁתַּדַּל גֻּבְרָא עִמֵּיהּ, עַד דִּסְלֵיק צַפְרָא. 
לב,כו וַיַּ֗רְא כִּ֣י לֹ֤א יָכֹל֙ ל֔וֹ וַיִּגַּ֖ע בְּכַף־יְרֵכ֑וֹ וַתֵּ֨קַע֙ כַּף־יֶ֣רֶךְ יַֽעֲקֹ֔ב בְּהֵֽאָבְק֖וֹ עִמּֽוֹ׃ וַחֲזָא, אֲרֵי לָא יְכֵיל לֵיהּ, וּקְרֵיב, בִּפְתֵי יִרְכֵּיהּ; וְזָע פְּתֵי יִרְכָּא דְּיַעֲקוֹב, בְּאִשְׁתַּדָּלוּתֵיהּ עִמֵּיהּ. 
לב,כז וַיֹּ֣אמֶר שַׁלְּחֵ֔נִי כִּ֥י עָלָ֖ה הַשָּׁ֑חַר וַיֹּ֨אמֶר֙ לֹ֣א אֲשַֽׁלֵּחֲךָ֔ כִּ֖י אִם־בֵּֽרַכְתָּֽנִי׃ וַאֲמַר שַׁלְּחַנִי, אֲרֵי סְלֵיק צַפְרָא; וַאֲמַר לָא אֲשַׁלְּחִנָּךְ, אֱלָהֵין בָּרֵיכְתָּנִי. 
לב,כח וַיֹּ֥אמֶר אֵלָ֖יו מַה־שְּׁמֶ֑ךָ וַיֹּ֖אמֶר יַֽעֲקֹֽב׃ וַאֲמַר לֵיהּ, מַאן שְׁמָךְ; וַאֲמַר, יַעֲקוֹב. 
לב,כט וַיֹּ֗אמֶר לֹ֤א יַֽעֲקֹב֙ יֵֽאָמֵ֥ר עוֹד֙ שִׁמְךָ֔ כִּ֖י אִם־יִשְׂרָאֵ֑ל כִּֽי־שָׂרִ֧יתָ עִם־אֱלֹהִ֛ים וְעִם־אֲנָשִׁ֖ים וַתּוּכָֽל׃ וַאֲמַר, לָא יַעֲקוֹב יִתְאֲמַר עוֹד שְׁמָךְ--אֱלָהֵין יִשְׂרָאֵל:  אֲרֵי רָב אַתְּ קֳדָם יְיָ וְעִם גֻּבְרַיָּא, וִיכֵילְתָּא. 
לב,ל וַיִּשְׁאַ֣ל יַֽעֲקֹ֗ב וַיֹּ֨אמֶר֙ הַגִּֽידָה־נָּ֣א שְׁמֶ֔ךָ וַיֹּ֕אמֶר לָ֥מָּה זֶּ֖ה תִּשְׁאַ֣ל לִשְׁמִ֑י וַיְבָ֥רֶךְ אֹת֖וֹ שָֽׁם׃ וּשְׁאֵיל יַעֲקוֹב, וַאֲמַר חַו כְּעַן שְׁמָךְ, וַאֲמַר, לְמָא דְּנָן אַתְּ שָׁאֵיל לִשְׁמִי; וּבָרֵיךְ יָתֵיהּ, תַּמָּן. 
לב,לא וַיִּקְרָ֧א יַֽעֲקֹ֛ב שֵׁ֥ם הַמָּק֖וֹם פְּנִיאֵ֑ל כִּֽי־רָאִ֤יתִי אֱלֹהִים֙ פָּנִ֣ים אֶל־פָּנִ֔ים וַתִּנָּצֵ֖ל נַפְשִֽׁי׃ וּקְרָא יַעֲקוֹב שְׁמֵיהּ דְּאַתְרָא, פְּנִיאֵל:  אֲרֵי חֲזֵיתִי מַלְאֲכָא דַּייָ אַפִּין בְּאַפִּין, וְאִשְׁתֵּיזַבַת נַפְשִׁי. 
לב,לב וַיִּֽזְרַֽח־ל֣וֹ הַשֶּׁ֔מֶשׁ כַּֽאֲשֶׁ֥ר עָבַ֖ר אֶת־פְּנוּאֵ֑ל וְה֥וּא צֹלֵ֖עַ עַל־יְרֵכֽוֹ׃ וּדְנַח לֵיהּ שִׁמְשָׁא, כַּד עֲבַר יָת פְּנוּאֵל; וְהוּא מַטְלַע, עַל יִרְכֵּיהּ. 
לב,לג עַל־כֵּ֡ן לֹֽא־יֹאכְל֨וּ בְנֵֽי־יִשְׂרָאֵ֜ל אֶת־גִּ֣יד הַנָּשֶׁ֗ה אֲשֶׁר֙ עַל־כַּ֣ף הַיָּרֵ֔ךְ עַ֖ד הַיּ֣וֹם הַזֶּ֑ה כִּ֤י נָגַע֙ בְּכַף־יֶ֣רֶךְ יַֽעֲקֹ֔ב בְּגִ֖יד הַנָּשֶֽׁה׃ עַל כֵּין לָא אָכְלִין בְּנֵי יִשְׂרָאֵל יָת גִּידָא נַשְׁיָא, דְּעַל פְּתֵי יִרְכָּא, עַד, יוֹמָא הָדֵין:  אֲרֵי קְרֵיב בִּפְתֵי יִרְכָּא דְּיַעֲקוֹב, בְּגִידָא נַשְׁיָא. 
לג,א וַיִּשָּׂ֨א יַֽעֲקֹ֜ב עֵינָ֗יו וַיַּרְא֙ וְהִנֵּ֣ה עֵשָׂ֣ו בָּ֔א וְעִמּ֕וֹ אַרְבַּ֥ע מֵא֖וֹת אִ֑ישׁ וַיַּ֣חַץ אֶת־הַיְלָדִ֗ים עַל־לֵאָה֙ וְעַל־רָחֵ֔ל וְעַ֖ל שְׁתֵּ֥י הַשְּׁפָחֽוֹת׃ וּזְקַף יַעֲקוֹב עֵינוֹהִי, וַחֲזָא וְהָא עֵשָׂו אָתֵי, וְעִמֵּיהּ, אַרְבַּע מְאָה גֻּבְרָא; וּפַלֵּיג יָת בְּנַיָּא, עַל לֵאָה וְעַל רָחֵל, וְעַל, תַּרְתֵּין לְחֵינָתָא. 
לג,ב וַיָּ֧שֶׂם אֶת־הַשְּׁפָח֛וֹת וְאֶת־יַלְדֵיהֶ֖ן רִֽאשֹׁנָ֑ה וְאֶת־לֵאָ֤ה וִֽילָדֶ֨יהָ֙ אַֽחֲרֹנִ֔ים וְאֶת־רָחֵ֥ל וְאֶת־יוֹסֵ֖ף אַֽחֲרֹנִֽים׃ וְשַׁוִּי יָת לְחֵינָתָא וְיָת בְּנֵיהוֹן, קַדְמוּתָא; וְיָת לֵאָה וּבְנַהָא בָּתְרָאִין, וְיָת רָחֵל וְיָת יוֹסֵף בָּתְרָאִין. 
לג,ג וְה֖וּא עָבַ֣ר לִפְנֵיהֶ֑ם וַיִּשְׁתַּ֤חוּ אַ֨רְצָה֙ שֶׁ֣בַע פְּעָמִ֔ים עַד־גִּשְׁתּ֖וֹ עַד־אָחִֽיו׃ וְהוּא, עֲבַר קֳדָמֵיהוֹן; וּסְגֵיד עַל אַרְעָא שְׁבַע זִמְנִין, עַד מִקְרְבֵיהּ לְוָת אֲחוּהִי. 
לג,ד וַיָּ֨רָץ עֵשָׂ֤ו לִקְרָאתוֹ֙ וַֽיְחַבְּקֵ֔הוּ וַיִּפֹּ֥ל עַל־צַוָּארָ֖ו וַׄיִּׄשָּׁׄקֵ֑ׄהׄוּׄ וַיִּבְכּֽוּ׃ וּרְהַט עֵשָׂו לְקַדָּמוּתֵיהּ וְגָפְפֵיהּ, וּנְפַל עַל צַוְרֵיהּ וְנַשְּׁקֵיהּ; וּבְכוֹ. 
לג,ה וַיִּשָּׂ֣א אֶת־עֵינָ֗יו וַיַּ֤רְא אֶת־הַנָּשִׁים֙ וְאֶת־הַיְלָדִ֔ים וַיֹּ֖אמֶר מִי־אֵ֣לֶּה לָּ֑ךְ וַיֹּאמַ֕ר הַיְלָדִ֕ים אֲשֶׁר־חָנַ֥ן אֱלֹהִ֖ים אֶת־עַבְדֶּֽךָ׃ וּזְקַף יָת עֵינוֹהִי, וַחֲזָא יָת נְשַׁיָּא וְיָת בְּנַיָּא, וַאֲמַר, מַאן אִלֵּין לָךְ; וַאֲמַר--בְּנַיָּא, דְּחָן יְיָ יָת עַבְדָּךְ. 
לג,ו וַתִּגַּ֧שְׁןָ הַשְּׁפָח֛וֹת הֵ֥נָּה וְיַלְדֵיהֶ֖ן וַתִּֽשְׁתַּחֲוֶֽיןָ׃ וּקְרִיבָא לְחֵינָתָא אִנִּין וּבְנֵיהוֹן, וּסְגִידָא. 
לג,ז וַתִּגַּ֧שׁ גַּם־לֵאָ֛ה וִֽילָדֶ֖יהָ וַיִּֽשְׁתַּחֲו֑וּ וְאַחַ֗ר נִגַּ֥שׁ יוֹסֵ֛ף וְרָחֵ֖ל וַיִּֽשְׁתַּחֲוֽוּ׃ וּקְרֵיבַת אַף לֵאָה וּבְנַהָא, וּסְגִידוּ; וּבָתַר כֵּין, קְרֵיב יוֹסֵף וְרָחֵל--וּסְגִידוּ. 
לג,ח וַיֹּ֕אמֶר מִ֥י לְךָ֛ כָּל־הַמַּֽחֲנֶ֥ה הַזֶּ֖ה אֲשֶׁ֣ר פָּגָ֑שְׁתִּי וַיֹּ֕אמֶר לִמְצֹא־חֵ֖ן בְּעֵינֵ֥י אֲדֹנִֽי׃ וַאֲמַר, מָא לָךְ כָּל מַשְׁרִיתָא הָדָא דְּעָרַעִית; וַאֲמַר, לְאַשְׁכָּחָא רַחֲמִין בְּעֵינֵי רִבּוֹנִי. 
לג,ט וַיֹּ֥אמֶר עֵשָׂ֖ו יֶשׁ־לִ֣י רָ֑ב אָחִ֕י יְהִ֥י לְךָ֖ אֲשֶׁר־לָֽךְ׃ וַאֲמַר עֵשָׂו, אִית לִי סַגִּי; אֲחִי, אַצְלַח בְּדִילָךְ. 
לג,י וַיֹּ֣אמֶר יַֽעֲקֹ֗ב אַל־נָא֙ אִם־נָ֨א מָצָ֤אתִי חֵן֙ בְּעֵינֶ֔יךָ וְלָֽקַחְתָּ֥ מִנְחָתִ֖י מִיָּדִ֑י כִּ֣י עַל־כֵּ֞ן רָאִ֣יתִי פָנֶ֗יךָ כִּרְאֹ֛ת פְּנֵ֥י אֱלֹהִ֖ים וַתִּרְצֵֽנִי׃ וַאֲמַר יַעֲקוֹב, בְּבָעוּ אִם כְּעַן אַשְׁכַּחִית רַחֲמִין בְּעֵינָךְ, וּתְקַבֵּיל תִּקְרֻבְתִּי, מִן יְדִי:  אֲרֵי עַל כֵּין חֲזֵיתִנּוּן לְאַפָּךְ, כְּחֵיזוּ אַפֵּי רַבְרְבַיָּא--וְאִתְרְעִית לִי. 
לג,יא קַח־נָ֤א אֶת־בִּרְכָתִי֙ אֲשֶׁ֣ר הֻבָ֣את לָ֔ךְ כִּֽי־חַנַּ֥נִי אֱלֹהִ֖ים וְכִ֣י יֶשׁ־לִי־כֹ֑ל וַיִּפְצַר־בּ֖וֹ וַיִּקָּֽח׃ קַבֵּיל כְּעַן יָת תִּקְרֻבְתִּי דְּאִתֵּיתִיאַת לָךְ, אֲרֵי רָחֵים עֲלַי יְיָ וַאֲרֵי אִית לִי כּוֹלָא; וְאַתְקֵיף בֵּיהּ, וְקַבֵּיל. 
לג,יב וַיֹּ֖אמֶר נִסְעָ֣ה וְנֵלֵ֑כָה וְאֵֽלְכָ֖ה לְנֶגְדֶּֽךָ׃ וַאֲמַר, נִטּוֹל וּנְהָךְ; וַאֲהָךְ, לְקִבְלָךְ. 
לג,יג וַיֹּ֣אמֶר אֵלָ֗יו אֲדֹנִ֤י יֹדֵ֨עַ֙ כִּֽי־הַיְלָדִ֣ים רַכִּ֔ים וְהַצֹּ֥אן וְהַבָּקָ֖ר עָל֣וֹת עָלָ֑י וּדְפָקוּם֙ י֣וֹם אֶחָ֔ד וָמֵ֖תוּ כָּל־הַצֹּֽאן׃ וַאֲמַר לֵיהּ, רִבּוֹנִי יָדַע אֲרֵי יָנְקַיָּא רַכִּיכִין, וְעָנָא וְתוֹרֵי, מֵינְקָתָא עֲלָי; אִם אֶדְחוֹקִנּוּן יוֹמָא חַד, וִימוּתוּן כָּל עָנָא. 
לג,יד יַֽעֲבָר־נָ֥א אֲדֹנִ֖י לִפְנֵ֣י עַבְדּ֑וֹ וַֽאֲנִ֞י אֶֽתְנָהֲלָ֣ה לְאִטִּ֗י לְרֶ֨גֶל הַמְּלָאכָ֤ה אֲשֶׁר־לְפָנַי֙ וּלְרֶ֣גֶל הַיְלָדִ֔ים עַ֛ד אֲשֶׁר־אָבֹ֥א אֶל־אֲדֹנִ֖י שֵׂעִֽירָה׃ יִעְבַּר כְּעַן רִבּוֹנִי, קֳדָם עַבְדֵּיהּ; וַאֲנָא אֲדַבַּר בִּנְיָח, לְרֶגֶל עֲבִידְתָא דִּקְדָמַי וּלְרֶגֶל יָנְקַיָּא, עַד דְּאֵיתֵי לְוָת רִבּוֹנִי, לְשֵׂעִיר. 
לג,טו וַיֹּ֣אמֶר עֵשָׂ֔ו אַצִּֽיגָה־נָּ֣א עִמְּךָ֔ מִן־הָעָ֖ם אֲשֶׁ֣ר אִתִּ֑י וַיֹּ֨אמֶר֙ לָ֣מָּה זֶּ֔ה אֶמְצָא־חֵ֖ן בְּעֵינֵ֥י אֲדֹנִֽי׃ וַאֲמַר עֵשָׂו--אֶשְׁבּוֹק כְּעַן עִמָּךְ, מִן עַמָּא דְּעִמִּי; וַאֲמַר לְמָא דְּנָן, אַשְׁכַּח רַחֲמִין בְּעֵינֵי רִבּוֹנִי. 
לג,טז וַיָּשָׁב֩ בַּיּ֨וֹם הַה֥וּא עֵשָׂ֛ו לְדַרְכּ֖וֹ שֵׂעִֽירָה׃ וְתָב בְּיוֹמָא הַהוּא עֵשָׂו לְאוֹרְחֵיהּ, לְשֵׂעִיר. 
לג,יז וְיַֽעֲקֹב֙ נָסַ֣ע סֻכֹּ֔תָה וַיִּ֥בֶן ל֖וֹ בָּ֑יִת וּלְמִקְנֵ֨הוּ֙ עָשָׂ֣ה סֻכֹּ֔ת עַל־כֵּ֛ן קָרָ֥א שֵׁם־הַמָּק֖וֹם סֻכּֽוֹת׃ {ס} וְיַעֲקוֹב נְטַל לְסֻכּוֹת, וּבְנָא לֵיהּ בֵּיתָא; וְלִבְעִירֵיהּ עֲבַד מְטַלָּן, עַל כֵּין קְרָא שְׁמֵיהּ דְּאַתְרָא סֻכּוֹת. {ס}
לג,יח וַיָּבֹא֩ יַֽעֲקֹ֨ב שָׁלֵ֜ם עִ֣יר שְׁכֶ֗ם אֲשֶׁר֙ בְּאֶ֣רֶץ כְּנַ֔עַן בְּבֹא֖וֹ מִפַּדַּ֣ן אֲרָ֑ם וַיִּ֖חַן אֶת־פְּנֵ֥י הָעִֽיר׃ וַאֲתָא יַעֲקוֹב שְׁלִים לְקַרְתָּא דִּשְׁכֶם, דִּבְאַרְעָא דִּכְנַעַן, בְּמֵיתוֹהִי, מִפַּדַּן אֲרָם; וּשְׁרָא, לָקֳבֵיל קַרְתָּא. 
לג,יט וַיִּ֜קֶן אֶת־חֶלְקַ֣ת הַשָּׂדֶ֗ה אֲשֶׁ֤ר נָֽטָה־שָׁם֙ אָֽהֳל֔וֹ מִיַּ֥ד בְּנֵֽי־חֲמ֖וֹר אֲבִ֣י שְׁכֶ֑ם בְּמֵאָ֖ה קְשִׂיטָֽה׃ וּזְבַן יָת אַחְסָנַת חַקְלָא, דְּפַרְסֵיהּ תַּמָּן לְמַשְׁכְּנֵיהּ, מִיַּד בְּנֵי חֲמוֹר, אֲבוּהִי דִּשְׁכֶם--בִּמְאָה, חֻרְפָן. 
לג,כ וַיַּצֶּב־שָׁ֖ם מִזְבֵּ֑חַ וַיִּ֨קְרָא־ל֔וֹ אֵ֖ל אֱלֹהֵ֥י יִשְׂרָאֵֽל׃ {ס} וַאֲקֵים תַּמָּן, מַדְבְּחָא; וּפְלַח עֲלוֹהִי--קֳדָם אֵל, אֱלָהָא דְּיִשְׂרָאֵל. {ס}
לד,א וַתֵּצֵ֤א דִינָה֙ בַּת־לֵאָ֔ה אֲשֶׁ֥ר יָֽלְדָ֖ה לְיַֽעֲקֹ֑ב לִרְא֖וֹת בִּבְנ֥וֹת הָאָֽרֶץ׃ וּנְפַקַת דִּינָה בַּת לֵאָה, דִּילֵידַת לְיַעֲקוֹב, לְמִחְזֵי, בִּבְנָת אַרְעָא. 
לד,ב וַיַּ֨רְא אֹתָ֜הּ שְׁכֶ֧ם בֶּן־חֲמ֛וֹר הַֽחִוִּ֖י נְשִׂ֣יא הָאָ֑רֶץ וַיִּקַּ֥ח אֹתָ֛הּ וַיִּשְׁכַּ֥ב אֹתָ֖הּ וַיְעַנֶּֽהָ׃ וַחֲזָא יָתַהּ שְׁכֶם בַּר חֲמוֹר, חִוָּאָה--רַבָּא דְּאַרְעָא; וּדְבַר יָתַהּ וּשְׁכֵיב יָתַהּ, וְעַנְּיַהּ. 
לד,ג וַתִּדְבַּ֣ק נַפְשׁ֔וֹ בְּדִינָ֖ה בַּֽת־יַעֲקֹ֑ב וַיֶּֽאֱהַב֙ אֶת־הַֽנַּעֲרָ֔ וַיְדַבֵּ֖ר עַל־לֵ֥ב הַֽנַּעֲרָֽ׃ וְאִתְרְעִיאַת נַפְשֵׁיהּ, בְּדִינָה בַּת יַעֲקוֹב; וּרְחֵים, יָת עוּלֵימְתָא, וּמַלִּי, עַל לִבַּהּ דְּעוּלֵימְתָא. 
לד,ד וַיֹּ֣אמֶר שְׁכֶ֔ם אֶל־חֲמ֥וֹר אָבִ֖יו לֵאמֹ֑ר קַֽח־לִ֛י אֶת־הַיַּלְדָּ֥ה הַזֹּ֖את לְאִשָּֽׁה׃ וַאֲמַר שְׁכֶם, לַחֲמוֹר אֲבוּהִי לְמֵימַר:  סַב לִי יָת עוּלֵימְתָא הָדָא, לְאִתּוּ. 
לד,ה וְיַֽעֲקֹ֣ב שָׁמַ֗ע כִּ֤י טִמֵּא֙ אֶת־דִּינָ֣ה בִתּ֔וֹ וּבָנָ֛יו הָי֥וּ אֶת־מִקְנֵ֖הוּ בַּשָּׂדֶ֑ה וְהֶֽחֱרִ֥שׁ יַֽעֲקֹ֖ב עַד־בֹּאָֽם׃ וְיַעֲקוֹב שְׁמַע, אֲרֵי סַאֵיב יָת דִּינָה בְּרַתֵּיהּ, וּבְנוֹהִי הֲווֹ עִם גֵּיתוֹהִי, בְּחַקְלָא; וּשְׁתֵיק יַעֲקוֹב, עַד מֵיתֵיהוֹן. 
לד,ו וַיֵּצֵ֛א חֲמ֥וֹר אֲבִֽי־שְׁכֶ֖ם אֶֽל־יַעֲקֹ֑ב לְדַבֵּ֖ר אִתּֽוֹ׃ וּנְפַק חֲמוֹר אֲבוּהִי דִּשְׁכֶם, לְוָת יַעֲקוֹב, לְמַלָּלָא, עִמֵּיהּ. 
לד,ז וּבְנֵ֨י יַֽעֲקֹ֜ב בָּ֤אוּ מִן־הַשָּׂדֶה֙ כְּשָׁמְעָ֔ם וַיִּֽתְעַצְּבוּ֙ הָֽאֲנָשִׁ֔ים וַיִּ֥חַר לָהֶ֖ם מְאֹ֑ד כִּֽי־נְבָלָ֞ה עָשָׂ֣ה בְיִשְׂרָאֵ֗ל לִשְׁכַּב֙ אֶת־בַּֽת־יַעֲקֹ֔ב וְכֵ֖ן לֹ֥א יֵֽעָשֶֽׂה׃ וּבְנֵי יַעֲקוֹב עָאלוּ מִן חַקְלָא, כַּד שְׁמַעוּ, וְאִתְנְסִיסוּ גֻּבְרַיָּא, וּתְקֵיף לְהוֹן לַחְדָּא:  אֲרֵי קְלָנָא עֲבַד בְּיִשְׂרָאֵל, לְמִשְׁכַּב יָת בַּת יַעֲקוֹב, וְכֵין, לָא כָּשַׁר דְּיִתְעֲבֵיד. 
לד,ח וַיְדַבֵּ֥ר חֲמ֖וֹר אִתָּ֣ם לֵאמֹ֑ר שְׁכֶ֣ם בְּנִ֗י חָֽשְׁקָ֤ה נַפְשׁוֹ֙ בְּבִתְּכֶ֔ם תְּנ֨וּ נָ֥א אֹתָ֛הּ ל֖וֹ לְאִשָּֽׁה׃ וּמַלֵּיל חֲמוֹר, עִמְּהוֹן לְמֵימַר:  שְׁכֶם בְּרִי, אִתְרְעִיאַת נַפְשֵׁיהּ בִּבְרַתְּכוֹן--הַבוּ כְּעַן יָתַהּ לֵיהּ, לְאִתּוּ. 
לד,ט וְהִֽתְחַתְּנ֖וּ אֹתָ֑נוּ בְּנֹֽתֵיכֶם֙ תִּתְּנוּ־לָ֔נוּ וְאֶת־בְּנֹתֵ֖ינוּ תִּקְח֥וּ לָכֶֽם׃ וְאִתְחַתַּנוּ, בַּנָא:  בְּנָתְכוֹן, תִּתְּנוּן לַנָא, וְיָת בְּנָתַנָא, תִּסְּבוּן לְכוֹן. 
לד,י וְאִתָּ֖נוּ תֵּשֵׁ֑בוּ וְהָאָ֨רֶץ֙ תִּֽהְיֶ֣ה לִפְנֵיכֶ֔ם שְׁבוּ֙ וּסְחָר֔וּהָ וְהֵֽאָחֲז֖וּ בָּֽהּ׃ וְעִמַּנָא, תִּתְּבוּן; וְאַרְעָא, תְּהֵי קֳדָמֵיכוֹן--תִּיבוּ וַעֲבִידוּ בַּהּ סְחוֹרְתָא, וְאַחְסִינוּ בַּהּ. 
לד,יא וַיֹּ֤אמֶר שְׁכֶם֙ אֶל־אָבִ֣יהָ וְאֶל־אַחֶ֔יהָ אֶמְצָא־חֵ֖ן בְּעֵֽינֵיכֶ֑ם וַֽאֲשֶׁ֥ר תֹּֽאמְר֛וּ אֵלַ֖י אֶתֵּֽן׃ וַאֲמַר שְׁכֶם לַאֲבוּהָא וּלְאַחַהָא, אַשְׁכַּח רַחֲמִין בְּעֵינֵיכוֹן; וּדְתֵימְרוּן לִי, אֶתֵּין. 
לד,יב הַרְבּ֨וּ עָלַ֤י מְאֹד֙ מֹ֣הַר וּמַתָּ֔ן וְאֶ֨תְּנָ֔ה כַּֽאֲשֶׁ֥ר תֹּֽאמְר֖וּ אֵלָ֑י וּתְנוּ־לִ֥י אֶת־הַֽנַּעֲרָ֖ לְאִשָּֽׁה׃ אַסְגּוֹ עֲלַי לַחְדָּא, מֻהְרִין וּמַתְּנָן, וְאֶתֵּין, כְּמָא דְּתֵימְרוּן לִי; וְהַבוּ לִי יָת עוּלֵימְתָא, לְאִתּוּ. 
לד,יג וַיַּֽעֲנ֨וּ בְנֵֽי־יַעֲקֹ֜ב אֶת־שְׁכֶ֨ם וְאֶת־חֲמ֥וֹר אָבִ֛יו בְּמִרְמָ֖ה וַיְדַבֵּ֑רוּ אֲשֶׁ֣ר טִמֵּ֔א אֵ֖ת דִּינָ֥ה אֲחֹתָֽם׃ וַאֲתִיבוּ בְּנֵי יַעֲקוֹב יָת שְׁכֶם וְיָת חֲמוֹר אֲבוּהִי, בְּחָכְמָא--וּמַלִּילוּ:  דְּסַאֵיב, יָת דִּינָה אֲחָתְהוֹן. 
לד,יד וַיֹּֽאמְר֣וּ אֲלֵיהֶ֗ם לֹ֤א נוּכַל֙ לַֽעֲשׂוֹת֙ הַדָּבָ֣ר הַזֶּ֔ה לָתֵת֙ אֶת־אֲחֹתֵ֔נוּ לְאִ֖ישׁ אֲשֶׁר־ל֣וֹ עָרְלָ֑ה כִּֽי־חֶרְפָּ֥ה הִ֖וא לָֽנוּ׃ וַאֲמַרוּ לְהוֹן, לָא נִכּוֹל לְמֶעֱבַד פִּתְגָמָא הָדֵין--לְמִתַּן יָת אֲחָתַנָא, לִגְבַר דְּלֵיהּ עָרְלָה:  אֲרֵי חִסּוּדָא הִיא, לַנָא. 
לד,טו אַךְ־בְּזֹ֖את נֵא֣וֹת לָכֶ֑ם אִ֚ם תִּֽהְי֣וּ כָמֹ֔נוּ לְהִמֹּ֥ל לָכֶ֖ם כָּל־זָכָֽר׃ בְּרַם בְּדָא, נִטְּפַס לְכוֹן:  אִם תְּהוֹן כְּוָתַנָא, לְמִגְזַר לְכוֹן כָּל דְּכוּרָא. 
לד,טז וְנָתַ֤נּוּ אֶת־בְּנֹתֵ֨ינוּ֙ לָכֶ֔ם וְאֶת־בְּנֹֽתֵיכֶ֖ם נִֽקַּֽח־לָ֑נוּ וְיָשַׁ֣בְנוּ אִתְּכֶ֔ם וְהָיִ֖ינוּ לְעַ֥ם אֶחָֽד׃ וְנִתֵּין יָת בְּנָתַנָא לְכוֹן, וְיָת בְּנָתְכוֹן נִסַּב לַנָא; וְנִתֵּיב עִמְּכוֹן, וּנְהֵי לְעַמָּא חַד. 
לד,יז וְאִם־לֹ֧א תִשְׁמְע֛וּ אֵלֵ֖ינוּ לְהִמּ֑וֹל וְלָקַ֥חְנוּ אֶת־בִּתֵּ֖נוּ וְהָלָֽכְנוּ׃ וְאִם לָא תְּקַבְּלוּן מִנַּנָא, לְמִגְזַר--וְנִדְבַּר יָת בְּרַתַּנָא, וְנֵיזֵיל. 
לד,יח וַיִּֽיטְב֥וּ דִבְרֵיהֶ֖ם בְּעֵינֵ֣י חֲמ֑וֹר וּבְעֵינֵ֖י שְׁכֶ֥ם בֶּן־חֲמֽוֹר׃ וּשְׁפַרוּ פִּתְגָמֵיהוֹן, בְּעֵינֵי חֲמוֹר, וּבְעֵינֵי, שְׁכֶם בַּר חֲמוֹר. 
לד,יט וְלֹֽא־אֵחַ֤ר הַנַּ֨עַר֙ לַֽעֲשׂ֣וֹת הַדָּבָ֔ר כִּ֥י חָפֵ֖ץ בְּבַֽת־יַעֲקֹ֑ב וְה֣וּא נִכְבָּ֔ד מִכֹּ֖ל בֵּ֥ית אָבִֽיו׃ וְלָא אוֹחַר עוּלֵימָא לְמֶעֱבַד פִּתְגָמָא, אֲרֵי אִתְרְעִי בְּבַת יַעֲקוֹב; וְהוּא יַקִּיר, מִכֹּל בֵּית אֲבוּהִי. 
לד,כ וַיָּבֹ֥א חֲמ֛וֹר וּשְׁכֶ֥ם בְּנ֖וֹ אֶל־שַׁ֣עַר עִירָ֑ם וַֽיְדַבְּר֛וּ אֶל־אַנְשֵׁ֥י עִירָ֖ם לֵאמֹֽר׃ וַאֲתָא חֲמוֹר וּשְׁכֶם בְּרֵיהּ, לִתְרַע קַרְתְּהוֹן; וּמַלִּילוּ עִם אֲנָשֵׁי קַרְתְּהוֹן, לְמֵימַר. 
לד,כא הָֽאֲנָשִׁ֨ים הָאֵ֜לֶּה שְֽׁלֵמִ֧ים הֵ֣ם אִתָּ֗נוּ וְיֵֽשְׁב֤וּ בָאָ֨רֶץ֙ וְיִסְחֲר֣וּ אֹתָ֔הּ וְהָאָ֛רֶץ הִנֵּ֥ה רַֽחֲבַת־יָדַ֖יִם לִפְנֵיהֶ֑ם אֶת־בְּנֹתָם֙ נִקַּֽח־לָ֣נוּ לְנָשִׁ֔ים וְאֶת־בְּנֹתֵ֖ינוּ נִתֵּ֥ן לָהֶֽם׃ גֻּבְרַיָּא הָאִלֵּין שְׁלָם אִנּוּן עִמַּנָא, וְיִתְּבוּן בְּאַרְעָא וְיַעְבְּדוּן בַּהּ סְחוֹרְתָא, וְאַרְעָא הָא פַּתְיוּת יְדִין, קֳדָמֵיהוֹן; יָת בְּנָתְהוֹן נִסַּב לַנָא לִנְשִׁין, וְיָת בְּנָתַנָא נִתֵּין לְהוֹן. 
לד,כב אַךְ־בְּ֠זֹאת יֵאֹ֨תוּ לָ֤נוּ הָֽאֲנָשִׁים֙ לָשֶׁ֣בֶת אִתָּ֔נוּ לִֽהְי֖וֹת לְעַ֣ם אֶחָ֑ד בְּהִמּ֥וֹל לָ֨נוּ֙ כָּל־זָכָ֔ר כַּֽאֲשֶׁ֖ר הֵ֥ם נִמֹּלִֽים׃ בְּרַם בְּדָא יִטַּפְסוּן לַנָא גֻּבְרַיָּא, לְמִתַּב עִמַּנָא--לְמִהְוֵי, לְעַמָּא חַד:  בְּמִגְזַר לַנָא כָּל דְּכוּרָא, כְּמָא דְּאִנּוּן גְּזִירִין. 
לד,כג מִקְנֵהֶ֤ם וְקִנְיָנָם֙ וְכָל־בְּהֶמְתָּ֔ם הֲל֥וֹא לָ֖נוּ הֵ֑ם אַ֚ךְ נֵא֣וֹתָה לָהֶ֔ם וְיֵֽשְׁב֖וּ אִתָּֽנוּ׃ גֵּיתֵיהוֹן וְקִנְיָנְהוֹן וְכָל בְּעִירְהוֹן, הֲלָא דִּילַנָא אִנּוּן; בְּרַם נִטְּפַס לְהוֹן, וְיִתְּבוּן עִמַּנָא. 
לד,כד וַיִּשְׁמְע֤וּ אֶל־חֲמוֹר֙ וְאֶל־שְׁכֶ֣ם בְּנ֔וֹ כָּל־יֹֽצְאֵ֖י שַׁ֣עַר עִיר֑וֹ וַיִּמֹּ֨לוּ֙ כָּל־זָכָ֔ר כָּל־יֹֽצְאֵ֖י שַׁ֥עַר עִירֽוֹ׃ וְקַבִּילוּ מִן חֲמוֹר וּמִן שְׁכֶם בְּרֵיהּ, כָּל נָפְקֵי תְּרַע קַרְתֵּיהּ; וּגְזַרוּ, כָּל דְּכוּרָא--כָּל נָפְקֵי, תְּרַע קַרְתֵּיהּ. 
לד,כה וַיְהִי֩ בַיּ֨וֹם הַשְּׁלִישִׁ֜י בִּֽהְיוֹתָ֣ם כֹּֽאֲבִ֗ים וַיִּקְח֣וּ שְׁנֵֽי־בְנֵי־יַ֠עֲקֹב שִׁמְע֨וֹן וְלֵוִ֜י אֲחֵ֤י דִינָה֙ אִ֣ישׁ חַרְבּ֔וֹ וַיָּבֹ֥אוּ עַל־הָעִ֖יר בֶּ֑טַח וַיַּֽהַרְג֖וּ כָּל־זָכָֽר׃ וַהֲוָה בְּיוֹמָא תְּלִיתָאָה כַּד תְּקִיפוּ עֲלֵיהוֹן כֵּיבֵיהוֹן, וּנְסִיבוּ תְּרֵין בְּנֵי יַעֲקוֹב שִׁמְעוֹן וְלֵוִי אֲחֵי דִּינָה גְּבַר חַרְבֵּיהּ, וְעָאלוּ עַל קַרְתָּא, דְּיָתְבָא לְרֻחְצָן; וּקְטַלוּ, כָּל דְּכוּרָא. 
לד,כו וְאֶת־חֲמוֹר֙ וְאֶת־שְׁכֶ֣ם בְּנ֔וֹ הָֽרְג֖וּ לְפִי־חָ֑רֶב וַיִּקְח֧וּ אֶת־דִּינָ֛ה מִבֵּ֥ית שְׁכֶ֖ם וַיֵּצֵֽאוּ׃ וְיָת חֲמוֹר וְיָת שְׁכֶם בְּרֵיהּ, קְטַלוּ לְפִתְגָם דְּחָרֶב; וּדְבַרוּ יָת דִּינָה מִבֵּית שְׁכֶם, וּנְפַקוּ. 
לד,כז בְּנֵ֣י יַֽעֲקֹ֗ב בָּ֚אוּ עַל־הַ֣חֲלָלִ֔ים וַיָּבֹ֖זּוּ הָעִ֑יר אֲשֶׁ֥ר טִמְּא֖וּ אֲחוֹתָֽם׃ בְּנֵי יַעֲקוֹב, עָאלוּ לְחַלָּצָא קְטִילַיָּא, וּבַזוּ, קַרְתָּא--דְּסַאִיבוּ, אֲחָתְהוֹן. 
לד,כח אֶת־צֹאנָ֥ם וְאֶת־בְּקָרָ֖ם וְאֶת־חֲמֹֽרֵיהֶ֑ם וְאֵ֧ת אֲשֶׁר־בָּעִ֛יר וְאֶת־אֲשֶׁ֥ר בַּשָּׂדֶ֖ה לָקָֽחוּ׃ יָת עָנְהוֹן וְיָת תּוֹרֵיהוֹן, וְיָת חֲמָרֵיהוֹן, וְיָת דִּבְקַרְתָּא וְיָת דִּבְחַקְלָא, בַּזוּ. 
לד,כט וְאֶת־כָּל־חֵילָ֤ם וְאֶת־כָּל־טַפָּם֙ וְאֶת־נְשֵׁיהֶ֔ם שָׁב֖וּ וַיָּבֹ֑זּוּ וְאֵ֖ת כָּל־אֲשֶׁ֥ר בַּבָּֽיִת׃ וְיָת כָּל נִכְסֵיהוֹן וְיָת כָּל טַפְלְהוֹן וְיָת נְשֵׁיהוֹן, שְׁבוֹ וּבַזוּ; וְיָת, כָּל דִּבְבֵּיתָא. 
לד,ל וַיֹּ֨אמֶר יַֽעֲקֹ֜ב אֶל־שִׁמְע֣וֹן וְאֶל־לֵוִי֮ עֲכַרְתֶּ֣ם אֹתִי֒ לְהַבְאִישֵׁ֨נִי֙ בְּיֹשֵׁ֣ב הָאָ֔רֶץ בַּֽכְּנַעֲנִ֖י וּבַפְּרִזִּ֑י וַֽאֲנִי֙ מְתֵ֣י מִסְפָּ֔ר וְנֶֽאֶסְפ֤וּ עָלַי֙ וְהִכּ֔וּנִי וְנִשְׁמַדְתִּ֖י אֲנִ֥י וּבֵיתִֽי׃ וַאֲמַר יַעֲקוֹב לְשִׁמְעוֹן וּלְלֵוִי, עֲכַרְתּוּן יָתִי, לְמִתַּן דְּבָבוּ בֵּינָא וּבֵין יָתֵיב אַרְעָא, בִּכְנַעֲנָאָה וּבִפְרִזָּאָה; וַאֲנָא, עַם דְּמִנְיָן, וְיִתְכַּנְשׁוּן עֲלַי וְיִמְחוֹנַנִי, וְאֶשְׁתֵּיצֵי אֲנָא וַאֲנָשׁ בֵּיתִי. 
לד,לא וַיֹּֽאמְר֑וּ הַכְזוֹנָ֕ה יַֽעֲשֶׂ֖ה אֶת־אֲחוֹתֵֽנוּ׃ {פ} וַאֲמַרוּ:  הַכְּנָפְקַת בָּרָא, יִתְעֲבֵיד לַאֲחָתַנָא. {פ}
לה,א וַיֹּ֤אמֶר אֱלֹהִים֙ אֶֽל־יַעֲקֹ֔ב ק֛וּם עֲלֵ֥ה בֵֽית־אֵ֖ל וְשֶׁב־שָׁ֑ם וַֽעֲשֵׂה־שָׁ֣ם מִזְבֵּ֔חַ לָאֵל֙ הַנִּרְאֶ֣ה אֵלֶ֔יךָ בְּבָ֨רְחֲךָ֔ מִפְּנֵ֖י עֵשָׂ֥ו אָחִֽיךָ׃ וַאֲמַר יְיָ לְיַעֲקוֹב, קוּם סַק לְבֵית אֵל וְתִיב תַּמָּן; וַעֲבֵיד תַּמָּן מַדְבַּח--לְאֵל דְּאִתְגְּלִי לָךְ, בְּמִעְרְקָךְ מִן קֳדָם עֵשָׂו אֲחוּךְ. 
לה,ב וַיֹּ֤אמֶר יַֽעֲקֹב֙ אֶל־בֵּית֔וֹ וְאֶ֖ל כָּל־אֲשֶׁ֣ר עִמּ֑וֹ הָסִ֜רוּ אֶת־אֱלֹהֵ֤י הַנֵּכָר֙ אֲשֶׁ֣ר בְּתֹֽכְכֶ֔ם וְהִֽטַּהֲר֔וּ וְהַֽחֲלִ֖יפוּ שִׂמְלֹֽתֵיכֶֽם׃ וַאֲמַר יַעֲקוֹב לַאֲנָשׁ בֵּיתֵיהּ, וּלְכָל דְּעִמֵּיהּ:  אַעְדּוֹ יָת טָעֲוָת עַמְמַיָּא, דְּבֵינֵיכוֹן, וְאִדַּכּוֹ, וְשַׁנּוֹ כְּסוּתְכוֹן. 
לה,ג וְנָק֥וּמָה וְנַֽעֲלֶ֖ה בֵּֽית־אֵ֑ל וְאֶֽעֱשֶׂה־שָּׁ֣ם מִזְבֵּ֗חַ לָאֵ֞ל הָֽעֹנֶ֤ה אֹתִי֙ בְּי֣וֹם צָֽרָתִ֔י וַֽיְהִי֙ עִמָּדִ֔י בַּדֶּ֖רֶךְ אֲשֶׁ֥ר הָלָֽכְתִּי׃ וּנְקוּם וְנִסַּק, לְבֵית אֵל; וְאַעֲבֵיד תַּמָּן מַדְבַּח, לְאֵל דְּקַבֵּיל צְלוֹתִי בְּיוֹמָא דְּעָקְתִי, וַהֲוָה מֵימְרֵיהּ בְּסַעֲדִי, בְּאוֹרְחָא דַּאֲזַלִית. 
לה,ד וַיִּתְּנ֣וּ אֶֽל־יַעֲקֹ֗ב אֵ֣ת כָּל־אֱלֹהֵ֤י הַנֵּכָר֙ אֲשֶׁ֣ר בְּיָדָ֔ם וְאֶת־הַנְּזָמִ֖ים אֲשֶׁ֣ר בְּאָזְנֵיהֶ֑ם וַיִּטְמֹ֤ן אֹתָם֙ יַֽעֲקֹ֔ב תַּ֥חַת הָֽאֵלָ֖ה אֲשֶׁ֥ר עִם־שְׁכֶֽם׃ וִיהַבוּ לְיַעֲקוֹב, יָת כָּל טָעֲוָת עַמְמַיָּא דִּבְיַדְהוֹן, וְיָת קְדָשַׁיָּא, דִּבְאֻדְנֵיהוֹן; וְטַמַּר יָתְהוֹן יַעֲקוֹב, תְּחוֹת בֻּטְמָא דְּעִם שְׁכֶם. 
לה,ה וַיִּסָּ֑עוּ וַיְהִ֣י ׀ חִתַּ֣ת אֱלֹהִ֗ים עַל־הֶֽעָרִים֙ אֲשֶׁר֙ סְבִיב֣וֹתֵיהֶ֔ם וְלֹ֣א רָֽדְפ֔וּ אַֽחֲרֵ֖י בְּנֵ֥י יַֽעֲקֹֽב׃ וּנְטַלוּ; וַהֲוָת דַּחְלָא מִן קֳדָם יְיָ, עַל עַמְמַיָּא דִּבְקִרְוֵי סַחְרָנֵיהוֹן, וְלָא רְדַפוּ, בָּתַר בְּנֵי יַעֲקוֹב. 
לה,ו וַיָּבֹ֨א יַֽעֲקֹ֜ב ל֗וּזָה אֲשֶׁר֙ בְּאֶ֣רֶץ כְּנַ֔עַן הִ֖וא בֵּֽית־אֵ֑ל ה֖וּא וְכָל־הָעָ֥ם אֲשֶׁר־עִמּֽוֹ׃ וַאֲתָא יַעֲקוֹב לְלוּז, דִּבְאַרְעָא דִּכְנַעַן--הִיא, בֵּית אֵל:  הוּא, וְכָל עַמָּא דְּעִמֵּיהּ. 
לה,ז וַיִּ֤בֶן שָׁם֙ מִזְבֵּ֔חַ וַיִּקְרָא֙ לַמָּק֔וֹם אֵ֖ל בֵּֽית־אֵ֑ל כִּ֣י שָׁ֗ם נִגְל֤וּ אֵלָיו֙ הָֽאֱלֹהִ֔ים בְּבָרְח֖וֹ מִפְּנֵ֥י אָחִֽיו׃ וּבְנָא תַּמָּן, מַדְבְּחָא, וּקְרָא לְאַתְרָא, אֵל בֵּית אֵל:  אֲרֵי תַּמָּן, אִתְגְּלִי לֵיהּ יְיָ, בְּמִעְרְקֵיהּ, מִן קֳדָם אֲחוּהִי. 
לה,ח וַתָּ֤מָת דְּבֹרָה֙ מֵינֶ֣קֶת רִבְקָ֔ה וַתִּקָּבֵ֛ר מִתַּ֥חַת לְבֵֽית־אֵ֖ל תַּ֣חַת הָֽאַלּ֑וֹן וַיִּקְרָ֥א שְׁמ֖וֹ אַלּ֥וֹן בָּכֽוּת׃ {פ} וּמֵיתַת דְּבוֹרָה מֵינִקְתַּהּ דְּרִבְקָה, וְאִתְקְבַרַת מִלְּרַע לְבֵית אֵל בְּשִׁפּוֹלֵי מֵישְׁרָא; וּקְרָא שְׁמֵיהּ, מֵישַׁר בְּכִיתָא. {פ}
לה,ט וַיֵּרָ֨א אֱלֹהִ֤ים אֶֽל־יַעֲקֹב֙ ע֔וֹד בְּבֹא֖וֹ מִפַּדַּ֣ן אֲרָ֑ם וַיְבָ֖רֶךְ אֹתֽוֹ׃ וְאִתְגְּלִי יְיָ לְיַעֲקוֹב עוֹד, בְּמֵיתוֹהִי מִפַּדַּן אֲרָם; וּבָרֵיךְ, יָתֵיהּ. 
לה,י וַיֹּֽאמֶר־ל֥וֹ אֱלֹהִ֖ים שִׁמְךָ֣ יַֽעֲקֹ֑ב לֹֽא־יִקָּרֵא֩ שִׁמְךָ֨ ע֜וֹד יַֽעֲקֹ֗ב כִּ֤י אִם־יִשְׂרָאֵל֙ יִֽהְיֶ֣ה שְׁמֶ֔ךָ וַיִּקְרָ֥א אֶת־שְׁמ֖וֹ יִשְׂרָאֵֽל׃ וַאֲמַר לֵיהּ יְיָ, שְׁמָךְ יַעֲקוֹב:  לָא יִתְקְרֵי שְׁמָךְ עוֹד יַעֲקוֹב, אֱלָהֵין יִשְׂרָאֵל יְהֵי שְׁמָךְ, וּקְרָא יָת שְׁמֵיהּ, יִשְׂרָאֵל. 
לה,יא וַיֹּאמֶר֩ ל֨וֹ אֱלֹהִ֜ים אֲנִ֨י אֵ֤ל שַׁדַּי֙ פְּרֵ֣ה וּרְבֵ֔ה גּ֛וֹי וּקְהַ֥ל גּוֹיִ֖ם יִֽהְיֶ֣ה מִמֶּ֑ךָּ וּמְלָכִ֖ים מֵֽחֲלָצֶ֥יךָ יֵצֵֽאוּ׃ וַאֲמַר לֵיהּ יְיָ אֲנָא אֵל שַׁדַּי, פּוּשׁ וּסְגִי--עַם וְכִנְשָׁת שִׁבְטִין, יְהוֹן מִנָּךְ; וּמַלְכִין דְּשָׁלְטִין בְּעַמְמַיָּא, מִנָּךְ יִפְּקוּן. 
לה,יב וְאֶת־הָאָ֗רֶץ אֲשֶׁ֥ר נָתַ֛תִּי לְאַבְרָהָ֥ם וּלְיִצְחָ֖ק לְךָ֣ אֶתְּנֶ֑נָּה וּֽלְזַרְעֲךָ֥ אַֽחֲרֶ֖יךָ אֶתֵּ֥ן אֶת־הָאָֽרֶץ׃ וְיָת אַרְעָא, דִּיהַבִית לְאַבְרָהָם וּלְיִצְחָק--לָךְ אֶתְּנִנַּהּ; וְלִבְנָךְ בָּתְרָךְ, אֶתֵּין יָת אַרְעָא. 
לה,יג וַיַּ֥עַל מֵֽעָלָ֖יו אֱלֹהִ֑ים בַּמָּק֖וֹם אֲשֶׁר־דִּבֶּ֥ר אִתּֽוֹ׃ וְאִסְתַּלַּק מֵעִלָּווֹהִי, יְקָרָא דַּייָ, בְּאַתְרָא, דְּמַלֵּיל עִמֵּיהּ. 
לה,יד וַיַּצֵּ֨ב יַֽעֲקֹ֜ב מַצֵּבָ֗ה בַּמָּק֛וֹם אֲשֶׁר־דִּבֶּ֥ר אִתּ֖וֹ מַצֶּ֣בֶת אָ֑בֶן וַיַּסֵּ֤ךְ עָלֶ֨יהָ֙ נֶ֔סֶךְ וַיִּצֹ֥ק עָלֶ֖יהָ שָֽׁמֶן׃ וַאֲקֵים יַעֲקוֹב קָמְתָא, בְּאַתְרָא דְּמַלֵּיל עִמֵּיהּ--קָמַת אַבְנָא; וְנַסֵּיךְ עֲלַהּ נִסְכִּין, וַאֲרֵיק עֲלַהּ מִשְׁחָא. 
לה,טו וַיִּקְרָ֨א יַֽעֲקֹ֜ב אֶת־שֵׁ֣ם הַמָּק֗וֹם אֲשֶׁר֩ דִּבֶּ֨ר אִתּ֥וֹ שָׁ֛ם אֱלֹהִ֖ים בֵּֽית־אֵֽל׃ וּקְרָא יַעֲקוֹב יָת שְׁמֵיהּ דְּאַתְרָא, דְּמַלֵּיל עִמֵּיהּ תַּמָּן יְיָ--בֵּית אֵל. 
לה,טז וַיִּסְעוּ֙ מִבֵּ֣ית אֵ֔ל וַֽיְהִי־ע֥וֹד כִּבְרַת־הָאָ֖רֶץ לָב֣וֹא אֶפְרָ֑תָה וַתֵּ֥לֶד רָחֵ֖ל וַתְּקַ֥שׁ בְּלִדְתָּֽהּ׃ וּנְטַלוּ מִבֵּית אֵל, וַהֲוָה עוֹד כְּרוֹב אַרְעָא לְמֵיעַל לְאֶפְרָת; וִילֵידַת רָחֵל, וְקַשִּׁיאַת בְּמֵילְדַהּ. 
לה,יז וַיְהִ֥י בְהַקְשֹׁתָ֖הּ בְּלִדְתָּ֑הּ וַתֹּ֨אמֶר לָ֤הּ הַמְיַלֶּ֨דֶת֙ אַל־תִּ֣ירְאִ֔י כִּֽי־גַם־זֶ֥ה לָ֖ךְ בֵּֽן׃ וַהֲוָה בְּקַשָּׁיוּתַהּ, בְּמֵילְדַהּ; וַאֲמַרַת לַהּ חָיְתָא לָא תִּדְחֲלִין, אֲרֵי אַף דֵּין לִיךְ בַּר. 
לה,יח וַיְהִ֞י בְּצֵ֤את נַפְשָׁהּ֙ כִּ֣י מֵ֔תָה וַתִּקְרָ֥א שְׁמ֖וֹ בֶּן־אוֹנִ֑י וְאָבִ֖יו קָֽרָא־ל֥וֹ בִנְיָמִֽין׃ וַהֲוָה בְּמִפַּק נַפְשַׁהּ, אֲרֵי מָיְתָא, וּקְרָת שְׁמֵיהּ, בַּר דְּוָיִי; וַאֲבוּהִי, קְרָא לֵיהּ בִּנְיָמִין. 
לה,יט וַתָּ֖מָת רָחֵ֑ל וַתִּקָּבֵר֙ בְּדֶ֣רֶךְ אֶפְרָ֔תָה הִ֖וא בֵּ֥ית לָֽחֶם׃ וּמֵיתַת, רָחֵל; וְאִתְקְבַרַת בְּאוֹרַח אֶפְרָת, הִיא בֵּית לָחֶם. 
לה,כ וַיַּצֵּ֧ב יַֽעֲקֹ֛ב מַצֵּבָ֖ה עַל־קְבֻֽרָתָ֑הּ הִ֛וא מַצֶּ֥בֶת קְבֻֽרַת־רָחֵ֖ל עַד־הַיּֽוֹם׃ וַאֲקֵים יַעֲקוֹב קָמְתָא, עַל קְבוּרְתַהּ--הִיא קָמַת קְבוּרְתָא דְּרָחֵל, עַד יוֹמָא דֵּין. 
לה,כא וַיִּסַּ֖ע יִשְׂרָאֵ֑ל וַיֵּ֣ט אָֽהֳלֹ֔ה מֵהָ֖לְאָה לְמִגְדַּל־עֵֽדֶר׃ וּנְטַל, יִשְׂרָאֵל; וּפַרְסֵיהּ לְמַשְׁכְּנֵיהּ, מֵהָלְאָה לְמִגְדְּלָא דְּעֵדֶר. 
לה,כב * וַיְהִ֗י בִּשְׁכֹּ֤ן יִשְׂרָאֵל֙ בָּאָ֣רֶץ הַהִ֔וא וַיֵּ֣לֶךְ רְאוּבֵ֗ן וַיִּשְׁכַּב֙ אֶת־בִּלְהָה֙ פִּילֶ֣גֶשׁ אָבִ֔יו וַיִּשְׁמַ֖ע יִשְׂרָאֵ֑ל {פ}
וַיִּֽהְי֥וּ בְנֵֽי־יַעֲקֹ֖ב שְׁנֵ֥ים עָשָֽׂר׃ 		וַהֲוָה, כַּד שְׁרָא יִשְׂרָאֵל בְּאַרְעָא הַהִיא, וַאֲזַל רְאוּבֵן וּשְׁכֵיב יָת בִּלְהָה לְחֵינְתָא דַּאֲבוּהִי, וּשְׁמַע יִשְׂרָאֵל;  {פ}
וַהֲווֹ בְּנֵי יַעֲקוֹב, תְּרֵי עֲסַר. 
לה,כג בְּנֵ֣י לֵאָ֔ה בְּכ֥וֹר יַֽעֲקֹ֖ב רְאוּבֵ֑ן וְשִׁמְעוֹן֙ וְלֵוִ֣י וִֽיהוּדָ֔ה וְיִשָּׂשכָ֖ר וּזְבֻלוּֽן׃ בְּנֵי לֵאָה, בֻּכְרֵיהּ דְּיַעֲקוֹב רְאוּבֵן; וְשִׁמְעוֹן וְלֵוִי וִיהוּדָה, וְיִשָּׂשכָר וּזְבוּלוּן. 
לה,כד בְּנֵ֣י רָחֵ֔ל יוֹסֵ֖ף וּבִנְיָמִֽן׃ בְּנֵי רָחֵל, יוֹסֵף וּבִנְיָמִין. 
לה,כה וּבְנֵ֤י בִלְהָה֙ שִׁפְחַ֣ת רָחֵ֔ל דָּ֖ן וְנַפְתָּלִֽי׃ וּבְנֵי בִּלְהָה אַמְתַּהּ דְּרָחֵל, דָּן וְנַפְתָּלִי. 
לה,כו וּבְנֵ֥י זִלְפָּ֛ה שִׁפְחַ֥ת לֵאָ֖ה גָּ֣ד וְאָשֵׁ֑ר אֵ֚לֶּה בְּנֵ֣י יַֽעֲקֹ֔ב אֲשֶׁ֥ר יֻלַּד־ל֖וֹ בְּפַדַּ֥ן אֲרָֽם׃ וּבְנֵי זִלְפָּה אַמְתַּהּ דְּלֵאָה, גָּד וְאָשֵׁר; אִלֵּין בְּנֵי יַעֲקוֹב, דְּאִתְיְלִידוּ לֵיהּ בְּפַדַּן אֲרָם. 
לה,כז וַיָּבֹ֤א יַֽעֲקֹב֙ אֶל־יִצְחָ֣ק אָבִ֔יו מַמְרֵ֖א קִרְיַ֣ת הָֽאַרְבַּ֑ע הִ֣וא חֶבְר֔וֹן אֲשֶׁר־גָּֽר־שָׁ֥ם אַבְרָהָ֖ם וְיִצְחָֽק׃ וַאֲתָא יַעֲקוֹב לְוָת יִצְחָק אֲבוּהִי, לְמַמְרֵא קִרְיַת אַרְבַּע--הִיא חֶבְרוֹן, דְּדָר תַּמָּן אַבְרָהָם וְיִצְחָק. 
לה,כח וַיִּֽהְי֖וּ יְמֵ֣י יִצְחָ֑ק מְאַ֥ת שָׁנָ֖ה וּשְׁמֹנִ֥ים שָׁנָֽה׃ וַהֲווֹ, יוֹמֵי יִצְחָק--מְאָה וּתְמָנַן, שְׁנִין. 
לה,כט וַיִּגְוַ֨ע יִצְחָ֤ק וַיָּ֨מָת֙ וַיֵּאָ֣סֶף אֶל־עַמָּ֔יו זָקֵ֖ן וּשְׂבַ֣ע יָמִ֑ים וַיִּקְבְּר֣וּ אֹת֔וֹ עֵשָׂ֥ו וְיַֽעֲקֹ֖ב בָּנָֽיו׃ {פ} וְאִתְנְגִיד יִצְחָק וּמִית וְאִתְכְּנֵישׁ לְעַמֵּיהּ, סִיב וּסְבַע יוֹמִין; וּקְבַרוּ יָתֵיהּ, עֵשָׂו וְיַעֲקוֹב בְּנוֹהִי. {פ}
לו,א וְאֵ֛לֶּה תֹּֽלְד֥וֹת עֵשָׂ֖ו ה֥וּא אֱדֽוֹם׃ וְאִלֵּין תּוֹלְדָת עֵשָׂו, הוּא אֱדוֹם. 
לו,ב עֵשָׂ֛ו לָקַ֥ח אֶת־נָשָׁ֖יו מִבְּנ֣וֹת כְּנָ֑עַן אֶת־עָדָ֗ה בַּת־אֵילוֹן֙ הַֽחִתִּ֔י וְאֶת־אָהֳלִֽיבָמָה֙ בַּת־עֲנָ֔ה בַּת־צִבְע֖וֹן הַֽחִוִּֽי׃ עֵשָׂו נְסֵיב יָת נְשׁוֹהִי, מִבְּנָת כְּנָעַן:  יָת עָדָה, בַּת אֵילוֹן חִתָּאָה, וְיָת אָהֳלִיבָמָה בַּת עֲנָה, בַּת צִבְעוֹן חִוָּאָה. 
לו,ג וְאֶת־בָּֽשְׂמַ֥ת בַּת־יִשְׁמָעֵ֖אל אֲח֥וֹת נְבָיֽוֹת׃ וְיָת בָּשְׂמַת בַּת יִשְׁמָעֵאל, אֲחָתֵיהּ דִּנְבָיוֹת. 
לו,ד וַתֵּ֧לֶד עָדָ֛ה לְעֵשָׂ֖ו אֶת־אֱלִיפָ֑ז וּבָ֣שְׂמַ֔ת יָֽלְדָ֖ה אֶת־רְעוּאֵֽל׃ וִילֵידַת עָדָה לְעֵשָׂו, יָת אֱלִיפָז; וּבָשְׂמַת, יְלֵידַת יָת רְעוּאֵל. 
לו,ה וְאָהֳלִֽיבָמָה֙ יָֽלְדָ֔ה אֶת־יעיש (יְע֥וּשׁ) וְאֶת־יַעְלָ֖ם וְאֶת־קֹ֑רַח אֵ֚לֶּה בְּנֵ֣י עֵשָׂ֔ו אֲשֶׁ֥ר יֻלְּדוּ־ל֖וֹ בְּאֶ֥רֶץ כְּנָֽעַן׃ וְאָהֳלִיבָמָה, יְלֵידַת, יָת יְעוּשׁ וְיָת יַעְלָם, וְיָת קֹרַח; אִלֵּין בְּנֵי עֵשָׂו, דְּאִתְיְלִידוּ לֵיהּ בְּאַרְעָא דִּכְנָעַן. 
לו,ו וַיִּקַּ֣ח עֵשָׂ֡ו אֶת־נָ֠שָׁיו וְאֶת־בָּנָ֣יו וְאֶת־בְּנֹתָיו֮ וְאֶת־כָּל־נַפְשׁ֣וֹת בֵּיתוֹ֒ וְאֶת־מִקְנֵ֣הוּ וְאֶת־כָּל־בְּהֶמְתּ֗וֹ וְאֵת֙ כָּל־קִנְיָנ֔וֹ אֲשֶׁ֥ר רָכַ֖שׁ בְּאֶ֣רֶץ כְּנָ֑עַן וַיֵּ֣לֶךְ אֶל־אֶ֔רֶץ מִפְּנֵ֖י יַֽעֲקֹ֥ב אָחִֽיו׃ וּדְבַר עֵשָׂו יָת נְשׁוֹהִי וְיָת בְּנוֹהִי וְיָת בְּנָתֵיהּ, וְיָת כָּל נַפְשָׁת בֵּיתֵיהּ, וְיָת גֵּיתוֹהִי וְיָת כָּל בְּעִירֵיהּ וְיָת כָּל קִנְיָנֵיהּ, דִּקְנָא בְּאַרְעָא דִּכְנָעַן; וַאֲזַל לַאֲרַע אֻחְרִי, מִן קֳדָם יַעֲקוֹב אֲחוּהִי. 
לו,ז כִּֽי־הָיָ֧ה רְכוּשָׁ֛ם רָ֖ב מִשֶּׁ֣בֶת יַחְדָּ֑ו וְלֹ֨א יָֽכְלָ֜ה אֶ֤רֶץ מְגֽוּרֵיהֶם֙ לָשֵׂ֣את אֹתָ֔ם מִפְּנֵ֖י מִקְנֵיהֶֽם׃ אֲרֵי הֲוָה קִנְיָנְהוֹן סַגִּי, מִלְּמִתַּב כַּחְדָּא; וְלָא יְכֵילַת אֲרַע תּוֹתָבוּתְהוֹן, לְסוֹבָרָא יָתְהוֹן--מִן קֳדָם, גֵּיתֵיהוֹן. 
לו,ח וַיֵּ֤שֶׁב עֵשָׂו֙ בְּהַ֣ר שֵׂעִ֔יר עֵשָׂ֖ו ה֥וּא אֱדֽוֹם׃ וִיתֵיב עֵשָׂו בְּטוּרָא דְּשֵׂעִיר, עֵשָׂו הוּא אֱדוֹם. 
לו,ט וְאֵ֛לֶּה תֹּֽלְד֥וֹת עֵשָׂ֖ו אֲבִ֣י אֱד֑וֹם בְּהַ֖ר שֵׂעִֽיר׃ וְאִלֵּין תּוֹלְדָת עֵשָׂו, אֲבוּהוֹן דֶּאֱדוֹמָאֵי, בְּטוּרָא, דְּשֵׂעִיר. 
לו,י אֵ֖לֶּה שְׁמ֣וֹת בְּנֵֽי־עֵשָׂ֑ו אֱלִיפַ֗ז בֶּן־עָדָה֙ אֵ֣שֶׁת עֵשָׂ֔ו רְעוּאֵ֕ל בֶּן־בָּֽשְׂמַ֖ת אֵ֥שֶׁת עֵשָֽׂו׃ אִלֵּין, שְׁמָהָת בְּנֵי עֵשָׂו:  אֱלִיפַז, בַּר עָדָה אִתַּת עֵשָׂו, רְעוּאֵל, בַּר בָּשְׂמַת אִתַּת עֵשָׂו. 
לו,יא וַיִּֽהְי֖וּ בְּנֵ֣י אֱלִיפָ֑ז תֵּימָ֣ן אוֹמָ֔ר צְפ֥וֹ וְגַעְתָּ֖ם וּקְנַֽז׃ וַהֲווֹ, בְּנֵי אֱלִיפַז--תֵּימָן אוֹמָר, צְפוֹ וְגַעְתָּם וּקְנַז. 
לו,יב וְתִמְנַ֣ע ׀ הָֽיְתָ֣ה פִילֶ֗גֶשׁ לֶֽאֱלִיפַז֙ בֶּן־עֵשָׂ֔ו וַתֵּ֥לֶד לֶֽאֱלִיפַ֖ז אֶת־עֲמָלֵ֑ק אֵ֕לֶּה בְּנֵ֥י עָדָ֖ה אֵ֥שֶׁת עֵשָֽׂו׃ וְתִמְנַע הֲוָת לְחֵינְתָא, לֶאֱלִיפַז בַּר עֵשָׂו, וִילֵידַת לֶאֱלִיפַז, יָת עֲמָלֵק; אִלֵּין, בְּנֵי עָדָה אִתַּת עֵשָׂו. 
לו,יג וְאֵ֨לֶּה֙ בְּנֵ֣י רְעוּאֵ֔ל נַ֥חַת וָזֶ֖רַח שַׁמָּ֣ה וּמִזָּ֑ה אֵ֣לֶּה הָי֔וּ בְּנֵ֥י בָֽשְׂמַ֖ת אֵ֥שֶׁת עֵשָֽׂו׃ וְאִלֵּין בְּנֵי רְעוּאֵל, נַחַת וָזֶרַח שַׁמָּה וּמִזָּה; אִלֵּין הֲווֹ, בְּנֵי בָּשְׂמַת אִתַּת עֵשָׂו. 
לו,יד וְאֵ֣לֶּה הָי֗וּ בְּנֵ֨י אָהֳלִֽיבָמָ֧ה בַת־עֲנָ֛ה בַּת־צִבְע֖וֹן אֵ֣שֶׁת עֵשָׂ֑ו וַתֵּ֣לֶד לְעֵשָׂ֔ו אֶת־יעיש (יְע֥וּשׁ) וְאֶת־יַעְלָ֖ם וְאֶת־קֹֽרַח׃ וְאִלֵּין הֲווֹ, בְּנֵי אָהֳלִיבָמָה בַּת עֲנָה בַּת צִבְעוֹן--אִתַּת עֵשָׂו; וִילֵידַת לְעֵשָׂו, יָת יְעוּשׁ וְיָת יַעְלָם וְיָת קֹרַח. 
לו,טו אֵ֖לֶּה אַלּוּפֵ֣י בְנֵֽי־עֵשָׂ֑ו בְּנֵ֤י אֱלִיפַז֙ בְּכ֣וֹר עֵשָׂ֔ו אַלּ֤וּף תֵּימָן֙ אַלּ֣וּף אוֹמָ֔ר אַלּ֥וּף צְפ֖וֹ אַלּ֥וּף קְנַֽז׃ אִלֵּין, רַבְרְבֵי בְּנֵי עֵשָׂו:  בְּנֵי אֱלִיפַז, בֻּכְרֵיהּ דְּעֵשָׂו--רַבָּא תֵּימָן רַבָּא אוֹמָר, רַבָּא צְפוֹ רַבָּא קְנַז. 
לו,טז אַלּֽוּף־קֹ֛רַח אַלּ֥וּף גַּעְתָּ֖ם אַלּ֣וּף עֲמָלֵ֑ק אֵ֣לֶּה אַלּוּפֵ֤י אֱלִיפַז֙ בְּאֶ֣רֶץ אֱד֔וֹם אֵ֖לֶּה בְּנֵ֥י עָדָֽה׃ רַבָּא קֹרַח רַבָּא גַּעְתָּם, רַבָּא עֲמָלֵק; אִלֵּין רַבְרְבֵי אֱלִיפַז בְּאַרְעָא דֶּאֱדוֹם, אִלֵּין בְּנֵי עָדָה. 
לו,יז וְאֵ֗לֶּה בְּנֵ֤י רְעוּאֵל֙ בֶּן־עֵשָׂ֔ו אַלּ֥וּף נַ֨חַת֙ אַלּ֣וּף זֶ֔רַח אַלּ֥וּף שַׁמָּ֖ה אַלּ֣וּף מִזָּ֑ה אֵ֣לֶּה אַלּוּפֵ֤י רְעוּאֵל֙ בְּאֶ֣רֶץ אֱד֔וֹם אֵ֕לֶּה בְּנֵ֥י בָֽשְׂמַ֖ת אֵ֥שֶׁת עֵשָֽׂו׃ וְאִלֵּין, בְּנֵי רְעוּאֵל בַּר עֵשָׂו--רַבָּא נַחַת רַבָּא זֶרַח, רַבָּא שַׁמָּה רַבָּא מִזָּה; אִלֵּין רַבְרְבֵי רְעוּאֵל, בְּאַרְעָא דֶּאֱדוֹם--אִלֵּין, בְּנֵי בָּשְׂמַת אִתַּת עֵשָׂו. 
לו,יח וְאֵ֗לֶּה בְּנֵ֤י אָהֳלִֽיבָמָה֙ אֵ֣שֶׁת עֵשָׂ֔ו אַלּ֥וּף יְע֛וּשׁ אַלּ֥וּף יַעְלָ֖ם אַלּ֣וּף קֹ֑רַח אֵ֣לֶּה אַלּוּפֵ֞י אָֽהֳלִיבָמָ֛ה בַּת־עֲנָ֖ה אֵ֥שֶׁת עֵשָֽׂו׃ וְאִלֵּין, בְּנֵי אָהֳלִיבָמָה אִתַּת עֵשָׂו--רַבָּא יְעוּשׁ רַבָּא יַעְלָם, רַבָּא קֹרַח; אִלֵּין רַבְרְבֵי אָהֳלִיבָמָה, בַּת עֲנָה--אִתַּת עֵשָׂו. 
לו,יט אֵ֧לֶּה בְנֵֽי־עֵשָׂ֛ו וְאֵ֥לֶּה אַלּֽוּפֵיהֶ֖ם ה֥וּא אֱדֽוֹם׃ {ס} אִלֵּין בְּנֵי עֵשָׂו וְאִלֵּין רַבְרְבָנֵיהוֹן, הוּא אֱדוֹם. {ס}
לו,כ אֵ֤לֶּה בְנֵֽי־שֵׂעִיר֙ הַֽחֹרִ֔י יֹֽשְׁבֵ֖י הָאָ֑רֶץ לוֹטָ֥ן וְשׁוֹבָ֖ל וְצִבְע֥וֹן וַֽעֲנָֽה׃ אִלֵּין בְּנֵי שֵׂעִיר חוֹרָאֵי, יָתְבֵי אַרְעָא:  לוֹטָן וְשׁוֹבָל, וְצִבְעוֹן וַעֲנָה. 
לו,כא וְדִשׁ֥וֹן וְאֵ֖צֶר וְדִישָׁ֑ן אֵ֣לֶּה אַלּוּפֵ֧י הַֽחֹרִ֛י בְּנֵ֥י שֵׂעִ֖יר בְּאֶ֥רֶץ אֱדֽוֹם׃ וְדִישׁוֹן וְאֵצֶר, וְדִישָׁן; אִלֵּין רַבְרְבֵי חוֹרָאֵי בְּנֵי שֵׂעִיר, בְּאַרְעָא דֶּאֱדוֹם. 
לו,כב וַיִּֽהְי֥וּ בְנֵֽי־לוֹטָ֖ן חֹרִ֣י וְהֵימָ֑ם וַֽאֲח֥וֹת לוֹטָ֖ן תִּמְנָֽע׃ וַהֲווֹ בְּנֵי לוֹטָן, חוֹרִי וְהֵימָם; וַאֲחָתֵיהּ דְּלוֹטָן, תִּמְנָע. 
לו,כג וְאֵ֨לֶּה֙ בְּנֵ֣י שׁוֹבָ֔ל עַלְוָ֥ן וּמָנַ֖חַת וְעֵיבָ֑ל שְׁפ֖וֹ וְאוֹנָֽם׃ וְאִלֵּין בְּנֵי שׁוֹבָל, עַלְוָן וּמָנַחַת וְעֵיבָל, שְׁפוֹ, וְאוֹנָם. 
לו,כד וְאֵ֥לֶּה בְנֵֽי־צִבְע֖וֹן וְאַיָּ֣ה וַֽעֲנָ֑ה ה֣וּא עֲנָ֗ה אֲשֶׁ֨ר מָצָ֤א אֶת־הַיֵּמִם֙ בַּמִּדְבָּ֔ר בִּרְעֹת֥וֹ אֶת־הַֽחֲמֹרִ֖ים לְצִבְע֥וֹן אָבִֽיו׃ וְאִלֵּין בְּנֵי צִבְעוֹן, וְאַיָּה וַעֲנָה; הוּא עֲנָה, דְּאַשְׁכַּח יָת גִּבָּרַיָּא בְּמַדְבְּרָא, כַּד הֲוָה רָעֵי יָת חֲמָרַיָּא, לְצִבְעוֹן אֲבוּהִי. 
לו,כה וְאֵ֥לֶּה בְנֵֽי־עֲנָ֖ה דִּשֹׁ֑ן וְאָהֳלִֽיבָמָ֖ה בַּת־עֲנָֽה׃ וְאִלֵּין בְּנֵי עֲנָה, דִּישׁוֹן; וְאָהֳלִיבָמָה, בַּת עֲנָה. 
לו,כו וְאֵ֖לֶּה בְּנֵ֣י דִישָׁ֑ן חֶמְדָּ֥ן וְאֶשְׁבָּ֖ן וְיִתְרָ֥ן וּכְרָֽן׃ וְאִלֵּין, בְּנֵי דִּישָׁן--חֶמְדָּן וְאֶשְׁבָּן, וְיִתְרָן וּכְרָן. 
לו,כז אֵ֖לֶּה בְּנֵי־אֵ֑צֶר בִּלְהָ֥ן וְזַֽעֲוָ֖ן וַֽעֲקָֽן׃ אִלֵּין, בְּנֵי אֵצֶר--בִּלְהָן וְזַעֲוָן, וַעֲקָן. 
לו,כח אֵ֥לֶּה בְנֵֽי־דִישָׁ֖ן ע֥וּץ וַֽאֲרָֽן׃ אִלֵּין בְּנֵי דִּישָׁן, עוּץ וַאֲרָן. 
לו,כט אֵ֖לֶּה אַלּוּפֵ֣י הַֽחֹרִ֑י אַלּ֤וּף לוֹטָן֙ אַלּ֣וּף שׁוֹבָ֔ל אַלּ֥וּף צִבְע֖וֹן אַלּ֥וּף עֲנָֽה׃ אִלֵּין, רַבְרְבֵי חוֹרָאֵי:  רַבָּא לוֹטָן רַבָּא שׁוֹבָל, רַבָּא צִבְעוֹן רַבָּא עֲנָה. 
לו,ל אַלּ֥וּף דִּשֹׁ֛ן אַלּ֥וּף אֵ֖צֶר אַלּ֣וּף דִּישָׁ֑ן אֵ֣לֶּה אַלּוּפֵ֧י הַֽחֹרִ֛י לְאַלֻּֽפֵיהֶ֖ם בְּאֶ֥רֶץ שֵׂעִֽיר׃ {פ} רַבָּא דִּישׁוֹן רַבָּא אֵצֶר, רַבָּא דִּישָׁן; אִלֵּין רַבְרְבֵי חוֹרָאֵי לְרַבְרְבָנֵיהוֹן, בְּאַרְעָא דְּשֵׂעִיר. {פ}
לו,לא וְאֵ֨לֶּה֙ הַמְּלָכִ֔ים אֲשֶׁ֥ר מָֽלְכ֖וּ בְּאֶ֣רֶץ אֱד֑וֹם לִפְנֵ֥י מְלָךְ־מֶ֖לֶךְ לִבְנֵ֥י יִשְׂרָאֵֽל׃ וְאִלֵּין, מַלְכַיָּא, דִּמְלַכוּ, בְּאַרְעָא דֶּאֱדוֹם--קֳדָם דְּיִמְלוֹךְ מַלְכָּא, לִבְנֵי יִשְׂרָאֵל. 
לו,לב וַיִּמְלֹ֣ךְ בֶּֽאֱד֔וֹם בֶּ֖לַע בֶּן־בְּע֑וֹר וְשֵׁ֥ם עִיר֖וֹ דִּנְהָֽבָה׃ וּמְלַךְ בֶּאֱדוֹם, בֶּלַע בַּר בְּעוֹר; וְשׁוֹם קַרְתֵּיהּ, דִּנְהָבָה. 
לו,לג וַיָּ֖מָת בָּ֑לַע וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו יוֹבָ֥ב בֶּן־זֶ֖רַח מִבָּצְרָֽה׃ וּמִית, בָּלַע; וּמְלַךְ תְּחוֹתוֹהִי, יוֹבָב בַּר זֶרַח מִבָּצְרָה. 
לו,לד וַיָּ֖מָת יוֹבָ֑ב וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו חֻשָׁ֖ם מֵאֶ֥רֶץ הַתֵּֽימָנִֽי׃ וּמִית, יוֹבָב; וּמְלַךְ תְּחוֹתוֹהִי, חוּשָׁם מֵאֲרַע דָּרוֹמָא. 
לו,לה וַיָּ֖מָת חֻשָׁ֑ם וַיִּמְלֹ֨ךְ תַּחְתָּ֜יו הֲדַ֣ד בֶּן־בְּדַ֗ד הַמַּכֶּ֤ה אֶת־מִדְיָן֙ בִּשְׂדֵ֣ה מוֹאָ֔ב וְשֵׁ֥ם עִיר֖וֹ עֲוִֽית׃ וּמִית, חוּשָׁם; וּמְלַךְ תְּחוֹתוֹהִי הֲדַד בַּר בְּדַד, דְּקַטֵּיל יָת מִדְיַנָאֵי בְּחַקְלֵי מוֹאָב, וְשׁוֹם קַרְתֵּיהּ, עֲוִית. 
לו,לו וַיָּ֖מָת הֲדָ֑ד וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו שַׂמְלָ֖ה מִמַּשְׂרֵקָֽה׃ וּמִית, הֲדָד; וּמְלַךְ תְּחוֹתוֹהִי, שַׂמְלָה מִמַּשְׂרֵקָה. 
לו,לז וַיָּ֖מָת שַׂמְלָ֑ה וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו שָׁא֖וּל מֵֽרְחֹב֥וֹת הַנָּהָֽר׃ וּמִית, שַׂמְלָה; וּמְלַךְ תְּחוֹתוֹהִי, שָׁאוּל מֵרְחוֹבוֹת דְּעַל פְּרָת. 
לו,לח וַיָּ֖מָת שָׁא֑וּל וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו בַּ֥עַל חָנָ֖ן בֶּן־עַכְבּֽוֹר׃ וּמִית, שָׁאוּל; וּמְלַךְ תְּחוֹתוֹהִי, בַּעַל חָנָן בַּר עַכְבּוֹר. 
לו,לט וַיָּמָת֮ בַּ֣עַל חָנָ֣ן בֶּן־עַכְבּוֹר֒ וַיִּמְלֹ֤ךְ תַּחְתָּיו֙ הֲדַ֔ר וְשֵׁ֥ם עִיר֖וֹ פָּ֑עוּ וְשֵׁ֨ם אִשְׁתּ֤וֹ מְהֵֽיטַבְאֵל֙ בַּת־מַטְרֵ֔ד בַּ֖ת מֵ֥י זָהָֽב׃ וּמִית, בַּעַל חָנָן בַּר עַכְבּוֹר, וּמְלַךְ תְּחוֹתוֹהִי הֲדַר, וְשׁוֹם קַרְתֵּיהּ פָּעוּ; וְשׁוֹם אִתְּתֵיהּ מְהֵיטַבְאֵל בַּת מַטְרֵד, בַּת מְצָרֵיף דַּהְבָּא. 
לו,מ וְ֠אֵלֶּה שְׁמ֞וֹת אַלּוּפֵ֤י עֵשָׂו֙ לְמִשְׁפְּחֹתָ֔ם לִמְקֹֽמֹתָ֖ם בִּשְׁמֹתָ֑ם אַלּ֥וּף תִּמְנָ֛ע אַלּ֥וּף עַֽלְוָ֖ה אַלּ֥וּף יְתֵֽת׃ וְאִלֵּין שְׁמָהָת רַבְרְבֵי עֵשָׂו, לְזַרְעֲיָתְהוֹן, לְאַתְרֵיהוֹן, בִּשְׁמָהָתְהוֹן:  רַבָּא תִּמְנָע רַבָּא עַלְוָה, רַבָּא יְתֵת. 
לו,מא אַלּ֧וּף אָהֳלִֽיבָמָ֛ה אַלּ֥וּף אֵלָ֖ה אַלּ֥וּף פִּינֹֽן׃ רַבָּא אָהֳלִיבָמָה רַבָּא אֵלָה, רַבָּא פִּינוֹן. 
לו,מב אַלּ֥וּף קְנַ֛ז אַלּ֥וּף תֵּימָ֖ן אַלּ֥וּף מִבְצָֽר׃ רַבָּא קְנַז רַבָּא תֵּימָן, רַבָּא מִבְצָר. 
לו,מג אַלּ֥וּף מַגְדִּיאֵ֖ל אַלּ֣וּף עִירָ֑ם אֵ֣לֶּה ׀ אַלּוּפֵ֣י אֱד֗וֹם לְמֹֽשְׁבֹתָם֙ בְּאֶ֣רֶץ אֲחֻזָּתָ֔ם ה֥וּא עֵשָׂ֖ו אֲבִ֥י אֱדֽוֹם׃ {פ} רַבָּא מַגְדִּיאֵל, רַבָּא עִירָם; אִלֵּין רַבְרְבֵי אֱדוֹם, לְמוֹתְבָנֵיהוֹן בַּאֲרַע אַחְסָנַתְהוֹן--הוּא עֵשָׂו, אֲבוּהוֹן דֶּאֱדוֹמָאֵי. {פ}
