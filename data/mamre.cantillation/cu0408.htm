פינחס

כה,י וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
כה,יא פִּֽינְחָ֨ס בֶּן־אֶלְעָזָ֜ר בֶּן־אַֽהֲרֹ֣ן הַכֹּהֵ֗ן הֵשִׁ֤יב אֶת־חֲמָתִי֙ מֵעַ֣ל בְּנֵֽי־יִשְׂרָאֵ֔ל בְּקַנְא֥וֹ אֶת־קִנְאָתִ֖י בְּתוֹכָ֑ם וְלֹֽא־כִלִּ֥יתִי אֶת־בְּנֵֽי־יִשְׂרָאֵ֖ל בְּקִנְאָתִֽי׃ פִּינְחָס בַּר אֶלְעָזָר בַּר אַהֲרוֹן כָּהֲנָא, אֲתֵיב יָת חִמְתִּי מֵעַל בְּנֵי יִשְׂרָאֵל, בִּדְקַנִּי יָת קִנְאֲתִי, בֵּינֵיהוֹן; וְלָא שֵׁיצִיתִי יָת בְּנֵי יִשְׂרָאֵל, בְּקִנְאֲתִי. 
כה,יב לָכֵ֖ן אֱמֹ֑ר הִנְנִ֨י נֹתֵ֥ן ל֛וֹ אֶת־בְּרִיתִ֖י שָׁלֽוֹם׃ בְּכֵין, אֵימַר:  הָאֲנָא גָּזַר לֵיהּ יָת קְיָמִי, שְׁלָם. 
כה,יג וְהָ֤יְתָה לּוֹ֙ וּלְזַרְע֣וֹ אַֽחֲרָ֔יו בְּרִ֖ית כְּהֻנַּ֣ת עוֹלָ֑ם תַּ֗חַת אֲשֶׁ֤ר קִנֵּא֙ לֵֽאלֹהָ֔יו וַיְכַפֵּ֖ר עַל־בְּנֵ֥י יִשְׂרָאֵֽל׃ וּתְהֵי לֵיהּ וְלִבְנוֹהִי בָּתְרוֹהִי, קְיָם כְּהֻנַּת עָלַם--חֲלָף, דְּקַנִּי קֳדָם אֱלָהֵיהּ, וְכַפַּר, עַל בְּנֵי יִשְׂרָאֵל. 
כה,יד וְשֵׁם֩ אִ֨ישׁ יִשְׂרָאֵ֜ל הַמֻּכֶּ֗ה אֲשֶׁ֤ר הֻכָּה֙ אֶת־הַמִּדְיָנִ֔ית זִמְרִ֖י בֶּן־סָל֑וּא נְשִׂ֥יא בֵֽית־אָ֖ב לַשִּׁמְעֹנִֽי׃ וְשׁוֹם גֻּבְרָא בַּר יִשְׂרָאֵל קְטִילָא, דְּאִתְקְטִיל עִם מִדְיַנֵיתָא--זִמְרִי, בַּר סָלוּא:  רַב בֵּית בָּא, לְבֵית שִׁמְעוֹן. 
כה,טו וְשֵׁ֨ם הָֽאִשָּׁ֧ה הַמֻּכָּ֛ה הַמִּדְיָנִ֖ית כָּזְבִּ֣י בַת־צ֑וּר רֹ֣אשׁ אֻמּ֥וֹת בֵּֽית־אָ֛ב בְּמִדְיָ֖ן הֽוּא׃ {פ} וְשׁוֹם אִתְּתָא דְּאִתְקְטֵילַת מִדְיַנֵיתָא, כָּזְבִּי בַּת צוּר:  רֵישׁ אֻמֵּי בֵּית אַבָּא בְּמִדְיָן, הוּא. {פ}
כה,טז וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
כה,יז צָר֖וֹר אֶת־הַמִּדְיָנִ֑ים וְהִכִּיתֶ֖ם אוֹתָֽם׃ אַעֵיק, לְמִדְיַנָאֵי; וּתְקַטֵּיל, יָתְהוֹן. 
כה,יח כִּ֣י צֹֽרְרִ֥ים הֵם֙ לָכֶ֔ם בְּנִכְלֵיהֶ֛ם אֲשֶׁר־נִכְּל֥וּ לָכֶ֖ם עַל־דְּבַר־פְּע֑וֹר וְעַל־דְּבַ֞ר כָּזְבִּ֨י בַת־נְשִׂ֤יא מִדְיָן֙ אֲחֹתָ֔ם הַמֻּכָּ֥ה בְיוֹם־הַמַּגֵּפָ֖ה עַל־דְּבַר־פְּעֽוֹר׃ אֲרֵי מְעִיקִין אִנּוּן לְכוֹן, בְּנִכְלֵיהוֹן דְּנַכִּילוּ לְכוֹן עַל עֵיסַק פְּעוֹר; וְעַל עֵיסַק כָּזְבִּי בַּת רַבָּא דְּמִדְיָן, אֲחָתְהוֹן, דְּאִתְקְטֵילַת בְּיוֹמָא דְּמוֹתָנָא, עַל עֵיסַק פְּעוֹר. 
כו,א וַיְהִ֖י אַֽחֲרֵ֣י הַמַּגֵּפָ֑ה {פ}
וַיֹּ֤אמֶר יְהוָה֙ אֶל־מֹשֶׁ֔ה וְאֶ֧ל אֶלְעָזָ֛ר בֶּן־אַֽהֲרֹ֥ן הַכֹּהֵ֖ן לֵאמֹֽר׃ 		וַהֲוָה, בָּתַר מוֹתָנָא;  {פ}
וַאֲמַר יְיָ לְמֹשֶׁה, וּלְאֶלְעָזָר בַּר אַהֲרוֹן כָּהֲנָא לְמֵימַר. 
כו,ב שְׂא֞וּ אֶת־רֹ֣אשׁ ׀ כָּל־עֲדַ֣ת בְּנֵֽי־יִשְׂרָאֵ֗ל מִבֶּ֨ן עֶשְׂרִ֥ים שָׁנָ֛ה וָמַ֖עְלָה לְבֵ֣ית אֲבֹתָ֑ם כָּל־יֹצֵ֥א צָבָ֖א בְּיִשְׂרָאֵֽל׃ קַבִּילוּ יָת חֻשְׁבַּן כָּל כְּנִשְׁתָּא דִּבְנֵי יִשְׂרָאֵל, מִבַּר עַסְרִין שְׁנִין וּלְעֵילָא--לְבֵית אֲבָהָתְהוֹן:  כָּל נָפֵיק חֵילָא, בְּיִשְׂרָאֵל. 
כו,ג וַיְדַבֵּ֨ר מֹשֶׁ֜ה וְאֶלְעָזָ֧ר הַכֹּהֵ֛ן אֹתָ֖ם בְּעַֽרְבֹ֣ת מוֹאָ֑ב עַל־יַרְדֵּ֥ן יְרֵח֖וֹ לֵאמֹֽר׃ וּמַלֵּיל מֹשֶׁה וְאֶלְעָזָר כָּהֲנָא, אֲמַרוּ לְמִמְנֵי יָתְהוֹן--בְּמֵישְׁרַיָּא דְּמוֹאָב:  עַל יַרְדְּנָא דִּירִיחוֹ, לְמֵימַר. 
כו,ד מִבֶּ֛ן עֶשְׂרִ֥ים שָׁנָ֖ה וָמָ֑עְלָה כַּֽאֲשֶׁר֩ צִוָּ֨ה יְהוָ֤ה אֶת־מֹשֶׁה֙ וּבְנֵ֣י יִשְׂרָאֵ֔ל הַיֹּֽצְאִ֖ים מֵאֶ֥רֶץ מִצְרָֽיִם׃ מִבַּר עַסְרִין שְׁנִין, וּלְעֵילָא, כְּמָא דְּפַקֵּיד יְיָ יָת מֹשֶׁה וּבְנֵי יִשְׂרָאֵל, דִּנְפַקוּ מֵאַרְעָא דְּמִצְרָיִם. 
כו,ה רְאוּבֵ֖ן בְּכ֣וֹר יִשְׂרָאֵ֑ל בְּנֵ֣י רְאוּבֵ֗ן חֲנוֹךְ֙ מִשְׁפַּ֣חַת הַֽחֲנֹכִ֔י לְפַלּ֕וּא מִשְׁפַּ֖חַת הַפַּלֻּאִֽי׃ רְאוּבֵן, בֻּכְרָא דְּיִשְׂרָאֵל; בְּנֵי רְאוּבֵן, חֲנוֹךְ זַרְעִית חֲנוֹךְ, לְפַלּוּא, זַרְעִית פַּלּוּא. 
כו,ו לְחֶצְרֹ֕ן מִשְׁפַּ֖חַת הַֽחֶצְרוֹנִ֑י לְכַרְמִ֕י מִשְׁפַּ֖חַת הַכַּרְמִֽי׃ לְחֶצְרוֹן, זַרְעִית חֶצְרוֹן; לְכַרְמִי, זַרְעִית כַּרְמִי. 
כו,ז אֵ֖לֶּה מִשְׁפְּחֹ֣ת הָרֽאוּבֵנִ֑י וַיִּֽהְי֣וּ פְקֻֽדֵיהֶ֗ם שְׁלֹשָׁ֤ה וְאַרְבָּעִים֙ אֶ֔לֶף וּשְׁבַ֥ע מֵא֖וֹת וּשְׁלֹשִֽׁים׃ אִלֵּין, זַרְעֲיָת רְאוּבֵן; וַהֲווֹ מִנְיָנֵיהוֹן, אַרְבְּעִין וּתְלָתָא אַלְפִין, וּשְׁבַע מְאָה, וּתְלָתִין. 
כו,ח וּבְנֵ֥י פַלּ֖וּא אֱלִיאָֽב׃ וּבְנֵי פַּלּוּא, אֱלִיאָב. 
כו,ט וּבְנֵ֣י אֱלִיאָ֔ב נְמוּאֵ֖ל וְדָתָ֣ן וַֽאֲבִירָ֑ם הֽוּא־דָתָ֨ן וַֽאֲבִירָ֜ם קרואי (קְרִיאֵ֣י) הָֽעֵדָ֗ה אֲשֶׁ֨ר הִצּ֜וּ עַל־מֹשֶׁ֤ה וְעַֽל־אַהֲרֹן֙ בַּֽעֲדַת־קֹ֔רַח בְּהַצֹּתָ֖ם עַל־יְהוָֽה׃ וּבְנֵי אֱלִיאָב, נְמוּאֵל וְדָתָן וַאֲבִירָם:  הוּא דָּתָן וַאֲבִירָם מְעָרְעֵי כְּנִשְׁתָּא, דְּאִתְכְּנִישׁוּ עַל מֹשֶׁה וְעַל אַהֲרוֹן בִּכְנִשְׁתָּא דְּקֹרַח, בְּאִתְכַּנּוֹשֵׁיהוֹן, עַל יְיָ. 
כו,י וַתִּפְתַּ֨ח הָאָ֜רֶץ אֶת־פִּ֗יהָ וַתִּבְלַ֥ע אֹתָ֛ם וְאֶת־קֹ֖רַח בְּמ֣וֹת הָֽעֵדָ֑ה בַּֽאֲכֹ֣ל הָאֵ֗שׁ אֵ֣ת חֲמִשִּׁ֤ים וּמָאתַ֨יִם֙ אִ֔ישׁ וַיִּֽהְי֖וּ לְנֵֽס׃ וּפְתַחַת אַרְעָא יָת פֻּמַּהּ, וּבְלַעַת יָתְהוֹן וְיָת קֹרַח--בְּמוֹתָא דִּכְנִשְׁתָּא:  כַּד אֲכַלַת אִישָׁתָא, יָת מָאתַן וְחַמְשִׁין גֻּבְרָא, וַהֲווֹ, לְאָת. 
כו,יא וּבְנֵי־קֹ֖רַח לֹא־מֵֽתוּ׃ {ס} וּבְנֵי קֹרַח, לָא מִיתוּ. {ס}
כו,יב בְּנֵ֣י שִׁמְעוֹן֮ לְמִשְׁפְּחֹתָם֒ לִנְמוּאֵ֗ל מִשְׁפַּ֨חַת֙ הַנְּמ֣וּאֵלִ֔י לְיָמִ֕ין מִשְׁפַּ֖חַת הַיָּֽמִינִ֑י לְיָכִ֕ין מִשְׁפַּ֖חַת הַיָּֽכִינִֽי׃ בְּנֵי שִׁמְעוֹן, לְזַרְעֲיָתְהוֹן--לִנְמוּאֵל זַרְעִית נְמוּאֵל, לְיָמִין זַרְעִית יָמִין; לְיָכִין, זַרְעִית יָכִין. 
כו,יג לְזֶ֕רַח מִשְׁפַּ֖חַת הַזַּרְחִ֑י לְשָׁא֕וּל מִשְׁפַּ֖חַת הַשָּֽׁאוּלִֽי׃ לְזֶרַח, זַרְעִית זֶרַח; לְשָׁאוּל, זַרְעִית שָׁאוּל. 
כו,יד אֵ֖לֶּה מִשְׁפְּחֹ֣ת הַשִּׁמְעֹנִ֑י שְׁנַ֧יִם וְעֶשְׂרִ֛ים אֶ֖לֶף וּמָאתָֽיִם׃ {ס} אִלֵּין, זַרְעֲיָת שִׁמְעוֹן--עַסְרִין וּתְרֵין אַלְפִין, וּמָאתַן. {ס}
כו,טו בְּנֵ֣י גָד֮ לְמִשְׁפְּחֹתָם֒ לִצְפ֗וֹן מִשְׁפַּ֨חַת֙ הַצְּפוֹנִ֔י לְחַגִּ֕י מִשְׁפַּ֖חַת הַֽחַגִּ֑י לְשׁוּנִ֕י מִשְׁפַּ֖חַת הַשּׁוּנִֽי׃ בְּנֵי גָּד, לְזַרְעֲיָתְהוֹן--לִצְפוֹן זַרְעִית צְפוֹן, לְחַגִּי זַרְעִית חַגִּי; לְשׁוּנִי, זַרְעִית שׁוּנִי. 
כו,טז לְאָזְנִ֕י מִשְׁפַּ֖חַת הָֽאָזְנִ֑י לְעֵרִ֕י מִשְׁפַּ֖חַת הָֽעֵרִֽי׃ לְאָזְנִי, זַרְעִית אָזְנִי; לְעֵרִי, זַרְעִית עֵרִי. 
כו,יז לַֽאֲר֕וֹד מִשְׁפַּ֖חַת הָֽאֲרוֹדִ֑י לְאַ֨רְאֵלִ֔י מִשְׁפַּ֖חַת הָֽאַרְאֵלִֽי׃ לַאֲרוֹד, זַרְעִית אֲרוֹד; לְאַרְאֵלִי--זַרְעִית, אַרְאֵלִי. 
כו,יח אֵ֛לֶּה מִשְׁפְּחֹ֥ת בְּנֵֽי־גָ֖ד לִפְקֻֽדֵיהֶ֑ם אַרְבָּעִ֥ים אֶ֖לֶף וַֽחֲמֵ֥שׁ מֵאֽוֹת׃ {ס} אִלֵּין זַרְעֲיָת בְּנֵי גָּד, לְמִנְיָנֵיהוֹן--אַרְבְּעִין אַלְפִין, וַחֲמֵישׁ מְאָה. {ס}
כו,יט בְּנֵ֥י יְהוּדָ֖ה עֵ֣ר וְאוֹנָ֑ן וַיָּ֥מָת עֵ֛ר וְאוֹנָ֖ן בְּאֶ֥רֶץ כְּנָֽעַן׃ בְּנֵי יְהוּדָה, עֵר וְאוֹנָן; וּמִית עֵר וְאוֹנָן, בְּאַרְעָא דִּכְנָעַן. 
כו,כ וַיִּֽהְי֣וּ בְנֵֽי־יְהוּדָה֮ לְמִשְׁפְּחֹתָם֒ לְשֵׁלָ֗ה מִשְׁפַּ֨חַת֙ הַשֵּׁ֣לָנִ֔י לְפֶ֕רֶץ מִשְׁפַּ֖חַת הַפַּרְצִ֑י לְזֶ֕רַח מִשְׁפַּ֖חַת הַזַּרְחִֽי׃ וַהֲווֹ בְּנֵי יְהוּדָה, לְזַרְעֲיָתְהוֹן--לְשֵׁלָה זַרְעִית שֵׁלָה, לְפֶרֶץ זַרְעִית פֶּרֶץ; לְזֶרַח, זַרְעִית זֶרַח. 
כו,כא וַיִּֽהְי֣וּ בְנֵי־פֶ֔רֶץ לְחֶצְרֹ֕ן מִשְׁפַּ֖חַת הַֽחֶצְרֹנִ֑י לְחָמ֕וּל מִשְׁפַּ֖חַת הֶחָֽמוּלִֽי׃ וַהֲווֹ בְּנֵי פֶּרֶץ--לְחֶצְרוֹן, זַרְעִית חֶצְרוֹן; לְחָמוּל, זַרְעִית חָמוּל. 
כו,כב אֵ֛לֶּה מִשְׁפְּחֹ֥ת יְהוּדָ֖ה לִפְקֻֽדֵיהֶ֑ם שִׁשָּׁ֧ה וְשִׁבְעִ֛ים אֶ֖לֶף וַֽחֲמֵ֥שׁ מֵאֽוֹת׃ {ס} אִלֵּין זַרְעֲיָת יְהוּדָה, לְמִנְיָנֵיהוֹן--שִׁבְעִין וְשִׁתָּא אַלְפִין, וַחֲמֵישׁ מְאָה. {ס}
כו,כג בְּנֵ֤י יִשָּׂשכָר֙ לְמִשְׁפְּחֹתָ֔ם תּוֹלָ֕ע מִשְׁפַּ֖חַת הַתּֽוֹלָעִ֑י לְפֻוָ֕ה מִשְׁפַּ֖חַת הַפּוּנִֽי׃ בְּנֵי יִשָּׂשכָר, לְזַרְעֲיָתְהוֹן--תּוֹלָע, זַרְעִית תּוֹלָע; לְפוּוָה, זַרְעִית פּוּוָה. 
כו,כד לְיָשׁ֕וּב מִשְׁפַּ֖חַת הַיָּֽשֻׁבִ֑י לְשִׁמְרֹ֕ן מִשְׁפַּ֖חַת הַשִּׁמְרֹנִֽי׃ לְיָשׁוּב, זַרְעִית יָשׁוּב; לְשִׁמְרוֹן, זַרְעִית שִׁמְרוֹן. 
כו,כה אֵ֛לֶּה מִשְׁפְּחֹ֥ת יִשָּׂשכָ֖ר לִפְקֻֽדֵיהֶ֑ם אַרְבָּעָ֧ה וְשִׁשִּׁ֛ים אֶ֖לֶף וּשְׁלֹ֥שׁ מֵאֽוֹת׃ {ס} אִלֵּין זַרְעֲיָת יִשָּׂשכָר, לְמִנְיָנֵיהוֹן--שִׁתִּין וְאַרְבְּעָא אַלְפִין, וּתְלָת מְאָה. {ס}
כו,כו בְּנֵ֣י זְבוּלֻן֮ לְמִשְׁפְּחֹתָם֒ לְסֶ֗רֶד מִשְׁפַּ֨חַת֙ הַסַּרְדִּ֔י לְאֵל֕וֹן מִשְׁפַּ֖חַת הָאֵֽלֹנִ֑י לְיַ֨חְלְאֵ֔ל מִשְׁפַּ֖חַת הַיַּחְלְאֵלִֽי׃ בְּנֵי זְבוּלוּן, לְזַרְעֲיָתְהוֹן--לְסֶרֶד זַרְעִית סֶרֶד, לְאֵילוֹן זַרְעִית אֵילוֹן; לְיַחְלְאֵל--זַרְעִית, יַחְלְאֵל. 
כו,כז אֵ֛לֶּה מִשְׁפְּחֹ֥ת הַזְּבֽוּלֹנִ֖י לִפְקֻֽדֵיהֶ֑ם שִׁשִּׁ֥ים אֶ֖לֶף וַֽחֲמֵ֥שׁ מֵאֽוֹת׃ {ס} אִלֵּין זַרְעֲיָת זְבוּלוּן, לְמִנְיָנֵיהוֹן--שִׁתִּין אַלְפִין, וַחֲמֵישׁ מְאָה. {ס}
כו,כח בְּנֵ֥י יוֹסֵ֖ף לְמִשְׁפְּחֹתָ֑ם מְנַשֶּׁ֖ה וְאֶפְרָֽיִם׃ בְּנֵי יוֹסֵף, לְזַרְעֲיָתְהוֹן--מְנַשֶּׁה, וְאֶפְרָיִם. 
כו,כט בְּנֵ֣י מְנַשֶּׁ֗ה לְמָכִיר֙ מִשְׁפַּ֣חַת הַמָּֽכִירִ֔י וּמָכִ֖יר הוֹלִ֣יד אֶת־גִּלְעָ֑ד לְגִלְעָ֕ד מִשְׁפַּ֖חַת הַגִּלְעָדִֽי׃ בְּנֵי מְנַשֶּׁה, לְמָכִיר זַרְעִית מָכִיר, וּמָכִיר, אוֹלֵיד יָת גִּלְעָד; לְגִלְעָד, זַרְעִית גִּלְעָד. 
כו,ל אֵ֚לֶּה בְּנֵ֣י גִלְעָ֔ד אִיעֶ֕זֶר מִשְׁפַּ֖חַת הָאִֽיעֶזְרִ֑י לְחֵ֕לֶק מִשְׁפַּ֖חַת הַֽחֶלְקִֽי׃ אִלֵּין, בְּנֵי גִּלְעָד--אִיעֶזֶר, זַרְעִית אִיעֶזֶר; לְחֵלֶק, זַרְעִית חֵלֶק. 
כו,לא וְאַ֨שְׂרִיאֵ֔ל מִשְׁפַּ֖חַת הָֽאַשְׂרִֽאֵלִ֑י וְשֶׁ֕כֶם מִשְׁפַּ֖חַת הַשִּׁכְמִֽי׃ וְאַשְׂרִיאֵל--זַרְעִית, אַשְׂרִיאֵל; וְשֶׁכֶם, זַרְעִית שֶׁכֶם. 
כו,לב וּשְׁמִידָ֕ע מִשְׁפַּ֖חַת הַשְּׁמִֽידָעִ֑י וְחֵ֕פֶר מִשְׁפַּ֖חַת הַֽחֶפְרִֽי׃ וּשְׁמִידָע, זַרְעִית שְׁמִידָע; וְחֵפֶר, זַרְעִית חֵפֶר. 
כו,לג וּצְלָפְחָ֣ד בֶּן־חֵ֗פֶר לֹא־הָ֥יוּ ל֛וֹ בָּנִ֖ים כִּ֣י אִם־בָּנ֑וֹת וְשֵׁם֙ בְּנ֣וֹת צְלָפְחָ֔ד מַחְלָ֣ה וְנֹעָ֔ה חָגְלָ֥ה מִלְכָּ֖ה וְתִרְצָֽה׃ וּצְלָפְחָד בַּר חֵפֶר, לָא הֲווֹ לֵיהּ בְּנִין--אֱלָהֵין בְּנָן:  וְשׁוֹם, בְּנָת צְלָפְחָד--מַחְלָה וְנוֹעָה, חָגְלָה מִלְכָּה וְתִרְצָה. 
כו,לד אֵ֖לֶּה מִשְׁפְּחֹ֣ת מְנַשֶּׁ֑ה וּפְקֻ֣דֵיהֶ֔ם שְׁנַ֧יִם וַֽחֲמִשִּׁ֛ים אֶ֖לֶף וּשְׁבַ֥ע מֵאֽוֹת׃ {ס} אִלֵּין, זַרְעֲיָת מְנַשֶּׁה; וּמִנְיָנֵיהוֹן, חַמְשִׁין וּתְרֵין אַלְפִין וּשְׁבַע מְאָה. {ס}
כו,לה אֵ֣לֶּה בְנֵֽי־אֶפְרַיִם֮ לְמִשְׁפְּחֹתָם֒ לְשׁוּתֶ֗לַח מִשְׁפַּ֨חַת֙ הַשֻּׁ֣תַלְחִ֔י לְבֶ֕כֶר מִשְׁפַּ֖חַת הַבַּכְרִ֑י לְתַ֕חַן מִשְׁפַּ֖חַת הַֽתַּחֲנִֽי׃ אִלֵּין בְּנֵי אֶפְרַיִם, לְזַרְעֲיָתְהוֹן--לְשׁוּתֶלַח זַרְעִית שׁוּתֶלַח, לְבֶכֶר זַרְעִית בֶּכֶר; לְתַחַן, זַרְעִית תַּחַן. 
כו,לו וְאֵ֖לֶּה בְּנֵ֣י שׁוּתָ֑לַח לְעֵרָ֕ן מִשְׁפַּ֖חַת הָעֵֽרָנִֽי׃ וְאִלֵּין, בְּנֵי שׁוּתָלַח--לְעֵרָן, זַרְעִית עֵרָן. 
כו,לז אֵ֣לֶּה מִשְׁפְּחֹ֤ת בְּנֵֽי־אֶפְרַ֨יִם֙ לִפְקֻ֣דֵיהֶ֔ם שְׁנַ֧יִם וּשְׁלֹשִׁ֛ים אֶ֖לֶף וַֽחֲמֵ֣שׁ מֵא֑וֹת אֵ֥לֶּה בְנֵֽי־יוֹסֵ֖ף לְמִשְׁפְּחֹתָֽם׃ {ס} אִלֵּין זַרְעֲיָת בְּנֵי אֶפְרַיִם לְמִנְיָנֵיהוֹן, תְּלָתִין וּתְרֵין אַלְפִין וַחֲמֵישׁ מְאָה; אִלֵּין בְּנֵי יוֹסֵף, לְזַרְעֲיָתְהוֹן. {ס}
כו,לח בְּנֵ֣י בִנְיָמִן֮ לְמִשְׁפְּחֹתָם֒ לְבֶ֗לַע מִשְׁפַּ֨חַת֙ הַבַּלְעִ֔י לְאַשְׁבֵּ֕ל מִשְׁפַּ֖חַת הָֽאַשְׁבֵּלִ֑י לַֽאֲחִירָ֕ם מִשְׁפַּ֖חַת הָאֲחִֽירָמִֽי׃ בְּנֵי בִּנְיָמִין, לְזַרְעֲיָתְהוֹן--לְבֶלַע זַרְעִית בֶּלַע, לְאַשְׁבֵּל זַרְעִית אַשְׁבֵּל; לַאֲחִירָם, זַרְעִית אֲחִירָם. 
כו,לט לִשְׁפוּפָ֕ם מִשְׁפַּ֖חַת הַשּֽׁוּפָמִ֑י לְחוּפָ֕ם מִשְׁפַּ֖חַת הַחֽוּפָמִֽי׃ לִשְׁפוּפָם, זַרְעִית שְׁפוּפָם; לְחוּפָם, זַרְעִית חוּפָם. 
כו,מ וַיִּֽהְי֥וּ בְנֵי־בֶ֖לַע אַ֣רְדְּ וְנַֽעֲמָ֑ן מִשְׁפַּ֨חַת֙ הָֽאַרְדִּ֔י לְנַֽעֲמָ֔ן מִשְׁפַּ֖חַת הַֽנַּעֲמִֽי׃ וַהֲווֹ בְּנֵי בֶּלַע, אַרְדְּ וְנַעֲמָן--זַרְעִית, אַרְדְּ, לְנַעֲמָן, זַרְעִית נַעֲמָן. 
כו,מא אֵ֥לֶּה בְנֵֽי־בִנְיָמִ֖ן לְמִשְׁפְּחֹתָ֑ם וּפְקֻ֣דֵיהֶ֔ם חֲמִשָּׁ֧ה וְאַרְבָּעִ֛ים אֶ֖לֶף וְשֵׁ֥שׁ מֵאֽוֹת׃ {ס} אִלֵּין בְּנֵי בִּנְיָמִין, לְזַרְעֲיָתְהוֹן; וּמִנְיָנֵיהוֹן, אַרְבְּעִין וְחַמְשָׁא אַלְפִין וְשֵׁית מְאָה. {ס}
כו,מב אֵ֤לֶּה בְנֵי־דָן֙ לְמִשְׁפְּחֹתָ֔ם לְשׁוּחָ֕ם מִשְׁפַּ֖חַת הַשּֽׁוּחָמִ֑י אֵ֛לֶּה מִשְׁפְּחֹ֥ת דָּ֖ן לְמִשְׁפְּחֹתָֽם׃ אִלֵּין בְּנֵי דָּן, לְזַרְעֲיָתְהוֹן--לְשׁוּחָם, זַרְעִית שׁוּחָם; אִלֵּין זַרְעֲיָת דָּן, לְזַרְעֲיָתְהוֹן. 
כו,מג כָּל־מִשְׁפְּחֹ֥ת הַשּֽׁוּחָמִ֖י לִפְקֻֽדֵיהֶ֑ם אַרְבָּעָ֧ה וְשִׁשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת׃ {ס} כָּל זַרְעֲיָת שׁוּחָם, לְמִנְיָנֵיהוֹן--שִׁתִּין וְאַרְבְּעָא אַלְפִין, וְאַרְבַּע מְאָה. {ס}
כו,מד בְּנֵ֣י אָשֵׁר֮ לְמִשְׁפְּחֹתָם֒ לְיִמְנָ֗ה מִשְׁפַּ֨חַת֙ הַיִּמְנָ֔ה לְיִשְׁוִ֕י מִשְׁפַּ֖חַת הַיִּשְׁוִ֑י לִבְרִיעָ֕ה מִשְׁפַּ֖חַת הַבְּרִיעִֽי׃ בְּנֵי אָשֵׁר, לְזַרְעֲיָתְהוֹן--לְיִמְנָה זַרְעִית יִמְנָה, לְיִשְׁוִי זַרְעִית יִשְׁוִי; לִבְרִיעָה, זַרְעִית בְּרִיעָה. 
כו,מה לִבְנֵ֣י בְרִיעָ֔ה לְחֶ֕בֶר מִשְׁפַּ֖חַת הַֽחֶבְרִ֑י לְמַ֨לְכִּיאֵ֔ל מִשְׁפַּ֖חַת הַמַּלְכִּֽיאֵלִֽי׃ לִבְנֵי בְּרִיעָה--לְחֶבֶר, זַרְעִית חֶבֶר; לְמַלְכִּיאֵל--זַרְעִית, מַלְכִּיאֵל. 
כו,מו וְשֵׁ֥ם בַּת־אָשֵׁ֖ר שָֽׂרַח׃ וְשׁוֹם בַּת אָשֵׁר, שָׂרַח. 
כו,מז אֵ֛לֶּה מִשְׁפְּחֹ֥ת בְּנֵֽי־אָשֵׁ֖ר לִפְקֻֽדֵיהֶ֑ם שְׁלֹשָׁ֧ה וַֽחֲמִשִּׁ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת׃ {ס} אִלֵּין זַרְעֲיָת בְּנֵי אָשֵׁר, לְמִנְיָנֵיהוֹן--חַמְשִׁין וּתְלָתָא אַלְפִין, וְאַרְבַּע מְאָה. {ס}
כו,מח בְּנֵ֤י נַפְתָּלִי֙ לְמִשְׁפְּחֹתָ֔ם לְיַ֨חְצְאֵ֔ל מִשְׁפַּ֖חַת הַיַּחְצְאֵלִ֑י לְגוּנִ֕י מִשְׁפַּ֖חַת הַגּוּנִֽי׃ בְּנֵי נַפְתָּלִי, לְזַרְעֲיָתְהוֹן--לְיַחְצְאֵל, זַרְעִית יַחְצְאֵל; לְגוּנִי, זַרְעִית גּוּנִי. 
כו,מט לְיֵ֕צֶר מִשְׁפַּ֖חַת הַיִּצְרִ֑י לְשִׁלֵּ֕ם מִשְׁפַּ֖חַת הַשִּׁלֵּמִֽי׃ לְיֵצֶר, זַרְעִית יֵצֶר; לְשִׁלֵּם, זַרְעִית שִׁלֵּם. 
כו,נ אֵ֛לֶּה מִשְׁפְּחֹ֥ת נַפְתָּלִ֖י לְמִשְׁפְּחֹתָ֑ם וּפְקֻ֣דֵיהֶ֔ם חֲמִשָּׁ֧ה וְאַרְבָּעִ֛ים אֶ֖לֶף וְאַרְבַּ֥ע מֵאֽוֹת׃ אִלֵּין זַרְעֲיָת נַפְתָּלִי, לְזַרְעֲיָתְהוֹן; וּמִנְיָנֵיהוֹן, אַרְבְּעִין וְחַמְשָׁא אַלְפִין וְאַרְבַּע מְאָה. 
כו,נא אֵ֗לֶּה פְּקוּדֵי֙ בְּנֵ֣י יִשְׂרָאֵ֔ל שֵׁשׁ־מֵא֥וֹת אֶ֖לֶף וָאָ֑לֶף שְׁבַ֥ע מֵא֖וֹת וּשְׁלֹשִֽׁים׃ {פ} אִלֵּין, מִנְיָנֵי בְּנֵי יִשְׂרָאֵל--שֵׁית מְאָה וְחַד, אַלְפִין; שְׁבַע מְאָה, וּתְלָתִין. {פ}
כו,נב וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
כו,נג לָאֵ֗לֶּה תֵּֽחָלֵ֥ק הָאָ֛רֶץ בְּנַֽחֲלָ֖ה בְּמִסְפַּ֥ר שֵׁמֽוֹת׃ לְאִלֵּין, תִּתְפְּלֵיג אַרְעָא בְּאַחְסָנָא--בְּמִנְיַן שְׁמָהָן. 
כו,נד לָרַ֗ב תַּרְבֶּה֙ נַֽחֲלָת֔וֹ וְלַמְעַ֕ט תַּמְעִ֖יט נַֽחֲלָת֑וֹ אִ֚ישׁ לְפִ֣י פְקֻדָ֔יו יֻתַּ֖ן נַֽחֲלָתֽוֹ׃ לְסַגִּיאֵי, תַּסְגּוֹן אַחְסָנַתְהוֹן, וְלִזְעֵירֵי, תַּזְעֲרוּן אַחְסָנַתְהוֹן:  גְּבַר לְפֹם מִנְיָנוֹהִי, תִּתְיְהֵיב אַחְסָנְתֵיהּ. 
כו,נה אַךְ־בְּגוֹרָ֕ל יֵֽחָלֵ֖ק אֶת־הָאָ֑רֶץ לִשְׁמ֥וֹת מַטּוֹת־אֲבֹתָ֖ם יִנְחָֽלוּ׃ בְּרַם בְּעַדְבָּא, תִּתְפְּלֵיג אַרְעָא:  לִשְׁמָהָת שִׁבְטֵי אֲבָהָתְהוֹן, יַחְסְנוּן. 
כו,נו עַל־פִּי֙ הַגּוֹרָ֔ל תֵּֽחָלֵ֖ק נַֽחֲלָת֑וֹ בֵּ֥ין רַ֖ב לִמְעָֽט׃ {ס} עַל פֹּם, עַדְבָּא, תִּתְפְּלֵיג, אַחְסָנַתְהוֹן--בֵּין סַגִּיאֵי, לִזְעֵירֵי. {ס}
כו,נז וְאֵ֨לֶּה פְקוּדֵ֣י הַלֵּוִי֮ לְמִשְׁפְּחֹתָם֒ לְגֵֽרְשׁ֗וֹן מִשְׁפַּ֨חַת֙ הַגֵּ֣רְשֻׁנִּ֔י לִקְהָ֕ת מִשְׁפַּ֖חַת הַקְּהָתִ֑י לִמְרָרִ֕י מִשְׁפַּ֖חַת הַמְּרָרִֽי׃ וְאִלֵּין מִנְיָנֵי לֵיוָאֵי, לְזַרְעֲיָתְהוֹן--לְגֵרְשׁוֹן זַרְעִית גֵּרְשׁוֹן, לִקְהָת זַרְעִית קְהָת; לִמְרָרִי, זַרְעִית מְרָרִי. 
כו,נח אֵ֣לֶּה ׀ מִשְׁפְּחֹ֣ת לֵוִ֗י מִשְׁפַּ֨חַת הַלִּבְנִ֜י מִשְׁפַּ֤חַת הַֽחֶבְרֹנִי֙ מִשְׁפַּ֤חַת הַמַּחְלִי֙ מִשְׁפַּ֣חַת הַמּוּשִׁ֔י מִשְׁפַּ֖חַת הַקָּרְחִ֑י וּקְהָ֖ת הוֹלִ֥ד אֶת־עַמְרָֽם׃ אִלֵּין זַרְעֲיָת לֵוִי, זַרְעִית לִבְנִי זַרְעִית חֶבְרוֹן זַרְעִית מַחְלִי זַרְעִית מוּשִׁי, זַרְעִית, קֹרַח; וּקְהָת, אוֹלֵיד יָת עַמְרָם. 
כו,נט וְשֵׁ֣ם ׀ אֵ֣שֶׁת עַמְרָ֗ם יוֹכֶ֨בֶד֙ בַּת־לֵוִ֔י אֲשֶׁ֨ר יָֽלְדָ֥ה אֹתָ֛הּ לְלֵוִ֖י בְּמִצְרָ֑יִם וַתֵּ֣לֶד לְעַמְרָ֗ם אֶֽת־אַהֲרֹן֙ וְאֶת־מֹשֶׁ֔ה וְאֵ֖ת מִרְיָ֥ם אֲחֹתָֽם׃ וְשׁוֹם אִתַּת עַמְרָם, יוֹכֶבֶד בַּת לֵוִי, דִּילֵידַת יָתַהּ לְלֵוִי, בְּמִצְרָיִם; וִילֵידַת לְעַמְרָם, יָת אַהֲרוֹן וְיָת מֹשֶׁה, וְיָת, מִרְיָם אֲחָתְהוֹן. 
כו,ס וַיִּוָּלֵ֣ד לְאַֽהֲרֹ֔ן אֶת־נָדָ֖ב וְאֶת־אֲבִיה֑וּא אֶת־אֶלְעָזָ֖ר וְאֶת־אִֽיתָמָֽר׃ וְאִתְיְלֵיד לְאַהֲרוֹן, יָת נָדָב וְיָת אֲבִיהוּא, יָת אֶלְעָזָר, וְיָת אִיתָמָר. 
כו,סא וַיָּ֥מָת נָדָ֖ב וַֽאֲבִיה֑וּא בְּהַקְרִיבָ֥ם אֵשׁ־זָרָ֖ה לִפְנֵ֥י יְהוָֽה׃ וּמִית נָדָב, וַאֲבִיהוּא, בְּקָרוֹבֵיהוֹן אִישָׁתָא נֻכְרֵיתָא, קֳדָם יְיָ. 
כו,סב וַיִּֽהְי֣וּ פְקֻֽדֵיהֶ֗ם שְׁלֹשָׁ֤ה וְעֶשְׂרִים֙ אֶ֔לֶף כָּל־זָכָ֖ר מִבֶּן־חֹ֣דֶשׁ וָמָ֑עְלָה כִּ֣י ׀ לֹ֣א הָתְפָּֽקְד֗וּ בְּתוֹךְ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל כִּ֠י לֹֽא־נִתַּ֤ן לָהֶם֙ נַֽחֲלָ֔ה בְּת֖וֹךְ בְּנֵ֥י יִשְׂרָאֵֽל׃ וַהֲווֹ מִנְיָנֵיהוֹן, עַסְרִין וּתְלָתָא אַלְפִין--כָּל דְּכוּרָא, מִבַּר יַרְחָא וּלְעֵילָא:  אֲרֵי לָא אִתְמְנִיאוּ, בְּגוֹ בְּנֵי יִשְׂרָאֵל, אֲרֵי לָא אִתְיְהֵיבַת לְהוֹן אַחְסָנָא, בְּגוֹ בְּנֵי יִשְׂרָאֵל. 
כו,סג אֵ֚לֶּה פְּקוּדֵ֣י מֹשֶׁ֔ה וְאֶלְעָזָ֖ר הַכֹּהֵ֑ן אֲשֶׁ֨ר פָּֽקְד֜וּ אֶת־בְּנֵ֤י יִשְׂרָאֵל֙ בְּעַֽרְבֹ֣ת מוֹאָ֔ב עַ֖ל יַרְדֵּ֥ן יְרֵחֽוֹ׃ אִלֵּין מִנְיָנֵי מֹשֶׁה, וְאֶלְעָזָר כָּהֲנָא--דִּמְנוֹ יָת בְּנֵי יִשְׂרָאֵל, בְּמֵישְׁרַיָּא דְּמוֹאָב, עַל, יַרְדְּנָא דִּירִיחוֹ. 
כו,סד וּבְאֵ֨לֶּה֙ לֹא־הָ֣יָה אִ֔ישׁ מִפְּקוּדֵ֣י מֹשֶׁ֔ה וְאַֽהֲרֹ֖ן הַכֹּהֵ֑ן אֲשֶׁ֥ר פָּֽקְד֛וּ אֶת־בְּנֵ֥י יִשְׂרָאֵ֖ל בְּמִדְבַּ֥ר סִינָֽי׃ וּבְאִלֵּין, לָא הֲוָה גְּבַר, מִמִּנְיָנֵי מֹשֶׁה, וְאַהֲרוֹן כָּהֲנָא--דִּמְנוֹ יָת בְּנֵי יִשְׂרָאֵל, בְּמַדְבְּרָא דְּסִינָי. 
כו,סה כִּֽי־אָמַ֤ר יְהוָה֙ לָהֶ֔ם מ֥וֹת יָמֻ֖תוּ בַּמִּדְבָּ֑ר וְלֹֽא־נוֹתַ֤ר מֵהֶם֙ אִ֔ישׁ כִּ֚י אִם־כָּלֵ֣ב בֶּן־יְפֻנֶּ֔ה וִֽיהוֹשֻׁ֖עַ בִּן־נֽוּן׃ {ס} אֲרֵי אֲמַר יְיָ לְהוֹן, מְמָת יְמוּתוּן בְּמַדְבְּרָא; וְלָא אִשְׁתְּאַר מִנְּהוֹן, אֲנָשׁ--אֱלָהֵין כָּלֵב בַּר יְפֻנֶּה, וִיהוֹשׁוּעַ בַּר נוּן. {ס}
כז,א וַתִּקְרַ֜בְנָה בְּנ֣וֹת צְלָפְחָ֗ד בֶּן־חֵ֤פֶר בֶּן־גִּלְעָד֙ בֶּן־מָכִ֣יר בֶּן־מְנַשֶּׁ֔ה לְמִשְׁפְּחֹ֖ת מְנַשֶּׁ֣ה בֶן־יוֹסֵ֑ף וְאֵ֨לֶּה֙ שְׁמ֣וֹת בְּנֹתָ֔יו מַחְלָ֣ה נֹעָ֔ה וְחָגְלָ֥ה וּמִלְכָּ֖ה וְתִרְצָֽה׃ וּקְרִיבָא בְּנָת צְלָפְחָד, בַּר חֵפֶר בַּר גִּלְעָד בַּר מָכִיר בַּר מְנַשֶּׁה, לְזַרְעֲיָת, מְנַשֶּׁה בַּר יוֹסֵף; וְאִלֵּין, שְׁמָהָת בְּנָתֵיהּ--מַחְלָה נוֹעָה, וְחָגְלָה וּמִלְכָּה וְתִרְצָה. 
כז,ב וַֽתַּעֲמֹ֜דְנָה לִפְנֵ֣י מֹשֶׁ֗ה וְלִפְנֵי֙ אֶלְעָזָ֣ר הַכֹּהֵ֔ן וְלִפְנֵ֥י הַנְּשִׂיאִ֖ם וְכָל־הָֽעֵדָ֑ה פֶּ֥תַח אֹֽהֶל־מוֹעֵ֖ד לֵאמֹֽר׃ וְקָמָא קֳדָם מֹשֶׁה, וּקְדָם אֶלְעָזָר כָּהֲנָא, וּקְדָם רַבְרְבַיָּא, וְכָל כְּנִשְׁתָּא--בִּתְרַע מַשְׁכַּן זִמְנָא, לְמֵימַר. 
כז,ג אָבִינוּ֮ מֵ֣ת בַּמִּדְבָּר֒ וְה֨וּא לֹֽא־הָיָ֜ה בְּת֣וֹךְ הָֽעֵדָ֗ה הַנּֽוֹעָדִ֛ים עַל־יְהוָ֖ה בַּֽעֲדַת־קֹ֑רַח כִּֽי־בְחֶטְא֣וֹ מֵ֔ת וּבָנִ֖ים לֹא־הָ֥יוּ לֽוֹ׃ אֲבוּנָא, מִית בְּמַדְבְּרָא, וְהוּא לָא הֲוָה בְּגוֹ כְּנִשְׁתָּא דְּאִזְדָּמַנוּ עַל יְיָ, בִּכְנִשְׁתָּא דְּקֹרַח:  אֲרֵי בְּחוֹבֵיהּ מִית, וּבְנִין לָא הֲווֹ לֵיהּ. 
כז,ד לָ֣מָּה יִגָּרַ֤ע שֵׁם־אָבִ֨ינוּ֙ מִתּ֣וֹךְ מִשְׁפַּחְתּ֔וֹ כִּ֛י אֵ֥ין ל֖וֹ בֵּ֑ן תְּנָה־לָּ֣נוּ אֲחֻזָּ֔ה בְּת֖וֹךְ אֲחֵ֥י אָבִֽינוּ׃ לְמָא יִתְמְנַע שְׁמָא דַּאֲבוּנָא מִגּוֹ זַרְעִיתֵיהּ, אֲרֵי לֵית לֵיהּ בַּר; הַב לַנָא אַחְסָנָא, בְּגוֹ אֲחֵי אֲבוּנָא. 
כז,ה וַיַּקְרֵ֥ב מֹשֶׁ֛ה אֶת־מִשְׁפָּטָ֖ן לִפְנֵ֥י יְהוָֽה׃ {פ} וְקָרֵיב מֹשֶׁה יָת דִּינְהוֹן, לִקְדָם יְיָ. {פ}
כז,ו וַיֹּ֥אמֶר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וַאֲמַר יְיָ, לְמֹשֶׁה לְמֵימַר. 
כז,ז כֵּ֗ן בְּנ֣וֹת צְלָפְחָד֮ דֹּֽבְרֹת֒ נָתֹ֨ן תִּתֵּ֤ן לָהֶם֙ אֲחֻזַּ֣ת נַֽחֲלָ֔ה בְּת֖וֹךְ אֲחֵ֣י אֲבִיהֶ֑ם וְהַֽעֲבַרְתָּ֛ אֶת־נַֽחֲלַ֥ת אֲבִיהֶ֖ן לָהֶֽן׃ יָאוּת, בְּנָת צְלָפְחָד מְמַלְּלָן--מִתָּן תִּתֵּין לְהוֹן אֲחֻדַּת אַחְסָנָא, בְּגוֹ אֲחֵי אֲבוּהוֹן; וְתַעְבַּר יָת אַחְסָנַת אֲבוּהוֹן, לְהוֹן. 
כז,ח וְאֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל תְּדַבֵּ֣ר לֵאמֹ֑ר אִ֣ישׁ כִּֽי־יָמ֗וּת וּבֵן֙ אֵ֣ין ל֔וֹ וְהַֽעֲבַרְתֶּ֥ם אֶת־נַֽחֲלָת֖וֹ לְבִתּֽוֹ׃ וְעִם בְּנֵי יִשְׂרָאֵל, תְּמַלֵּיל לְמֵימַר:  גְּבַר אֲרֵי יְמוּת, וּבַר לֵית לֵיהּ--וְתַעְבְּרוּן יָת אַחְסָנְתֵיהּ, לִבְרַתֵּיהּ. 
כז,ט וְאִם־אֵ֥ין ל֖וֹ בַּ֑ת וּנְתַתֶּ֥ם אֶת־נַֽחֲלָת֖וֹ לְאֶחָֽיו׃ וְאִם לֵית לֵיהּ, בְּרַתָּא--וְתִתְּנוּן יָת אַחְסָנְתֵיהּ, לַאֲחוֹהִי. 
כז,י וְאִם־אֵ֥ין ל֖וֹ אַחִ֑ים וּנְתַתֶּ֥ם אֶת־נַֽחֲלָת֖וֹ לַֽאֲחֵ֥י אָבִֽיו׃ וְאִם לֵית לֵיהּ, אַחִין--וְתִתְּנוּן יָת אַחְסָנְתֵיהּ, לַאֲחֵי אֲבוּהִי. 
כז,יא וְאִם־אֵ֣ין אַחִים֮ לְאָבִיו֒ וּנְתַתֶּ֣ם אֶת־נַֽחֲלָת֗וֹ לִשְׁאֵר֞וֹ הַקָּרֹ֥ב אֵלָ֛יו מִמִּשְׁפַּחְתּ֖וֹ וְיָרַ֣שׁ אֹתָ֑הּ וְֽהָ֨יְתָ֜ה לִבְנֵ֤י יִשְׂרָאֵל֙ לְחֻקַּ֣ת מִשְׁפָּ֔ט כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ {פ} וְאִם לֵית אַחִין, לַאֲבוּהִי--וְתִתְּנוּן יָת אַחְסָנְתֵיהּ לְקָרִיבֵיהּ דְּקָרִיב לֵיהּ מִזַּרְעִיתֵיהּ, וְיֵירַת יָתַהּ; וּתְהֵי לִבְנֵי יִשְׂרָאֵל, לִגְזֵירַת דִּין, כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. {פ}
כז,יב וַיֹּ֤אמֶר יְהוָה֙ אֶל־מֹשֶׁ֔ה עֲלֵ֛ה אֶל־הַ֥ר הָֽעֲבָרִ֖ים הַזֶּ֑ה וּרְאֵה֙ אֶת־הָאָ֔רֶץ אֲשֶׁ֥ר נָתַ֖תִּי לִבְנֵ֥י יִשְׂרָאֵֽל׃ וַאֲמַר יְיָ לְמֹשֶׁה, סַק לְטוּרָא דְּעִבְרָאֵי הָדֵין; וַחְזִי, יָת אַרְעָא, דִּיהַבִית, לִבְנֵי יִשְׂרָאֵל. 
כז,יג וְרָאִ֣יתָה אֹתָ֔הּ וְנֶֽאֱסַפְתָּ֥ אֶל־עַמֶּ֖יךָ גַּם־אָ֑תָּה כַּֽאֲשֶׁ֥ר נֶֽאֱסַ֖ף אַֽהֲרֹ֥ן אָחִֽיךָ׃ וְתִחְזֵי יָתַהּ, וְתִתְכְּנֵישׁ לְעַמָּךְ אַף אַתְּ, כְּמָא דְּאִתְכְּנֵישׁ אַהֲרוֹן, אֲחוּךְ. 
כז,יד כַּֽאֲשֶׁר֩ מְרִיתֶ֨ם פִּ֜י בְּמִדְבַּר־צִ֗ן בִּמְרִיבַת֙ הָֽעֵדָ֔ה לְהַקְדִּישֵׁ֥נִי בַמַּ֖יִם לְעֵֽינֵיהֶ֑ם הֵ֛ם מֵֽי־מְרִיבַ֥ת קָדֵ֖שׁ מִדְבַּר־צִֽן׃ {ס} כְּמָא דְּסָרֵיבְתּוּן עַל מֵימְרִי בְּמַדְבְּרָא דְּצִין, בְּמַצּוּת כְּנִשְׁתָּא, לְקַדָּשׁוּתִי בְּמַיָּא, לְעֵינֵיהוֹן:  אִנּוּן מֵי מַצּוּת רְקַם, מַדְבְּרָא דְּצִין. {ס}
כז,טו וַיְדַבֵּ֣ר מֹשֶׁ֔ה אֶל־יְהוָ֖ה לֵאמֹֽר׃ וּמַלֵּיל מֹשֶׁה, קֳדָם יְיָ לְמֵימַר. 
כז,טז יִפְקֹ֣ד יְהוָ֔ה אֱלֹהֵ֥י הָֽרוּחֹ֖ת לְכָל־בָּשָׂ֑ר אִ֖ישׁ עַל־הָֽעֵדָֽה׃ יְמַנֵּי יְיָ, אֱלָהּ רוּחַיָּא לְכָל בִּסְרָא, גֻּבְרָא, עַל כְּנִשְׁתָּא. 
כז,יז אֲשֶׁר־יֵצֵ֣א לִפְנֵיהֶ֗ם וַֽאֲשֶׁ֤ר יָבֹא֙ לִפְנֵיהֶ֔ם וַֽאֲשֶׁ֥ר יֽוֹצִיאֵ֖ם וַֽאֲשֶׁ֣ר יְבִיאֵ֑ם וְלֹ֤א תִֽהְיֶה֙ עֲדַ֣ת יְהוָ֔ה כַּצֹּ֕אן אֲשֶׁ֥ר אֵין־לָהֶ֖ם רֹעֶֽה׃ דְּיִפּוֹק קֳדָמֵיהוֹן, וּדְיֵיעוֹל קֳדָמֵיהוֹן, וּדְיַפֵּיקִנּוּן, וּדְיַעֵילִנּוּן; וְלָא תְּהֵי, כְּנִשְׁתָּא דַּייָ, כְּעָנָא, דְּלֵית לְהוֹן רָעֵי. 
כז,יח וַיֹּ֨אמֶר יְהוָ֜ה אֶל־מֹשֶׁ֗ה קַח־לְךָ֙ אֶת־יְהוֹשֻׁ֣עַ בִּן־נ֔וּן אִ֖ישׁ אֲשֶׁר־ר֣וּחַ בּ֑וֹ וְסָֽמַכְתָּ֥ אֶת־יָֽדְךָ֖ עָלָֽיו׃ וַאֲמַר יְיָ לְמֹשֶׁה, סַב לָךְ יָת יְהוֹשׁוּעַ בַּר נוּן--גְּבַר, דְּרוּחַ נְבוּאָה בֵּיהּ; וְתִסְמוֹךְ יָת יְדָךְ, עֲלוֹהִי. 
כז,יט וְהַֽעֲמַדְתָּ֣ אֹת֗וֹ לִפְנֵי֙ אֶלְעָזָ֣ר הַכֹּהֵ֔ן וְלִפְנֵ֖י כָּל־הָֽעֵדָ֑ה וְצִוִּיתָ֥ה אֹת֖וֹ לְעֵֽינֵיהֶֽם׃ וּתְקִים יָתֵיהּ, קֳדָם אֶלְעָזָר כָּהֲנָא, וּקְדָם, כָּל כְּנִשְׁתָּא; וּתְפַקֵּיד יָתֵיהּ, לְעֵינֵיהוֹן. 
כז,כ וְנָֽתַתָּ֥ה מֵהֽוֹדְךָ֖ עָלָ֑יו לְמַ֣עַן יִשְׁמְע֔וּ כָּל־עֲדַ֖ת בְּנֵ֥י יִשְׂרָאֵֽל׃ וְתִתֵּין מִזִּיוָךְ, עֲלוֹהִי--בְּדִיל דִּיקַבְּלוּן מִנֵּיהּ, כָּל כְּנִשְׁתָּא דִּבְנֵי יִשְׂרָאֵל. 
כז,כא וְלִפְנֵ֨י אֶלְעָזָ֤ר הַכֹּהֵן֙ יַֽעֲמֹ֔ד וְשָׁ֥אַל ל֛וֹ בְּמִשְׁפַּ֥ט הָֽאוּרִ֖ים לִפְנֵ֣י יְהוָ֑ה עַל־פִּ֨יו יֵֽצְא֜וּ וְעַל־פִּ֣יו יָבֹ֗אוּ ה֛וּא וְכָל־בְּנֵֽי־יִשְׂרָאֵ֥ל אִתּ֖וֹ וְכָל־הָֽעֵדָֽה׃ וּקְדָם אֶלְעָזָר כָּהֲנָא יְקוּם, וְיִשְׁאַל לֵיהּ בְּדִין אוּרַיָּא קֳדָם יְיָ:  עַל מֵימְרֵיהּ יְהוֹן נָפְקִין וְעַל מֵימְרֵיהּ יְהוֹן עָאלִין, הוּא וְכָל בְּנֵי יִשְׂרָאֵל עִמֵּיהּ--וְכָל כְּנִשְׁתָּא. 
כז,כב וַיַּ֣עַשׂ מֹשֶׁ֔ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֹת֑וֹ וַיִּקַּ֣ח אֶת־יְהוֹשֻׁ֗עַ וַיַּֽעֲמִדֵ֨הוּ֙ לִפְנֵי֙ אֶלְעָזָ֣ר הַכֹּהֵ֔ן וְלִפְנֵ֖י כָּל־הָֽעֵדָֽה׃ וַעֲבַד מֹשֶׁה, כְּמָא דְּפַקֵּיד יְיָ יָתֵיהּ; וּדְבַר יָת יְהוֹשׁוּעַ, וַאֲקִימֵיהּ קֳדָם אֶלְעָזָר כָּהֲנָא, וּקְדָם, כָּל כְּנִשְׁתָּא. 
כז,כג וַיִּסְמֹ֧ךְ אֶת־יָדָ֛יו עָלָ֖יו וַיְצַוֵּ֑הוּ כַּֽאֲשֶׁ֛ר דִּבֶּ֥ר יְהוָ֖ה בְּיַד־מֹשֶֽׁה׃ {פ} וּסְמַךְ יָת יְדוֹהִי עֲלוֹהִי, וּפַקְּדֵיהּ, כְּמָא דְּמַלֵּיל יְיָ, בִּידָא דְּמֹשֶׁה. {פ}
כח,א וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
כח,ב צַ֚ו אֶת־בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָֽמַרְתָּ֖ אֲלֵהֶ֑ם אֶת־קָרְבָּנִ֨י לַחְמִ֜י לְאִשַּׁ֗י רֵ֚יחַ נִֽיחֹחִ֔י תִּשְׁמְר֕וּ לְהַקְרִ֥יב לִ֖י בְּמֽוֹעֲדֽוֹ׃ פַּקֵּיד יָת בְּנֵי יִשְׂרָאֵל, וְתֵימַר לְהוֹן:  יָת קֻרְבָּנִי לְחֵים סִדּוּר לְקֻרְבָּנַי, לְאִתְקַבָּלָא בְּרַעֲוָא, תִּטְּרוּן, לְקָרָבָא קֳדָמַי בְּזִמְנֵיהּ. 
כח,ג וְאָֽמַרְתָּ֣ לָהֶ֔ם זֶ֚ה הָֽאִשֶּׁ֔ה אֲשֶׁ֥ר תַּקְרִ֖יבוּ לַֽיהוָ֑ה כְּבָשִׂ֨ים בְּנֵֽי־שָׁנָ֧ה תְמִימִ֛ם שְׁנַ֥יִם לַיּ֖וֹם עֹלָ֥ה תָמִֽיד׃ וְתֵימַר לְהוֹן--דֵּין קֻרְבָּנָא, דִּתְקָרְבוּן קֳדָם יְיָ:  אִמְּרִין בְּנֵי שְׁנָא שַׁלְמִין תְּרֵין לְיוֹמָא, עֲלָתָא תְּדִירָא. 
כח,ד אֶת־הַכֶּ֥בֶשׂ אֶחָ֖ד תַּֽעֲשֶׂ֣ה בַבֹּ֑קֶר וְאֵת֙ הַכֶּ֣בֶשׂ הַשֵּׁנִ֔י תַּֽעֲשֶׂ֖ה בֵּ֥ין הָֽעַרְבָּֽיִם׃ יָת אִמְּרָא חַד, תַּעֲבֵיד בְּצַפְרָא; וְיָת אִמְּרָא תִּנְיָנָא, תַּעֲבֵיד בֵּין שִׁמְשַׁיָּא. 
כח,ה וַֽעֲשִׂירִ֧ית הָֽאֵיפָ֛ה סֹ֖לֶת לְמִנְחָ֑ה בְּלוּלָ֛ה בְּשֶׁ֥מֶן כָּתִ֖ית רְבִיעִ֥ת הַהִֽין׃ וְחַד מִן עַסְרָא בִּתְלָת סְאִין סֻלְתָּא, לְמִנְחָתָא, דְּפִילָא בִּמְשַׁח כָּתִישָׁא, רַבְעוּת הִינָא. 
כח,ו עֹלַ֖ת תָּמִ֑יד הָֽעֲשֻׂיָה֙ בְּהַ֣ר סִינַ֔י לְרֵ֣יחַ נִיחֹ֔חַ אִשֶּׁ֖ה לַֽיהוָֽה׃ עֲלַת, תְּדִירָא--דְּאִתְעֲבֵידַת, בְּטוּרָא דְּסִינַי, לְאִתְקַבָּלָא בְּרַעֲוָא, קֻרְבָּנָא קֳדָם יְיָ. 
כח,ז וְנִסְכּוֹ֙ רְבִיעִ֣ת הַהִ֔ין לַכֶּ֖בֶשׂ הָֽאֶחָ֑ד בַּקֹּ֗דֶשׁ הַסֵּ֛ךְ נֶ֥סֶךְ שֵׁכָ֖ר לַֽיהוָֽה׃ וְנִסְכֵּיהּ רַבְעוּת הִינָא, לְאִמְּרָא חַד; בְּקֻדְשָׁא, יִתְנַסַּךְ נִסּוּךְ דַּחֲמַר עַתִּיק--קֳדָם יְיָ. 
כח,ח וְאֵת֙ הַכֶּ֣בֶשׂ הַשֵּׁנִ֔י תַּֽעֲשֶׂ֖ה בֵּ֣ין הָֽעַרְבָּ֑יִם כְּמִנְחַ֨ת הַבֹּ֤קֶר וּכְנִסְכּוֹ֙ תַּֽעֲשֶׂ֔ה אִשֵּׁ֛ה רֵ֥יחַ נִיחֹ֖חַ לַֽיהוָֽה׃ {פ} וְיָת אִמְּרָא תִּנְיָנָא, תַּעֲבֵיד בֵּין שִׁמְשַׁיָּא:  כְּמִנְחַת צַפְרָא וּכְנִסְכֵּיהּ תַּעֲבֵיד, קֻרְבַּן דְּמִתְקַבַּל בְּרַעֲוָא קֳדָם יְיָ. {פ}
כח,ט וּבְיוֹם֙ הַשַּׁבָּ֔ת שְׁנֵֽי־כְבָשִׂ֥ים בְּנֵֽי־שָׁנָ֖ה תְּמִימִ֑ם וּשְׁנֵ֣י עֶשְׂרֹנִ֗ים סֹ֧לֶת מִנְחָ֛ה בְּלוּלָ֥ה בַשֶּׁ֖מֶן וְנִסְכּֽוֹ׃ וּבְיוֹמָא, דְּשַׁבְּתָא--תְּרֵין אִמְּרִין בְּנֵי שְׁנָא, שַׁלְמִין; וּתְרֵין עֶסְרוֹנִין, סֻלְתָּא מִנְחָתָא דְּפִילָא בִּמְשַׁח--וְנִסְכֵּיהּ. 
כח,י עֹלַ֥ת שַׁבַּ֖ת בְּשַׁבַּתּ֑וֹ עַל־עֹלַ֥ת הַתָּמִ֖יד וְנִסְכָּֽהּ׃ {פ} עֲלַת שַׁבָּא, תִּתְעֲבֵיד בְּשַׁבָּא, עַל עֲלַת תְּדִירָא, וְנִסְכַּהּ. {פ}
כח,יא וּבְרָאשֵׁי֙ חָדְשֵׁיכֶ֔ם תַּקְרִ֥יבוּ עֹלָ֖ה לַֽיהוָ֑ה פָּרִ֨ים בְּנֵֽי־בָקָ֤ר שְׁנַ֨יִם֙ וְאַ֣יִל אֶחָ֔ד כְּבָשִׂ֧ים בְּנֵֽי־שָׁנָ֛ה שִׁבְעָ֖ה תְּמִימִֽם׃ וּבְרֵישֵׁי, יַרְחֵיכוֹן--תְּקָרְבוּן עֲלָתָא, קֳדָם יְיָ:  תּוֹרֵי בְּנֵי תּוֹרֵי תְּרֵין וּדְכַר חַד, אִמְּרִין בְּנֵי שְׁנָא שִׁבְעָא שַׁלְמִין. 
כח,יב וּשְׁלֹשָׁ֣ה עֶשְׂרֹנִ֗ים סֹ֤לֶת מִנְחָה֙ בְּלוּלָ֣ה בַשֶּׁ֔מֶן לַפָּ֖ר הָֽאֶחָ֑ד וּשְׁנֵ֣י עֶשְׂרֹנִ֗ים סֹ֤לֶת מִנְחָה֙ בְּלוּלָ֣ה בַשֶּׁ֔מֶן לָאַ֖יִל הָֽאֶחָֽד׃ וּתְלָתָא עֶסְרוֹנִין, סֻלְתָּא מִנְחָתָא דְּפִילָא בִּמְשַׁח, לְתוֹרָא, חַד; וּתְרֵין עֶסְרוֹנִין, סֻלְתָּא מִנְחָתָא דְּפִילָא בִּמְשַׁח, לְדִכְרָא, חַד. 
כח,יג וְעִשָּׂרֹ֣ן עִשָּׂר֗וֹן סֹ֤לֶת מִנְחָה֙ בְּלוּלָ֣ה בַשֶּׁ֔מֶן לַכֶּ֖בֶשׂ הָֽאֶחָ֑ד עֹלָה֙ רֵ֣יחַ נִיחֹ֔חַ אִשֶּׁ֖ה לַֽיהוָֽה׃ וְעֶסְרוֹנָא עֶסְרוֹנָא, סֻלְתָּא מִנְחָתָא דְּפִילָא בִּמְשַׁח, לְאִמְּרָא, חַד; עֲלָתָא לְאִתְקַבָּלָא בְּרַעֲוָא, קֻרְבָּנָא קֳדָם יְיָ. 
כח,יד וְנִסְכֵּיהֶ֗ם חֲצִ֣י הַהִין֩ יִֽהְיֶ֨ה לַפָּ֜ר וּשְׁלִישִׁ֧ת הַהִ֣ין לָאַ֗יִל וּרְבִיעִ֥ת הַהִ֛ין לַכֶּ֖בֶשׂ יָ֑יִן זֹ֣את עֹלַ֥ת חֹ֨דֶשׁ֙ בְּחָדְשׁ֔וֹ לְחָדְשֵׁ֖י הַשָּׁנָֽה׃ וְנִסְכֵּיהוֹן, פַּלְגוּת הִינָא יְהֵי לְתוֹרָא וְתַלְתוּת הִינָא לְדִכְרָא וְרַבְעוּת הִינָא לְאִמְּרָא--חַמְרָא:  דָּא עֲלַת רֵישׁ יַרְחָא בְּאִתְחַדָּתוּתֵיהּ, כֵּין לְכָל רֵישֵׁי יַרְחֵי שַׁתָּא. 
כח,טו וּשְׂעִ֨יר עִזִּ֥ים אֶחָ֛ד לְחַטָּ֖את לַֽיהוָ֑ה עַל־עֹלַ֧ת הַתָּמִ֛יד יֵֽעָשֶׂ֖ה וְנִסְכּֽוֹ׃ {ס} וּצְפִיר בַּר עִזִּין חַד לְחַטָּתָא, קֳדָם יְיָ, עַל עֲלַת תְּדִירָא יִתְעֲבֵיד, וְנִסְכֵּיהּ. {ס}
כח,טז וּבַחֹ֣דֶשׁ הָֽרִאשׁ֗וֹן בְּאַרְבָּעָ֥ה עָשָׂ֛ר י֖וֹם לַחֹ֑דֶשׁ פֶּ֖סַח לַֽיהוָֽה׃ וּבְיַרְחָא קַדְמָאָה, בְּאַרְבְּעַת עַסְרָא יוֹמָא--לְיַרְחָא:  פִּסְחָא, קֳדָם יְיָ. 
כח,יז וּבַֽחֲמִשָּׁ֨ה עָשָׂ֥ר י֛וֹם לַחֹ֥דֶשׁ הַזֶּ֖ה חָ֑ג שִׁבְעַ֣ת יָמִ֔ים מַצּ֖וֹת יֵֽאָכֵֽל׃ וּבַחֲמֵישְׁתְּ עַסְרָא יוֹמָא לְיַרְחָא הָדֵין, חַגָּא:  שִׁבְעָא יוֹמִין, פַּטִּירָא יִתְאֲכִיל. 
כח,יח בַּיּ֥וֹם הָֽרִאשׁ֖וֹן מִקְרָא־קֹ֑דֶשׁ כָּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֥א תַֽעֲשֽׂוּ׃ בְּיוֹמָא קַדְמָאָה, מְעָרַע קַדִּישׁ:  כָּל עֲבִידַת פֻּלְחַן, לָא תַּעְבְּדוּן. 
כח,יט וְהִקְרַבְתֶּ֨ם אִשֶּׁ֤ה עֹלָה֙ לַֽיהוָ֔ה פָּרִ֧ים בְּנֵֽי־בָקָ֛ר שְׁנַ֖יִם וְאַ֣יִל אֶחָ֑ד וְשִׁבְעָ֤ה כְבָשִׂים֙ בְּנֵ֣י שָׁנָ֔ה תְּמִימִ֖ם יִֽהְי֥וּ לָכֶֽם׃ וּתְקָרְבוּן קֻרְבָּנָא עֲלָתָא קֳדָם יְיָ, תּוֹרֵי בְּנֵי תּוֹרֵי תְּרֵין וּדְכַר חַד; וְשִׁבְעָא אִמְּרִין בְּנֵי שְׁנָא, שַׁלְמִין יְהוֹן לְכוֹן. 
כח,כ וּמִ֨נְחָתָ֔ם סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן שְׁלֹשָׁ֨ה עֶשְׂרֹנִ֜ים לַפָּ֗ר וּשְׁנֵ֧י עֶשְׂרֹנִ֛ים לָאַ֖יִל תַּֽעֲשֽׂוּ׃ וּמִנְחָתְהוֹן--סֻלְתָּא, דְּפִילָא בִּמְשַׁח:  תְּלָתָא עֶסְרוֹנִין לְתוֹרָא, וּתְרֵין עֶסְרוֹנִין לְדִכְרָא--תַּעְבְּדוּן. 
כח,כא עִשָּׂר֤וֹן עִשָּׂרוֹן֙ תַּֽעֲשֶׂ֔ה לַכֶּ֖בֶשׂ הָֽאֶחָ֑ד לְשִׁבְעַ֖ת הַכְּבָשִֽׂים׃ עֶסְרוֹנָא עֶסְרוֹנָא, תַּעֲבֵיד, לְאִמְּרָא, חַד--כֵּין לְשִׁבְעָא, אִמְּרִין. 
כח,כב וּשְׂעִ֥יר חַטָּ֖את אֶחָ֑ד לְכַפֵּ֖ר עֲלֵיכֶֽם׃ וּצְפִירָא דְּחַטָּתָא, חַד, לְכַפָּרָא, עֲלֵיכוֹן. 
כח,כג מִלְּבַד֙ עֹלַ֣ת הַבֹּ֔קֶר אֲשֶׁ֖ר לְעֹלַ֣ת הַתָּמִ֑יד תַּֽעֲשׂ֖וּ אֶת־אֵֽלֶּה׃ בָּר מֵעֲלַת צַפְרָא, דְּהִיא עֲלַת תְּדִירָא--תַּעְבְּדוּן, יָת אִלֵּין. 
כח,כד כָּאֵ֜לֶּה תַּֽעֲשׂ֤וּ לַיּוֹם֙ שִׁבְעַ֣ת יָמִ֔ים לֶ֛חֶם אִשֵּׁ֥ה רֵֽיחַ־נִיחֹ֖חַ לַֽיהוָ֑ה עַל־עוֹלַ֧ת הַתָּמִ֛יד יֵֽעָשֶׂ֖ה וְנִסְכּֽוֹ׃ כְּאִלֵּין תַּעְבְּדוּן לְיוֹמָא, שִׁבְעָא יוֹמִין--לְחֵים קֻרְבַּן דְּמִתְקַבַּל בְּרַעֲוָא, קֳדָם יְיָ; עַל עֲלַת תְּדִירָא יִתְעֲבֵיד, וְנִסְכֵּיהּ. 
כח,כה וּבַיּוֹם֙ הַשְּׁבִיעִ֔י מִקְרָא־קֹ֖דֶשׁ יִֽהְיֶ֣ה לָכֶ֑ם כָּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֥א תַֽעֲשֽׂוּ׃ {ס} וּבְיוֹמָא, שְׁבִיעָאָה--מְעָרַע קַדִּישׁ, יְהֵי לְכוֹן:  כָּל עֲבִידַת פֻּלְחָן, לָא תַּעְבְּדוּן. {ס}
כח,כו וּבְי֣וֹם הַבִּכּוּרִ֗ים בְּהַקְרִ֨יבְכֶ֜ם מִנְחָ֤ה חֲדָשָׁה֙ לַֽיהוָ֔ה בְּשָׁבֻעֹ֖תֵיכֶ֑ם מִֽקְרָא־קֹ֨דֶשׁ֙ יִֽהְיֶ֣ה לָכֶ֔ם כָּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֥א תַֽעֲשֽׂוּ׃ וּבְיוֹמָא דְּבִכּוּרַיָּא, בְּקָרוֹבֵיכוֹן מִנְחָתָא חֲדַתָּא קֳדָם יְיָ--בְּעַצְרָתְכוֹן:  מְעָרַע קַדִּישׁ יְהֵי לְכוֹן, כָּל עֲבִידַת פֻּלְחַן לָא תַּעְבְּדוּן. 
כח,כז וְהִקְרַבְתֶּ֨ם עוֹלָ֜ה לְרֵ֤יחַ נִיחֹ֨חַ֙ לַֽיהוָ֔ה פָּרִ֧ים בְּנֵֽי־בָקָ֛ר שְׁנַ֖יִם אַ֣יִל אֶחָ֑ד שִׁבְעָ֥ה כְבָשִׂ֖ים בְּנֵ֥י שָׁנָֽה׃ וּתְקָרְבוּן עֲלָתָא לְאִתְקַבָּלָא בְּרַעֲוָא, קֳדָם יְיָ--תּוֹרֵי בְּנֵי תּוֹרֵי תְּרֵין, דְּכַר חַד; שִׁבְעָא אִמְּרִין, בְּנֵי שְׁנָא. 
כח,כח וּמִ֨נְחָתָ֔ם סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן שְׁלֹשָׁ֤ה עֶשְׂרֹנִים֙ לַפָּ֣ר הָֽאֶחָ֔ד שְׁנֵי֙ עֶשְׂרֹנִ֔ים לָאַ֖יִל הָֽאֶחָֽד׃ וּמִנְחָתְהוֹן--סֻלְתָּא, דְּפִילָא בִּמְשַׁח:  תְּלָתָא עֶסְרוֹנִין, לְתוֹרָא חַד, תְּרֵין עֶסְרוֹנִין, לְדִכְרָא חַד. 
כח,כט עִשָּׂרוֹן֙ עִשָּׂר֔וֹן לַכֶּ֖בֶשׂ הָֽאֶחָ֑ד לְשִׁבְעַ֖ת הַכְּבָשִֽׂים׃ עֶסְרוֹנָא, עֶסְרוֹנָא, לְאִמְּרָא, חַד--כֵּין לְשִׁבְעָא, אִמְּרִין. 
כח,ל שְׂעִ֥יר עִזִּ֖ים אֶחָ֑ד לְכַפֵּ֖ר עֲלֵיכֶֽם׃ צְפִיר בַּר עִזִּין, חַד, לְכַפָּרָא, עֲלֵיכוֹן. 
כח,לא מִלְּבַ֞ד עֹלַ֧ת הַתָּמִ֛יד וּמִנְחָת֖וֹ תַּֽעֲשׂ֑וּ תְּמִימִ֥ם יִֽהְיוּ־לָכֶ֖ם וְנִסְכֵּיהֶֽם׃ {פ} בָּר מֵעֲלַת תְּדִירָא, וּמִנְחָתֵיהּ--תַּעְבְּדוּן; שַׁלְמִין יְהוֹן לְכוֹן, וְנִסְכֵּיהוֹן. {פ}
כט,א וּבַחֹ֨דֶשׁ הַשְּׁבִיעִ֜י בְּאֶחָ֣ד לַחֹ֗דֶשׁ מִֽקְרָא־קֹ֨דֶשׁ֙ יִֽהְיֶ֣ה לָכֶ֔ם כָּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֣א תַֽעֲשׂ֑וּ י֥וֹם תְּרוּעָ֖ה יִֽהְיֶ֥ה לָכֶֽם׃ וּבְיַרְחָא שְׁבִיעָאָה בְּחַד לְיַרְחָא, מְעָרַע קַדִּישׁ יְהֵי לְכוֹן--כָּל עֲבִידַת פֻּלְחַן, לָא תַּעְבְּדוּן:  יוֹם יַבָּבָא, יְהֵי לְכוֹן. 
כט,ב וַֽעֲשִׂיתֶ֨ם עֹלָ֜ה לְרֵ֤יחַ נִיחֹ֨חַ֙ לַֽיהוָ֔ה פַּ֧ר בֶּן־בָּקָ֛ר אֶחָ֖ד אַ֣יִל אֶחָ֑ד כְּבָשִׂ֧ים בְּנֵֽי־שָׁנָ֛ה שִׁבְעָ֖ה תְּמִימִֽם׃ וְתַעְבְּדוּן עֲלָתָא לְאִתְקַבָּלָא בְּרַעֲוָא, קֳדָם יְיָ--תּוֹר בַּר תּוֹרֵי חַד, דְּכַר חַד; אִמְּרִין בְּנֵי שְׁנָא שִׁבְעָא, שַׁלְמִין. 
כט,ג וּמִ֨נְחָתָ֔ם סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן שְׁלֹשָׁ֤ה עֶשְׂרֹנִים֙ לַפָּ֔ר שְׁנֵ֥י עֶשְׂרֹנִ֖ים לָאָֽיִל׃ וּמִנְחָתְהוֹן--סֻלְתָּא, דְּפִילָא בִּמְשַׁח:  תְּלָתָא עֶסְרוֹנִין לְתוֹרָא, תְּרֵין עֶסְרוֹנִין לְדִכְרָא. 
כט,ד וְעִשָּׂר֣וֹן אֶחָ֔ד לַכֶּ֖בֶשׂ הָֽאֶחָ֑ד לְשִׁבְעַ֖ת הַכְּבָשִֽׂים׃ וְעֶסְרוֹנָא חַד, לְאִמְּרָא חַד, כֵּין לְשִׁבְעָא, אִמְּרִין. 
כט,ה וּשְׂעִיר־עִזִּ֥ים אֶחָ֖ד חַטָּ֑את לְכַפֵּ֖ר עֲלֵיכֶֽם׃ וּצְפִיר בַּר עִזִּין חַד, חַטָּתָא, לְכַפָּרָא, עֲלֵיכוֹן. 
כט,ו מִלְּבַד֩ עֹלַ֨ת הַחֹ֜דֶשׁ וּמִנְחָתָ֗הּ וְעֹלַ֤ת הַתָּמִיד֙ וּמִנְחָתָ֔הּ וְנִסְכֵּיהֶ֖ם כְּמִשְׁפָּטָ֑ם לְרֵ֣יחַ נִיחֹ֔חַ אִשֶּׁ֖ה לַֽיהוָֽה׃ {ס} בָּר מֵעֲלַת יַרְחָא וּמִנְחָתַהּ, וַעֲלַת תְּדִירָא וּמִנְחָתַהּ, וְנִסְכֵּיהוֹן, כְּדַחְזֵי לְהוֹן:  לְאִתְקַבָּלָא בְּרַעֲוָא, קֻרְבָּנָא קֳדָם יְיָ. {ס}
כט,ז וּבֶֽעָשׂוֹר֩ לַחֹ֨דֶשׁ הַשְּׁבִיעִ֜י הַזֶּ֗ה מִֽקְרָא־קֹ֨דֶשׁ֙ יִֽהְיֶ֣ה לָכֶ֔ם וְעִנִּיתֶ֖ם אֶת־נַפְשֹֽׁתֵיכֶ֑ם כָּל־מְלָאכָ֖ה לֹ֥א תַֽעֲשֽׂוּ׃ וּבְעַסְרָא לְיַרְחָא שְׁבִיעָאָה הָדֵין, מְעָרַע קַדִּישׁ יְהֵי לְכוֹן, וּתְעַנּוֹן, יָת נַפְשָׁתְכוֹן; כָּל עֲבִידָא, לָא תַּעְבְּדוּן. 
כט,ח וְהִקְרַבְתֶּ֨ם עֹלָ֤ה לַֽיהוָה֙ רֵ֣יחַ נִיחֹ֔חַ פַּ֧ר בֶּן־בָּקָ֛ר אֶחָ֖ד אַ֣יִל אֶחָ֑ד כְּבָשִׂ֤ים בְּנֵֽי־שָׁנָה֙ שִׁבְעָ֔ה תְּמִימִ֖ם יִֽהְי֥וּ לָכֶֽם׃ וּתְקָרְבוּן עֲלָתָא קֳדָם יְיָ לְאִתְקַבָּלָא בְּרַעֲוָא, תּוֹר בַּר תּוֹרֵי חַד דְּכַר חַד; אִמְּרִין בְּנֵי שְׁנָא שִׁבְעָא, שַׁלְמִין יְהוֹן לְכוֹן. 
כט,ט וּמִ֨נְחָתָ֔ם סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן שְׁלֹשָׁ֤ה עֶשְׂרֹנִים֙ לַפָּ֔ר שְׁנֵי֙ עֶשְׂרֹנִ֔ים לָאַ֖יִל הָֽאֶחָֽד׃ וּמִנְחָתְהוֹן--סֻלְתָּא, דְּפִילָא בִּמְשַׁח:  תְּלָתָא עֶסְרוֹנִין, לְתוֹרָא, תְּרֵין עֶסְרוֹנִין, לְדִכְרָא חַד. 
כט,י עִשָּׂרוֹן֙ עִשָּׂר֔וֹן לַכֶּ֖בֶשׂ הָֽאֶחָ֑ד לְשִׁבְעַ֖ת הַכְּבָשִֽׂים׃ עֶסְרוֹנָא, עֶסְרוֹנָא, לְאִמְּרָא, חַד--כֵּין לְשִׁבְעָא, אִמְּרִין. 
כט,יא שְׂעִיר־עִזִּ֥ים אֶחָ֖ד חַטָּ֑את מִלְּבַ֞ד חַטַּ֤את הַכִּפֻּרִים֙ וְעֹלַ֣ת הַתָּמִ֔יד וּמִנְחָתָ֖הּ וְנִסְכֵּיהֶֽם׃ {ס} צְפִיר בַּר עִזִּין חַד, חַטָּתָא; בָּר מֵחַטַּת כִּפּוּרַיָּא, וַעֲלַת תְּדִירָא, וּמִנְחָתַהּ, וְנִסְכֵּיהוֹן. {ס}
כט,יב וּבַֽחֲמִשָּׁה֩ עָשָׂ֨ר י֜וֹם לַחֹ֣דֶשׁ הַשְּׁבִיעִ֗י מִֽקְרָא־קֹ֨דֶשׁ֙ יִֽהְיֶ֣ה לָכֶ֔ם כָּל־מְלֶ֥אכֶת עֲבֹדָ֖ה לֹ֣א תַֽעֲשׂ֑וּ וְחַגֹּתֶ֥ם חַ֛ג לַֽיהוָ֖ה שִׁבְעַ֥ת יָמִֽים׃ וּבַחֲמֵישְׁתְּ עַסְרָא יוֹמָא לְיַרְחָא שְׁבִיעָאָה, מְעָרַע קַדִּישׁ יְהֵי לְכוֹן--כָּל עֲבִידַת פֻּלְחַן, לָא תַּעְבְּדוּן; וְתֵיחֲגוּן חַגָּא קֳדָם יְיָ, שִׁבְעָא יוֹמִין. 
כט,יג וְהִקְרַבְתֶּ֨ם עֹלָ֜ה אִשֵּׁ֨ה רֵ֤יחַ נִיחֹ֨חַ֙ לַֽיהוָ֔ה פָּרִ֧ים בְּנֵֽי־בָקָ֛ר שְׁלֹשָׁ֥ה עָשָׂ֖ר אֵילִ֣ם שְׁנָ֑יִם כְּבָשִׂ֧ים בְּנֵֽי־שָׁנָ֛ה אַרְבָּעָ֥ה עָשָׂ֖ר תְּמִימִ֥ם יִֽהְיֽוּ׃ וּתְקָרְבוּן עֲלָתָא קֻרְבַּן דְּמִתְקַבַּל בְּרַעֲוָא, קֳדָם יְיָ--תּוֹרֵי בְּנֵי תּוֹרֵי תְּלָת עֲסַר, דִּכְרִין תְּרֵין; אִמְּרִין בְּנֵי שְׁנָא אַרְבְּעַת עֲסַר, שַׁלְמִין יְהוֹן. 
כט,יד וּמִ֨נְחָתָ֔ם סֹ֖לֶת בְּלוּלָ֣ה בַשָּׁ֑מֶן שְׁלֹשָׁ֨ה עֶשְׂרֹנִ֜ים לַפָּ֣ר הָֽאֶחָ֗ד לִשְׁלֹשָׁ֤ה עָשָׂר֙ פָּרִ֔ים שְׁנֵ֤י עֶשְׂרֹנִים֙ לָאַ֣יִל הָֽאֶחָ֔ד לִשְׁנֵ֖י הָֽאֵילִֽם׃ וּמִנְחָתְהוֹן--סֻלְתָּא, דְּפִילָא בִּמְשַׁח:  תְּלָתָא עֶסְרוֹנִין לְתוֹרָא חַד, לִתְלָת עֲסַר תּוֹרִין, תְּרֵין עֶסְרוֹנִין לְדִכְרָא חַד, לִתְרֵין דִּכְרִין.
