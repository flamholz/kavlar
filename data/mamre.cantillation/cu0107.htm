וייצא

כח,י וַיֵּצֵ֥א יַֽעֲקֹ֖ב מִבְּאֵ֣ר שָׁ֑בַע וַיֵּ֖לֶךְ חָרָֽנָה׃ וּנְפַק יַעֲקוֹב, מִבְּאֵר שָׁבַע; וַאֲזַל, לְחָרָן. 
כח,יא וַיִּפְגַּ֨ע בַּמָּק֜וֹם וַיָּ֤לֶן שָׁם֙ כִּי־בָ֣א הַשֶּׁ֔מֶשׁ וַיִּקַּח֙ מֵֽאַבְנֵ֣י הַמָּק֔וֹם וַיָּ֖שֶׂם מְרַֽאֲשֹׁתָ֑יו וַיִּשְׁכַּ֖ב בַּמָּק֥וֹם הַהֽוּא׃ וְעָרַע בְּאַתְרָא וּבָת תַּמָּן, אֲרֵי עָאל שִׁמְשָׁא, וּנְסֵיב מֵאַבְנֵי אַתְרָא, וְשַׁוִּי אִיסָדוֹהִי; וּשְׁכֵיב, בְּאַתְרָא הַהוּא. 
כח,יב וַֽיַּחֲלֹ֗ם וְהִנֵּ֤ה סֻלָּם֙ מֻצָּ֣ב אַ֔רְצָה וְרֹאשׁ֖וֹ מַגִּ֣יעַ הַשָּׁמָ֑יְמָה וְהִנֵּה֙ מַלְאֲכֵ֣י אֱלֹהִ֔ים עֹלִ֥ים וְיֹֽרְדִ֖ים בּֽוֹ׃ וַחֲלַם, וְהָא סֻלְמָא נְעִיץ בְּאַרְעָא, וְרֵישֵׁיהּ, מָטֵי עַד צֵית שְׁמַיָּא; וְהָא מַלְאֲכַיָּא דַּייָ, סָלְקִין וְנָחֲתִין בֵּיהּ. 
כח,יג וְהִנֵּ֨ה יְהוָ֜ה נִצָּ֣ב עָלָיו֮ וַיֹּאמַר֒ אֲנִ֣י יְהוָ֗ה אֱלֹהֵי֙ אַבְרָהָ֣ם אָבִ֔יךָ וֵֽאלֹהֵ֖י יִצְחָ֑ק הָאָ֗רֶץ אֲשֶׁ֤ר אַתָּה֙ שֹׁכֵ֣ב עָלֶ֔יהָ לְךָ֥ אֶתְּנֶ֖נָּה וּלְזַרְעֶֽךָ׃ וְהָא יְקָרָא דַּייָ מְעַתַּד עִלָּווֹהִי, וַאֲמַר, אֲנָא יְיָ אֱלָהֵיהּ דְּאַבְרָהָם אֲבוּךְ, וֵאלָהֵיהּ דְּיִצְחָק; אַרְעָא, דְּאַתְּ שָׁכֵיב עֲלַהּ--לָךְ אֶתְּנִנַּהּ, וְלִבְנָךְ. 
כח,יד וְהָיָ֤ה זַרְעֲךָ֙ כַּֽעֲפַ֣ר הָאָ֔רֶץ וּפָֽרַצְתָּ֛ יָ֥מָּה וָקֵ֖דְמָה וְצָפֹ֣נָה וָנֶ֑גְבָּה וְנִבְרְכ֥וּ בְךָ֛ כָּל־מִשְׁפְּחֹ֥ת הָֽאֲדָמָ֖ה וּבְזַרְעֶֽךָ׃ וִיהוֹן בְּנָךְ סַגִּיאִין כְּעַפְרָא דְּאַרְעָא, וְתִתַּקַּף לְמַעְרְבָא וּלְמַדְנְחָא וּלְצִפּוּנָא וּלְדָרוֹמָא; וְיִתְבָּרְכוּן בְּדִילָךְ כָּל זַרְעֲיָת אַרְעָא, וּבְדִיל בְּנָךְ. 
כח,טו וְהִנֵּ֨ה אָֽנֹכִ֜י עִמָּ֗ךְ וּשְׁמַרְתִּ֨יךָ֙ בְּכֹ֣ל אֲשֶׁר־תֵּלֵ֔ךְ וַהֲשִׁ֣בֹתִ֔יךָ אֶל־הָֽאֲדָמָ֖ה הַזֹּ֑את כִּ֚י לֹ֣א אֶֽעֱזָבְךָ֔ עַ֚ד אֲשֶׁ֣ר אִם־עָשִׂ֔יתִי אֵ֥ת אֲשֶׁר־דִּבַּ֖רְתִּי לָֽךְ׃ וְהָא מֵימְרִי בְּסַעֲדָךְ, וְאַטְּרִנָּךְ בְּכֹל אֲתַר דִּתְהָךְ, וַאֲתִיבִנָּךְ, לְאַרְעָא הָדָא:  אֲרֵי, לָא אֶשְׁבְּקִנָּךְ, עַד דְּאַעֲבֵיד, יָת דְּמַלֵּילִית לָךְ. 
כח,טז וַיִּיקַ֣ץ יַֽעֲקֹב֮ מִשְּׁנָתוֹ֒ וַיֹּ֕אמֶר אָכֵן֙ יֵ֣שׁ יְהוָ֔ה בַּמָּק֖וֹם הַזֶּ֑ה וְאָֽנֹכִ֖י לֹ֥א יָדָֽעְתִּי׃ וְאִתְּעַר יַעֲקוֹב, מִשִּׁינְתֵיהּ, וַאֲמַר, בְּקֻשְׁטָא יְקָרָא דַּייָ שְׁרִי בְּאַתְרָא הָדֵין; וַאֲנָא, לָא הֲוֵיתִי יָדַע. 
כח,יז וַיִּירָא֙ וַיֹּאמַ֔ר מַה־נּוֹרָ֖א הַמָּק֣וֹם הַזֶּ֑ה אֵ֣ין זֶ֗ה כִּ֚י אִם־בֵּ֣ית אֱלֹהִ֔ים וְזֶ֖ה שַׁ֥עַר הַשָּׁמָֽיִם׃ וּדְחֵיל, וַאֲמַר, מָא דְּחִילוּ, אַתְרָא הָדֵין:  לֵית דֵּין אֲתַר הִדְיוֹט, אֱלָהֵין אֲתַר דְּרַעֲוָא בֵּיהּ מִן קֳדָם יְיָ, וְדֵין, תְּרַע קֳבֵיל שְׁמַיָּא. 
כח,יח וַיַּשְׁכֵּ֨ם יַֽעֲקֹ֜ב בַּבֹּ֗קֶר וַיִּקַּ֤ח אֶת־הָאֶ֨בֶן֙ אֲשֶׁר־שָׂ֣ם מְרַֽאֲשֹׁתָ֔יו וַיָּ֥שֶׂם אֹתָ֖הּ מַצֵּבָ֑ה וַיִּצֹ֥ק שֶׁ֖מֶן עַל־רֹאשָֽׁהּ׃ וְאַקְדֵּים יַעֲקוֹב בְּצַפְרָא, וּנְסֵיב יָת אַבְנָא דְּשַׁוִּי אִיסָדוֹהִי, וְשַׁוִּי יָתַהּ, קָמָא; וַאֲרֵיק מִשְׁחָא, עַל רֵישַׁהּ. 
כח,יט וַיִּקְרָ֛א אֶת־שֵֽׁם־הַמָּק֥וֹם הַה֖וּא בֵּֽית־אֵ֑ל וְאוּלָ֛ם ל֥וּז שֵׁם־הָעִ֖יר לָרִֽאשֹׁנָֽה׃ וּקְרָא יָת שְׁמֵיהּ דְּאַתְרָא הַהוּא, בֵּית אֵל; וּבְרַם לוּז שְׁמַהּ דְּקַרְתָּא, מִלְּקַדְמִין. 
כח,כ וַיִּדַּ֥ר יַֽעֲקֹ֖ב נֶ֣דֶר לֵאמֹ֑ר אִם־יִֽהְיֶ֨ה אֱלֹהִ֜ים עִמָּדִ֗י וּשְׁמָרַ֨נִי֙ בַּדֶּ֤רֶךְ הַזֶּה֙ אֲשֶׁ֣ר אָֽנֹכִ֣י הוֹלֵ֔ךְ וְנָֽתַן־לִ֥י לֶ֛חֶם לֶֽאֱכֹ֖ל וּבֶ֥גֶד לִלְבֹּֽשׁ׃ וְקַיֵּים יַעֲקוֹב, קְיָם לְמֵימַר:  אִם יְהֵי מֵימְרָא דַּייָ בְּסַעֲדִי, וְיִטְּרִנַּנִי בְּאוֹרְחָא הָדָא דַּאֲנָא אָזֵיל, וְיִתֵּין לִי לְחֵים לְמֵיכַל, וּכְסוּ לְמִלְבַּשׁ. 
כח,כא וְשַׁבְתִּ֥י בְשָׁל֖וֹם אֶל־בֵּ֣ית אָבִ֑י וְהָיָ֧ה יְהוָ֛ה לִ֖י לֵֽאלֹהִֽים׃ וַאֲתוּב בִּשְׁלָם, לְבֵית אַבָּא; וִיהֵי מֵימְרָא דַּייָ לִי, לֶאֱלָהּ. 
כח,כב וְהָאֶ֣בֶן הַזֹּ֗את אֲשֶׁר־שַׂ֨מְתִּי֙ מַצֵּבָ֔ה יִֽהְיֶ֖ה בֵּ֣ית אֱלֹהִ֑ים וְכֹל֙ אֲשֶׁ֣ר תִּתֶּן־לִ֔י עַשֵּׂ֖ר אֲעַשְּׂרֶ֥נּוּ לָֽךְ׃ וְאַבְנָא הָדָא, דְּשַׁוִּיתִי קָמָא--תְּהֵי, דְּאֵיהֵי פָּלַח עֲלַהּ קֳדָם יְיָ; וְכֹל דְּתִתֵּין לִי, חַד מִן עַסְרָא אַפְרְשִׁנֵּיהּ קֳדָמָךְ. 
כט,א וַיִּשָּׂ֥א יַֽעֲקֹ֖ב רַגְלָ֑יו וַיֵּ֖לֶךְ אַ֥רְצָה בְנֵי־קֶֽדֶם׃ וּנְטַל יַעֲקוֹב, רַגְלוֹהִי; וַאֲזַל, לַאֲרַע בְּנֵי מַדְנְחָא. 
כט,ב וַיַּ֞רְא וְהִנֵּ֧ה בְאֵ֣ר בַּשָּׂדֶ֗ה וְהִנֵּה־שָׁ֞ם שְׁלֹשָׁ֤ה עֶדְרֵי־צֹאן֙ רֹֽבְצִ֣ים עָלֶ֔יהָ כִּ֚י מִן־הַבְּאֵ֣ר הַהִ֔וא יַשְׁק֖וּ הָֽעֲדָרִ֑ים וְהָאֶ֥בֶן גְּדֹלָ֖ה עַל־פִּ֥י הַבְּאֵֽר׃ וַחֲזָא וְהָא בֵּירָא בְּחַקְלָא, וְהָא תַּמָּן תְּלָתָא עֶדְרִין דְּעָן רְבִיעִין עֲלַהּ--אֲרֵי מִן בֵּירָא הַהִיא, מַשְׁקַן עֶדְרַיָּא; וְאַבְנָא רַבְּתָא, עַל פֻּמָּא דְּבֵירָא. 
כט,ג וְנֶֽאֶסְפוּ־שָׁ֣מָּה כָל־הָֽעֲדָרִ֗ים וְגָֽלְל֤וּ אֶת־הָאֶ֨בֶן֙ מֵעַל֙ פִּ֣י הַבְּאֵ֔ר וְהִשְׁק֖וּ אֶת־הַצֹּ֑אן וְהֵשִׁ֧יבוּ אֶת־הָאֶ֛בֶן עַל־פִּ֥י הַבְּאֵ֖ר לִמְקֹמָֽהּ׃ וּמִתְכַּנְשִׁין לְתַמָּן כָּל עֶדְרַיָּא, וּמְגַנְדְּרִין יָת אַבְנָא מֵעַל פֻּמָּא דְּבֵירָא, וּמַשְׁקַן, יָת עָנָא; וּמְתִיבִין יָת אַבְנָא עַל פֻּמָּא דְּבֵירָא, לְאַתְרַהּ. 
כט,ד וַיֹּ֤אמֶר לָהֶם֙ יַֽעֲקֹ֔ב אַחַ֖י מֵאַ֣יִן אַתֶּ֑ם וַיֹּ֣אמְר֔וּ מֵֽחָרָ֖ן אֲנָֽחְנוּ׃ וַאֲמַר לְהוֹן יַעֲקוֹב, אַחַי מְנָן אַתּוּן; וַאֲמַרוּ, מֵחָרָן אֲנַחְנָא. 
כט,ה וַיֹּ֣אמֶר לָהֶ֔ם הַיְדַעְתֶּ֖ם אֶת־לָבָ֣ן בֶּן־נָח֑וֹר וַיֹּֽאמְר֖וּ יָדָֽעְנוּ׃ וַאֲמַר לְהוֹן, הַיְדַעְתּוּן יָת לָבָן בַּר נָחוֹר; וַאֲמַרוּ, יָדְעִין. 
כט,ו וַיֹּ֥אמֶר לָהֶ֖ם הֲשָׁל֣וֹם ל֑וֹ וַיֹּֽאמְר֣וּ שָׁל֔וֹם וְהִנֵּה֙ רָחֵ֣ל בִּתּ֔וֹ בָּאָ֖ה עִם־הַצֹּֽאן׃ וַאֲמַר לְהוֹן, הַשְׁלָם לֵיהּ; וַאֲמַרוּ שְׁלָם--וְהָא רָחֵל בְּרַתֵּיהּ, אָתְיָא עִם עָנָא. 
כט,ז וַיֹּ֗אמֶר הֵ֥ן עוֹד֙ הַיּ֣וֹם גָּד֔וֹל לֹא־עֵ֖ת הֵֽאָסֵ֣ף הַמִּקְנֶ֑ה הַשְׁק֥וּ הַצֹּ֖אן וּלְכ֥וּ רְעֽוּ׃ וַאֲמַר, הָא עוֹד יוֹמָא סַגִּי--לָא עִדָּן, לְמִכְנַשׁ בְּעִיר; אַשְׁקוֹ עָנָא, וְאִיזִילוּ רְעוֹ. 
כט,ח וַיֹּֽאמְרוּ֮ לֹ֣א נוּכַל֒ עַ֣ד אֲשֶׁ֤ר יֵאָֽסְפוּ֙ כָּל־הָ֣עֲדָרִ֔ים וְגָֽלְלוּ֙ אֶת־הָאֶ֔בֶן מֵעַ֖ל פִּ֣י הַבְּאֵ֑ר וְהִשְׁקִ֖ינוּ הַצֹּֽאן׃ וַאֲמַרוּ, לָא נִכּוֹל, עַד דְּיִתְכַּנְשׁוּן כָּל עֶדְרַיָּא, וִיגַנְדְּרוּן יָת אַבְנָא מֵעַל פֻּמָּא דְּבֵירָא; וְנַשְׁקֵי, עָנָא. 
כט,ט עוֹדֶ֖נּוּ מְדַבֵּ֣ר עִמָּ֑ם וְרָחֵ֣ל ׀ בָּ֗אָה עִם־הַצֹּאן֙ אֲשֶׁ֣ר לְאָבִ֔יהָ כִּ֥י רֹעָ֖ה הִֽוא׃ עַד דְּהוּא, מְמַלֵּיל עִמְּהוֹן; וְרָחֵל אֲתָת, עִם עָנָא דִּלְאֲבוּהָא--אֲרֵי רָעִיתָא, הִיא. 
כט,י וַיְהִ֡י כַּֽאֲשֶׁר֩ רָאָ֨ה יַֽעֲקֹ֜ב אֶת־רָחֵ֗ל בַּת־לָבָן֙ אֲחִ֣י אִמּ֔וֹ וְאֶת־צֹ֥אן לָבָ֖ן אֲחִ֣י אִמּ֑וֹ וַיִּגַּ֣שׁ יַֽעֲקֹ֗ב וַיָּ֤גֶל אֶת־הָאֶ֨בֶן֙ מֵעַל֙ פִּ֣י הַבְּאֵ֔ר וַיַּ֕שְׁקְ אֶת־צֹ֥אן לָבָ֖ן אֲחִ֥י אִמּֽוֹ׃ וַהֲוָה כַּד חֲזָא יַעֲקוֹב יָת רָחֵל, בַּת לָבָן אֲחוּהָא דְּאִמֵּיהּ, וְיָת עָנָא דְּלָבָן, אֲחוּהָא דְּאִמֵּיהּ; וּקְרֵיב יַעֲקוֹב, וְגַנְדַּר יָת אַבְנָא מֵעַל פֻּמָּא דְּבֵירָא, וְאַשְׁקִי, יָת עָנָא דְּלָבָן אֲחוּהָא דְּאִמֵּיהּ. 
כט,יא וַיִּשַּׁ֥ק יַֽעֲקֹ֖ב לְרָחֵ֑ל וַיִּשָּׂ֥א אֶת־קֹל֖וֹ וַיֵּֽבְךְּ׃ וְנַשֵּׁיק יַעֲקוֹב, לְרָחֵל; וַאֲרֵים יָת קָלֵיהּ, וּבְכָא. 
כט,יב וַיַּגֵּ֨ד יַֽעֲקֹ֜ב לְרָחֵ֗ל כִּ֣י אֲחִ֤י אָבִ֨יהָ֙ ה֔וּא וְכִ֥י בֶן־רִבְקָ֖ה ה֑וּא וַתָּ֖רָץ וַתַּגֵּ֥ד לְאָבִֽיהָ׃ וְחַוִּי יַעֲקוֹב לְרָחֵל, אֲרֵי בַּר אֲחָת אֲבוּהָא הוּא, וַאֲרֵי בַּר רִבְקָה, הוּא; וּרְהַטַת, וְחַוִּיאַת לַאֲבוּהָא. 
כט,יג וַיְהִי֩ כִשְׁמֹ֨עַ לָבָ֜ן אֶת־שֵׁ֣מַע ׀ יַֽעֲקֹ֣ב בֶּן־אֲחֹת֗וֹ וַיָּ֤רָץ לִקְרָאתוֹ֙ וַיְחַבֶּק־לוֹ֙ וַיְנַשֶּׁק־ל֔וֹ וַיְבִיאֵ֖הוּ אֶל־בֵּית֑וֹ וַיְסַפֵּ֣ר לְלָבָ֔ן אֵ֥ת כָּל־הַדְּבָרִ֖ים הָאֵֽלֶּה׃ וַהֲוָה כַּד שְׁמַע לָבָן יָת שֵׁימַע יַעֲקוֹב בַּר אֲחָתֵיהּ, וּרְהַט לְקַדָּמוּתֵיהּ וְגָפֵיף לֵיהּ וְנַשֵּׁיק לֵיהּ, וְאַעֲלֵיהּ, לְבֵיתֵיהּ; וְאִשְׁתַּעִי לְלָבָן, יָת כָּל פִּתְגָמַיָּא הָאִלֵּין. 
כט,יד וַיֹּ֤אמֶר לוֹ֙ לָבָ֔ן אַ֛ךְ עַצְמִ֥י וּבְשָׂרִ֖י אָ֑תָּה וַיֵּ֥שֶׁב עִמּ֖וֹ חֹ֥דֶשׁ יָמִֽים׃ וַאֲמַר לֵיהּ לָבָן, בְּרַם קָרִיבִי וּבִסְרִי אַתְּ; וִיתֵיב עִמֵּיהּ, יְרַח יוֹמִין. 
כט,טו וַיֹּ֤אמֶר לָבָן֙ לְיַֽעֲקֹ֔ב הֲכִֽי־אָחִ֣י אַ֔תָּה וַֽעֲבַדְתַּ֖נִי חִנָּ֑ם הַגִּ֥ידָה לִּ֖י מַה־מַּשְׂכֻּרְתֶּֽךָ׃ וַאֲמַר לָבָן, לְיַעֲקוֹב, הֲמִדַּאֲחִי אַתְּ, וְתִפְלְחִנַּנִי מַגָּן; חַו לִי, מָא אַגְרָךְ. 
כט,טז וּלְלָבָ֖ן שְׁתֵּ֣י בָנ֑וֹת שֵׁ֤ם הַגְּדֹלָה֙ לֵאָ֔ה וְשֵׁ֥ם הַקְּטַנָּ֖ה רָחֵֽל׃ וּלְלָבָן, תַּרְתֵּין בְּנָן:  שׁוֹם רַבְּתָא לֵאָה, וְשׁוֹם זְעֵירְתָא רָחֵל. 
כט,יז וְעֵינֵ֥י לֵאָ֖ה רַכּ֑וֹת וְרָחֵל֙ הָֽיְתָ֔ה יְפַת־תֹּ֖אַר וִיפַ֥ת מַרְאֶֽה׃ וְעֵינֵי לֵאָה, יָאֲיָן; וְרָחֵל, הֲוָת, שַׁפִּירָא בְּרֵיוָא, וְיָאֲיָא בְּחֶזְוָא. 
כט,יח וַיֶּֽאֱהַ֥ב יַֽעֲקֹ֖ב אֶת־רָחֵ֑ל וַיֹּ֗אמֶר אֶֽעֱבָדְךָ֙ שֶׁ֣בַע שָׁנִ֔ים בְּרָחֵ֥ל בִּתְּךָ֖ הַקְּטַנָּֽה׃ וּרְחֵים יַעֲקוֹב, יָת רָחֵל; וַאֲמַר, אֶפְלְחִנָּךְ שְׁבַע שְׁנִין, בְּרָחֵל בְּרַתָּךְ, זְעֵירְתָא. 
כט,יט וַיֹּ֣אמֶר לָבָ֗ן ט֚וֹב תִּתִּ֣י אֹתָ֣הּ לָ֔ךְ מִתִּתִּ֥י אֹתָ֖הּ לְאִ֣ישׁ אַחֵ֑ר שְׁבָ֖ה עִמָּדִֽי׃ וַאֲמַר לָבָן, טָב דְּאֶתֵּין יָתַהּ לָךְ, מִדְּאֶתֵּין יָתַהּ, לִגְבַר אָחֳרָן; תִּיב, עִמִּי. 
כט,כ וַיַּֽעֲבֹ֧ד יַֽעֲקֹ֛ב בְּרָחֵ֖ל שֶׁ֣בַע שָׁנִ֑ים וַיִּֽהְי֤וּ בְעֵינָיו֙ כְּיָמִ֣ים אֲחָדִ֔ים בְּאַֽהֲבָת֖וֹ אֹתָֽהּ׃ וּפְלַח יַעֲקוֹב בְּרָחֵל, שְׁבַע שְׁנִין; וַהֲווֹ בְּעֵינוֹהִי כְּיוֹמִין זְעֵירִין, בִּדְרָחֵים יָתַהּ. 
כט,כא וַיֹּ֨אמֶר יַֽעֲקֹ֤ב אֶל־לָבָן֙ הָבָ֣ה אֶת־אִשְׁתִּ֔י כִּ֥י מָֽלְא֖וּ יָמָ֑י וְאָב֖וֹאָה אֵלֶֽיהָ׃ וַאֲמַר יַעֲקוֹב לְלָבָן הַב יָת אִתְּתִי, אֲרֵי שְׁלִימוּ יוֹמֵי פֻּלְחָנִי; וְאֵיעוֹל, לְוָתַהּ. 
כט,כב וַיֶּֽאֱסֹ֥ף לָבָ֛ן אֶת־כָּל־אַנְשֵׁ֥י הַמָּק֖וֹם וַיַּ֥עַשׂ מִשְׁתֶּֽה׃ וּכְנַשׁ לָבָן יָת כָּל אֲנָשֵׁי אַתְרָא, וַעֲבַד מִשְׁתְּיָא. 
כט,כג וַיְהִ֣י בָעֶ֔רֶב וַיִּקַּח֙ אֶת־לֵאָ֣ה בִתּ֔וֹ וַיָּבֵ֥א אֹתָ֖הּ אֵלָ֑יו וַיָּבֹ֖א אֵלֶֽיהָ׃ וַהֲוָה בְּרַמְשָׁא--וּדְבַר יָת לֵאָה בְּרַתֵּיהּ, וְאַעֵיל יָתַהּ לְוָתֵיהּ; וְעָאל, לְוָתַהּ. 
כט,כד וַיִּתֵּ֤ן לָבָן֙ לָ֔הּ אֶת־זִלְפָּ֖ה שִׁפְחָת֑וֹ לְלֵאָ֥ה בִתּ֖וֹ שִׁפְחָֽה׃ וִיהַב לָבָן לַהּ, יָת זִלְפָּה אַמְתֵּיהּ--לְלֵאָה בְּרַתֵּיהּ, לְאַמְהוּ. 
כט,כה וַיְהִ֣י בַבֹּ֔קֶר וְהִנֵּה־הִ֖וא לֵאָ֑ה וַיֹּ֣אמֶר אֶל־לָבָ֗ן מַה־זֹּאת֙ עָשִׂ֣יתָ לִּ֔י הֲלֹ֤א בְרָחֵל֙ עָבַ֣דְתִּי עִמָּ֔ךְ וְלָ֖מָּה רִמִּיתָֽנִי׃ וַהֲוָה בְּצַפְרָא, וְהָא הִיא לֵאָה; וַאֲמַר לְלָבָן, מָא דָּא עֲבַדְתְּ לִי--הֲלָא בְּרָחֵל פְּלַחִית עִמָּךְ, וּלְמָא שַׁקַּרְתְּ בִּי. 
כט,כו וַיֹּ֣אמֶר לָבָ֔ן לֹֽא־יֵעָשֶׂ֥ה כֵ֖ן בִּמְקוֹמֵ֑נוּ לָתֵ֥ת הַצְּעִירָ֖ה לִפְנֵ֥י הַבְּכִירָֽה׃ וַאֲמַר לָבָן, לָא מִתְעֲבֵיד כֵּין בְּאַתְרַנָא--לְמִתַּן זְעֵירְתָא, קֳדָם רַבְּתָא. 
כט,כז מַלֵּ֖א שְׁבֻ֣עַ זֹ֑את וְנִתְּנָ֨ה לְךָ֜ גַּם־אֶת־זֹ֗את בַּֽעֲבֹדָה֙ אֲשֶׁ֣ר תַּֽעֲבֹ֣ד עִמָּדִ֔י ע֖וֹד שֶֽׁבַע־שָׁנִ֥ים אֲחֵרֽוֹת׃ אַשְׁלֵים, שְׁבוּעֲתָא דְּדָא; וְנִתֵּין לָךְ אַף יָת דָּא, בְּפֻלְחָנָא דְּתִפְלַח עִמִּי, עוֹד, שְׁבַע שְׁנִין אָחֳרָנְיָן. 
כט,כח וַיַּ֤עַשׂ יַֽעֲקֹב֙ כֵּ֔ן וַיְמַלֵּ֖א שְׁבֻ֣עַ זֹ֑את וַיִּתֶּן־ל֛וֹ אֶת־רָחֵ֥ל בִּתּ֖וֹ ל֥וֹ לְאִשָּֽׁה׃ וַעֲבַד יַעֲקוֹב כֵּין, וְאַשְׁלֵים שְׁבוּעֲתָא דְּדָא; וִיהַב לֵיהּ יָת רָחֵל בְּרַתֵּיהּ, לֵיהּ לְאִתּוּ. 
כט,כט וַיִּתֵּ֤ן לָבָן֙ לְרָחֵ֣ל בִּתּ֔וֹ אֶת־בִּלְהָ֖ה שִׁפְחָת֑וֹ לָ֖הּ לְשִׁפְחָֽה׃ וִיהַב לָבָן לְרָחֵל בְּרַתֵּיהּ, יָת בִּלְהָה אַמְתֵּיהּ--לַהּ, לְאַמְהוּ. 
כט,ל וַיָּבֹא֙ גַּ֣ם אֶל־רָחֵ֔ל וַיֶּֽאֱהַ֥ב גַּֽם־אֶת־רָחֵ֖ל מִלֵּאָ֑ה וַיַּֽעֲבֹ֣ד עִמּ֔וֹ ע֖וֹד שֶֽׁבַע־שָׁנִ֥ים אֲחֵרֽוֹת׃ וְעָאל אַף לְוָת רָחֵל, וּרְחֵים אַף יָת רָחֵל מִלֵּאָה; וּפְלַח עִמֵּיהּ, עוֹד שְׁבַע שְׁנִין אָחֳרָנְיָן. 
כט,לא וַיַּ֤רְא יְהוָה֙ כִּֽי־שְׂנוּאָ֣ה לֵאָ֔ה וַיִּפְתַּ֖ח אֶת־רַחְמָ֑הּ וְרָחֵ֖ל עֲקָרָֽה׃ וַחֲזָא יְיָ אֲרֵי סְנוּאֲתָא לֵאָה, וִיהַב לַהּ עִדּוּי; וְרָחֵל, עַקְרָא. 
כט,לב וַתַּ֤הַר לֵאָה֙ וַתֵּ֣לֶד בֵּ֔ן וַתִּקְרָ֥א שְׁמ֖וֹ רְאוּבֵ֑ן כִּ֣י אָֽמְרָ֗ה כִּֽי־רָאָ֤ה יְהוָה֙ בְּעָנְיִ֔י כִּ֥י עַתָּ֖ה יֶֽאֱהָבַ֥נִי אִישִֽׁי׃ וְעַדִּיאַת לֵאָה וִילֵידַת בַּר, וּקְרָת שְׁמֵיהּ רְאוּבֵן:  אֲרֵי אֲמַרַת, אֲרֵי גְּלֵי קֳדָם יְיָ עֻלְבָּנִי--אֲרֵי כְּעַן, יִרְחֲמִנַּנִי בַּעְלִי. 
כט,לג וַתַּ֣הַר עוֹד֮ וַתֵּ֣לֶד בֵּן֒ וַתֹּ֗אמֶר כִּֽי־שָׁמַ֤ע יְהוָה֙ כִּֽי־שְׂנוּאָ֣ה אָנֹ֔כִי וַיִּתֶּן־לִ֖י גַּם־אֶת־זֶ֑ה וַתִּקְרָ֥א שְׁמ֖וֹ שִׁמְעֽוֹן׃ וְעַדִּיאַת עוֹד, וִילֵידַת בַּר, וַאֲמַרַת אֲרֵי שְׁמִיעַ קֳדָם יְיָ אֲרֵי סְנוּאֲתָא אֲנָא, וִיהַב לִי אַף יָת דֵּין; וּקְרָת שְׁמֵיהּ, שִׁמְעוֹן. 
כט,לד וַתַּ֣הַר עוֹד֮ וַתֵּ֣לֶד בֵּן֒ וַתֹּ֗אמֶר עַתָּ֤ה הַפַּ֨עַם֙ יִלָּוֶ֤ה אִישִׁי֙ אֵלַ֔י כִּֽי־יָלַ֥דְתִּי ל֖וֹ שְׁלֹשָׁ֣ה בָנִ֑ים עַל־כֵּ֥ן קָרָֽא־שְׁמ֖וֹ לֵוִֽי׃ וְעַדִּיאַת עוֹד, וִילֵידַת בַּר, וַאֲמַרַת הָדָא זִמְנָא יִתְחַבַּר לִי בַּעְלִי, אֲרֵי יְלֵידִית לֵיהּ תְּלָתָא בְּנִין; עַל כֵּין קְרָא שְׁמֵיהּ, לֵוִי. 
כט,לה וַתַּ֨הַר ע֜וֹד וַתֵּ֣לֶד בֵּ֗ן וַתֹּ֨אמֶר֙ הַפַּ֨עַם֙ אוֹדֶ֣ה אֶת־יְהוָ֔ה עַל־כֵּ֛ן קָֽרְאָ֥ה שְׁמ֖וֹ יְהוּדָ֑ה וַֽתַּעֲמֹ֖ד מִלֶּֽדֶת׃ וְעַדִּיאַת עוֹד וִילֵידַת בַּר, וַאֲמַרַת הָדָא זִמְנָא אוֹדֵי קֳדָם יְיָ--עַל כֵּין קְרָת שְׁמֵיהּ, יְהוּדָה; וְקַמַת, מִלְּמֵילַד. 
ל,א וַתֵּ֣רֶא רָחֵ֗ל כִּ֣י לֹ֤א יָֽלְדָה֙ לְיַֽעֲקֹ֔ב וַתְּקַנֵּ֥א רָחֵ֖ל בַּֽאֲחֹתָ֑הּ וַתֹּ֤אמֶר אֶֽל־יַעֲקֹב֙ הָֽבָה־לִּ֣י בָנִ֔ים וְאִם־אַ֖יִן מֵתָ֥ה אָנֹֽכִי׃ וַחֲזָת רָחֵל, אֲרֵי לָא יְלֵידַת לְיַעֲקוֹב, וְקַנִּיאַת רָחֵל, בַּאֲחָתַהּ; וַאֲמַרַת לְיַעֲקוֹב הַב לִי בְּנִין, וְאִם לָא מָיְתָא אֲנָא. 
ל,ב וַיִּֽחַר־אַ֥ף יַֽעֲקֹ֖ב בְּרָחֵ֑ל וַיֹּ֗אמֶר הֲתַ֤חַת אֱלֹהִים֙ אָנֹ֔כִי אֲשֶׁר־מָנַ֥ע מִמֵּ֖ךְ פְּרִי־בָֽטֶן׃ וּתְקֵיף רֻגְזָא דְּיַעֲקוֹב, בְּרָחֵל; וַאֲמַר, הֲמִנִּי אַתְּ בָּעְיָא הֲלָא מִן קֳדָם יְיָ תִּבְעַן, דִּמְנַע מִנִּיךְ, וַלְדָּא דִּמְעֵי. 
ל,ג וַתֹּ֕אמֶר הִנֵּ֛ה אֲמָתִ֥י בִלְהָ֖ה בֹּ֣א אֵלֶ֑יהָ וְתֵלֵד֙ עַל־בִּרְכַּ֔י וְאִבָּנֶ֥ה גַם־אָֽנֹכִ֖י מִמֶּֽנָּה׃ וַאֲמַרַת, הָא אַמְתִּי בִּלְהָה עוֹל לְוָתַהּ; תְּלִיד, וַאֲנָא אֲרַבֵּי, וְאֶתְבְּנֵי אַף אֲנָא, מִנַּהּ. 
ל,ד וַתִּתֶּן־ל֛וֹ אֶת־בִּלְהָ֥ה שִׁפְחָתָ֖הּ לְאִשָּׁ֑ה וַיָּבֹ֥א אֵלֶ֖יהָ יַֽעֲקֹֽב׃ וִיהַבַת לֵיהּ יָת בִּלְהָה אַמְתַּהּ, לְאִתּוּ; וְעָאל לְוָתַהּ, יַעֲקוֹב. 
ל,ה וַתַּ֣הַר בִּלְהָ֔ה וַתֵּ֥לֶד לְיַֽעֲקֹ֖ב בֵּֽן׃ וְעַדִּיאַת בִּלְהָה, וִילֵידַת לְיַעֲקוֹב בַּר. 
ל,ו וַתֹּ֤אמֶר רָחֵל֙ דָּנַ֣נִּי אֱלֹהִ֔ים וְגַם֙ שָׁמַ֣ע בְּקֹלִ֔י וַיִּתֶּן־לִ֖י בֵּ֑ן עַל־כֵּ֛ן קָֽרְאָ֥ה שְׁמ֖וֹ דָּֽן׃ וַאֲמַרַת רָחֵל, דָּנַנִּי יְיָ, וְאַף קַבֵּיל צְלוֹתִי, וִיהַב לִי בַּר; עַל כֵּין קְרָת שְׁמֵיהּ, דָּן. 
ל,ז וַתַּ֣הַר ע֔וֹד וַתֵּ֕לֶד בִּלְהָ֖ה שִׁפְחַ֣ת רָחֵ֑ל בֵּ֥ן שֵׁנִ֖י לְיַֽעֲקֹֽב׃ וְעַדִּיאַת עוֹד--וִילֵידַת, בִּלְהָה אַמְתַּהּ דְּרָחֵל:  בַּר תִּנְיָן, לְיַעֲקוֹב. 
ל,ח וַתֹּ֣אמֶר רָחֵ֗ל נַפְתּוּלֵ֨י אֱלֹהִ֧ים ׀ נִפְתַּ֛לְתִּי עִם־אֲחֹתִ֖י גַּם־יָכֹ֑לְתִּי וַתִּקְרָ֥א שְׁמ֖וֹ נַפְתָּלִֽי׃ וַאֲמַרַת רָחֵל, קַבֵּיל בָּעוּתִי יְיָ בְּאִתְחַנָּנוּתִי בִּצְלוֹתִי חַמֵּידִית דִּיהֵי לִי וְלַד כַּאֲחָתִי--אַף אִתְיְהֵיב לִי; וּקְרָת שְׁמֵיהּ, נַפְתָּלִי. 
ל,ט וַתֵּ֣רֶא לֵאָ֔ה כִּ֥י עָֽמְדָ֖ה מִלֶּ֑דֶת וַתִּקַּח֙ אֶת־זִלְפָּ֣ה שִׁפְחָתָ֔הּ וַתִּתֵּ֥ן אֹתָ֛הּ לְיַֽעֲקֹ֖ב לְאִשָּֽׁה׃ וַחֲזָת לֵאָה, אֲרֵי קַמַת מִלְּמֵילַד; וּדְבַרַת יָת זִלְפָּה אַמְתַּהּ, וִיהַבַת יָתַהּ לְיַעֲקוֹב לְאִתּוּ. 
ל,י וַתֵּ֗לֶד זִלְפָּ֛ה שִׁפְחַ֥ת לֵאָ֖ה לְיַֽעֲקֹ֥ב בֵּֽן׃ וִילֵידַת, זִלְפָּה אַמְתַּהּ דְּלֵאָה--לְיַעֲקוֹב בַּר. 
ל,יא וַתֹּ֥אמֶר לֵאָ֖ה בגד (בָּ֣א גָ֑ד) וַתִּקְרָ֥א אֶת־שְׁמ֖וֹ גָּֽד׃ וַאֲמַרַת לֵאָה, אֲתָא גָּד; וּקְרָת יָת שְׁמֵיהּ, גָּד. 
ל,יב וַתֵּ֗לֶד זִלְפָּה֙ שִׁפְחַ֣ת לֵאָ֔ה בֵּ֥ן שֵׁנִ֖י לְיַֽעֲקֹֽב׃ וִילֵידַת, זִלְפָּה אַמְתַּהּ דְּלֵאָה, בַּר תִּנְיָן, לְיַעֲקוֹב. 
ל,יג וַתֹּ֣אמֶר לֵאָ֔ה בְּאָשְׁרִ֕י כִּ֥י אִשְּׁר֖וּנִי בָּנ֑וֹת וַתִּקְרָ֥א אֶת־שְׁמ֖וֹ אָשֵֽׁר׃ וַאֲמַרַת לֵאָה--תֻּשְׁבְּחָא הֲוָת לִי, אֲרֵי בְּכֵין יְשַׁבְּחוּנַּנִי נְשַׁיָּא; וּקְרָת יָת שְׁמֵיהּ, אָשֵׁר. 
ל,יד וַיֵּ֨לֶךְ רְאוּבֵ֜ן בִּימֵ֣י קְצִיר־חִטִּ֗ים וַיִּמְצָ֤א דֽוּדָאִים֙ בַּשָּׂדֶ֔ה וַיָּבֵ֣א אֹתָ֔ם אֶל־לֵאָ֖ה אִמּ֑וֹ וַתֹּ֤אמֶר רָחֵל֙ אֶל־לֵאָ֔ה תְּנִי־נָ֣א לִ֔י מִדּֽוּדָאֵ֖י בְּנֵֽךְ׃ וַאֲזַל רְאוּבֵן בְּיוֹמֵי חֲצַד חִטִּין, וְאַשְׁכַּח יַבְרוּחִין בְּחַקְלָא, וְאַיְתִי יָתְהוֹן, לְלֵאָה אִמֵּיהּ; וַאֲמַרַת רָחֵל, לְלֵאָה, הַבִי כְּעַן לִי, מִיַּבְרוּחֵי דִּבְרִיךְ. 
ל,טו וַתֹּ֣אמֶר לָ֗הּ הַמְעַט֙ קַחְתֵּ֣ךְ אֶת־אִישִׁ֔י וְלָקַ֕חַת גַּ֥ם אֶת־דּֽוּדָאֵ֖י בְּנִ֑י וַתֹּ֣אמֶר רָחֵ֗ל לָכֵן֙ יִשְׁכַּ֤ב עִמָּךְ֙ הַלַּ֔יְלָה תַּ֖חַת דּֽוּדָאֵ֥י בְנֵֽךְ׃ וַאֲמַרַת לַהּ, הַזְעֵיר דִּדְבַרְתְּ יָת בַּעְלִי, וְתִסְּבִין, אַף יָת יַבְרוּחֵי דִּבְרִי; וַאֲמַרַת רָחֵל, בְּכֵין יִשְׁכּוֹב עִמִּיךְ בְּלֵילְיָא, חֲלָף, יַבְרוּחֵי דִּבְרִיךְ. 
ל,טז וַיָּבֹ֨א יַֽעֲקֹ֣ב מִן־הַשָּׂדֶה֮ בָּעֶרֶב֒ וַתֵּצֵ֨א לֵאָ֜ה לִקְרָאת֗וֹ וַתֹּ֨אמֶר֙ אֵלַ֣י תָּב֔וֹא כִּ֚י שָׂכֹ֣ר שְׂכַרְתִּ֔יךָ בְּדֽוּדָאֵ֖י בְּנִ֑י וַיִּשְׁכַּ֥ב עִמָּ֖הּ בַּלַּ֥יְלָה הֽוּא׃ וְעָאל יַעֲקוֹב מִן חַקְלָא, בְּרַמְשָׁא, וּנְפַקַת לֵאָה לְקַדָּמוּתֵיהּ וַאֲמַרַת לְוָתִי תֵּיעוֹל, אֲרֵי מֵיגָר אֲגַרְתָּךְ בְּיַבְרוּחֵי דִּבְרִי; וּשְׁכֵיב עִמַּהּ, בְּלֵילְיָא הוּא. 
ל,יז וַיִּשְׁמַ֥ע אֱלֹהִ֖ים אֶל־לֵאָ֑ה וַתַּ֛הַר וַתֵּ֥לֶד לְיַֽעֲקֹ֖ב בֵּ֥ן חֲמִישִֽׁי׃ וְקַבֵּיל יְיָ, צְלוֹתַהּ דְּלֵאָה; וְעַדִּיאַת וִילֵידַת לְיַעֲקוֹב, בַּר חֲמִישָׁאי. 
ל,יח וַתֹּ֣אמֶר לֵאָ֗ה נָתַ֤ן אֱלֹהִים֙ שְׂכָרִ֔י אֲשֶׁר־נָתַ֥תִּי שִׁפְחָתִ֖י לְאִישִׁ֑י וַתִּקְרָ֥א שְׁמ֖וֹ יִשָּׂשכָֽר׃ וַאֲמַרַת לֵאָה, יְהַב יְיָ אַגְרִי, דִּיהַבִית אַמְתִּי, לְבַעְלִי; וּקְרָת שְׁמֵיהּ, יִשָּׂשכָר. 
ל,יט וַתַּ֤הַר עוֹד֙ לֵאָ֔ה וַתֵּ֥לֶד בֵּן־שִׁשִּׁ֖י לְיַֽעֲקֹֽב׃ וְעַדִּיאַת עוֹד לֵאָה, וִילֵידַת בַּר שְׁתִיתַאי לְיַעֲקוֹב. 
ל,כ וַתֹּ֣אמֶר לֵאָ֗ה זְבָדַ֨נִי אֱלֹהִ֥ים ׀ אֹתִי֮ זֵ֣בֶד טוֹב֒ הַפַּ֨עַם֙ יִזְבְּלֵ֣נִי אִישִׁ֔י כִּֽי־יָלַ֥דְתִּי ל֖וֹ שִׁשָּׁ֣ה בָנִ֑ים וַתִּקְרָ֥א אֶת־שְׁמ֖וֹ זְבֻלֽוּן׃ וַאֲמַרַת לֵאָה, יְהַב יְיָ יָתֵיהּ לִי חוּלָק טָב--הָדָא זִמְנָא יְהֵי מְדוֹרֵיהּ דְּבַעְלִי לְוָתִי, אֲרֵי יְלֵידִית לֵיהּ שִׁתָּא בְּנִין; וּקְרָת יָת שְׁמֵיהּ, זְבוּלוּן. 
ל,כא וְאַחַ֖ר יָ֣לְדָה בַּ֑ת וַתִּקְרָ֥א אֶת־שְׁמָ֖הּ דִּינָֽה׃ וּבָתַר כֵּין, יְלֵידַת בְּרַתָּא; וּקְרָת יָת שְׁמַהּ, דִּינָה. 
ל,כב וַיִּזְכֹּ֥ר אֱלֹהִ֖ים אֶת־רָחֵ֑ל וַיִּשְׁמַ֤ע אֵלֶ֨יהָ֙ אֱלֹהִ֔ים וַיִּפְתַּ֖ח אֶת־רַחְמָֽהּ׃ וְעָאל דֻּכְרָנַהּ דְּרָחֵל, קֳדָם יְיָ; וְקַבֵּיל צְלוֹתַהּ יְיָ, וִיהַב לַהּ עִדּוּי. 
ל,כג וַתַּ֖הַר וַתֵּ֣לֶד בֵּ֑ן וַתֹּ֕אמֶר אָסַ֥ף אֱלֹהִ֖ים אֶת־חֶרְפָּתִֽי׃ וְעַדִּיאַת, וִילֵידַת בַּר; וַאֲמַרַת, כְּנַשׁ יְיָ יָת חִסּוּדָי. 
ל,כד וַתִּקְרָ֧א אֶת־שְׁמ֛וֹ יוֹסֵ֖ף לֵאמֹ֑ר יֹסֵ֧ף יְהוָ֛ה לִ֖י בֵּ֥ן אַחֵֽר׃ וּקְרָת יָת שְׁמֵיהּ יוֹסֵף, לְמֵימַר:  יוֹסֵיף יְיָ לִי, בַּר אָחֳרָן. 
ל,כה וַיְהִ֕י כַּֽאֲשֶׁ֛ר יָֽלְדָ֥ה רָחֵ֖ל אֶת־יוֹסֵ֑ף וַיֹּ֤אמֶר יַֽעֲקֹב֙ אֶל־לָבָ֔ן שַׁלְּחֵ֨נִי֙ וְאֵ֣לְכָ֔ה אֶל־מְקוֹמִ֖י וּלְאַרְצִֽי׃ וַהֲוָה, כַּד יְלֵידַת רָחֵל יָת יוֹסֵף; וַאֲמַר יַעֲקוֹב, לְלָבָן, שַׁלְּחַנִי וַאֲהָךְ, לְאַתְרִי וּלְאַרְעִי. 
ל,כו תְּנָ֞ה אֶת־נָשַׁ֣י וְאֶת־יְלָדַ֗י אֲשֶׁ֨ר עָבַ֧דְתִּי אֹֽתְךָ֛ בָּהֵ֖ן וְאֵלֵ֑כָה כִּ֚י אַתָּ֣ה יָדַ֔עְתָּ אֶת־עֲבֹֽדָתִ֖י אֲשֶׁ֥ר עֲבַדְתִּֽיךָ׃ הַב יָת נְשַׁי וְיָת בְּנַי, דִּפְלַחִית יָתָךְ בְּהוֹן--וְאֵיזֵיל:  אֲרֵי אַתְּ יְדַעְתְּ, יָת פֻּלְחָנִי דִּפְלַחְתָּךְ. 
ל,כז וַיֹּ֤אמֶר אֵלָיו֙ לָבָ֔ן אִם־נָ֛א מָצָ֥אתִי חֵ֖ן בְּעֵינֶ֑יךָ נִחַ֕שְׁתִּי וַיְבָֽרְכֵ֥נִי יְהוָ֖ה בִּגְלָלֶֽךָ׃ וַאֲמַר לֵיהּ לָבָן, אִם כְּעַן אַשְׁכַּחִית רַחֲמִין בְּעֵינָךְ; נַסִּיתִי, וּבָרְכַנִי יְיָ בְּדִילָךְ. 
ל,כח וַיֹּאמַ֑ר נָקְבָ֧ה שְׂכָֽרְךָ֛ עָלַ֖י וְאֶתֵּֽנָה׃ וַאֲמַר:  פָּרֵישׁ אַגְרָךְ עֲלַי, וְאֶתֵּין. 
ל,כט וַיֹּ֣אמֶר אֵלָ֔יו אַתָּ֣ה יָדַ֔עְתָּ אֵ֖ת אֲשֶׁ֣ר עֲבַדְתִּ֑יךָ וְאֵ֛ת אֲשֶׁר־הָיָ֥ה מִקְנְךָ֖ אִתִּֽי׃ וַאֲמַר לֵיהּ--אַתְּ יְדַעְתְּ, יָת דִּפְלַחְתָּךְ; וְיָת דַּהֲוָה בְּעִירָךְ, עִמִּי. 
ל,ל כִּ֡י מְעַט֩ אֲשֶׁר־הָיָ֨ה לְךָ֤ לְפָנַי֙ וַיִּפְרֹ֣ץ לָרֹ֔ב וַיְבָ֧רֶךְ יְהוָ֛ה אֹֽתְךָ֖ לְרַגְלִ֑י וְעַתָּ֗ה מָתַ֛י אֶֽעֱשֶׂ֥ה גַם־אָֽנֹכִ֖י לְבֵיתִֽי׃ אֲרֵי זְעֵיר דַּהֲוָה לָךְ קֳדָמַי, וּתְקֵיף לִסְגֵּי, וּבָרֵיךְ יְיָ יָתָךְ, בְּדִילִי; וּכְעַן, אִמַּתִּי אַעֲבֵיד אַף אֲנָא--לְבֵיתִי. 
ל,לא וַיֹּ֖אמֶר מָ֣ה אֶתֶּן־לָ֑ךְ וַיֹּ֤אמֶר יַֽעֲקֹב֙ לֹֽא־תִתֶּן־לִ֣י מְא֔וּמָה אִם־תַּֽעֲשֶׂה־לִּי֙ הַדָּבָ֣ר הַזֶּ֔ה אָשׁ֛וּבָה אֶרְעֶ֥ה צֹֽאנְךָ֖ אֶשְׁמֹֽר׃ וַאֲמַר, מָא אֶתֵּין לָךְ; וַאֲמַר יַעֲקוֹב, לָא תִּתֵּין לִי מִדָּעַם--אִם תַּעֲבֵיד לִי פִּתְגָמָא הָדֵין, אֲתוּב אֶרְעֵי עָנָךְ אֶטַּר. 
ל,לב אֶֽעֱבֹ֨ר בְּכָל־צֹֽאנְךָ֜ הַיּ֗וֹם הָסֵ֨ר מִשָּׁ֜ם כָּל־שֶׂ֣ה ׀ נָקֹ֣ד וְטָל֗וּא וְכָל־שֶׂה־חוּם֙ בַּכְּשָׂבִ֔ים וְטָל֥וּא וְנָקֹ֖ד בָּֽעִזִּ֑ים וְהָיָ֖ה שְׂכָרִֽי׃ אֶעְבַּר בְּכָל עָנָךְ יוֹמָא דֵּין, אַעְדִּי מִתַּמָּן כָּל אִמַּר נְמוֹר וּרְקוֹעַ וְכָל אִמַּר שְׁחוֹם בְּאִמְּרַיָּא, וּרְקוֹעַ וּנְמוֹר, בְּעִזַּיָּא; וִיהֵי, אַגְרִי. 
ל,לג וְעָֽנְתָה־בִּ֤י צִדְקָתִי֙ בְּי֣וֹם מָחָ֔ר כִּֽי־תָב֥וֹא עַל־שְׂכָרִ֖י לְפָנֶ֑יךָ כֹּ֣ל אֲשֶׁר־אֵינֶנּוּ֩ נָקֹ֨ד וְטָל֜וּא בָּֽעִזִּ֗ים וְחוּם֙ בַּכְּשָׂבִ֔ים גָּנ֥וּב ה֖וּא אִתִּֽי׃ וְתַסְהֵיד עֲלַי זְכוּתִי בְּיוֹם מְחַר, אֲרֵי תֵּיעוֹל עַל אַגְרִי קֳדָמָךְ:  כֹּל דְּלָיְתוֹהִי נְמוֹר וּרְקוֹעַ בְּעִזַּיָּא, וּשְׁחוֹם בְּאִמְּרַיָּא--גָּנוֹבָא הוּא, עִמִּי. 
ל,לד וַיֹּ֥אמֶר לָבָ֖ן הֵ֑ן ל֖וּ יְהִ֥י כִדְבָרֶֽךָ׃ וַאֲמַר לָבָן, בְּרַם:  לְוֵי, יְהֵי כְּפִתְגָמָךְ. 
ל,לה וַיָּ֣סַר בַּיּוֹם֩ הַה֨וּא אֶת־הַתְּיָשִׁ֜ים הָֽעֲקֻדִּ֣ים וְהַטְּלֻאִ֗ים וְאֵ֤ת כָּל־הָֽעִזִּים֙ הַנְּקֻדּ֣וֹת וְהַטְּלֻאֹ֔ת כֹּ֤ל אֲשֶׁר־לָבָן֙ בּ֔וֹ וְכָל־ח֖וּם בַּכְּשָׂבִ֑ים וַיִּתֵּ֖ן בְּיַד־בָּנָֽיו׃ וְאַעְדִּי בְּיוֹמָא הַהוּא יָת תֵּישַׁיָּא רְגוֹלַיָּא וּרְקוֹעַיָּא, וְיָת כָּל עִזַּיָּא נְמוֹרָתָא וּרְקוֹעָתָא, כֹּל דְּחִיוָר בֵּיהּ, וְכָל דִּשְׁחוֹם בְּאִמְּרַיָּא; וִיהַב, בְּיַד בְּנוֹהִי. 
ל,לו וַיָּ֗שֶׂם דֶּ֚רֶךְ שְׁלֹ֣שֶׁת יָמִ֔ים בֵּינ֖וֹ וּבֵ֣ין יַֽעֲקֹ֑ב וְיַֽעֲקֹ֗ב רֹעֶ֛ה אֶת־צֹ֥אן לָבָ֖ן הַנּֽוֹתָרֹֽת׃ וְשַׁוִּי, מַהְלַךְ תְּלָתָא יוֹמִין, בֵּינוֹהִי, וּבֵין יַעֲקוֹב; וְיַעֲקוֹב, רָעֵי יָת עָנָא דְּלָבָן--דְּאִשְׁתְּאַרָא. 
ל,לז וַיִּֽקַּֽח־ל֣וֹ יַֽעֲקֹ֗ב מַקַּ֥ל לִבְנֶ֛ה לַ֖ח וְל֣וּז וְעַרְמ֑וֹן וַיְפַצֵּ֤ל בָּהֵן֙ פְּצָל֣וֹת לְבָנ֔וֹת מַחְשֹׂף֙ הַלָּבָ֔ן אֲשֶׁ֖ר עַל־הַמַּקְלֽוֹת׃ וּנְסֵיב לֵיהּ יַעֲקוֹב, חֻטְרִין דִּלְבַן רַטִּיבִין--וּדְלוּז וְדִדְלוּב; וְקַלֵּיף בְּהוֹן, קַלְפִין חִיוָרִין--קִלּוּף חִיוָר, דְּעַל חֻטְרַיָּא. 
ל,לח וַיַּצֵּ֗ג אֶת־הַמַּקְלוֹת֙ אֲשֶׁ֣ר פִּצֵּ֔ל בָּֽרְהָטִ֖ים בְּשִֽׁקְת֣וֹת הַמָּ֑יִם אֲשֶׁר֩ תָּבֹ֨אןָ הַצֹּ֤אן לִשְׁתּוֹת֙ לְנֹ֣כַח הַצֹּ֔אן וַיֵּחַ֖מְנָה בְּבֹאָ֥ן לִשְׁתּֽוֹת׃ וְדַעֵיץ, יָת חֻטְרַיָּא דְּקַלֵּיף, בְּרָטַיָּא, אֲתַר בֵּית שִׁקְיָא דְּמַיָּא--אֲתַר דְּאָתְיָן עָנָא לְמִשְׁתֵּי לְקִבְלְהוֹן דְּעָנָא, וּמִתְיַחֲמָן בְּמֵיתֵיהוֹן לְמִשְׁתֵּי. 
ל,לט וַיֶּֽחֱמ֥וּ הַצֹּ֖אן אֶל־הַמַּקְל֑וֹת וַתֵּלַ֣דְןָ הַצֹּ֔אן עֲקֻדִּ֥ים נְקֻדִּ֖ים וּטְלֻאִֽים׃ וְאִתְיַחֲמָא עָנָא, בְּחֻטְרַיָּא; וִילִידָן עָנָא, רְגוֹלִין נְמוֹרִין וּרְקוֹעִין. 
ל,מ וְהַכְּשָׂבִים֮ הִפְרִ֣יד יַֽעֲקֹב֒ וַ֠יִּתֵּן פְּנֵ֨י הַצֹּ֧אן אֶל־עָקֹ֛ד וְכָל־ח֖וּם בְּצֹ֣אן לָבָ֑ן וַיָּֽשֶׁת־ל֤וֹ עֲדָרִים֙ לְבַדּ֔וֹ וְלֹ֥א שָׁתָ֖ם עַל־צֹ֥אן לָבָֽן׃ וְאִמְּרַיָּא, אַפְרֵישׁ יַעֲקוֹב, וִיהַב בְּרֵישׁ עָנָא כָּל דִּרְגוֹל וְכָל דִּשְׁחוֹם, בְּעָנָא דְּלָבָן; וְשַׁוִּי לֵיהּ עֶדְרִין בִּלְחוֹדוֹהִי, וְלָא עָרֵיבִנּוּן עִם עָנָא דְּלָבָן. 
ל,מא וְהָיָ֗ה בְּכָל־יַחֵם֮ הַצֹּ֣אן הַמְקֻשָּׁרוֹת֒ וְשָׂ֨ם יַֽעֲקֹ֧ב אֶת־הַמַּקְל֛וֹת לְעֵינֵ֥י הַצֹּ֖אן בָּֽרְהָטִ֑ים לְיַחְמֵ֖נָּה בַּמַּקְלֽוֹת׃ וְהָוֵי, בְּכָל עִדָּן דְּמִתְיַחֲמָן עָנָא מְבַכְּרָתָא, וּמְשַׁוֵּי יַעֲקוֹב יָת חֻטְרַיָּא לְעֵינֵי עָנָא, בְּרָטַיָּא--לְיַחָמוּתְהוֹן, בְּחֻטְרַיָּא. 
ל,מב וּבְהַֽעֲטִ֥יף הַצֹּ֖אן לֹ֣א יָשִׂ֑ים וְהָיָ֤ה הָֽעֲטֻפִים֙ לְלָבָ֔ן וְהַקְּשֻׁרִ֖ים לְיַֽעֲקֹֽב׃ וּבְלַקִּישׁוּת עָנָא, לָא מְשַׁוֵּי; וְהָוַן לַקִּישַׁיָּא לְלָבָן, וּבַכִּירַיָּא לְיַעֲקוֹב. 
ל,מג וַיִּפְרֹ֥ץ הָאִ֖ישׁ מְאֹ֣ד מְאֹ֑ד וַֽיְהִי־לוֹ֙ צֹ֣אן רַבּ֔וֹת וּשְׁפָחוֹת֙ וַֽעֲבָדִ֔ים וּגְמַלִּ֖ים וַֽחֲמֹרִֽים׃ וּתְקֵיף גֻּבְרָא, לַחְדָּא לַחְדָּא; וַהֲווֹ לֵיהּ, עָן סַגִּיאָן, וְאַמְהָן וְעַבְדִּין, וְגַמְלִין וּחְמָרִין. 
לא,א וַיִּשְׁמַ֗ע אֶת־דִּבְרֵ֤י בְנֵֽי־לָבָן֙ לֵאמֹ֔ר לָקַ֣ח יַֽעֲקֹ֔ב אֵ֖ת כָּל־אֲשֶׁ֣ר לְאָבִ֑ינוּ וּמֵֽאֲשֶׁ֣ר לְאָבִ֔ינוּ עָשָׂ֕ה אֵ֥ת כָּל־הַכָּבֹ֖ד הַזֶּֽה׃ וּשְׁמַע, יָת פִּתְגָמֵי בְּנֵי לָבָן דְּאָמְרִין, נְסֵיב יַעֲקוֹב, יָת כָּל דִּלְאֲבוּנָא; וּמִדִּלְאֲבוּנָא--קְנָא, יָת כָּל נִכְסַיָּא הָאִלֵּין. 
לא,ב וַיַּ֥רְא יַֽעֲקֹ֖ב אֶת־פְּנֵ֣י לָבָ֑ן וְהִנֵּ֥ה אֵינֶנּ֛וּ עִמּ֖וֹ כִּתְמ֥וֹל שִׁלְשֽׁוֹם׃ וַחֲזָא יַעֲקוֹב, יָת סְבַר אַפֵּי לָבָן; וְהָא לֵיתִנּוּן עִמֵּיהּ, כְּמֵאֶתְמָלִי וּמִדְּקַמּוֹהִי. 
לא,ג וַיֹּ֤אמֶר יְהוָה֙ אֶֽל־יַעֲקֹ֔ב שׁ֛וּב אֶל־אֶ֥רֶץ אֲבוֹתֶ֖יךָ וּלְמֽוֹלַדְתֶּ֑ךָ וְאֶֽהְיֶ֖ה עִמָּֽךְ׃ וַאֲמַר יְיָ לְיַעֲקוֹב, תּוּב לַאֲרַע אֲבָהָתָךְ וּלְיַלָּדוּתָךְ; וִיהֵי מֵימְרִי, בְּסַעֲדָךְ. 
לא,ד וַיִּשְׁלַ֣ח יַֽעֲקֹ֔ב וַיִּקְרָ֖א לְרָחֵ֣ל וּלְלֵאָ֑ה הַשָּׂדֶ֖ה אֶל־צֹאנֽוֹ׃ וּשְׁלַח יַעֲקוֹב, וּקְרָא לְרָחֵל וּלְלֵאָה, לְחַקְלָא, לְוָת עָנֵיהּ. 
לא,ה וַיֹּ֣אמֶר לָהֶ֗ן רֹאֶ֤ה אָֽנֹכִי֙ אֶת־פְּנֵ֣י אֲבִיכֶ֔ן כִּֽי־אֵינֶ֥נּוּ אֵלַ֖י כִּתְמֹ֣ל שִׁלְשֹׁ֑ם וֵֽאלֹהֵ֣י אָבִ֔י הָיָ֖ה עִמָּדִֽי׃ וַאֲמַר לְהוֹן, חָזֵי אֲנָא יָת סְבַר אַפֵּי אֲבוּכוֹן, אֲרֵי לֵיתִנּוּן עִמִּי, כְּמֵאֶתְמָלִי וּמִדְּקַמּוֹהִי; וֵאלָהֵיהּ דְּאַבָּא, הֲוָה בְּסַעֲדִי. 
לא,ו וְאַתֵּ֖נָה יְדַעְתֶּ֑ן כִּ֚י בְּכָל־כֹּחִ֔י עָבַ֖דְתִּי אֶת־אֲבִיכֶֽן׃ וְאַתִּין, יְדַעְתִּין:  אֲרֵי, בְּכָל חֵילִי, פְּלַחִית, יָת אֲבוּכוֹן. 
לא,ז וַֽאֲבִיכֶן֙ הֵ֣תֶל בִּ֔י וְהֶֽחֱלִ֥ף אֶת־מַשְׂכֻּרְתִּ֖י עֲשֶׂ֣רֶת מֹנִ֑ים וְלֹֽא־נְתָנ֣וֹ אֱלֹהִ֔ים לְהָרַ֖ע עִמָּדִֽי׃ וַאֲבוּכוֹן שַׁקַּר בִּי, וְאַשְׁנִי יָת אַגְרִי עֲסַר זִמְנִין; וְלָא שַׁבְקֵיהּ יְיָ, לְאַבְאָשָׁא עִמִּי. 
לא,ח אִם־כֹּ֣ה יֹאמַ֗ר נְקֻדִּים֙ יִֽהְיֶ֣ה שְׂכָרֶ֔ךָ וְיָֽלְד֥וּ כָל־הַצֹּ֖אן נְקֻדִּ֑ים וְאִם־כֹּ֣ה יֹאמַ֗ר עֲקֻדִּים֙ יִֽהְיֶ֣ה שְׂכָרֶ֔ךָ וְיָֽלְד֥וּ כָל־הַצֹּ֖אן עֲקֻדִּֽים׃ אִם כְּדֵין הֲוָה אָמַר, נְמוֹרִין יְהֵי אַגְרָךְ--וִילִידָן כָּל עָנָא, נְמוֹרִין; וְאִם כְּדֵין הֲוָה אָמַר, רְגוֹלִין יְהֵי אַגְרָךְ--וִילִידָן כָּל עָנָא, רְגוֹלִין. 
לא,ט וַיַּצֵּ֧ל אֱלֹהִ֛ים אֶת־מִקְנֵ֥ה אֲבִיכֶ֖ם וַיִּתֶּן־לִֽי׃ וְאַפְרֵישׁ יְיָ מִן בְּעִירָא דַּאֲבוּכוֹן, וִיהַב לִי. 
לא,י וַיְהִ֗י בְּעֵת֙ יַחֵ֣ם הַצֹּ֔אן וָֽאֶשָּׂ֥א עֵינַ֛י וָאֵ֖רֶא בַּֽחֲל֑וֹם וְהִנֵּ֤ה הָֽעַתֻּדִים֙ הָֽעֹלִ֣ים עַל־הַצֹּ֔אן עֲקֻדִּ֥ים נְקֻדִּ֖ים וּבְרֻדִּֽים׃ וַהֲוָה, בְּעִדָּן דְּאִתְיַחֲמָא עָנָא, וּזְקַפִית עֵינַי וַחֲזֵית, בְּחֶלְמָא; וְהָא תֵּישַׁיָּא דְּסָלְקִין עַל עָנָא, רְגוֹלִין נְמוֹרִין וּפַצִּיחִין. 
לא,יא וַיֹּ֨אמֶר אֵלַ֜י מַלְאַ֧ךְ הָֽאֱלֹהִ֛ים בַּֽחֲל֖וֹם יַֽעֲקֹ֑ב וָֽאֹמַ֖ר הִנֵּֽנִי׃ וַאֲמַר לִי מַלְאֲכָא דַּייָ, בְּחֶלְמָא--יַעֲקוֹב; וַאֲמַרִית, הָאֲנָא. 
לא,יב וַיֹּ֗אמֶר שָׂא־נָ֨א עֵינֶ֤יךָ וּרְאֵה֙ כָּל־הָֽעַתֻּדִים֙ הָֽעֹלִ֣ים עַל־הַצֹּ֔אן עֲקֻדִּ֥ים נְקֻדִּ֖ים וּבְרֻדִּ֑ים כִּ֣י רָאִ֔יתִי אֵ֛ת כָּל־אֲשֶׁ֥ר לָבָ֖ן עֹ֥שֶׂה לָּֽךְ׃ וַאֲמַר, זְקוֹף כְּעַן עֵינָךְ וַחְזִי כָּל תֵּישַׁיָּא דְּסָלְקִין עַל עָנָא, רְגוֹלִין נְמוֹרִין, וּפַצִּיחִין:  אֲרֵי גְּלֵי קֳדָמַי, יָת כָּל דְּלָבָן עָבֵיד לָךְ. 
לא,יג אָֽנֹכִ֤י הָאֵל֙ בֵּֽית־אֵ֔ל אֲשֶׁ֨ר מָשַׁ֤חְתָּ שָּׁם֙ מַצֵּבָ֔ה אֲשֶׁ֨ר נָדַ֥רְתָּ לִּ֛י שָׁ֖ם נֶ֑דֶר עַתָּ֗ה ק֥וּם צֵא֙ מִן־הָאָ֣רֶץ הַזֹּ֔את וְשׁ֖וּב אֶל־אֶ֥רֶץ מֽוֹלַדְתֶּֽךָ׃ אֲנָא אֱלָהָא דְּאִתְגְּלִיתִי עֲלָךְ, בְּבֵית אֵל, דִּמְשַׁחְתָּא תַּמָּן קָמָא, דְּקַיֵּימְתָּא קֳדָמַי תַּמָּן קְיָם; כְּעַן, קוּם פּוֹק מִן אַרְעָא הָדָא, וְתוּב, לַאֲרַע יַלָּדוּתָךְ. 
לא,יד וַתַּ֤עַן רָחֵל֙ וְלֵאָ֔ה וַתֹּאמַ֖רְנָה ל֑וֹ הַע֥וֹד לָ֛נוּ חֵ֥לֶק וְנַֽחֲלָ֖ה בְּבֵ֥ית אָבִֽינוּ׃ וַאֲתֵיבַת רָחֵל וְלֵאָה, וַאֲמַרָא לֵיהּ:  הַעוֹד לַנָא חוּלָק וְאַחְסָנָא, בְּבֵית אֲבוּנָא. 
לא,טו הֲל֧וֹא נָכְרִיּ֛וֹת נֶחְשַׁ֥בְנוּ ל֖וֹ כִּ֣י מְכָרָ֑נוּ וַיֹּ֥אכַל גַּם־אָכ֖וֹל אֶת־כַּסְפֵּֽנוּ׃ הֲלָא נֻכְרָאָן אִתְחֲשַׁבְנָא לֵיהּ, אֲרֵי זַבְּנַנָא; וַאֲכַל אַף מֵיכָל, יָת כַּסְפַּנָא. 
לא,טז כִּ֣י כָל־הָעֹ֗שֶׁר אֲשֶׁ֨ר הִצִּ֤יל אֱלֹהִים֙ מֵֽאָבִ֔ינוּ לָ֥נוּ ה֖וּא וּלְבָנֵ֑ינוּ וְעַתָּ֗ה כֹּל֩ אֲשֶׁ֨ר אָמַ֧ר אֱלֹהִ֛ים אֵלֶ֖יךָ עֲשֵֽׂה׃ אֲרֵי כָּל עֻתְרָא, דְּאַפְרֵישׁ יְיָ מֵאֲבוּנָא--דִּילַנָא הוּא, וְדִבְנַנָא; וּכְעַן, כֹּל דַּאֲמַר יְיָ לָךְ--עֲבֵיד. 
לא,יז וַיָּ֖קָם יַֽעֲקֹ֑ב וַיִּשָּׂ֛א אֶת־בָּנָ֥יו וְאֶת־נָשָׁ֖יו עַל־הַגְּמַלִּֽים׃ וְקָם, יַעֲקוֹב; וּנְטַל יָת בְּנוֹהִי וְיָת נְשׁוֹהִי, עַל גַּמְלַיָּא. 
לא,יח וַיִּנְהַ֣ג אֶת־כָּל־מִקְנֵ֗הוּ וְאֶת־כָּל־רְכֻשׁוֹ֙ אֲשֶׁ֣ר רָכָ֔שׁ מִקְנֵה֙ קִנְיָנ֔וֹ אֲשֶׁ֥ר רָכַ֖שׁ בְּפַדַּ֣ן אֲרָ֑ם לָב֛וֹא אֶל־יִצְחָ֥ק אָבִ֖יו אַ֥רְצָה כְּנָֽעַן׃ וְדַבַּר יָת כָּל גֵּיתוֹהִי, וְיָת כָּל קִנְיָנֵיהּ דִּקְנָא--גֵּיתוֹהִי וְקִנְיָנֵיהּ, דִּקְנָא בְּפַדַּן אֲרָם:  לְמֵיתֵי לְוָת יִצְחָק אֲבוּהִי, לְאַרְעָא דִּכְנָעַן. 
לא,יט וְלָבָ֣ן הָלַ֔ךְ לִגְזֹ֖ז אֶת־צֹאנ֑וֹ וַתִּגְנֹ֣ב רָחֵ֔ל אֶת־הַתְּרָפִ֖ים אֲשֶׁ֥ר לְאָבִֽיהָ׃ וְלָבָן אֲזַל, לְמִגַּז יָת עָנֵיהּ; וּנְסֵיבַת רָחֵל, יָת צַלְמָנַיָּא דִּלְאֲבוּהָא. 
לא,כ וַיִּגְנֹ֣ב יַֽעֲקֹ֔ב אֶת־לֵ֥ב לָבָ֖ן הָֽאֲרַמִּ֑י עַל־בְּלִי֙ הִגִּ֣יד ל֔וֹ כִּ֥י בֹרֵ֖חַ הֽוּא׃ וְכַסִּי יַעֲקוֹב, מִן לָבָן אֲרַמָּאָה--עַל דְּלָא חַוִּי לֵיהּ, אֲרֵי אָזֵיל הוּא. 
לא,כא וַיִּבְרַ֥ח הוּא֙ וְכָל־אֲשֶׁר־ל֔וֹ וַיָּ֖קָם וַיַּֽעֲבֹ֣ר אֶת־הַנָּהָ֑ר וַיָּ֥שֶׂם אֶת־פָּנָ֖יו הַ֥ר הַגִּלְעָֽד׃ וַאֲזַל הוּא וְכָל דְּלֵיהּ, וְקָם וַעֲבַר יָת פְּרָת; וְשַׁוִּי יָת אַפּוֹהִי, לְטוּרָא דְּגִלְעָד. 
לא,כב וַיֻּגַּ֥ד לְלָבָ֖ן בַּיּ֣וֹם הַשְּׁלִישִׁ֑י כִּ֥י בָרַ֖ח יַֽעֲקֹֽב׃ וְאִתְחַוַּא לְלָבָן, בְּיוֹמָא תְּלִיתָאָה:  אֲרֵי אֲזַל, יַעֲקוֹב. 
לא,כג וַיִּקַּ֤ח אֶת־אֶחָיו֙ עִמּ֔וֹ וַיִּרְדֹּ֣ף אַֽחֲרָ֔יו דֶּ֖רֶךְ שִׁבְעַ֣ת יָמִ֑ים וַיַּדְבֵּ֥ק אֹת֖וֹ בְּהַ֥ר הַגִּלְעָֽד׃ וּדְבַר יָת אֲחוֹהִי, עִמֵּיהּ, וּרְדַף בָּתְרוֹהִי, מַהְלַךְ שִׁבְעָא יוֹמִין; וְאַדְבֵּיק יָתֵיהּ, בְּטוּרָא דְּגִלְעָד. 
לא,כד וַיָּבֹ֧א אֱלֹהִ֛ים אֶל־לָבָ֥ן הָֽאֲרַמִּ֖י בַּֽחֲלֹ֣ם הַלָּ֑יְלָה וַיֹּ֣אמֶר ל֗וֹ הִשָּׁ֧מֶר לְךָ֛ פֶּן־תְּדַבֵּ֥ר עִֽם־יַעֲקֹ֖ב מִטּ֥וֹב עַד־רָֽע׃ וַאֲתָא מֵימַר מִן קֳדָם יְיָ לְוָת לָבָן אֲרַמָּאָה, בְּחֶלְמָא דְּלֵילְיָא; וַאֲמַר לֵיהּ, אִסְתְּמַר לָךְ דִּלְמָא תְּמַלֵּיל עִם יַעֲקוֹב--מִטָּב עַד בִּישׁ. 
לא,כה וַיַּשֵּׂ֥ג לָבָ֖ן אֶֽת־יַעֲקֹ֑ב וְיַֽעֲקֹ֗ב תָּקַ֤ע אֶֽת־אָהֳלוֹ֙ בָּהָ֔ר וְלָבָ֛ן תָּקַ֥ע אֶת־אֶחָ֖יו בְּהַ֥ר הַגִּלְעָֽד׃ וְאַדְבֵּיק לָבָן, יָת יַעֲקוֹב; וְיַעֲקוֹב, פְּרַס יָת מַשְׁכְּנֵיהּ בְּטוּרָא, וְלָבָן אַשְׁרִי יָת אֲחוֹהִי, בְּטוּרָא דְּגִלְעָד. 
לא,כו וַיֹּ֤אמֶר לָבָן֙ לְיַֽעֲקֹ֔ב מֶ֣ה עָשִׂ֔יתָ וַתִּגְנֹ֖ב אֶת־לְבָבִ֑י וַתְּנַהֵג֙ אֶת־בְּנֹתַ֔י כִּשְׁבֻי֖וֹת חָֽרֶב׃ וַאֲמַר לָבָן, לְיַעֲקוֹב, מָא עֲבַדְתָּא, וְכַסִּיתָא מִנִּי; וְדַבַּרְתְּ, יָת בְּנָתַי, כְּשִׁבְיָת, חֲרַב. 
לא,כז לָ֤מָּה נַחְבֵּ֨אתָ֙ לִבְרֹ֔חַ וַתִּגְנֹ֖ב אֹתִ֑י וְלֹֽא־הִגַּ֣דְתָּ לִּ֔י וָֽאֲשַׁלֵּחֲךָ֛ בְּשִׂמְחָ֥ה וּבְשִׁרִ֖ים בְּתֹ֥ף וּבְכִנּֽוֹר׃ לְמָא אִטְּמַרְתָּא לְמֵיזַל, וְכַסִּיתָא מִנִּי; וְלָא חַוֵּית לִי, וְשַׁלַּחְתָּךְ פּוֹן בְּחֶדְוָא וּבְתֻשְׁבְּחָן בְּתֻפִּין וּבְכִנָּרִין. 
לא,כח וְלֹ֣א נְטַשְׁתַּ֔נִי לְנַשֵּׁ֥ק לְבָנַ֖י וְלִבְנֹתָ֑י עַתָּ֖ה הִסְכַּ֥לְתָּֽ עֲשֽׂוֹ׃ וְלָא שְׁבַקְתַּנִי, לְנַשָּׁקָא לִבְנַי וְלִבְנָתָי; כְּעַן, אַסְכֵּילְתָּא לְמֶעֱבַד. 
לא,כט יֶשׁ־לְאֵ֣ל יָדִ֔י לַֽעֲשׂ֥וֹת עִמָּכֶ֖ם רָ֑ע וֵֽאלֹהֵ֨י אֲבִיכֶ֜ם אֶ֣מֶשׁ ׀ אָמַ֧ר אֵלַ֣י לֵאמֹ֗ר הִשָּׁ֧מֶר לְךָ֛ מִדַּבֵּ֥ר עִֽם־יַעֲקֹ֖ב מִטּ֥וֹב עַד־רָֽע׃ אִית חֵילָא בִּידִי, לְמֶעֱבַד עִמְּכוֹן בִּישָׁא; וֵאלָהָא דַּאֲבוּכוֹן בְּרַמְשָׁא אֲמַר לִי לְמֵימַר, אִסְתְּמַר לָךְ מִלְּמַלָּלָא עִם יַעֲקוֹב--מִטָּב עַד בִּישׁ. 
לא,ל וְעַתָּה֙ הָלֹ֣ךְ הָלַ֔כְתָּ כִּֽי־נִכְסֹ֥ף נִכְסַ֖פְתָּה לְבֵ֣ית אָבִ֑יךָ לָ֥מָּה גָנַ֖בְתָּ אֶת־אֱלֹהָֽי׃ וּכְעַן מֵיזָל אֲזַלְתָּא, אֲרֵי חַמָּדָא חַמֵּידְתָּא לְבֵית אֲבוּךְ; לְמָא נְסֵיבְתָּא, יָת דַּחְלְתִי. 
לא,לא וַיַּ֥עַן יַֽעֲקֹ֖ב וַיֹּ֣אמֶר לְלָבָ֑ן כִּ֣י יָרֵ֔אתִי כִּ֣י אָמַ֔רְתִּי פֶּן־תִּגְזֹ֥ל אֶת־בְּנוֹתֶ֖יךָ מֵֽעִמִּֽי׃ וַאֲתֵיב יַעֲקוֹב, וַאֲמַר לְלָבָן:  אֲרֵי דְּחֵילִית--אֲרֵי אֲמַרִית, דִּלְמָא תֵּינוֹס יָת בְּנָתָךְ מִנִּי. 
לא,לב עִ֠ם אֲשֶׁ֨ר תִּמְצָ֣א אֶת־אֱלֹהֶיךָ֮ לֹ֣א יִֽחְיֶה֒ נֶ֣גֶד אַחֵ֧ינוּ הַֽכֶּר־לְךָ֛ מָ֥ה עִמָּדִ֖י וְקַֽח־לָ֑ךְ וְלֹֽא־יָדַ֣ע יַֽעֲקֹ֔ב כִּ֥י רָחֵ֖ל גְּנָבָֽתַם׃ אֲתַר דְּתַשְׁכַּח יָת דַּחְלָתָךְ, לָא יִתְקַיַּם--קֳדָם אֲחַנָא אִשְׁתְּמוֹדַע לָךְ מָא דְּעִמִּי, וְסַב לָךְ; וְלָא יְדַע יַעֲקוֹב, אֲרֵי רָחֵל נְסֵיבַתְנוּן. 
לא,לג וַיָּבֹ֨א לָבָ֜ן בְּאֹ֥הֶל יַֽעֲקֹ֣ב ׀ וּבְאֹ֣הֶל לֵאָ֗ה וּבְאֹ֛הֶל שְׁתֵּ֥י הָֽאֲמָהֹ֖ת וְלֹ֣א מָצָ֑א וַיֵּצֵא֙ מֵאֹ֣הֶל לֵאָ֔ה וַיָּבֹ֖א בְּאֹ֥הֶל רָחֵֽל׃ וְעָאל לָבָן בְּמַשְׁכְּנָא דְּיַעֲקוֹב וּבְמַשְׁכְּנָא דְּלֵאָה, וּבְמַשְׁכְּנָא דְּתַרְתֵּין לְחֵינָתָא--וְלָא אַשְׁכַּח; וּנְפַק מִמַּשְׁכְּנָא דְּלֵאָה, וְעָאל בְּמַשְׁכְּנָא דְּרָחֵל. 
לא,לד וְרָחֵ֞ל לָֽקְחָ֣ה אֶת־הַתְּרָפִ֗ים וַתְּשִׂמֵ֛ם בְּכַ֥ר הַגָּמָ֖ל וַתֵּ֣שֶׁב עֲלֵיהֶ֑ם וַיְמַשֵּׁ֥שׁ לָבָ֛ן אֶת־כָּל־הָאֹ֖הֶל וְלֹ֥א מָצָֽא׃ וְרָחֵל נְסֵיבַת יָת צַלְמָנַיָּא, וְשַׁוִּיאַתְנוּן בַּעֲבִיטָא דְּגַמְלָא--וִיתֵיבַת עֲלֵיהוֹן; וּמַשֵּׁישׁ לָבָן יָת כָּל מַשְׁכְּנָא, וְלָא אַשְׁכַּח. 
לא,לה וַתֹּ֣אמֶר אֶל־אָבִ֗יהָ אַל־יִ֨חַר֙ בְּעֵינֵ֣י אֲדֹנִ֔י כִּ֣י ל֤וֹא אוּכַל֙ לָק֣וּם מִפָּנֶ֔יךָ כִּי־דֶ֥רֶךְ נָשִׁ֖ים לִ֑י וַיְחַפֵּ֕שׂ וְלֹ֥א מָצָ֖א אֶת־הַתְּרָפִֽים׃ וַאֲמַרַת לַאֲבוּהָא, לָא יִתְקַף בְּעֵינֵי רִבּוֹנִי, אֲרֵי לָא אִכּוֹל לִמְקָם מִן קֳדָמָךְ, אֲרֵי אוֹרַח נְשִׁין לִי; וּבְלַשׁ, וְלָא אַשְׁכַּח יָת צַלְמָנַיָּא. 
לא,לו וַיִּ֥חַר לְיַֽעֲקֹ֖ב וַיָּ֣רֶב בְּלָבָ֑ן וַיַּ֤עַן יַֽעֲקֹב֙ וַיֹּ֣אמֶר לְלָבָ֔ן מַה־פִּשְׁעִי֙ מַ֣ה חַטָּאתִ֔י כִּ֥י דָלַ֖קְתָּ אַֽחֲרָֽי׃ וּתְקֵיף לְיַעֲקוֹב, וּנְצָא עִם לָבָן; וַאֲתֵיב יַעֲקוֹב, וַאֲמַר לְלָבָן, מָא חוֹבִי מָא סֻרְחָנִי, אֲרֵי רְדַפְתָּא בָּתְרָי. 
לא,לז כִּֽי־מִשַּׁ֣שְׁתָּ אֶת־כָּל־כֵּלַ֗י מַה־מָּצָ֨אתָ֙ מִכֹּ֣ל כְּלֵֽי־בֵיתֶ֔ךָ שִׂ֣ים כֹּ֔ה נֶ֥גֶד אַחַ֖י וְאַחֶ֑יךָ וְיוֹכִ֖יחוּ בֵּ֥ין שְׁנֵֽינוּ׃ אֲרֵי מַשֵּׁישְׁתָּא יָת כָּל מָנַי, מָא אַשְׁכַּחְתָּא מִכֹּל מָנֵי בֵּיתָךְ--שַׁו הָכָא, קֳדָם אֲחַי וַאֲחָךְ; וְיוֹכְחוּן, בֵּין תַּרְוַנָא. 
לא,לח זֶה֩ עֶשְׂרִ֨ים שָׁנָ֤ה אָֽנֹכִי֙ עִמָּ֔ךְ רְחֵלֶ֥יךָ וְעִזֶּ֖יךָ לֹ֣א שִׁכֵּ֑לוּ וְאֵילֵ֥י צֹֽאנְךָ֖ לֹ֥א אָכָֽלְתִּי׃ דְּנָן עַסְרִין שְׁנִין אֲנָא עִמָּךְ, רַחְלָךְ וְעִזָּךְ לָא אַתְכִּילוּ; וְדִכְרֵי עָנָךְ, לָא אֲכַלִית. 
לא,לט טְרֵפָה֙ לֹֽא־הֵבֵ֣אתִי אֵלֶ֔יךָ אָֽנֹכִ֣י אֲחַטֶּ֔נָּה מִיָּדִ֖י תְּבַקְשֶׁ֑נָּה גְּנֻֽבְתִ֣י י֔וֹם וּגְנֻֽבְתִ֖י לָֽיְלָה׃ דִּתְבִירָא, לָא אֵיתִיתִי לָךְ--דַּהֲוָת שָׁגְיָא מִמִּנְיָנָא, מִנִּי אַתְּ בָּעֵי לַהּ; נְטַרִית בִּימָמָא, וּנְטַרִית בְּלֵילְיָא. 
לא,מ הָיִ֧יתִי בַיּ֛וֹם אֲכָלַ֥נִי חֹ֖רֶב וְקֶ֣רַח בַּלָּ֑יְלָה וַתִּדַּ֥ד שְׁנָתִ֖י מֵֽעֵינָֽי׃ הֲוֵיתִי בִּימָמָא אַכְלַנִי שַׁרְבָּא, וּגְלִידָא נְחַת עֲלַי בְּלֵילְיָא; וְנַדַת שִׁינְתִי, מֵעֵינָי. 
לא,מא זֶה־לִּ֞י עֶשְׂרִ֣ים שָׁנָה֮ בְּבֵיתֶךָ֒ עֲבַדְתִּ֜יךָ אַרְבַּֽע־עֶשְׂרֵ֤ה שָׁנָה֙ בִּשְׁתֵּ֣י בְנֹתֶ֔יךָ וְשֵׁ֥שׁ שָׁנִ֖ים בְּצֹאנֶ֑ךָ וַתַּֽחֲלֵ֥ף אֶת־מַשְׂכֻּרְתִּ֖י עֲשֶׂ֥רֶת מֹנִֽים׃ דְּנָן לִי עַסְרִין שְׁנִין, בְּבֵיתָךְ, פְּלַחְתָּךְ אַרְבַּע עַסְרֵי שְׁנִין בְּתַרְתֵּין בְּנָתָךְ, וְשֵׁית שְׁנִין בְּעָנָךְ; וְאַשְׁנִיתָא יָת אַגְרִי, עֲסַר זִמְנִין. 
לא,מב לוּלֵ֡י אֱלֹהֵ֣י אָבִי֩ אֱלֹהֵ֨י אַבְרָהָ֜ם וּפַ֤חַד יִצְחָק֙ הָ֣יָה לִ֔י כִּ֥י עַתָּ֖ה רֵיקָ֣ם שִׁלַּחְתָּ֑נִי אֶת־עָנְיִ֞י וְאֶת־יְגִ֧יעַ כַּפַּ֛י רָאָ֥ה אֱלֹהִ֖ים וַיּ֥וֹכַח אָֽמֶשׁ׃ אִלּוּ לָא פּוֹן אֱלָהֵיהּ דְּאַבָּא אֱלָהֵיהּ דְּאַבְרָהָם וּדְדָחֵיל לֵיהּ יִצְחָק, הֲוָה בְּסַעֲדִי--אֲרֵי כְּעַן, רֵיקָן שַׁלַּחְתָּנִי; יָת עַמְלִי וְיָת לֵיאוּת יְדַי, גְּלֵי קֳדָם יְיָ--וְאוֹכַח בְּרַמְשָׁא. 
לא,מג וַיַּ֨עַן לָבָ֜ן וַיֹּ֣אמֶר אֶֽל־יַעֲקֹ֗ב הַבָּנ֨וֹת בְּנֹתַ֜י וְהַבָּנִ֤ים בָּנַי֙ וְהַצֹּ֣אן צֹאנִ֔י וְכֹ֛ל אֲשֶׁר־אַתָּ֥ה רֹאֶ֖ה לִי־ה֑וּא וְלִבְנֹתַ֞י מָֽה־אֶעֱשֶׂ֤ה לָאֵ֨לֶּה֙ הַיּ֔וֹם א֥וֹ לִבְנֵיהֶ֖ן אֲשֶׁ֥ר יָלָֽדוּ׃ וַאֲתֵיב לָבָן וַאֲמַר לְיַעֲקוֹב, בְּנָתָא בְּנָתַי וּבְנַיָּא בְּנַי וְעָנָא עָנִי, וְכֹל דְּאַתְּ חָזֵי, דִּילִי הוּא; וְלִבְנָתַי מָא אַעֲבֵיד לְאִלֵּין, יוֹמָא דֵּין, אוֹ לִבְנֵיהוֹן, דִּילִידָא. 
לא,מד וְעַתָּ֗ה לְכָ֛ה נִכְרְתָ֥ה בְרִ֖ית אֲנִ֣י וָאָ֑תָּה וְהָיָ֥ה לְעֵ֖ד בֵּינִ֥י וּבֵינֶֽךָ׃ וּכְעַן, אֵיתַא נִגְזַר קְיָם--אֲנָא וְאַתְּ; וִיהֵי לְסָהִיד, בֵּינָא וּבֵינָךְ. 
לא,מה וַיִּקַּ֥ח יַֽעֲקֹ֖ב אָ֑בֶן וַיְרִימֶ֖הָ מַצֵּבָֽה׃ וּנְסֵיב יַעֲקוֹב, אַבְנָא; וְזַקְפַהּ, קָמָא. 
לא,מו וַיֹּ֨אמֶר יַֽעֲקֹ֤ב לְאֶחָיו֙ לִקְט֣וּ אֲבָנִ֔ים וַיִּקְח֥וּ אֲבָנִ֖ים וַיַּֽעֲשׂוּ־גָ֑ל וַיֹּ֥אכְלוּ שָׁ֖ם עַל־הַגָּֽל׃ וַאֲמַר יַעֲקוֹב לַאֲחוֹהִי לְקוּטוּ אַבְנִין, וּנְסִיבוּ אַבְנִין וַעֲבַדוּ דְּגוֹרָא; וַאֲכַלוּ תַּמָּן, עַל דְּגוֹרָא. 
לא,מז וַיִּקְרָא־ל֣וֹ לָבָ֔ן יְגַ֖ר שָֽׂהֲדוּתָ֑א וְיַֽעֲקֹ֔ב קָ֥רָא ל֖וֹ גַּלְעֵֽד׃ וּקְרָא לֵיהּ לָבָן, יְגַר סָהֲדוּתָא; וְיַעֲקוֹב, קְרָא לֵיהּ גַּלְעֵד. 
לא,מח וַיֹּ֣אמֶר לָבָ֔ן הַגַּ֨ל הַזֶּ֥ה עֵ֛ד בֵּינִ֥י וּבֵֽינְךָ֖ הַיּ֑וֹם עַל־כֵּ֥ן קָרָֽא־שְׁמ֖וֹ גַּלְעֵֽד׃ וַאֲמַר לָבָן, דְּגוֹרָא הָדֵין סָהִיד בֵּינָא וּבֵינָךְ יוֹמָא דֵּין; עַל כֵּין קְרָא שְׁמֵיהּ, גַּלְעֵד. 
לא,מט וְהַמִּצְפָּה֙ אֲשֶׁ֣ר אָמַ֔ר יִ֥צֶף יְהוָ֖ה בֵּינִ֣י וּבֵינֶ֑ךָ כִּ֥י נִסָּתֵ֖ר אִ֥ישׁ מֵֽרֵעֵֽהוּ׃ וְסָכוּתָא דַּאֲמַר, יִיסַךְ מֵימְרָא דַּייָ בֵּינָא וּבֵינָךְ:  אֲרֵי נִתְכַּסֵּי, גְּבַר מֵחַבְרֵיהּ. 
לא,נ אִם־תְּעַנֶּ֣ה אֶת־בְּנֹתַ֗י וְאִם־תִּקַּ֤ח נָשִׁים֙ עַל־בְּנֹתַ֔י אֵ֥ין אִ֖ישׁ עִמָּ֑נוּ רְאֵ֕ה אֱלֹהִ֥ים עֵ֖ד בֵּינִ֥י וּבֵינֶֽךָ׃ אִם תְּעַנֵּי יָת בְּנָתַי, וְאִם תִּסַּב נְשִׁין עַל בְּנָתַי--לֵית אֲנָשׁ, עִמַּנָא; חֲזִי, מֵימְרָא דַּייָ סָהִיד בֵּינָא וּבֵינָךְ. 
לא,נא וַיֹּ֥אמֶר לָבָ֖ן לְיַֽעֲקֹ֑ב הִנֵּ֣ה ׀ הַגַּ֣ל הַזֶּ֗ה וְהִנֵּה֙ הַמַּצֵּבָ֔ה אֲשֶׁ֥ר יָרִ֖יתִי בֵּינִ֥י וּבֵינֶֽךָ׃ וַאֲמַר לָבָן, לְיַעֲקוֹב:  הָא דְּגוֹרָא הָדֵין, וְהָא קָמְתָא, דַּאֲקֵימִית, בֵּינָא וּבֵינָךְ. 
לא,נב עֵ֚ד הַגַּ֣ל הַזֶּ֔ה וְעֵדָ֖ה הַמַּצֵּבָ֑ה אִם־אָ֗נִי לֹֽא־אֶעֱבֹ֤ר אֵלֶ֨יךָ֙ אֶת־הַגַּ֣ל הַזֶּ֔ה וְאִם־אַ֠תָּה לֹֽא־תַעֲבֹ֨ר אֵלַ֜י אֶת־הַגַּ֥ל הַזֶּ֛ה וְאֶת־הַמַּצֵּבָ֥ה הַזֹּ֖את לְרָעָֽה׃ סָהִיד דְּגוֹרָא הָדֵין, וְסָהֲדָא קָמְתָא:  אִם אֲנָא, לָא אֶעְבַּר לְוָתָךְ יָת דְּגוֹרָא הָדֵין, וְאִם אַתְּ לָא תִּעְבַּר לְוָתִי יָת דְּגוֹרָא הָדֵין וְיָת קָמְתָא הָדָא, לְבִישׁוּ. 
לא,נג אֱלֹהֵ֨י אַבְרָהָ֜ם וֵֽאלֹהֵ֤י נָחוֹר֙ יִשְׁפְּט֣וּ בֵינֵ֔ינוּ אֱלֹהֵ֖י אֲבִיהֶ֑ם וַיִּשָּׁבַ֣ע יַֽעֲקֹ֔ב בְּפַ֖חַד אָבִ֥יו יִצְחָֽק׃ אֱלָהֵיהּ דְּאַבְרָהָם וֵאלָהֵיהּ דְּנָחוֹר, יְדִינוּן בֵּינַנָא--אֱלָהָא, דַּאֲבוּהוֹן; וְקַיֵּים יַעֲקוֹב, בִּדְדָחֵיל לֵיהּ אֲבוּהִי יִצְחָק. 
לא,נד וַיִּזְבַּ֨ח יַֽעֲקֹ֥ב זֶ֨בַח֙ בָּהָ֔ר וַיִּקְרָ֥א לְאֶחָ֖יו לֶֽאֱכָל־לָ֑חֶם וַיֹּ֣אכְלוּ לֶ֔חֶם וַיָּלִ֖ינוּ בָּהָֽר׃ וּנְכַס יַעֲקוֹב נִכְסְתָא בְּטוּרָא, וּקְרָא לַאֲחוֹהִי לְמֵיכַל לַחְמָא; וַאֲכַלוּ לַחְמָא, וּבָתוּ בְּטוּרָא. 
לב,א וַיַּשְׁכֵּ֨ם לָבָ֜ן בַּבֹּ֗קֶר וַיְנַשֵּׁ֧ק לְבָנָ֛יו וְלִבְנוֹתָ֖יו וַיְבָ֣רֶךְ אֶתְהֶ֑ם וַיֵּ֛לֶךְ וַיָּ֥שָׁב לָבָ֖ן לִמְקֹמֽוֹ׃ וְאַקְדֵּים לָבָן בְּצַפְרָא, וְנַשֵּׁיק לִבְנוֹהִי וְלִבְנָתֵיהּ--וּבָרֵיךְ יָתְהוֹן; וַאֲזַל וְתָב לָבָן, לְאַתְרֵיהּ. 
לב,ב וְיַֽעֲקֹ֖ב הָלַ֣ךְ לְדַרְכּ֑וֹ וַיִּפְגְּעוּ־ב֖וֹ מַלְאֲכֵ֥י אֱלֹהִֽים׃ וְיַעֲקוֹב, אֲזַל לְאוֹרְחֵיהּ; וְעָרַעוּ בֵּיהּ, מַלְאֲכַיָּא דַּייָ. 
לב,ג וַיֹּ֤אמֶר יַֽעֲקֹב֙ כַּֽאֲשֶׁ֣ר רָאָ֔ם מַֽחֲנֵ֥ה אֱלֹהִ֖ים זֶ֑ה וַיִּקְרָ֛א שֵֽׁם־הַמָּק֥וֹם הַה֖וּא מַֽחֲנָֽיִם׃ {פ} וַאֲמַר יַעֲקוֹב כַּד חֲזָנוּן, מַשְׁרִי מִן קֳדָם יְיָ דָּא; וּקְרָא שְׁמֵיהּ דְּאַתְרָא הַהוּא, מַחֲנָיִם. {פ}
