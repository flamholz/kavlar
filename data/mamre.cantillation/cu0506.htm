כי-תצא

כא,י כִּֽי־תֵצֵ֥א לַמִּלְחָמָ֖ה עַל־אֹֽיְבֶ֑יךָ וּנְתָנ֞וֹ יְהוָ֧ה אֱלֹהֶ֛יךָ בְּיָדֶ֖ךָ וְשָׁבִ֥יתָ שִׁבְיֽוֹ׃ אֲרֵי תִּפּוֹק לְאָגָחָא קְרָבָא, עַל בַּעֲלֵי דְּבָבָךְ; וְיִמְסְרִנּוּן יְיָ אֱלָהָךְ, בִּידָךְ--וְתִשְׁבֵּי שִׁבְיְהוֹן. 
כא,יא וְרָאִ֨יתָ֙ בַּשִּׁבְיָ֔ה אֵ֖שֶׁת יְפַת־תֹּ֑אַר וְחָֽשַׁקְתָּ֣ בָ֔הּ וְלָֽקַחְתָּ֥ לְךָ֖ לְאִשָּֽׁה׃ וְתִחְזֵי, בְּשִׁבְיָא, אִתְּתָא, שַׁפִּירַת רֵיו; וְתִתְרְעֵי בַּהּ, וְתִסְּבַהּ לָךְ לְאִתּוּ. 
כא,יב וַֽהֲבֵאתָ֖הּ אֶל־תּ֣וֹךְ בֵּיתֶ֑ךָ וְגִלְּחָה֙ אֶת־רֹאשָׁ֔הּ וְעָֽשְׂתָ֖ה אֶת־צִפָּרְנֶֽיהָ׃ וְתַעֲלִנַּהּ, לְגוֹ בֵּיתָךְ; וּתְגַלַּח, יָת רֵישַׁהּ, וּתְרַבֵּי, יָת טֻפְרַהָא. 
כא,יג וְהֵסִירָה֩ אֶת־שִׂמְלַ֨ת שִׁבְיָ֜הּ מֵֽעָלֶ֗יהָ וְיָֽשְׁבָה֙ בְּבֵיתֶ֔ךָ וּבָ֥כְתָ֛ה אֶת־אָבִ֥יהָ וְאֶת־אִמָּ֖הּ יֶ֣רַח יָמִ֑ים וְאַ֨חַר כֵּ֜ן תָּב֤וֹא אֵלֶ֨יהָ֙ וּבְעַלְתָּ֔הּ וְהָֽיְתָ֥ה לְךָ֖ לְאִשָּֽׁה׃ וְתַעְדֵּי יָת כְּסוּת שִׁבְיַהּ מִנַּהּ, וְתִתֵּיב בְּבֵיתָךְ, וְתִבְכֵּי יָת אֲבוּהָא וְיָת אִמַּהּ, יְרַח יוֹמִין; וּבָתַר כֵּין תֵּיעוֹל לְוָתַהּ, וְתִבְעֲלִנַּהּ, וּתְהֵי לָךְ, לְאִתּוּ. 
כא,יד וְהָיָ֞ה אִם־לֹ֧א חָפַ֣צְתָּ בָּ֗הּ וְשִׁלַּחְתָּהּ֙ לְנַפְשָׁ֔הּ וּמָכֹ֥ר לֹֽא־תִמְכְּרֶ֖נָּה בַּכָּ֑סֶף לֹֽא־תִתְעַמֵּ֣ר בָּ֔הּ תַּ֖חַת אֲשֶׁ֥ר עִנִּיתָֽהּ׃ {ס} וִיהֵי אִם לָא תִּתְרְעֵי בַּהּ, וְתִפְטְרִנַּהּ לְנַפְשַׁהּ, וְזַבָּנָא לָא תְּזַבְּנִנַּהּ, בְּכַסְפָּא; לָא תִּתַּגַר בַּהּ, חֲלָף דְּעַנִּיתַהּ. {ס}
כא,טו כִּֽי־תִהְיֶ֨יןָ לְאִ֜ישׁ שְׁתֵּ֣י נָשִׁ֗ים הָֽאַחַ֤ת אֲהוּבָה֙ וְהָֽאַחַ֣ת שְׂנוּאָ֔ה וְיָֽלְדוּ־ל֣וֹ בָנִ֔ים הָֽאֲהוּבָ֖ה וְהַשְּׂנוּאָ֑ה וְהָיָ֛ה הַבֵּ֥ן הַבְּכֹ֖ר לַשְּׂנִיאָֽה׃ אֲרֵי יִהְוְיָן לִגְבַר תַּרְתֵּין נְשִׁין, חֲדָא רְחוּמְתָא וַחֲדָא סְנוּאֲתָא, וִילִידָן לֵיהּ בְּנִין, רְחוּמְתָא וּסְנוּאֲתָא; וִיהֵי בְּרָא בֻּכְרָא, לִסְנִיאֲתָא. 
כא,טז וְהָיָ֗ה בְּיוֹם֙ הַנְחִיל֣וֹ אֶת־בָּנָ֔יו אֵ֥ת אֲשֶׁר־יִֽהְיֶ֖ה ל֑וֹ לֹ֣א יוּכַ֗ל לְבַכֵּר֙ אֶת־בֶּן־הָ֣אֲהוּבָ֔ה עַל־פְּנֵ֥י בֶן־הַשְּׂנוּאָ֖ה הַבְּכֹֽר׃ וִיהֵי, בְּיוֹמָא דְּיַחְסֵין לִבְנוֹהִי, יָת דִּיהֵי, לֵיהּ--לֵית לֵיהּ רְשׁוּ, לְבַכָּרָא יָת בַּר רְחוּמְתָא, עַל אַפֵּי בַּר סְנוּאֲתָא, בֻּכְרָא. 
כא,יז כִּי֩ אֶת־הַבְּכֹ֨ר בֶּן־הַשְּׂנוּאָ֜ה יַכִּ֗יר לָ֤תֶת לוֹ֙ פִּ֣י שְׁנַ֔יִם בְּכֹ֥ל אֲשֶׁר־יִמָּצֵ֖א ל֑וֹ כִּי־הוּא֙ רֵאשִׁ֣ית אֹנ֔וֹ ל֖וֹ מִשְׁפַּ֥ט הַבְּכֹרָֽה׃ {ס} אֲרֵי יָת בֻּכְרָא בַּר סְנוּאֲתָא יְפָרֵישׁ, לְמִתַּן לֵיהּ תְּרֵין חוּלָקִין, בְּכֹל דְּיִשְׁתְּכַח, לֵיהּ:  אֲרֵי הוּא רֵישׁ תֻּקְפֵיהּ, לֵיהּ חָזְיָא בְּכֵירוּתָא. {ס}
כא,יח כִּֽי־יִהְיֶ֣ה לְאִ֗ישׁ בֵּ֚ן סוֹרֵ֣ר וּמוֹרֶ֔ה אֵינֶ֣נּוּ שֹׁמֵ֔עַ בְּק֥וֹל אָבִ֖יו וּבְק֣וֹל אִמּ֑וֹ וְיִסְּר֣וּ אֹת֔וֹ וְלֹ֥א יִשְׁמַ֖ע אֲלֵיהֶֽם׃ אֲרֵי יְהֵי לִגְבַר, בַּר סָטֵי וּמָרוֹד--לָיְתוֹהִי מְקַבֵּיל, לְמֵימַר אֲבוּהִי וּלְמֵימַר אִמֵּיהּ; וּמַלְּפִין יָתֵיהּ, וְלָא מְקַבֵּיל מִנְּהוֹן. 
כא,יט וְתָ֥פְשׂוּ ב֖וֹ אָבִ֣יו וְאִמּ֑וֹ וְהוֹצִ֧יאוּ אֹת֛וֹ אֶל־זִקְנֵ֥י עִיר֖וֹ וְאֶל־שַׁ֥עַר מְקֹמֽוֹ׃ וְיֵיחֲדוּן בֵּיהּ, אֲבוּהִי וְאִמֵּיהּ; וְיַפְּקוּן יָתֵיהּ לִקְדָם סָבֵי קַרְתֵּיהּ, וְלִתְרַע בֵּית דִּין אַתְרֵיהּ. 
כא,כ וְאָֽמְר֞וּ אֶל־זִקְנֵ֣י עִיר֗וֹ בְּנֵ֤נוּ זֶה֙ סוֹרֵ֣ר וּמֹרֶ֔ה אֵינֶ֥נּוּ שֹׁמֵ֖עַ בְּקֹלֵ֑נוּ זוֹלֵ֖ל וְסֹבֵֽא׃ וְיֵימְרוּן לְסָבֵי קַרְתֵּיהּ, בְּרַנָא דֵּין סָטֵי וּמָרוֹד--לָיְתוֹהִי מְקַבֵּיל, לְמֵימְרַנָא; זָלֵיל בְּסַר, וְסָבֵי חֲמַר. 
כא,כא וּ֠רְגָמֻהוּ כָּל־אַנְשֵׁ֨י עִיר֤וֹ בָֽאֲבָנִים֙ וָמֵ֔ת וּבִֽעַרְתָּ֥ הָרָ֖ע מִקִּרְבֶּ֑ךָ וְכָל־יִשְׂרָאֵ֖ל יִשְׁמְע֥וּ וְיִרָֽאוּ׃ {ס} וְיִרְגְּמוּנֵּיהּ כָּל אֲנָשֵׁי קַרְתֵּיהּ בְּאַבְנַיָּא, וִימוּת, וּתְפַלֵּי עָבֵיד דְּבִישׁ, מִבֵּינָךְ; וְכָל יִשְׂרָאֵל, יִשְׁמְעוּן וְיִדְחֲלוּן. {ס}
כא,כב וְכִֽי־יִהְיֶ֣ה בְאִ֗ישׁ חֵ֛טְא מִשְׁפַּט־מָ֖וֶת וְהוּמָ֑ת וְתָלִ֥יתָ אֹת֖וֹ עַל־עֵֽץ׃ וַאֲרֵי יְהֵי בִּגְבַר, חוֹבַת דִּין דִּקְטוֹל--וְיִתְקְטִיל:  וְתִצְלוֹב יָתֵיהּ, עַל צְלִיבָא. 
כא,כג לֹֽא־תָלִ֨ין נִבְלָת֜וֹ עַל־הָעֵ֗ץ כִּֽי־קָב֤וֹר תִּקְבְּרֶ֨נּוּ֙ בַּיּ֣וֹם הַה֔וּא כִּֽי־קִלְלַ֥ת אֱלֹהִ֖ים תָּל֑וּי וְלֹ֤א תְטַמֵּא֙ אֶת־אַדְמָ֣תְךָ֔ אֲשֶׁר֙ יְהוָ֣ה אֱלֹהֶ֔יךָ נֹתֵ֥ן לְךָ֖ נַֽחֲלָֽה׃ {ס} לָא תְּבִית נְבִילְתֵיהּ עַל צְלִיבָא, אֲרֵי מִקְבָּר תִּקְבְּרִנֵּיהּ בְּיוֹמָא הַהוּא--אֲרֵי עַל דְּחָב קֳדָם יְיָ, אִצְטְלִיב; וְלָא תְּסַאֵיב, יָת אַרְעָךְ, דַּייָ אֱלָהָךְ, יָהֵיב לָךְ אַחְסָנָא. {ס}
כב,א לֹֽא־תִרְאֶה֩ אֶת־שׁ֨וֹר אָחִ֜יךָ א֤וֹ אֶת־שֵׂיוֹ֙ נִדָּחִ֔ים וְהִתְעַלַּמְתָּ֖ מֵהֶ֑ם הָשֵׁ֥ב תְּשִׁיבֵ֖ם לְאָחִֽיךָ׃ לָא תִּחְזֵי יָת תּוֹרָא דַּאֲחוּךְ אוֹ יָת אִמְּרֵיהּ, דְּטָעַן, וְתִכְבּוֹשׁ, מִנְּהוֹן:  אָתָבָא תָּתִיבִנּוּן, לַאֲחוּךְ. 
כב,ב וְאִם־לֹ֨א קָר֥וֹב אָחִ֛יךָ אֵלֶ֖יךָ וְלֹ֣א יְדַעְתּ֑וֹ וַֽאֲסַפְתּוֹ֙ אֶל־תּ֣וֹךְ בֵּיתֶ֔ךָ וְהָיָ֣ה עִמְּךָ֗ עַ֣ד דְּרֹ֤שׁ אָחִ֨יךָ֙ אֹת֔וֹ וַהֲשֵֽׁבֹת֖וֹ לֽוֹ׃ וְאִם לָא קָרִיב אֲחוּךְ לְוָתָךְ, וְלָא יְדַעְתְּ לֵיהּ--וְתִכְנְשִׁנֵּיהּ, לְגוֹ בֵּיתָךְ, וִיהֵי עִמָּךְ עַד דְּיִתְבַּע אֲחוּךְ יָתֵיהּ, וְתָתִיבִנֵּיהּ לֵיהּ. 
כב,ג וְכֵ֧ן תַּֽעֲשֶׂ֣ה לַֽחֲמֹר֗וֹ וְכֵ֣ן תַּֽעֲשֶׂה֮ לְשִׂמְלָתוֹ֒ וְכֵ֣ן תַּֽעֲשֶׂ֗ה לְכָל־אֲבֵדַ֥ת אָחִ֛יךָ אֲשֶׁר־תֹּאבַ֥ד מִמֶּ֖נּוּ וּמְצָאתָ֑הּ לֹ֥א תוּכַ֖ל לְהִתְעַלֵּֽם׃ {ס} וְכֵין תַּעֲבֵיד לִחְמָרֵיהּ, וְכֵין תַּעֲבֵיד לִכְסוּתֵיהּ, וְכֵין תַּעֲבֵיד לְכָל אֲבֵידְתָא דַּאֲחוּךְ דְּתֵיבַד מִנֵּיהּ, וְתַשְׁכְּחִנַּהּ:  לֵית לָךְ רְשׁוּ, לְכַסָּיוּתַהּ. {ס}
כב,ד לֹֽא־תִרְאֶה֩ אֶת־חֲמ֨וֹר אָחִ֜יךָ א֤וֹ שׁוֹרוֹ֙ נֹֽפְלִ֣ים בַּדֶּ֔רֶךְ וְהִתְעַלַּמְתָּ֖ מֵהֶ֑ם הָקֵ֥ם תָּקִ֖ים עִמּֽוֹ׃ {ס} לָא תִּחְזֵי יָת חֲמָרָא דַּאֲחוּךְ אוֹ תּוֹרֵיהּ, רְמַן בְּאוֹרְחָא, וְתִכְבּוֹשׁ, מִנְּהוֹן:  אָקָמָא תָּקִים, עִמֵּיהּ. {ס}
כב,ה לֹֽא־יִהְיֶ֤ה כְלִי־גֶ֨בֶר֙ עַל־אִשָּׁ֔ה וְלֹֽא־יִלְבַּ֥שׁ גֶּ֖בֶר שִׂמְלַ֣ת אִשָּׁ֑ה כִּ֧י תֽוֹעֲבַ֛ת יְהוָ֥ה אֱלֹהֶ֖יךָ כָּל־עֹ֥שֵׂה אֵֽלֶּה׃ {פ} לָא יְהֵי תִּקּוּן זֵין דִּגְבַר עַל אִתָּא, וְלָא יִתַּקַּן גְּבַר בְּתִקּוּנֵי אִתָּא:  אֲרֵי מְרַחַק קֳדָם יְיָ אֱלָהָךְ, כָּל עָבֵיד אִלֵּין. {פ}
כב,ו כִּ֣י יִקָּרֵ֣א קַן־צִפּ֣וֹר ׀ לְפָנֶ֡יךָ בַּדֶּ֜רֶךְ בְּכָל־עֵ֣ץ ׀ א֣וֹ עַל־הָאָ֗רֶץ אֶפְרֹחִים֙ א֣וֹ בֵיצִ֔ים וְהָאֵ֤ם רֹבֶ֨צֶת֙ עַל־הָֽאֶפְרֹחִ֔ים א֖וֹ עַל־הַבֵּיצִ֑ים לֹֽא־תִקַּ֥ח הָאֵ֖ם עַל־הַבָּנִֽים׃ אֲרֵי תְּעָרַע קִנָּא דְּצִפְּרָא קֳדָמָךְ בְּאוֹרְחָא בְּכָל אִילָן אוֹ עַל אַרְעָא, אֶפְרוֹחִין אוֹ בֵּיעִין, וְאִמָּא רְבִיעָא עַל אֶפְרוֹחִין, אוֹ עַל בֵּיעִין--לָא תִּסַּב אִמָּא, עַל בְּנַיָּא. 
כב,ז שַׁלֵּ֤חַ תְּשַׁלַּח֙ אֶת־הָאֵ֔ם וְאֶת־הַבָּנִ֖ים תִּֽקַּֽח־לָ֑ךְ לְמַ֨עַן֙ יִ֣יטַב לָ֔ךְ וְהַֽאֲרַכְתָּ֖ יָמִֽים׃ {ס} שַׁלָּחָא תְּשַׁלַּח יָת אִמָּא, וְיָת בְּנַיָּא תִּסַּב לָךְ, בְּדִיל דְּיִיטַב לָךְ, וְתוֹרֵיךְ יוֹמִין. {ס}
כב,ח כִּ֤י תִבְנֶה֙ בַּ֣יִת חָדָ֔שׁ וְעָשִׂ֥יתָ מַֽעֲקֶ֖ה לְגַגֶּ֑ךָ וְלֹֽא־תָשִׂ֤ים דָּמִים֙ בְּבֵיתֶ֔ךָ כִּֽי־יִפֹּ֥ל הַנֹּפֵ֖ל מִמֶּֽנּוּ׃ אֲרֵי תִּבְנֵי בֵּיתָא חֲדַתָּא, וְתַעֲבֵיד תְּיָקָא לְאִגָּרָךְ; וְלָא תְּשַׁוֵּי חוֹבַת קְטוֹל בְּבֵיתָךְ, אֲרֵי יִפּוֹל דְּנָפֵיל מִנֵּיהּ. 
כב,ט לֹֽא־תִזְרַ֥ע כַּרְמְךָ֖ כִּלְאָ֑יִם פֶּן־תִּקְדַּ֗שׁ הַֽמְלֵאָ֤ה הַזֶּ֨רַע֙ אֲשֶׁ֣ר תִּזְרָ֔ע וּתְבוּאַ֖ת הַכָּֽרֶם׃ {ס} לָא תִּזְרַע כַּרְמָךְ, עֵירוּבִין:  דִּלְמָא תִּסְתָּאַב, דִּמְעַת זַרְעָךְ דְּתִזְרַע, וְעַלְלַת, כַּרְמָא. {ס}
כב,י לֹֽא־תַחֲרֹ֥שׁ בְּשׁוֹר־וּבַֽחֲמֹ֖ר יַחְדָּֽו׃ לָא תִּרְדֵּי בְּתוֹרָא וּבִחְמָרָא, כַּחְדָּא. 
כב,יא לֹ֤א תִלְבַּשׁ֙ שַֽׁעַטְנֵ֔ז צֶ֥מֶר וּפִשְׁתִּ֖ים יַחְדָּֽו׃ {ס} לָא תִּלְבַּשׁ שַׁעַטְנֵיזָא, עַמַּר וְכִתָּן מְחֻבַּר כַּחְדָּא. {ס}
כב,יב גְּדִלִ֖ים תַּֽעֲשֶׂה־לָּ֑ךְ עַל־אַרְבַּ֛ע כַּנְפ֥וֹת כְּסֽוּתְךָ֖ אֲשֶׁ֥ר תְּכַסֶּה־בָּֽהּ׃ {ס} כְּרֻסְפְּדִין, תַּעֲבֵיד לָךְ, עַל אַרְבַּע כַּנְפֵי כְּסוּתָךְ, דְּתִתְכַסֵּי בַּהּ. {ס}
כב,יג כִּֽי־יִקַּ֥ח אִ֖ישׁ אִשָּׁ֑ה וּבָ֥א אֵלֶ֖יהָ וּשְׂנֵאָֽהּ׃ אֲרֵי יִסַּב גְּבַר, אִתְּתָא; וְיֵיעוֹל לְוָתַהּ, וְיִסְנֵינַהּ. 
כב,יד וְשָׂ֥ם לָהּ֙ עֲלִילֹ֣ת דְּבָרִ֔ים וְהוֹצִ֥א עָלֶ֖יהָ שֵׁ֣ם רָ֑ע וְאָמַ֗ר אֶת־הָֽאִשָּׁ֤ה הַזֹּאת֙ לָקַ֔חְתִּי וָֽאֶקְרַ֣ב אֵלֶ֔יהָ וְלֹֽא־מָצָ֥אתִי לָ֖הּ בְּתוּלִֽים׃ וִישַׁוֵּי לַהּ תַּסְקֻפֵּי מִלִּין, וְיַפֵּיק עֲלַהּ שׁוֹם בִּישׁ; וְיֵימַר, יָת אִתְּתָא הָדָא נְסֵיבִית, וְעַאלִית לְוָתַהּ, וְלָא אַשְׁכַּחִית לַהּ בְּתוּלִין. 
כב,טו וְלָקַ֛ח אֲבִ֥י הַֽנַּעֲרָ֖ וְאִמָּ֑הּ וְהוֹצִ֜יאוּ אֶת־בְּתוּלֵ֧י הַֽנַּעֲרָ֛ אֶל־זִקְנֵ֥י הָעִ֖יר הַשָּֽׁעְרָה׃ וְיִסַּב אֲבוּהָא דְּעוּלֵימְתָא, וְאִמַּהּ; וְיַפְּקוּן יָת בְּתוּלֵי עוּלֵימְתָא, לִקְדָם סָבֵי קַרְתָּא--וְלִתְרַע בֵּית דִּין אַתְרָא. 
כב,טז וְאָמַ֛ר אֲבִ֥י הַֽנַּעֲרָ֖ אֶל־הַזְּקֵנִ֑ים אֶת־בִּתִּ֗י נָתַ֜תִּי לָאִ֥ישׁ הַזֶּ֛ה לְאִשָּׁ֖ה וַיִּשְׂנָאֶֽהָ׃ וְיֵימַר אֲבוּהָא דְּעוּלֵימְתָא, לְסָבַיָּא:  יָת בְּרַתִּי, יְהַבִית לְגֻבְרָא הָדֵין לְאִתּוּ--וּסְנַהּ. 
כב,יז וְהִנֵּה־ה֡וּא שָׂם֩ עֲלִילֹ֨ת דְּבָרִ֜ים לֵאמֹ֗ר לֹֽא־מָצָ֤אתִי לְבִתְּךָ֙ בְּתוּלִ֔ים וְאֵ֖לֶּה בְּתוּלֵ֣י בִתִּ֑י וּפָֽרְשׂוּ֙ הַשִּׂמְלָ֔ה לִפְנֵ֖י זִקְנֵ֥י הָעִֽיר׃ וְהָא הוּא שַׁוִּי תַּסְקֻפֵּי מִלִּין לְמֵימַר, לָא אַשְׁכַּחִית לִבְרַתָּךְ בְּתוּלִין, וְאִלֵּין, בְּתוּלֵי בְּרַתִּי; וְיִפְרְסוּן, שׁוֹשִׁפָּא, קֳדָם, סָבֵי קַרְתָּא. 
כב,יח וְלָ֥קְח֛וּ זִקְנֵ֥י הָֽעִיר־הַהִ֖וא אֶת־הָאִ֑ישׁ וְיִסְּר֖וּ אֹתֽוֹ׃ וְיִדְבְּרוּן סָבֵי קַרְתָּא הַהִיא, יָת גֻּבְרָא; וְיַלְקוֹן, יָתֵיהּ. 
כב,יט וְעָֽנְשׁ֨וּ אֹת֜וֹ מֵ֣אָה כֶ֗סֶף וְנָֽתְנוּ֙ לַֽאֲבִ֣י הַֽנַּעֲרָ֔ה כִּ֤י הוֹצִיא֙ שֵׁ֣ם רָ֔ע עַ֖ל בְּתוּלַ֣ת יִשְׂרָאֵ֑ל וְלֽוֹ־תִהְיֶ֣ה לְאִשָּׁ֔ה לֹֽא־יוּכַ֥ל לְשַׁלְּחָ֖הּ כָּל־יָמָֽיו׃ {ס} וְיִגְבּוֹן מִנֵּיהּ מְאָה סִלְעִין דִּכְסַף, וְיִתְּנוּן לַאֲבוּהָא דְּעוּלֵימְתָא--אֲרֵי אַפֵּיק שׁוֹם בִּישׁ, עַל בְּתוּלְתָא בַּת יִשְׂרָאֵל; וְלֵיהּ תְּהֵי לְאִתּוּ, לֵית לֵיהּ רְשׁוּ לְמִפְטְרַהּ כָּל יוֹמוֹהִי. {ס}
כב,כ וְאִם־אֱמֶ֣ת הָיָ֔ה הַדָּבָ֖ר הַזֶּ֑ה לֹֽא־נִמְצְא֥וּ בְתוּלִ֖ים לַֽנַּעֲרָֽ׃ וְאִם קֻשְׁטָא הֲוָה, פִּתְגָמָא הָדֵין:  לָא אִשְׁתְּכַחוּ בְּתוּלִין, לְעוּלֵימְתָא. 
כב,כא וְהוֹצִ֨יאוּ אֶת־הַֽנַּעֲרָ֜ אֶל־פֶּ֣תַח בֵּית־אָבִ֗יהָ וּסְקָלוּהָ֩ אַנְשֵׁ֨י עִירָ֤הּ בָּֽאֲבָנִים֙ וָמֵ֔תָה כִּֽי־עָשְׂתָ֤ה נְבָלָה֙ בְּיִשְׂרָאֵ֔ל לִזְנ֖וֹת בֵּ֣ית אָבִ֑יהָ וּבִֽעַרְתָּ֥ הָרָ֖ע מִקִּרְבֶּֽךָ׃ {ס} וְיַפְּקוּן יָת עוּלֵימְתָא לִתְרַע בֵּית אֲבוּהָא, וְיִרְגְּמוּנַּהּ אֲנָשֵׁי קַרְתַּהּ בְּאַבְנַיָּא וּתְמוּת--אֲרֵי עֲבַדַת קְלָנָא בְּיִשְׂרָאֵל, לְזַנָּאָה בֵּית אֲבוּהָא; וּתְפַלֵּי עָבֵיד דְּבִישׁ, מִבֵּינָךְ. {ס}
כב,כב כִּֽי־יִמָּצֵ֨א אִ֜ישׁ שֹׁכֵ֣ב ׀ עִם־אִשָּׁ֣ה בְעֻֽלַת־בַּ֗עַל וּמֵ֨תוּ֙ גַּם־שְׁנֵיהֶ֔ם הָאִ֛ישׁ הַשֹּׁכֵ֥ב עִם־הָֽאִשָּׁ֖ה וְהָֽאִשָּׁ֑ה וּבִֽעַרְתָּ֥ הָרָ֖ע מִיִּשְׂרָאֵֽל׃ {ס} אֲרֵי יִשְׁתְּכַח גְּבַר שָׁכֵיב עִם אִתְּתָא אִתַּת גְּבַר, וְיִתְקַטְלוּן אַף תַּרְוֵיהוֹן--גֻּבְרָא דְּשָׁכֵיב עִם אִתְּתָא, וְאִתְּתָא; וּתְפַלֵּי עָבֵיד דְּבִישׁ, מִיִּשְׂרָאֵל. {ס}
כב,כג כִּ֤י יִֽהְיֶה֙ נַֽעֲרָ֣ בְתוּלָ֔ה מְאֹֽרָשָׂ֖ה לְאִ֑ישׁ וּמְצָאָ֥הּ אִ֛ישׁ בָּעִ֖יר וְשָׁכַ֥ב עִמָּֽהּ׃ אֲרֵי תְּהֵי עוּלֵימְתָא בְּתוּלְתָא, דִּמְאָרְסָא לִגְבַר; וְיַשְׁכְּחִנַּהּ גְּבַר בְּקַרְתָּא, וְיִשְׁכּוֹב עִמַּהּ. 
כב,כד וְהֽוֹצֵאתֶ֨ם אֶת־שְׁנֵיהֶ֜ם אֶל־שַׁ֣עַר ׀ הָעִ֣יר הַהִ֗וא וּסְקַלְתֶּ֨ם אֹתָ֥ם בָּֽאֲבָנִים֮ וָמֵתוּ֒ אֶת־הַֽנַּעֲרָ֗ עַל־דְּבַר֙ אֲשֶׁ֣ר לֹֽא־צָעֲקָ֣ה בָעִ֔יר וְאֶ֨ת־הָאִ֔ישׁ עַל־דְּבַ֥ר אֲשֶׁר־עִנָּ֖ה אֶת־אֵ֣שֶׁת רֵעֵ֑הוּ וּבִֽעַרְתָּ֥ הָרָ֖ע מִקִּרְבֶּֽךָ׃ {ס} וְתַפְּקוּן יָת תַּרְוֵיהוֹן לִתְרַע קַרְתָּא הַהִיא, וְתִרְגְּמוּן יָתְהוֹן בְּאַבְנַיָּא וִימוּתוּן--יָת עוּלֵימְתָא עַל עֵיסַק דְּלָא צְוַחַת בְּקַרְתָּא, וְיָת גֻּבְרָא עַל עֵיסַק דְּעַנִּי יָת אִתַּת חַבְרֵיהּ; וּתְפַלֵּי עָבֵיד דְּבִישׁ, מִבֵּינָךְ. {ס}
כב,כה וְֽאִם־בַּשָּׂדֶ֞ה יִמְצָ֣א הָאִ֗ישׁ אֶת־הַֽנַּעֲרָ֙ הַמְאֹ֣רָשָׂ֔ה וְהֶחֱזִֽיק־בָּ֥הּ הָאִ֖ישׁ וְשָׁכַ֣ב עִמָּ֑הּ וּמֵ֗ת הָאִ֛ישׁ אֲשֶׁר־שָׁכַ֥ב עִמָּ֖הּ לְבַדּֽוֹ׃ וְאִם בְּחַקְלָא יַשְׁכַּח גֻּבְרָא, יָת עוּלֵימְתָא דִּמְאָרְסָא, וְיַתְקֵיף בַּהּ גֻּבְרָא, וְיִשְׁכּוֹב עִמַּהּ:  וְיִתְקְטִיל, גֻּבְרָא דִּשָׁכֵיב עִמַּהּ--בִּלְחוֹדוֹהִי. 
כב,כו וְלַֽנַּעֲרָ֙ לֹֽא־תַעֲשֶׂ֣ה דָבָ֔ר אֵ֥ין לַֽנַּעֲרָ֖ חֵ֣טְא מָ֑וֶת כִּ֡י כַּֽאֲשֶׁר֩ יָק֨וּם אִ֤ישׁ עַל־רֵעֵ֨הוּ֙ וּרְצָח֣וֹ נֶ֔פֶשׁ כֵּ֖ן הַדָּבָ֥ר הַזֶּֽה׃ וּלְעוּלֵימְתָא לָא תַּעֲבֵיד מִדָּעַם, לֵית לְעוּלֵימְתָא חוֹבַת דִּין דִּקְטוֹל:  אֲרֵי כְּמָא דִּיקוּם גֻּבְרָא עַל חַבְרֵיהּ, וְיִקְטְלִנֵּיהּ נְפַשׁ--כֵּין, פִּתְגָמָא הָדֵין. 
כב,כז כִּ֥י בַשָּׂדֶ֖ה מְצָאָ֑הּ צָֽעֲקָ֗ה הַֽנַּעֲרָ֙ הַמְאֹ֣רָשָׂ֔ה וְאֵ֥ין מוֹשִׁ֖יעַ לָֽהּ׃ {ס} אֲרֵי בְּחַקְלָא, אַשְׁכְּחַהּ; צְוַחַת, עוּלֵימְתָא דִּמְאָרְסָא, וְלֵית דְּפָרִיק, לַהּ. {ס}
כב,כח כִּֽי־יִמְצָ֣א אִ֗ישׁ נַֽעֲרָ֤ בְתוּלָה֙ אֲשֶׁ֣ר לֹֽא־אֹרָ֔שָׂה וּתְפָשָׂ֖הּ וְשָׁכַ֣ב עִמָּ֑הּ וְנִמְצָֽאוּ׃ אֲרֵי יַשְׁכַּח גְּבַר, עוּלֵימְתָא בְּתוּלְתָא דְּלָא מְאָרְסָא, וְיַחֲדִנַּהּ, וְיִשְׁכּוֹב עִמַּהּ; וְיִשְׁתַּכְחוּן. 
כב,כט וְ֠נָתַן הָאִ֨ישׁ הַשֹּׁכֵ֥ב עִמָּ֛הּ לַֽאֲבִ֥י הַֽנַּעֲרָ֖ חֲמִשִּׁ֣ים כָּ֑סֶף וְלֽוֹ־תִהְיֶ֣ה לְאִשָּׁ֗ה תַּ֚חַת אֲשֶׁ֣ר עִנָּ֔הּ לֹֽא־יוּכַ֥ל שַׁלְּחָ֖הּ כָּל־יָמָֽיו׃ {ס} וְיִתֵּין גֻּבְרָא דְּשָׁכֵיב עִמַּהּ, לַאֲבוּהָא דְּעוּלֵימְתָא--חַמְשִׁין סִלְעִין דִּכְסַף; וְלֵיהּ תְּהֵי לְאִתּוּ, חֲלָף דְּעַנְּיַהּ--לֵית לֵיהּ רְשׁוּ לְמִפְטְרַהּ, כָּל יוֹמוֹהִי. {ס}
כג,א לֹֽא־יִקַּ֥ח אִ֖ישׁ אֶת־אֵ֣שֶׁת אָבִ֑יו וְלֹ֥א יְגַלֶּ֖ה כְּנַ֥ף אָבִֽיו׃ {ס} לָא יִסַּב גְּבַר, יָת אִתַּת אֲבוּהִי; וְלָא יְגַלֵּי, כַּנְפָא דַּאֲבוּהִי. {ס}
כג,ב לֹֽא־יָבֹ֧א פְצֽוּעַ־דַּכָּ֛א וּכְר֥וּת שָׁפְכָ֖ה בִּקְהַ֥ל יְהוָֽה׃ {ס} לָא יִדְכֵּי דִּפְסִיק וְדִמְחַבַּל, לְמֵיעַל בִּקְהָלָא דַּייָ. {ס}
כג,ג לֹֽא־יָבֹ֥א מַמְזֵ֖ר בִּקְהַ֣ל יְהוָ֑ה גַּ֚ם דּ֣וֹר עֲשִׂירִ֔י לֹא־יָ֥בֹא ל֖וֹ בִּקְהַ֥ל יְהוָֽה׃ {ס} לָא יִדְכֵּי מַמְזֵירָא, לְמֵיעַל בִּקְהָלָא דַּייָ:  אַף דָּרָא עֲסִירָאָה, לָא יִדְכֵּי לֵיהּ לְמֵיעַל בִּקְהָלָא דַּייָ. {ס}
כג,ד לֹֽא־יָבֹ֧א עַמּוֹנִ֛י וּמֽוֹאָבִ֖י בִּקְהַ֣ל יְהוָ֑ה גַּ֚ם דּ֣וֹר עֲשִׂירִ֔י לֹֽא־יָבֹ֥א לָהֶ֛ם בִּקְהַ֥ל יְהוָ֖ה עַד־עוֹלָֽם׃ לָא יִדְכּוֹן עַמּוֹנָאֵי וּמוֹאֲבָאֵי, לְמֵיעַל בִּקְהָלָא דַּייָ:  אַף דָּרָא עֲסִירָאָה, לָא יִדְכֵּי לְהוֹן לְמֵיעַל בִּקְהָלָא דַּייָ עַד עָלְמָא. 
כג,ה עַל־דְּבַ֞ר אֲשֶׁ֨ר לֹֽא־קִדְּמ֤וּ אֶתְכֶם֙ בַּלֶּ֣חֶם וּבַמַּ֔יִם בַּדֶּ֖רֶךְ בְּצֵֽאתְכֶ֣ם מִמִּצְרָ֑יִם וַֽאֲשֶׁר֩ שָׂכַ֨ר עָלֶ֜יךָ אֶת־בִּלְעָ֣ם בֶּן־בְּע֗וֹר מִפְּת֛וֹר אֲרַ֥ם נַֽהֲרַ֖יִם לְקַֽלְלֶֽךָּ׃ עַל עֵיסַק דְּלָא עָרַעוּ יָתְכוֹן, בְּלַחְמָא וּבְמַיָּא, בְּאוֹרְחָא, בְּמִפַּקְכוֹן מִמִּצְרָיִם; וְדַאֲגַר עֲלָךְ יָת בִּלְעָם בַּר בְּעוֹר, מִפְּתוֹר אֲרַם דְּעַל פְּרָת--לְלָטָטוּתָךְ. 
כג,ו וְלֹֽא־אָבָ֞ה יְהוָ֤ה אֱלֹהֶ֨יךָ֙ לִשְׁמֹ֣עַ אֶל־בִּלְעָ֔ם וַיַּֽהֲפֹךְ֩ יְהוָ֨ה אֱלֹהֶ֧יךָ לְּךָ֛ אֶת־הַקְּלָלָ֖ה לִבְרָכָ֑ה כִּ֥י אֲהֵֽבְךָ֖ יְהוָ֥ה אֱלֹהֶֽיךָ׃ וְלָא אֲבָא יְיָ אֱלָהָךְ, לְקַבָּלָא מִן בִּלְעָם, וַהֲפַךְ יְיָ אֱלָהָךְ לָךְ יָת לְוָטִין, לְבִרְכָן:  אֲרֵי רִיחֲמָךְ, יְיָ אֱלָהָךְ. 
כג,ז לֹֽא־תִדְרֹ֥שׁ שְׁלֹמָ֖ם וְטֹֽבָתָ֑ם כָּל־יָמֶ֖יךָ לְעוֹלָֽם׃ {ס} לָא תִּתְבַּע שְׁלָמְהוֹן, וְטָבָתְהוֹן, כָּל יוֹמָךְ, לְעָלַם. {ס}
כג,ח לֹֽא־תְתַעֵ֣ב אֲדֹמִ֔י כִּ֥י אָחִ֖יךָ ה֑וּא לֹֽא־תְתַעֵ֣ב מִצְרִ֔י כִּי־גֵ֖ר הָיִ֥יתָ בְאַרְצֽוֹ׃ לָא תְּרַחֵיק אֱדוֹמָאָה, אֲרֵי אֲחוּךְ הוּא; לָא תְּרַחֵיק מִצְרָאָה, אֲרֵי דַּיָּר הֲוֵיתָא בְּאַרְעֵיהּ. 
כג,ט בָּנִ֛ים אֲשֶׁר־יִוָּֽלְד֥וּ לָהֶ֖ם דּ֣וֹר שְׁלִישִׁ֑י יָבֹ֥א לָהֶ֖ם בִּקְהַ֥ל יְהוָֽה׃ {ס} בְּנִין דְּיִתְיַלְּדוּן לְהוֹן, דָּרָא תְּלִיתָאָה--יִדְכֵּי לְהוֹן, לְמֵיעַל בִּקְהָלָא דַּייָ. {ס}
כג,י כִּֽי־תֵצֵ֥א מַֽחֲנֶ֖ה עַל־אֹֽיְבֶ֑יךָ וְנִ֨שְׁמַרְתָּ֔ מִכֹּ֖ל דָּבָ֥ר רָֽע׃ אֲרֵי תִּפּוֹק מַשְׁרִיתָא, עַל בַּעֲלֵי דְּבָבָךְ:  וְתִסְתְּמַר--מִכֹּל, מִדָּעַם בִּישׁ. 
כג,יא כִּֽי־יִהְיֶ֤ה בְךָ֙ אִ֔ישׁ אֲשֶׁ֛ר לֹֽא־יִהְיֶ֥ה טָה֖וֹר מִקְּרֵה־לָ֑יְלָה וְיָצָא֙ אֶל־מִח֣וּץ לַֽמַּחֲנֶ֔ה לֹ֥א יָבֹ֖א אֶל־תּ֥וֹךְ הַֽמַּחֲנֶֽה׃ אֲרֵי יְהֵי בָּךְ גְּבַר, דְּלָא יְהֵי דְּכֵי מִקְּרֵי לֵילְיָא--וְיִפּוֹק לְמִבַּרָא לְמַשְׁרִיתָא, לָא יֵיעוֹל לְגוֹ מַשְׁרִיתָא. 
כג,יב וְהָיָ֥ה לִפְנֽוֹת־עֶ֖רֶב יִרְחַ֣ץ בַּמָּ֑יִם וּכְבֹ֣א הַשֶּׁ֔מֶשׁ יָבֹ֖א אֶל־תּ֥וֹךְ הַֽמַּחֲנֶֽה׃ וִיהֵי לְמִפְנֵי רַמְשָׁא, יִסְחֵי בְּמַיָּא; וּכְמֵיעַל שִׁמְשָׁא, יֵיעוֹל לְגוֹ מַשְׁרִיתָא. 
כג,יג וְיָד֙ תִּֽהְיֶ֣ה לְךָ֔ מִח֖וּץ לַֽמַּחֲנֶ֑ה וְיָצָ֥אתָ שָּׁ֖מָּה חֽוּץ׃ וַאֲתַר מְתֻקַּן יְהֵי לָךְ, מִבַּרָא לְמַשְׁרִיתָא; וְתִפּוֹק תַּמָּן, לְבָרָא. 
כג,יד וְיָתֵ֛ד תִּֽהְיֶ֥ה לְךָ֖ עַל־אֲזֵנֶ֑ךָ וְהָיָה֙ בְּשִׁבְתְּךָ֣ ח֔וּץ וְחָֽפַרְתָּ֣ה בָ֔הּ וְשַׁבְתָּ֖ וְכִסִּ֥יתָ אֶת־צֵֽאָתֶֽךָ׃ וְסִכְּתָא תְּהֵי לָךְ, עַל זֵינָךְ; וִיהֵי, בְּמִתְּבָךְ בְּבָרָא, וְתַחְפַּר בַּהּ, וּתְתוּב וּתְכַסֵּי יָת מַפַּקְתָּךְ. 
כג,טו כִּי֩ יְהוָ֨ה אֱלֹהֶ֜יךָ מִתְהַלֵּ֣ךְ ׀ בְּקֶ֣רֶב מַֽחֲנֶ֗ךָ לְהַצִּֽילְךָ֙ וְלָתֵ֤ת אֹֽיְבֶ֨יךָ֙ לְפָנֶ֔יךָ וְהָיָ֥ה מַֽחֲנֶ֖יךָ קָד֑וֹשׁ וְלֹֽא־יִרְאֶ֤ה בְךָ֙ עֶרְוַ֣ת דָּבָ֔ר וְשָׁ֖ב מֵאַֽחֲרֶֽיךָ׃ {ס} אֲרֵי יְיָ אֱלָהָךְ שְׁכִינְתֵיהּ מְהַלְּכָא בְּגוֹ מַשְׁרִיתָךְ, לְשֵׁיזָבוּתָךְ וּלְמִמְסַר סָנְאָךְ קֳדָמָךְ, וּתְהֵי מַשְׁרִיתָךְ, קַדִּישָׁא:  וְלָא יִתַּחְזֵי בָּךְ עֲבֵירַת פִּתְגָם, וִיתוּב מֵימְרֵיהּ מִלְּאֵיטָבָא לָךְ. {ס}
כג,טז לֹֽא־תַסְגִּ֥יר עֶ֖בֶד אֶל־אֲדֹנָ֑יו אֲשֶׁר־יִנָּצֵ֥ל אֵלֶ֖יךָ מֵעִ֥ם אֲדֹנָֽיו׃ לָא תִּמְסַר עֶבֶד עַמְמִין, לְיַד רִבּוֹנֵיהּ, דְּיִשְׁתֵּיזַב לְוָתָךְ, מִן קֳדָם רִבּוֹנֵיהּ. 
כג,יז עִמְּךָ֞ יֵשֵׁ֣ב בְּקִרְבְּךָ֗ בַּמָּק֧וֹם אֲשֶׁר־יִבְחַ֛ר בְּאַחַ֥ד שְׁעָרֶ֖יךָ בַּטּ֣וֹב ל֑וֹ לֹ֖א תּוֹנֶֽנּוּ׃ {ס} עִמָּךְ יִתֵּיב בֵּינָךְ, בְּאַתְרָא דְּיִתְרְעֵי בַּחֲדָא מִן קִרְוָךְ--בִּדְיִיטַב לֵיהּ; לָא, תּוֹנֵינֵיהּ. {ס}
כג,יח לֹֽא־תִהְיֶ֥ה קְדֵשָׁ֖ה מִבְּנ֣וֹת יִשְׂרָאֵ֑ל וְלֹֽא־יִהְיֶ֥ה קָדֵ֖שׁ מִבְּנֵ֥י יִשְׂרָאֵֽל׃ לָא תְּהֵי אִתְּתָא מִבְּנָת יִשְׂרָאֵל, לִגְבַר עֶבֶד; וְלָא יִסַּב גֻּבְרָא מִבְּנֵי יִשְׂרָאֵל, אִתָּא אָמָא. 
כג,יט לֹֽא־תָבִיא֩ אֶתְנַ֨ן זוֹנָ֜ה וּמְחִ֣יר כֶּ֗לֶב בֵּ֛ית יְהוָ֥ה אֱלֹהֶ֖יךָ לְכָל־נֶ֑דֶר כִּ֧י תֽוֹעֲבַ֛ת יְהוָ֥ה אֱלֹהֶ֖יךָ גַּם־שְׁנֵיהֶֽם׃ {ס} לָא תַּעֵיל אֲגַר זָנִיתָא וְחֻלְפַן כַּלְבָּא, לְבֵית מַקְדְּשָׁא דַּייָ אֱלָהָךְ--לְכָל נְדַר:  אֲרֵי מְרַחַק קֳדָם יְיָ אֱלָהָךְ, אַף תַּרְוֵיהוֹן. {ס}
כג,כ לֹֽא־תַשִּׁ֣יךְ לְאָחִ֔יךָ נֶ֥שֶׁךְ כֶּ֖סֶף נֶ֣שֶׁךְ אֹ֑כֶל נֶ֕שֶׁךְ כָּל־דָּבָ֖ר אֲשֶׁ֥ר יִשָּֽׁךְ׃ לָא תְּרַבֵּי לַאֲחוּךְ, רִבִּית כְּסַף רִבִּית עֲבוּר:  רִבִּית, כָּל מִדָּעַם דְּמִתְרַבֵּי. 
כג,כא לַנָּכְרִ֣י תַשִּׁ֔יךְ וּלְאָחִ֖יךָ לֹ֣א תַשִּׁ֑יךְ לְמַ֨עַן יְבָֽרֶכְךָ֜ יְהוָ֣ה אֱלֹהֶ֗יךָ בְּכֹל֙ מִשְׁלַ֣ח יָדֶ֔ךָ עַל־הָאָ֕רֶץ אֲשֶׁר־אַתָּ֥ה בָא־שָׁ֖מָּה לְרִשְׁתָּֽהּ׃ {ס} לְבַר עַמְמִין תְּרַבֵּי, וְלַאֲחוּךְ לָא תְּרַבֵּי--בְּדִיל דִּיבָרְכִנָּךְ יְיָ אֱלָהָךְ, בְּכֹל אוֹשָׁטוּת יְדָךְ, עַל אַרְעָא, דְּאַתְּ עָלֵיל לְתַמָּן לְמֵירְתַהּ. {ס}
כג,כב כִּֽי־תִדֹּ֥ר נֶ֨דֶר֙ לַֽיהוָ֣ה אֱלֹהֶ֔יךָ לֹ֥א תְאַחֵ֖ר לְשַׁלְּמ֑וֹ כִּֽי־דָרֹ֨שׁ יִדְרְשֶׁ֜נּוּ יְהוָ֤ה אֱלֹהֶ֨יךָ֙ מֵֽעִמָּ֔ךְ וְהָיָ֥ה בְךָ֖ חֵֽטְא׃ אֲרֵי תִּדַּר נְדַר קֳדָם יְיָ אֱלָהָךְ, לָא תְּאַחַר לְשַׁלָּמוּתֵיהּ:  אֲרֵי מִתְבָּע יִתְבְּעִנֵּיהּ יְיָ אֱלָהָךְ, מִנָּךְ, וִיהֵי בָּךְ, חוֹבָא. 
כג,כג וְכִ֥י תֶחְדַּ֖ל לִנְדֹּ֑ר לֹֽא־יִהְיֶ֥ה בְךָ֖ חֵֽטְא׃ וַאֲרֵי תִּתְמְנַע, מִלְּמִדַּר--לָא יְהֵי בָּךְ, חוֹבָא. 
כג,כד מוֹצָ֥א שְׂפָתֶ֖יךָ תִּשְׁמֹ֣ר וְעָשִׂ֑יתָ כַּֽאֲשֶׁ֨ר נָדַ֜רְתָּ לַֽיהוָ֤ה אֱלֹהֶ֨יךָ֙ נְדָבָ֔ה אֲשֶׁ֥ר דִּבַּ֖רְתָּ בְּפִֽיךָ׃ {ס} אַפָּקוּת סִפְוָתָךְ, תִּטַּר וְתַעֲבֵיד:  כְּמָא דִּנְדַרְתָּא קֳדָם יְיָ אֱלָהָךְ, נְדַבְתָּא, דְּמַלֵּילְתָּא, בְּפֻמָּךְ. {ס}
כג,כה כִּ֤י תָבֹא֙ בְּכֶ֣רֶם רֵעֶ֔ךָ וְאָֽכַלְתָּ֧ עֲנָבִ֛ים כְּנַפְשְׁךָ֖ שָׂבְעֶ֑ךָ וְאֶֽל־כֶּלְיְךָ֖ לֹ֥א תִתֵּֽן׃ {ס} אֲרֵי תִּתַּגַר בְּכַרְמָא דְּחַבְרָךְ, וְתֵיכוֹל עִנְבִּין כְּנַפְשָׁךְ סִבְעָךְ; וּלְמָנָךְ, לָא תִּתֵּין. {ס}
כג,כו כִּ֤י תָבֹא֙ בְּקָמַ֣ת רֵעֶ֔ךָ וְקָֽטַפְתָּ֥ מְלִילֹ֖ת בְּיָדֶ֑ךָ וְחֶרְמֵשׁ֙ לֹ֣א תָנִ֔יף עַ֖ל קָמַ֥ת רֵעֶֽךָ׃ {ס} אֲרֵי תִּתַּגַר בְּקָמְתָא דְּחַבְרָךְ, וְתִקְטוֹף דְּמַלְיָן בִּידָךְ; וּמַגְּלָא לָא תְּרִים, עַל קָמְתָא דְּחַבְרָךְ. {ס}
כד,א כִּֽי־יִקַּ֥ח אִ֛ישׁ אִשָּׁ֖ה וּבְעָלָ֑הּ וְהָיָ֞ה אִם־לֹ֧א תִמְצָא־חֵ֣ן בְּעֵינָ֗יו כִּי־מָ֤צָא בָהּ֙ עֶרְוַ֣ת דָּבָ֔ר וְכָ֨תַב לָ֜הּ סֵ֤פֶר כְּרִיתֻת֙ וְנָתַ֣ן בְּיָדָ֔הּ וְשִׁלְּחָ֖הּ מִבֵּיתֽוֹ׃ אֲרֵי יִסַּב גְּבַר אִתְּתָא, וְיִבְעֲלִנַּהּ; וִיהֵי אִם לָא תַּשְׁכַּח רַחֲמִין בְּעֵינוֹהִי, אֲרֵי אַשְׁכַּח בַּהּ עֲבֵירַת פִּתְגָם--וְיִכְתוֹב לַהּ גֶּט פִּטּוּרִין וְיִתֵּין בִּידַהּ, וְיִפְטְרִנַּהּ מִבֵּיתֵיהּ. 
כד,ב וְיָֽצְאָ֖ה מִבֵּית֑וֹ וְהָֽלְכָ֖ה וְהָֽיְתָ֥ה לְאִישׁ־אַחֵֽר׃ וְתִפּוֹק, מִבֵּיתֵיהּ; וּתְהָךְ, וּתְהֵי לִגְבַר אָחֳרָן. 
כד,ג וּשְׂנֵאָהּ֮ הָאִ֣ישׁ הָאַֽחֲרוֹן֒ וְכָ֨תַב לָ֜הּ סֵ֤פֶר כְּרִיתֻת֙ וְנָתַ֣ן בְּיָדָ֔הּ וְשִׁלְּחָ֖הּ מִבֵּית֑וֹ א֣וֹ כִ֤י יָמוּת֙ הָאִ֣ישׁ הָאַֽחֲר֔וֹן אֲשֶׁר־לְקָחָ֥הּ ל֖וֹ לְאִשָּֽׁה׃ וְיִסְנֵינַהּ, גֻּבְרָא בָּתְרָאָה, וְיִכְתּוֹב לַהּ גֶּט פִּטּוּרִין וְיִתֵּין בִּידַהּ, וְיִפְטְרִנַּהּ מִבֵּיתֵיהּ; אוֹ אֲרֵי יְמוּת גֻּבְרָא בָּתְרָאָה, דְּנַסְבַּהּ לֵיהּ לְאִתּוּ. 
כד,ד לֹֽא־יוּכַ֣ל בַּעְלָ֣הּ הָֽרִאשׁ֣וֹן אֲשֶֽׁר־שִׁ֠לְּחָהּ לָשׁ֨וּב לְקַחְתָּ֜הּ לִֽהְי֧וֹת ל֣וֹ לְאִשָּׁ֗ה אַֽחֲרֵי֙ אֲשֶׁ֣ר הֻטַּמָּ֔אָה כִּֽי־תוֹעֵבָ֥ה הִ֖וא לִפְנֵ֣י יְהוָ֑ה וְלֹ֤א תַֽחֲטִיא֙ אֶת־הָאָ֔רֶץ אֲשֶׁר֙ יְהוָ֣ה אֱלֹהֶ֔יךָ נֹתֵ֥ן לְךָ֖ נַֽחֲלָֽה׃ {ס} לֵית לֵיהּ רְשׁוּ לְבַעְלַהּ קַדְמָאָה דְּפַטְרַהּ לִמְתָּב לְמִסְּבַהּ לְמִהְוֵי לֵיהּ לְאִתּוּ, בָּתַר דְּאִסְתָּאַבַת--אֲרֵי מְרַחֲקָא הִיא, קֳדָם יְיָ; וְלָא תְּחַיֵּיב, יָת אַרְעָא, דַּייָ אֱלָהָךְ, יָהֵיב לָךְ אַחְסָנָא. {ס}
כד,ה כִּֽי־יִקַּ֥ח אִישׁ֙ אִשָּׁ֣ה חֲדָשָׁ֔ה לֹ֤א יֵצֵא֙ בַּצָּבָ֔א וְלֹֽא־יַעֲבֹ֥ר עָלָ֖יו לְכָל־דָּבָ֑ר נָקִ֞י יִֽהְיֶ֤ה לְבֵיתוֹ֙ שָׁנָ֣ה אֶחָ֔ת וְשִׂמַּ֖ח אֶת־אִשְׁתּ֥וֹ אֲשֶׁר־לָקָֽח׃ אֲרֵי יִסַּב גְּבַר, אִתְּתָא חֲדַתָּא--לָא יִפּוֹק בְּחֵילָא, וְלָא יִעְבַּר עֲלוֹהִי לְכָל פִּתְגָם:  פְּנִי יְהֵי לְבֵיתֵיהּ, שַׁתָּא חֲדָא, וִיחַדֵּי, יָת אִתְּתֵיהּ דִּנְסֵיב. 
כד,ו לֹֽא־יַחֲבֹ֥ל רֵחַ֖יִם וָרָ֑כֶב כִּי־נֶ֖פֶשׁ ה֥וּא חֹבֵֽל׃ {ס} לָא תִּסַּב מִשְׁכּוֹנָא רִחְיָא, וְרָכְבֵיהּ:  אֲרֵי בְּהוֹן מִתְעֲבֵיד מָזוֹן, לְכָל נְפַשׁ. {ס}
כד,ז כִּֽי־יִמָּצֵ֣א אִ֗ישׁ גֹּנֵ֨ב נֶ֤פֶשׁ מֵֽאֶחָיו֙ מִבְּנֵ֣י יִשְׂרָאֵ֔ל וְהִתְעַמֶּר־בּ֖וֹ וּמְכָר֑וֹ וּמֵת֙ הַגַּנָּ֣ב הַה֔וּא וּבִֽעַרְתָּ֥ הָרָ֖ע מִקִּרְבֶּֽךָ׃ {ס} אֲרֵי יִשְׁתְּכַח גְּבַר, גָּנֵיב נַפְשָׁא מֵאֲחוֹהִי מִבְּנֵי יִשְׂרָאֵל, וְיִתַּגַר בֵּיהּ, וִיזַבְּנִנֵּיהּ--וְיִתְקְטִיל גַּנָּבָא הַהוּא, וּתְפַלֵּי עָבֵיד דְּבִישׁ מִבֵּינָךְ. {ס}
כד,ח הִשָּׁ֧מֶר בְּנֶֽגַע־הַצָּרַ֛עַת לִשְׁמֹ֥ר מְאֹ֖ד וְלַֽעֲשׂ֑וֹת כְּכֹל֩ אֲשֶׁר־יוֹר֨וּ אֶתְכֶ֜ם הַכֹּֽהֲנִ֧ים הַלְוִיִּ֛ם כַּֽאֲשֶׁ֥ר צִוִּיתִ֖ם תִּשְׁמְר֥וּ לַֽעֲשֽׂוֹת׃ אִסְתְּמַר בְּמַכְתָּשׁ סְגִירוּ לְמִטַּר לַחְדָּא, וּלְמֶעֱבַד:  כְּכֹל דְּיַלְּפוּן יָתְכוֹן כָּהֲנַיָּא לֵיוָאֵי, כְּמָא דְּפַקֵּידְתִּנּוּן--תִּטְּרוּן לְמֶעֱבַד. 
כד,ט זָכ֕וֹר אֵ֧ת אֲשֶׁר־עָשָׂ֛ה יְהוָ֥ה אֱלֹהֶ֖יךָ לְמִרְיָ֑ם בַּדֶּ֖רֶךְ בְּצֵֽאתְכֶ֥ם מִמִּצְרָֽיִם׃ {ס} הֱוִי דְּכִיר, יָת דַּעֲבַד יְיָ אֱלָהָךְ לְמִרְיָם, בְּאוֹרְחָא, בְּמִפַּקְכוֹן מִמִּצְרָיִם. {ס}
כד,י כִּֽי־תַשֶּׁ֥ה בְרֵֽעֲךָ֖ מַשַּׁ֣את מְא֑וּמָה לֹֽא־תָבֹ֥א אֶל־בֵּית֖וֹ לַֽעֲבֹ֥ט עֲבֹטֽוֹ׃ אֲרֵי תִּרְשֵׁי בְּחַבְרָךְ, רְשׁוּת מִדָּעַם--לָא תֵּיעוֹל לְבֵיתֵיהּ, לְמִסַּב מִשְׁכּוֹנֵיהּ. 
כד,יא בַּח֖וּץ תַּֽעֲמֹ֑ד וְהָאִ֗ישׁ אֲשֶׁ֤ר אַתָּה֙ נֹשֶׁ֣ה ב֔וֹ יוֹצִ֥יא אֵלֶ֛יךָ אֶֽת־הַעֲב֖וֹט הַחֽוּצָה׃ בְּבָרָא, תְּקוּם; וְגֻבְרָא, דְּאַתְּ רָשֵׁי בֵּיהּ, יַפֵּיק לָךְ יָת מִשְׁכּוֹנָא, לְבָרָא. 
כד,יב וְאִם־אִ֥ישׁ עָנִ֖י ה֑וּא לֹ֥א תִשְׁכַּ֖ב בַּֽעֲבֹטֽוֹ׃ וְאִם גְּבַר מִסְכֵּין, הוּא--לָא תִּשְׁכּוֹב, בְּמִשְׁכּוֹנֵיהּ. 
כד,יג הָשֵׁב֩ תָּשִׁ֨יב ל֤וֹ אֶֽת־הַעֲבוֹט֙ כְּב֣וֹא הַשֶּׁ֔מֶשׁ וְשָׁכַ֥ב בְּשַׂלְמָת֖וֹ וּבֵֽרְכֶ֑ךָּ וּלְךָ֙ תִּֽהְיֶ֣ה צְדָקָ֔ה לִפְנֵ֖י יְהוָ֥ה אֱלֹהֶֽיךָ׃ {ס} אָתָבָא תָּתִיב לֵיהּ יָת מִשְׁכּוֹנָא כְּמֵיעַל שִׁמְשָׁא, וְיִשְׁכּוֹב בִּכְסוּתֵיהּ וִיבָרְכִנָּךְ; וְלָךְ תְּהֵי זָכוּ, קֳדָם יְיָ אֱלָהָךְ. {ס}
כד,יד לֹֽא־תַעֲשֹׁ֥ק שָׂכִ֖יר עָנִ֣י וְאֶבְי֑וֹן מֵֽאַחֶ֕יךָ א֧וֹ מִגֵּֽרְךָ֛ אֲשֶׁ֥ר בְּאַרְצְךָ֖ בִּשְׁעָרֶֽיךָ׃ לָא תַּעְשׁוֹק אֲגִירָא, עַנְיָא וּמִסְכֵּינָא, מֵאֲחָךְ, אוֹ מִגִּיּוֹרָךְ דִּבְאַרְעָךְ בְּקִרְוָךְ. 
כד,טו בְּיוֹמוֹ֩ תִתֵּ֨ן שְׂכָר֜וֹ וְֽלֹא־תָב֧וֹא עָלָ֣יו הַשֶּׁ֗מֶשׁ כִּ֤י עָנִי֙ ה֔וּא וְאֵלָ֕יו ה֥וּא נֹשֵׂ֖א אֶת־נַפְשׁ֑וֹ וְלֹֽא־יִקְרָ֤א עָלֶ֨יךָ֙ אֶל־יְהוָ֔ה וְהָיָ֥ה בְךָ֖ חֵֽטְא׃ {ס} בְּיוֹמֵיהּ תִּתֵּין אַגְרֵיהּ וְלָא תֵּיעוֹל עֲלוֹהִי שִׁמְשָׁא, אֲרֵי עַנְיָא הוּא, וְלֵיהּ, הוּא מָסַר יָת נַפְשֵׁיהּ; וְלָא יִקְרֵי עֲלָךְ קֳדָם יְיָ, וִיהֵי בָּךְ חוֹבָא. {ס}
כד,טז לֹֽא־יוּמְת֤וּ אָבוֹת֙ עַל־בָּנִ֔ים וּבָנִ֖ים לֹֽא־יוּמְת֣וּ עַל־אָב֑וֹת אִ֥ישׁ בְּחֶטְא֖וֹ יוּמָֽתוּ׃ {ס} לָא יְמוּתוּן אֲבָהָן עַל פֹּם בְּנִין, וּבְנִין לָא יְמוּתוּן עַל פֹּם אֲבָהָן:  אֲנָשׁ בְּחוֹבֵיהּ, יְמוּתוּן. {ס}
כד,יז לֹ֣א תַטֶּ֔ה מִשְׁפַּ֖ט גֵּ֣ר יָת֑וֹם וְלֹ֣א תַֽחֲבֹ֔ל בֶּ֖גֶד אַלְמָנָֽה׃ לָא תַּצְלֵי, דִּין גִּיּוֹר יִיתַם; וְלָא תִּסַּב מִשְׁכּוֹנָא, כְּסוּת אַרְמְלָא. 
כד,יח וְזָֽכַרְתָּ֗ כִּ֣י עֶ֤בֶד הָיִ֨יתָ֙ בְּמִצְרַ֔יִם וַֽיִּפְדְּךָ֛ יְהוָ֥ה אֱלֹהֶ֖יךָ מִשָּׁ֑ם עַל־כֵּ֞ן אָֽנֹכִ֤י מְצַוְּךָ֙ לַֽעֲשׂ֔וֹת אֶת־הַדָּבָ֖ר הַזֶּֽה׃ {ס} וְתִדְכַר, אֲרֵי עַבְדָּא הֲוֵיתָא בְּמִצְרַיִם, וּפַרְקָךְ יְיָ אֱלָהָךְ, מִתַּמָּן; עַל כֵּין אֲנָא מְפַקֵּיד לָךְ, לְמֶעֱבַד, יָת פִּתְגָמָא, הָדֵין. {ס}
כד,יט כִּ֣י תִקְצֹר֩ קְצִֽירְךָ֨ בְשָׂדֶ֜ךָ וְֽשָׁכַחְתָּ֧ עֹ֣מֶר בַּשָּׂדֶ֗ה לֹ֤א תָשׁוּב֙ לְקַחְתּ֔וֹ לַגֵּ֛ר לַיָּת֥וֹם וְלָֽאַלְמָנָ֖ה יִֽהְיֶ֑ה לְמַ֤עַן יְבָֽרֶכְךָ֙ יְהוָ֣ה אֱלֹהֶ֔יךָ בְּכֹ֖ל מַֽעֲשֵׂ֥ה יָדֶֽיךָ׃ {ס} אֲרֵי תִּחְצוּד חֲצָדָךְ בְּחַקְלָךְ וְתִתְנְשֵׁי עֻמְרָא בְּחַקְלָא, לָא תְּתוּב לְמִסְּבֵיהּ--לְגִיּוֹרָא לְיַתְמָא וּלְאַרְמַלְתָּא, יְהֵי:  בְּדִיל דְּיבָרְכִנָּךְ יְיָ אֱלָהָךְ, בְּכֹל עוּבָדֵי יְדָךְ. {ס}
כד,כ כִּ֤י תַחְבֹּט֙ זֵֽיתְךָ֔ לֹ֥א תְפַאֵ֖ר אַֽחֲרֶ֑יךָ לַגֵּ֛ר לַיָּת֥וֹם וְלָֽאַלְמָנָ֖ה יִֽהְיֶֽה׃ אֲרֵי תַּחְבּוֹט זֵיתָךְ, לָא תְּפַלֵּי בָּתְרָךְ:  לְגִיּוֹרָא לְיַתְמָא וּלְאַרְמַלְתָּא, יְהֵי. 
כד,כא כִּ֤י תִבְצֹר֙ כַּרְמְךָ֔ לֹ֥א תְעוֹלֵ֖ל אַֽחֲרֶ֑יךָ לַגֵּ֛ר לַיָּת֥וֹם וְלָֽאַלְמָנָ֖ה יִֽהְיֶֽה׃ אֲרֵי תִּקְטוֹף כַּרְמָךְ, לָא תְּעָלֵיל בָּתְרָךְ:  לְגִיּוֹרָא לְיַתְמָא וּלְאַרְמַלְתָּא, יְהֵי. 
כד,כב וְזָ֣כַרְתָּ֔ כִּי־עֶ֥בֶד הָיִ֖יתָ בְּאֶ֣רֶץ מִצְרָ֑יִם עַל־כֵּ֞ן אָֽנֹכִ֤י מְצַוְּךָ֙ לַֽעֲשׂ֔וֹת אֶת־הַדָּבָ֖ר הַזֶּֽה׃ {ס} וְתִדְכַר, אֲרֵי עַבְדָּא הֲוֵיתָא בְּאַרְעָא דְּמִצְרָיִם; עַל כֵּין אֲנָא מְפַקֵּיד לָךְ, לְמֶעֱבַד, יָת פִּתְגָמָא, הָדֵין. {ס}
כה,א כִּֽי־יִהְיֶ֥ה רִיב֙ בֵּ֣ין אֲנָשִׁ֔ים וְנִגְּשׁ֥וּ אֶל־הַמִּשְׁפָּ֖ט וּשְׁפָט֑וּם וְהִצְדִּ֨יקוּ֙ אֶת־הַצַּדִּ֔יק וְהִרְשִׁ֖יעוּ אֶת־הָֽרָשָֽׁע׃ אֲרֵי יְהֵי דִּין בֵּין גֻּבְרַיָּא, וְיִתְקָרְבוּן לְדִינָא וִידִינוּנוּן; וִיזַכּוֹן, יָת זַכָּאָה, וִיחַיְּבוּן, יָת חַיָּבָא. 
כה,ב וְהָיָ֛ה אִם־בִּ֥ן הַכּ֖וֹת הָֽרָשָׁ֑ע וְהִפִּיל֤וֹ הַשֹּׁפֵט֙ וְהִכָּ֣הוּ לְפָנָ֔יו כְּדֵ֥י רִשְׁעָת֖וֹ בְּמִסְפָּֽר׃ וִיהֵי אִם בַּר חַיָּב לְאַלְקָאָה, חַיָּבָא--וְיִרְמֵינֵיהּ דַּיָּנָא וְיַלְקֵינֵיהּ קֳדָמוֹהִי, כְּמִסַּת חוֹבְתֵיהּ בְּמִנְיָן. 
כה,ג אַרְבָּעִ֥ים יַכֶּ֖נּוּ לֹ֣א יֹסִ֑יף פֶּן־יֹסִ֨יף לְהַכֹּת֤וֹ עַל־אֵ֨לֶּה֙ מַכָּ֣ה רַבָּ֔ה וְנִקְלָ֥ה אָחִ֖יךָ לְעֵינֶֽיךָ׃ אַרְבְּעִין יַלְקֵינֵיהּ, לָא יוֹסֵיף:  דִּלְמָא יוֹסֵיף לְאַלְקָיוּתֵיהּ עַל אִלֵּין מַחָא רַבָּא, וְיֵיקַל אֲחוּךְ לְעֵינָךְ. 
כה,ד לֹֽא־תַחְסֹ֥ם שׁ֖וֹר בְּדִישֽׁוֹ׃ {ס} לָא תֵּיחוֹד פֹּם תּוֹרָא, בִּדְיָשֵׁיהּ. {ס}
כה,ה כִּֽי־יֵשְׁב֨וּ אַחִ֜ים יַחְדָּ֗ו וּמֵ֨ת אַחַ֤ד מֵהֶם֙ וּבֵ֣ן אֵֽין־ל֔וֹ לֹֽא־תִהְיֶ֧ה אֵֽשֶׁת־הַמֵּ֛ת הַח֖וּצָה לְאִ֣ישׁ זָ֑ר יְבָמָהּ֙ יָבֹ֣א עָלֶ֔יהָ וּלְקָחָ֥הּ ל֛וֹ לְאִשָּׁ֖ה וְיִבְּמָֽהּ׃ אֲרֵי יִתְּבוּן אַחִין כַּחְדָּא, וִימוּת חַד מִנְּהוֹן וּבַר לֵית לֵיהּ--לָא תְּהֵי אִתַּת מִיתָא לְבָרָא, לִגְבַר אָחֳרָן:  יְבָמַהּ יֵיעוֹל עֲלַהּ, וְיִסְּבַהּ לֵיהּ לְאִתּוּ וְיַבְּמִנַּהּ. 
כה,ו וְהָיָ֗ה הַבְּכוֹר֙ אֲשֶׁ֣ר תֵּלֵ֔ד יָק֕וּם עַל־שֵׁ֥ם אָחִ֖יו הַמֵּ֑ת וְלֹֽא־יִמָּחֶ֥ה שְׁמ֖וֹ מִיִּשְׂרָאֵֽל׃ וִיהֵי, בֻּכְרָא דִּתְלִיד--יְקוּם, עַל שְׁמָא דַּאֲחוּהִי מִיתָא; וְלָא יִתְמְחֵי שְׁמֵיהּ, מִיִּשְׂרָאֵל. 
כה,ז וְאִם־לֹ֤א יַחְפֹּץ֙ הָאִ֔ישׁ לָקַ֖חַת אֶת־יְבִמְתּ֑וֹ וְעָֽלְתָה֩ יְבִמְתּ֨וֹ הַשַּׁ֜עְרָה אֶל־הַזְּקֵנִ֗ים וְאָֽמְרָה֙ מֵאֵ֨ן יְבָמִ֜י לְהָקִ֨ים לְאָחִ֥יו שֵׁם֙ בְּיִשְׂרָאֵ֔ל לֹ֥א אָבָ֖ה יַבְּמִֽי׃ וְאִם לָא יִצְבֵּי גֻּבְרָא, לְמִסַּב יָת יְבִמְתֵּיהּ; וְתִסַּק יְבִמְתֵּיהּ לִתְרַע בֵּית דִּינָא לִקְדָם סָבַיָּא, וְתֵימַר סָרֵיב יְבָמִי לְאָקָמָא לַאֲחוּהִי שְׁמָא בְּיִשְׂרָאֵל--לָא אֲבָא, לְיַבָּמוּתִי. 
כה,ח וְקָֽרְאוּ־ל֥וֹ זִקְנֵֽי־עִיר֖וֹ וְדִבְּר֣וּ אֵלָ֑יו וְעָמַ֣ד וְאָמַ֔ר לֹ֥א חָפַ֖צְתִּי לְקַחְתָּֽהּ׃ וְיִקְרוֹן לֵיהּ סָבֵי קַרְתֵּיהּ, וִימַלְּלוּן עִמֵּיהּ; וִיקוּם וְיֵימַר, לָא רְעֵינָא לְמִסְּבַהּ. 
כה,ט וְנִגְּשָׁ֨ה יְבִמְתּ֣וֹ אֵלָיו֮ לְעֵינֵ֣י הַזְּקֵנִים֒ וְחָֽלְצָ֤ה נַֽעֲלוֹ֙ מֵעַ֣ל רַגְל֔וֹ וְיָֽרְקָ֖ה בְּפָנָ֑יו וְעָֽנְתָה֙ וְאָ֣מְרָ֔ה כָּ֚כָה יֵֽעָשֶׂ֣ה לָאִ֔ישׁ אֲשֶׁ֥ר לֹֽא־יִבְנֶ֖ה אֶת־בֵּ֥ית אָחִֽיו׃ וְתִתְקָרַב יְבִמְתֵּיהּ לְוָתֵיהּ, לִקְדָם סָבַיָּא, וְתִשְׁרֵי סֵינֵיהּ מֵעַל רַגְלֵיהּ, וְתִרּוֹק בְּאַנְפּוֹהִי; וְתָתִיב, וְתֵימַר, כְּדֵין יִתְעֲבֵיד לְגֻבְרָא, דְּלָא יִבְנֵי יָת בֵּיתָא דַּאֲחוּהִי. 
כה,י וְנִקְרָ֥א שְׁמ֖וֹ בְּיִשְׂרָאֵ֑ל בֵּ֖ית חֲל֥וּץ הַנָּֽעַל׃ {ס} וְיִתְקְרֵי שְׁמֵיהּ, בְּיִשְׂרָאֵל:  בֵּית, שְׁרִי סֵינָא. {ס}
כה,יא כִּֽי־יִנָּצ֨וּ אֲנָשִׁ֤ים יַחְדָּו֙ אִ֣ישׁ וְאָחִ֔יו וְקָֽרְבָה֙ אֵ֣שֶׁת הָֽאֶחָ֔ד לְהַצִּ֥יל אֶת־אִישָׁ֖הּ מִיַּ֣ד מַכֵּ֑הוּ וְשָֽׁלְחָ֣ה יָדָ֔הּ וְהֶֽחֱזִ֖יקָה בִּמְבֻשָֽׁיו׃ אֲרֵי יִנְצוֹן גֻּבְרִין כַּחְדָּא, גְּבַר וַאֲחוּהִי, וְתִתְקָרַב אִתַּת חַד, לְשֵׁיזָבָא יָת בַּעְלַהּ מִיַּד מָחוֹהִי; וְתוֹשֵׁיט יְדַהּ, וְתַתְקֵיף בְּבֵית בַּהְתְּתֵיהּ. 
כה,יב וְקַצֹּתָ֖ה אֶת־כַּפָּ֑הּ לֹ֥א תָח֖וֹס עֵינֶֽךָ׃ {ס} וּתְקוּץ, יָת יְדַהּ:  לָא תְּחוּס, עֵינָךְ. {ס}
כה,יג לֹֽא־יִהְיֶ֥ה לְךָ֛ בְּכִֽיסְךָ֖ אֶ֣בֶן וָאָ֑בֶן גְּדוֹלָ֖ה וּקְטַנָּֽה׃ לָא יְהֵי לָךְ בְּכִיסָךְ, מַתְקַל וּמַתְקַל:  רַב, וּזְעֵיר. 
כה,יד לֹֽא־יִהְיֶ֥ה לְךָ֛ בְּבֵֽיתְךָ֖ אֵיפָ֣ה וְאֵיפָ֑ה גְּדוֹלָ֖ה וּקְטַנָּֽה׃ לָא יְהֵי לָךְ בְּבֵיתָךְ, מְכִילָא וּמְכִילָא:  רַבְּתָא, וּזְעֵירְתָא. 
כה,טו אֶ֣בֶן שְׁלֵמָ֤ה וָצֶ֨דֶק֙ יִֽהְיֶה־לָּ֔ךְ אֵיפָ֧ה שְׁלֵמָ֛ה וָצֶ֖דֶק יִֽהְיֶה־לָּ֑ךְ לְמַ֨עַן֙ יַֽאֲרִ֣יכוּ יָמֶ֔יךָ עַ֚ל הָֽאֲדָמָ֔ה אֲשֶׁר־יְהוָ֥ה אֱלֹהֶ֖יךָ נֹתֵ֥ן לָֽךְ׃ מַתְקָלִין שַׁלְמִין דִּקְשׁוֹט יְהוֹן לָךְ, מְכִילָן שַׁלְמָן דִּקְשׁוֹט יְהוֹן לָךְ--בְּדִיל, דְּיֵירְכוּן יוֹמָךְ, עַל אַרְעָא, דַּייָ אֱלָהָךְ יָהֵיב לָךְ. 
כה,טז כִּ֧י תֽוֹעֲבַ֛ת יְהוָ֥ה אֱלֹהֶ֖יךָ כָּל־עֹ֣שֵׂה אֵ֑לֶּה כֹּ֖ל עֹ֥שֵׂה עָֽוֶל׃ {פ} אֲרֵי מְרַחַק קֳדָם יְיָ אֱלָהָךְ, כָּל עָבֵיד אִלֵּין:  כֹּל, עָבֵיד שְׁקַר. {פ}
כה,יז זָכ֕וֹר אֵ֛ת אֲשֶׁר־עָשָׂ֥ה לְךָ֖ עֲמָלֵ֑ק בַּדֶּ֖רֶךְ בְּצֵֽאתְכֶ֥ם מִמִּצְרָֽיִם׃ הֱוִי דְּכִיר, יָת דַּעֲבַד לָךְ עֲמָלֵק, בְּאוֹרְחָא, בְּמִפַּקְכוֹן מִמִּצְרָיִם. 
כה,יח אֲשֶׁ֨ר קָֽרְךָ֜ בַּדֶּ֗רֶךְ וַיְזַנֵּ֤ב בְּךָ֙ כָּל־הַנֶּֽחֱשָׁלִ֣ים אַֽחֲרֶ֔יךָ וְאַתָּ֖ה עָיֵ֣ף וְיָגֵ֑עַ וְלֹ֥א יָרֵ֖א אֱלֹהִֽים׃ דְּעָרְעָךְ בְּאוֹרְחָא, וְקַטֵּיל בָּךְ כָּל דַּהֲווֹ מִתְאַחֲרִין בָּתְרָךְ--וְאַתְּ, מְשַׁלְהֵי וְלָאֵי; וְלָא דָּחֵיל, מִן קֳדָם יְיָ. 
כה,יט וְהָיָ֡ה בְּהָנִ֣יחַ יְהוָ֣ה אֱלֹהֶ֣יךָ ׀ לְ֠ךָ מִכָּל־אֹ֨יְבֶ֜יךָ מִסָּבִ֗יב בָּאָ֨רֶץ֙ אֲשֶׁ֣ר יְהוָֽה־אֱ֠לֹהֶיךָ נֹתֵ֨ן לְךָ֤ נַֽחֲלָה֙ לְרִשְׁתָּ֔הּ תִּמְחֶה֙ אֶת־זֵ֣כֶר עֲמָלֵ֔ק מִתַּ֖חַת הַשָּׁמָ֑יִם לֹ֖א תִּשְׁכָּֽח׃ {פ} וִיהֵי כַּד יְנִיחַ יְיָ אֱלָהָךְ לָךְ מִכָּל בַּעֲלֵי דְּבָבָךְ מִסְּחוֹר סְחוֹר, בְּאַרְעָא דַּייָ אֱלָהָךְ יָהֵיב לָךְ אַחְסָנָא לְמֵירְתַהּ--תִּמְחֵי יָת דֻּכְרָנֵיהּ דַּעֲמָלֵק, מִתְּחוֹת שְׁמַיָּא; לָא, תִּתְנְשֵׁי. {פ}
