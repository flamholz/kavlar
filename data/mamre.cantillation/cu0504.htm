ראה

יא,כו רְאֵ֗ה אָֽנֹכִ֛י נֹתֵ֥ן לִפְנֵיכֶ֖ם הַיּ֑וֹם בְּרָכָ֖ה וּקְלָלָֽה׃ חֲזוֹ, דַּאֲנָא יָהֵיב קֳדָמֵיכוֹן--יוֹמָא דֵּין:  בִּרְכָן, וּלְוָטִין. 
יא,כז אֶֽת־הַבְּרָכָ֑ה אֲשֶׁ֣ר תִּשְׁמְע֗וּ אֶל־מִצְו‍ֹת֙ יְהוָ֣ה אֱלֹֽהֵיכֶ֔ם אֲשֶׁ֧ר אָֽנֹכִ֛י מְצַוֶּ֥ה אֶתְכֶ֖ם הַיּֽוֹם׃ יָת בִּרְכָן--אִם תְּקַבְּלוּן, לְפִקּוֹדַיָּא דַּייָ אֱלָהֲכוֹן, דַּאֲנָא מְפַקֵּיד יָתְכוֹן, יוֹמָא דֵּין. 
יא,כח וְהַקְּלָלָ֗ה אִם־לֹ֤א תִשְׁמְעוּ֙ אֶל־מִצְו‍ֹת֙ יְהוָ֣ה אֱלֹֽהֵיכֶ֔ם וְסַרְתֶּ֣ם מִן־הַדֶּ֔רֶךְ אֲשֶׁ֧ר אָֽנֹכִ֛י מְצַוֶּ֥ה אֶתְכֶ֖ם הַיּ֑וֹם לָלֶ֗כֶת אַֽחֲרֵ֛י אֱלֹהִ֥ים אֲחֵרִ֖ים אֲשֶׁ֥ר לֹֽא־יְדַעְתֶּֽם׃ {ס} וּלְוָטִין, אִם לָא תְּקַבְּלוּן לְפִקּוֹדַיָּא דַּייָ אֱלָהֲכוֹן, וְתִסְטוֹן מִן אוֹרְחָא, דַּאֲנָא מְפַקֵּיד יָתְכוֹן יוֹמָא דֵּין:  לִמְהָךְ, בָּתַר טָעֲוָת עַמְמַיָּא--דְּלָא יְדַעְתּוּן. {ס}
יא,כט וְהָיָ֗ה כִּ֤י יְבִֽיאֲךָ֙ יְהוָ֣ה אֱלֹהֶ֔יךָ אֶל־הָאָ֕רֶץ אֲשֶׁר־אַתָּ֥ה בָא־שָׁ֖מָּה לְרִשְׁתָּ֑הּ וְנָֽתַתָּ֤ה אֶת־הַבְּרָכָה֙ עַל־הַ֣ר גְּרִזִ֔ים וְאֶת־הַקְּלָלָ֖ה עַל־הַ֥ר עֵיבָֽל׃ וִיהֵי, אֲרֵי יַעֵילִנָּךְ יְיָ אֱלָהָךְ, לְאַרְעָא, דְּאַתְּ עָלֵיל לְתַמָּן לְמֵירְתַהּ--וְתִתֵּין יָת מְבָרְכַיָּא עַל טוּרָא דִּגְרִיזִין, וְיָת מְלָטְטַיָּא עַל טוּרָא דְּעֵיבָל. 
יא,ל הֲלֹא־הֵ֜מָּה בְּעֵ֣בֶר הַיַּרְדֵּ֗ן אַֽחֲרֵי֙ דֶּ֚רֶךְ מְב֣וֹא הַשֶּׁ֔מֶשׁ בְּאֶ֨רֶץ֙ הַֽכְּנַעֲנִ֔י הַיֹּשֵׁ֖ב בָּֽעֲרָבָ֑ה מ֚וּל הַגִּלְגָּ֔ל אֵ֖צֶל אֵֽלוֹנֵ֥י מֹרֶֽה׃ הֲלָא אִנּוּן בְּעִבְרָא דְּיַרְדְּנָא, אֲחוֹרֵי אוֹרַח מַעֲלָנֵי שִׁמְשָׁא, בַּאֲרַע כְּנַעֲנָאָה, דְּיָתֵיב בְּמֵישְׁרָא--לָקֳבֵיל, גִּלְגָּלָא, בִּסְטַר, מֵישְׁרֵי מוֹרֶה. 
יא,לא כִּ֤י אַתֶּם֙ עֹֽבְרִ֣ים אֶת־הַיַּרְדֵּ֔ן לָבֹא֙ לָרֶ֣שֶׁת אֶת־הָאָ֔רֶץ אֲשֶׁר־יְהוָ֥ה אֱלֹֽהֵיכֶ֖ם נֹתֵ֣ן לָכֶ֑ם וִֽירִשְׁתֶּ֥ם אֹתָ֖הּ וִֽישַׁבְתֶּם־בָּֽהּ׃ אֲרֵי אַתּוּן, עָבְרִין יָת יַרְדְּנָא, לְמֵיעַל לְמֵירַת יָת אַרְעָא, דַּייָ אֱלָהֲכוֹן יָהֵיב לְכוֹן; וְתֵירְתוּן יָתַהּ, וְתִתְּבוּן בַּהּ. 
יא,לב וּשְׁמַרְתֶּ֣ם לַֽעֲשׂ֔וֹת אֵ֥ת כָּל־הַֽחֻקִּ֖ים וְאֶת־הַמִּשְׁפָּטִ֑ים אֲשֶׁ֧ר אָֽנֹכִ֛י נֹתֵ֥ן לִפְנֵיכֶ֖ם הַיּֽוֹם׃ וְתִטְּרוּן לְמֶעֱבַד, יָת כָּל קְיָמַיָּא וְיָת דִּינַיָּא, דַּאֲנָא יָהֵיב קֳדָמֵיכוֹן, יוֹמָא דֵּין. 
יב,א אֵ֠לֶּה הַֽחֻקִּ֣ים וְהַמִּשְׁפָּטִים֮ אֲשֶׁ֣ר תִּשְׁמְר֣וּן לַֽעֲשׂוֹת֒ בָּאָ֕רֶץ אֲשֶׁר֩ נָתַ֨ן יְהוָ֜ה אֱלֹהֵ֧י אֲבֹתֶ֛יךָ לְךָ֖ לְרִשְׁתָּ֑הּ כָּ֨ל־הַיָּמִ֔ים אֲשֶׁר־אַתֶּ֥ם חַיִּ֖ים עַל־הָֽאֲדָמָֽה׃ אִלֵּין קְיָמַיָּא וְדִינַיָּא, דְּתִטְּרוּן לְמֶעֱבַד, בְּאַרְעָא, דִּיהַב יְיָ אֱלָהָא דַּאֲבָהָתָךְ לָךְ לְמֵירְתַהּ:  כָּל יוֹמַיָּא--דְּאַתּוּן קַיָּמִין, עַל אַרְעָא. 
יב,ב אַבֵּ֣ד תְּ֠אַבְּדוּן אֶֽת־כָּל־הַמְּקֹמ֞וֹת אֲשֶׁ֧ר עָֽבְדוּ־שָׁ֣ם הַגּוֹיִ֗ם אֲשֶׁ֥ר אַתֶּ֛ם יֹֽרְשִׁ֥ים אֹתָ֖ם אֶת־אֱלֹֽהֵיהֶ֑ם עַל־הֶֽהָרִ֤ים הָֽרָמִים֙ וְעַל־הַגְּבָע֔וֹת וְתַ֖חַת כָּל־עֵ֥ץ רַֽעֲנָֽן׃ אַבָּדָא תְּאַבְּדוּן יָת כָּל אַתְרַיָּא דִּפְלַחוּ תַּמָּן עַמְמַיָּא, דְּאַתּוּן יָרְתִין יָתְהוֹן--יָת טָעֲוָתְהוֹן:  עַל טוּרַיָּא רָמַיָּא וְעַל רָמָתָא, וּתְחוֹת כָּל אִילָן עַבּוֹף. 
יב,ג וְנִתַּצְתֶּ֣ם אֶת־מִזְבְּחֹתָ֗ם וְשִׁבַּרְתֶּם֙ אֶת־מַצֵּ֣בֹתָ֔ם וַאֲשֵֽׁרֵיהֶם֙ תִּשְׂרְפ֣וּן בָּאֵ֔שׁ וּפְסִילֵ֥י אֱלֹֽהֵיהֶ֖ם תְּגַדֵּע֑וּן וְאִבַּדְתֶּ֣ם אֶת־שְׁמָ֔ם מִן־הַמָּק֖וֹם הַהֽוּא׃ וּתְתָרְעוּן יָת אֵיגוֹרֵיהוֹן, וּתְתַבְּרוּן יָת קָמָתְהוֹן, וַאֲשֵׁירֵיהוֹן תֵּיקְדוּן בְּנוּרָא, וְצַלְמֵי טָעֲוָתְהוֹן תְּקָצְצוּן; וְתַבְּדוּן יָת שׁוֹמְהוֹן, מִן אַתְרָא הַהוּא. 
יב,ד לֹֽא־תַעֲשׂ֣וּן כֵּ֔ן לַֽיהוָ֖ה אֱלֹֽהֵיכֶֽם׃ לָא תַּעְבְּדוּן כֵּין, קֳדָם יְיָ אֱלָהֲכוֹן. 
יב,ה כִּ֠י אִֽם־אֶל־הַמָּק֞וֹם אֲשֶׁר־יִבְחַ֨ר יְהוָ֤ה אֱלֹֽהֵיכֶם֙ מִכָּל־שִׁבְטֵיכֶ֔ם לָשׂ֥וּם אֶת־שְׁמ֖וֹ שָׁ֑ם לְשִׁכְנ֥וֹ תִדְרְשׁ֖וּ וּבָ֥אתָ שָּֽׁמָּה׃ אֱלָהֵין לְאַתְרָא דְּיִתְרְעֵי יְיָ אֱלָהֲכוֹן, מִכָּל שִׁבְטֵיכוֹן, לְאַשְׁרָאָה שְׁכִינְתֵיהּ, תַּמָּן--לְבֵית שְׁכִינְתֵיהּ תִּתְבְּעוּן, וְתֵיתוֹן לְתַמָּן. 
יב,ו וַֽהֲבֵאתֶ֣ם שָׁ֗מָּה עֹלֹֽתֵיכֶם֙ וְזִבְחֵיכֶ֔ם וְאֵת֙ מַעְשְׂרֹ֣תֵיכֶ֔ם וְאֵ֖ת תְּרוּמַ֣ת יֶדְכֶ֑ם וְנִדְרֵיכֶם֙ וְנִדְבֹ֣תֵיכֶ֔ם וּבְכֹרֹ֥ת בְּקַרְכֶ֖ם וְצֹֽאנְכֶֽם׃ וְתַיְתוֹן לְתַמָּן, עֲלָוָתְכוֹן וְנִכְסַת קֻדְשֵׁיכוֹן, וְיָת מַעְסְרֵיכוֹן, וְיָת אַפְרָשׁוּת יַדְכוֹן; וְנִדְרֵיכוֹן, וְנִדְבָתְכוֹן, וּבְכוֹרֵי תּוֹרֵיכוֹן, וְעָנְכוֹן. 
יב,ז וַֽאֲכַלְתֶּם־שָׁ֗ם לִפְנֵי֙ יְהוָ֣ה אֱלֹֽהֵיכֶ֔ם וּשְׂמַחְתֶּ֗ם בְּכֹל֙ מִשְׁלַ֣ח יֶדְכֶ֔ם אַתֶּ֖ם וּבָֽתֵּיכֶ֑ם אֲשֶׁ֥ר בֵּֽרַכְךָ֖ יְהוָ֥ה אֱלֹהֶֽיךָ׃ וְתֵיכְלוּן תַּמָּן, קֳדָם יְיָ אֱלָהֲכוֹן, וְתִחְדוֹן בְּכֹל אוֹשָׁטוּת יַדְכוֹן, אַתּוּן וַאֲנָשׁ בָּתֵּיכוֹן--דְּבָרְכָךְ, יְיָ אֱלָהָךְ. 
יב,ח לֹ֣א תַֽעֲשׂ֔וּן כְּ֠כֹל אֲשֶׁ֨ר אֲנַ֧חְנוּ עֹשִׂ֛ים פֹּ֖ה הַיּ֑וֹם אִ֖ישׁ כָּל־הַיָּשָׁ֥ר בְּעֵינָֽיו׃ לָא תַּעְבְּדוּן--כְּכֹל דַּאֲנַחְנָא עָבְדִין כָּא, יוֹמָא דֵּין:  גְּבַר, כָּל דְּכָשַׁר בְּעֵינוֹהִי. 
יב,ט כִּ֥י לֹֽא־בָאתֶ֖ם עַד־עָ֑תָּה אֶל־הַמְּנוּחָה֙ וְאֶל־הַֽנַּחֲלָ֔ה אֲשֶׁר־יְהוָ֥ה אֱלֹהֶ֖יךָ נֹתֵ֥ן לָֽךְ׃ אֲרֵי לָא אֲתֵיתוֹן, עַד כְּעַן--לְבֵית נְיָחָא, וּלְאַחְסָנְתָא, דַּייָ אֱלָהָךְ, יָהֵיב לָךְ. 
יב,י וַֽעֲבַרְתֶּם֮ אֶת־הַיַּרְדֵּן֒ וִֽישַׁבְתֶּ֣ם בָּאָ֔רֶץ אֲשֶׁר־יְהוָ֥ה אֱלֹֽהֵיכֶ֖ם מַנְחִ֣יל אֶתְכֶ֑ם וְהֵנִ֨יחַ לָכֶ֧ם מִכָּל־אֹֽיְבֵיכֶ֛ם מִסָּבִ֖יב וִֽישַׁבְתֶּם־בֶּֽטַח׃ וְתִעְבְּרוּן, יָת יַרְדְּנָא, וְתִתְּבוּן בְּאַרְעָא, דַּייָ אֱלָהֲכוֹן מַחְסֵין יָתְכוֹן; וִינִיחַ לְכוֹן מִכָּל בַּעֲלֵי דְּבָבֵיכוֹן מִסְּחוֹר סְחוֹר, וְתִתְּבוּן לְרֻחְצָן. 
יב,יא וְהָיָ֣ה הַמָּק֗וֹם אֲשֶׁר־יִבְחַר֩ יְהוָ֨ה אֱלֹֽהֵיכֶ֥ם בּוֹ֙ לְשַׁכֵּ֤ן שְׁמוֹ֙ שָׁ֔ם שָׁ֣מָּה תָבִ֔יאוּ אֵ֛ת כָּל־אֲשֶׁ֥ר אָֽנֹכִ֖י מְצַוֶּ֣ה אֶתְכֶ֑ם עוֹלֹֽתֵיכֶ֣ם וְזִבְחֵיכֶ֗ם מַעְשְׂרֹֽתֵיכֶם֙ וּתְרֻמַ֣ת יֶדְכֶ֔ם וְכֹל֙ מִבְחַ֣ר נִדְרֵיכֶ֔ם אֲשֶׁ֥ר תִּדְּר֖וּ לַֽיהוָֽה׃ וִיהֵי אַתְרָא, דְּיִתְרְעֵי יְיָ אֱלָהֲכוֹן בֵּיהּ לְאַשְׁרָאָה שְׁכִינְתֵיהּ תַּמָּן--לְתַמָּן תַּיְתוֹן, יָת כָּל דַּאֲנָא מְפַקֵּיד יָתְכוֹן:  עֲלָוָתְכוֹן וְנִכְסַת קֻדְשֵׁיכוֹן, מַעְסְרֵיכוֹן וְאַפְרָשׁוּת יַדְכוֹן, וְכֹל שְׁפַר נִדְרֵיכוֹן, דְּתִדְּרוּן קֳדָם יְיָ. 
יב,יב וּשְׂמַחְתֶּ֗ם לִפְנֵי֮ יְהוָ֣ה אֱלֹֽהֵיכֶם֒ אַתֶּ֗ם וּבְנֵיכֶם֙ וּבְנֹ֣תֵיכֶ֔ם וְעַבְדֵיכֶ֖ם וְאַמְהֹֽתֵיכֶ֑ם וְהַלֵּוִי֙ אֲשֶׁ֣ר בְּשַֽׁעֲרֵיכֶ֔ם כִּ֣י אֵ֥ין ל֛וֹ חֵ֥לֶק וְנַֽחֲלָ֖ה אִתְּכֶֽם׃ וְתִחְדּוֹן, קֳדָם יְיָ אֱלָהֲכוֹן--אַתּוּן וּבְנֵיכוֹן וּבְנָתְכוֹן, וְעַבְדֵיכוֹן וְאַמְהָתְכוֹן; וְלֵיוָאָה דִּבְקִרְוֵיכוֹן, אֲרֵי לֵית לֵיהּ חוּלָק וְאַחְסָנָא עִמְּכוֹן. 
יב,יג הִשָּׁ֣מֶר לְךָ֔ פֶּֽן־תַּעֲלֶ֖ה עֹֽלֹתֶ֑יךָ בְּכָל־מָק֖וֹם אֲשֶׁ֥ר תִּרְאֶֽה׃ אִסְתְּמַר לָךְ, דִּלְמָא תַּסֵּיק עֲלָוָתָךְ, בְּכָל אַתְרָא, דְּתִחְזֵי. 
יב,יד כִּ֣י אִם־בַּמָּק֞וֹם אֲשֶׁר־יִבְחַ֤ר יְהוָה֙ בְּאַחַ֣ד שְׁבָטֶ֔יךָ שָׁ֖ם תַּֽעֲלֶ֣ה עֹֽלֹתֶ֑יךָ וְשָׁ֣ם תַּֽעֲשֶׂ֔ה כֹּ֛ל אֲשֶׁ֥ר אָֽנֹכִ֖י מְצַוֶּֽךָּ׃ אֱלָהֵין בְּאַתְרָא דְּיִתְרְעֵי יְיָ, בַּחֲדָא מִן שִׁבְטָךְ--תַּמָּן, תַּסֵּיק עֲלָוָתָךְ; וְתַמָּן תַּעֲבֵיד, כֹּל דַּאֲנָא מְפַקֵּיד לָךְ. 
יב,טו רַק֩ בְּכָל־אַוַּ֨ת נַפְשְׁךָ֜ תִּזְבַּ֣ח ׀ וְאָֽכַלְתָּ֣ בָשָׂ֗ר כְּבִרְכַּ֨ת יְהוָ֧ה אֱלֹהֶ֛יךָ אֲשֶׁ֥ר נָֽתַן־לְךָ֖ בְּכָל־שְׁעָרֶ֑יךָ הַטָּמֵ֤א וְהַטָּהוֹר֙ יֹֽאכְלֶ֔נּוּ כַּצְּבִ֖י וְכָֽאַיָּֽל׃ לְחוֹד בְּכָל רְעוּת נַפְשָׁךְ תִּכּוֹס וְתֵיכוֹל בִּסְרָא, כְּבִרְכְּתָא דַּייָ אֱלָהָךְ דִּיהַב לָךְ--בְּכָל קִרְוָךְ; מְסָאֲבָא וְדָכְיָא יֵיכְלוּנֵּיהּ, כִּבְסַר טַבְיָא וְאַיְלָא. 
יב,טז רַ֥ק הַדָּ֖ם לֹ֣א תֹאכֵ֑לוּ עַל־הָאָ֥רֶץ תִּשְׁפְּכֶ֖נּוּ כַּמָּֽיִם׃ לְחוֹד דְּמָא, לָא תֵּיכְלוּן:  עַל אַרְעָא תֵּישְׁדִנֵּיהּ, כְּמַיָּא. 
יב,יז לֹֽא־תוּכַ֞ל לֶֽאֱכֹ֣ל בִּשְׁעָרֶ֗יךָ מַעְשַׂ֤ר דְּגָֽנְךָ֙ וְתִֽירֹשְׁךָ֣ וְיִצְהָרֶ֔ךָ וּבְכֹרֹ֥ת בְּקָֽרְךָ֖ וְצֹאנֶ֑ךָ וְכָל־נְדָרֶ֨יךָ֙ אֲשֶׁ֣ר תִּדֹּ֔ר וְנִדְבֹתֶ֖יךָ וּתְרוּמַ֥ת יָדֶֽךָ׃ לֵית לָךְ רְשׁוּ לְמֵיכַל בְּקִרְוָךְ, מַעְסַר עֲבוּרָךְ וְחַמְרָךְ וּמִשְׁחָךְ, וּבְכוֹרֵי תּוֹרָךְ, וְעָנָךְ; וְכָל נִדְרָךְ דְּתִדַּר, וְנִדְבָתָךְ וְאַפְרָשׁוּת יְדָךְ. 
יב,יח כִּ֡י אִם־לִפְנֵי֩ יְהוָ֨ה אֱלֹהֶ֜יךָ תֹּֽאכְלֶ֗נּוּ בַּמָּקוֹם֙ אֲשֶׁ֨ר יִבְחַ֜ר יְהוָ֣ה אֱלֹהֶיךָ֮ בּוֹ֒ אַתָּ֨ה וּבִנְךָ֤ וּבִתֶּ֨ךָ֙ וְעַבְדְּךָ֣ וַֽאֲמָתֶ֔ךָ וְהַלֵּוִ֖י אֲשֶׁ֣ר בִּשְׁעָרֶ֑יךָ וְשָֽׂמַחְתָּ֗ לִפְנֵי֙ יְהוָ֣ה אֱלֹהֶ֔יךָ בְּכֹ֖ל מִשְׁלַ֥ח יָדֶֽךָ׃ אֱלָהֵין קֳדָם יְיָ אֱלָהָךְ תֵּיכְלִנֵּיהּ, בְּאַתְרָא דְּיִתְרְעֵי יְיָ אֱלָהָךְ בֵּיהּ--אַתְּ וּבְרָךְ וּבְרַתָּךְ וְעַבְדָּךְ וְאַמְתָּךְ, וְלֵיוָאָה דִּבְקִרְוָךְ; וְתִחְדֵי, קֳדָם יְיָ אֱלָהָךְ, בְּכֹל, אוֹשָׁטוּת יְדָךְ. 
יב,יט הִשָּׁ֣מֶר לְךָ֔ פֶּֽן־תַּעֲזֹ֖ב אֶת־הַלֵּוִ֑י כָּל־יָמֶ֖יךָ עַל־אַדְמָתֶֽךָ׃ {ס} אִסְתְּמַר לָךְ, דִּלְמָא תְּרַחֵיק יָת לֵיוָאָה, כָּל יוֹמָךְ, עַל אַרְעָךְ. {ס}
יב,כ כִּֽי־יַרְחִיב֩ יְהוָ֨ה אֱלֹהֶ֥יךָ אֶֽת־גְּבֻלְךָ֮ כַּֽאֲשֶׁ֣ר דִּבֶּר־לָךְ֒ וְאָֽמַרְתָּ֙ אֹֽכְלָ֣ה בָשָׂ֔ר כִּֽי־תְאַוֶּ֥ה נַפְשְׁךָ֖ לֶֽאֱכֹ֣ל בָּשָׂ֑ר בְּכָל־אַוַּ֥ת נַפְשְׁךָ֖ תֹּאכַ֥ל בָּשָֽׂר׃ אֲרֵי יַפְתֵּי יְיָ אֱלָהָךְ יָת תְּחוּמָךְ, כְּמָא דְּמַלֵּיל לָךְ, וְתֵימַר אֵיכוֹל בִּסְרָא, אֲרֵי תִּתְרְעֵי נַפְשָׁךְ לְמֵיכַל בִּסְרָא--בְּכָל רְעוּת נַפְשָׁךְ, תֵּיכוֹל בִּסְרָא. 
יב,כא כִּֽי־יִרְחַ֨ק מִמְּךָ֜ הַמָּק֗וֹם אֲשֶׁ֨ר יִבְחַ֜ר יְהוָ֣ה אֱלֹהֶיךָ֮ לָשׂ֣וּם שְׁמ֣וֹ שָׁם֒ וְזָֽבַחְתָּ֞ מִבְּקָֽרְךָ֣ וּמִצֹּֽאנְךָ֗ אֲשֶׁ֨ר נָתַ֤ן יְהוָה֙ לְךָ֔ כַּֽאֲשֶׁ֖ר צִוִּיתִ֑ךָ וְאָֽכַלְתָּ֙ בִּשְׁעָרֶ֔יךָ בְּכֹ֖ל אַוַּ֥ת נַפְשֶֽׁךָ׃ אֲרֵי יִתְרַחַק מִנָּךְ אַתְרָא, דְּיִתְרְעֵי יְיָ אֱלָהָךְ לְאַשְׁרָאָה שְׁכִינְתֵיהּ תַּמָּן, וְתִכּוֹס מִתּוֹרָךְ וּמֵעָנָךְ דִּיהַב יְיָ לָךְ, כְּמָא דְּפַקֵּידְתָּךְ--וְתֵיכוֹל, בְּקִרְוָךְ, בְּכֹל, רְעוּת נַפְשָׁךְ. 
יב,כב אַ֗ךְ כַּֽאֲשֶׁ֨ר יֵֽאָכֵ֤ל אֶֽת־הַצְּבִי֙ וְאֶת־הָ֣אַיָּ֔ל כֵּ֖ן תֹּֽאכְלֶ֑נּוּ הַטָּמֵא֙ וְהַטָּה֔וֹר יַחְדָּ֖ו יֹֽאכְלֶֽנּוּ׃ בְּרַם, כְּמָא דְּמִתְאֲכִיל בְּסַר טַבְיָא וְאַיְלָא--כֵּין, תֵּיכְלִנֵּיהּ:  מְסָאֲבָא, וְדָכְיָא, כַּחְדָּא, יֵיכְלִנֵּיהּ. 
יב,כג רַ֣ק חֲזַ֗ק לְבִלְתִּי֙ אֲכֹ֣ל הַדָּ֔ם כִּ֥י הַדָּ֖ם ה֣וּא הַנָּ֑פֶשׁ וְלֹֽא־תֹאכַ֥ל הַנֶּ֖פֶשׁ עִם־הַבָּשָֽׂר׃ לְחוֹד תְּקַף, בְּדִיל דְּלָא לְמֵיכַל דְּמָא, אֲרֵי דְּמָא, הוּא נַפְשָׁא; וְלָא תֵּיכוֹל נַפְשָׁא, עִם בִּסְרָא. 
יב,כד לֹ֖א תֹּֽאכְלֶ֑נּוּ עַל־הָאָ֥רֶץ תִּשְׁפְּכֶ֖נּוּ כַּמָּֽיִם׃ לָא, תֵּיכְלִנֵּיהּ:  עַל אַרְעָא תֵּישְׁדִנֵּיהּ, כְּמַיָּא. 
יב,כה לֹ֖א תֹּֽאכְלֶ֑נּוּ לְמַ֨עַן יִיטַ֤ב לְךָ֙ וּלְבָנֶ֣יךָ אַֽחֲרֶ֔יךָ כִּֽי־תַעֲשֶׂ֥ה הַיָּשָׁ֖ר בְּעֵינֵ֥י יְהוָֽה׃ לָא, תֵּיכְלִנֵּיהּ--בְּדִיל דְּיִיטַב לָךְ וְלִבְנָךְ בָּתְרָךְ, אֲרֵי תַּעֲבֵיד דְּכָשַׁר קֳדָם יְיָ. 
יב,כו רַ֧ק קָֽדָשֶׁ֛יךָ אֲשֶׁר־יִֽהְי֥וּ לְךָ֖ וּנְדָרֶ֑יךָ תִּשָּׂ֣א וּבָ֔אתָ אֶל־הַמָּק֖וֹם אֲשֶׁר־יִבְחַ֥ר יְהוָֽה׃ לְחוֹד מַעְסַר קֻדְשָׁךְ דִּיהוֹן לָךְ, וְנִדְרָךְ, תִּטּוֹל וְתֵיתֵי, לְאַתְרָא דְּיִתְרְעֵי יְיָ. 
יב,כז וְעָשִׂ֤יתָ עֹֽלֹתֶ֨יךָ֙ הַבָּשָׂ֣ר וְהַדָּ֔ם עַל־מִזְבַּ֖ח יְהוָ֣ה אֱלֹהֶ֑יךָ וְדַם־זְבָחֶ֗יךָ יִשָּׁפֵךְ֙ עַל־מִזְבַּח֙ יְהוָ֣ה אֱלֹהֶ֔יךָ וְהַבָּשָׂ֖ר תֹּאכֵֽל׃ וְתַעֲבֵיד עֲלָוָתָךְ בִּסְרָא וּדְמָא, עַל מַדְבְּחָא דַּייָ אֱלָהָךְ; וְדַם נִכְסַת קֻדְשָׁךְ, יִשְׁתְּפֵיךְ עַל מַדְבְּחָא דַּייָ אֱלָהָךְ, וּבִסְרָא, תֵּיכוֹל. 
יב,כח שְׁמֹ֣ר וְשָֽׁמַעְתָּ֗ אֵ֚ת כָּל־הַדְּבָרִ֣ים הָאֵ֔לֶּה אֲשֶׁ֥ר אָֽנֹכִ֖י מְצַוֶּ֑ךָּ לְמַעַן֩ יִיטַ֨ב לְךָ֜ וּלְבָנֶ֤יךָ אַֽחֲרֶ֨יךָ֙ עַד־עוֹלָ֔ם כִּ֤י תַֽעֲשֶׂה֙ הַטּ֣וֹב וְהַיָּשָׁ֔ר בְּעֵינֵ֖י יְהוָ֥ה אֱלֹהֶֽיךָ׃ {ס} טַר וּתְקַבֵּיל, יָת כָּל פִּתְגָמַיָּא הָאִלֵּין, דַּאֲנָא, מְפַקֵּיד לָךְ:  בְּדִיל דְּיִיטַב לָךְ וְלִבְנָךְ בָּתְרָךְ, עַד עָלְמָא--אֲרֵי תַּעֲבֵיד דְּתַקִּין וּדְכָשַׁר, קֳדָם יְיָ אֱלָהָךְ. {ס}
יב,כט כִּֽי־יַכְרִית֩ יְהוָ֨ה אֱלֹהֶ֜יךָ אֶת־הַגּוֹיִ֗ם אֲשֶׁ֨ר אַתָּ֥ה בָא־שָׁ֛מָּה לָרֶ֥שֶׁת אוֹתָ֖ם מִפָּנֶ֑יךָ וְיָֽרַשְׁתָּ֣ אֹתָ֔ם וְיָֽשַׁבְתָּ֖ בְּאַרְצָֽם׃ אֲרֵי יְשֵׁיצֵי יְיָ אֱלָהָךְ יָת עַמְמַיָּא, דְּאַתְּ עָלֵיל לְתַמָּן לְמֵירַת יָתְהוֹן--מִן קֳדָמָךְ; וְתֵירַת יָתְהוֹן, וְתִתֵּיב בַּאֲרַעְהוֹן. 
יב,ל הִשָּׁ֣מֶר לְךָ֗ פֶּן־תִּנָּקֵשׁ֙ אַֽחֲרֵיהֶ֔ם אַֽחֲרֵ֖י הִשָּֽׁמְדָ֣ם מִפָּנֶ֑יךָ וּפֶן־תִּדְרֹ֨שׁ לֵֽאלֹהֵיהֶ֜ם לֵאמֹ֗ר אֵיכָ֨ה יַֽעַבְד֜וּ הַגּוֹיִ֤ם הָאֵ֨לֶּה֙ אֶת־אֱלֹ֣הֵיהֶ֔ם וְאֶֽעֱשֶׂה־כֵּ֖ן גַּם־אָֽנִי׃ אִסְתְּמַר לָךְ, דִּלְמָא תִּתְּקֵיל בָּתְרֵיהוֹן, בָּתַר, דְּיִשְׁתֵּיצוֹן מִן קֳדָמָךְ; וְדִלְמָא תִּתְבַּע לְטָעֲוָתְהוֹן לְמֵימַר, אֵיכְדֵין פָּלְחִין עַמְמַיָּא הָאִלֵּין יָת טָעֲוָתְהוֹן, וְאַעֲבֵיד כֵּין, אַף אֲנָא. 
יב,לא לֹֽא־תַעֲשֶׂ֣ה כֵ֔ן לַֽיהוָ֖ה אֱלֹהֶ֑יךָ כִּי֩ כָל־תּֽוֹעֲבַ֨ת יְהוָ֜ה אֲשֶׁ֣ר שָׂנֵ֗א עָשׂוּ֙ לֵאלֹ֣הֵיהֶ֔ם כִּ֣י גַ֤ם אֶת־בְּנֵיהֶם֙ וְאֶת־בְּנֹ֣תֵיהֶ֔ם יִשְׂרְפ֥וּ בָאֵ֖שׁ לֵֽאלֹהֵיהֶֽם׃ לָא תַּעֲבֵיד כֵּין, קֳדָם יְיָ אֱלָהָךְ:  אֲרֵי כָּל דִּמְרַחַק קֳדָם יְיָ דְּסָנֵי, עָבְדִין לְטָעֲוָתְהוֹן--אֲרֵי אַף יָת בְּנֵיהוֹן וְיָת בְּנָתְהוֹן, מוֹקְדִין בְּנוּרָא לְטָעֲוָתְהוֹן. 
יג,א אֵ֣ת כָּל־הַדָּבָ֗ר אֲשֶׁ֤ר אָֽנֹכִי֙ מְצַוֶּ֣ה אֶתְכֶ֔ם אֹת֥וֹ תִשְׁמְר֖וּ לַֽעֲשׂ֑וֹת לֹֽא־תֹסֵ֣ף עָלָ֔יו וְלֹ֥א תִגְרַ֖ע מִמֶּֽנּוּ׃ {פ} יָת כָּל פִּתְגָמָא, דַּאֲנָא מְפַקֵּיד יָתְכוֹן--יָתֵיהּ תִּטְּרוּן, לְמֶעֱבַד:  לָא תֵּיסְפוּן עֲלוֹהִי, וְלָא תִּמְנְעוּן מִנֵּיהּ. {פ}
יג,ב כִּֽי־יָק֤וּם בְּקִרְבְּךָ֙ נָבִ֔יא א֖וֹ חֹלֵ֣ם חֲל֑וֹם וְנָתַ֥ן אֵלֶ֛יךָ א֖וֹת א֥וֹ מוֹפֵֽת׃ אֲרֵי יְקוּם בֵּינָךְ נְבִיָּא, אוֹ חָלֵים חֶלְמָא; וְיִתֵּין לָךְ אָת, אוֹ מוֹפָת. 
יג,ג וּבָ֤א הָאוֹת֙ וְהַמּוֹפֵ֔ת אֲשֶׁר־דִּבֶּ֥ר אֵלֶ֖יךָ לֵאמֹ֑ר נֵֽלְכָ֞ה אַֽחֲרֵ֨י אֱלֹהִ֧ים אֲחֵרִ֛ים אֲשֶׁ֥ר לֹֽא־יְדַעְתָּ֖ם וְנָֽעָבְדֵֽם׃ וְיֵיתֵי אָתָא וּמוֹפְתָא, דְּמַלֵּיל עִמָּךְ לְמֵימַר:  נְהָךְ בָּתַר טָעֲוָת עַמְמַיָּא, דְּלָא יְדַעְתָּנִין--וְנִפְלְחִנִּין. 
יג,ד לֹ֣א תִשְׁמַ֗ע אֶל־דִּבְרֵי֙ הַנָּבִ֣יא הַה֔וּא א֛וֹ אֶל־חוֹלֵ֥ם הַֽחֲל֖וֹם הַה֑וּא כִּ֣י מְנַסֶּ֞ה יְהוָ֤ה אֱלֹֽהֵיכֶם֙ אֶתְכֶ֔ם לָדַ֗עַת הֲיִשְׁכֶ֤ם אֹֽהֲבִים֙ אֶת־יְהוָ֣ה אֱלֹֽהֵיכֶ֔ם בְּכָל־לְבַבְכֶ֖ם וּבְכָל־נַפְשְׁכֶֽם׃ לָא תְּקַבֵּיל, לְפִתְגָמֵי נְבִיָּא הַהוּא, אוֹ מִן חָלֵים חֶלְמָא, הַהוּא:  אֲרֵי מְנַסֵּי יְיָ אֱלָהֲכוֹן, יָתְכוֹן, לְמִדַּע הֲאִיתֵיכוֹן רָחֲמִין יָת יְיָ אֱלָהֲכוֹן, בְּכָל לִבְּכוֹן וּבְכָל נַפְשְׁכוֹן. 
יג,ה אַֽחֲרֵ֨י יְהוָ֧ה אֱלֹֽהֵיכֶ֛ם תֵּלֵ֖כוּ וְאֹת֣וֹ תִירָ֑אוּ וְאֶת־מִצְו‍ֹתָ֤יו תִּשְׁמֹ֨רוּ֙ וּבְקֹל֣וֹ תִשְׁמָ֔עוּ וְאֹת֥וֹ תַֽעֲבֹ֖דוּ וּב֥וֹ תִדְבָּקֽוּן׃ בָּתַר פֻּלְחָנָא דַּייָ אֱלָהֲכוֹן תְּהָכוּן, וְיָתֵיהּ תִּדְחֲלוּן; וְיָת פִּקּוֹדוֹהִי תִּטְּרוּן וּלְמֵימְרֵיהּ תְּקַבְּלוּן, וּקְדָמוֹהִי תִּפְלְחוּן וּלְדַחְלְתֵיהּ תִּתְקָרְבוּן. 
יג,ו וְהַנָּבִ֣יא הַה֡וּא א֣וֹ חֹלֵם֩ הַֽחֲל֨וֹם הַה֜וּא יוּמָ֗ת כִּ֣י דִבֶּר־סָ֠רָה עַל־יְהוָ֨ה אֱלֹֽהֵיכֶ֜ם הַמּוֹצִ֥יא אֶתְכֶ֣ם ׀ מֵאֶ֣רֶץ מִצְרַ֗יִם וְהַפֹּֽדְךָ֙ מִבֵּ֣ית עֲבָדִ֔ים לְהַדִּֽיחֲךָ֙ מִן־הַדֶּ֔רֶךְ אֲשֶׁ֧ר צִוְּךָ֛ יְהוָ֥ה אֱלֹהֶ֖יךָ לָלֶ֣כֶת בָּ֑הּ וּבִֽעַרְתָּ֥ הָרָ֖ע מִקִּרְבֶּֽךָ׃ {ס} וּנְבִיָּא הַהוּא אוֹ חָלֵים חֶלְמָא הַהוּא יִתְקְטִיל, אֲרֵי מַלֵּיל סָטְיָא עַל יְיָ אֱלָהֲכוֹן דְּאַפֵּיק יָתְכוֹן מֵאַרְעָא דְּמִצְרַיִם וּדְפַרְקָךְ מִבֵּית עַבְדּוּתָא--לְאַטְעָיוּתָךְ מִן אוֹרְחָא, דְּפַקְּדָךְ יְיָ אֱלָהָךְ לִמְהָךְ בַּהּ; וּתְפַלֵּי עָבֵיד דְּבִישׁ, מִבֵּינָךְ. {ס}
יג,ז כִּ֣י יְסִֽיתְךָ֡ אָחִ֣יךָ בֶן־אִ֠מֶּךָ אֽוֹ־בִנְךָ֨ אֽוֹ־בִתְּךָ֜ א֣וֹ ׀ אֵ֣שֶׁת חֵיקֶ֗ךָ א֧וֹ רֵֽעֲךָ֛ אֲשֶׁ֥ר כְּנַפְשְׁךָ֖ בַּסֵּ֣תֶר לֵאמֹ֑ר נֵֽלְכָ֗ה וְנַֽעַבְדָה֙ אֱלֹהִ֣ים אֲחֵרִ֔ים אֲשֶׁר֙ לֹ֣א יָדַ֔עְתָּ אַתָּ֖ה וַֽאֲבֹתֶֽיךָ׃ אֲרֵי יִמְלְכִנָּךְ אֲחוּךְ בַּר אִמָּךְ אוֹ בְּרָךְ אוֹ בְּרַתָּךְ אוֹ אִתַּת קְיָמָךְ, אוֹ חַבְרָךְ דִּכְנַפְשָׁךְ--בְּסִתְרָא לְמֵימַר:  נְהָךְ, וְנִפְלַח לְטָעֲוָת עַמְמַיָּא, דְּלָא יְדַעְתָּא, אַתְּ וַאֲבָהָתָךְ. 
יג,ח מֵֽאֱלֹהֵ֣י הָֽעַמִּ֗ים אֲשֶׁר֙ סְבִיבֹ֣תֵיכֶ֔ם הַקְּרֹבִ֣ים אֵלֶ֔יךָ א֖וֹ הָֽרְחֹקִ֣ים מִמֶּ֑ךָּ מִקְצֵ֥ה הָאָ֖רֶץ וְעַד־קְצֵ֥ה הָאָֽרֶץ׃ מִטָּעֲוָת עַמְמַיָּא, דִּבְסַחְרָנֵיכוֹן, דְּקָרִיבִין לָךְ, אוֹ דְּרַחִיקִין מִנָּךְ--מִסְּיָפֵי אַרְעָא, וְעַד סְיָפֵי אַרְעָא. 
יג,ט לֹֽא־תֹאבֶ֣ה ל֔וֹ וְלֹ֥א תִשְׁמַ֖ע אֵלָ֑יו וְלֹֽא־תָח֤וֹס עֵֽינְךָ֙ עָלָ֔יו וְלֹֽא־תַחְמֹ֥ל וְלֹֽא־תְכַסֶּ֖ה עָלָֽיו׃ לָא תֵּיבֵי לֵיהּ, וְלָא תְּקַבֵּיל מִנֵּיהּ; וְלָא תְּחוּס עֵינָךְ עֲלוֹהִי, וְלָא תְּרַחֵים וְלָא תְּכַסֵּי עֲלוֹהִי. 
יג,י כִּ֤י הָרֹג֙ תַּֽהַרְגֶ֔נּוּ יָ֥דְךָ֛ תִּֽהְיֶה־בּ֥וֹ בָרִֽאשׁוֹנָ֖ה לַֽהֲמִית֑וֹ וְיַ֥ד כָּל־הָעָ֖ם בָּאַֽחֲרֹנָֽה׃ אֲרֵי מִקְטָל תִּקְטְלִנֵּיהּ, יְדָךְ תְּהֵי בֵּיהּ בְּקַדְמֵיתָא לְמִקְטְלֵיהּ; וִידָא דְּכָל עַמָּא, בְּבָתְרֵיתָא. 
יג,יא וּסְקַלְתּ֥וֹ בָֽאֲבָנִ֖ים וָמֵ֑ת כִּ֣י בִקֵּ֗שׁ לְהַדִּֽיחֲךָ֙ מֵעַל֙ יְהוָ֣ה אֱלֹהֶ֔יךָ הַמּוֹצִֽיאֲךָ֛ מֵאֶ֥רֶץ מִצְרַ֖יִם מִבֵּ֥ית עֲבָדִֽים׃ וְתִרְגְּמִנֵּיהּ בְּאַבְנַיָּא, וִימוּת:  אֲרֵי בְּעָא, לְאַטְעָיוּתָךְ מִדַּחְלְתָא דַּייָ אֱלָהָךְ, דְּאַפְּקָךְ מֵאַרְעָא דְּמִצְרַיִם, מִבֵּית עַבְדּוּתָא. 
יג,יב וְכָ֨ל־יִשְׂרָאֵ֔ל יִשְׁמְע֖וּ וְיִֽרָא֑וּן וְלֹֽא־יוֹסִ֣פוּ לַֽעֲשׂ֗וֹת כַּדָּבָ֥ר הָרָ֛ע הַזֶּ֖ה בְּקִרְבֶּֽךָ׃ {ס} וְכָל יִשְׂרָאֵל--יִשְׁמְעוּן, וְיִדְחֲלוּן; וְלָא יֵיסְפוּן לְמֶעֱבַד, כְּפִתְגָמָא בִּישָׁא הָדֵין--בֵּינָךְ. {ס}
יג,יג כִּֽי־תִשְׁמַ֞ע בְּאַחַ֣ת עָרֶ֗יךָ אֲשֶׁר֩ יְהוָ֨ה אֱלֹהֶ֜יךָ נֹתֵ֥ן לְךָ֛ לָשֶׁ֥בֶת שָׁ֖ם לֵאמֹֽר׃ אֲרֵי תִּשְׁמַע בַּחֲדָא מִקִּרְוָךְ, דַּייָ אֱלָהָךְ יָהֵיב לָךְ לְמִתַּב תַּמָּן--לְמֵימַר. 
יג,יד יָֽצְא֞וּ אֲנָשִׁ֤ים בְּנֵֽי־בְלִיַּ֨עַל֙ מִקִּרְבֶּ֔ךָ וַיַּדִּ֛יחוּ אֶת־יֹֽשְׁבֵ֥י עִירָ֖ם לֵאמֹ֑ר נֵֽלְכָ֗ה וְנַֽעַבְדָ֛ה אֱלֹהִ֥ים אֲחֵרִ֖ים אֲשֶׁ֥ר לֹֽא־יְדַעְתֶּֽם׃ נְפַקוּ גֻּבְרִין בְּנֵי רִשְׁעָא, מִבֵּינָךְ, וְאַטְעִיאוּ יָת יָתְבֵי קַרְתְּהוֹן, לְמֵימַר:  נְהָךְ, וְנִפְלַח לְטָעֲוָת עַמְמַיָּא--דְּלָא יְדַעְתּוּן. 
יג,טו וְדָֽרַשְׁתָּ֧ וְחָֽקַרְתָּ֛ וְשָֽׁאַלְתָּ֖ הֵיטֵ֑ב וְהִנֵּ֤ה אֱמֶת֙ נָכ֣וֹן הַדָּבָ֔ר נֶֽעֶשְׂתָ֛ה הַתּֽוֹעֵבָ֥ה הַזֹּ֖את בְּקִרְבֶּֽךָ׃ וְתִתְבַּע וְתִבְדּוֹק וְתִשְׁאַל, יָאוּת; וְהָא קֻשְׁטָא כֵּיוָן פִּתְגָמָא, אִתְעֲבֵידַת תּוֹעֵיבְתָא הָדָא בֵּינָךְ. 
יג,טז הַכֵּ֣ה תַכֶּ֗ה אֶת־יֹ֥שְׁבֵ֛י הָעִ֥יר הַהִ֖וא לְפִי־חָ֑רֶב הַֽחֲרֵ֨ם אֹתָ֧הּ וְאֶת־כָּל־אֲשֶׁר־בָּ֛הּ וְאֶת־בְּהֶמְתָּ֖הּ לְפִי־חָֽרֶב׃ מִמְחָא תִּמְחֵי, יָת יָתְבֵי קַרְתָּא הַהִיא--לְפִתְגָם דְּחָרֶב:  גַּמַּר יָתַהּ וְיָת כָּל דְּבַהּ וְיָת בְּעִירַהּ, לְפִתְגָם דְּחָרֶב. 
יג,יז וְאֶת־כָּל־שְׁלָלָ֗הּ תִּקְבֹּץ֮ אֶל־תּ֣וֹךְ רְחֹבָהּ֒ וְשָֽׂרַפְתָּ֨ בָאֵ֜שׁ אֶת־הָעִ֤יר וְאֶת־כָּל־שְׁלָלָהּ֙ כָּלִ֔יל לַֽיהוָ֖ה אֱלֹהֶ֑יךָ וְהָֽיְתָה֙ תֵּ֣ל עוֹלָ֔ם לֹ֥א תִבָּנֶ֖ה עֽוֹד׃ וְיָת כָּל עֲדָאַהּ, תִּכְנוֹשׁ לְגוֹ פְּתָאַהּ, וְתוֹקֵיד בְּנוּרָא יָת קַרְתָּא וְיָת כָּל עֲדָאַהּ גְּמִיר, קֳדָם יְיָ אֱלָהָךְ; וּתְהֵי תֵּל חָרוּב לְעָלַם, לָא תִּתְבְּנֵי עוֹד. 
יג,יח וְלֹֽא־יִדְבַּ֧ק בְּיָֽדְךָ֛ מְא֖וּמָה מִן־הַחֵ֑רֶם לְמַעַן֩ יָשׁ֨וּב יְהוָ֜ה מֵֽחֲר֣וֹן אַפּ֗וֹ וְנָֽתַן־לְךָ֤ רַֽחֲמִים֙ וְרִֽחַמְךָ֣ וְהִרְבֶּ֔ךָ כַּֽאֲשֶׁ֥ר נִשְׁבַּ֖ע לַֽאֲבֹתֶֽיךָ׃ וְלָא יִדְבַּק בִּידָךְ מִדָּעַם, מִן חֶרְמָא--בְּדִיל דִּיתוּב יְיָ מִתְּקוֹף רֻגְזֵיהּ, וְיִתֵּין לָךְ רַחֲמִין וִירַחֵים עֲלָךְ וְיַסְגֵּינָךְ, כְּמָא דְּקַיֵּים, לַאֲבָהָתָךְ. 
יג,יט כִּ֣י תִשְׁמַ֗ע בְּקוֹל֙ יְהוָ֣ה אֱלֹהֶ֔יךָ לִשְׁמֹר֙ אֶת־כָּל־מִצְו‍ֹתָ֔יו אֲשֶׁ֛ר אָֽנֹכִ֥י מְצַוְּךָ֖ הַיּ֑וֹם לַֽעֲשׂוֹת֙ הַיָּשָׁ֔ר בְּעֵינֵ֖י יְהוָ֥ה אֱלֹהֶֽיךָ׃ {ס} אֲרֵי תְּקַבֵּיל, לְמֵימְרָא דַּייָ אֱלָהָךְ, לְמִטַּר יָת כָּל פִּקּוֹדוֹהִי, דַּאֲנָא מְפַקֵּיד לָךְ יוֹמָא דֵּין--לְמֶעֱבַד, דְּכָשַׁר, קֳדָם, יְיָ אֱלָהָךְ. {ס}
יד,א בָּנִ֣ים אַתֶּ֔ם לַֽיהוָ֖ה אֱלֹֽהֵיכֶ֑ם לֹ֣א תִתְגֹּֽדְד֗וּ וְלֹֽא־תָשִׂ֧ימוּ קָרְחָ֛ה בֵּ֥ין עֵֽינֵיכֶ֖ם לָמֵֽת׃ בְּנִין אַתּוּן, קֳדָם יְיָ אֱלָהֲכוֹן:  לָא תִּתְהָמְמוּן, וְלָא תְּשַׁוּוֹן מְרַט בֵּין עֵינֵיכוֹן--עַל מִית. 
יד,ב כִּ֣י עַ֤ם קָדוֹשׁ֙ אַתָּ֔ה לַֽיהוָ֖ה אֱלֹהֶ֑יךָ וּבְךָ֞ בָּחַ֣ר יְהוָ֗ה לִֽהְי֥וֹת לוֹ֙ לְעַ֣ם סְגֻלָּ֔ה מִכֹּל֙ הָֽעַמִּ֔ים אֲשֶׁ֖ר עַל־פְּנֵ֥י הָֽאֲדָמָֽה׃ {ס} אֲרֵי עַם קַדִּישׁ אַתְּ, קֳדָם יְיָ אֱלָהָךְ; וּבָךְ אִתְרְעִי יְיָ, לְמִהְוֵי לֵיהּ לְעַם חַבִּיב, מִכֹּל עַמְמַיָּא, דְּעַל אַפֵּי אַרְעָא. {ס}
יד,ג לֹ֥א תֹאכַ֖ל כָּל־תּֽוֹעֵבָֽה׃ לָא תֵּיכוֹל, כָּל דִּמְרַחַק. 
יד,ד זֹ֥את הַבְּהֵמָ֖ה אֲשֶׁ֣ר תֹּאכֵ֑לוּ שׁ֕וֹר שֵׂ֥ה כְשָׂבִ֖ים וְשֵׂ֥ה עִזִּֽים׃ דֵּין בְּעִירָא, דְּתֵיכְלוּן:  תּוֹרִין, אִמְּרִין דְּרַחְלִין וּגְדִין דְּעִזִּין. 
יד,ה אַיָּ֥ל וּצְבִ֖י וְיַחְמ֑וּר וְאַקּ֥וֹ וְדִישֹׁ֖ן וּתְא֥וֹ וָזָֽמֶר׃ אַיְלָא וְטַבְיָא, וְיַחְמוּרָא; וְיַעְלָא וְרֵימָא, וְתֻרְבְּלָא וְדֵיצָא. 
יד,ו וְכָל־בְּהֵמָ֞ה מַפְרֶ֣סֶת פַּרְסָ֗ה וְשֹׁסַ֤עַת שֶׁ֨סַע֙ שְׁתֵּ֣י פְרָס֔וֹת מַֽעֲלַ֥ת גֵּרָ֖ה בַּבְּהֵמָ֑ה אֹתָ֖הּ תֹּאכֵֽלוּ׃ וְכָל בְּעִירָא דִּסְדִיקָא פַּרְסְתַהּ, וּמַטְלְפָן טִלְפִין תַּרְתֵּין פַּרְסָתַהּ, מַסְּקָא פִּשְׁרָא, בִּבְעִירָא--יָתַהּ, תֵּיכְלוּן. 
יד,ז אַ֣ךְ אֶת־זֶ֞ה לֹ֤א תֹֽאכְלוּ֙ מִמַּֽעֲלֵ֣י הַגֵּרָ֔ה וּמִמַּפְרִיסֵ֥י הַפַּרְסָ֖ה הַשְּׁסוּעָ֑ה אֶֽת־הַ֠גָּמָל וְאֶת־הָֽאַרְנֶ֨בֶת וְאֶת־הַשָּׁפָ֜ן כִּֽי־מַעֲלֵ֧ה גֵרָ֣ה הֵ֗מָּה וּפַרְסָה֙ לֹ֣א הִפְרִ֔יסוּ טְמֵאִ֥ים הֵ֖ם לָכֶֽם׃ בְּרַם יָת דֵּין לָא תֵּיכְלוּן, מִמַּסְּקֵי פִּשְׁרָא, וּמִסְּדִיקֵי פַּרְסְתָא, מַטְלְפֵי טִלְפַיָּא:  יָת גַּמְלָא וְיָת אַרְנְבָא וְיָת טַבְזָא אֲרֵי מַסְּקֵי פִּשְׁרָא אִנּוּן, וּפַרְסַתְהוֹן לָא סְדִיקָא--מְסָאֲבִין אִנּוּן, לְכוֹן. 
יד,ח וְאֶת־הַֽ֠חֲזִיר כִּֽי־מַפְרִ֨יס פַּרְסָ֥ה הוּא֙ וְלֹ֣א גֵרָ֔ה טָמֵ֥א ה֖וּא לָכֶ֑ם מִבְּשָׂרָם֙ לֹ֣א תֹאכֵ֔לוּ וּבְנִבְלָתָ֖ם לֹ֥א תִגָּֽעוּ׃ {ס} וְיָת חֲזִירָא אֲרֵי סְדִיק פַּרְסְתָא הוּא, וְלָא פָּשַׁר--מְסָאַב הוּא, לְכוֹן; מִבִּסְרְהוֹן לָא תֵּיכְלוּן, וּבִנְבִילַתְהוֹן לָא תִּקְרְבוּן. {ס}
יד,ט אֶת־זֶה֙ תֹּֽאכְל֔וּ מִכֹּ֖ל אֲשֶׁ֣ר בַּמָּ֑יִם כֹּ֧ל אֲשֶׁר־ל֛וֹ סְנַפִּ֥יר וְקַשְׂקֶ֖שֶׂת תֹּאכֵֽלוּ׃ יָת דֵּין, תֵּיכְלוּן, מִכֹּל, דִּבְמַיָּא:  כֹּל דְּלֵיהּ צִיצִין וְקַלְפִין, תֵּיכְלוּן. 
יד,י וְכֹ֨ל אֲשֶׁ֧ר אֵֽין־ל֛וֹ סְנַפִּ֥יר וְקַשְׂקֶ֖שֶׂת לֹ֣א תֹאכֵ֑לוּ טָמֵ֥א ה֖וּא לָכֶֽם׃ {ס} וְכֹל דְּלֵית לֵיהּ צִיצִין וְקַלְפִין, לָא תֵּיכְלוּן--מְסָאַב הוּא, לְכוֹן. {ס}
יד,יא כָּל־צִפּ֥וֹר טְהֹרָ֖ה תֹּאכֵֽלוּ׃ כָּל צִפַּר דְּכֵי, תֵּיכְלוּן. 
יד,יב וְזֶ֕ה אֲשֶׁ֥ר לֹֽא־תֹאכְל֖וּ מֵהֶ֑ם הַנֶּ֥שֶׁר וְהַפֶּ֖רֶס וְהָֽעָזְנִיָּֽה׃ וְדֵין, דְּלָא תֵּיכְלוּן מִנְּהוֹן:  נִשְׁרָא וְעָר, וְעָזְיָא. 
יד,יג וְהָֽרָאָה֙ וְאֶת־הָ֣אַיָּ֔ה וְהַדַּיָּ֖ה לְמִינָֽהּ׃ וּבַת כַּנְפָא, וְטָרָפִיתָא, וְדַיְתָא, לִזְנַהּ. 
יד,יד וְאֵ֥ת כָּל־עֹרֵ֖ב לְמִינֽוֹ׃ וְיָת כָּל עוֹרְבָא, לִזְנֵיהּ. 
יד,טו וְאֵת֙ בַּ֣ת הַֽיַּעֲנָ֔ה וְאֶת־הַתַּחְמָ֖ס וְאֶת־הַשָּׁ֑חַף וְאֶת־הַנֵּ֖ץ לְמִינֵֽהוּ׃ וְיָת בַּת נַעָמִיתָא, וְצִיצָא וְצִפַּר שַׁחְפָּא; וְנַצָּא, לִזְנוֹהִי. 
יד,טז אֶת־הַכּ֥וֹס וְאֶת־הַיַּנְשׁ֖וּף וְהַתִּנְשָֽׁמֶת׃ וְקָדְיָא וְקִפּוּפָא, וּבָוְתָא. 
יד,יז וְהַקָּאָ֥ת וְאֶֽת־הָרָחָ֖מָה וְאֶת־הַשָּׁלָֽךְ׃ וְקָתָא וִירַקְרֵיקָא, וְשָׁלֵינוּנָא. 
יד,יח וְהַ֣חֲסִידָ֔ה וְהָֽאֲנָפָ֖ה לְמִינָ֑הּ וְהַדּֽוּכִיפַ֖ת וְהָֽעֲטַלֵּֽף׃ וְחָוָרִיתָא, וְאִבּוֹ לִזְנַהּ; וְנַגָּר טוּרָא, וַעֲטַלֵּיפָא. 
יד,יט וְכֹל֙ שֶׁ֣רֶץ הָע֔וֹף טָמֵ֥א ה֖וּא לָכֶ֑ם לֹ֖א יֵֽאָכֵֽלוּ׃ וְכֹל רִחְשָׁא דְּעוֹפָא, מְסָאַב הוּא לְכוֹן:  לָא, יִתְאַכְלוּן. 
יד,כ כָּל־ע֥וֹף טָה֖וֹר תֹּאכֵֽלוּ׃ כָּל עוֹף דְּכֵי, תֵּיכְלוּן. 
יד,כא לֹ֣א תֹֽאכְל֣וּ כָל־נְ֠בֵלָה לַגֵּ֨ר אֲשֶׁר־בִּשְׁעָרֶ֜יךָ תִּתְּנֶ֣נָּה וַֽאֲכָלָ֗הּ א֤וֹ מָכֹר֙ לְנָכְרִ֔י כִּ֣י עַ֤ם קָדוֹשׁ֙ אַתָּ֔ה לַֽיהוָ֖ה אֱלֹהֶ֑יךָ לֹֽא־תְבַשֵּׁ֥ל גְּדִ֖י בַּֽחֲלֵ֥ב אִמּֽוֹ׃ {פ} לָא תֵּיכְלוּן כָּל נְבִילָא לְתוֹתַב עֲרַל דִּבְקִרְוָךְ תִּתְּנִנַּהּ וְיֵיכְלִנַּהּ, אוֹ תְּזַבְּנִנַּהּ לְבַר עַמְמִין--אֲרֵי עַם קַדִּישׁ אַתְּ, קֳדָם יְיָ אֱלָהָךְ; לָא תֵּיכְלוּן בְּסַר, בַּחֲלַב. {פ}
יד,כב עַשֵּׂ֣ר תְּעַשֵּׂ֔ר אֵ֖ת כָּל־תְּבוּאַ֣ת זַרְעֶ֑ךָ הַיֹּצֵ֥א הַשָּׂדֶ֖ה שָׁנָ֥ה שָׁנָֽה׃ עַסָּרָא תְּעַסַּר, יָת כָּל עַלְלַת זַרְעָךְ, דְּיַפֵּיק חַקְלָא, שְׁנָא שְׁנָא. 
יד,כג וְאָֽכַלְתָּ֞ לִפְנֵ֣י ׀ יְהוָ֣ה אֱלֹהֶ֗יךָ בַּמָּק֣וֹם אֲשֶׁר־יִבְחַר֮ לְשַׁכֵּ֣ן שְׁמ֣וֹ שָׁם֒ מַעְשַׂ֤ר דְּגָֽנְךָ֙ תִּֽירֹשְׁךָ֣ וְיִצְהָרֶ֔ךָ וּבְכֹרֹ֥ת בְּקָֽרְךָ֖ וְצֹאנֶ֑ךָ לְמַ֣עַן תִּלְמַ֗ד לְיִרְאָ֛ה אֶת־יְהוָ֥ה אֱלֹהֶ֖יךָ כָּל־הַיָּמִֽים׃ וְתֵיכוֹל קֳדָם יְיָ אֱלָהָךְ, בְּאַתְרָא דְּיִתְרְעֵי לְאַשְׁרָאָה שְׁכִינְתֵיהּ תַּמָּן, מַעְסַר עֲבוּרָךְ חַמְרָךְ וּמִשְׁחָךְ, וּבְכוֹרֵי תּוֹרָךְ וְעָנָךְ:  בְּדִיל דְּתֵילַף, לְמִדְחַל קֳדָם יְיָ אֱלָהָךְ--כָּל יוֹמַיָּא. 
יד,כד וְכִֽי־יִרְבֶּ֨ה מִמְּךָ֜ הַדֶּ֗רֶךְ כִּ֣י לֹ֣א תוּכַל֮ שְׂאֵתוֹ֒ כִּֽי־יִרְחַ֤ק מִמְּךָ֙ הַמָּק֔וֹם אֲשֶׁ֤ר יִבְחַר֙ יְהוָ֣ה אֱלֹהֶ֔יךָ לָשׂ֥וּם שְׁמ֖וֹ שָׁ֑ם כִּ֥י יְבָֽרֶכְךָ֖ יְהוָ֥ה אֱלֹהֶֽיךָ׃ וַאֲרֵי יִסְגֵּי מִנָּךְ אוֹרְחָא, אֲרֵי לָא תִּכּוֹל לְמִטְּלֵיהּ--אֲרֵי יִתְרַחַק מִנָּךְ אַתְרָא, דְּיִתְרְעֵי יְיָ אֱלָהָךְ לְאַשְׁרָאָה שְׁכִינְתֵיהּ תַּמָּן:  אֲרֵי יְבָרְכִנָּךְ, יְיָ אֱלָהָךְ. 
יד,כה וְנָֽתַתָּ֖ה בַּכָּ֑סֶף וְצַרְתָּ֤ הַכֶּ֨סֶף֙ בְּיָ֣דְךָ֔ וְהָֽלַכְתָּ֙ אֶל־הַמָּק֔וֹם אֲשֶׁ֥ר יִבְחַ֛ר יְהוָ֥ה אֱלֹהֶ֖יךָ בּֽוֹ׃ וְתִתֵּין, בְּכַסְפָּא; וּתְצוּר כַּסְפָּא, בִּידָךְ, וּתְהָךְ לְאַתְרָא, דְּיִתְרְעֵי יְיָ אֱלָהָךְ בֵּיהּ. 
יד,כו וְנָֽתַתָּ֣ה הַכֶּ֡סֶף בְּכֹל֩ אֲשֶׁר־תְּאַוֶּ֨ה נַפְשְׁךָ֜ בַּבָּקָ֣ר וּבַצֹּ֗אן וּבַיַּ֨יִן֙ וּבַשֵּׁכָ֔ר וּבְכֹ֛ל אֲשֶׁ֥ר תִּֽשְׁאָלְךָ֖ נַפְשֶׁ֑ךָ וְאָכַ֣לְתָּ שָּׁ֗ם לִפְנֵי֙ יְהוָ֣ה אֱלֹהֶ֔יךָ וְשָֽׂמַחְתָּ֖ אַתָּ֥ה וּבֵיתֶֽךָ׃ וְתִתֵּין כַּסְפָּא בְּכֹל דְּתִתְרְעֵי נַפְשָׁךְ בְּתוֹרֵי וּבְעָנָא, וּבַחֲמַר חֲדַת וְעַתִּיק, וּבְכֹל דְּתִשְׁאֲלִנָּךְ, נַפְשָׁךְ; וְתֵיכוֹל תַּמָּן, קֳדָם יְיָ אֱלָהָךְ, וְתִחְדֵּי, אַתְּ וַאֲנָשׁ בֵּיתָךְ. 
יד,כז וְהַלֵּוִ֥י אֲשֶׁר־בִּשְׁעָרֶ֖יךָ לֹ֣א תַֽעַזְבֶ֑נּוּ כִּ֣י אֵ֥ין ל֛וֹ חֵ֥לֶק וְנַֽחֲלָ֖ה עִמָּֽךְ׃ {ס} וְלֵיוָאָה דִּבְקִרְוָךְ, לָא תְּרַחַקִנֵּיהּ:  אֲרֵי לֵית לֵיהּ חוּלָק וְאַחְסָנָא, עִמָּךְ. {ס}
יד,כח מִקְצֵ֣ה ׀ שָׁלֹ֣שׁ שָׁנִ֗ים תּוֹצִיא֙ אֶת־כָּל־מַעְשַׂר֙ תְּבוּאָ֣תְךָ֔ בַּשָּׁנָ֖ה הַהִ֑וא וְהִנַּחְתָּ֖ בִּשְׁעָרֶֽיךָ׃ מִסּוֹף תְּלָת שְׁנִין, תַּפֵּיק יָת כָּל מַעְסַר עֲלַלְתָּךְ, בְּשַׁתָּא, הַהִיא; וְתַצְנַע, בְּקִרְוָךְ. 
יד,כט וּבָ֣א הַלֵּוִ֡י כִּ֣י אֵֽין־לוֹ֩ חֵ֨לֶק וְנַֽחֲלָ֜ה עִמָּ֗ךְ וְ֠הַגֵּר וְהַיָּת֤וֹם וְהָֽאַלְמָנָה֙ אֲשֶׁ֣ר בִּשְׁעָרֶ֔יךָ וְאָֽכְל֖וּ וְשָׂבֵ֑עוּ לְמַ֤עַן יְבָֽרֶכְךָ֙ יְהוָ֣ה אֱלֹהֶ֔יךָ בְּכָל־מַֽעֲשֵׂ֥ה יָֽדְךָ֖ אֲשֶׁ֥ר תַּֽעֲשֶֽׂה׃ {ס} וְיֵיתֵי לֵיוָאָה אֲרֵי לֵית לֵיהּ חוּלָק וְאַחְסָנָא עִמָּךְ, וְגִיּוֹרָא וְיַתְמָא וְאַרְמַלְתָּא דִּבְקִרְוָךְ, וְיֵיכְלוּן, וְיִסְבְּעוּן--בְּדִיל דִּיבָרְכִנָּךְ יְיָ אֱלָהָךְ, בְּכָל עוּבָדֵי יְדָךְ דְּתַעֲבֵיד. {ס}
טו,א מִקֵּ֥ץ שֶֽׁבַע־שָׁנִ֖ים תַּֽעֲשֶׂ֥ה שְׁמִטָּֽה׃ מִסּוֹף שְׁבַע שְׁנִין, תַּעֲבֵיד שְׁמִטְּתָא. 
טו,ב וְזֶה֮ דְּבַ֣ר הַשְּׁמִטָּה֒ שָׁמ֗וֹט כָּל־בַּ֨עַל֙ מַשֵּׁ֣ה יָד֔וֹ אֲשֶׁ֥ר יַשֶּׁ֖ה בְּרֵעֵ֑הוּ לֹֽא־יִגֹּ֤שׂ אֶת־רֵעֵ֨הוּ֙ וְאֶת־אָחִ֔יו כִּֽי־קָרָ֥א שְׁמִטָּ֖ה לַֽיהוָֽה׃ וְדֵין, פִּתְגָם שְׁמִטְּתָא--דְּיַשְׁמֵיט כָּל גְּבַר מָרֵי רְשׁוּ, דְּיִרְשֵׁי בְּחַבְרֵיהּ:  לָא יִתְבַּע מִן חַבְרֵיהּ וּמִן אֲחוּהִי, אֲרֵי קְרָא שְׁמִטְּתָא קֳדָם יְיָ. 
טו,ג אֶת־הַנָּכְרִ֖י תִּגֹּ֑שׂ וַֽאֲשֶׁ֨ר יִֽהְיֶ֥ה לְךָ֛ אֶת־אָחִ֖יךָ תַּשְׁמֵ֥ט יָדֶֽךָ׃ מִן בַּר עַמְמִין, תִּתְבַּע; וְדִיהֵי לָךְ עִם אֲחוּךְ, תַּשְׁמֵיט יְדָךְ. 
טו,ד אֶ֕פֶס כִּ֛י לֹ֥א יִֽהְיֶה־בְּךָ֖ אֶבְי֑וֹן כִּֽי־בָרֵ֤ךְ יְבָֽרֶכְךָ֙ יְהוָ֔ה בָּאָ֕רֶץ אֲשֶׁר֙ יְהוָ֣ה אֱלֹהֶ֔יךָ נֹֽתֵן־לְךָ֥ נַֽחֲלָ֖ה לְרִשְׁתָּֽהּ׃ לְחוֹד, אֲרֵי לָא יְהֵי בָּךְ מִסְכֵּינָא:  אֲרֵי בָּרָכָא יְבָרְכִנָּךְ, יְיָ, בְּאַרְעָא, דַּייָ אֱלָהָךְ יָהֵיב לָךְ אַחְסָנָא לְמֵירְתַהּ. 
טו,ה רַ֚ק אִם־שָׁמ֣וֹעַ תִּשְׁמַ֔ע בְּק֖וֹל יְהוָ֣ה אֱלֹהֶ֑יךָ לִשְׁמֹ֤ר לַֽעֲשׂוֹת֙ אֶת־כָּל־הַמִּצְוָ֣ה הַזֹּ֔את אֲשֶׁ֛ר אָֽנֹכִ֥י מְצַוְּךָ֖ הַיּֽוֹם׃ לְחוֹד אִם קַבָּלָא תְּקַבֵּיל, לְמֵימְרָא דַּייָ אֱלָהָךְ, לְמִטַּר לְמֶעֱבַד יָת כָּל תַּפְקֵידְתָּא הָדָא, דַּאֲנָא מְפַקֵּיד לָךְ יוֹמָא דֵּין. 
טו,ו כִּֽי־יְהוָ֤ה אֱלֹהֶ֨יךָ֙ בֵּֽרַכְךָ֔ כַּֽאֲשֶׁ֖ר דִּבֶּר־לָ֑ךְ וְהַֽעֲבַטְתָּ֞ גּוֹיִ֣ם רַבִּ֗ים וְאַתָּה֙ לֹ֣א תַֽעֲבֹ֔ט וּמָֽשַׁלְתָּ֙ בְּגוֹיִ֣ם רַבִּ֔ים וּבְךָ֖ לֹ֥א יִמְשֹֽׁלוּ׃ {ס} אֲרֵי יְיָ אֱלָהָךְ בָּרְכָךְ, כְּמָא דְּמַלֵּיל לָךְ; וְתוֹזֵיף לְעַמְמִין סַגִּיאִין, וְאַתְּ לָא תְּזִיף, וְתִשְׁלוֹט בְּעַמְמִין סַגִּיאִין, וּבָךְ לָא יִשְׁלְטוּן. {ס}
טו,ז כִּֽי־יִהְיֶה֩ בְךָ֨ אֶבְי֜וֹן מֵֽאַחַ֤ד אַחֶ֨יךָ֙ בְּאַחַ֣ד שְׁעָרֶ֔יךָ בְּאַ֨רְצְךָ֔ אֲשֶׁר־יְהוָ֥ה אֱלֹהֶ֖יךָ נֹתֵ֣ן לָ֑ךְ לֹ֧א תְאַמֵּ֣ץ אֶת־לְבָֽבְךָ֗ וְלֹ֤א תִקְפֹּץ֙ אֶת־יָ֣דְךָ֔ מֵֽאָחִ֖יךָ הָֽאֶבְיֽוֹן׃ אֲרֵי יְהֵי בָּךְ מִסְכֵּינָא חַד מֵאֲחָךְ, בַּחֲדָא מִקִּרְוָךְ, בְּאַרְעָךְ, דַּייָ אֱלָהָךְ יָהֵיב לָךְ--לָא תְּתַקֵּיף יָת לִבָּךְ, וְלָא תִּקְפּוֹץ יָת יְדָךְ, מֵאֲחוּךְ, מִסְכֵּינָא. 
טו,ח כִּֽי־פָתֹ֧חַ תִּפְתַּ֛ח אֶת־יָֽדְךָ֖ ל֑וֹ וְהַֽעֲבֵט֙ תַּֽעֲבִיטֶ֔נּוּ דֵּ֚י מַחְסֹר֔וֹ אֲשֶׁ֥ר יֶחְסַ֖ר לֽוֹ׃ אֲרֵי מִפְתָּח תִּפְתַּח יָת יְדָךְ, לֵיהּ; וְאוֹזָפָא, תּוֹזְפִנֵּיהּ, כְּמִסַּת חֻסְרָנֵיהּ, דְּחַסִּיר לֵיהּ. 
טו,ט הִשָּׁ֣מֶר לְךָ֡ פֶּן־יִֽהְיֶ֣ה דָבָר֩ עִם־לְבָֽבְךָ֨ בְלִיַּ֜עַל לֵאמֹ֗ר קָֽרְבָ֣ה שְׁנַֽת־הַשֶּׁבַע֮ שְׁנַ֣ת הַשְּׁמִטָּה֒ וְרָעָ֣ה עֵֽינְךָ֗ בְּאָחִ֨יךָ֙ הָֽאֶבְי֔וֹן וְלֹ֥א תִתֵּ֖ן ל֑וֹ וְקָרָ֤א עָלֶ֨יךָ֙ אֶל־יְהוָ֔ה וְהָיָ֥ה בְךָ֖ חֵֽטְא׃ אִסְתְּמַר לָךְ דִּלְמָא יְהֵי פִּתְגָם עִם לִבָּךְ בִּרְשַׁע לְמֵימַר, קְרֵיבַת שַׁתָּא דִּשְׁבִיעֵיתָא שַׁתָּא דִּשְׁמִטְּתָא, וְתִבְאַשׁ עֵינָךְ בַּאֲחוּךְ מִסְכֵּינָא, וְלָא תִּתֵּין לֵיהּ; וְיִקְרֵי עֲלָךְ קֳדָם יְיָ, וִיהֵי בָּךְ חוֹבָא. 
טו,י נָת֤וֹן תִּתֵּן֙ ל֔וֹ וְלֹֽא־יֵרַ֥ע לְבָֽבְךָ֖ בְּתִתְּךָ֣ ל֑וֹ כִּ֞י בִּגְלַ֣ל ׀ הַדָּבָ֣ר הַזֶּ֗ה יְבָֽרֶכְךָ֙ יְהוָ֣ה אֱלֹהֶ֔יךָ בְּכָֽל־מַעֲשֶׂ֔ךָ וּבְכֹ֖ל מִשְׁלַ֥ח יָדֶֽךָ׃ מִתָּן תִּתֵּין לֵיהּ, וְלָא יִבְאַשׁ לִבָּךְ בְּמִתְּנָךְ לֵיהּ:  אֲרֵי בְּדִיל פִּתְגָמָא הָדֵין, יְבָרְכִנָּךְ יְיָ אֱלָהָךְ, בְּכָל עוּבָדָךְ, וּבְכֹל אוֹשָׁטוּת יְדָךְ. 
טו,יא כִּ֛י לֹֽא־יֶחְדַּ֥ל אֶבְי֖וֹן מִקֶּ֣רֶב הָאָ֑רֶץ עַל־כֵּ֞ן אָֽנֹכִ֤י מְצַוְּךָ֙ לֵאמֹ֔ר פָּ֠תֹחַ תִּפְתַּ֨ח אֶת־יָֽדְךָ֜ לְאָחִ֧יךָ לַֽעֲנִיֶּ֛ךָ וּלְאֶבְיֹֽנְךָ֖ בְּאַרְצֶֽךָ׃ {ס} אֲרֵי לָא יִפְסוּק מִסְכֵּינָא, מִגּוֹ אַרְעָא; עַל כֵּין אֲנָא מְפַקֵּיד לָךְ, לְמֵימַר, מִפְתָּח תִּפְתַּח יָת יְדָךְ לַאֲחוּךְ לְעַנְיָךְ וּלְמִסְכֵּינָךְ, בְּאַרְעָךְ. {ס}
טו,יב כִּֽי־יִמָּכֵ֨ר לְךָ֜ אָחִ֣יךָ הָֽעִבְרִ֗י א֚וֹ הָֽעִבְרִיָּ֔ה וַעֲבָֽדְךָ֖ שֵׁ֣שׁ שָׁנִ֑ים וּבַשָּׁנָה֙ הַשְּׁבִיעִ֔ת תְּשַׁלְּחֶ֥נּוּ חָפְשִׁ֖י מֵֽעִמָּֽךְ׃ אֲרֵי יִזְדַּבַּן לָךְ אֲחוּךְ בַּר יִשְׂרָאֵל, אוֹ בַּת יִשְׂרָאֵל--וְיִפְלְחִנָּךְ, שֵׁית שְׁנִין; וּבְשַׁתָּא, שְׁבִיעֵיתָא, תִּפְטְרִנֵּיהּ בַּר חוֹרִין, מֵעִמָּךְ. 
טו,יג וְכִֽי־תְשַׁלְּחֶ֥נּוּ חָפְשִׁ֖י מֵֽעִמָּ֑ךְ לֹ֥א תְשַׁלְּחֶ֖נּוּ רֵיקָֽם׃ וַאֲרֵי תִּפְטְרִנֵּיהּ בַּר חוֹרִין, מֵעִמָּךְ--לָא תִּפְטְרִנֵּיהּ, רֵיקָן. 
טו,יד הַֽעֲנֵ֤יק תַּֽעֲנִיק֙ ל֔וֹ מִצֹּ֣אנְךָ֔ וּמִֽגָּרְנְךָ֖ וּמִיִּקְבֶ֑ךָ אֲשֶׁ֧ר בֵּֽרַכְךָ֛ יְהוָ֥ה אֱלֹהֶ֖יךָ תִּתֶּן־לֽוֹ׃ אַפְרָשָׁא תַּפְרֵישׁ, לֵיהּ, מֵעָנָךְ, וּמֵאִדְּרָךְ וּמִמַּעֲצַרְתָּךְ:  דְּבָרְכָךְ יְיָ אֱלָהָךְ, תִּתֵּין לֵיהּ. 
טו,טו וְזָֽכַרְתָּ֗ כִּ֣י עֶ֤בֶד הָיִ֨יתָ֙ בְּאֶ֣רֶץ מִצְרַ֔יִם וַֽיִּפְדְּךָ֖ יְהוָ֣ה אֱלֹהֶ֑יךָ עַל־כֵּ֞ן אָֽנֹכִ֧י מְצַוְּךָ֛ אֶת־הַדָּבָ֥ר הַזֶּ֖ה הַיּֽוֹם׃ וְתִדְכַר, אֲרֵי עַבְדָּא הֲוֵיתָא בְּאַרְעָא דְּמִצְרַיִם, וּפַרְקָךְ, יְיָ אֱלָהָךְ; עַל כֵּין אֲנָא מְפַקֵּיד לָךְ, יָת פִּתְגָמָא הָדֵין--יוֹמָא דֵּין. 
טו,טז וְהָיָה֙ כִּֽי־יֹאמַ֣ר אֵלֶ֔יךָ לֹ֥א אֵצֵ֖א מֵֽעִמָּ֑ךְ כִּ֤י אֲהֵֽבְךָ֙ וְאֶת־בֵּיתֶ֔ךָ כִּי־ט֥וֹב ל֖וֹ עִמָּֽךְ׃ וִיהֵי אֲרֵי יֵימַר לָךְ, לָא אֶפּוֹק מֵעִמָּךְ:  אֲרֵי רִיחֲמָךְ וְלַאֲנָשׁ בֵּיתָךְ, אֲרֵי טָב לֵיהּ עִמָּךְ. 
טו,יז וְלָֽקַחְתָּ֣ אֶת־הַמַּרְצֵ֗עַ וְנָֽתַתָּ֤ה בְאָזְנוֹ֙ וּבַדֶּ֔לֶת וְהָיָ֥ה לְךָ֖ עֶ֣בֶד עוֹלָ֑ם וְאַ֥ף לַאֲמָֽתְךָ֖ תַּֽעֲשֶׂה־כֵּֽן׃ וְתִסַּב יָת מַרְצְעָא, וְתִתֵּין בְּאֻדְנֵיהּ וּבְדַשָּׁא, וִיהֵי לָךְ, עֶבֶד פָּלַח לְעָלַם; וְאַף לְאַמְתָּךְ, תַּעֲבֵיד כֵּין. 
טו,יח לֹֽא־יִקְשֶׁ֣ה בְעֵינֶ֗ךָ בְּשַׁלֵּֽחֲךָ֨ אֹת֤וֹ חָפְשִׁי֙ מֵֽעִמָּ֔ךְ כִּ֗י מִשְׁנֶה֙ שְׂכַ֣ר שָׂכִ֔יר עֲבָֽדְךָ֖ שֵׁ֣שׁ שָׁנִ֑ים וּבֵֽרַכְךָ֙ יְהוָ֣ה אֱלֹהֶ֔יךָ בְּכֹ֖ל אֲשֶׁ֥ר תַּֽעֲשֶֽׂה׃ {פ} לָא יִקְשֵׁי בְּעֵינָךְ, בְּמִפְטְרָךְ יָתֵיהּ בַּר חוֹרִין מֵעִמָּךְ--אֲרֵי עַל חַד תְּרֵין כַּאֲגַר אֲגִירָא, פַּלְחָךְ שֵׁית שְׁנִין; וִיבָרְכִנָּךְ יְיָ אֱלָהָךְ, בְּכֹל דְּתַעֲבֵיד. {פ}
טו,יט כָּֽל־הַבְּכ֡וֹר אֲשֶׁר֩ יִוָּלֵ֨ד בִּבְקָֽרְךָ֤ וּבְצֹֽאנְךָ֙ הַזָּכָ֔ר תַּקְדִּ֖ישׁ לַֽיהוָ֣ה אֱלֹהֶ֑יךָ לֹ֤א תַֽעֲבֹד֙ בִּבְכֹ֣ר שׁוֹרֶ֔ךָ וְלֹ֥א תָגֹ֖ז בְּכ֥וֹר צֹאנֶֽךָ׃ כָּל בֻּכְרָא דְּיִתְיְלֵיד בְּתוֹרָךְ וּבְעָנָךְ, דִּכְרִין--תַּקְדֵּישׁ, קֳדָם יְיָ אֱלָהָךְ:  לָא תִּפְלַח בְּבֻכְרָא דְּתוֹרָךְ, וְלָא תִּגּוֹז בֻּכְרָא דְּעָנָךְ. 
טו,כ לִפְנֵי֩ יְהוָ֨ה אֱלֹהֶ֤יךָ תֹֽאכְלֶ֨נּוּ֙ שָׁנָ֣ה בְשָׁנָ֔ה בַּמָּק֖וֹם אֲשֶׁר־יִבְחַ֣ר יְהוָ֑ה אַתָּ֖ה וּבֵיתֶֽךָ׃ קֳדָם יְיָ אֱלָהָךְ תֵּיכְלִנֵּיהּ שְׁנָא בִּשְׁנָא, בְּאַתְרָא דְּיִתְרְעֵי יְיָ--אַתְּ, וַאֲנָשׁ בֵּיתָךְ. 
טו,כא וְכִֽי־יִהְיֶ֨ה ב֜וֹ מ֗וּם פִּסֵּ֨חַ֙ א֣וֹ עִוֵּ֔ר כֹּ֖ל מ֣וּם רָ֑ע לֹ֣א תִזְבָּחֶ֔נּוּ לַֽיהוָ֖ה אֱלֹהֶֽיךָ׃ וַאֲרֵי יְהֵי בֵּיהּ מוּמָא, חֲגִיר אוֹ עֲוִיר, כֹּל, מוּם בִּישׁ--לָא תִּכְּסִנֵּיהּ, קֳדָם יְיָ אֱלָהָךְ. 
טו,כב בִּשְׁעָרֶ֖יךָ תֹּֽאכְלֶ֑נּוּ הַטָּמֵ֤א וְהַטָּהוֹר֙ יַחְדָּ֔ו כַּצְּבִ֖י וְכָֽאַיָּֽל׃ בְּקִרְוָךְ, תֵּיכְלִנֵּיהּ--מְסָאֲבָא וְדָכְיָא כַּחְדָּא, כִּבְסַר טַבְיָא וְאַיְלָא. 
טו,כג רַ֥ק אֶת־דָּמ֖וֹ לֹ֣א תֹאכֵ֑ל עַל־הָאָ֥רֶץ תִּשְׁפְּכֶ֖נּוּ כַּמָּֽיִם׃ {פ} לְחוֹד יָת דְּמֵיהּ, לָא תֵּיכוֹל:  עַל אַרְעָא תֵּישְׁדִנֵּיהּ, כְּמַיָּא. {פ}
טז,א שָׁמוֹר֙ אֶת־חֹ֣דֶשׁ הָֽאָבִ֔יב וְעָשִׂ֣יתָ פֶּ֔סַח לַֽיהוָ֖ה אֱלֹהֶ֑יךָ כִּ֞י בְּחֹ֣דֶשׁ הָֽאָבִ֗יב הוֹצִ֨יאֲךָ֜ יְהוָ֧ה אֱלֹהֶ֛יךָ מִמִּצְרַ֖יִם לָֽיְלָה׃ טַר, יָת יַרְחָא דַּאֲבִיבָא, וְתַעֲבֵיד פִּסְחָא, קֳדָם יְיָ אֱלָהָךְ:  אֲרֵי בְּיַרְחָא דַּאֲבִיבָא, אַפְּקָךְ יְיָ אֱלָהָךְ מִמִּצְרַיִם וַעֲבַד לָךְ נִסִּין--בְּלֵילְיָא. 
טז,ב וְזָבַ֥חְתָּ פֶּ֛סַח לַֽיהוָ֥ה אֱלֹהֶ֖יךָ צֹ֣אן וּבָקָ֑ר בַּמָּקוֹם֙ אֲשֶׁר־יִבְחַ֣ר יְהוָ֔ה לְשַׁכֵּ֥ן שְׁמ֖וֹ שָֽׁם׃ וְתִכּוֹס פִּסְחָא קֳדָם יְיָ אֱלָהָךְ מִן בְּנֵי עָנָא, וְנִכְסַת קֻדְשַׁיָּא מִן תּוֹרֵי, בְּאַתְרָא דְּיִתְרְעֵי יְיָ, לְאַשְׁרָאָה שְׁכִינְתֵיהּ תַּמָּן. 
טז,ג לֹֽא־תֹאכַ֤ל עָלָיו֙ חָמֵ֔ץ שִׁבְעַ֥ת יָמִ֛ים תֹּֽאכַל־עָלָ֥יו מַצּ֖וֹת לֶ֣חֶם עֹ֑נִי כִּ֣י בְחִפָּז֗וֹן יָצָ֨אתָ֙ מֵאֶ֣רֶץ מִצְרַ֔יִם לְמַ֣עַן תִּזְכֹּ֗ר אֶת־י֤וֹם צֵֽאתְךָ֙ מֵאֶ֣רֶץ מִצְרַ֔יִם כֹּ֖ל יְמֵ֥י חַיֶּֽיךָ׃ לָא תֵּיכוֹל עֲלוֹהִי חֲמִיעַ, שִׁבְעָא יוֹמִין תֵּיכוֹל עֲלוֹהִי פַּטִּירָא לְחֵים עַנְיֵי:  אֲרֵי בִּבְהִילוּ, נְפַקְתָּא מֵאַרְעָא דְּמִצְרַיִם--בְּדִיל דְּתִדְכַר יָת יוֹם מִפְּקָךְ מֵאַרְעָא דְּמִצְרַיִם, כֹּל יוֹמֵי חַיָּךְ. 
טז,ד וְלֹֽא־יֵרָאֶ֨ה לְךָ֥ שְׂאֹ֛ר בְּכָל־גְּבֻֽלְךָ֖ שִׁבְעַ֣ת יָמִ֑ים וְלֹֽא־יָלִ֣ין מִן־הַבָּשָׂ֗ר אֲשֶׁ֨ר תִּזְבַּ֥ח בָּעֶ֛רֶב בַּיּ֥וֹם הָֽרִאשׁ֖וֹן לַבֹּֽקֶר׃ וְלָא יִתַּחְזֵי לָךְ חֲמִיר בְּכָל תְּחוּמָךְ, שִׁבְעָא יוֹמִין; וְלָא יְבִית מִן בִּסְרָא, דְּתִכּוֹס בְּרַמְשָׁא בְּיוֹמָא קַדְמָאָה--לְצַפְרָא. 
טז,ה לֹ֥א תוּכַ֖ל לִזְבֹּ֣חַ אֶת־הַפָּ֑סַח בְּאַחַ֣ד שְׁעָרֶ֔יךָ אֲשֶׁר־יְהוָ֥ה אֱלֹהֶ֖יךָ נֹתֵ֥ן לָֽךְ׃ לֵית לָךְ רְשׁוּ, לְמִכַּס יָת פִּסְחָא, בַּחֲדָא מִן קִרְוָךְ, דַּייָ אֱלָהָךְ יָהֵיב לָךְ. 
טז,ו כִּ֠י אִֽם־אֶל־הַמָּק֞וֹם אֲשֶׁר־יִבְחַ֨ר יְהוָ֤ה אֱלֹהֶ֨יךָ֙ לְשַׁכֵּ֣ן שְׁמ֔וֹ שָׁ֛ם תִּזְבַּ֥ח אֶת־הַפֶּ֖סַח בָּעָ֑רֶב כְּב֣וֹא הַשֶּׁ֔מֶשׁ מוֹעֵ֖ד צֵֽאתְךָ֥ מִמִּצְרָֽיִם׃ אֱלָהֵין בְּאַתְרָא דְּיִתְרְעֵי יְיָ אֱלָהָךְ, לְאַשְׁרָאָה שְׁכִינְתֵיהּ--תַּמָּן תִּכּוֹס יָת פִּסְחָא, בְּרַמְשָׁא:  כְּמֵיעַל שִׁמְשָׁא, זְמָן מִפְּקָךְ מִמִּצְרָיִם. 
טז,ז וּבִשַּׁלְתָּ֙ וְאָ֣כַלְתָּ֔ בַּמָּק֕וֹם אֲשֶׁ֥ר יִבְחַ֛ר יְהוָ֥ה אֱלֹהֶ֖יךָ בּ֑וֹ וּפָנִ֣יתָ בַבֹּ֔קֶר וְהָֽלַכְתָּ֖ לְאֹֽהָלֶֽיךָ׃ וּתְבַשֵּׁיל, וְתֵיכוֹל, בְּאַתְרָא, דְּיִתְרְעֵי יְיָ אֱלָהָךְ בֵּיהּ; וְתִתְפְּנֵי בְּצַפְרָא, וּתְהָךְ לְמַשְׁכְּנָךְ. 
טז,ח שֵׁ֥שֶׁת יָמִ֖ים תֹּאכַ֣ל מַצּ֑וֹת וּבַיּ֣וֹם הַשְּׁבִיעִ֗י עֲצֶ֨רֶת֙ לַֽיהוָ֣ה אֱלֹהֶ֔יךָ לֹ֥א תַֽעֲשֶׂ֖ה מְלָאכָֽה׃ {ס} שִׁתָּא יוֹמִין, תֵּיכוֹל פַּטִּירָא; וּבְיוֹמָא שְׁבִיעָאָה, כְּנִישׁ קֳדָם יְיָ אֱלָהָךְ--לָא תַּעֲבֵיד, עֲבִידָא. {ס}
טז,ט שִׁבְעָ֥ה שָֽׁבֻעֹ֖ת תִּסְפָּר־לָ֑ךְ מֵֽהָחֵ֤ל חֶרְמֵשׁ֙ בַּקָּמָ֔ה תָּחֵ֣ל לִסְפֹּ֔ר שִׁבְעָ֖ה שָֽׁבֻעֽוֹת׃ שִׁבְעָא שָׁבוּעִין, תִּמְנֵי לָךְ:  מִשֵּׁירָיוּת מַגְּלָא, בַּחֲצַד עֻמְרָא דַּאֲרָמוּתָא, תְּשָׁרֵי לְמִמְנֵי, שִׁבְעָא שָׁבוּעִין. 
טז,י וְעָשִׂ֜יתָ חַ֤ג שָֽׁבֻעוֹת֙ לַֽיהוָ֣ה אֱלֹהֶ֔יךָ מִסַּ֛ת נִדְבַ֥ת יָֽדְךָ֖ אֲשֶׁ֣ר תִּתֵּ֑ן כַּֽאֲשֶׁ֥ר יְבָֽרֶכְךָ֖ יְהוָ֥ה אֱלֹהֶֽיךָ׃ וְתַעֲבֵיד חַגָּא דְּשָׁבוּעַיָּא, קֳדָם יְיָ אֱלָהָךְ--מִסַּת נִדְבַת יְדָךְ, דְּתִתֵּין:  כְּמָא דִּיבָרְכִנָּךְ, יְיָ אֱלָהָךְ. 
טז,יא וְשָֽׂמַחְתָּ֞ לִפְנֵ֣י ׀ יְהוָ֣ה אֱלֹהֶ֗יךָ אַתָּ֨ה וּבִנְךָ֣ וּבִתֶּךָ֮ וְעַבְדְּךָ֣ וַֽאֲמָתֶךָ֒ וְהַלֵּוִי֙ אֲשֶׁ֣ר בִּשְׁעָרֶ֔יךָ וְהַגֵּ֛ר וְהַיָּת֥וֹם וְהָֽאַלְמָנָ֖ה אֲשֶׁ֣ר בְּקִרְבֶּ֑ךָ בַּמָּק֗וֹם אֲשֶׁ֤ר יִבְחַר֙ יְהוָ֣ה אֱלֹהֶ֔יךָ לְשַׁכֵּ֥ן שְׁמ֖וֹ שָֽׁם׃ וְתִחְדֵּי קֳדָם יְיָ אֱלָהָךְ, אַתְּ וּבְרָךְ וּבְרַתָּךְ וְעַבְדָּךְ וְאַמְתָּךְ, וְלֵיוָאָה דִּבְקִרְוָךְ, וְגִיּוֹרָא וְיַתְמָא וְאַרְמַלְתָּא דְּבֵינָךְ--בְּאַתְרָא, דְּיִתְרְעֵי יְיָ אֱלָהָךְ, לְאַשְׁרָאָה שְׁכִינְתֵיהּ, תַּמָּן. 
טז,יב וְזָ֣כַרְתָּ֔ כִּי־עֶ֥בֶד הָיִ֖יתָ בְּמִצְרָ֑יִם וְשָֽׁמַרְתָּ֣ וְעָשִׂ֔יתָ אֶת־הַֽחֻקִּ֖ים הָאֵֽלֶּה׃ {פ} וְתִדְכַר, אֲרֵי עַבְדָּא הֲוֵיתָא בְּמִצְרָיִם; וְתִטַּר וְתַעֲבֵיד, יָת קְיָמַיָּא הְאִלֵּין. {פ}
טז,יג חַ֧ג הַסֻּכֹּ֛ת תַּֽעֲשֶׂ֥ה לְךָ֖ שִׁבְעַ֣ת יָמִ֑ים בְּאָ֨סְפְּךָ֔ מִֽגָּרְנְךָ֖ וּמִיִּקְבֶֽךָ׃ חַגָּא דִּמְטַלַּיָּא תַּעֲבֵיד לָךְ, שִׁבְעָא יוֹמִין:  בְּמִכְנְשָׁךְ--מֵאִדְּרָךְ, וּמִמַּעֲצַרְתָּךְ. 
טז,יד וְשָֽׂמַחְתָּ֖ בְּחַגֶּ֑ךָ אַתָּ֨ה וּבִנְךָ֤ וּבִתֶּ֨ךָ֙ וְעַבְדְּךָ֣ וַֽאֲמָתֶ֔ךָ וְהַלֵּוִ֗י וְהַגֵּ֛ר וְהַיָּת֥וֹם וְהָֽאַלְמָנָ֖ה אֲשֶׁ֥ר בִּשְׁעָרֶֽיךָ׃ וְתִחְדֵּי, בְּחַגָּךְ:  אַתְּ וּבְרָךְ וּבְרַתָּךְ, וְעַבְדָּךְ וְאַמְתָּךְ, וְלֵיוָאָה וְגִיּוֹרָא וְיַתְמָא וְאַרְמַלְתָּא, דִּבְקִרְוָךְ. 
טז,טו שִׁבְעַ֣ת יָמִ֗ים תָּחֹג֙ לַֽיהוָ֣ה אֱלֹהֶ֔יךָ בַּמָּק֖וֹם אֲשֶׁר־יִבְחַ֣ר יְהוָ֑ה כִּ֣י יְבָֽרֶכְךָ֞ יְהוָ֣ה אֱלֹהֶ֗יךָ בְּכֹ֤ל תְּבוּאָֽתְךָ֙ וּבְכֹל֙ מַֽעֲשֵׂ֣ה יָדֶ֔יךָ וְהָיִ֖יתָ אַ֥ךְ שָׂמֵֽחַ׃ שִׁבְעָא יוֹמִין, תֵּיחוֹג קֳדָם יְיָ אֱלָהָךְ, בְּאַתְרָא, דְּיִתְרְעֵי יְיָ:  אֲרֵי יְבָרְכִנָּךְ יְיָ אֱלָהָךְ, בְּכֹל עֲלַלְתָּךְ וּבְכֹל עוּבָדֵי יְדָךְ, וּתְהֵי, בְּרַם חָדֵי. 
טז,טז שָׁל֣וֹשׁ פְּעָמִ֣ים ׀ בַּשָּׁנָ֡ה יֵֽרָאֶ֨ה כָל־זְכֽוּרְךָ֜ אֶת־פְּנֵ֣י ׀ יְהוָ֣ה אֱלֹהֶ֗יךָ בַּמָּקוֹם֙ אֲשֶׁ֣ר יִבְחָ֔ר בְּחַ֧ג הַמַּצּ֛וֹת וּבְחַ֥ג הַשָּֽׁבֻע֖וֹת וּבְחַ֣ג הַסֻּכּ֑וֹת וְלֹ֧א יֵֽרָאֶ֛ה אֶת־פְּנֵ֥י יְהוָ֖ה רֵיקָֽם׃ תְּלָת זִמְנִין בְּשַׁתָּא יִתַּחְזוֹן כָּל דְּכוּרָךְ קֳדָם יְיָ אֱלָהָךְ, בְּאַתְרָא דְּיִתְרְעֵי--בְּחַגָּא דְּפַטִּירַיָּא וּבְחַגָּא דְּשָׁבוּעַיָּא, וּבְחַגָּא דִּמְטַלַּיָּא; וְלָא יִתַּחְזוֹן קֳדָם יְיָ, רֵיקָנִין. 
טז,יז אִ֖ישׁ כְּמַתְּנַ֣ת יָד֑וֹ כְּבִרְכַּ֛ת יְהוָ֥ה אֱלֹהֶ֖יךָ אֲשֶׁ֥ר נָֽתַן־לָֽךְ׃ {ס} גְּבַר, כְּמַתְּנַת יְדֵיהּ, כְּבִרְכְּתָא דַּייָ אֱלָהָךְ, דִּיהַב לָךְ. {ס}
