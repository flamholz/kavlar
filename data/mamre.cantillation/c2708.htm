איוב פרק ח

א                וַ֭יַּעַן בִּלְדַּ֥ד הַשּׁוּחִ֗י וַיֹּאמַֽר׃
ב עַד־אָ֥ן תְּמַלֶּל־אֵ֑לֶּה    וְר֥וּחַ כַּ֝בִּ֗יר אִמְרֵי־פִֽיךָ׃
ג הַ֭אֵל יְעַוֵּ֣ת מִשְׁפָּ֑ט    וְאִם־שַׁ֝דַּ֗י יְעַוֵּֽת־צֶֽדֶק׃
ד אִם־בָּנֶ֥יךָ חָֽטְאוּ־ל֑וֹ    וַֽ֝יְשַׁלְּחֵ֗ם בְּיַד־פִּשְׁעָֽם׃
ה אִם־אַ֭תָּה תְּשַׁחֵ֣ר אֶל־אֵ֑ל    וְאֶל־שַׁ֝דַּ֗י תִּתְחַנָּֽן׃
ו אִם־זַ֥ךְ וְיָשָׁ֗ר אָ֥תָּה    כִּֽי־עַ֭תָּה יָעִ֣יר עָלֶ֑יךָ וְ֝שִׁלַּ֗ם נְוַ֣ת צִדְקֶֽךָ׃
ז וְהָיָ֣ה רֵאשִֽׁיתְךָ֣ מִצְעָ֑ר    וְ֝אַחֲרִֽיתְךָ֗ יִשְׂגֶּ֥ה מְאֹֽד׃
ח כִּֽי־שְׁאַל־נָ֭א לְדֹ֣ר רִישׁ֑וֹן    וְ֝כוֹנֵ֗ן לְחֵ֣קֶר אֲבוֹתָֽם׃
ט כִּֽי־תְמ֣וֹל אֲ֭נַחְנוּ וְלֹ֣א נֵדָ֑ע    כִּ֤י צֵ֖ל יָמֵ֣ינוּ עֲלֵי־אָֽרֶץ׃
י הֲלֹא־הֵ֣ם י֭וֹרוּךָ יֹ֣אמְרוּ לָ֑ךְ    וּ֝מִלִּבָּ֗ם יוֹצִ֥אוּ מִלִּֽים׃
יא הֲיִֽגְאֶה־גֹּ֭מֶא בְּלֹ֣א בִצָּ֑ה    יִשְׂגֶּה־אָ֥חוּ בְלִי־מָֽיִם׃
יב עֹדֶ֣נּוּ בְ֭אִבּוֹ לֹ֣א יִקָּטֵ֑ף    וְלִפְנֵ֖י כָל־חָצִ֣יר יִיבָֽשׁ׃
יג כֵּ֗ן אָ֭רְחוֹת כָּל־שֹׁ֣כְחֵי אֵ֑ל    וְתִקְוַ֖ת חָנֵ֣ף תֹּאבֵֽד׃
יד אֲשֶׁר־יָק֥וֹט כִּסְל֑וֹ    וּבֵ֥ית עַ֝כָּבִ֗ישׁ מִבְטַחֽוֹ׃
טו יִשָּׁעֵ֣ן עַל־בֵּ֭יתוֹ וְלֹ֣א יַֽעֲמֹ֑ד    יַֽחֲזִ֥יק בּ֝֗וֹ וְלֹ֣א יָקֽוּם׃
טז רָטֹ֣ב ה֭וּא לִפְנֵי־שָׁ֑מֶשׁ    וְעַ֥ל גַּ֝נָּת֗וֹ יֹֽנַקְתּ֥וֹ תֵצֵֽא׃
יז עַל־גַּ֭ל שָֽׁרָשָׁ֣יו יְסֻבָּ֑כוּ    בֵּ֖ית אֲבָנִ֣ים יֶֽחֱזֶֽה׃
יח אִם־יְבַלְּעֶ֥נּוּ מִמְּקֹמ֑וֹ    וְכִ֥חֶשׁ בּ֝֗וֹ לֹ֣א רְאִיתִֽיךָ׃
יט הֶן־ה֭וּא מְשׂ֣וֹשׂ דַּרְכּ֑וֹ    וּ֝מֵֽעָפָ֗ר אַחֵ֥ר יִצְמָֽחוּ׃
כ הֶן־אֵ֭ל לֹ֣א יִמְאַס־תָּ֑ם    וְלֹֽא־יַ֝חֲזִ֗יק בְּיַד־מְרֵעִֽים׃
כא עַד־יְמַלֵּ֣ה שְׂח֣וֹק פִּ֑יךָ    וּשְׂפָתֶ֥יךָ תְרוּעָֽה׃
כב שֹֽׂנְאֶ֥יךָ יִלְבְּשׁוּ־בֹ֑שֶׁת    וְאֹ֖הֶל רְשָׁעִ֣ים אֵינֶֽנּוּ׃
