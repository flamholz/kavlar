תולדות

כה,יט וְאֵ֛לֶּה תּֽוֹלְדֹ֥ת יִצְחָ֖ק בֶּן־אַבְרָהָ֑ם אַבְרָהָ֖ם הוֹלִ֥יד אֶת־יִצְחָֽק׃ וְאִלֵּין תּוֹלְדָת יִצְחָק, בַּר אַבְרָהָם:  אַבְרָהָם, אוֹלֵיד יָת יִצְחָק. 
כה,כ וַיְהִ֤י יִצְחָק֙ בֶּן־אַרְבָּעִ֣ים שָׁנָ֔ה בְּקַחְתּ֣וֹ אֶת־רִבְקָ֗ה בַּת־בְּתוּאֵל֙ הָֽאֲרַמִּ֔י מִפַּדַּ֖ן אֲרָ֑ם אֲח֛וֹת לָבָ֥ן הָֽאֲרַמִּ֖י ל֥וֹ לְאִשָּֽׁה׃ וַהֲוָה יִצְחָק, בַּר אַרְבְּעִין שְׁנִין, כַּד נְסֵיב יָת רִבְקָה בַּת בְּתוּאֵל אֲרַמָּאָה, מִפַּדַּן אֲרָם--אֲחָתֵיהּ דְּלָבָן אֲרַמָּאָה, לֵיהּ לְאִתּוּ. 
כה,כא וַיֶּעְתַּ֨ר יִצְחָ֤ק לַֽיהוָה֙ לְנֹ֣כַח אִשְׁתּ֔וֹ כִּ֥י עֲקָרָ֖ה הִ֑וא וַיֵּעָ֤תֶר לוֹ֙ יְהוָ֔ה וַתַּ֖הַר רִבְקָ֥ה אִשְׁתּֽוֹ׃ וְצַלִּי יִצְחָק קֳדָם יְיָ לָקֳבֵיל אִתְּתֵיהּ, אֲרֵי עַקְרָא הִיא; וְקַבֵּיל צְלוֹתֵיהּ יְיָ, וְעַדִּיאַת רִבְקָה אִתְּתֵיהּ. 
כה,כב וַיִּתְרֹֽצְצ֤וּ הַבָּנִים֙ בְּקִרְבָּ֔הּ וַתֹּ֣אמֶר אִם־כֵּ֔ן לָ֥מָּה זֶּ֖ה אָנֹ֑כִי וַתֵּ֖לֶךְ לִדְרֹ֥שׁ אֶת־יְהוָֽה׃ וְדָחֲקִין בְּנַיָּא, בִּמְעַהָא, וַאֲמַרַת אִם כֵּין, לְמָא דְּנָן אֲנָא; וַאֲזַלַת, לְמִתְבַּע אֻלְפָן מִן קֳדָם יְיָ. 
כה,כג וַיֹּ֨אמֶר יְהוָ֜ה לָ֗הּ שְׁנֵ֤י גֹיִים֙ בְּבִטְנֵ֔ךְ וּשְׁנֵ֣י לְאֻמִּ֔ים מִמֵּעַ֖יִךְ יִפָּרֵ֑דוּ וּלְאֹם֙ מִלְאֹ֣ם יֶֽאֱמָ֔ץ וְרַ֖ב יַֽעֲבֹ֥ד צָעִֽיר׃ וַאֲמַר יְיָ לַהּ, תְּרֵין עַמְמִין בִּמְעַכִי, וְתַרְתֵּין מַלְכְוָן, מִמְּעַכִי יִתְפָּרְשָׁן; וּמַלְכוּ מִמַּלְכוּ תִּתַּקַּף, וְרַבָּא יִשְׁתַּעֲבַד לִזְעֵירָא. 
כה,כד וַיִּמְלְא֥וּ יָמֶ֖יהָ לָלֶ֑דֶת וְהִנֵּ֥ה תוֹמִ֖ם בְּבִטְנָֽהּ׃ וּשְׁלִימוּ יוֹמַהָא, לְמֵילַד; וְהָא תְּיוֹמִין, בִּמְעַהָא. 
כה,כה וַיֵּצֵ֤א הָֽרִאשׁוֹן֙ אַדְמוֹנִ֔י כֻּלּ֖וֹ כְּאַדֶּ֣רֶת שֵׂעָ֑ר וַיִּקְרְא֥וּ שְׁמ֖וֹ עֵשָֽׂו׃ וּנְפַק קַדְמָאָה סָמוֹק, כֻּלֵּיהּ כִּגְלִים דִּסְעַר; וּקְרוֹ שְׁמֵיהּ, עֵשָׂו. 
כה,כו וְאַֽחֲרֵי־כֵ֞ן יָצָ֣א אָחִ֗יו וְיָד֤וֹ אֹחֶ֨זֶת֙ בַּֽעֲקֵ֣ב עֵשָׂ֔ו וַיִּקְרָ֥א שְׁמ֖וֹ יַֽעֲקֹ֑ב וְיִצְחָ֛ק בֶּן־שִׁשִּׁ֥ים שָׁנָ֖ה בְּלֶ֥דֶת אֹתָֽם׃ וּבָתַר כֵּין נְפַק אֲחוּהִי, וִידֵיהּ אֲחִידָא בְּעִקְבָא דְּעֵשָׂו, וּקְרָא שְׁמֵיהּ, יַעֲקוֹב; וְיִצְחָק בַּר שִׁתִּין שְׁנִין, כַּד יְלֵידַת יָתְהוֹן. 
כה,כז וַֽיִּגְדְּלוּ֙ הַנְּעָרִ֔ים וַיְהִ֣י עֵשָׂ֗ו אִ֛ישׁ יֹדֵ֥עַ צַ֖יִד אִ֣ישׁ שָׂדֶ֑ה וְיַֽעֲקֹב֙ אִ֣ישׁ תָּ֔ם יֹשֵׁ֖ב אֹֽהָלִֽים׃ וּרְבִיאוּ, עוּלֵימַיָּא, וַהֲוָה עֵשָׂו גְּבַר נְחַשׁ יִרְכָן, גְּבַר נָפֵיק חֲקַל; וְיַעֲקוֹב גְּבַר שְׁלִים, מְשַׁמֵּישׁ בֵּית אֻלְפָנָא. 
כה,כח וַיֶּֽאֱהַ֥ב יִצְחָ֛ק אֶת־עֵשָׂ֖ו כִּי־צַ֣יִד בְּפִ֑יו וְרִבְקָ֖ה אֹהֶ֥בֶת אֶֽת־יַעֲקֹֽב׃ וּרְחֵים יִצְחָק יָת עֵשָׂו, אֲרֵי מִצֵּידֵיהּ הֲוָה אָכֵיל; וְרִבְקָה, רְחֵימַת יָת יַעֲקוֹב. 
כה,כט וַיָּ֥זֶד יַֽעֲקֹ֖ב נָזִ֑יד וַיָּבֹ֥א עֵשָׂ֛ו מִן־הַשָּׂדֶ֖ה וְה֥וּא עָיֵֽף׃ וּבַשֵּׁיל יַעֲקוֹב, תַּבְשִׁילָא; וְעָאל עֵשָׂו מִן חַקְלָא, וְהוּא מְשַׁלְהֵי. 
כה,ל וַיֹּ֨אמֶר עֵשָׂ֜ו אֶֽל־יַעֲקֹ֗ב הַלְעִיטֵ֤נִי נָא֙ מִן־הָֽאָדֹ֤ם הָֽאָדֹם֙ הַזֶּ֔ה כִּ֥י עָיֵ֖ף אָנֹ֑כִי עַל־כֵּ֥ן קָרָֽא־שְׁמ֖וֹ אֱדֽוֹם׃ וַאֲמַר עֵשָׂו לְיַעֲקוֹב, אַטְעֵימְנִי כְּעַן מִן סִמּוֹקָא סָמְקָא הָדֵין--אֲרֵי מְשַׁלְהֵי, אֲנָא; עַל כֵּין קְרָא שְׁמֵיהּ, אֱדוֹם. 
כה,לא וַיֹּ֖אמֶר יַֽעֲקֹ֑ב מִכְרָ֥ה כַיּ֛וֹם אֶת־בְּכֹֽרָתְךָ֖ לִֽי׃ וַאֲמַר, יַעֲקוֹב:  זַבֵּין כְּיוֹם דִּלְהֵין יָת בְּכֵירוּתָךְ, לִי. 
כה,לב וַיֹּ֣אמֶר עֵשָׂ֔ו הִנֵּ֛ה אָֽנֹכִ֥י הוֹלֵ֖ךְ לָמ֑וּת וְלָמָּה־זֶּ֥ה לִ֖י בְּכֹרָֽה׃ וַאֲמַר עֵשָׂו, הָא אֲנָא אָזֵיל לִמְמָת; וּלְמָא דְּנָן לִי, בְּכֵירוּתָא. 
כה,לג וַיֹּ֣אמֶר יַֽעֲקֹ֗ב הִשָּׁ֤בְעָה לִּי֙ כַּיּ֔וֹם וַיִּשָּׁבַ֖ע ל֑וֹ וַיִּמְכֹּ֥ר אֶת־בְּכֹֽרָת֖וֹ לְיַֽעֲקֹֽב׃ וַאֲמַר יַעֲקוֹב, קַיֵּים לִי כְּיוֹם דִּלְהֵין, וְקַיֵּים, לֵיהּ; וְזַבֵּין יָת בְּכֵירוּתֵיהּ, לְיַעֲקוֹב. 
כה,לד וְיַֽעֲקֹ֞ב נָתַ֣ן לְעֵשָׂ֗ו לֶ֚חֶם וּנְזִ֣יד עֲדָשִׁ֔ים וַיֹּ֣אכַל וַיֵּ֔שְׁתְּ וַיָּ֖קָם וַיֵּלַ֑ךְ וַיִּ֥בֶז עֵשָׂ֖ו אֶת־הַבְּכֹרָֽה׃ {פ} וְיַעֲקוֹב יְהַב לְעֵשָׂו, לְחֵים וְתַבְשִׁיל דִּטְלוֹפְחִין, וַאֲכַל וּשְׁתִי, וְקָם וַאֲזַל; וְשָׁט עֵשָׂו, יָת בְּכֵירוּתָא. {פ}
כו,א וַיְהִ֤י רָעָב֙ בָּאָ֔רֶץ מִלְּבַד֙ הָֽרָעָ֣ב הָֽרִאשׁ֔וֹן אֲשֶׁ֥ר הָיָ֖ה בִּימֵ֣י אַבְרָהָ֑ם וַיֵּ֧לֶךְ יִצְחָ֛ק אֶל־אֲבִימֶ֥לֶךְ מֶֽלֶךְ־פְּלִשְׁתִּ֖ים גְּרָֽרָה׃ וַהֲוָה כַּפְנָא, בְּאַרְעָא, בָּר מִכַּפְנָא קַדְמָאָה, דַּהֲוָה בְּיוֹמֵי אַבְרָהָם; וַאֲזַל יִצְחָק לְוָת אֲבִימֶלֶךְ מַלְכָּא דִּפְלִשְׁתָּאֵי, לִגְרָר. 
כו,ב וַיֵּרָ֤א אֵלָיו֙ יְהוָ֔ה וַיֹּ֖אמֶר אַל־תֵּרֵ֣ד מִצְרָ֑יְמָה שְׁכֹ֣ן בָּאָ֔רֶץ אֲשֶׁ֖ר אֹמַ֥ר אֵלֶֽיךָ׃ וְאִתְגְּלִי לֵיהּ יְיָ, וַאֲמַר לָא תֵּיחוֹת לְמִצְרָיִם:  שְׁרִי בְּאַרְעָא, דְּאֵימַר לָךְ. 
כו,ג גּ֚וּר בָּאָ֣רֶץ הַזֹּ֔את וְאֶֽהְיֶ֥ה עִמְּךָ֖ וַאֲבָֽרְכֶ֑ךָּ כִּֽי־לְךָ֣ וּֽלְזַרְעֲךָ֗ אֶתֵּן֙ אֶת־כָּל־הָֽאֲרָצֹ֣ת הָאֵ֔ל וַהֲקִֽמֹתִי֙ אֶת־הַשְּׁבֻעָ֔ה אֲשֶׁ֥ר נִשְׁבַּ֖עְתִּי לְאַבְרָהָ֥ם אָבִֽיךָ׃ דּוּר בְּאַרְעָא הָדָא, וִיהֵי מֵימְרִי בְּסַעֲדָךְ וַאֲבָרְכִנָּךְ:  אֲרֵי לָךְ וְלִבְנָךְ, אֶתֵּין יָת כָּל אַרְעָתָא הָאִלֵּין, וַאֲקַיֵּם יָת קְיָמָא, דְּקַיֵּימִית לְאַבְרָהָם אֲבוּךְ. 
כו,ד וְהִרְבֵּיתִ֤י אֶֽת־זַרְעֲךָ֙ כְּכֽוֹכְבֵ֣י הַשָּׁמַ֔יִם וְנָֽתַתִּ֣י לְזַרְעֲךָ֔ אֵ֥ת כָּל־הָֽאֲרָצֹ֖ת הָאֵ֑ל וְהִתְבָּֽרְכ֣וּ בְזַרְעֲךָ֔ כֹּ֖ל גּוֹיֵ֥י הָאָֽרֶץ׃ וְאַסְגֵּי יָת בְּנָךְ, כְּכוֹכְבֵי שְׁמַיָּא, וְאֶתֵּין לִבְנָךְ, יָת כָּל אַרְעָתָא הָאִלֵּין; וְיִתְבָּרְכוּן בְּדִיל בְּנָךְ, כֹּל עַמְמֵי אַרְעָא. 
כו,ה עֵ֕קֶב אֲשֶׁר־שָׁמַ֥ע אַבְרָהָ֖ם בְּקֹלִ֑י וַיִּשְׁמֹר֙ מִשְׁמַרְתִּ֔י מִצְו‍ֹתַ֖י חֻקּוֹתַ֥י וְתֽוֹרֹתָֽי׃ חֲלָף, דְּקַבֵּיל אַבְרָהָם לְמֵימְרִי; וּנְטַר, מַטְּרַת מֵימְרִי, פִּקּוֹדַי, קְיָמַי וְאוֹרָיָתָי. 
כו,ו וַיֵּ֥שֶׁב יִצְחָ֖ק בִּגְרָֽר׃ וִיתֵיב יִצְחָק, בִּגְרָר. 
כו,ז וַֽיִּשְׁאֲל֞וּ אַנְשֵׁ֤י הַמָּקוֹם֙ לְאִשְׁתּ֔וֹ וַיֹּ֖אמֶר אֲחֹ֣תִי הִ֑וא כִּ֤י יָרֵא֙ לֵאמֹ֣ר אִשְׁתִּ֔י פֶּן־יַֽהַרְגֻ֜נִי אַנְשֵׁ֤י הַמָּקוֹם֙ עַל־רִבְקָ֔ה כִּֽי־טוֹבַ֥ת מַרְאֶ֖ה הִֽוא׃ וּשְׁאִילוּ אֲנָשֵׁי אַתְרָא, עַל עֵיסַק אִתְּתֵיהּ, וַאֲמַר, אֲחָת הִיא:  אֲרֵי דְּחֵיל, לְמֵימַר אִתְּתִי, דִּלְמָא יִקְטְלוּנַּנִי אֲנָשֵׁי אַתְרָא עַל רִבְקָה, אֲרֵי שַׁפִּירַת חֵיזוּ הִיא. 
כו,ח וַיְהִ֗י כִּ֣י אָֽרְכוּ־ל֥וֹ שָׁם֙ הַיָּמִ֔ים וַיַּשְׁקֵ֗ף אֲבִימֶ֨לֶךְ֙ מֶ֣לֶךְ פְּלִשְׁתִּ֔ים בְּעַ֖ד הַֽחַלּ֑וֹן וַיַּ֗רְא וְהִנֵּ֤ה יִצְחָק֙ מְצַחֵ֔ק אֵ֖ת רִבְקָ֥ה אִשְׁתּֽוֹ׃ וַהֲוָה, כַּד סְגִיאוּ לֵיהּ תַּמָּן יוֹמַיָּא, וְאִסְתַּכִי אֲבִימֶלֶךְ מַלְכָּא דִּפְלִשְׁתָּאֵי, מִן חֲרַכָּא; וַחֲזָא, וְהָא יִצְחָק מְחַיֵּיךְ, עִם, רִבְקָה אִתְּתֵיהּ. 
כו,ט וַיִּקְרָ֨א אֲבִימֶ֜לֶךְ לְיִצְחָ֗ק וַיֹּ֨אמֶר֙ אַ֣ךְ הִנֵּ֤ה אִשְׁתְּךָ֙ הִ֔וא וְאֵ֥יךְ אָמַ֖רְתָּ אֲחֹ֣תִי הִ֑וא וַיֹּ֤אמֶר אֵלָיו֙ יִצְחָ֔ק כִּ֣י אָמַ֔רְתִּי פֶּן־אָמ֖וּת עָלֶֽיהָ׃ וּקְרָא אֲבִימֶלֶךְ לְיִצְחָק, וַאֲמַר בְּרַם הָא אִתְּתָךְ הִיא, וְאֵיכְדֵין אֲמַרְתְּ, אֲחָת הִיא; וַאֲמַר לֵיהּ, יִצְחָק, אֲרֵי אֲמַרִית, דִּלְמָא אֶתְקְטִיל עֲלַהּ. 
כו,י וַיֹּ֣אמֶר אֲבִימֶ֔לֶךְ מַה־זֹּ֖את עָשִׂ֣יתָ לָּ֑נוּ כִּ֠מְעַט שָׁכַ֞ב אַחַ֤ד הָעָם֙ אֶת־אִשְׁתֶּ֔ךָ וְהֵֽבֵאתָ֥ עָלֵ֖ינוּ אָשָֽׁם׃ וַאֲמַר אֲבִימֶלֶךְ, מָא דָּא עֲבַדְתְּ לַנָא; כִּזְעֵיר פּוֹן שְׁכֵיב דִּמְיַחַד בְּעַמָּא, עִם אִתְּתָךְ, וְאֵיתִיתָא עֲלַנָא, חוֹבָא. 
כו,יא וַיְצַ֣ו אֲבִימֶ֔לֶךְ אֶת־כָּל־הָעָ֖ם לֵאמֹ֑ר הַנֹּגֵ֜עַ בָּאִ֥ישׁ הַזֶּ֛ה וּבְאִשְׁתּ֖וֹ מ֥וֹת יוּמָֽת׃ וּפַקֵּיד אֲבִימֶלֶךְ, יָת כָּל עַמָּא לְמֵימַר:  דְּיַנְזֵיק לְגֻבְרָא הָדֵין, וּלְאִתְּתֵיהּ--אִתְקְטָלָא יִתְקְטִיל. 
כו,יב וַיִּזְרַ֤ע יִצְחָק֙ בָּאָ֣רֶץ הַהִ֔וא וַיִּמְצָ֛א בַּשָּׁנָ֥ה הַהִ֖וא מֵאָ֣ה שְׁעָרִ֑ים וַֽיְבָרְכֵ֖הוּ יְהוָֽה׃ וּזְרַע יִצְחָק בְּאַרְעָא הַהִיא, וְאַשְׁכַּח בְּשַׁתָּא הַהִיא עַל חַד מְאָה בִּדְשַׁעֲרוּהִי; וּבָרְכֵיהּ, יְיָ. 
כו,יג וַיִּגְדַּ֖ל הָאִ֑ישׁ וַיֵּ֤לֶךְ הָלוֹךְ֙ וְגָדֵ֔ל עַ֥ד כִּֽי־גָדַ֖ל מְאֹֽד׃ וּרְבָא, גֻּבְרָא; וַאֲזַל אָזֵיל סָגֵי וְרָבֵי, עַד דִּרְבָא לַחְדָּא. 
כו,יד וַֽיְהִי־ל֤וֹ מִקְנֵה־צֹאן֙ וּמִקְנֵ֣ה בָקָ֔ר וַֽעֲבֻדָּ֖ה רַבָּ֑ה וַיְקַנְא֥וּ אֹת֖וֹ פְּלִשְׁתִּֽים׃ וַהֲווֹ לֵיהּ גֵּיתֵי עָנָא וְגֵיתֵי תּוֹרֵי, וַעֲבֻדָּה סַגִּי; וְקַנִּיאוּ בֵּיהּ, פְּלִשְׁתָּאֵי. 
כו,טו וְכָל־הַבְּאֵרֹ֗ת אֲשֶׁ֤ר חָֽפְרוּ֙ עַבְדֵ֣י אָבִ֔יו בִּימֵ֖י אַבְרָהָ֣ם אָבִ֑יו סִתְּמ֣וּם פְּלִשְׁתִּ֔ים וַיְמַלְא֖וּם עָפָֽר׃ וְכָל בֵּירֵי, דַּחֲפַרוּ עַבְדֵי אֲבוּהִי, בְּיוֹמֵי, אַבְרָהָם אֲבוּהִי--טָמוּנִין פְּלִשְׁתָּאֵי, וּמְלוֹנִין עַפְרָא. 
כו,טז וַיֹּ֥אמֶר אֲבִימֶ֖לֶךְ אֶל־יִצְחָ֑ק לֵ֚ךְ מֵֽעִמָּ֔נוּ כִּֽי־עָצַ֥מְתָּ מִמֶּ֖נּוּ מְאֹֽד׃ וַאֲמַר אֲבִימֶלֶךְ, לְיִצְחָק:  אִיזֵיל, מֵעִמַּנָא, אֲרֵי תְּקֵיפְתְּ מִנַּנָא, לַחְדָּא. 
כו,יז וַיֵּ֥לֶךְ מִשָּׁ֖ם יִצְחָ֑ק וַיִּ֥חַן בְּנַֽחַל־גְּרָ֖ר וַיֵּ֥שֶׁב שָֽׁם׃ וַאֲזַל מִתַּמָּן, יִצְחָק; וּשְׁרָא בְּנַחְלָא דִּגְרָר, וִיתֵיב תַּמָּן. 
כו,יח וַיָּ֨שָׁב יִצְחָ֜ק וַיַּחְפֹּ֣ר ׀ אֶת־בְּאֵרֹ֣ת הַמַּ֗יִם אֲשֶׁ֤ר חָֽפְרוּ֙ בִּימֵי֙ אַבְרָהָ֣ם אָבִ֔יו וַיְסַתְּמ֣וּם פְּלִשְׁתִּ֔ים אַֽחֲרֵ֖י מ֣וֹת אַבְרָהָ֑ם וַיִּקְרָ֤א לָהֶן֙ שֵׁמ֔וֹת כַּשֵּׁמֹ֕ת אֲשֶׁר־קָרָ֥א לָהֶ֖ן אָבִֽיו׃ וְתָב יִצְחָק וַחֲפַר יָת בֵּירֵי דְּמַיָּא, דַּחֲפַרוּ בְּיוֹמֵי אַבְרָהָם אֲבוּהִי, וְטָמוּנִין פְּלִשְׁתָּאֵי, בָּתַר דְּמִית אַבְרָהָם; וּקְרָא לְהוֹן, שְׁמָהָן, כִּשְׁמָהָן, דַּהֲוָה קָרֵי לְהוֹן אֲבוּהִי. 
כו,יט וַיַּחְפְּר֥וּ עַבְדֵֽי־יִצְחָ֖ק בַּנָּ֑חַל וַיִּ֨מְצְאוּ־שָׁ֔ם בְּאֵ֖ר מַ֥יִם חַיִּֽים׃ וַחֲפַרוּ עַבְדֵי יִצְחָק, בְּנַחְלָא; וְאַשְׁכַּחוּ תַּמָּן--בֵּיר, דְּמַיִין נָבְעִין. 
כו,כ וַיָּרִ֜יבוּ רֹעֵ֣י גְרָ֗ר עִם־רֹעֵ֥י יִצְחָ֛ק לֵאמֹ֖ר לָ֣נוּ הַמָּ֑יִם וַיִּקְרָ֤א שֵֽׁם־הַבְּאֵר֙ עֵ֔שֶׂק כִּ֥י הִֽתְעַשְּׂק֖וּ עִמּֽוֹ׃ וּנְצוֹ רָעֲוָתָא דִּגְרָר, עִם רָעֲוָתָא דְּיִצְחָק לְמֵימַר--דִּילַנָא מַיָּא; וּקְרָא שְׁמַהּ דְּבֵירָא עֵשֶׂק, אֲרֵי אִתְעַסַּקוּ עִמֵּיהּ. 
כו,כא וַֽיַּחְפְּרוּ֙ בְּאֵ֣ר אַחֶ֔רֶת וַיָּרִ֖יבוּ גַּם־עָלֶ֑יהָ וַיִּקְרָ֥א שְׁמָ֖הּ שִׂטְנָֽה׃ וַחֲפַרוּ בֵּיר אֻחְרִי, וּנְצוֹ אַף עֲלַהּ; וּקְרָא שְׁמַהּ, שִׂטְנָה. 
כו,כב וַיַּעְתֵּ֣ק מִשָּׁ֗ם וַיַּחְפֹּר֙ בְּאֵ֣ר אַחֶ֔רֶת וְלֹ֥א רָב֖וּ עָלֶ֑יהָ וַיִּקְרָ֤א שְׁמָהּ֙ רְחֹב֔וֹת וַיֹּ֗אמֶר כִּֽי־עַתָּ֞ה הִרְחִ֧יב יְהוָ֛ה לָ֖נוּ וּפָרִ֥ינוּ בָאָֽרֶץ׃ וְאִסְתַּלַּק מִתַּמָּן, וַחֲפַר בֵּיר אֻחְרִי, וְלָא נְצוֹ, עֲלַהּ; וּקְרָא שְׁמַהּ, רְחוֹבוֹת, וַאֲמַר אֲרֵי כְּעַן אַפְתִּי יְיָ לַנָא, וְנִפּוֹשׁ בְּאַרְעָא. 
כו,כג וַיַּ֥עַל מִשָּׁ֖ם בְּאֵ֥ר שָֽׁבַע׃ וּסְלֵיק מִתַּמָּן, לִבְאֵר שָׁבַע. 
כו,כד וַיֵּרָ֨א אֵלָ֤יו יְהוָה֙ בַּלַּ֣יְלָה הַה֔וּא וַיֹּ֕אמֶר אָֽנֹכִ֕י אֱלֹהֵ֖י אַבְרָהָ֣ם אָבִ֑יךָ אַל־תִּירָא֙ כִּֽי־אִתְּךָ֣ אָנֹ֔כִי וּבֵֽרַכְתִּ֨יךָ֙ וְהִרְבֵּיתִ֣י אֶֽת־זַרְעֲךָ֔ בַּֽעֲב֖וּר אַבְרָהָ֥ם עַבְדִּֽי׃ וְאִתְגְּלִי לֵיהּ יְיָ, בְּלֵילְיָא הַהוּא, וַאֲמַר, אֲנָא אֱלָהֵיהּ דְּאַבְרָהָם אֲבוּךְ; לָא תִּדְחַל, אֲרֵי בְּסַעֲדָךְ מֵימְרִי, וַאֲבָרְכִנָּךְ וְאַסְגֵּי יָת בְּנָךְ, בְּדִיל אַבְרָהָם עַבְדִּי. 
כו,כה וַיִּ֧בֶן שָׁ֣ם מִזְבֵּ֗חַ וַיִּקְרָא֙ בְּשֵׁ֣ם יְהוָ֔ה וַיֶּט־שָׁ֖ם אָֽהֳל֑וֹ וַיִּכְרוּ־שָׁ֥ם עַבְדֵי־יִצְחָ֖ק בְּאֵֽר׃ וּבְנָא תַּמָּן מַדְבְּחָא, וְצַלִּי בִּשְׁמָא דַּייָ, וּפַרְסֵיהּ תַּמָּן, לְמַשְׁכְּנֵיהּ; וּכְרוֹ תַּמָּן עַבְדֵי יִצְחָק, בֵּירָא. 
כו,כו וַֽאֲבִימֶ֕לֶךְ הָלַ֥ךְ אֵלָ֖יו מִגְּרָ֑ר וַֽאֲחֻזַּת֙ מֵֽרֵעֵ֔הוּ וּפִיכֹ֖ל שַׂר־צְבָאֽוֹ׃ וַאֲבִימֶלֶךְ, אֲתָא לְוָתֵיהּ מִגְּרָר; וְסִיעַת, מֵרָחֲמוֹהִי, וּפִיכוֹל, רַב חֵילֵיהּ. 
כו,כז וַיֹּ֤אמֶר אֲלֵהֶם֙ יִצְחָ֔ק מַדּ֖וּעַ בָּאתֶ֣ם אֵלָ֑י וְאַתֶּם֙ שְׂנֵאתֶ֣ם אֹתִ֔י וַתְּשַׁלְּח֖וּנִי מֵֽאִתְּכֶֽם׃ וַאֲמַר לְהוֹן יִצְחָק, מָדֵין אֲתֵיתוֹן לְוָתִי; וְאַתּוּן סְנֵיתוֹן יָתִי, וְשַׁלַּחְתּוּנִי מִלְּוָתְכוֹן. 
כו,כח וַיֹּֽאמְר֗וּ רָא֣וֹ רָאִינוּ֮ כִּֽי־הָיָ֣ה יְהוָ֣ה ׀ עִמָּךְ֒ וַנֹּ֗אמֶר תְּהִ֨י נָ֥א אָלָ֛ה בֵּֽינוֹתֵ֖ינוּ בֵּינֵ֣ינוּ וּבֵינֶ֑ךָ וְנִכְרְתָ֥ה בְרִ֖ית עִמָּֽךְ׃ וַאֲמַרוּ, מִחְזָא חֲזֵינָא אֲרֵי הֲוָה מֵימְרָא דַּייָ בְּסַעֲדָךְ, וַאֲמַרְנָא תִּתְקַיַּם כְּעַן מוֹמָתָא דַּהֲוָת בֵּין אֲבָהָתַנָא, בֵּינַנָא וּבֵינָךְ; וְנִגְזַר קְיָם, עִמָּךְ. 
כו,כט אִם־תַּֽעֲשֵׂ֨ה עִמָּ֜נוּ רָעָ֗ה כַּֽאֲשֶׁר֙ לֹ֣א נְגַֽעֲנ֔וּךָ וְכַֽאֲשֶׁ֨ר עָשִׂ֤ינוּ עִמְּךָ֙ רַק־ט֔וֹב וַנְּשַׁלֵּֽחֲךָ֖ בְּשָׁל֑וֹם אַתָּ֥ה עַתָּ֖ה בְּר֥וּךְ יְהוָֽה׃ אִם תַּעֲבֵיד עִמַּנָא בִּישָׁא, כְּמָא דְּלָא אַנְזֵיקְנָךְ, וּכְמָא דַּעֲבַדְנָא עִמָּךְ לְחוֹד טָב, וְשַׁלַּחְנָךְ בִּשְׁלָם; אַתְּ כְּעַן, בְּרִיכָא דַּייָ. 
כו,ל וַיַּ֤עַשׂ לָהֶם֙ מִשְׁתֶּ֔ה וַיֹּֽאכְל֖וּ וַיִּשְׁתּֽוּ׃ וַעֲבַד לְהוֹן מִשְׁתְּיָא, וַאֲכַלוּ וּשְׁתִיאוּ. 
כו,לא וַיַּשְׁכִּ֣ימוּ בַבֹּ֔קֶר וַיִּשָּֽׁבְע֖וּ אִ֣ישׁ לְאָחִ֑יו וַיְשַׁלְּחֵ֣ם יִצְחָ֔ק וַיֵּֽלְכ֥וּ מֵֽאִתּ֖וֹ בְּשָׁלֽוֹם׃ וְאַקְדִּימוּ בְּצַפְרָא, וְקַיִּימוּ גְּבַר לַאֲחוּהִי; וְשַׁלְּחִנּוּן יִצְחָק, וַאֲזַלוּ מִלְּוָתֵיהּ בִּשְׁלָם. 
כו,לב וַיְהִ֣י ׀ בַּיּ֣וֹם הַה֗וּא וַיָּבֹ֨אוּ֙ עַבְדֵ֣י יִצְחָ֔ק וַיַּגִּ֣דוּ ל֔וֹ עַל־אֹד֥וֹת הַבְּאֵ֖ר אֲשֶׁ֣ר חָפָ֑רוּ וַיֹּ֥אמְרוּ ל֖וֹ מָצָ֥אנוּ מָֽיִם׃ וַהֲוָה בְּיוֹמָא הַהוּא, וַאֲתוֹ עַבְדֵי יִצְחָק, וְחַוִּיאוּ לֵיהּ, עַל עֵיסַק בֵּירָא דַּחֲפַרוּ; וַאֲמַרוּ לֵיהּ, אַשְׁכַּחְנָא מַיָּא. 
כו,לג וַיִּקְרָ֥א אֹתָ֖הּ שִׁבְעָ֑ה עַל־כֵּ֤ן שֵׁם־הָעִיר֙ בְּאֵ֣ר שֶׁ֔בַע עַ֖ד הַיּ֥וֹם הַזֶּֽה׃ {ס} וּקְרָא יָתַהּ, שִׁבְעָה; עַל כֵּין שְׁמַהּ דְּקַרְתָּא בְּאֵר שֶׁבַע, עַד יוֹמָא הָדֵין. {ס}
כו,לד וַיְהִ֤י עֵשָׂו֙ בֶּן־אַרְבָּעִ֣ים שָׁנָ֔ה וַיִּקַּ֤ח אִשָּׁה֙ אֶת־יְהוּדִ֔ית בַּת־בְּאֵרִ֖י הַֽחִתִּ֑י וְאֶת־בָּ֣שְׂמַ֔ת בַּת־אֵילֹ֖ן הַֽחִתִּֽי׃ וַהֲוָה עֵשָׂו, בַּר אַרְבְּעִין שְׁנִין, וּנְסֵיב אִתְּתָא יָת יְהוּדִית, בַּת בְּאֵרִי חִתָּאָה--וְיָת בָּשְׂמַת, בַּת אֵילוֹן חִתָּאָה. 
כו,לה וַתִּֽהְיֶ֖יןָ מֹ֣רַת ר֑וּחַ לְיִצְחָ֖ק וּלְרִבְקָֽה׃ {ס} וַהֲוַאָה, מְסָרְבָן וּמַרְגְּזָן, עַל מֵימַר יִצְחָק, וְרִבְקָה. {ס}
כז,א וַֽיְהִי֙ כִּֽי־זָקֵ֣ן יִצְחָ֔ק וַתִּכְהֶ֥יןָ עֵינָ֖יו מֵֽרְאֹ֑ת וַיִּקְרָ֞א אֶת־עֵשָׂ֣ו ׀ בְּנ֣וֹ הַגָּדֹ֗ל וַיֹּ֤אמֶר אֵלָיו֙ בְּנִ֔י וַיֹּ֥אמֶר אֵלָ֖יו הִנֵּֽנִי׃ וַהֲוָה כַּד סִיב יִצְחָק, וּכְהַיָא עֵינוֹהִי מִלְּמִחְזֵי; וּקְרָא יָת עֵשָׂו בְּרֵיהּ רַבָּא, וַאֲמַר לֵיהּ בְּרִי, וַאֲמַר לֵיהּ, הָאֲנָא. 
כז,ב וַיֹּ֕אמֶר הִנֵּה־נָ֖א זָקַ֑נְתִּי לֹ֥א יָדַ֖עְתִּי י֥וֹם מוֹתִֽי׃ וַאֲמַר, הָא כְּעַן סֵיבִית; לֵית אֲנָא יָדַע, יוֹם מוֹתִי. 
כז,ג וְעַתָּה֙ שָׂא־נָ֣א כֵלֶ֔יךָ תֶּלְיְךָ֖ וְקַשְׁתֶּ֑ךָ וְצֵא֙ הַשָּׂדֶ֔ה וְצ֥וּדָה לִּ֖י צידה (צָֽיִד)׃ וּכְעַן סַב כְּעַן זֵינָךְ, סֵיפָךְ וְקַשְׁתָּךְ; וּפוֹק, לְחַקְלָא, וְצוּד לִי, צֵידָא. 
כז,ד וַֽעֲשֵׂה־לִ֨י מַטְעַמִּ֜ים כַּֽאֲשֶׁ֥ר אָהַ֛בְתִּי וְהָבִ֥יאָה לִּ֖י וְאֹכֵ֑לָה בַּֽעֲב֛וּר תְּבָֽרֶכְךָ֥ נַפְשִׁ֖י בְּטֶ֥רֶם אָמֽוּת׃ וַעֲבֵיד לִי תַּבְשִׁילִין כְּמָא דִּרְחֵימִית, וְאַעֵיל לִי--וְאֵיכוֹל:  בְּדִיל דִּתְבָרְכִנָּךְ נַפְשִׁי, עַד לָא אֲמוּת. 
כז,ה וְרִבְקָ֣ה שֹׁמַ֔עַת בְּדַבֵּ֣ר יִצְחָ֔ק אֶל־עֵשָׂ֖ו בְּנ֑וֹ וַיֵּ֤לֶךְ עֵשָׂו֙ הַשָּׂדֶ֔ה לָצ֥וּד צַ֖יִד לְהָבִֽיא׃ וְרִבְקָה שְׁמַעַת--כַּד מַלֵּיל יִצְחָק, עִם עֵשָׂו בְּרֵיהּ; וַאֲזַל עֵשָׂו לְחַקְלָא, לִמְצָד צֵידָא לְאֵיתָאָה. 
כז,ו וְרִבְקָה֙ אָֽמְרָ֔ה אֶל־יַֽעֲקֹ֥ב בְּנָ֖הּ לֵאמֹ֑ר הִנֵּ֤ה שָׁמַ֨עְתִּי֙ אֶת־אָבִ֔יךָ מְדַבֵּ֛ר אֶל־עֵשָׂ֥ו אָחִ֖יךָ לֵאמֹֽר׃ וְרִבְקָה, אֲמַרַת, לְיַעֲקוֹב בְּרַהּ, לְמֵימַר:  הָא שְׁמַעִית מִן אֲבוּךְ, מְמַלֵּיל עִם עֵשָׂו אֲחוּךְ לְמֵימַר. 
כז,ז הָבִ֨יאָה לִּ֥י צַ֛יִד וַֽעֲשֵׂה־לִ֥י מַטְעַמִּ֖ים וְאֹכֵ֑לָה וַאֲבָֽרֶכְכָ֛ה לִפְנֵ֥י יְהוָ֖ה לִפְנֵ֥י מוֹתִֽי׃ אֵיתַא לִי צֵידָא וַעֲבֵיד לִי תַּבְשִׁילִין, וְאֵיכוֹל; וַאֲבָרְכִנָּךְ קֳדָם יְיָ, קֳדָם מוֹתִי. 
כז,ח וְעַתָּ֥ה בְנִ֖י שְׁמַ֣ע בְּקֹלִ֑י לַֽאֲשֶׁ֥ר אֲנִ֖י מְצַוָּ֥ה אֹתָֽךְ׃ וּכְעַן בְּרִי, קַבֵּיל מִנִּי--לְמָא דַּאֲנָא, מְפַקְּדָא יָתָךְ. 
כז,ט לֶךְ־נָא֙ אֶל־הַצֹּ֔אן וְקַֽח־לִ֣י מִשָּׁ֗ם שְׁנֵ֛י גְּדָיֵ֥י עִזִּ֖ים טֹבִ֑ים וְאֶֽעֱשֶׂ֨ה אֹתָ֧ם מַטְעַמִּ֛ים לְאָבִ֖יךָ כַּֽאֲשֶׁ֥ר אָהֵֽב׃ אִיזֵיל כְּעַן, לְעָנָא, וְסַב לִי מִתַּמָּן תְּרֵין גְּדָיֵי עִזִּין, טָבִין; וְאַעֲבֵיד יָתְהוֹן תַּבְשִׁילִין לַאֲבוּךְ, כְּמָא דְּרָחֵים. 
כז,י וְהֵֽבֵאתָ֥ לְאָבִ֖יךָ וְאָכָ֑ל בַּֽעֲבֻ֛ר אֲשֶׁ֥ר יְבָֽרֶכְךָ֖ לִפְנֵ֥י מוֹתֽוֹ׃ וְתַעֵיל לַאֲבוּךְ, וְיֵיכוֹל, בְּדִיל דִּיבָרְכִנָּךְ, קֳדָם מוֹתֵיהּ. 
כז,יא וַיֹּ֣אמֶר יַֽעֲקֹ֔ב אֶל־רִבְקָ֖ה אִמּ֑וֹ הֵ֣ן עֵשָׂ֤ו אָחִי֙ אִ֣ישׁ שָׂעִ֔ר וְאָֽנֹכִ֖י אִ֥ישׁ חָלָֽק׃ וַאֲמַר יַעֲקוֹב, לְרִבְקָה אִמֵּיהּ:  הָא עֵשָׂו אֲחִי גְּבַר סַעֲרָן, וַאֲנָא גְּבַר שְׁעִיעַ. 
כז,יב אוּלַ֤י יְמֻשֵּׁ֨נִי֙ אָבִ֔י וְהָיִ֥יתִי בְעֵינָ֖יו כִּמְתַעְתֵּ֑עַ וְהֵֽבֵאתִ֥י עָלַ֛י קְלָלָ֖ה וְלֹ֥א בְרָכָֽה׃ מָאִם יְמוּשִׁנַּנִי אַבָּא, וְאֵיהֵי בְּעֵינוֹהִי כְּמִתְלָעַב; וְאַיְתִי עָלַי לְוָטִין, וְלָא בִּרְכָן. 
כז,יג וַתֹּ֤אמֶר לוֹ֙ אִמּ֔וֹ עָלַ֥י קִלְלָֽתְךָ֖ בְּנִ֑י אַ֛ךְ שְׁמַ֥ע בְּקֹלִ֖י וְלֵ֥ךְ קַֽח־לִֽי׃ וַאֲמַרַת לֵיהּ אִמֵּיהּ, עֲלַי אִתְאֲמַר בִּנְבוּאָה דְּלָא יֵיתוֹן לְוָטַיָּא עֲלָךְ בְּרִי; בְּרַם קַבֵּיל מִנִּי, וְאִיזֵיל סַב לִי. 
כז,יד וַיֵּ֨לֶךְ֙ וַיִּקַּ֔ח וַיָּבֵ֖א לְאִמּ֑וֹ וַתַּ֤עַשׂ אִמּוֹ֙ מַטְעַמִּ֔ים כַּֽאֲשֶׁ֖ר אָהֵ֥ב אָבִֽיו׃ וַאֲזַל, וּנְסֵיב, וְאַיְתִי, לְאִמֵּיהּ; וַעֲבַדַת אִמֵּיהּ תַּבְשִׁילִין, כְּמָא דְּרָחֵים אֲבוּהִי. 
כז,טו וַתִּקַּ֣ח רִ֠בְקָה אֶת־בִּגְדֵ֨י עֵשָׂ֜ו בְּנָ֤הּ הַגָּדֹל֙ הַֽחֲמֻדֹ֔ת אֲשֶׁ֥ר אִתָּ֖הּ בַּבָּ֑יִת וַתַּלְבֵּ֥שׁ אֶֽת־יַעֲקֹ֖ב בְּנָ֥הּ הַקָּטָֽן׃ וּנְסֵיבַת רִבְקָה יָת לְבוּשֵׁי עֵשָׂו בְּרַהּ רַבָּא, דָּכְיָתָא, דְּעִמַּהּ, בְּבֵיתָא; וְאַלְבֵּישַׁת יָת יַעֲקוֹב, בְּרַהּ זְעֵירָא. 
כז,טז וְאֵ֗ת עֹרֹת֙ גְּדָיֵ֣י הָֽעִזִּ֔ים הִלְבִּ֖ישָׁה עַל־יָדָ֑יו וְעַ֖ל חֶלְקַ֥ת צַוָּארָֽיו׃ וְיָת, מַשְׁכֵּי דִּגְדֵי בְּנֵי עִזֵּי, אַלְבֵּישַׁת, עַל יְדוֹהִי--וְעַל שְׁעִיעוּת, צַוְרֵיהּ. 
כז,יז וַתִּתֵּ֧ן אֶת־הַמַּטְעַמִּ֛ים וְאֶת־הַלֶּ֖חֶם אֲשֶׁ֣ר עָשָׂ֑תָה בְּיַ֖ד יַֽעֲקֹ֥ב בְּנָֽהּ׃ וִיהַבַת יָת תַּבְשִׁילַיָּא וְיָת לַחְמָא, דַּעֲבַדַת, בִּידָא, דְּיַעֲקוֹב בְּרַהּ. 
כז,יח וַיָּבֹ֥א אֶל־אָבִ֖יו וַיֹּ֣אמֶר אָבִ֑י וַיֹּ֣אמֶר הִנֶּ֔נִּי מִ֥י אַתָּ֖ה בְּנִֽי׃ וְעָאל לְוָת אֲבוּהִי, וַאֲמַר אַבָּא; וַאֲמַר הָאֲנָא, מַאן אַתְּ בְּרִי. 
כז,יט וַיֹּ֨אמֶר יַֽעֲקֹ֜ב אֶל־אָבִ֗יו אָֽנֹכִי֙ עֵשָׂ֣ו בְּכֹרֶ֔ךָ עָשִׂ֕יתִי כַּֽאֲשֶׁ֥ר דִּבַּ֖רְתָּ אֵלָ֑י קֽוּם־נָ֣א שְׁבָ֗ה וְאָכְלָה֙ מִצֵּידִ֔י בַּֽעֲב֖וּר תְּבָֽרְכַ֥נִּי נַפְשֶֽׁךָ׃ וַאֲמַר יַעֲקוֹב לַאֲבוּהִי, אֲנָא עֵשָׂו בֻּכְרָךְ--עֲבַדִית, כְּמָא דְּמַלֵּילְתָּא עִמִּי; קוּם כְּעַן אִסְתְּחַר, וֶאֱכוֹל מִצֵּידִי--בְּדִיל, דִּתְבָרְכִנַּנִי נַפְשָׁךְ. 
כז,כ וַיֹּ֤אמֶר יִצְחָק֙ אֶל־בְּנ֔וֹ מַה־זֶּ֛ה מִהַ֥רְתָּ לִמְצֹ֖א בְּנִ֑י וַיֹּ֕אמֶר כִּ֥י הִקְרָ֛ה יְהוָ֥ה אֱלֹהֶ֖יךָ לְפָנָֽי׃ וַאֲמַר יִצְחָק לִבְרֵיהּ, מָא דֵּין אוֹחִיתָא לְאַשְׁכָּחָא בְּרִי; וַאֲמַר, אֲרֵי זָמֵין יְיָ אֱלָהָךְ קֳדָמָי. 
כז,כא וַיֹּ֤אמֶר יִצְחָק֙ אֶֽל־יַעֲקֹ֔ב גְּשָׁה־נָּ֥א וַאֲמֻֽשְׁךָ֖ בְּנִ֑י הַֽאַתָּ֥ה זֶ֛ה בְּנִ֥י עֵשָׂ֖ו אִם־לֹֽא׃ וַאֲמַר יִצְחָק לְיַעֲקוֹב, קְרַב כְּעַן וַאֲמוּשִׁנָּךְ בְּרִי:  הַאַתְּ דֵּין בְּרִי עֵשָׂו, אִם לָא. 
כז,כב וַיִּגַּ֧שׁ יַֽעֲקֹ֛ב אֶל־יִצְחָ֥ק אָבִ֖יו וַיְמֻשֵּׁ֑הוּ וַיֹּ֗אמֶר הַקֹּל֙ ק֣וֹל יַֽעֲקֹ֔ב וְהַיָּדַ֖יִם יְדֵ֥י עֵשָֽׂו׃ וּקְרֵיב יַעֲקוֹב לְוָת יִצְחָק אֲבוּהִי, וּמָשֵׁיהּ; וַאֲמַר, קָלָא קָלֵיהּ דְּיַעֲקוֹב, וִידַיָּא, יְדֵי עֵשָׂו. 
כז,כג וְלֹ֣א הִכִּיר֔וֹ כִּֽי־הָי֣וּ יָדָ֗יו כִּידֵ֛י עֵשָׂ֥ו אָחִ֖יו שְׂעִרֹ֑ת וַֽיְבָרְכֵֽהוּ׃ וְלָא אִשְׁתְּמוֹדְעֵיהּ--אֲרֵי הֲוַאָה יְדוֹהִי כִּידֵי עֵשָׂו אֲחוּהִי, סַעֲרָנְיָן; וּבָרְכֵיהּ. 
כז,כד וַיֹּ֕אמֶר אַתָּ֥ה זֶ֖ה בְּנִ֣י עֵשָׂ֑ו וַיֹּ֖אמֶר אָֽנִי׃ וַאֲמַר, אַתְּ דֵּין בְּרִי עֵשָׂו; וַאֲמַר, אֲנָא. 
כז,כה וַיֹּ֗אמֶר הַגִּ֤שָׁה לִּי֙ וְאֹֽכְלָה֙ מִצֵּ֣יד בְּנִ֔י לְמַ֥עַן תְּבָֽרֶכְךָ֖ נַפְשִׁ֑י וַיַּגֶּשׁ־לוֹ֙ וַיֹּאכַ֔ל וַיָּ֧בֵא ל֦וֹ יַ֖יִן וַיֵּֽשְׁתְּ׃ וַאֲמַר, קָרֵיב לִי וְאֵיכוֹל מִצֵּידָא דִּבְרִי--בְּדִיל דִּתְבָרְכִנָּךְ, נַפְשִׁי; וְקָרֵיב לֵיהּ, וַאֲכַל, וְאַעֵיל לֵיהּ חַמְרָא, וּשְׁתִי. 
כז,כו וַיֹּ֥אמֶר אֵלָ֖יו יִצְחָ֣ק אָבִ֑יו גְּשָׁה־נָּ֥א וּשְׁקָה־לִּ֖י בְּנִֽי׃ וַאֲמַר לֵיהּ, יִצְחָק אֲבוּהִי:  קְרַב כְּעַן וְשַׁק לִי, בְּרִי. 
כז,כז וַיִּגַּשׁ֙ וַיִּשַּׁק־ל֔וֹ וַיָּ֛רַח אֶת־רֵ֥יחַ בְּגָדָ֖יו וַֽיְבָרְכֵ֑הוּ וַיֹּ֗אמֶר רְאֵה֙ רֵ֣יחַ בְּנִ֔י כְּרֵ֣יחַ שָׂדֶ֔ה אֲשֶׁ֥ר בֵּֽרְכ֖וֹ יְהוָֽה׃ וּקְרֵיב, וְנַשֵּׁיק לֵיהּ, וַאֲרִיחַ יָת רֵיחַ לְבוּשׁוֹהִי, וּבָרְכֵיהּ; וַאֲמַר, חֲזִי רֵיחָא דִּבְרִי, כְּרֵיחַ חַקְלָא, דְּבָרְכֵיהּ יְיָ. 
כז,כח וְיִֽתֶּן־לְךָ֙ הָֽאֱלֹהִ֔ים מִטַּל֙ הַשָּׁמַ֔יִם וּמִשְׁמַנֵּ֖י הָאָ֑רֶץ וְרֹ֥ב דָּגָ֖ן וְתִירֹֽשׁ׃ וְיִתֵּין לָךְ, יְיָ, מִטַּלָּא דִּשְׁמַיָּא, וּמִטּוּבָא דְּאַרְעָא--וְסַגְיוּת עֲבוּר, וַחֲמַר. 
כז,כט יַֽעַבְד֣וּךָ עַמִּ֗ים וישתחו (וְיִֽשְׁתַּחֲו֤וּ) לְךָ֙ לְאֻמִּ֔ים הֱוֵ֤ה גְבִיר֙ לְאַחֶ֔יךָ וְיִשְׁתַּֽחֲו֥וּ לְךָ֖ בְּנֵ֣י אִמֶּ֑ךָ אֹֽרְרֶ֣יךָ אָר֔וּר וּֽמְבָרְכֶ֖יךָ בָּרֽוּךְ׃ יִפְלְחוּנָּךְ עַמְמִין, וְיִשְׁתַּעְבְּדָן לָךְ מַלְכְוָן--הֱוִי רָב לַאֲחָךְ, וְיִסְגְּדוּן לָךְ בְּנֵי אִמָּךְ; לִיטָךְ יְהוֹן לִיטִין, וּבְרִיכָךְ יְהוֹן בְּרִיכִין. 
כז,ל וַיְהִ֗י כַּֽאֲשֶׁ֨ר כִּלָּ֣ה יִצְחָק֮ לְבָרֵ֣ךְ אֶֽת־יַעֲקֹב֒ וַיְהִ֗י אַ֣ךְ יָצֹ֤א יָצָא֙ יַֽעֲקֹ֔ב מֵאֵ֥ת פְּנֵ֖י יִצְחָ֣ק אָבִ֑יו וְעֵשָׂ֣ו אָחִ֔יו בָּ֖א מִצֵּידֽוֹ׃ וַהֲוָה, כַּד שֵׁיצִי יִצְחָק לְבָרָכָא יָת יַעֲקוֹב, וַהֲוָה בְּרַם מִפָּק נְפַק יַעֲקוֹב, מִן קֳדָם יִצְחָק אֲבוּהִי; וְעֵשָׂו אֲחוּהִי, עָאל מִצֵּידֵיהּ. 
כז,לא וַיַּ֤עַשׂ גַּם־הוּא֙ מַטְעַמִּ֔ים וַיָּבֵ֖א לְאָבִ֑יו וַיֹּ֣אמֶר לְאָבִ֗יו יָקֻ֤ם אָבִי֙ וְיֹאכַל֙ מִצֵּ֣יד בְּנ֔וֹ בַּֽעֲבֻ֖ר תְּבָֽרְכַ֥נִּי נַפְשֶֽׁךָ׃ וַעֲבַד אַף הוּא תַּבְשִׁילִין, וְאַעֵיל לַאֲבוּהִי; וַאֲמַר לַאֲבוּהִי, יְקוּם אַבָּא וְיֵיכוֹל מִצֵּידָא דִּבְרֵיהּ--בְּדִיל, דִּתְבָרְכִנַּנִי נַפְשָׁךְ. 
כז,לב וַיֹּ֥אמֶר ל֛וֹ יִצְחָ֥ק אָבִ֖יו מִי־אָ֑תָּה וַיֹּ֕אמֶר אֲנִ֛י בִּנְךָ֥ בְכֹֽרְךָ֖ עֵשָֽׂו׃ וַאֲמַר לֵיהּ יִצְחָק אֲבוּהִי, מַאן אַתְּ; וַאֲמַר, אֲנָא בְּרָךְ בֻּכְרָךְ עֵשָׂו. 
כז,לג וַיֶּֽחֱרַ֨ד יִצְחָ֣ק חֲרָדָה֮ גְּדֹלָ֣ה עַד־מְאֹד֒ וַיֹּ֡אמֶר מִֽי־אֵפ֡וֹא ה֣וּא הַצָּֽד־צַיִד֩ וַיָּ֨בֵא לִ֜י וָֽאֹכַ֥ל מִכֹּ֛ל בְּטֶ֥רֶם תָּב֖וֹא וָאֲבָֽרְכֵ֑הוּ גַּם־בָּר֖וּךְ יִֽהְיֶֽה׃ וּתְוַהּ יִצְחָק תִּוְהָא, רַבָּא עַד לַחְדָּא, וַאֲמַר מַאן הוּא דֵּיכִי דְּצָד צֵידָא וְאַעֵיל לִי וַאֲכַלִית מִכּוֹלָא עַד לָא תֵּיעוֹל, וּבָרֵיכְתֵּיהּ; אַף בְּרִיךְ, יְהֵי. 
כז,לד כִּשְׁמֹ֤עַ עֵשָׂו֙ אֶת־דִּבְרֵ֣י אָבִ֔יו וַיִּצְעַ֣ק צְעָקָ֔ה גְּדֹלָ֥ה וּמָרָ֖ה עַד־מְאֹ֑ד וַיֹּ֣אמֶר לְאָבִ֔יו בָּֽרְכֵ֥נִי גַם־אָ֖נִי אָבִֽי׃ כַּד שְׁמַע עֵשָׂו, יָת פִּתְגָמֵי אֲבוּהִי, וּצְוַח צִוְחָא, רַבָּא וּמָרִירָא עַד לַחְדָּא; וַאֲמַר לַאֲבוּהִי, בָּרֵיכְנִי אַף לִי אַבָּא. 
כז,לה וַיֹּ֕אמֶר בָּ֥א אָחִ֖יךָ בְּמִרְמָ֑ה וַיִּקַּ֖ח בִּרְכָתֶֽךָ׃ וַאֲמַר, עָאל אֲחוּךְ בְּחָכְמָא; וְקַבֵּיל, בִּרְכְתָךְ. 
כז,לו וַיֹּ֡אמֶר הֲכִי֩ קָרָ֨א שְׁמ֜וֹ יַֽעֲקֹ֗ב וַֽיַּעְקְבֵ֨נִי֙ זֶ֣ה פַֽעֲמַ֔יִם אֶת־בְּכֹֽרָתִ֣י לָקָ֔ח וְהִנֵּ֥ה עַתָּ֖ה לָקַ֣ח בִּרְכָתִ֑י וַיֹּאמַ֕ר הֲלֹֽא־אָצַ֥לְתָּ לִּ֖י בְּרָכָֽה׃ וַאֲמַר יָאוּת קְרָא שְׁמֵיהּ יַעֲקוֹב, וְחַכְמַנִי דְּנָן תַּרְתֵּין זִמְנִין--יָת בְּכֵירוּתִי נְסֵיב, וְהָא כְּעַן קַבֵּיל בִּרְכְתִי; וַאֲמַר, הֲלָא שְׁבַקְתְּ לִי בִּרְכָא. 
כז,לז וַיַּ֨עַן יִצְחָ֜ק וַיֹּ֣אמֶר לְעֵשָׂ֗ו הֵ֣ן גְּבִ֞יר שַׂמְתִּ֥יו לָךְ֙ וְאֶת־כָּל־אֶחָ֗יו נָתַ֤תִּי לוֹ֙ לַֽעֲבָדִ֔ים וְדָגָ֥ן וְתִירֹ֖שׁ סְמַכְתִּ֑יו וּלְכָ֣ה אֵפ֔וֹא מָ֥ה אֶֽעֱשֶׂ֖ה בְּנִֽי׃ וַאֲתֵיב יִצְחָק וַאֲמַר לְעֵשָׂו, הָא רָב שַׁוִּיתֵיהּ עִלָּוָךְ וְיָת כָּל אֲחוֹהִי יְהַבִית לֵיהּ לְעַבְדִּין, וּבַעֲבוּר וּבַחֲמַר, סְעַדְתֵּיהּ; וְלָךְ כְּעַן, מָא אַעֲבֵיד בְּרִי. 
כז,לח וַיֹּ֨אמֶר עֵשָׂ֜ו אֶל־אָבִ֗יו הַֽבְרָכָ֨ה אַחַ֤ת הִֽוא־לְךָ֙ אָבִ֔י בָּֽרְכֵ֥נִי גַם־אָ֖נִי אָבִ֑י וַיִּשָּׂ֥א עֵשָׂ֛ו קֹל֖וֹ וַיֵּֽבְךְּ׃ וַאֲמַר עֵשָׂו לַאֲבוּהִי, הֲבִרְכְתָא חֲדָא הִיא לָךְ אַבָּא--בָּרֵיכְנִי אַף לִי, אַבָּא; וַאֲרֵים עֵשָׂו קָלֵיהּ, וּבְכָא. 
כז,לט וַיַּ֛עַן יִצְחָ֥ק אָבִ֖יו וַיֹּ֣אמֶר אֵלָ֑יו הִנֵּ֞ה מִשְׁמַנֵּ֤י הָאָ֨רֶץ֙ יִֽהְיֶ֣ה מֽוֹשָׁבֶ֔ךָ וּמִטַּ֥ל הַשָּׁמַ֖יִם מֵעָֽל׃ וַאֲתֵיב יִצְחָק אֲבוּהִי, וַאֲמַר לֵיהּ:  הָא מִטּוּבָא דְּאַרְעָא, יְהֵי מוֹתְבָךְ, וּמִטַּלָּא דִּשְׁמַיָּא, מִלְּעֵילָא. 
כז,מ וְעַל־חַרְבְּךָ֣ תִֽחְיֶ֔ה וְאֶת־אָחִ֖יךָ תַּֽעֲבֹ֑ד וְהָיָה֙ כַּֽאֲשֶׁ֣ר תָּרִ֔יד וּפָֽרַקְתָּ֥ עֻלּ֖וֹ מֵעַ֥ל צַוָּארֶֽךָ׃ וְעַל חַרְבָּךְ תֵּיחֵי, וְיָת אֲחוּךְ תִּפְלַח; וִיהֵי כַּד יִעְבְּרוּן בְּנוֹהִי עַל פִּתְגָמֵי אוֹרָיְתָא, וְתַעְדֵּי נִירֵיהּ מֵעַל צַוְרָךְ. 
כז,מא וַיִּשְׂטֹ֤ם עֵשָׂו֙ אֶֽת־יַעֲקֹ֔ב עַ֨ל־הַבְּרָכָ֔ה אֲשֶׁ֥ר בֵּֽרְכ֖וֹ אָבִ֑יו וַיֹּ֨אמֶר עֵשָׂ֜ו בְּלִבּ֗וֹ יִקְרְבוּ֙ יְמֵי֙ אֵ֣בֶל אָבִ֔י וְאַֽהַרְגָ֖ה אֶת־יַֽעֲקֹ֥ב אָחִֽי׃ וּנְטַר עֵשָׂו, דְּבָבוּ לְיַעֲקוֹב, עַל בִּרְכְתָא, דְּבָרְכֵיהּ אֲבוּהִי; וַאֲמַר עֵשָׂו בְּלִבֵּיהּ, יִקְרְבוּן יוֹמֵי אֶבְלֵיהּ דְּאַבָּא, וְאֶקְטוֹל, יָת יַעֲקוֹב אֲחִי. 
כז,מב וַיֻּגַּ֣ד לְרִבְקָ֔ה אֶת־דִּבְרֵ֥י עֵשָׂ֖ו בְּנָ֣הּ הַגָּדֹ֑ל וַתִּשְׁלַ֞ח וַתִּקְרָ֤א לְיַֽעֲקֹב֙ בְּנָ֣הּ הַקָּטָ֔ן וַתֹּ֣אמֶר אֵלָ֔יו הִנֵּה֙ עֵשָׂ֣ו אָחִ֔יךָ מִתְנַחֵ֥ם לְךָ֖ לְהָרְגֶֽךָ׃ וְאִתְחַוַּא לְרִבְקָה, יָת פִּתְגָמֵי עֵשָׂו בְּרַהּ רַבָּא; וּשְׁלַחַת וּקְרָת לְיַעֲקוֹב, בְּרַהּ זְעֵירָא, וַאֲמַרַת לֵיהּ, הָא עֵשָׂו אֲחוּךְ כָּמֵין לָךְ לְמִקְטְלָךְ. 
כז,מג וְעַתָּ֥ה בְנִ֖י שְׁמַ֣ע בְּקֹלִ֑י וְק֧וּם בְּרַח־לְךָ֛ אֶל־לָבָ֥ן אָחִ֖י חָרָֽנָה׃ וּכְעַן בְּרִי, קַבֵּיל מִנִּי; וְקוּם אִיזֵיל לָךְ לְוָת לָבָן אֲחִי, לְחָרָן. 
כז,מד וְיָֽשַׁבְתָּ֥ עִמּ֖וֹ יָמִ֣ים אֲחָדִ֑ים עַ֥ד אֲשֶׁר־תָּשׁ֖וּב חֲמַ֥ת אָחִֽיךָ׃ וְתִתֵּיב עִמֵּיהּ, יוֹמִין זְעֵירִין--עַד דִּתְתוּב, חִמְתָּא דַּאֲחוּךְ. 
כז,מה עַד־שׁ֨וּב אַף־אָחִ֜יךָ מִמְּךָ֗ וְשָׁכַח֙ אֵ֣ת אֲשֶׁר־עָשִׂ֣יתָ לּ֔וֹ וְשָֽׁלַחְתִּ֖י וּלְקַחְתִּ֣יךָ מִשָּׁ֑ם לָמָ֥ה אֶשְׁכַּ֛ל גַּם־שְׁנֵיכֶ֖ם י֥וֹם אֶחָֽד׃ עַד דִּיתוּב רֻגְזָא דַּאֲחוּךְ מִנָּךְ, וְיִתְנְשֵׁי יָת דַּעֲבַדְתְּ לֵיהּ, וְאֶשְׁלַח, וְאֶדְבְּרִנָּךְ מִתַּמָּן; לְמָא אֶתְכּוּל אַף תַּרְוֵיכוֹן, יוֹמָא חַד. 
כז,מו וַתֹּ֤אמֶר רִבְקָה֙ אֶל־יִצְחָ֔ק קַ֣צְתִּי בְחַיַּ֔י מִפְּנֵ֖י בְּנ֣וֹת חֵ֑ת אִם־לֹקֵ֣חַ יַֽ֠עֲקֹב אִשָּׁ֨ה מִבְּנֽוֹת־חֵ֤ת כָּאֵ֨לֶּה֙ מִבְּנ֣וֹת הָאָ֔רֶץ לָ֥מָּה לִּ֖י חַיִּֽים׃ וַאֲמַרַת רִבְקָה, לְיִצְחָק, עַקִית בְּחַיַּי, מִן קֳדָם בְּנָת חִתָּאָה; אִם נָסֵיב יַעֲקוֹב אִתְּתָא מִבְּנָת חִתָּאָה כְּאִלֵּין, מִבְּנָת אַרְעָא--לְמָא לִי, חַיִין. 
כח,א וַיִּקְרָ֥א יִצְחָ֛ק אֶֽל־יַעֲקֹ֖ב וַיְבָ֣רֶךְ אֹת֑וֹ וַיְצַוֵּ֨הוּ֙ וַיֹּ֣אמֶר ל֔וֹ לֹֽא־תִקַּ֥ח אִשָּׁ֖ה מִבְּנ֥וֹת כְּנָֽעַן׃ וּקְרָא יִצְחָק לְיַעֲקוֹב, וּבָרֵיךְ יָתֵיהּ; וּפַקְּדֵיהּ וַאֲמַר לֵיהּ, לָא תִּסַּב אִתְּתָא מִבְּנָת כְּנָעַן. 
כח,ב ק֥וּם לֵךְ֙ פַּדֶּ֣נָֽה אֲרָ֔ם בֵּ֥יתָה בְתוּאֵ֖ל אֲבִ֣י אִמֶּ֑ךָ וְקַח־לְךָ֤ מִשָּׁם֙ אִשָּׁ֔ה מִבְּנ֥וֹת לָבָ֖ן אֲחִ֥י אִמֶּֽךָ׃ קוּם אִיזֵיל לְפַדַּן דַּאֲרָם, לְבֵית בְּתוּאֵל אֲבוּהָא דְּאִמָּךְ; וְסַב לָךְ מִתַּמָּן אִתְּתָא, מִבְּנָת לָבָן אֲחוּהָא דְּאִמָּךְ. 
כח,ג וְאֵ֤ל שַׁדַּי֙ יְבָרֵ֣ךְ אֹֽתְךָ֔ וְיַפְרְךָ֖ וְיַרְבֶּ֑ךָ וְהָיִ֖יתָ לִקְהַ֥ל עַמִּֽים׃ וְאֵל שַׁדַּי יְבָרֵיךְ יָתָךְ, וְיַפְּשִׁנָּךְ וְיַסְגֵּינָךְ; וּתְהֵי, לְכִנְשָׁת שִׁבְטִין. 
כח,ד וְיִֽתֶּן־לְךָ֙ אֶת־בִּרְכַּ֣ת אַבְרָהָ֔ם לְךָ֖ וּלְזַרְעֲךָ֣ אִתָּ֑ךְ לְרִשְׁתְּךָ֙ אֶת־אֶ֣רֶץ מְגֻרֶ֔יךָ אֲשֶׁר־נָתַ֥ן אֱלֹהִ֖ים לְאַבְרָהָֽם׃ וְיִתֵּין לָךְ יָת בִּרְכְתָא דְּאַבְרָהָם, לָךְ וְלִבְנָךְ עִמָּךְ--לְמֵירְתָךְ יָת אֲרַע תּוֹתָבוּתָךְ, דִּיהַב יְיָ לְאַבְרָהָם. 
כח,ה וַיִּשְׁלַ֤ח יִצְחָק֙ אֶֽת־יַעֲקֹ֔ב וַיֵּ֖לֶךְ פַּדֶּ֣נָֽה אֲרָ֑ם אֶל־לָבָ֤ן בֶּן־בְּתוּאֵל֙ הָֽאֲרַמִּ֔י אֲחִ֣י רִבְקָ֔ה אֵ֥ם יַֽעֲקֹ֖ב וְעֵשָֽׂו׃ וּשְׁלַח יִצְחָק יָת יַעֲקוֹב, וַאֲזַל לְפַדַּן דַּאֲרָם--לְוָת לָבָן בַּר בְּתוּאֵל, אֲרַמָּאָה, אֲחוּהָא דְּרִבְקָה, אִמֵּיהּ דְּיַעֲקוֹב וְעֵשָׂו. 
כח,ו וַיַּ֣רְא עֵשָׂ֗ו כִּֽי־בֵרַ֣ךְ יִצְחָק֮ אֶֽת־יַעֲקֹב֒ וְשִׁלַּ֤ח אֹתוֹ֙ פַּדֶּ֣נָֽה אֲרָ֔ם לָקַֽחַת־ל֥וֹ מִשָּׁ֖ם אִשָּׁ֑ה בְּבָֽרְכ֣וֹ אֹת֔וֹ וַיְצַ֤ו עָלָיו֙ לֵאמֹ֔ר לֹֽא־תִקַּ֥ח אִשָּׁ֖ה מִבְּנ֥וֹת כְּנָֽעַן׃ וַחֲזָא עֵשָׂו, אֲרֵי בָּרֵיךְ יִצְחָק יָת יַעֲקוֹב, וְשַׁלַּח יָתֵיהּ לְפַדַּן דַּאֲרָם, לְמִסַּב לֵיהּ מִתַּמָּן אִתְּתָא:  בִּדְבָרֵיךְ יָתֵיהּ--וּפַקֵּיד עֲלוֹהִי לְמֵימַר, לָא תִּסַּב אִתְּתָא מִבְּנָת כְּנָעַן. 
כח,ז וַיִּשְׁמַ֣ע יַֽעֲקֹ֔ב אֶל־אָבִ֖יו וְאֶל־אִמּ֑וֹ וַיֵּ֖לֶךְ פַּדֶּ֥נָֽה אֲרָֽם׃ וְקַבֵּיל יַעֲקוֹב, מִן אֲבוּהִי וּמִן אִמֵּיהּ; וַאֲזַל, לְפַדַּן דַּאֲרָם. 
כח,ח וַיַּ֣רְא עֵשָׂ֔ו כִּ֥י רָע֖וֹת בְּנ֣וֹת כְּנָ֑עַן בְּעֵינֵ֖י יִצְחָ֥ק אָבִֽיו׃ וַחֲזָא עֵשָׂו, אֲרֵי בִּישָׁן בְּנָת כְּנָעַן, בְּעֵינֵי, יִצְחָק אֲבוּהִי. 
כח,ט וַיֵּ֥לֶךְ עֵשָׂ֖ו אֶל־יִשְׁמָעֵ֑אל וַיִּקַּ֡ח אֶֽת־מָחֲלַ֣ת ׀ בַּת־יִשְׁמָעֵ֨אל בֶּן־אַבְרָהָ֜ם אֲח֧וֹת נְבָי֛וֹת עַל־נָשָׁ֖יו ל֥וֹ לְאִשָּֽׁה׃ {ס} וַאֲזַל עֵשָׂו, לְוָת יִשְׁמָעֵאל; וּנְסֵיב יָת מָחֲלַת בַּת יִשְׁמָעֵאל בַּר אַבְרָהָם אֲחָתֵיהּ דִּנְבָיוֹת, עַל נְשׁוֹהִי--לֵיהּ לְאִתּוּ. {ס}
