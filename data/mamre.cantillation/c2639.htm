תהילים פרק לט

א לַמְנַצֵּ֥חַ לידיתון (לִֽידוּת֗וּן)    מִזְמ֥וֹר לְדָוִֽד׃
ב אָמַ֗רְתִּי    אֶ֥שְׁמְרָ֣ה דְרָכַי֮ מֵֽחֲט֪וֹא בִלְשׁ֫וֹנִ֥י
אֶשְׁמְרָ֥ה לְפִ֥י מַחְס֑וֹם    בְּעֹ֖ד רָשָׁ֣ע לְנֶגְדִּֽי׃
ג נֶֽאֱלַ֣מְתִּי דֽ֭וּמִיָּה הֶֽחֱשֵׁ֣יתִי מִטּ֑וֹב    וּכְאֵבִ֥י נֶעְכָּֽר׃
ד חַם־לִבִּ֨י ׀ בְּקִרְבִּ֗י בַּֽהֲגִיגִ֥י תִבְעַר־אֵ֑שׁ    דִּ֝בַּ֗רְתִּי בִּלְשׁוֹנִֽי׃
ה הֽוֹדִ֘יעֵ֤נִי יְהוָ֨ה ׀ קִצִּ֗י וּמִדַּ֣ת יָמַ֣י מַה־הִ֑יא    אֵֽ֝דְעָ֗ה מֶֽה־חָדֵ֥ל אָֽנִי׃
ו הִנֵּ֤ה טְפָח֨וֹת ׀ נָ֘תַ֤תָּה יָמַ֗י    וְחֶלְדִּ֣י כְאַ֣יִן נֶגְדֶּ֑ךָ
אַ֥ךְ כָּֽל־הֶ֥בֶל כָּל־אָ֝דָ֗ם    נִצָּ֥ב סֶֽלָה׃
ז אַךְ־בְּצֶ֤לֶם ׀ יִֽתְהַלֶּךְ־אִ֗ישׁ    אַךְ־הֶ֥בֶל יֶֽהֱמָי֑וּן
יִ֝צְבֹּ֗ר    וְֽלֹא־יֵדַ֥ע מִֽי־אֹסְפָֽם׃
ח וְעַתָּ֣ה מַה־קִּוִּ֣יתִי אֲדֹנָ֑י    תּֽ֝וֹחַלְתִּ֗י לְךָ֣ הִֽיא׃
ט מִכָּל־פְּשָׁעַ֥י הַצִּילֵ֑נִי    חֶרְפַּ֥ת נָ֝בָ֗ל אַל־תְּשִׂימֵֽנִי׃
י נֶֽ֭אֱלַמְתִּי לֹ֣א אֶפְתַּח־פִּ֑י    כִּ֖י אַתָּ֣ה עָשִֽׂיתָ׃
יא הָסֵ֣ר מֵֽעָלַ֣י נִגְעֶ֑ךָ    מִתִּגְרַ֥ת יָֽ֝דְךָ֗ אֲנִ֣י כָלִֽיתִי׃
יב בְּֽתוֹכָ֘ח֤וֹת עַל־עָו֨‍ֹן ׀ יִסַּ֬רְתָּ אִ֗ישׁ    וַתֶּ֣מֶס כָּעָ֣שׁ חֲמוּד֑וֹ
אַ֤ךְ הֶ֖בֶל כָּל־אָדָ֣ם    סֶֽלָה׃
יג שִׁ֥מְעָֽה תְפִלָּתִ֨י ׀ יְהוָ֡ה    וְשַׁוְעָתִ֨י ׀ הַֽאֲזִינָה֮
אֶֽל־דִּמְעָתִ֗י    אַֽל־תֶּ֫חֱרַ֥שׁ
כִּ֤י גֵ֣ר אָֽנֹכִ֣י עִמָּ֑ךְ    תּ֝וֹשָׁ֗ב כְּכָל־אֲבוֹתָֽי׃
יד הָשַׁ֣ע מִמֶּ֣נִּי וְאַבְלִ֑יגָה    בְּטֶ֖רֶם אֵלֵ֣ךְ וְאֵינֶֽנִּי׃
