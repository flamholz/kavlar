וייגש

מד,יח וַיִּגַּ֨שׁ אֵלָ֜יו יְהוּדָ֗ה וַיֹּאמֶר֮ בִּ֣י אֲדֹנִי֒ יְדַבֶּר־נָ֨א עַבְדְּךָ֤ דָבָר֙ בְּאָזְנֵ֣י אֲדֹנִ֔י וְאַל־יִ֥חַר אַפְּךָ֖ בְּעַבְדֶּ֑ךָ כִּ֥י כָמ֖וֹךָ כְּפַרְעֹֽה׃ וּקְרֵיב לְוָתֵיהּ יְהוּדָה, וַאֲמַר בְּבָעוּ רִבּוֹנִי, יְמַלֵּיל כְּעַן עַבְדָּךְ פִּתְגָמָא קֳדָם רִבּוֹנִי, וְלָא יִתְקַף רֻגְזָךְ בְּעַבְדָּךְ:  אֲרֵי כְּפַרְעֹה, כֵּין אַתְּ. 
מד,יט אֲדֹנִ֣י שָׁאַ֔ל אֶת־עֲבָדָ֖יו לֵאמֹ֑ר הֲיֵשׁ־לָכֶ֥ם אָ֖ב אוֹ־אָֽח׃ רִבּוֹנִי שַׁאֵיל, יָת עַבְדּוֹהִי לְמֵימַר:  הַאִית לְכוֹן אַבָּא, אוֹ אֲחָא. 
מד,כ וַנֹּ֨אמֶר֙ אֶל־אֲדֹנִ֔י יֶשׁ־לָ֨נוּ֙ אָ֣ב זָקֵ֔ן וְיֶ֥לֶד זְקֻנִ֖ים קָטָ֑ן וְאָחִ֣יו מֵ֔ת וַיִּוָּתֵ֨ר ה֧וּא לְבַדּ֛וֹ לְאִמּ֖וֹ וְאָבִ֥יו אֲהֵבֽוֹ׃ וַאֲמַרְנָא, לְרִבּוֹנִי, אִית לַנָא אַבָּא סָבָא, וּבַר סֵיבְתִין זְעֵיר; וַאֲחוּהִי מִית, וְאִשְׁתְּאַר הוּא בִּלְחוֹדוֹהִי לְאִמֵּיהּ וַאֲבוּהִי רָחֵים לֵיהּ. 
מד,כא וַתֹּ֨אמֶר֙ אֶל־עֲבָדֶ֔יךָ הֽוֹרִדֻ֖הוּ אֵלָ֑י וְאָשִׂ֥ימָה עֵינִ֖י עָלָֽיו׃ וַאֲמַרְתְּ, לְעַבְדָּךְ, אַחֲתוּהִי, לְוָתִי; וַאֲשַׁוֵּי עֵינִי, עֲלוֹהִי. 
מד,כב וַנֹּ֨אמֶר֙ אֶל־אֲדֹנִ֔י לֹֽא־יוּכַ֥ל הַנַּ֖עַר לַֽעֲזֹ֣ב אֶת־אָבִ֑יו וְעָזַ֥ב אֶת־אָבִ֖יו וָמֵֽת׃ וַאֲמַרְנָא, לְרִבּוֹנִי, לָא יִכּוֹל עוּלֵימָא, לְמִשְׁבַּק יָת אֲבוּהִי:  אִם יִשְׁבּוֹק יָת אֲבוּהִי, יְמוּת. 
מד,כג וַתֹּ֨אמֶר֙ אֶל־עֲבָדֶ֔יךָ אִם־לֹ֥א יֵרֵ֛ד אֲחִיכֶ֥ם הַקָּטֹ֖ן אִתְּכֶ֑ם לֹ֥א תֹֽסִפ֖וּן לִרְא֥וֹת פָּנָֽי׃ וַאֲמַרְתְּ, לְעַבְדָּךְ, אִם לָא יֵיחוֹת אֲחוּכוֹן זְעֵירָא, עִמְּכוֹן--לָא תֵּיסְפוּן, לְמִחְזֵי אַפָּי. 
מד,כד וַֽיְהִי֙ כִּ֣י עָלִ֔ינוּ אֶֽל־עַבְדְּךָ֖ אָבִ֑י וַנַּ֨גֶּד־ל֔וֹ אֵ֖ת דִּבְרֵ֥י אֲדֹנִֽי׃ וַהֲוָה כַּד סְלֵיקְנָא, לְוָת עַבְדָּךְ אַבָּא; וְחַוִּינָא לֵיהּ--יָת, פִּתְגָמֵי רִבּוֹנִי. 
מד,כה וַיֹּ֖אמֶר אָבִ֑ינוּ שֻׁ֖בוּ שִׁבְרוּ־לָ֥נוּ מְעַט־אֹֽכֶל׃ וַאֲמַר, אֲבוּנָא:  תּוּבוּ, זְבוּנוּ לַנָא זְעֵיר עֲבוּרָא. 
מד,כו וַנֹּ֕אמֶר לֹ֥א נוּכַ֖ל לָרֶ֑דֶת אִם־יֵשׁ֩ אָחִ֨ינוּ הַקָּטֹ֤ן אִתָּ֨נוּ֙ וְיָרַ֔דְנוּ כִּי־לֹ֣א נוּכַ֗ל לִרְאוֹת֙ פְּנֵ֣י הָאִ֔ישׁ וְאָחִ֥ינוּ הַקָּטֹ֖ן אֵינֶ֥נּוּ אִתָּֽנוּ׃ וַאֲמַרְנָא, לָא נִכּוֹל לְמֵיחַת:  אִם אִית אֲחוּנָא זְעֵירָא עִמַּנָא, וְנֵיחוֹת--אֲרֵי לָא נִכּוֹל לְמִחְזֵי אַפֵּי גֻּבְרָא, וְאַחוּנָא זְעֵירָא לָיְתוֹהִי עִמַּנָא. 
מד,כז וַיֹּ֛אמֶר עַבְדְּךָ֥ אָבִ֖י אֵלֵ֑ינוּ אַתֶּ֣ם יְדַעְתֶּ֔ם כִּ֥י שְׁנַ֖יִם יָֽלְדָה־לִּ֥י אִשְׁתִּֽי׃ וַאֲמַר עַבְדָּךְ אַבָּא, לַנָא:  אַתּוּן יְדַעְתּוּן, אֲרֵי תְּרֵין יְלֵידַת לִי אִתְּתִי. 
מד,כח וַיֵּצֵ֤א הָֽאֶחָד֙ מֵֽאִתִּ֔י וָֽאֹמַ֕ר אַ֖ךְ טָרֹ֣ף טֹרָ֑ף וְלֹ֥א רְאִיתִ֖יו עַד־הֵֽנָּה׃ וּנְפַק חַד, מִלְּוָתִי, וַאֲמַרִית, בְּרַם מִקְטָל קְטִיל; וְלָא חֲזֵיתֵיהּ, עַד כְּעַן. 
מד,כט וּלְקַחְתֶּ֧ם גַּם־אֶת־זֶ֛ה מֵעִ֥ם פָּנַ֖י וְקָרָ֣הוּ אָס֑וֹן וְהֽוֹרַדְתֶּ֧ם אֶת־שֵֽׂיבָתִ֛י בְּרָעָ֖ה שְׁאֹֽלָה׃ וְתִדְבְּרוּן אַף יָת דֵּין מִן קֳדָמַי, וִיעָרְעִנֵּיהּ מוֹתָא--וְתַחֲתוּן יָת סֵיבְתִי, בְּבִשְׁתָּא לִשְׁאוֹל. 
מד,ל וְעַתָּ֗ה כְּבֹאִי֙ אֶל־עַבְדְּךָ֣ אָבִ֔י וְהַנַּ֖עַר אֵינֶנּ֣וּ אִתָּ֑נוּ וְנַפְשׁ֖וֹ קְשׁוּרָ֥ה בְנַפְשֽׁוֹ׃ וּכְעַן, כְּמֵיתַא לְוָת עַבְדָּךְ אַבָּא, וְעוּלֵימָא לָיְתוֹהִי, עִמַּנָא; וְנַפְשֵׁיהּ, חַבִּיבָא לֵיהּ כְּנַפְשֵׁיהּ. 
מד,לא וְהָיָ֗ה כִּרְאוֹת֛וֹ כִּי־אֵ֥ין הַנַּ֖עַר וָמֵ֑ת וְהוֹרִ֨ידוּ עֲבָדֶ֜יךָ אֶת־שֵׂיבַ֨ת עַבְדְּךָ֥ אָבִ֛ינוּ בְּיָג֖וֹן שְׁאֹֽלָה׃ וִיהֵי, כַּד יִחְזֵי אֲרֵי לֵית עוּלֵימָא--וִימוּת; וְיַחֲתוּן עַבְדָּךְ יָת סֵיבַת עַבְדָּךְ אֲבוּנָא, בְּדָווֹנָא--לִשְׁאוֹל. 
מד,לב כִּ֤י עַבְדְּךָ֙ עָרַ֣ב אֶת־הַנַּ֔עַר מֵעִ֥ם אָבִ֖י לֵאמֹ֑ר אִם־לֹ֤א אֲבִיאֶ֨נּוּ֙ אֵלֶ֔יךָ וְחָטָ֥אתִי לְאָבִ֖י כָּל־הַיָּמִֽים׃ אֲרֵי עַבְדָּךְ מְעָרַב בְּעוּלֵימָא, מִן אַבָּא לְמֵימַר:  אִם לָא אַיְתֵינֵיהּ לָךְ, וְאֵיהֵי חָטֵי לְאַבָּא כָּל יוֹמַיָּא. 
מד,לג וְעַתָּ֗ה יֵֽשֶׁב־נָ֤א עַבְדְּךָ֙ תַּ֣חַת הַנַּ֔עַר עֶ֖בֶד לַֽאדֹנִ֑י וְהַנַּ֖עַר יַ֥עַל עִם־אֶחָֽיו׃ וּכְעַן, יִתֵּיב כְּעַן עַבְדָּךְ תְּחוֹת עוּלֵימָא--עַבְדָּא, לְרִבּוֹנִי; וְעוּלֵימָא, יִסַּק עִם אֲחוֹהִי. 
מד,לד כִּי־אֵיךְ֙ אֶֽעֱלֶ֣ה אֶל־אָבִ֔י וְהַנַּ֖עַר אֵינֶנּ֣וּ אִתִּ֑י פֶּ֚ן אֶרְאֶ֣ה בָרָ֔ע אֲשֶׁ֥ר יִמְצָ֖א אֶת־אָבִֽי׃ אֲרֵי אֵיכְדֵין אֶסַּק לְוָת אַבָּא, וְעוּלֵימָא לָיְתוֹהִי עִמִּי:  דִּלְמָא אֶחְזֵי בְּבִשְׁתָּא, דְּתַשְׁכַּח יָת אַבָּא. 
מה,א וְלֹֽא־יָכֹ֨ל יוֹסֵ֜ף לְהִתְאַפֵּ֗ק לְכֹ֤ל הַנִּצָּבִים֙ עָלָ֔יו וַיִּקְרָ֕א הוֹצִ֥יאוּ כָל־אִ֖ישׁ מֵֽעָלָ֑י וְלֹא־עָ֤מַד אִישׁ֙ אִתּ֔וֹ בְּהִתְוַדַּ֥ע יוֹסֵ֖ף אֶל־אֶחָֽיו׃ וְלָא יְכֵיל יוֹסֵף לְאִתְחַסָּנָא, לְכֹל דְּקָיְמִין עִלָּווֹהִי, וּקְרָא, אַפִּיקוּ כָּל אֲנָשׁ מֵעִלָּוָי; וְלָא קָם אֲנָשׁ עִמֵּיהּ, כַּד אִתְיְדַע יוֹסֵף לַאֲחוֹהִי. 
מה,ב וַיִּתֵּ֥ן אֶת־קֹל֖וֹ בִּבְכִ֑י וַיִּשְׁמְע֣וּ מִצְרַ֔יִם וַיִּשְׁמַ֖ע בֵּ֥ית פַּרְעֹֽה׃ וִיהַב יָת קָלֵיהּ, בִּבְכִיתָא; וּשְׁמַעוּ מִצְרָאֵי, וּשְׁמַע אֲנָשׁ בֵּית פַּרְעֹה. 
מה,ג וַיֹּ֨אמֶר יוֹסֵ֤ף אֶל־אֶחָיו֙ אֲנִ֣י יוֹסֵ֔ף הַע֥וֹד אָבִ֖י חָ֑י וְלֹֽא־יָכְל֤וּ אֶחָיו֙ לַֽעֲנ֣וֹת אֹת֔וֹ כִּ֥י נִבְהֲל֖וּ מִפָּנָֽיו׃ וַאֲמַר יוֹסֵף לַאֲחוֹהִי אֲנָא יוֹסֵף, הַעַד כְּעַן אַבָּא קַיָּם; וְלָא יְכִילוּ אֲחוֹהִי לְאָתָבָא יָתֵיהּ פִּתְגָּם, אֲרֵי אִתְבְּהִילוּ מִן קֳדָמוֹהִי. 
מה,ד וַיֹּ֨אמֶר יוֹסֵ֧ף אֶל־אֶחָ֛יו גְּשׁוּ־נָ֥א אֵלַ֖י וַיִּגָּ֑שׁוּ וַיֹּ֗אמֶר אֲנִי֙ יוֹסֵ֣ף אֲחִיכֶ֔ם אֲשֶׁר־מְכַרְתֶּ֥ם אֹתִ֖י מִצְרָֽיְמָה׃ וַאֲמַר יוֹסֵף לַאֲחוֹהִי קְרוּבוּ כְּעַן לְוָתִי, וּקְרִיבוּ; וַאֲמַר, אֲנָא יוֹסֵף אֲחוּכוֹן, דְּזַבֵּינְתּוּן יָתִי, לְמִצְרָיִם. 
מה,ה וְעַתָּ֣ה ׀ אַל־תֵּעָ֣צְב֗וּ וְאַל־יִ֨חַר֙ בְּעֵ֣ינֵיכֶ֔ם כִּֽי־מְכַרְתֶּ֥ם אֹתִ֖י הֵ֑נָּה כִּ֣י לְמִֽחְיָ֔ה שְׁלָחַ֥נִי אֱלֹהִ֖ים לִפְנֵיכֶֽם׃ וּכְעַן לָא תִּתְנָסְסוּן, וְלָא יִתְקַף בְּעֵינֵיכוֹן, אֲרֵי זַבֵּינְתּוּן יָתִי, הָלְכָא:  אֲרֵי לְקַיָּמָא, שַׁלְחַנִי יְיָ קֳדָמֵיכוֹן. 
מה,ו כִּי־זֶ֛ה שְׁנָתַ֥יִם הָֽרָעָ֖ב בְּקֶ֣רֶב הָאָ֑רֶץ וְעוֹד֙ חָמֵ֣שׁ שָׁנִ֔ים אֲשֶׁ֥ר אֵין־חָרִ֖ישׁ וְקָצִֽיר׃ אֲרֵי דְּנָן תַּרְתֵּין שְׁנִין כַּפְנָא, בְּגוֹ אַרְעָא; וְעוֹד חֲמֵישׁ שְׁנִין, דְּלֵית זְרוּעָא וּחְצָדָא. 
מה,ז וַיִּשְׁלָחֵ֤נִי אֱלֹהִים֙ לִפְנֵיכֶ֔ם לָשׂ֥וּם לָכֶ֛ם שְׁאֵרִ֖ית בָּאָ֑רֶץ וּלְהַֽחֲי֣וֹת לָכֶ֔ם לִפְלֵיטָ֖ה גְּדֹלָֽה׃ וְשַׁלְחַנִי יְיָ קֳדָמֵיכוֹן, לְשַׁוָּאָה לְכוֹן שְׁאָרָא בְּאַרְעָא, וּלְקַיָּמָא לְכוֹן, לְשֵׁיזָבָא רַבָּא. 
מה,ח וְעַתָּ֗ה לֹֽא־אַתֶּ֞ם שְׁלַחְתֶּ֤ם אֹתִי֙ הֵ֔נָּה כִּ֖י הָֽאֱלֹהִ֑ים וַיְשִׂימֵ֨נִֽי לְאָ֜ב לְפַרְעֹ֗ה וּלְאָדוֹן֙ לְכָל־בֵּית֔וֹ וּמֹשֵׁ֖ל בְּכָל־אֶ֥רֶץ מִצְרָֽיִם׃ וּכְעַן, לָא אַתּוּן שְׁלַחְתּוּן יָתִי הָלְכָא, אֱלָהֵין, מִן קֳדָם יְיָ; וְשַׁוְיַנִי אַבָּא לְפַרְעֹה, וְרִבּוֹן לְכָל בֵּיתֵיהּ, וְשַׁלִּיט, בְּכָל אַרְעָא דְּמִצְרָיִם. 
מה,ט מַֽהֲרוּ֮ וַֽעֲל֣וּ אֶל־אָבִי֒ וַֽאֲמַרְתֶּ֣ם אֵלָ֗יו כֹּ֤ה אָמַר֙ בִּנְךָ֣ יוֹסֵ֔ף שָׂמַ֧נִי אֱלֹהִ֛ים לְאָד֖וֹן לְכָל־מִצְרָ֑יִם רְדָ֥ה אֵלַ֖י אַֽל־תַּעֲמֹֽד׃ אוֹחוֹ, וְסַקוּ לְוָת אַבָּא, וְתֵימְרוּן לֵיהּ כִּדְנָן אֲמַר בְּרָךְ יוֹסֵף, שַׁוְיַנִי יְיָ לְרִבּוֹן לְכָל מִצְרָאֵי; חוֹת לְוָתִי, לָא תִּתְעַכַּב. 
מה,י וְיָֽשַׁבְתָּ֣ בְאֶֽרֶץ־גֹּ֗שֶׁן וְהָיִ֤יתָ קָרוֹב֙ אֵלַ֔י אַתָּ֕ה וּבָנֶ֖יךָ וּבְנֵ֣י בָנֶ֑יךָ וְצֹֽאנְךָ֥ וּבְקָֽרְךָ֖ וְכָל־אֲשֶׁר־לָֽךְ׃ וְתִתֵּיב בְּאַרְעָא דְּגֹשֶׁן, וּתְהֵי קָרִיב לִי--אַתְּ, וּבְנָךְ וּבְנֵי בְּנָךְ; וְעָנָךְ וְתוֹרָךְ, וְכָל דְּלָךְ. 
מה,יא וְכִלְכַּלְתִּ֤י אֹֽתְךָ֙ שָׁ֔ם כִּי־ע֛וֹד חָמֵ֥שׁ שָׁנִ֖ים רָעָ֑ב פֶּן־תִּוָּרֵ֛שׁ אַתָּ֥ה וּבֵֽיתְךָ֖ וְכָל־אֲשֶׁר־לָֽךְ׃ וַאֲזוּן יָתָךְ תַּמָּן, אֲרֵי עוֹד חֲמֵישׁ שְׁנִין כַּפְנָא:  דִּלְמָא תִּשְׁתֵּיצֵי אַתְּ וַאֲנָשׁ בֵּיתָךְ, וְכָל דְּלָךְ. 
מה,יב וְהִנֵּ֤ה עֵֽינֵיכֶם֙ רֹא֔וֹת וְעֵינֵ֖י אָחִ֣י בִנְיָמִ֑ין כִּי־פִ֖י הַֽמְדַבֵּ֥ר אֲלֵיכֶֽם׃ וְהָא עֵינֵיכוֹן חָזְיָן, וְעֵינֵי אֲחִי בִּנְיָמִין:  אֲרֵי כְּלִישָׁנְכוֹן, אֲנָא מְמַלֵּיל עִמְּכוֹן. 
מה,יג וְהִגַּדְתֶּ֣ם לְאָבִ֗י אֶת־כָּל־כְּבוֹדִי֙ בְּמִצְרַ֔יִם וְאֵ֖ת כָּל־אֲשֶׁ֣ר רְאִיתֶ֑ם וּמִֽהַרְתֶּ֛ם וְהֽוֹרַדְתֶּ֥ם אֶת־אָבִ֖י הֵֽנָּה׃ וּתְחַוּוֹן לְאַבָּא, יָת כָּל יְקָרִי דִּבְמִצְרַיִם, וְיָת, כָּל דַּחֲזֵיתוֹן; וְתוֹחוֹן וְתַחֲתוּן יָת אַבָּא, הָלְכָא. 
מה,יד וַיִּפֹּ֛ל עַל־צַוְּארֵ֥י בִנְיָֽמִן־אָחִ֖יו וַיֵּ֑בְךְּ וּבִ֨נְיָמִ֔ן בָּכָ֖ה עַל־צַוָּארָֽיו׃ וּנְפַל עַל צַוְרָא דְּבִנְיָמִין אֲחוּהִי, וּבְכָא; וּבִנְיָמִין--בְּכָא, עַל צַוְרֵיהּ. 
מה,טו וַיְנַשֵּׁ֥ק לְכָל־אֶחָ֖יו וַיֵּ֣בְךְּ עֲלֵהֶ֑ם וְאַ֣חֲרֵי כֵ֔ן דִּבְּר֥וּ אֶחָ֖יו אִתּֽוֹ׃ וְנַשֵּׁיק לְכָל אֲחוֹהִי, וּבְכָא עֲלֵיהוֹן; וּבָתַר כֵּין, מַלִּילוּ אֲחוֹהִי עִמֵּיהּ. 
מה,טז וְהַקֹּ֣ל נִשְׁמַ֗ע בֵּ֤ית פַּרְעֹה֙ לֵאמֹ֔ר בָּ֖אוּ אֲחֵ֣י יוֹסֵ֑ף וַיִּיטַב֙ בְּעֵינֵ֣י פַרְעֹ֔ה וּבְעֵינֵ֖י עֲבָדָֽיו׃ וְקָלָא אִשְׁתְּמַע, בֵּית פַּרְעֹה לְמֵימַר, אֲתוֹ, אֲחֵי יוֹסֵף; וּשְׁפַר בְּעֵינֵי פַּרְעֹה, וּבְעֵינֵי עַבְדּוֹהִי. 
מה,יז וַיֹּ֤אמֶר פַּרְעֹה֙ אֶל־יוֹסֵ֔ף אֱמֹ֥ר אֶל־אַחֶ֖יךָ זֹ֣את עֲשׂ֑וּ טַֽעֲנוּ֙ אֶת־בְּעִ֣ירְכֶ֔ם וּלְכוּ־בֹ֖אוּ אַ֥רְצָה כְּנָֽעַן׃ וַאֲמַר פַּרְעֹה לְיוֹסֵף, אֵימַר לַאֲחָךְ דָּא עֲבִידוּ:  טְעוּנוּ, יָת בְּעִירְכוֹן, וְאִיזִילוּ אוֹבִילוּ, לְאַרְעָא דִּכְנָעַן. 
מה,יח וּקְח֧וּ אֶת־אֲבִיכֶ֛ם וְאֶת־בָּֽתֵּיכֶ֖ם וּבֹ֣אוּ אֵלָ֑י וְאֶתְּנָ֣ה לָכֶ֗ם אֶת־טוּב֙ אֶ֣רֶץ מִצְרַ֔יִם וְאִכְל֖וּ אֶת־חֵ֥לֶב הָאָֽרֶץ׃ וּדְבַרוּ יָת אֲבוּכוֹן וְיָת אֲנָשׁ בָּתֵּיכוֹן, וְאֵיתוֹ לְוָתִי; וְאֶתֵּין לְכוֹן, יָת טוּב אַרְעָא דְּמִצְרַיִם, וְתֵיכְלוּן, יָת טוּבָא דְּאַרְעָא. 
מה,יט וְאַתָּ֥ה צֻוֵּ֖יתָה זֹ֣את עֲשׂ֑וּ קְחוּ־לָכֶם֩ מֵאֶ֨רֶץ מִצְרַ֜יִם עֲגָל֗וֹת לְטַפְּכֶם֙ וְלִנְשֵׁיכֶ֔ם וּנְשָׂאתֶ֥ם אֶת־אֲבִיכֶ֖ם וּבָאתֶֽם׃ וְאַתְּ מְפֻקַּד, דָּא עֲבִידוּ:  סַבוּ לְכוֹן מֵאַרְעָא דְּמִצְרַיִם עֶגְלָן, לְטַפְלְכוֹן וְלִנְשֵׁיכוֹן, וְתִטְּלוּן יָת אֲבוּכוֹן, וְתֵיתוֹן. 
מה,כ וְעֵ֣ינְכֶ֔ם אַל־תָּחֹ֖ס עַל־כְּלֵיכֶ֑ם כִּי־ט֛וּב כָּל־אֶ֥רֶץ מִצְרַ֖יִם לָכֶ֥ם הֽוּא׃ וְעֵינְכוֹן, לָא תְּחוּס עַל מָנֵיכוֹן:  אֲרֵי טוּב כָּל אַרְעָא דְּמִצְרַיִם, דִּילְכוֹן הוּא. 
מה,כא וַיַּֽעֲשׂוּ־כֵן֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וַיִּתֵּ֨ן לָהֶ֥ם יוֹסֵ֛ף עֲגָל֖וֹת עַל־פִּ֣י פַרְעֹ֑ה וַיִּתֵּ֥ן לָהֶ֛ם צֵדָ֖ה לַדָּֽרֶךְ׃ וַעֲבַדוּ כֵּין בְּנֵי יִשְׂרָאֵל, וִיהַב לְהוֹן יוֹסֵף עֶגְלָן עַל מֵימְרָא דְּפַרְעֹה; וִיהַב לְהוֹן זְוָדִין, לְאוֹרְחָא. 
מה,כב לְכֻלָּ֥ם נָתַ֛ן לָאִ֖ישׁ חֲלִפ֣וֹת שְׂמָלֹ֑ת וּלְבִנְיָמִ֤ן נָתַן֙ שְׁלֹ֣שׁ מֵא֣וֹת כֶּ֔סֶף וְחָמֵ֖שׁ חֲלִפֹ֥ת שְׂמָלֹֽת׃ לְכֻלְּהוֹן יְהַב לִגְבַר, אִצְטְלָוָן דִּלְבוּשָׁא; וּלְבִנְיָמִין יְהַב תְּלָת מְאָה סִלְעִין דִּכְסַף, וְחַמְשָׁא אִצְטְלָוָן דִּלְבוּשָׁא. 
מה,כג וּלְאָבִ֞יו שָׁלַ֤ח כְּזֹאת֙ עֲשָׂרָ֣ה חֲמֹרִ֔ים נֹֽשְׂאִ֖ים מִטּ֣וּב מִצְרָ֑יִם וְעֶ֣שֶׂר אֲתֹנֹ֡ת נֹֽ֠שְׂאֹת בָּ֣ר וָלֶ֧חֶם וּמָז֛וֹן לְאָבִ֖יו לַדָּֽרֶךְ׃ וְלַאֲבוּהִי שְׁלַח כְּדֵין, עַסְרָא חֲמָרִין, טְעִינִין, מִטּוּב מִצְרָיִם; וַעֲסַר אֲתָנָן טְעִינָן עֲבוּר וּלְחֵים וּזְוָדִין, לַאֲבוּהִי--לְאוֹרְחָא. 
מה,כד וַיְשַׁלַּ֥ח אֶת־אֶחָ֖יו וַיֵּלֵ֑כוּ וַיֹּ֣אמֶר אֲלֵהֶ֔ם אַֽל־תִּרְגְּז֖וּ בַּדָּֽרֶךְ׃ וְשַׁלַּח יָת אֲחוֹהִי, וַאֲזַלוּ; וַאֲמַר לְהוֹן, לָא תִּתְנְצוֹן בְּאוֹרְחָא. 
מה,כה וַֽיַּעֲל֖וּ מִמִּצְרָ֑יִם וַיָּבֹ֨אוּ֙ אֶ֣רֶץ כְּנַ֔עַן אֶֽל־יַעֲקֹ֖ב אֲבִיהֶֽם׃ וּסְלִיקוּ, מִמִּצְרָיִם; וַאֲתוֹ לְאַרְעָא דִּכְנַעַן, לְוָת יַעֲקוֹב אֲבוּהוֹן. 
מה,כו וַיַּגִּ֨דוּ ל֜וֹ לֵאמֹ֗ר ע֚וֹד יוֹסֵ֣ף חַ֔י וְכִֽי־ה֥וּא מֹשֵׁ֖ל בְּכָל־אֶ֣רֶץ מִצְרָ֑יִם וַיָּ֣פָג לִבּ֔וֹ כִּ֥י לֹֽא־הֶאֱמִ֖ין לָהֶֽם׃ וְחַוִּיאוּ לֵיהּ לְמֵימַר, עַד כְּעַן יוֹסֵף קַיָּם, וַאֲרֵי הוּא שַׁלִּיט, בְּכָל אַרְעָא דְּמִצְרָיִם; וַהֲוַאָה מִלַּיָּא פָּיְגָן עַל לִבֵּיהּ, אֲרֵי לָא הֵימֵין לְהוֹן. 
מה,כז וַיְדַבְּר֣וּ אֵלָ֗יו אֵ֣ת כָּל־דִּבְרֵ֤י יוֹסֵף֙ אֲשֶׁ֣ר דִּבֶּ֣ר אֲלֵהֶ֔ם וַיַּרְא֙ אֶת־הָ֣עֲגָל֔וֹת אֲשֶׁר־שָׁלַ֥ח יוֹסֵ֖ף לָשֵׂ֣את אֹת֑וֹ וַתְּחִ֕י ר֖וּחַ יַֽעֲקֹ֥ב אֲבִיהֶֽם׃ וּמַלִּילוּ עִמֵּיהּ, יָת כָּל פִּתְגָמֵי יוֹסֵף דְּמַלֵּיל עִמְּהוֹן, וַחֲזָא יָת עֶגְלָתָא, דִּשְׁלַח יוֹסֵף לְמִטַּל יָתֵיהּ; וּשְׁרָת, רוּחַ קֻדְשָׁא עַל יַעֲקוֹב אֲבוּהוֹן. 
מה,כח וַיֹּ֨אמֶר֙ יִשְׂרָאֵ֔ל רַ֛ב עוֹד־יוֹסֵ֥ף בְּנִ֖י חָ֑י אֵֽלְכָ֥ה וְאֶרְאֶ֖נּוּ בְּטֶ֥רֶם אָמֽוּת׃ וַאֲמַר, יִשְׂרָאֵל, סַגִּי לִי חֶדְוָא עַד כְּעַן יוֹסֵף בְּרִי, קַיָּם; אֵיזֵיל וְאֶחְזֵינֵיהּ, עַד לָא אֲמוּת. 
מו,א וַיִּסַּ֤ע יִשְׂרָאֵל֙ וְכָל־אֲשֶׁר־ל֔וֹ וַיָּבֹ֖א בְּאֵ֣רָה שָּׁ֑בַע וַיִּזְבַּ֣ח זְבָחִ֔ים לֵֽאלֹהֵ֖י אָבִ֥יו יִצְחָֽק׃ וּנְטַל יִשְׂרָאֵל וְכָל דְּלֵיהּ, וַאֲתָא לִבְאֵר שָׁבַע; וְדַבַּח דִּבְחִין, לֵאלָהָא דַּאֲבוּהִי יִצְחָק. 
מו,ב וַיֹּ֨אמֶר אֱלֹהִ֤ים ׀ לְיִשְׂרָאֵל֙ בְּמַרְאֹ֣ת הַלַּ֔יְלָה וַיֹּ֖אמֶר יַֽעֲקֹ֣ב ׀ יַֽעֲקֹ֑ב וַיֹּ֖אמֶר הִנֵּֽנִי׃ וַאֲמַר יְיָ לְיִשְׂרָאֵל בְּחֶזְוָא דְּלֵילְיָא, וַאֲמַר יַעֲקוֹב יַעֲקוֹב; וַאֲמַר, הָאֲנָא. 
מו,ג וַיֹּ֕אמֶר אָֽנֹכִ֥י הָאֵ֖ל אֱלֹהֵ֣י אָבִ֑יךָ אַל־תִּירָא֙ מֵֽרְדָ֣ה מִצְרַ֔יְמָה כִּֽי־לְג֥וֹי גָּד֖וֹל אֲשִֽׂימְךָ֥ שָֽׁם׃ וַאֲמַר, אֲנָא אֵל אֱלָהָא דַּאֲבוּךְ; לָא תִּדְחַל מִלְּמֵיחַת לְמִצְרַיִם, אֲרֵי לְעַם סַגִּי אֲשַׁוֵּינָךְ תַּמָּן. 
מו,ד אָֽנֹכִ֗י אֵרֵ֤ד עִמְּךָ֙ מִצְרַ֔יְמָה וְאָֽנֹכִ֖י אַֽעַלְךָ֣ גַם־עָלֹ֑ה וְיוֹסֵ֕ף יָשִׁ֥ית יָד֖וֹ עַל־עֵינֶֽיךָ׃ אֲנָא, אֵיחוֹת עִמָּךְ לְמִצְרַיִם, וַאֲנָא, אַסְּקִנָּךְ אַף אַסָּקָא; וְיוֹסֵף, יְשַׁוֵּי יְדוֹהִי עַל עֵינָךְ. 
מו,ה וַיָּ֥קָם יַֽעֲקֹ֖ב מִבְּאֵ֣ר שָׁ֑בַע וַיִּשְׂא֨וּ בְנֵֽי־יִשְׂרָאֵ֜ל אֶת־יַֽעֲקֹ֣ב אֲבִיהֶ֗ם וְאֶת־טַפָּם֙ וְאֶת־נְשֵׁיהֶ֔ם בָּֽעֲגָל֕וֹת אֲשֶׁר־שָׁלַ֥ח פַּרְעֹ֖ה לָשֵׂ֥את אֹתֽוֹ׃ וְקָם יַעֲקוֹב, מִבְּאֵר שָׁבַע; וּנְטַלוּ בְּנֵי יִשְׂרָאֵל יָת יַעֲקוֹב אֲבוּהוֹן, וְיָת טַפְלְהוֹן וְיָת נְשֵׁיהוֹן, בְּעֶגְלָתָא, דִּשְׁלַח פַּרְעֹה לְמִטַּל יָתֵיהּ. 
מו,ו וַיִּקְח֣וּ אֶת־מִקְנֵיהֶ֗ם וְאֶת־רְכוּשָׁם֙ אֲשֶׁ֤ר רָֽכְשׁוּ֙ בְּאֶ֣רֶץ כְּנַ֔עַן וַיָּבֹ֖אוּ מִצְרָ֑יְמָה יַֽעֲקֹ֖ב וְכָל־זַרְע֥וֹ אִתּֽוֹ׃ וּדְבַרוּ יָת גֵּיתֵיהוֹן, וְיָת קִנְיָנְהוֹן דִּקְנוֹ בְּאַרְעָא דִּכְנַעַן, וַאֲתוֹ, לְמִצְרָיִם:  יַעֲקוֹב, וְכָל בְּנוֹהִי עִמֵּיהּ. 
מו,ז בָּנָ֞יו וּבְנֵ֤י בָנָיו֙ אִתּ֔וֹ בְּנֹתָ֛יו וּבְנ֥וֹת בָּנָ֖יו וְכָל־זַרְע֑וֹ הֵבִ֥יא אִתּ֖וֹ מִצְרָֽיְמָה׃ {ס} בְּנוֹהִי וּבְנֵי בְּנוֹהִי, עִמֵּיהּ, בְּנָתֵיהּ וּבְנָת בְּנוֹהִי, וְכָל בְּנוֹהִי--אַיְתִי עִמֵּיהּ, לְמִצְרָיִם. {ס}
מו,ח וְאֵ֨לֶּה שְׁמ֧וֹת בְּנֵֽי־יִשְׂרָאֵ֛ל הַבָּאִ֥ים מִצְרַ֖יְמָה יַֽעֲקֹ֣ב וּבָנָ֑יו בְּכֹ֥ר יַֽעֲקֹ֖ב רְאוּבֵֽן׃ וְאִלֵּין שְׁמָהָת בְּנֵי יִשְׂרָאֵל דְּעָאלוּ לְמִצְרַיִם, יַעֲקוֹב וּבְנוֹהִי:  בֻּכְרֵיהּ דְּיַעֲקוֹב, רְאוּבֵן. 
מו,ט וּבְנֵ֖י רְאוּבֵ֑ן חֲנ֥וֹךְ וּפַלּ֖וּא וְחֶצְרֹ֥ן וְכַרְמִֽי׃ וּבְנֵי, רְאוּבֵן--חֲנוֹךְ וּפַלּוּא, וְחֶצְרוֹן וְכַרְמִי. 
מו,י וּבְנֵ֣י שִׁמְע֗וֹן יְמוּאֵ֧ל וְיָמִ֛ין וְאֹ֖הַד וְיָכִ֣ין וְצֹ֑חַר וְשָׁא֖וּל בֶּן־הַֽכְּנַעֲנִֽית׃ וּבְנֵי שִׁמְעוֹן, יְמוּאֵל וְיָמִין וְאֹהַד--וְיָכִין וְצֹחַר; וְשָׁאוּל, בַּר כְּנַעֲנֵיתָא. 
מו,יא וּבְנֵ֖י לֵוִ֑י גֵּֽרְשׁ֕וֹן קְהָ֖ת וּמְרָרִֽי׃ וּבְנֵי, לֵוִי--גֵּרְשׁוֹן, קְהָת וּמְרָרִי. 
מו,יב וּבְנֵ֣י יְהוּדָ֗ה עֵ֧ר וְאוֹנָ֛ן וְשֵׁלָ֖ה וָפֶ֣רֶץ וָזָ֑רַח וַיָּ֨מָת עֵ֤ר וְאוֹנָן֙ בְּאֶ֣רֶץ כְּנַ֔עַן וַיִּֽהְי֥וּ בְנֵי־פֶ֖רֶץ חֶצְרֹ֥ן וְחָמֽוּל׃ וּבְנֵי יְהוּדָה, עֵר וְאוֹנָן וְשֵׁלָה--וָפֶרֶץ וָזָרַח; וּמִית עֵר וְאוֹנָן בְּאַרְעָא דִּכְנַעַן, וַהֲווֹ בְּנֵי פֶּרֶץ חֶצְרוֹן וְחָמוּל. 
מו,יג וּבְנֵ֖י יִשָּׂשכָ֑ר תּוֹלָ֥ע וּפֻוָ֖ה וְי֥וֹב וְשִׁמְרֹֽן׃ וּבְנֵי, יִשָּׂשכָר--תּוֹלָע וּפוּוָה, וְיוֹב וְשִׁמְרוֹן. 
מו,יד וּבְנֵ֖י זְבֻל֑וּן סֶ֥רֶד וְאֵל֖וֹן וְיַחְלְאֵֽל׃ וּבְנֵי, זְבוּלוּן--סֶרֶד וְאֵילוֹן, וְיַחְלְאֵל. 
מו,טו אֵ֣לֶּה ׀ בְּנֵ֣י לֵאָ֗ה אֲשֶׁ֨ר יָֽלְדָ֤ה לְיַֽעֲקֹב֙ בְּפַדַּ֣ן אֲרָ֔ם וְאֵ֖ת דִּינָ֣ה בִתּ֑וֹ כָּל־נֶ֧פֶשׁ בָּנָ֛יו וּבְנוֹתָ֖יו שְׁלֹשִׁ֥ים וְשָׁלֹֽשׁ׃ אִלֵּין בְּנֵי לֵאָה, דִּילֵידַת לְיַעֲקוֹב בְּפַדַּן אֲרָם, וְיָת, דִּינָה בְּרַתֵּיהּ:  כָּל נְפַשׁ בְּנוֹהִי וּבְנָתֵיהּ, תְּלָתִין וּתְלָת. 
מו,טז וּבְנֵ֣י גָ֔ד צִפְי֥וֹן וְחַגִּ֖י שׁוּנִ֣י וְאֶצְבֹּ֑ן עֵרִ֥י וַֽאֲרוֹדִ֖י וְאַרְאֵלִֽי׃ וּבְנֵי גָּד, צִפְיוֹן וְחַגִּי שׁוּנִי וְאֶצְבּוֹן, עֵרִי וַאֲרוֹדִי, וְאַרְאֵלִי. 
מו,יז וּבְנֵ֣י אָשֵׁ֗ר יִמְנָ֧ה וְיִשְׁוָ֛ה וְיִשְׁוִ֥י וּבְרִיעָ֖ה וְשֶׂ֣רַח אֲחֹתָ֑ם וּבְנֵ֣י בְרִיעָ֔ה חֶ֖בֶר וּמַלְכִּיאֵֽל׃ וּבְנֵי אָשֵׁר, יִמְנָה וְיִשְׁוָה וְיִשְׁוִי וּבְרִיעָה--וְשֶׂרַח אֲחָתְהוֹן; וּבְנֵי בְּרִיעָה, חֶבֶר וּמַלְכִּיאֵל. 
מו,יח אֵ֚לֶּה בְּנֵ֣י זִלְפָּ֔ה אֲשֶׁר־נָתַ֥ן לָבָ֖ן לְלֵאָ֣ה בִתּ֑וֹ וַתֵּ֤לֶד אֶת־אֵ֨לֶּה֙ לְיַֽעֲקֹ֔ב שֵׁ֥שׁ עֶשְׂרֵ֖ה נָֽפֶשׁ׃ אִלֵּין בְּנֵי זִלְפָּה, דִּיהַב לָבָן לְלֵאָה בְּרַתֵּיהּ; וִילֵידַת יָת אִלֵּין לְיַעֲקוֹב, שֵׁית עַסְרֵי נַפְשָׁן. 
מו,יט בְּנֵ֤י רָחֵל֙ אֵ֣שֶׁת יַֽעֲקֹ֔ב יוֹסֵ֖ף וּבִנְיָמִֽן׃ בְּנֵי רָחֵל אִתַּת יַעֲקוֹב, יוֹסֵף וּבִנְיָמִין. 
מו,כ וַיִּוָּלֵ֣ד לְיוֹסֵף֮ בְּאֶ֣רֶץ מִצְרַיִם֒ אֲשֶׁ֤ר יָֽלְדָה־לּוֹ֙ אָֽסְנַ֔ת בַּת־פּ֥וֹטִי פֶ֖רַע כֹּהֵ֣ן אֹ֑ן אֶת־מְנַשֶּׁ֖ה וְאֶת־אֶפְרָֽיִם׃ וְאִתְיְלֵיד לְיוֹסֵף, בְּאַרְעָא דְּמִצְרַיִם, דִּילֵידַת לֵיהּ אָסְנַת, בַּת פּוֹטִי פֶרַע רַבָּא דְּאוֹן--יָת מְנַשֶּׁה, וְיָת אֶפְרָיִם. 
מו,כא וּבְנֵ֣י בִנְיָמִ֗ן בֶּ֤לַע וָבֶ֨כֶר֙ וְאַשְׁבֵּ֔ל גֵּרָ֥א וְנַֽעֲמָ֖ן אֵחִ֣י וָרֹ֑אשׁ מֻפִּ֥ים וְחֻפִּ֖ים וָאָֽרְדְּ׃ וּבְנֵי בִּנְיָמִין, בֶּלַע וָבֶכֶר וְאַשְׁבֵּל, גֵּרָא וְנַעֲמָן, אֵחִי וָרֹאשׁ; מֻפִּים וְחֻפִּים, וָאָרְדְּ. 
מו,כב אֵ֚לֶּה בְּנֵ֣י רָחֵ֔ל אֲשֶׁ֥ר יֻלַּ֖ד לְיַֽעֲקֹ֑ב כָּל־נֶ֖פֶשׁ אַרְבָּעָ֥ה עָשָֽׂר׃ אִלֵּין בְּנֵי רָחֵל, דְּאִתְיְלִידוּ לְיַעֲקוֹב--כָּל נַפְשָׁתָא, אַרְבַּע עַסְרֵי. 
מו,כג וּבְנֵי־דָ֖ן חֻשִֽׁים׃ וּבְנֵי דָּן, חוּשִׁים. 
מו,כד וּבְנֵ֖י נַפְתָּלִ֑י יַחְצְאֵ֥ל וְגוּנִ֖י וְיֵ֥צֶר וְשִׁלֵּֽם׃ וּבְנֵי, נַפְתָּלִי--יַחְצְאֵל וְגוּנִי, וְיֵצֶר וְשִׁלֵּם. 
מו,כה אֵ֚לֶּה בְּנֵ֣י בִלְהָ֔ה אֲשֶׁר־נָתַ֥ן לָבָ֖ן לְרָחֵ֣ל בִּתּ֑וֹ וַתֵּ֧לֶד אֶת־אֵ֛לֶּה לְיַֽעֲקֹ֖ב כָּל־נֶ֥פֶשׁ שִׁבְעָֽה׃ אִלֵּין בְּנֵי בִּלְהָה, דִּיהַב לָבָן לְרָחֵל בְּרַתֵּיהּ; וִילֵידַת יָת אִלֵּין לְיַעֲקוֹב, כָּל נַפְשָׁתָא שְׁבַע. 
מו,כו כָּל־הַ֠נֶּפֶשׁ הַבָּאָ֨ה לְיַֽעֲקֹ֤ב מִצְרַ֨יְמָה֙ יֹֽצְאֵ֣י יְרֵכ֔וֹ מִלְּבַ֖ד נְשֵׁ֣י בְנֵֽי־יַעֲקֹ֑ב כָּל־נֶ֖פֶשׁ שִׁשִּׁ֥ים וָשֵֽׁשׁ׃ כָּל נַפְשָׁתָא דְּעַאלָא עִם יַעֲקוֹב לְמִצְרַיִם, נָפְקֵי יִרְכֵּיהּ, בָּר, מִנְּשֵׁי בְּנֵי יַעֲקוֹב--כָּל נַפְשָׁתָא, שִׁתִּין וְשֵׁית. 
מו,כז וּבְנֵ֥י יוֹסֵ֛ף אֲשֶׁר־יֻלַּד־ל֥וֹ בְמִצְרַ֖יִם נֶ֣פֶשׁ שְׁנָ֑יִם כָּל־הַנֶּ֧פֶשׁ לְבֵֽית־יַעֲקֹ֛ב הַבָּ֥אָה מִצְרַ֖יְמָה שִׁבְעִֽים׃ {ס} וּבְנֵי יוֹסֵף דְּאִתְיְלִידוּ לֵיהּ בְּמִצְרַיִם, נַפְשָׁתָא תַּרְתֵּין:  כָּל נַפְשָׁתָא לְבֵית יַעֲקוֹב דְּעַאלָא לְמִצְרַיִם, שִׁבְעִין. {ס}
מו,כח וְאֶת־יְהוּדָ֞ה שָׁלַ֤ח לְפָנָיו֙ אֶל־יוֹסֵ֔ף לְהוֹרֹ֥ת לְפָנָ֖יו גֹּ֑שְׁנָה וַיָּבֹ֖אוּ אַ֥רְצָה גֹּֽשֶׁן׃ וְיָת יְהוּדָה שְׁלַח קֳדָמוֹהִי, לְוָת יוֹסֵף, לְפַנָּאָה קֳדָמוֹהִי, לְגֹשֶׁן; וַאֲתוֹ, לְאַרְעָא דְּגֹשֶׁן. 
מו,כט וַיֶּאְסֹ֤ר יוֹסֵף֙ מֶרְכַּבְתּ֔וֹ וַיַּ֛עַל לִקְרַֽאת־יִשְׂרָאֵ֥ל אָבִ֖יו גֹּ֑שְׁנָה וַיֵּרָ֣א אֵלָ֗יו וַיִּפֹּל֙ עַל־צַוָּארָ֔יו וַיֵּ֥בְךְּ עַל־צַוָּארָ֖יו עֽוֹד׃ וְטַקֵּיס יוֹסֵף רְתִכּוֹהִי, וּסְלֵיק לְקַדָּמוּת יִשְׂרָאֵל אֲבוּהִי לְגֹשֶׁן; וְאִתַּחְזִי לֵיהּ, וּנְפַל עַל צַוְרֵיהּ, וּבְכָא עַל צַוְרֵיהּ, עוֹד. 
מו,ל וַיֹּ֧אמֶר יִשְׂרָאֵ֛ל אֶל־יוֹסֵ֖ף אָמ֣וּתָה הַפָּ֑עַם אַֽחֲרֵי֙ רְאוֹתִ֣י אֶת־פָּנֶ֔יךָ כִּ֥י עֽוֹדְךָ֖ חָֽי׃ וַאֲמַר יִשְׂרָאֵל לְיוֹסֵף, אִלּוּ אֲנָא מָאִית זִמְנָא הָדָא מִנַּחַם אֲנָא, בָּתַר דַּחֲזֵיתִנּוּן לְאַפָּךְ, אֲרֵי עַד כְּעַן קַיָּם אַתְּ. 
מו,לא וַיֹּ֨אמֶר יוֹסֵ֤ף אֶל־אֶחָיו֙ וְאֶל־בֵּ֣ית אָבִ֔יו אֶֽעֱלֶ֖ה וְאַגִּ֣ידָה לְפַרְעֹ֑ה וְאֹֽמְרָ֣ה אֵלָ֔יו אַחַ֧י וּבֵית־אָבִ֛י אֲשֶׁ֥ר בְּאֶֽרֶץ־כְּנַ֖עַן בָּ֥אוּ אֵלָֽי׃ וַאֲמַר יוֹסֵף לַאֲחוֹהִי וּלְבֵית אֲבוּהִי, אֶסַּק וַאֲחַוֵּי לְפַרְעֹה; וְאֵימַר לֵיהּ, אַחַי וּבֵית אַבָּא דִּבְאַרְעָא דִּכְנַעַן אֲתוֹ לְוָתִי. 
מו,לב וְהָֽאֲנָשִׁים֙ רֹ֣עֵי צֹ֔אן כִּֽי־אַנְשֵׁ֥י מִקְנֶ֖ה הָי֑וּ וְצֹאנָ֧ם וּבְקָרָ֛ם וְכָל־אֲשֶׁ֥ר לָהֶ֖ם הֵבִֽיאוּ׃ וְגֻבְרַיָּא רָעַן עָנָא, אֲרֵי גֻּבְרֵי מָרֵי גֵּיתֵי הֲווֹ; וְעָנְהוֹן וְתוֹרֵיהוֹן וְכָל דִּילְהוֹן, אֵיתִיאוּ. 
מו,לג וְהָיָ֕ה כִּֽי־יִקְרָ֥א לָכֶ֖ם פַּרְעֹ֑ה וְאָמַ֖ר מַה־מַּֽעֲשֵׂיכֶֽם׃ וִיהֵי, אֲרֵי יִקְרֵי לְכוֹן פַּרְעֹה; וְיֵימַר, מָא עוּבָדֵיכוֹן. 
מו,לד וַֽאֲמַרְתֶּ֗ם אַנְשֵׁ֨י מִקְנֶ֜ה הָי֤וּ עֲבָדֶ֨יךָ֙ מִנְּעוּרֵ֣ינוּ וְעַד־עַ֔תָּה גַּם־אֲנַ֖חְנוּ גַּם־אֲבֹתֵ֑ינוּ בַּֽעֲב֗וּר תֵּֽשְׁבוּ֙ בְּאֶ֣רֶץ גֹּ֔שֶׁן כִּֽי־תוֹעֲבַ֥ת מִצְרַ֖יִם כָּל־רֹ֥עֵה צֹֽאן׃ וְתֵימְרוּן, גֻּבְרֵי מָרֵי גֵּיתֵי הֲווֹ עַבְדָּךְ מִזְּעוּרַנָא וְעַד כְּעַן--אַף אֲנַחְנָא, אַף אֲבָהָתַנָא:  בְּדִיל, דְּתִתְּבוּן בְּאַרְעָא דְּגֹשֶׁן, אֲרֵי מְרַחֲקִין מִצְרָאֵי, כָּל רָעֵי עָנָא. 
מז,א וַיָּבֹ֣א יוֹסֵף֮ וַיַּגֵּ֣ד לְפַרְעֹה֒ וַיֹּ֗אמֶר אָבִ֨י וְאַחַ֜י וְצֹאנָ֤ם וּבְקָרָם֙ וְכָל־אֲשֶׁ֣ר לָהֶ֔ם בָּ֖אוּ מֵאֶ֣רֶץ כְּנָ֑עַן וְהִנָּ֖ם בְּאֶ֥רֶץ גֹּֽשֶׁן׃ וַאֲתָא יוֹסֵף, וְחַוִּי לְפַרְעֹה, וַאֲמַר אַבָּא וְאַחַי וְעָנְהוֹן וְתוֹרֵיהוֹן וְכָל דִּילְהוֹן, אֲתוֹ מֵאַרְעָא דִּכְנָעַן; וְהָא אִנּוּן, בְּאַרְעָא דְּגֹשֶׁן. 
מז,ב וּמִקְצֵ֣ה אֶחָ֔יו לָקַ֖ח חֲמִשָּׁ֣ה אֲנָשִׁ֑ים וַיַּצִּגֵ֖ם לִפְנֵ֥י פַרְעֹֽה׃ וּמִקְצָת אֲחוֹהִי, דְּבַר חַמְשָׁא גֻּבְרִין; וַאֲקֵימִנּוּן, קֳדָם פַּרְעֹה. 
מז,ג וַיֹּ֧אמֶר פַּרְעֹ֛ה אֶל־אֶחָ֖יו מַה־מַּֽעֲשֵׂיכֶ֑ם וַיֹּֽאמְר֣וּ אֶל־פַּרְעֹ֗ה רֹעֵ֥ה צֹאן֙ עֲבָדֶ֔יךָ גַּם־אֲנַ֖חְנוּ גַּם־אֲבוֹתֵֽינוּ׃ וַאֲמַר פַּרְעֹה לַאֲחוֹהִי, מָא עוּבָדֵיכוֹן; וַאֲמַרוּ לְפַרְעֹה, רָעַן עָנָא עַבְדָּךְ--אַף אֲנַחְנָא, אַף אֲבָהָתַנָא. 
מז,ד וַיֹּֽאמְר֣וּ אֶל־פַּרְעֹ֗ה לָג֣וּר בָּאָרֶץ֮ בָּאנוּ֒ כִּי־אֵ֣ין מִרְעֶ֗ה לַצֹּאן֙ אֲשֶׁ֣ר לַֽעֲבָדֶ֔יךָ כִּֽי־כָבֵ֥ד הָֽרָעָ֖ב בְּאֶ֣רֶץ כְּנָ֑עַן וְעַתָּ֛ה יֵֽשְׁבוּ־נָ֥א עֲבָדֶ֖יךָ בְּאֶ֥רֶץ גֹּֽשֶׁן׃ וַאֲמַרוּ לְפַרְעֹה, לְאִתּוֹתָבָא בְּאַרְעָא אֲתֵינָא, אֲרֵי לֵית רִעְיָא לְעָנָא דִּלְעַבְדָּךְ, אֲרֵי תְּקֵיף כַּפְנָא בְּאַרְעָא דִּכְנָעַן; וּכְעַן יִתְּבוּן כְּעַן עַבְדָּךְ, בְּאַרְעָא דְּגֹשֶׁן. 
מז,ה וַיֹּ֣אמֶר פַּרְעֹ֔ה אֶל־יוֹסֵ֖ף לֵאמֹ֑ר אָבִ֥יךָ וְאַחֶ֖יךָ בָּ֥אוּ אֵלֶֽיךָ׃ וַאֲמַר פַּרְעֹה, לְיוֹסֵף לְמֵימַר:  אֲבוּךְ וַאֲחָךְ, אֲתוֹ לְוָתָךְ. 
מז,ו אֶ֤רֶץ מִצְרַ֨יִם֙ לְפָנֶ֣יךָ הִ֔וא בְּמֵיטַ֣ב הָאָ֔רֶץ הוֹשֵׁ֥ב אֶת־אָבִ֖יךָ וְאֶת־אַחֶ֑יךָ יֵֽשְׁבוּ֙ בְּאֶ֣רֶץ גֹּ֔שֶׁן וְאִם־יָדַ֗עְתָּ וְיֶשׁ־בָּם֙ אַנְשֵׁי־חַ֔יִל וְשַׂמְתָּ֛ם שָׂרֵ֥י מִקְנֶ֖ה עַל־אֲשֶׁר־לִֽי׃ אַרְעָא דְּמִצְרַיִם, קֳדָמָךְ הִיא--בִּדְשַׁפִּיר בְּאַרְעָא, אוֹתֵיב יָת אֲבוּךְ וְיָת אֲחָךְ:  יִתְּבוּן, בְּאַרְעָא דְּגֹשֶׁן--וְאִם יְדַעְתְּ וְאִית בְּהוֹן גֻּבְרִין דְּחֵילָא, וּתְמַנֵּינוּן רַבָּנֵי גֵּיתֵי עַל דִּילִי. 
מז,ז וַיָּבֵ֤א יוֹסֵף֙ אֶת־יַֽעֲקֹ֣ב אָבִ֔יו וַיַּֽעֲמִדֵ֖הוּ לִפְנֵ֣י פַרְעֹ֑ה וַיְבָ֥רֶךְ יַֽעֲקֹ֖ב אֶת־פַּרְעֹֽה׃ וְאַיְתִי יוֹסֵף יָת יַעֲקוֹב אֲבוּהִי, וַאֲקִימֵיהּ קֳדָם פַּרְעֹה; וּבָרֵיךְ יַעֲקוֹב, יָת פַּרְעֹה. 
מז,ח וַיֹּ֥אמֶר פַּרְעֹ֖ה אֶֽל־יַעֲקֹ֑ב כַּמָּ֕ה יְמֵ֖י שְׁנֵ֥י חַיֶּֽיךָ׃ וַאֲמַר פַּרְעֹה, לְיַעֲקוֹב:  כַּמָּה, יוֹמֵי שְׁנֵי חַיָּךְ. 
מז,ט וַיֹּ֤אמֶר יַֽעֲקֹב֙ אֶל־פַּרְעֹ֔ה יְמֵי֙ שְׁנֵ֣י מְגוּרַ֔י שְׁלֹשִׁ֥ים וּמְאַ֖ת שָׁנָ֑ה מְעַ֣ט וְרָעִ֗ים הָיוּ֙ יְמֵי֙ שְׁנֵ֣י חַיַּ֔י וְלֹ֣א הִשִּׂ֗יגוּ אֶת־יְמֵי֙ שְׁנֵי֙ חַיֵּ֣י אֲבֹתַ֔י בִּימֵ֖י מְגֽוּרֵיהֶֽם׃ וַאֲמַר יַעֲקוֹב, לְפַרְעֹה, יוֹמֵי שְׁנֵי תּוֹתָבוּתִי, מְאָה וּתְלָתִין שְׁנִין:  זְעֵירִין וּבִישִׁין, הֲווֹ יוֹמֵי שְׁנֵי חַיַּי, וְלָא אַדְבִּיקוּ יָת יוֹמֵי שְׁנֵי חַיֵּי אֲבָהָתַי, בְּיוֹמֵי תּוֹתָבוּתְהוֹן. 
מז,י וַיְבָ֥רֶךְ יַֽעֲקֹ֖ב אֶת־פַּרְעֹ֑ה וַיֵּצֵ֖א מִלִּפְנֵ֥י פַרְעֹֽה׃ וּבָרֵיךְ יַעֲקוֹב, יָת פַּרְעֹה; וּנְפַק, מִן קֳדָם פַּרְעֹה. 
מז,יא וַיּוֹשֵׁ֣ב יוֹסֵף֮ אֶת־אָבִ֣יו וְאֶת־אֶחָיו֒ וַיִּתֵּ֨ן לָהֶ֤ם אֲחֻזָּה֙ בְּאֶ֣רֶץ מִצְרַ֔יִם בְּמֵיטַ֥ב הָאָ֖רֶץ בְּאֶ֣רֶץ רַעְמְסֵ֑ס כַּֽאֲשֶׁ֖ר צִוָּ֥ה פַרְעֹֽה׃ וְאוֹתֵיב יוֹסֵף, יָת אֲבוּהִי וְיָת אֲחוֹהִי, וִיהַב לְהוֹן אַחְסָנָא בְּאַרְעָא דְּמִצְרַיִם, בִּדְשַׁפִּיר בְּאַרְעָא בַּאֲרַע רַעְמְסֵס--כְּמָא דְּפַקֵּיד פַּרְעֹה. 
מז,יב וַיְכַלְכֵּ֤ל יוֹסֵף֙ אֶת־אָבִ֣יו וְאֶת־אֶחָ֔יו וְאֵ֖ת כָּל־בֵּ֣ית אָבִ֑יו לֶ֖חֶם לְפִ֥י הַטָּֽף׃ וְזָן יוֹסֵף יָת אֲבוּהִי וְיָת אֲחוֹהִי, וְיָת כָּל בֵּית אֲבוּהִי--לַחְמָא, לְפֹם טַפְלָא. 
מז,יג וְלֶ֤חֶם אֵין֙ בְּכָל־הָאָ֔רֶץ כִּֽי־כָבֵ֥ד הָֽרָעָ֖ב מְאֹ֑ד וַתֵּ֜לַהּ אֶ֤רֶץ מִצְרַ֨יִם֙ וְאֶ֣רֶץ כְּנַ֔עַן מִפְּנֵ֖י הָֽרָעָֽב׃ וְלַחְמָא לֵית בְּכָל אַרְעָא, אֲרֵי תְּקֵיף כַּפְנָא לַחְדָּא; וְאִשְׁתַּלְהִי, עַמָּא דְּאַרְעָא דְּמִצְרַיִם וְעַמָּא דְּאַרְעָא דִּכְנַעַן, מִן קֳדָם, כַּפְנָא. 
מז,יד וַיְלַקֵּ֣ט יוֹסֵ֗ף אֶת־כָּל־הַכֶּ֨סֶף֙ הַנִּמְצָ֤א בְאֶֽרֶץ־מִצְרַ֨יִם֙ וּבְאֶ֣רֶץ כְּנַ֔עַן בַּשֶּׁ֖בֶר אֲשֶׁר־הֵ֣ם שֹֽׁבְרִ֑ים וַיָּבֵ֥א יוֹסֵ֛ף אֶת־הַכֶּ֖סֶף בֵּ֥יתָה פַרְעֹֽה׃ וְלַקֵּיט יוֹסֵף, יָת כָּל כַּסְפָּא דְּאִשְׁתְּכַח בְּאַרְעָא דְּמִצְרַיִם וּבְאַרְעָא דִּכְנַעַן, בַּעֲבוּרָא, דְּאִנּוּן זָבְנִין; וְאַיְתִי יוֹסֵף יָת כַּסְפָּא, לְבֵית פַּרְעֹה. 
מז,טו וַיִּתֹּ֣ם הַכֶּ֗סֶף מֵאֶ֣רֶץ מִצְרַיִם֮ וּמֵאֶ֣רֶץ כְּנַעַן֒ וַיָּבֹאוּ֩ כָל־מִצְרַ֨יִם אֶל־יוֹסֵ֤ף לֵאמֹר֙ הָֽבָה־לָּ֣נוּ לֶ֔חֶם וְלָ֥מָּה נָמ֖וּת נֶגְדֶּ֑ךָ כִּ֥י אָפֵ֖ס כָּֽסֶף׃ וּשְׁלִים כַּסְפָּא, מֵאַרְעָא דְּמִצְרַיִם וּמֵאַרְעָא דִּכְנַעַן, וַאֲתוֹ כָּל מִצְרָאֵי לְוָת יוֹסֵף לְמֵימַר הַב לַנָא לַחְמָא, וּלְמָא נְמוּת לְקִבְלָךְ:  אֲרֵי שְׁלִים, כַּסְפָּא. 
מז,טז וַיֹּ֤אמֶר יוֹסֵף֙ הָב֣וּ מִקְנֵיכֶ֔ם וְאֶתְּנָ֥ה לָכֶ֖ם בְּמִקְנֵיכֶ֑ם אִם־אָפֵ֖ס כָּֽסֶף׃ וַאֲמַר יוֹסֵף הַבוּ גֵּיתֵיכוֹן, וְאֶתֵּין לְכוֹן בְּגֵיתֵיכוֹן--אִם שְׁלִים, כַּסְפָּא. 
מז,יז וַיָּבִ֣יאוּ אֶת־מִקְנֵיהֶם֮ אֶל־יוֹסֵף֒ וַיִּתֵּ֣ן לָהֶם֩ יוֹסֵ֨ף לֶ֜חֶם בַּסּוּסִ֗ים וּבְמִקְנֵ֥ה הַצֹּ֛אן וּבְמִקְנֵ֥ה הַבָּקָ֖ר וּבַֽחֲמֹרִ֑ים וַיְנַֽהֲלֵ֤ם בַּלֶּ֨חֶם֙ בְּכָל־מִקְנֵהֶ֔ם בַּשָּׁנָ֖ה הַהִֽוא׃ וְאֵיתִיאוּ יָת גֵּיתֵיהוֹן, לְוָת יוֹסֵף, וִיהַב לְהוֹן יוֹסֵף לַחְמָא בְּסוּסָוָתָא וּבְגֵיתֵי עָנָא וּבְגֵיתֵי תּוֹרֵי, וּבִחְמָרַיָּא; וְזָנִנּוּן בְּלַחְמָא בְּכָל גֵּיתֵיהוֹן, בְּשַׁתָּא הַהִיא. 
מז,יח וַתִּתֹּם֮ הַשָּׁנָ֣ה הַהִוא֒ וַיָּבֹ֨אוּ אֵלָ֜יו בַּשָּׁנָ֣ה הַשֵּׁנִ֗ית וַיֹּ֤אמְרוּ לוֹ֙ לֹֽא־נְכַחֵ֣ד מֵֽאֲדֹנִ֔י כִּ֚י אִם־תַּ֣ם הַכֶּ֔סֶף וּמִקְנֵ֥ה הַבְּהֵמָ֖ה אֶל־אֲדֹנִ֑י לֹ֤א נִשְׁאַר֙ לִפְנֵ֣י אֲדֹנִ֔י בִּלְתִּ֥י אִם־גְּוִיָּתֵ֖נוּ וְאַדְמָתֵֽנוּ׃ וּשְׁלֵימַת, שַׁתָּא הַהִיא, וַאֲתוֹ לְוָתֵיהּ בְּשַׁתָּא תִּנְיֵיתָא וַאֲמַרוּ לֵיהּ לָא נְכַסֵּי מִן רִבּוֹנִי, אֱלָהֵין שְׁלִים כַּסְפָּא וְגֵיתֵי בְּעִירָא לְוָת רִבּוֹנִי:  לָא אִשְׁתְּאַר קֳדָם רִבּוֹנִי, אֱלָהֵין גִּוְיָתַנָא וְאַרְעַנָא. 
מז,יט לָ֧מָּה נָמ֣וּת לְעֵינֶ֗יךָ גַּם־אֲנַ֨חְנוּ֙ גַּ֣ם אַדְמָתֵ֔נוּ קְנֵֽה־אֹתָ֥נוּ וְאֶת־אַדְמָתֵ֖נוּ בַּלָּ֑חֶם וְנִֽהְיֶ֞ה אֲנַ֤חְנוּ וְאַדְמָתֵ֨נוּ֙ עֲבָדִ֣ים לְפַרְעֹ֔ה וְתֶן־זֶ֗רַע וְנִֽחְיֶה֙ וְלֹ֣א נָמ֔וּת וְהָֽאֲדָמָ֖ה לֹ֥א תֵשָֽׁם׃ לְמָא נְמוּת לְעֵינָךְ, אַף אֲנַחְנָא אַף אַרְעַנָא--קְנֵי יָתַנָא וְיָת אַרְעַנָא, בְּלַחְמָא; וּנְהֵי אֲנַחְנָא וְאַרְעַנָא, עַבְדִּין לְפַרְעֹה, וְהַב בַּר זְרַע וְנֵיחֵי וְלָא נְמוּת, וְאַרְעָא לָא תְּבוּר. 
מז,כ וַיִּ֨קֶן יוֹסֵ֜ף אֶת־כָּל־אַדְמַ֤ת מִצְרַ֨יִם֙ לְפַרְעֹ֔ה כִּֽי־מָכְר֤וּ מִצְרַ֨יִם֙ אִ֣ישׁ שָׂדֵ֔הוּ כִּֽי־חָזַ֥ק עֲלֵהֶ֖ם הָֽרָעָ֑ב וַתְּהִ֥י הָאָ֖רֶץ לְפַרְעֹֽה׃ וּקְנָא יוֹסֵף יָת כָּל אַרְעָא דְּמִצְרַיִם, לְפַרְעֹה, אֲרֵי זַבִּינוּ מִצְרָאֵי גְּבַר חַקְלֵיהּ, אֲרֵי תְּקֵיף עֲלֵיהוֹן כַּפְנָא; וַהֲוָת אַרְעָא, לְפַרְעֹה. 
מז,כא וְאֶ֨ת־הָעָ֔ם הֶֽעֱבִ֥יר אֹת֖וֹ לֶֽעָרִ֑ים מִקְצֵ֥ה גְבוּל־מִצְרַ֖יִם וְעַד־קָצֵֽהוּ׃ וְיָת עַמָּא--אַעְבַּר יָתֵיהּ, מִקְּרֵי לִקְרֵי:  מִסּוֹף תְּחוּם מִצְרַיִם, וְעַד סוֹפֵיהּ. 
מז,כב רַ֛ק אַדְמַ֥ת הַכֹּֽהֲנִ֖ים לֹ֣א קָנָ֑ה כִּי֩ חֹ֨ק לַכֹּֽהֲנִ֜ים מֵאֵ֣ת פַּרְעֹ֗ה וְאָֽכְל֤וּ אֶת־חֻקָּם֙ אֲשֶׁ֨ר נָתַ֤ן לָהֶם֙ פַּרְעֹ֔ה עַל־כֵּ֕ן לֹ֥א מָֽכְר֖וּ אֶת־אַדְמָתָֽם׃ לְחוֹד אֲרַע כֻּמְרַיָּא, לָא קְנָא:  אֲרֵי חוּלָקָא לְכֻמְרַיָּא מִן קֳדָם פַּרְעֹה, וְאָכְלִין יָת חוּלָקְהוֹן דִּיהַב לְהוֹן פַּרְעֹה--עַל כֵּין, לָא זַבִּינוּ יָת אֲרַעְהוֹן. 
מז,כג וַיֹּ֤אמֶר יוֹסֵף֙ אֶל־הָעָ֔ם הֵן֩ קָנִ֨יתִי אֶתְכֶ֥ם הַיּ֛וֹם וְאֶת־אַדְמַתְכֶ֖ם לְפַרְעֹ֑ה הֵֽא־לָכֶ֣ם זֶ֔רַע וּזְרַעְתֶּ֖ם אֶת־הָֽאֲדָמָֽה׃ וַאֲמַר יוֹסֵף לְעַמָּא, הָא קְנֵיתִי יָתְכוֹן יוֹמָא דֵּין וְיָת אֲרַעְכוֹן לְפַרְעֹה; הָא לְכוֹן בַּר זַרְעָא, וְתִזְרְעוּן יָת אַרְעָא. 
מז,כד וְהָיָה֙ בַּתְּבוּאֹ֔ת וּנְתַתֶּ֥ם חֲמִישִׁ֖ית לְפַרְעֹ֑ה וְאַרְבַּ֣ע הַיָּדֹ֡ת יִֽהְיֶ֣ה לָכֶם֩ לְזֶ֨רַע הַשָּׂדֶ֧ה וּֽלְאָכְלְכֶ֛ם וְלַֽאֲשֶׁ֥ר בְּבָֽתֵּיכֶ֖ם וְלֶֽאֱכֹ֥ל לְטַפְּכֶֽם׃ וִיהֵי, בְּאֵיעוֹלֵי עֲלַלְתָּא, וְתִתְּנוּן חַד מִן חַמְשָׁא, לְפַרְעֹה; וְאַרְבְּעָא חוּלָקִין יְהֵי לְכוֹן לְבַר זְרַע חַקְלָא וּלְמֵיכַלְכוֹן, וְלַאֲנָשׁ בָּתֵּיכוֹן--וּלְמֵיכַל לְטַפְלְכוֹן. 
מז,כה וַיֹּֽאמְר֖וּ הֶֽחֱיִתָ֑נוּ נִמְצָא־חֵן֙ בְּעֵינֵ֣י אֲדֹנִ֔י וְהָיִ֥ינוּ עֲבָדִ֖ים לְפַרְעֹֽה׃ וַאֲמַרוּ, קַיֵּימְתַּנָא; נַשְׁכַּח רַחֲמִין בְּעֵינֵי רִבּוֹנִי, וּנְהֵי עַבְדִּין לְפַרְעֹה. 
מז,כו וַיָּ֣שֶׂם אֹתָ֣הּ יוֹסֵ֡ף לְחֹק֩ עַד־הַיּ֨וֹם הַזֶּ֜ה עַל־אַדְמַ֥ת מִצְרַ֛יִם לְפַרְעֹ֖ה לַחֹ֑מֶשׁ רַ֞ק אַדְמַ֤ת הַכֹּֽהֲנִים֙ לְבַדָּ֔ם לֹ֥א הָֽיְתָ֖ה לְפַרְעֹֽה׃ וְשַׁוִּי יָתַהּ יוֹסֵף לִגְזֵירָא עַד יוֹמָא הָדֵין עַל אַרְעָא דְּמִצְרַיִם, דִּיהוֹן יָהֲבִין חַד מִן חַמְשָׁא--לְפַרְעֹה:  לְחוֹד אֲרַע כֻּמְרַיָּא, בִּלְחוֹדֵיהוֹן--לָא הֲוָת, לְפַרְעֹה. 
מז,כז וַיֵּ֧שֶׁב יִשְׂרָאֵ֛ל בְּאֶ֥רֶץ מִצְרַ֖יִם בְּאֶ֣רֶץ גֹּ֑שֶׁן וַיֵּאָֽחֲז֣וּ בָ֔הּ וַיִּפְר֥וּ וַיִּרְבּ֖וּ מְאֹֽד׃ וִיתֵיב יִשְׂרָאֵל בְּאַרְעָא דְּמִצְרַיִם, בְּאַרְעָא דְּגֹשֶׁן; וְאַחְסִינוּ בַּהּ, וּנְפִישׁוּ וּסְגִיאוּ לַחְדָּא.
