פקודי

לח,כא אֵ֣לֶּה פְקוּדֵ֤י הַמִּשְׁכָּן֙ מִשְׁכַּ֣ן הָֽעֵדֻ֔ת אֲשֶׁ֥ר פֻּקַּ֖ד עַל־פִּ֣י מֹשֶׁ֑ה עֲבֹדַת֙ הַלְוִיִּ֔ם בְּיַד֙ אִֽיתָמָ֔ר בֶּֽן־אַהֲרֹ֖ן הַכֹּהֵֽן׃ אִלֵּין מִנְיָנֵי מַשְׁכְּנָא מַשְׁכְּנָא דְּסָהֲדוּתָא, דְּאִתְמְנִיאוּ עַל מֵימְרָא דְּמֹשֶׁה:  פֻּלְחַן, לֵיוָאֵי, בִּידָא דְּאִיתָמָר, בַּר אַהֲרוֹן כָּהֲנָא. 
לח,כב וּבְצַלְאֵ֛ל בֶּן־אוּרִ֥י בֶן־ח֖וּר לְמַטֵּ֣ה יְהוּדָ֑ה עָשָׂ֕ה אֵ֛ת כָּל־אֲשֶׁר־צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ וּבְצַלְאֵל בַּר אוּרִי בַּר חוּר, לְשִׁבְטָא דִּיהוּדָה, עֲבַד, יָת כָּל דְּפַקֵּיד יְיָ יָת מֹשֶׁה. 
לח,כג וְאִתּ֗וֹ אָֽהֳלִיאָ֞ב בֶּן־אֲחִֽיסָמָ֛ךְ לְמַטֵּה־דָ֖ן חָרָ֣שׁ וְחֹשֵׁ֑ב וְרֹקֵ֗ם בַּתְּכֵ֨לֶת֙ וּבָֽאַרְגָּמָ֔ן וּבְתוֹלַ֥עַת הַשָּׁנִ֖י וּבַשֵּֽׁשׁ׃ {ס} וְעִמֵּיהּ, אָהֳלִיאָב בַּר אֲחִיסָמָךְ לְשִׁבְטָא דְּדָן--נַגָּר וְאֻמָּן; וְצַיָּר, בְּתַכְלָא וּבְאַרְגְּוָנָא, וּבִצְבַע זְהוֹרִי, וּבְבוּצָא. {ס}
לח,כד כָּל־הַזָּהָ֗ב הֶֽעָשׂוּי֙ לַמְּלָאכָ֔ה בְּכֹ֖ל מְלֶ֣אכֶת הַקֹּ֑דֶשׁ וַיְהִ֣י ׀ זְהַ֣ב הַתְּנוּפָ֗ה תֵּ֤שַׁע וְעֶשְׂרִים֙ כִּכָּ֔ר וּשְׁבַ֨ע מֵא֧וֹת וּשְׁלֹשִׁ֛ים שֶׁ֖קֶל בְּשֶׁ֥קֶל הַקֹּֽדֶשׁ׃ כָּל דַּהְבָּא, דְּאִתְעֲבֵיד לַעֲבִידְתָא, בְּכֹל, עֲבִידַת קֻדְשָׁא--וַהֲוָה דְּהַב אֲרָמוּתָא, עַסְרִין וּתְשַׁע כַּכְּרִין, וּשְׁבַע מְאָה וּתְלָתִין סִלְעִין, בְּסִלְעֵי קֻדְשָׁא. 
לח,כה וְכֶ֛סֶף פְּקוּדֵ֥י הָֽעֵדָ֖ה מְאַ֣ת כִּכָּ֑ר וְאֶלֶף֩ וּשְׁבַ֨ע מֵא֜וֹת וַֽחֲמִשָּׁ֧ה וְשִׁבְעִ֛ים שֶׁ֖קֶל בְּשֶׁ֥קֶל הַקֹּֽדֶשׁ׃ וּכְסַף מִנְיָנֵי כְּנִשְׁתָּא, מְאָה כַּכְּרִין; וְאֶלֶף וּשְׁבַע מְאָה וְשִׁבְעִין וַחֲמֵישׁ, סִלְעִין--בְּסִלְעֵי קֻדְשָׁא. 
לח,כו בֶּ֚קַע לַגֻּלְגֹּ֔לֶת מַֽחֲצִ֥ית הַשֶּׁ֖קֶל בְּשֶׁ֣קֶל הַקֹּ֑דֶשׁ לְכֹ֨ל הָֽעֹבֵ֜ר עַל־הַפְּקֻדִ֗ים מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה לְשֵׁשׁ־מֵא֥וֹת אֶ֨לֶף֙ וּשְׁלֹ֣שֶׁת אֲלָפִ֔ים וַֽחֲמֵ֥שׁ מֵא֖וֹת וַֽחֲמִשִּֽׁים׃ תִּקְלָא, לְגֻלְגֻּלְתָּא, פַּלְגוּת סִלְעָא, בְּסִלְעֵי קֻדְשָׁא--לְכֹל דְּעָבַר עַל מִנְיָנַיָּא, מִבַּר עַסְרִין שְׁנִין וּלְעֵילָא, לְשֵׁית מְאָה וּתְלָתָא אַלְפִין, וַחֲמֵישׁ מְאָה וְחַמְשִׁין. 
לח,כז וַיְהִ֗י מְאַת֙ כִּכַּ֣ר הַכֶּ֔סֶף לָצֶ֗קֶת אֵ֚ת אַדְנֵ֣י הַקֹּ֔דֶשׁ וְאֵ֖ת אַדְנֵ֣י הַפָּרֹ֑כֶת מְאַ֧ת אֲדָנִ֛ים לִמְאַ֥ת הַכִּכָּ֖ר כִּכָּ֥ר לָאָֽדֶן׃ וַהֲוָאָה, מְאָה כַּכְּרֵי כַּסְפָּא, לְאַתָּכָא יָת סָמְכֵי קֻדְשָׁא, וְיָת סָמְכֵי פָּרֻכְתָּא:  מְאָה סָמְכִין לִמְאָה כַּכְּרִין, כַּכְּרָא לְסָמְכָא. 
לח,כח וְאֶת־הָאֶ֜לֶף וּשְׁבַ֤ע הַמֵּאוֹת֙ וַֽחֲמִשָּׁ֣ה וְשִׁבְעִ֔ים עָשָׂ֥ה וָוִ֖ים לָֽעַמּוּדִ֑ים וְצִפָּ֥ה רָֽאשֵׁיהֶ֖ם וְחִשַּׁ֥ק אֹתָֽם׃ וְיָת אֶלֶף וּשְׁבַע מְאָה, וְשִׁבְעִין וַחֲמֵישׁ, עֲבַד וָוִין, לְעַמּוּדַיָּא; וְחַפִּי רֵישֵׁיהוֹן, וְכַבֵּישׁ יָתְהוֹן. 
לח,כט וּנְחֹ֥שֶׁת הַתְּנוּפָ֖ה שִׁבְעִ֣ים כִּכָּ֑ר וְאַלְפַּ֥יִם וְאַרְבַּע־מֵא֖וֹת שָֽׁקֶל׃ וּנְחָשׁ אֲרָמוּתָא, שִׁבְעִין כַּכְּרִין, וּתְרֵין אַלְפִין וְאַרְבַּע מְאָה, סִלְעִין. 
לח,ל וַיַּ֣עַשׂ בָּ֗הּ אֶת־אַדְנֵי֙ פֶּ֚תַח אֹ֣הֶל מוֹעֵ֔ד וְאֵת֙ מִזְבַּ֣ח הַנְּחֹ֔שֶׁת וְאֶת־מִכְבַּ֥ר הַנְּחֹ֖שֶׁת אֲשֶׁר־ל֑וֹ וְאֵ֖ת כָּל־כְּלֵ֥י הַמִּזְבֵּֽחַ׃ וַעֲבַד בַּהּ, יָת סָמְכֵי תְּרַע מַשְׁכַּן זִמְנָא, וְיָת מַדְבְּחָא דִּנְחָשָׁא, וְיָת סְרָדָא דִּנְחָשָׁא דִּילֵיהּ; וְיָת, כָּל מָנֵי מַדְבְּחָא. 
לח,לא וְאֶת־אַדְנֵ֤י הֶֽחָצֵר֙ סָבִ֔יב וְאֶת־אַדְנֵ֖י שַׁ֣עַר הֶֽחָצֵ֑ר וְאֵ֨ת כָּל־יִתְדֹ֧ת הַמִּשְׁכָּ֛ן וְאֶת־כָּל־יִתְדֹ֥ת הֶֽחָצֵ֖ר סָבִֽיב׃ וְיָת סָמְכֵי דָּרְתָא סְחוֹר סְחוֹר, וְיָת סָמְכֵי תְּרַע דָּרְתָא; וְיָת כָּל סִכֵּי מַשְׁכְּנָא וְיָת כָּל סִכֵּי דָּרְתָא, סְחוֹר סְחוֹר. 
לט,א וּמִן־הַתְּכֵ֤לֶת וְהָֽאַרְגָּמָן֙ וְתוֹלַ֣עַת הַשָּׁנִ֔י עָשׂ֥וּ בִגְדֵי־שְׂרָ֖ד לְשָׁרֵ֣ת בַּקֹּ֑דֶשׁ וַֽיַּעֲשׂ֞וּ אֶת־בִּגְדֵ֤י הַקֹּ֨דֶשׁ֙ אֲשֶׁ֣ר לְאַֽהֲרֹ֔ן כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ {פ} וּמִן תַּכְלָא וְאַרְגְּוָנָא וּצְבַע זְהוֹרִי, עֲבַדוּ לְבוּשֵׁי שִׁמּוּשָׁא לְשַׁמָּשָׁא בְּקֻדְשָׁא; וַעֲבַדוּ יָת לְבוּשֵׁי קֻדְשָׁא, דִּלְאַהֲרוֹן, כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. {פ}
לט,ב וַיַּ֖עַשׂ אֶת־הָֽאֵפֹ֑ד זָהָ֗ב תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֥שׁ מָשְׁזָֽר׃ וַעֲבַד, יָת אֵיפוֹדָא:  דַּהְבָּא, תַּכְלָא וְאַרְגְּוָנָא וּצְבַע זְהוֹרִי--וּבוּץ שְׁזִיר. 
לט,ג וַֽיְרַקְּע֞וּ אֶת־פַּחֵ֣י הַזָּהָב֮ וְקִצֵּ֣ץ פְּתִילִם֒ לַֽעֲשׂ֗וֹת בְּת֤וֹךְ הַתְּכֵ֨לֶת֙ וּבְת֣וֹךְ הָֽאַרְגָּמָ֔ן וּבְת֛וֹךְ תּוֹלַ֥עַת הַשָּׁנִ֖י וּבְת֣וֹךְ הַשֵּׁ֑שׁ מַֽעֲשֵׂ֖ה חֹשֵֽׁב׃ וְרַדִידוּ יָת טַסֵי דְּדַהְבָּא, וְקַצִּיצוּ חוּטִין, לְמֶעֱבַד בְּגוֹ תַּכְלָא וּבְגוֹ אַרְגְּוָנָא, וּבְגוֹ צְבַע זְהוֹרִי וּבְגוֹ בּוּצָא--עוֹבָד, אֻמָּן. 
לט,ד כְּתֵפֹ֥ת עָֽשׂוּ־ל֖וֹ חֹֽבְרֹ֑ת עַל־שְׁנֵ֥י קצוותו (קְצוֹתָ֖יו) חֻבָּֽר׃ כִּתְפִין עֲבַדוּ לֵיהּ, מְלָפְפָן--עַל תְּרֵין סִטְרוֹהִי, מְלָפַף. 
לט,ה וְחֵ֨שֶׁב אֲפֻדָּת֜וֹ אֲשֶׁ֣ר עָלָ֗יו מִמֶּ֣נּוּ הוּא֮ כְּמַֽעֲשֵׂהוּ֒ זָהָ֗ב תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֣שׁ מָשְׁזָ֑ר כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ {ס} וְהִמְיַן תִּקּוּנֵיהּ דַּעֲלוֹהִי, מִנֵּיהּ הוּא כְּעוֹבָדוֹהִי, דַּהְבָּא תַּכְלָא וְאַרְגְּוָנָא וּצְבַע זְהוֹרִי, וּבוּץ שְׁזִיר--כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. {ס}
לט,ו וַֽיַּעֲשׂוּ֙ אֶת־אַבְנֵ֣י הַשֹּׁ֔הַם מֻֽסַבֹּ֖ת מִשְׁבְּצֹ֣ת זָהָ֑ב מְפֻתָּחֹת֙ פִּתּוּחֵ֣י חוֹתָ֔ם עַל־שְׁמ֖וֹת בְּנֵ֥י יִשְׂרָאֵֽל׃ וַעֲבַדוּ יָת אַבְנֵי בֻּרְלָא, מְשַׁקְּעָן מְרַמְּצָן דִּדְהַב, גְּלִיפָן כְּתָב מְפָרַשׁ, עַל שְׁמָהָת בְּנֵי יִשְׂרָאֵל. 
לט,ז וַיָּ֣שֶׂם אֹתָ֗ם עַ֚ל כִּתְפֹ֣ת הָֽאֵפֹ֔ד אַבְנֵ֥י זִכָּר֖וֹן לִבְנֵ֣י יִשְׂרָאֵ֑ל כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ {פ} וְשַׁוִּי יָתְהוֹן, עַל כִּתְפֵי אֵיפוֹדָא--אַבְנֵי דֻּכְרָנָא, לִבְנֵי יִשְׂרָאֵל:  כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. {פ}
לט,ח וַיַּ֧עַשׂ אֶת־הַחֹ֛שֶׁן מַֽעֲשֵׂ֥ה חֹשֵׁ֖ב כְּמַֽעֲשֵׂ֣ה אֵפֹ֑ד זָהָ֗ב תְּכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י וְשֵׁ֥שׁ מָשְׁזָֽר׃ וַעֲבַד יָת חֻשְׁנָא עוֹבָד אֻמָּן, כְּעוֹבָד אֵיפוֹדָא:  דַּהְבָּא, תַּכְלָא וְאַרְגְּוָנָא וּצְבַע זְהוֹרִי--וּבוּץ שְׁזִיר. 
לט,ט רָב֧וּעַ הָיָ֛ה כָּפ֖וּל עָשׂ֣וּ אֶת־הַחֹ֑שֶׁן זֶ֧רֶת אָרְכּ֛וֹ וְזֶ֥רֶת רָחְבּ֖וֹ כָּפֽוּל׃ מְרֻבַּע הֲוָה עִיף, עֲבַדוּ יָת חֻשְׁנָא--זַרְתָּא אֻרְכֵּיהּ וְזַרְתָּא פֻּתְיֵיהּ, עִיף. 
לט,י וַיְמַ֨לְאוּ־ב֔וֹ אַרְבָּעָ֖ה ט֣וּרֵי אָ֑בֶן ט֗וּר אֹ֤דֶם פִּטְדָה֙ וּבָרֶ֔קֶת הַטּ֖וּר הָֽאֶחָֽד׃ וְאַשְׁלִימוּ בֵּיהּ--אַרְבְּעָא, סִדְרִין דְּאֶבֶן טָבָא:  סִדְרָא קַדְמָאָה, סָמְקָן יָרְקָן וּבָרְקָן--סִדְרָא, חַד. 
לט,יא וְהַטּ֖וּר הַשֵּׁנִ֑י נֹ֥פֶךְ סַפִּ֖יר וְיָֽהֲלֹֽם׃ וְסִדְרָא, תִּנְיָנָא--אִזְמַרַגְדִּין שַׁבְזֵיז, וְסַבְהֲלוֹם. 
לט,יב וְהַטּ֖וּר הַשְּׁלִישִׁ֑י לֶ֥שֶׁם שְׁב֖וֹ וְאַחְלָֽמָה׃ וְסִדְרָא, תְּלִיתָאָה--קַנְכֵּירִי טְרַקְיָא, וְעֵין עִגְלָא. 
לט,יג וְהַטּוּר֙ הָֽרְבִיעִ֔י תַּרְשִׁ֥ישׁ שֹׁ֖הַם וְיָֽשְׁפֵ֑ה מֽוּסַבֹּ֛ת מִשְׁבְּצֹ֥ת זָהָ֖ב בְּמִלֻּֽאֹתָֽם׃ וְסִדְרָא, רְבִיעָאָה--כְּרוּם יַמָּא וּבֻרְלָא, וּפַנְתֵּירִי:  מְשַׁקְּעָן מְרַמְּצָן בִּדְהַב, בְּאַשְׁלָמוּתְהוֹן. 
לט,יד וְ֠הָֽאֲבָנִים עַל־שְׁמֹ֨ת בְּנֵֽי־יִשְׂרָאֵ֥ל הֵ֛נָּה שְׁתֵּ֥ים עֶשְׂרֵ֖ה עַל־שְׁמֹתָ֑ם פִּתּוּחֵ֤י חֹתָם֙ אִ֣ישׁ עַל־שְׁמ֔וֹ לִשְׁנֵ֥ים עָשָׂ֖ר שָֽׁבֶט׃ וְאַבְנַיָּא עַל שְׁמָהָת בְּנֵי יִשְׂרָאֵל אִנִּין, תַּרְתַּא עַסְרֵי--עַל שְׁמָהָתְהוֹן:  כְּתָב מְפָרַשׁ כִּגְלָף דְּעִזְקָא גְּבַר עַל שְׁמֵיהּ, לִתְרֵי עֲסַר שִׁבְטִין. 
לט,טו וַיַּֽעֲשׂ֧וּ עַל־הַחֹ֛שֶׁן שַׁרְשְׁרֹ֥ת גַּבְלֻ֖ת מַֽעֲשֵׂ֣ה עֲבֹ֑ת זָהָ֖ב טָהֽוֹר׃ וַעֲבַדוּ עַל חֻשְׁנָא תִּכִּין מְתַחֲמָן, עוֹבָד גְּדִילוּ, דִּדְהַב, דְּכֵי. 
לט,טז וַֽיַּעֲשׂ֗וּ שְׁתֵּי֙ מִשְׁבְּצֹ֣ת זָהָ֔ב וּשְׁתֵּ֖י טַבְּעֹ֣ת זָהָ֑ב וַֽיִּתְּנ֗וּ אֶת־שְׁתֵּי֙ הַטַּבָּעֹ֔ת עַל־שְׁנֵ֖י קְצ֥וֹת הַחֹֽשֶׁן׃ וַעֲבַדוּ, תַּרְתֵּין מְרַמְּצָן דִּדְהַב, וְתַרְתֵּין, עִזְקָן דִּדְהַב; וִיהַבוּ, יָת תַּרְתֵּין עִזְקָתָא, עַל תְּרֵין, סִטְרֵי חֻשְׁנָא. 
לט,יז וַֽיִּתְּנ֗וּ שְׁתֵּי֙ הָֽעֲבֹתֹ֣ת הַזָּהָ֔ב עַל־שְׁתֵּ֖י הַטַּבָּעֹ֑ת עַל־קְצ֖וֹת הַחֹֽשֶׁן׃ וִיהַבוּ, תַּרְתֵּין גְּדִילָן דִּדְהַב, עַל תַּרְתֵּין, עִזְקָתָא--עַל סִטְרֵי, חֻשְׁנָא. 
לט,יח וְאֵ֨ת שְׁתֵּ֤י קְצוֹת֙ שְׁתֵּ֣י הָֽעֲבֹתֹ֔ת נָֽתְנ֖וּ עַל־שְׁתֵּ֣י הַֽמִּשְׁבְּצֹ֑ת וַֽיִּתְּנֻ֛ם עַל־כִּתְפֹ֥ת הָֽאֵפֹ֖ד אֶל־מ֥וּל פָּנָֽיו׃ וְיָת תַּרְתֵּין גְּדִילָן דְּעַל תְּרֵין סִטְרוֹהִי, יְהַבוּ עַל תַּרְתֵּין מְרַמְּצָתָא; וִיהַבוּנִין עַל כִּתְפֵי אֵיפוֹדָא, לָקֳבֵיל אַפּוֹהִי. 
לט,יט וַֽיַּעֲשׂ֗וּ שְׁתֵּי֙ טַבְּעֹ֣ת זָהָ֔ב וַיָּשִׂ֕ימוּ עַל־שְׁנֵ֖י קְצ֣וֹת הַחֹ֑שֶׁן עַל־שְׂפָת֕וֹ אֲשֶׁ֛ר אֶל־עֵ֥בֶר הָֽאֵפֹ֖ד בָּֽיְתָה׃ וַעֲבַדוּ, תַּרְתֵּין עִזְקָן דִּדְהַב, וְשַׁוִּיאוּ, עַל תְּרֵין סִטְרֵי חֻשְׁנָא:  עַל סִפְתֵּיהּ--דִּלְעִבְרָא דְּאֵיפוֹדָא, לְגָיו. 
לט,כ וַֽיַּעֲשׂוּ֮ שְׁתֵּ֣י טַבְּעֹ֣ת זָהָב֒ וַֽיִּתְּנֻ֡ם עַל־שְׁתֵּי֩ כִתְפֹ֨ת הָֽאֵפֹ֤ד מִלְּמַ֨טָּה֙ מִמּ֣וּל פָּנָ֔יו לְעֻמַּ֖ת מַחְבַּרְתּ֑וֹ מִמַּ֕עַל לְחֵ֖שֶׁב הָֽאֵפֹֽד׃ וַעֲבַדוּ, תַּרְתֵּין עִזְקָן דִּדְהַב, וִיהַבוּנִין עַל תְּרֵין כִּתְפֵי אֵיפוֹדָא מִלְּרַע מִלָּקֳבֵיל אַפּוֹהִי, לָקֳבֵיל בֵּית לוֹפֵי--מֵעִלָּוֵי, לְהִמְיַן אֵיפוֹדָא. 
לט,כא וַיִּרְכְּס֣וּ אֶת־הַחֹ֡שֶׁן מִטַּבְּעֹתָיו֩ אֶל־טַבְּעֹ֨ת הָֽאֵפֹ֜ד בִּפְתִ֣יל תְּכֵ֗לֶת לִֽהְיֹת֙ עַל־חֵ֣שֶׁב הָֽאֵפֹ֔ד וְלֹֽא־יִזַּ֣ח הַחֹ֔שֶׁן מֵעַ֖ל הָֽאֵפֹ֑ד כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ {פ} וַאֲחַדוּ יָת חֻשְׁנָא מֵעִזְקָתֵיהּ לְעִזְקָת אֵיפוֹדָא בְּחוּטָא דִּתְכִילְתָא, לְמִהְוֵי עַל הִמְיַן אֵיפוֹדָא, וְלָא יִתְפָּרַק חֻשְׁנָא, מֵעִלָּוֵי אֵיפוֹדָא--כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. {פ}
לט,כב וַיַּ֛עַשׂ אֶת־מְעִ֥יל הָֽאֵפֹ֖ד מַֽעֲשֵׂ֣ה אֹרֵ֑ג כְּלִ֖יל תְּכֵֽלֶת׃ וַעֲבַד יָת מְעִיל אֵיפוֹדָא, עוֹבָד מָחֵי, גְּמִיר, תַּכְלָא. 
לט,כג וּפִֽי־הַמְּעִ֥יל בְּתוֹכ֖וֹ כְּפִ֣י תַחְרָ֑א שָׂפָ֥ה לְפִ֛יו סָבִ֖יב לֹ֥א יִקָּרֵֽעַ׃ וּפֻמֵּיהּ דִּמְעִילָא כְּפִיל לְגַוֵּיהּ, כְּפֹם שִׁרְיָן; תּוּרָא מַקַּף לְפֻמֵּיהּ סְחוֹר סְחוֹר, לָא יִתְבְּזַע. 
לט,כד וַֽיַּעֲשׂוּ֙ עַל־שׁוּלֵ֣י הַמְּעִ֔יל רִמּוֹנֵ֕י תְּכֵ֥לֶת וְאַרְגָּמָ֖ן וְתוֹלַ֣עַת שָׁנִ֑י מָשְׁזָֽר׃ וַעֲבַדוּ, עַל שִׁפּוֹלֵי מְעִילָא, רִמּוֹנֵי, תַּכְלָא וְאַרְגְּוָנָא וּצְבַע זְהוֹרִי--שְׁזִיר. 
לט,כה וַיַּֽעֲשׂ֥וּ פַֽעֲמֹנֵ֖י זָהָ֣ב טָה֑וֹר וַיִּתְּנ֨וּ אֶת־הַפַּֽעֲמֹנִ֜ים בְּת֣וֹךְ הָֽרִמֹּנִ֗ים עַל־שׁוּלֵ֤י הַמְּעִיל֙ סָבִ֔יב בְּת֖וֹךְ הָֽרִמֹּנִֽים׃ וַעֲבַדוּ זַגִּין, דִּדְהַב דְּכֵי; וִיהַבוּ יָת זַגַּיָּא בְּגוֹ רִמּוֹנַיָּא, עַל שִׁפּוֹלֵי מְעִילָא סְחוֹר סְחוֹר--בְּגוֹ, רִמּוֹנַיָּא. 
לט,כו פַּֽעֲמֹ֤ן וְרִמֹּן֙ פַּֽעֲמֹ֣ן וְרִמֹּ֔ן עַל־שׁוּלֵ֥י הַמְּעִ֖יל סָבִ֑יב לְשָׁרֵ֕ת כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ {ס} זַגָּא וְרִמּוֹנָא זַגָּא וְרִמּוֹנָא, עַל שִׁפּוֹלֵי מְעִילָא סְחוֹר סְחוֹר, לְשַׁמָּשָׁא, כְּמָא דְּפַקֵּיד יְיָ יָת מֹשֶׁה. {ס}
לט,כז וַֽיַּעֲשׂ֛וּ אֶת־הַכָּתְנֹ֥ת שֵׁ֖שׁ מַֽעֲשֵׂ֣ה אֹרֵ֑ג לְאַֽהֲרֹ֖ן וּלְבָנָֽיו׃ וַעֲבַדוּ יָת כִּתּוּנִין דְּבוּצָא, עוֹבָד מָחֵי, לְאַהֲרוֹן, וְלִבְנוֹהִי. 
לט,כח וְאֵת֙ הַמִּצְנֶ֣פֶת שֵׁ֔שׁ וְאֶת־פַּֽאֲרֵ֥י הַמִּגְבָּעֹ֖ת שֵׁ֑שׁ וְאֶת־מִכְנְסֵ֥י הַבָּ֖ד שֵׁ֥שׁ מָשְׁזָֽר׃ וְיָת מַצְנַפְתָּא דְּבוּצָא, וְיָת שְׁבָח קוֹבְעַיָּא דְּבוּצָא, וְיָת מִכְנְסֵי בּוּצָא, דְּבוּץ שְׁזִיר. 
לט,כט וְֽאֶת־הָאַבְנֵ֞ט שֵׁ֣שׁ מָשְׁזָ֗ר וּתְכֵ֧לֶת וְאַרְגָּמָ֛ן וְתוֹלַ֥עַת שָׁנִ֖י מַֽעֲשֵׂ֣ה רֹקֵ֑ם כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ {ס} וְיָת הִמְיָנָא דְּבוּץ שְׁזִיר, וְתַכְלָא וְאַרְגְּוָנָא וּצְבַע זְהוֹרִי--עוֹבָד צַיָּר:  כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. {ס}
לט,ל וַֽיַּעֲשׂ֛וּ אֶת־צִ֥יץ נֵֽזֶר־הַקֹּ֖דֶשׁ זָהָ֣ב טָה֑וֹר וַיִּכְתְּב֣וּ עָלָ֗יו מִכְתַּב֙ פִּתּוּחֵ֣י חוֹתָ֔ם קֹ֖דֶשׁ לַֽיהוָֽה׃ וַעֲבַדוּ יָת צִיצָא כְּלִילָא דְּקֻדְשָׁא, דִּדְהַב דְּכֵי; וּכְתַבוּ עֲלוֹהִי, כְּתָב מְפָרַשׁ--קֹדֶשׁ, לַייָ. 
לט,לא וַיִּתְּנ֤וּ עָלָיו֙ פְּתִ֣יל תְּכֵ֔לֶת לָתֵ֥ת עַל־הַמִּצְנֶ֖פֶת מִלְמָ֑עְלָה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ {ס} וִיהַבוּ עֲלוֹהִי חוּטָא דִּתְכִילְתָא, לְמִתַּן עַל מַצְנַפְתָּא מִלְּעֵילָא, כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. {ס}
לט,לב וַתֵּ֕כֶל כָּל־עֲבֹדַ֕ת מִשְׁכַּ֖ן אֹ֣הֶל מוֹעֵ֑ד וַֽיַּעֲשׂוּ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל כְּ֠כֹל אֲשֶׁ֨ר צִוָּ֧ה יְהוָ֛ה אֶת־מֹשֶׁ֖ה כֵּ֥ן עָשֽׂוּ׃ {פ} וּשְׁלֵימַת--כָּל עֲבִידַת, מַשְׁכְּנָא מַשְׁכַּן זִמְנָא; וַעֲבַדוּ, בְּנֵי יִשְׂרָאֵל--כְּכֹל דְּפַקֵּיד יְיָ יָת מֹשֶׁה, כֵּין עֲבַדוּ. {פ}
לט,לג וַיָּבִ֤יאוּ אֶת־הַמִּשְׁכָּן֙ אֶל־מֹשֶׁ֔ה אֶת־הָאֹ֖הֶל וְאֶת־כָּל־כֵּלָ֑יו קְרָסָ֣יו קְרָשָׁ֔יו בְּרִיחָ֖ו וְעַמֻּדָ֥יו וַֽאֲדָנָֽיו׃ וְאֵיתִיאוּ יָת מַשְׁכְּנָא לְוָת מֹשֶׁה, יָת מַשְׁכְּנָא וְיָת כָּל מָנוֹהִי, פֻּרְפוֹהִי דַּפּוֹהִי, עָבְרוֹהִי וְעַמּוּדוֹהִי וְסָמְכוֹהִי. 
לט,לד וְאֶת־מִכְסֵ֞ה עוֹרֹ֤ת הָֽאֵילִם֙ הַמְאָדָּמִ֔ים וְאֶת־מִכְסֵ֖ה עֹרֹ֣ת הַתְּחָשִׁ֑ים וְאֵ֖ת פָּרֹ֥כֶת הַמָּסָֽךְ׃ וְיָת חוּפָאָה דְּמַשְׁכֵּי דִּכְרֵי, מְסֻמְּקֵי, וְיָת חוּפָאָה, דְּמַשְׁכֵּי סָסְגוֹנָא; וְיָת, פָּרֻכְתָּא דִּפְרָסָא. 
לט,לה אֶת־אֲר֥וֹן הָֽעֵדֻ֖ת וְאֶת־בַּדָּ֑יו וְאֵ֖ת הַכַּפֹּֽרֶת׃ יָת אֲרוֹנָא דְּסָהֲדוּתָא, וְיָת אֲרִיחוֹהִי; וְיָת, כָּפֻרְתָּא. 
לט,לו אֶת־הַשֻּׁלְחָן֙ אֶת־כָּל־כֵּלָ֔יו וְאֵ֖ת לֶ֥חֶם הַפָּנִֽים׃ יָת פָּתוּרָא, יָת כָּל מָנוֹהִי, וְיָת, לְחֵים אַפַּיָּא. 
לט,לז אֶת־הַמְּנֹרָ֨ה הַטְּהֹרָ֜ה אֶת־נֵֽרֹתֶ֗יהָ נֵרֹ֛ת הַמַּֽעֲרָכָ֖ה וְאֶת־כָּל־כֵּלֶ֑יהָ וְאֵ֖ת שֶׁ֥מֶן הַמָּאֽוֹר׃ יָת מְנָרְתָא דָּכִיתָא יָת בּוֹצִינַהָא, בּוֹצִינֵי סִדְרָא--וְיָת כָּל מָנַהָא; וְיָת מִשְׁחָא, דְּאַנְהָרוּתָא. 
לט,לח וְאֵת֙ מִזְבַּ֣ח הַזָּהָ֔ב וְאֵת֙ שֶׁ֣מֶן הַמִּשְׁחָ֔ה וְאֵ֖ת קְטֹ֣רֶת הַסַּמִּ֑ים וְאֵ֕ת מָסַ֖ךְ פֶּ֥תַח הָאֹֽהֶל׃ וְיָת, מַדְבְּחָא דְּדַהְבָּא, וְיָת מִשְׁחָא דִּרְבוּתָא, וְיָת קְטֹרֶת בֻּסְמַיָּא; וְיָת, פְּרָסָא דִּתְרַע מַשְׁכְּנָא. 
לט,לט אֵ֣ת ׀ מִזְבַּ֣ח הַנְּחֹ֗שֶׁת וְאֶת־מִכְבַּ֤ר הַנְּחֹ֨שֶׁת֙ אֲשֶׁר־ל֔וֹ אֶת־בַּדָּ֖יו וְאֶת־כָּל־כֵּלָ֑יו אֶת־הַכִּיֹּ֖ר וְאֶת־כַּנּֽוֹ׃ יָת מַדְבְּחָא דִּנְחָשָׁא, וְיָת סְרָדָא דִּנְחָשָׁא דִּילֵיהּ, יָת אֲרִיחוֹהִי, וְיָת כָּל מָנוֹהִי; יָת כִּיּוֹרָא, וְיָת בְּסִיסֵיהּ. 
לט,מ אֵת֩ קַלְעֵ֨י הֶֽחָצֵ֜ר אֶת־עַמֻּדֶ֣יהָ וְאֶת־אֲדָנֶ֗יהָ וְאֶת־הַמָּסָךְ֙ לְשַׁ֣עַר הֶֽחָצֵ֔ר אֶת־מֵֽיתָרָ֖יו וִיתֵֽדֹתֶ֑יהָ וְאֵ֗ת כָּל־כְּלֵ֛י עֲבֹדַ֥ת הַמִּשְׁכָּ֖ן לְאֹ֥הֶל מוֹעֵֽד׃ יָת סְרָדֵי דָּרְתָא יָת עַמּוּדַהָא וְיָת סָמְכַהָא, וְיָת פְּרָסָא לִתְרַע דָּרְתָא, יָת אַטּוּנוֹהִי, וְסִכַּהָא; וְיָת, כָּל מָנֵי פֻּלְחַן מַשְׁכְּנָא--לְמַשְׁכַּן זִמְנָא. 
לט,מא אֶת־בִּגְדֵ֥י הַשְּׂרָ֖ד לְשָׁרֵ֣ת בַּקֹּ֑דֶשׁ אֶת־בִּגְדֵ֤י הַקֹּ֨דֶשׁ֙ לְאַֽהֲרֹ֣ן הַכֹּהֵ֔ן וְאֶת־בִּגְדֵ֥י בָנָ֖יו לְכַהֵֽן׃ יָת לְבוּשֵׁי שִׁמּוּשָׁא, לְשַׁמָּשָׁא בְּקֻדְשָׁא; יָת לְבוּשֵׁי קֻדְשָׁא לְאַהֲרוֹן כָּהֲנָא, וְיָת לְבוּשֵׁי בְּנוֹהִי לְשַׁמָּשָׁא. 
לט,מב כְּכֹ֛ל אֲשֶׁר־צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶׁ֑ה כֵּ֤ן עָשׂוּ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל אֵ֖ת כָּל־הָֽעֲבֹדָֽה׃ כְּכֹל דְּפַקֵּיד יְיָ, יָת מֹשֶׁה:  כֵּין עֲבַדוּ בְּנֵי יִשְׂרָאֵל, יָת כָּל פֻּלְחָנָא. 
לט,מג וַיַּ֨רְא מֹשֶׁ֜ה אֶת־כָּל־הַמְּלָאכָ֗ה וְהִנֵּה֙ עָשׂ֣וּ אֹתָ֔הּ כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה כֵּ֣ן עָשׂ֑וּ וַיְבָ֥רֶךְ אֹתָ֖ם מֹשֶֽׁה׃ {פ} וַחֲזָא מֹשֶׁה יָת כָּל עֲבִידְתָא, וְהָא עֲבַדוּ יָתַהּ--כְּמָא דְּפַקֵּיד יְיָ, כֵּין עֲבַדוּ; וּבָרֵיךְ יָתְהוֹן, מֹשֶׁה. {פ}
מ,א וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
מ,ב בְּיוֹם־הַחֹ֥דֶשׁ הָֽרִאשׁ֖וֹן בְּאֶחָ֣ד לַחֹ֑דֶשׁ תָּקִ֕ים אֶת־מִשְׁכַּ֖ן אֹ֥הֶל מוֹעֵֽד׃ בְּיוֹם יַרְחָא קַדְמָאָה, בְּחַד לְיַרְחָא, תְּקִים, יָת מַשְׁכְּנָא מַשְׁכַּן זִמְנָא. 
מ,ג וְשַׂמְתָּ֣ שָׁ֔ם אֵ֖ת אֲר֣וֹן הָֽעֵד֑וּת וְסַכֹּתָ֥ עַל־הָֽאָרֹ֖ן אֶת־הַפָּרֹֽכֶת׃ וּתְשַׁוֵּי תַּמָּן, יָת אֲרוֹנָא דְּסָהֲדוּתָא; וְתַטֵּיל עַל אֲרוֹנָא, יָת פָּרֻכְתָּא. 
מ,ד וְהֵֽבֵאתָ֙ אֶת־הַשֻּׁלְחָ֔ן וְעָֽרַכְתָּ֖ אֶת־עֶרְכּ֑וֹ וְהֵֽבֵאתָ֙ אֶת־הַמְּנֹרָ֔ה וְהַֽעֲלֵיתָ֖ אֶת־נֵֽרֹתֶֽיהָ׃ וְתַעֵיל, יָת פָּתוּרָא, וְתַסְדַּר, יָת סִדְרֵיהּ; וְתַעֵיל, יָת מְנָרְתָא, וְתַדְלֵיק, יָת בּוֹצִינַהָא. 
מ,ה וְנָֽתַתָּ֞ה אֶת־מִזְבַּ֤ח הַזָּהָב֙ לִקְטֹ֔רֶת לִפְנֵ֖י אֲר֣וֹן הָֽעֵדֻ֑ת וְשַׂמְתָּ֛ אֶת־מָסַ֥ךְ הַפֶּ֖תַח לַמִּשְׁכָּֽן׃ וְתִתֵּין יָת מַדְבְּחָא דְּדַהְבָּא, לִקְטֹרֶת בֻּסְמַיָּא, קֳדָם, אֲרוֹנָא דְּסָהֲדוּתָא; וּתְשַׁוֵּי יָת פְּרָסָא דְּתַרְעָא, לְמַשְׁכְּנָא. 
מ,ו וְנָ֣תַתָּ֔ה אֵ֖ת מִזְבַּ֣ח הָֽעֹלָ֑ה לִפְנֵ֕י פֶּ֖תַח מִשְׁכַּ֥ן אֹֽהֶל־מוֹעֵֽד׃ וְתִתֵּין, יָת מַדְבְּחָא דַּעֲלָתָא, קֳדָם, תְּרַע מַשְׁכְּנָא מַשְׁכַּן זִמְנָא. 
מ,ז וְנָֽתַתָּ֙ אֶת־הַכִּיֹּ֔ר בֵּֽין־אֹ֥הֶל מוֹעֵ֖ד וּבֵ֣ין הַמִּזְבֵּ֑חַ וְנָֽתַתָּ֥ שָׁ֖ם מָֽיִם׃ וְתִתֵּין, יָת כִּיּוֹרָא, בֵּין מַשְׁכַּן זִמְנָא, וּבֵין מַדְבְּחָא; וְתִתֵּין תַּמָּן, מַיָּא. 
מ,ח וְשַׂמְתָּ֥ אֶת־הֶֽחָצֵ֖ר סָבִ֑יב וְנָ֣תַתָּ֔ אֶת־מָסַ֖ךְ שַׁ֥עַר הֶֽחָצֵֽר׃ וּתְשַׁוֵּי יָת דָּרְתָא, סְחוֹר סְחוֹר; וְתִתֵּין, יָת פְּרָסָא דִּתְרַע דָּרְתָא. 
מ,ט וְלָֽקַחְתָּ֙ אֶת־שֶׁ֣מֶן הַמִּשְׁחָ֔ה וּמָֽשַׁחְתָּ֥ אֶת־הַמִּשְׁכָּ֖ן וְאֶת־כָּל־אֲשֶׁר־בּ֑וֹ וְקִדַּשְׁתָּ֥ אֹת֛וֹ וְאֶת־כָּל־כֵּלָ֖יו וְהָ֥יָה קֹֽדֶשׁ׃ וְתִסַּב יָת מִשְׁחָא דִּרְבוּתָא, וּתְרַבֵּי יָת מַשְׁכְּנָא וְיָת כָּל דְּבֵיהּ; וּתְקַדֵּישׁ יָתֵיהּ וְיָת כָּל מָנוֹהִי, וִיהֵי קֻדְשָׁא. 
מ,י וּמָֽשַׁחְתָּ֛ אֶת־מִזְבַּ֥ח הָֽעֹלָ֖ה וְאֶת־כָּל־כֵּלָ֑יו וְקִדַּשְׁתָּ֙ אֶת־הַמִּזְבֵּ֔חַ וְהָיָ֥ה הַמִּזְבֵּ֖חַ קֹ֥דֶשׁ קָֽדָשִֽׁים׃ וּתְרַבֵּי יָת מַדְבְּחָא דַּעֲלָתָא, וְיָת כָּל מָנוֹהִי; וּתְקַדֵּישׁ, יָת מַדְבְּחָא, וִיהֵי מַדְבְּחָא, קֹדֶשׁ קֻדְשִׁין. 
מ,יא וּמָֽשַׁחְתָּ֥ אֶת־הַכִּיֹּ֖ר וְאֶת־כַּנּ֑וֹ וְקִדַּשְׁתָּ֖ אֹתֽוֹ׃ וּתְרַבֵּי יָת כִּיּוֹרָא, וְיָת בְּסִיסֵיהּ; וּתְקַדֵּישׁ, יָתֵיהּ. 
מ,יב וְהִקְרַבְתָּ֤ אֶֽת־אַהֲרֹן֙ וְאֶת־בָּנָ֔יו אֶל־פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד וְרָֽחַצְתָּ֥ אֹתָ֖ם בַּמָּֽיִם׃ וּתְקָרֵיב יָת אַהֲרוֹן וְיָת בְּנוֹהִי, לִתְרַע מַשְׁכַּן זִמְנָא; וְתַסְחֵי יָתְהוֹן, בְּמַיָּא. 
מ,יג וְהִלְבַּשְׁתָּ֙ אֶֽת־אַהֲרֹ֔ן אֵ֖ת בִּגְדֵ֣י הַקֹּ֑דֶשׁ וּמָֽשַׁחְתָּ֥ אֹת֛וֹ וְקִדַּשְׁתָּ֥ אֹת֖וֹ וְכִהֵ֥ן לִֽי׃ וְתַלְבֵּישׁ, יָת אַהֲרוֹן, יָת, לְבוּשֵׁי קֻדְשָׁא; וּתְרַבֵּי יָתֵיהּ וּתְקַדֵּישׁ יָתֵיהּ, וִישַׁמֵּישׁ קֳדָמָי. 
מ,יד וְאֶת־בָּנָ֖יו תַּקְרִ֑יב וְהִלְבַּשְׁתָּ֥ אֹתָ֖ם כֻּתֳּנֹֽת׃ וְיָת בְּנוֹהִי, תְּקָרֵיב; וְתַלְבֵּישׁ יָתְהוֹן, כִּתּוּנִין. 
מ,טו וּמָֽשַׁחְתָּ֣ אֹתָ֗ם כַּֽאֲשֶׁ֤ר מָשַׁ֨חְתָּ֙ אֶת־אֲבִיהֶ֔ם וְכִֽהֲנ֖וּ לִ֑י וְ֠הָֽיְתָה לִֽהְיֹ֨ת לָהֶ֧ם מָשְׁחָתָ֛ם לִכְהֻנַּ֥ת עוֹלָ֖ם לְדֹֽרֹתָֽם׃ וּתְרַבֵּי יָתְהוֹן, כְּמָא דְּרַבִּיתָא יָת אֲבוּהוֹן, וִישַׁמְּשׁוּן, קֳדָמָי; וּתְהֵי לְמִהְוֵי לְהוֹן רְבוּתְהוֹן, לִכְהֻנַּת עָלַם--לְדָרֵיהוֹן. 
מ,טז וַיַּ֖עַשׂ מֹשֶׁ֑ה כְּ֠כֹל אֲשֶׁ֨ר צִוָּ֧ה יְהוָ֛ה אֹת֖וֹ כֵּ֥ן עָשָֽׂה׃ {ס} וַעֲבַד, מֹשֶׁה:  כְּכֹל דְּפַקֵּיד יְיָ, יָתֵיהּ--כֵּין עֲבַד. {ס}
מ,יז וַיְהִ֞י בַּחֹ֧דֶשׁ הָֽרִאשׁ֛וֹן בַּשָּׁנָ֥ה הַשֵּׁנִ֖ית בְּאֶחָ֣ד לַחֹ֑דֶשׁ הוּקַ֖ם הַמִּשְׁכָּֽן׃ וַהֲוָה בְּיַרְחָא קַדְמָאָה, בְּשַׁתָּא תִּנְיֵיתָא--בְּחַד לְיַרְחָא:  אִתָּקַם, מַשְׁכְּנָא. 
מ,יח וַיָּ֨קֶם מֹשֶׁ֜ה אֶת־הַמִּשְׁכָּ֗ן וַיִּתֵּן֙ אֶת־אֲדָנָ֔יו וַיָּ֨שֶׂם֙ אֶת־קְרָשָׁ֔יו וַיִּתֵּ֖ן אֶת־בְּרִיחָ֑יו וַיָּ֖קֶם אֶת־עַמּוּדָֽיו׃ וַאֲקֵים מֹשֶׁה יָת מַשְׁכְּנָא, וִיהַב יָת סָמְכוֹהִי, וְשַׁוִּי יָת דַּפּוֹהִי, וִיהַב יָת עָבְרוֹהִי; וַאֲקֵים, יָת עַמּוּדוֹהִי. 
מ,יט וַיִּפְרֹ֤שׂ אֶת־הָאֹ֨הֶל֙ עַל־הַמִּשְׁכָּ֔ן וַיָּ֜שֶׂם אֶת־מִכְסֵ֥ה הָאֹ֛הֶל עָלָ֖יו מִלְמָ֑עְלָה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ {ס} וּפְרַס יָת פְּרָסָא, עַל מַשְׁכְּנָא, וְשַׁוִּי יָת חוּפָאָה דְּמַשְׁכְּנָא עֲלוֹהִי, מִלְּעֵילָא--כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. {ס}
מ,כ וַיִּקַּ֞ח וַיִּתֵּ֤ן אֶת־הָֽעֵדֻת֙ אֶל־הָ֣אָרֹ֔ן וַיָּ֥שֶׂם אֶת־הַבַּדִּ֖ים עַל־הָֽאָרֹ֑ן וַיִּתֵּ֧ן אֶת־הַכַּפֹּ֛רֶת עַל־הָֽאָרֹ֖ן מִלְמָֽעְלָה׃ וּנְסֵיב וִיהַב יָת סָהֲדוּתָא, בַּאֲרוֹנָא, וְשַׁוִּי יָת אֲרִיחַיָּא, עַל אֲרוֹנָא; וִיהַב יָת כָּפֻרְתָּא עַל אֲרוֹנָא, מִלְּעֵילָא. 
מ,כא וַיָּבֵ֣א אֶת־הָֽאָרֹן֮ אֶל־הַמִּשְׁכָּן֒ וַיָּ֗שֶׂם אֵ֚ת פָּרֹ֣כֶת הַמָּסָ֔ךְ וַיָּ֕סֶךְ עַ֖ל אֲר֣וֹן הָֽעֵד֑וּת כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ {ס} וְאַעֵיל יָת אֲרוֹנָא, לְמַשְׁכְּנָא, וְשַׁוִּי יָת פָּרֻכְתָּא דִּפְרָסָא, וְאַטֵּיל עַל אֲרוֹנָא דְּסָהֲדוּתָא--כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. {ס}
מ,כב וַיִּתֵּ֤ן אֶת־הַשֻּׁלְחָן֙ בְּאֹ֣הֶל מוֹעֵ֔ד עַ֛ל יֶ֥רֶךְ הַמִּשְׁכָּ֖ן צָפֹ֑נָה מִח֖וּץ לַפָּרֹֽכֶת׃ וִיהַב יָת פָּתוּרָא בְּמַשְׁכַּן זִמְנָא, עַל שִׁדָּא דְּמַשְׁכְּנָא צִפּוּנָא, מִבַּרָא, לְפָרֻכְתָּא. 
מ,כג וַיַּֽעֲרֹ֥ךְ עָלָ֛יו עֵ֥רֶךְ לֶ֖חֶם לִפְנֵ֣י יְהוָ֑ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ {ס} וְסַדַּר עֲלוֹהִי סִדְרִין דִּלְחֵים, קֳדָם יְיָ--כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. {ס}
מ,כד וַיָּ֤שֶׂם אֶת־הַמְּנֹרָה֙ בְּאֹ֣הֶל מוֹעֵ֔ד נֹ֖כַח הַשֻּׁלְחָ֑ן עַ֛ל יֶ֥רֶךְ הַמִּשְׁכָּ֖ן נֶֽגְבָּה׃ וְשַׁוִּי יָת מְנָרְתָא בְּמַשְׁכַּן זִמְנָא, לָקֳבֵיל פָּתוּרָא, עַל שִׁדָּא דְּמַשְׁכְּנָא, דָּרוֹמָא. 
מ,כה וַיַּ֥עַל הַנֵּרֹ֖ת לִפְנֵ֣י יְהוָ֑ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ {ס} וְאַדְלֵיק בּוֹצִינַיָּא, קֳדָם יְיָ--כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. {ס}
מ,כו וַיָּ֛שֶׂם אֶת־מִזְבַּ֥ח הַזָּהָ֖ב בְּאֹ֣הֶל מוֹעֵ֑ד לִפְנֵ֖י הַפָּרֹֽכֶת׃ וְשַׁוִּי יָת מַדְבְּחָא דְּדַהְבָּא, בְּמַשְׁכַּן זִמְנָא, קֳדָם, פָּרֻכְתָּא. 
מ,כז וַיַּקְטֵ֥ר עָלָ֖יו קְטֹ֣רֶת סַמִּ֑ים כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ {ס} וְאַקְטַר עֲלוֹהִי, קְטֹרֶת בֻּסְמִין--כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. {ס}
מ,כח וַיָּ֛שֶׂם אֶת־מָסַ֥ךְ הַפֶּ֖תַח לַמִּשְׁכָּֽן׃ וְשַׁוִּי יָת פְּרָסָא דְּתַרְעָא, לְמַשְׁכְּנָא. 
מ,כט וְאֵת֙ מִזְבַּ֣ח הָֽעֹלָ֔ה שָׂ֕ם פֶּ֖תַח מִשְׁכַּ֣ן אֹֽהֶל־מוֹעֵ֑ד וַיַּ֣עַל עָלָ֗יו אֶת־הָֽעֹלָה֙ וְאֶת־הַמִּנְחָ֔ה כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ {ס} וְיָת, מַדְבְּחָא דַּעֲלָתָא, שַׁוִּי, בִּתְרַע מַשְׁכְּנָא מַשְׁכַּן זִמְנָא; וְאַסֵּיק עֲלוֹהִי, יָת עֲלָתָא וְיָת מִנְחָתָא--כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. {ס}
מ,ל וַיָּ֨שֶׂם֙ אֶת־הַכִּיֹּ֔ר בֵּֽין־אֹ֥הֶל מוֹעֵ֖ד וּבֵ֣ין הַמִּזְבֵּ֑חַ וַיִּתֵּ֥ן שָׁ֛מָּה מַ֖יִם לְרָחְצָֽה׃ וְשַׁוִּי, יָת כִּיּוֹרָא, בֵּין מַשְׁכַּן זִמְנָא, וּבֵין מַדְבְּחָא; וִיהַב תַּמָּן מַיָּא, לְקִדּוּשׁ. 
מ,לא וְרָֽחֲצ֣וּ מִמֶּ֔נּוּ מֹשֶׁ֖ה וְאַֽהֲרֹ֣ן וּבָנָ֑יו אֶת־יְדֵיהֶ֖ם וְאֶת־רַגְלֵיהֶֽם׃ וּמְקַדְּשִׁין מִנֵּיהּ, מֹשֶׁה וְאַהֲרוֹן וּבְנוֹהִי, יָת יְדֵיהוֹן, וְיָת רַגְלֵיהוֹן. 
מ,לב בְּבֹאָ֞ם אֶל־אֹ֣הֶל מוֹעֵ֗ד וּבְקָרְבָתָ֛ם אֶל־הַמִּזְבֵּ֖חַ יִרְחָ֑צוּ כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ {ס} בְּמֵיעַלְהוֹן לְמַשְׁכַּן זִמְנָא, וּבְמִקְרַבְהוֹן לְמַדְבְּחָא--מְקַדְּשִׁין:  כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. {ס}
מ,לג וַיָּ֣קֶם אֶת־הֶֽחָצֵ֗ר סָבִיב֙ לַמִּשְׁכָּ֣ן וְלַמִּזְבֵּ֔חַ וַיִּתֵּ֕ן אֶת־מָסַ֖ךְ שַׁ֣עַר הֶֽחָצֵ֑ר וַיְכַ֥ל מֹשֶׁ֖ה אֶת־הַמְּלָאכָֽה׃ {פ} וַאֲקֵים יָת דָּרְתָא, סְחוֹר סְחוֹר לְמַשְׁכְּנָא וּלְמַדְבְּחָא, וִיהַב, יָת פְּרָסָא דִּתְרַע דָּרְתָא; וְשֵׁיצִי מֹשֶׁה, יָת עֲבִידְתָא. {פ}
מ,לד וַיְכַ֥ס הֶֽעָנָ֖ן אֶת־אֹ֣הֶל מוֹעֵ֑ד וּכְב֣וֹד יְהוָ֔ה מָלֵ֖א אֶת־הַמִּשְׁכָּֽן׃ וַחֲפָא עֲנָנָא, יָת מַשְׁכַּן זִמְנָא; וִיקָרָא דַּייָ, אִתְמְלִי יָת מַשְׁכְּנָא. 
מ,לה וְלֹֽא־יָכֹ֣ל מֹשֶׁ֗ה לָבוֹא֙ אֶל־אֹ֣הֶל מוֹעֵ֔ד כִּֽי־שָׁכַ֥ן עָלָ֖יו הֶֽעָנָ֑ן וּכְב֣וֹד יְהוָ֔ה מָלֵ֖א אֶת־הַמִּשְׁכָּֽן׃ וְלָא יָכֵיל מֹשֶׁה, לְמֵיעַל לְמַשְׁכַּן זִמְנָא--אֲרֵי שְׁרָא עֲלוֹהִי, עֲנָנָא; וִיקָרָא דַּייָ, אִתְמְלִי יָת מַשְׁכְּנָא. 
מ,לו וּבְהֵֽעָל֤וֹת הֶֽעָנָן֙ מֵעַ֣ל הַמִּשְׁכָּ֔ן יִסְע֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל בְּכֹ֖ל מַסְעֵיהֶֽם׃ וּבְאִסְתַּלָּקוּת עֲנָנָא מֵעִלָּוֵי מַשְׁכְּנָא, נָטְלִין בְּנֵי יִשְׂרָאֵל, בְּכֹל, מַטְּלָנֵיהוֹן. 
מ,לז וְאִם־לֹ֥א יֵֽעָלֶ֖ה הֶֽעָנָ֑ן וְלֹ֣א יִסְע֔וּ עַד־י֖וֹם הֵעָֽלֹתֽוֹ׃ וְאִם לָא מִסְתַּלַּק, עֲנָנָא--וְלָא נָטְלִין, עַד יוֹם אִסְתַּלָּקוּתֵיהּ. 
מ,לח כִּי֩ עֲנַ֨ן יְהוָ֤ה עַֽל־הַמִּשְׁכָּן֙ יוֹמָ֔ם וְאֵ֕שׁ תִּֽהְיֶ֥ה לַ֖יְלָה בּ֑וֹ לְעֵינֵ֥י כָל־בֵּֽית־יִשְׂרָאֵ֖ל בְּכָל־מַסְעֵיהֶֽם׃ {ש} אֲרֵי עֲנַן יְקָרָא דַּייָ עַל מַשְׁכְּנָא, בִּימָמָא, וְחֵיזוּ אִישָׁתָא, הָוֵי בְּלֵילְיָא בֵּיהּ--לְעֵינֵי כָּל בֵּית יִשְׂרָאֵל, בְּכָל מַטְּלָנֵיהוֹן. {ש}
