מסעי

לג,א אֵ֜לֶּה מַסְעֵ֣י בְנֵֽי־יִשְׂרָאֵ֗ל אֲשֶׁ֥ר יָֽצְא֛וּ מֵאֶ֥רֶץ מִצְרַ֖יִם לְצִבְאֹתָ֑ם בְּיַד־מֹשֶׁ֖ה וְאַֽהֲרֹֽן׃ אִלֵּין מַטְּלָנֵי בְּנֵי יִשְׂרָאֵל, דִּנְפַקוּ מֵאַרְעָא דְּמִצְרַיִם--לְחֵילֵיהוֹן:  בִּידָא דְּמֹשֶׁה, וְאַהֲרוֹן. 
לג,ב וַיִּכְתֹּ֨ב מֹשֶׁ֜ה אֶת־מוֹצָֽאֵיהֶ֛ם לְמַסְעֵיהֶ֖ם עַל־פִּ֣י יְהוָ֑ה וְאֵ֥לֶּה מַסְעֵיהֶ֖ם לְמוֹצָֽאֵיהֶֽם׃ וּכְתַב מֹשֶׁה יָת מַפְּקָנֵיהוֹן, לְמַטְּלָנֵיהוֹן--עַל מֵימְרָא דַּייָ; וְאִלֵּין מַטְּלָנֵיהוֹן, לְמַפְּקָנֵיהוֹן. 
לג,ג וַיִּסְע֤וּ מֵֽרַעְמְסֵס֙ בַּחֹ֣דֶשׁ הָֽרִאשׁ֔וֹן בַּֽחֲמִשָּׁ֥ה עָשָׂ֛ר י֖וֹם לַחֹ֣דֶשׁ הָֽרִאשׁ֑וֹן מִֽמָּחֳרַ֣ת הַפֶּ֗סַח יָֽצְא֤וּ בְנֵֽי־יִשְׂרָאֵל֙ בְּיָ֣ד רָמָ֔ה לְעֵינֵ֖י כָּל־מִצְרָֽיִם׃ וּנְטַלוּ מֵרַעְמְסֵס בְּיַרְחָא קַדְמָאָה, בַּחֲמֵישְׁתְּ עַסְרָא יוֹמָא לְיַרְחָא קַדְמָאָה:  מִבָּתַר פִּסְחָא, נְפַקוּ בְּנֵי יִשְׂרָאֵל בְּרֵישׁ גְּלֵי--לְעֵינֵי, כָּל מִצְרָאֵי. 
לג,ד וּמִצְרַ֣יִם מְקַבְּרִ֗ים אֵת֩ אֲשֶׁ֨ר הִכָּ֧ה יְהוָ֛ה בָּהֶ֖ם כָּל־בְּכ֑וֹר וּבֵאלֹ֣הֵיהֶ֔ם עָשָׂ֥ה יְהוָ֖ה שְׁפָטִֽים׃ וּמִצְרָאֵי מְקַבְּרִין, יָת דִּקְטַל יְיָ בְּהוֹן--כָּל בֻּכְרָא; וּבְטָעֲוָתְהוֹן, עֲבַד יְיָ דִּינִין. 
לג,ה וַיִּסְע֥וּ בְנֵֽי־יִשְׂרָאֵ֖ל מֵֽרַעְמְסֵ֑ס וַֽיַּחֲנ֖וּ בְּסֻכֹּֽת׃ וּנְטַלוּ בְּנֵי יִשְׂרָאֵל, מֵרַעְמְסֵס; וּשְׁרוֹ, בְּסֻכּוֹת. 
לג,ו וַיִּסְע֖וּ מִסֻּכֹּ֑ת וַיַּֽחֲנ֣וּ בְאֵתָ֔ם אֲשֶׁ֖ר בִּקְצֵ֥ה הַמִּדְבָּֽר׃ וּנְטַלוּ, מִסֻּכּוֹת; וּשְׁרוֹ בְּאֵיתָם, דְּבִסְטַר מַדְבְּרָא. 
לג,ז וַיִּסְעוּ֙ מֵֽאֵתָ֔ם וַיָּ֨שָׁב֙ עַל־פִּ֣י הַֽחִירֹ֔ת אֲשֶׁ֥ר עַל־פְּנֵ֖י בַּ֣עַל צְפ֑וֹן וַֽיַּחֲנ֖וּ לִפְנֵ֥י מִגְדֹּֽל׃ וּנְטַלוּ, מֵאֵיתָם, וְתָב עַל פֹּם חִירָתָא, דִּקְדָם בְּעֵיל צְפוֹן; וּשְׁרוֹ, קֳדָם מִגְדּוֹל. 
לג,ח וַיִּסְעוּ֙ מִפְּנֵ֣י הַֽחִירֹ֔ת וַיַּֽעַבְר֥וּ בְתוֹךְ־הַיָּ֖ם הַמִּדְבָּ֑רָה וַיֵּ֨לְכ֜וּ דֶּ֣רֶךְ שְׁלֹ֤שֶׁת יָמִים֙ בְּמִדְבַּ֣ר אֵתָ֔ם וַֽיַּחֲנ֖וּ בְּמָרָֽה׃ וּנְטַלוּ מִן קֳדָם חִירָתָא, וַעֲבַרוּ בְּגוֹ יַמָּא לְמַדְבְּרָא; וַאֲזַלוּ מַהְלַךְ תְּלָתָא יוֹמִין, בְּמַדְבְּרָא דְּאֵיתָם, וּשְׁרוֹ, בְּמָרָה. 
לג,ט וַיִּסְעוּ֙ מִמָּרָ֔ה וַיָּבֹ֖אוּ אֵילִ֑מָה וּ֠בְאֵילִם שְׁתֵּ֣ים עֶשְׂרֵ֞ה עֵינֹ֥ת מַ֛יִם וְשִׁבְעִ֥ים תְּמָרִ֖ים וַיַּֽחֲנוּ־שָֽׁם׃ וּנְטַלוּ, מִמָּרָה, וַאֲתוֹ, לְאֵילִים; וּבְאֵילִים תְּרֵי עֲסַר מַבּוּעִין דְּמַיִין, וְשִׁבְעִין דִּקְלִין--וּשְׁרוֹ תַּמָּן. 
לג,י וַיִּסְע֖וּ מֵֽאֵילִ֑ם וַֽיַּחֲנ֖וּ עַל־יַם־סֽוּף׃ וּנְטַלוּ, מֵאֵילִים; וּשְׁרוֹ, עַל יַמָּא דְּסוּף. 
לג,יא וַיִּסְע֖וּ מִיַּם־ס֑וּף וַֽיַּחֲנ֖וּ בְּמִדְבַּר־סִֽין׃ וּנְטַלוּ, מִיַּמָּא דְּסוּף; וּשְׁרוֹ, בְּמַדְבְּרָא דְּסִין. 
לג,יב וַיִּסְע֖וּ מִמִּדְבַּר־סִ֑ין וַֽיַּחֲנ֖וּ בְּדָפְקָֽה׃ וּנְטַלוּ, מִמַּדְבְּרָא דְּסִין; וּשְׁרוֹ, בְּדָפְקָה. 
לג,יג וַיִּסְע֖וּ מִדָּפְקָ֑ה וַֽיַּחֲנ֖וּ בְּאָלֽוּשׁ׃ וּנְטַלוּ, מִדָּפְקָה; וּשְׁרוֹ, בְּאָלוּשׁ. 
לג,יד וַיִּסְע֖וּ מֵֽאָל֑וּשׁ וַֽיַּחֲנוּ֙ בִּרְפִידִ֔ם וְלֹא־הָ֨יָה שָׁ֥ם מַ֛יִם לָעָ֖ם לִשְׁתּֽוֹת׃ וּנְטַלוּ, מֵאָלוּשׁ; וּשְׁרוֹ, בִּרְפִידִים, וְלָא הֲוָה תַּמָּן מַיָּא לְעַמָּא, לְמִשְׁתֵּי. 
לג,טו וַיִּסְע֖וּ מֵֽרְפִידִ֑ם וַֽיַּחֲנ֖וּ בְּמִדְבַּ֥ר סִינָֽי׃ וּנְטַלוּ, מֵרְפִידִים; וּשְׁרוֹ, בְּמַדְבְּרָא דְּסִינָי. 
לג,טז וַיִּסְע֖וּ מִמִּדְבַּ֣ר סִינָ֑י וַֽיַּחֲנ֖וּ בְּקִבְרֹ֥ת הַֽתַּאֲוָֽה׃ וּנְטַלוּ, מִמַּדְבְּרָא דְּסִינָי; וּשְׁרוֹ, בְּקִבְרֵי דִּמְשַׁאֲלֵי. 
לג,יז וַיִּסְע֖וּ מִקִּבְרֹ֣ת הַֽתַּאֲוָ֑ה וַֽיַּחֲנ֖וּ בַּֽחֲצֵרֹֽת׃ וּנְטַלוּ, מִקִּבְרֵי דִּמְשַׁאֲלֵי; וּשְׁרוֹ, בַּחֲצֵרוֹת. 
לג,יח וַיִּסְע֖וּ מֵֽחֲצֵרֹ֑ת וַֽיַּחֲנ֖וּ בְּרִתְמָֽה׃ וּנְטַלוּ, מֵחֲצֵרוֹת; וּשְׁרוֹ, בְּרִתְמָה. 
לג,יט וַיִּסְע֖וּ מֵֽרִתְמָ֑ה וַֽיַּחֲנ֖וּ בְּרִמֹּ֥ן פָּֽרֶץ׃ וּנְטַלוּ, מֵרִתְמָה; וּשְׁרוֹ, בְּרִמּוֹן פָּרֶץ. 
לג,כ וַיִּסְע֖וּ מֵֽרִמֹּ֣ן פָּ֑רֶץ וַֽיַּחֲנ֖וּ בְּלִבְנָֽה׃ וּנְטַלוּ, מֵרִמּוֹן פָּרֶץ; וּשְׁרוֹ, בְּלִבְנָה. 
לג,כא וַיִּסְע֖וּ מִלִּבְנָ֑ה וַֽיַּחֲנ֖וּ בְּרִסָּֽה׃ וּנְטַלוּ, מִלִּבְנָה; וּשְׁרוֹ, בְּרִסָּה. 
לג,כב וַיִּסְע֖וּ מֵֽרִסָּ֑ה וַֽיַּחֲנ֖וּ בִּקְהֵלָֽתָה׃ וּנְטַלוּ, מֵרִסָּה; וּשְׁרוֹ, בִּקְהֵלָתָה. 
לג,כג וַיִּסְע֖וּ מִקְּהֵלָ֑תָה וַֽיַּחֲנ֖וּ בְּהַר־שָֽׁפֶר׃ וּנְטַלוּ, מִקְּהֵלָתָה; וּשְׁרוֹ, בְּטוּרָא דְּשָׁפֶר. 
לג,כד וַיִּסְע֖וּ מֵֽהַר־שָׁ֑פֶר וַֽיַּחֲנ֖וּ בַּֽחֲרָדָֽה׃ וּנְטַלוּ, מִטּוּרָא דְּשָׁפֶר; וּשְׁרוֹ, בַּחֲרָדָה. 
לג,כה וַיִּסְע֖וּ מֵֽחֲרָדָ֑ה וַֽיַּחֲנ֖וּ בְּמַקְהֵלֹֽת׃ וּנְטַלוּ, מֵחֲרָדָה; וּשְׁרוֹ, בְּמַקְהֵלוֹת. 
לג,כו וַיִּסְע֖וּ מִמַּקְהֵלֹ֑ת וַֽיַּחֲנ֖וּ בְּתָֽחַת׃ וּנְטַלוּ, מִמַּקְהֵלוֹת; וּשְׁרוֹ, בְּתָחַת. 
לג,כז וַיִּסְע֖וּ מִתָּ֑חַת וַֽיַּחֲנ֖וּ בְּתָֽרַח׃ וּנְטַלוּ, מִתָּחַת; וּשְׁרוֹ, בְּתָרַח. 
לג,כח וַיִּסְע֖וּ מִתָּ֑רַח וַֽיַּחֲנ֖וּ בְּמִתְקָֽה׃ וּנְטַלוּ, מִתָּרַח; וּשְׁרוֹ, בְּמִתְקָה. 
לג,כט וַיִּסְע֖וּ מִמִּתְקָ֑ה וַֽיַּחֲנ֖וּ בְּחַשְׁמֹנָֽה׃ וּנְטַלוּ, מִמִּתְקָה; וּשְׁרוֹ, בְּחַשְׁמוֹנָה. 
לג,ל וַיִּסְע֖וּ מֵֽחַשְׁמֹנָ֑ה וַֽיַּחֲנ֖וּ בְּמֹֽסֵרֽוֹת׃ וּנְטַלוּ, מֵחַשְׁמוֹנָה; וּשְׁרוֹ, בְּמוֹסֵרוֹת. 
לג,לא וַיִּסְע֖וּ מִמֹּֽסֵר֑וֹת וַֽיַּחֲנ֖וּ בִּבְנֵ֥י יַֽעֲקָֽן׃ וּנְטַלוּ, מִמּוֹסֵרוֹת; וּשְׁרוֹ, בִּבְנֵי יַעֲקָן. 
לג,לב וַיִּסְע֖וּ מִבְּנֵ֣י יַֽעֲקָ֑ן וַֽיַּחֲנ֖וּ בְּחֹ֥ר הַגִּדְגָּֽד׃ וּנְטַלוּ, מִבְּנֵי יַעֲקָן; וּשְׁרוֹ, בְּחוֹר גִּדְגָּד. 
לג,לג וַיִּסְע֖וּ מֵחֹ֣ר הַגִּדְגָּ֑ד וַֽיַּחֲנ֖וּ בְּיָטְבָֽתָה׃ וּנְטַלוּ, מֵחוֹר גִּדְגָד; וּשְׁרוֹ, בְּיָטְבָתָה. 
לג,לד וַיִּסְע֖וּ מִיָּטְבָ֑תָה וַֽיַּחֲנ֖וּ בְּעַבְרֹנָֽה׃ וּנְטַלוּ, מִיָּטְבָתָה; וּשְׁרוֹ, בְּעַבְרוֹנָה. 
לג,לה וַיִּסְע֖וּ מֵֽעַבְרֹנָ֑ה וַֽיַּחֲנ֖וּ בְּעֶצְיֹ֥ן גָּֽבֶר׃ וּנְטַלוּ, מֵעַבְרוֹנָה; וּשְׁרוֹ, בְּעֶצְיוֹן גָּבֶר. 
לג,לו וַיִּסְע֖וּ מֵֽעֶצְיֹ֣ן גָּ֑בֶר וַיַּֽחֲנ֥וּ בְמִדְבַּר־צִ֖ן הִ֥וא קָדֵֽשׁ׃ וּנְטַלוּ, מֵעֶצְיוֹן גָּבֶר; וּשְׁרוֹ בְּמַדְבְּרָא דְּצִין, הִיא רְקַם. 
לג,לז וַיִּסְע֖וּ מִקָּדֵ֑שׁ וַֽיַּחֲנוּ֙ בְּהֹ֣ר הָהָ֔ר בִּקְצֵ֖ה אֶ֥רֶץ אֱדֽוֹם׃ וּנְטַלוּ, מֵרְקַם; וּשְׁרוֹ בְּהוֹר טוּרָא, בִּסְיָפֵי אַרְעָא דֶּאֱדוֹם. 
לג,לח וַיַּעַל֩ אַֽהֲרֹ֨ן הַכֹּהֵ֜ן אֶל־הֹ֥ר הָהָ֛ר עַל־פִּ֥י יְהוָ֖ה וַיָּ֣מָת שָׁ֑ם בִּשְׁנַ֣ת הָֽאַרְבָּעִ֗ים לְצֵ֤את בְּנֵֽי־יִשְׂרָאֵל֙ מֵאֶ֣רֶץ מִצְרַ֔יִם בַּחֹ֥דֶשׁ הַֽחֲמִישִׁ֖י בְּאֶחָ֥ד לַחֹֽדֶשׁ׃ וּסְלֵיק אַהֲרוֹן כָּהֲנָא לְהוֹר טוּרָא, עַל מֵימְרָא דַּייָ--וּמִית תַּמָּן:  בִּשְׁנַת אַרְבְּעִין, לְמִפַּק בְּנֵי יִשְׂרָאֵל מֵאַרְעָא דְּמִצְרַיִם, בְּיַרְחָא חֲמִישָׁאָה, בְּחַד לְיַרְחָא. 
לג,לט וְאַֽהֲרֹ֔ן בֶּן־שָׁלֹ֧שׁ וְעֶשְׂרִ֛ים וּמְאַ֖ת שָׁנָ֑ה בְּמֹת֖וֹ בְּהֹ֥ר הָהָֽר׃ {ס} וְאַהֲרוֹן, בַּר מְאָה וְעַסְרִין וּתְלָת שְׁנִין, כַּד מִית, בְּהוֹר טוּרָא. {ס}
לג,מ וַיִּשְׁמַ֗ע הַֽכְּנַעֲנִי֙ מֶ֣לֶךְ עֲרָ֔ד וְהֽוּא־יֹשֵׁ֥ב בַּנֶּ֖גֶב בְּאֶ֣רֶץ כְּנָ֑עַן בְּבֹ֖א בְּנֵ֥י יִשְׂרָאֵֽל׃ וּשְׁמַע, כְּנַעֲנָאָה מַלְכָּא דַּעֲרָד, וְהוּא יָתֵיב בְּדָרוֹמָא, בְּאַרְעָא דִּכְנָעַן--בְּמֵיתֵי, בְּנֵי יִשְׂרָאֵל. 
לג,מא וַיִּסְע֖וּ מֵהֹ֣ר הָהָ֑ר וַֽיַּחֲנ֖וּ בְּצַלְמֹנָֽה׃ וּנְטַלוּ, מֵהוֹר טוּרָא; וּשְׁרוֹ, בְּצַלְמוֹנָה. 
לג,מב וַיִּסְע֖וּ מִצַּלְמֹנָ֑ה וַֽיַּחֲנ֖וּ בְּפוּנֹֽן׃ וּנְטַלוּ, מִצַּלְמוֹנָה; וּשְׁרוֹ, בְּפוּנוֹן. 
לג,מג וַיִּסְע֖וּ מִפּוּנֹ֑ן וַֽיַּחֲנ֖וּ בְּאֹבֹֽת׃ וּנְטַלוּ, מִפּוּנוֹן; וּשְׁרוֹ, בְּאוֹבוֹת. 
לג,מד וַיִּסְע֖וּ מֵֽאֹבֹ֑ת וַֽיַּחֲנ֛וּ בְּעִיֵּ֥י הָֽעֲבָרִ֖ים בִּגְב֥וּל מוֹאָֽב׃ וּנְטַלוּ, מֵאוֹבוֹת; וּשְׁרוֹ בִּמְגָזַת עִבְרָאֵי, בִּתְחוּם מוֹאָב. 
לג,מה וַיִּסְע֖וּ מֵֽעִיִּ֑ים וַֽיַּחֲנ֖וּ בְּדִיבֹ֥ן גָּֽד׃ וּנְטַלוּ, מִמְּגָזָתָא; וּשְׁרוֹ, בְּדִיבוֹן גָּד. 
לג,מו וַיִּסְע֖וּ מִדִּיבֹ֣ן גָּ֑ד וַֽיַּחֲנ֖וּ בְּעַלְמֹ֥ן דִּבְלָתָֽיְמָה׃ וּנְטַלוּ, מִדִּיבוֹן גָּד; וּשְׁרוֹ, בְּעַלְמוֹן דִּבְלָתָיְמָה. 
לג,מז וַיִּסְע֖וּ מֵֽעַלְמֹ֣ן דִּבְלָתָ֑יְמָה וַֽיַּחֲנ֛וּ בְּהָרֵ֥י הָֽעֲבָרִ֖ים לִפְנֵ֥י נְבֽוֹ׃ וּנְטַלוּ, מֵעַלְמוֹן דִּבְלָתָיְמָה; וּשְׁרוֹ בְּטוּרֵי דְּעִבְרָאֵי, דִּקְדָם נְבוֹ. 
לג,מח וַיִּסְע֖וּ מֵֽהָרֵ֣י הָֽעֲבָרִ֑ים וַֽיַּחֲנוּ֙ בְּעַֽרְבֹ֣ת מוֹאָ֔ב עַ֖ל יַרְדֵּ֥ן יְרֵחֽוֹ׃ וּנְטַלוּ, מִטּוּרֵי דְּעִבְרָאֵי; וּשְׁרוֹ בְּמֵישְׁרַיָּא דְּמוֹאָב, עַל יַרְדְּנָא דִּירִיחוֹ. 
לג,מט וַיַּֽחֲנ֤וּ עַל־הַיַּרְדֵּן֙ מִבֵּ֣ית הַיְשִׁמֹ֔ת עַ֖ד אָבֵ֣ל הַשִּׁטִּ֑ים בְּעַֽרְבֹ֖ת מוֹאָֽב׃ {ס} וּשְׁרוֹ עַל יַרְדְּנָא מִבֵּית יְשִׁימוֹת, עַד מֵישַׁר שִׁטִּין, בְּמֵישְׁרַיָּא, דְּמוֹאָב. {ס}
לג,נ וַיְדַבֵּ֧ר יְהוָ֛ה אֶל־מֹשֶׁ֖ה בְּעַֽרְבֹ֣ת מוֹאָ֑ב עַל־יַרְדֵּ֥ן יְרֵח֖וֹ לֵאמֹֽר׃ וּמַלֵּיל יְיָ עִם מֹשֶׁה, בְּמֵישְׁרַיָּא דְּמוֹאָב, עַל יַרְדְּנָא דִּירִיחוֹ, לְמֵימַר. 
לג,נא דַּבֵּר֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָֽמַרְתָּ֖ אֲלֵהֶ֑ם כִּ֥י אַתֶּ֛ם עֹֽבְרִ֥ים אֶת־הַיַּרְדֵּ֖ן אֶל־אֶ֥רֶץ כְּנָֽעַן׃ מַלֵּיל עִם בְּנֵי יִשְׂרָאֵל, וְתֵימַר לְהוֹן:  אֲרֵי אַתּוּן עָבְרִין יָת יַרְדְּנָא, לְאַרְעָא דִּכְנָעַן. 
לג,נב וְה֨וֹרַשְׁתֶּ֜ם אֶת־כָּל־יֹֽשְׁבֵ֤י הָאָ֨רֶץ֙ מִפְּנֵיכֶ֔ם וְאִ֨בַּדְתֶּ֔ם אֵ֖ת כָּל־מַשְׂכִּיֹּתָ֑ם וְאֵ֨ת כָּל־צַלְמֵ֤י מַסֵּֽכֹתָם֙ תְּאַבֵּ֔דוּ וְאֵ֥ת כָּל־בָּֽמוֹתָ֖ם תַּשְׁמִֽידוּ׃ וּתְתָרְכוּן יָת כָּל יָתְבֵי אַרְעָא, מִן קֳדָמֵיכוֹן, וְתַבְּדוּן, יָת כָּל בֵּית סִגְדַּתְהוֹן; וְיָת כָּל צַלְמֵי מַתְּכָתְהוֹן תַּבְּדוּן, וְיָת כָּל בָּמָתְהוֹן תְּשֵׁיצוֹן. 
לג,נג וְהֽוֹרַשְׁתֶּ֥ם אֶת־הָאָ֖רֶץ וִֽישַׁבְתֶּם־בָּ֑הּ כִּ֥י לָכֶ֛ם נָתַ֥תִּי אֶת־הָאָ֖רֶץ לָרֶ֥שֶׁת אֹתָֽהּ׃ וּתְתָרְכוּן יָת יָתְבֵי אַרְעָא, וְתִתְּבוּן בַּהּ:  אֲרֵי לְכוֹן יְהַבִית יָת אַרְעָא, לְמֵירַת יָתַהּ. 
לג,נד וְהִתְנַֽחַלְתֶּם֩ אֶת־הָאָ֨רֶץ בְּגוֹרָ֜ל לְמִשְׁפְּחֹֽתֵיכֶ֗ם לָרַ֞ב תַּרְבּ֤וּ אֶת־נַֽחֲלָתוֹ֙ וְלַמְעַט֙ תַּמְעִ֣יט אֶת־נַֽחֲלָת֔וֹ אֶל֩ אֲשֶׁר־יֵ֨צֵא ל֥וֹ שָׁ֛מָּה הַגּוֹרָ֖ל ל֣וֹ יִֽהְיֶ֑ה לְמַטּ֥וֹת אֲבֹֽתֵיכֶ֖ם תִּתְנֶחָֽלוּ׃ וְתַחְסְנוּן יָת אַרְעָא בְּעַדְבָּא לְזַרְעֲיָתְכוֹן, לְסַגִּיאֵי תַּסְגּוֹן יָת אַחְסָנַתְהוֹן וְלִזְעֵירֵי תַּזְעֲרוּן יָת אַחְסָנַתְהוֹן--לִדְיִפּוֹק לֵיהּ תַּמָּן עַדְבָּא, דִּילֵיהּ יְהֵי:  לְשִׁבְטֵי אֲבָהָתְכוֹן, תַּחְסְנוּן. 
לג,נה וְאִם־לֹ֨א תוֹרִ֜ישׁוּ אֶת־יֹֽשְׁבֵ֣י הָאָרֶץ֮ מִפְּנֵיכֶם֒ וְהָיָה֙ אֲשֶׁ֣ר תּוֹתִ֣ירוּ מֵהֶ֔ם לְשִׂכִּים֙ בְּעֵ֣ינֵיכֶ֔ם וְלִצְנִינִ֖ם בְּצִדֵּיכֶ֑ם וְצָֽרְר֣וּ אֶתְכֶ֔ם עַל־הָאָ֕רֶץ אֲשֶׁ֥ר אַתֶּ֖ם יֹֽשְׁבִ֥ים בָּֽהּ׃ וְאִם לָא תְּתָרְכוּן יָת יָתְבֵי אַרְעָא, מִן קֳדָמֵיכוֹן--וִיהֵי דְּתַשְׁאֲרוּן מִנְּהוֹן, לְסִיעָן נָטְלָן זֵין לְקִבְלְכוֹן וּלְמַשְׁרְיָן מַקְּפָנֵיכוֹן; וִיעִיקוּן לְכוֹן--עַל אַרְעָא, דְּאַתּוּן יָתְבִין בַּהּ. 
לג,נו וְהָיָ֗ה כַּֽאֲשֶׁ֥ר דִּמִּ֛יתִי לַֽעֲשׂ֥וֹת לָהֶ֖ם אֶֽעֱשֶׂ֥ה לָכֶֽם׃ {פ} וִיהֵי, כְּמָא דְּחַשֵּׁיבִית לְמֶעֱבַד לְהוֹן--אַעֲבֵיד לְכוֹן. {פ}
לד,א וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
לד,ב צַ֞ו אֶת־בְּנֵ֤י יִשְׂרָאֵל֙ וְאָֽמַרְתָּ֣ אֲלֵהֶ֔ם כִּֽי־אַתֶּ֥ם בָּאִ֖ים אֶל־הָאָ֣רֶץ כְּנָ֑עַן זֹ֣את הָאָ֗רֶץ אֲשֶׁ֨ר תִּפֹּ֤ל לָכֶם֙ בְּֽנַחֲלָ֔ה אֶ֥רֶץ כְּנַ֖עַן לִגְבֻֽלֹתֶֽיהָ׃ פַּקֵּיד יָת בְּנֵי יִשְׂרָאֵל, וְתֵימַר לְהוֹן, אֲרֵי אַתּוּן עָאלִין, לְאַרְעָא דִּכְנָעַן:  דָּא אַרְעָא, דְּתִתְפְּלֵיג לְכוֹן בְּאַחְסָנָא, אַרְעָא דִּכְנַעַן, לִתְחוּמַהָא. 
לד,ג וְהָיָ֨ה לָכֶ֧ם פְּאַת־נֶ֛גֶב מִמִּדְבַּר־צִ֖ן עַל־יְדֵ֣י אֱד֑וֹם וְהָיָ֤ה לָכֶם֙ גְּב֣וּל נֶ֔גֶב מִקְצֵ֥ה יָם־הַמֶּ֖לַח קֵֽדְמָה׃ וִיהֵי לְכוֹן רוּחַ דָּרוֹמָא מִמַּדְבְּרָא דְּצִין, עַל תְּחוּמֵי אֱדוֹם; וִיהֵי לְכוֹן תְּחוּם דָּרוֹמָא, מִסְּיָפֵי יַמָּא דְּמִלְחָא קִדּוּמָא. 
לד,ד וְנָסַ֣ב לָכֶם֩ הַגְּב֨וּל מִנֶּ֜גֶב לְמַֽעֲלֵ֤ה עַקְרַבִּים֙ וְעָ֣בַר צִ֔נָה והיה (וְהָיוּ֙) תּֽוֹצְאֹתָ֔יו מִנֶּ֖גֶב לְקָדֵ֣שׁ בַּרְנֵ֑עַ וְיָצָ֥א חֲצַר־אַדָּ֖ר וְעָבַ֥ר עַצְמֹֽנָה׃ וְיַסְחַר לְכוֹן תְּחוּמָא מִדָּרוֹמָא לְמַסְּקַנָא דְּעַקְרַבִּין, וְיִעְבַּר לְצִין, וִיהוֹן מַפְּקָנוֹהִי, מִדָּרוֹמָא לִרְקַם גֵּיאָה; וְיִפּוֹק לַחֲצַר אַדָּר, וְיִעְבַּר לְעַצְמוֹן. 
לד,ה וְנָסַ֧ב הַגְּב֛וּל מֵֽעַצְמ֖וֹן נַ֣חְלָה מִצְרָ֑יִם וְהָי֥וּ תֽוֹצְאֹתָ֖יו הַיָּֽמָּה׃ וְיַסְחַר תְּחוּמָא מֵעַצְמוֹן, לְנַחְלָא דְּמִצְרָיִם; וִיהוֹן מַפְּקָנוֹהִי, לְיַמָּא. 
לד,ו וּגְב֣וּל יָ֔ם וְהָיָ֥ה לָכֶ֛ם הַיָּ֥ם הַגָּד֖וֹל וּגְב֑וּל זֶֽה־יִהְיֶ֥ה לָכֶ֖ם גְּב֥וּל יָֽם׃ וּתְחוּם מַעְרְבָא, וִיהֵי לְכוֹן יַמָּא רַבָּא וּתְחוּמֵיהּ; דֵּין יְהֵי לְכוֹן, תְּחוּם מַעְרְבָא. 
לד,ז וְזֶֽה־יִהְיֶ֥ה לָכֶ֖ם גְּב֣וּל צָפ֑וֹן מִן־הַיָּם֙ הַגָּדֹ֔ל תְּתָא֥וּ לָכֶ֖ם הֹ֥ר הָהָֽר׃ וְדֵין יְהֵי לְכוֹן, תְּחוּם צִפּוּנָא:  מִן יַמָּא, רַבָּא, תְּכָוְנוּן לְכוֹן, לְהוֹר טוּרָא. 
לד,ח מֵהֹ֣ר הָהָ֔ר תְּתָא֖וּ לְבֹ֣א חֲמָ֑ת וְהָי֛וּ תּֽוֹצְאֹ֥ת הַגְּבֻ֖ל צְדָֽדָה׃ מֵהוֹר טוּרָא, תְּכָוְנוּן לְמָטֵי חֲמָת; וִיהוֹן מַפְּקָנוֹהִי דִּתְחוּמָא, לִצְדָד. 
לד,ט וְיָצָ֤א הַגְּבֻל֙ זִפְרֹ֔נָה וְהָי֥וּ תֽוֹצְאֹתָ֖יו חֲצַ֣ר עֵינָ֑ן זֶֽה־יִהְיֶ֥ה לָכֶ֖ם גְּב֥וּל צָפֽוֹן׃ וְיִפּוֹק תְּחוּמָא לְזִפְרוֹן, וִיהוֹן מַפְּקָנוֹהִי לַחֲצַר עֵינָן; דֵּין יְהֵי לְכוֹן, תְּחוּם צִפּוּנָא. 
לד,י וְהִתְאַוִּיתֶ֥ם לָכֶ֖ם לִגְב֣וּל קֵ֑דְמָה מֵֽחֲצַ֥ר עֵינָ֖ן שְׁפָֽמָה׃ וּתְכָוְנוּן לְכוֹן, לִתְחוּם קִדּוּמָא, מֵחֲצַר עֵינָן, לִשְׁפָם. 
לד,יא וְיָרַ֨ד הַגְּבֻ֧ל מִשְּׁפָ֛ם הָֽרִבְלָ֖ה מִקֶּ֣דֶם לָעָ֑יִן וְיָרַ֣ד הַגְּבֻ֔ל וּמָחָ֛ה עַל־כֶּ֥תֶף יָם־כִּנֶּ֖רֶת קֵֽדְמָה׃ וְיֵיחוֹת תְּחוּמָא מִשְּׁפָם לְרִבְלָה, מִמַּדְנַח לְעָיִן; וְיֵיחוֹת תְּחוּמָא, וְיִמְטֵי עַל כֵּיף יָם גִּנֵּיסַר קִדּוּמָא. 
לד,יב וְיָרַ֤ד הַגְּבוּל֙ הַיַּרְדֵּ֔נָה וְהָי֥וּ תֽוֹצְאֹתָ֖יו יָ֣ם הַמֶּ֑לַח זֹאת֩ תִּֽהְיֶ֨ה לָכֶ֥ם הָאָ֛רֶץ לִגְבֻֽלֹתֶ֖יהָ סָבִֽיב׃ וְיֵיחוֹת תְּחוּמָא לְיַרְדְּנָא, וִיהוֹן מַפְּקָנוֹהִי לְיַמָּא דְּמִלְחָא; דָּא תְּהֵי לְכוֹן אַרְעָא לִתְחוּמַהָא, סְחוֹר סְחוֹר. 
לד,יג וַיְצַ֣ו מֹשֶׁ֔ה אֶת־בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר זֹ֣את הָאָ֗רֶץ אֲשֶׁ֨ר תִּתְנַֽחֲל֤וּ אֹתָהּ֙ בְּגוֹרָ֔ל אֲשֶׁר֙ צִוָּ֣ה יְהוָ֔ה לָתֵ֛ת לְתִשְׁעַ֥ת הַמַּטּ֖וֹת וַֽחֲצִ֥י הַמַּטֶּֽה׃ וּפַקֵּיד מֹשֶׁה, יָת בְּנֵי יִשְׂרָאֵל לְמֵימַר:  דָּא אַרְעָא, דְּתַחְסְנוּן יָתַהּ בְּעַדְבָּא, דְּפַקֵּיד יְיָ, לְמִתַּן לְתִשְׁעַת שִׁבְטִין וּפַלְגוּת שִׁבְטָא. 
לד,יד כִּ֣י לָֽקְח֞וּ מַטֵּ֨ה בְנֵ֤י הָרֽאוּבֵנִי֙ לְבֵ֣ית אֲבֹתָ֔ם וּמַטֵּ֥ה בְנֵֽי־הַגָּדִ֖י לְבֵ֣ית אֲבֹתָ֑ם וַֽחֲצִי֙ מַטֵּ֣ה מְנַשֶּׁ֔ה לָֽקְח֖וּ נַֽחֲלָתָֽם׃ אֲרֵי קַבִּילוּ שִׁבְטָא דִּבְנֵי רְאוּבֵן, לְבֵית אֲבָהָתְהוֹן, וְשִׁבְטָא דִּבְנֵי גָּד, לְבֵית אֲבָהָתְהוֹן; וּפַלְגוּת שִׁבְטָא דִּמְנַשֶּׁה, קַבִּילוּ אַחְסָנַתְהוֹן. 
לד,טו שְׁנֵ֥י הַמַּטּ֖וֹת וַֽחֲצִ֣י הַמַּטֶּ֑ה לָֽקְח֣וּ נַֽחֲלָתָ֗ם מֵעֵ֛בֶר לְיַרְדֵּ֥ן יְרֵח֖וֹ קֵ֥דְמָה מִזְרָֽחָה׃ {פ} תְּרֵין שִׁבְטִין, וּפַלְגוּת שִׁבְטָא:  קַבִּילוּ אַחְסָנַתְהוֹן, מֵעִבְרָא לְיַרְדְּנָא דִּירִיחוֹ--קִדּוּמָא מַדְנְחָא. {פ}
לד,טז וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
לד,יז אֵ֚לֶּה שְׁמ֣וֹת הָֽאֲנָשִׁ֔ים אֲשֶׁר־יִנְחֲל֥וּ לָכֶ֖ם אֶת־הָאָ֑רֶץ אֶלְעָזָר֙ הַכֹּהֵ֔ן וִֽיהוֹשֻׁ֖עַ בִּן־נֽוּן׃ אִלֵּין שְׁמָהָת גֻּבְרַיָּא, דְּיַחְסְנוּן לְכוֹן יָת אַרְעָא:  אֶלְעָזָר, כָּהֲנָא, וִיהוֹשׁוּעַ, בַּר נוּן. 
לד,יח וְנָשִׂ֥יא אֶחָ֛ד נָשִׂ֥יא אֶחָ֖ד מִמַּטֶּ֑ה תִּקְח֖וּ לִנְחֹ֥ל אֶת־הָאָֽרֶץ׃ וְרַבָּא חַד רַבָּא חַד, מִשִּׁבְטָא--תִּדְבְּרוּן, לְאַחְסָנָא יָת אַרְעָא. 
לד,יט וְאֵ֖לֶּה שְׁמ֣וֹת הָֽאֲנָשִׁ֑ים לְמַטֵּ֣ה יְהוּדָ֔ה כָּלֵ֖ב בֶּן־יְפֻנֶּֽה׃ וְאִלֵּין, שְׁמָהָת גֻּבְרַיָּא:  לְשִׁבְטָא דִּיהוּדָה, כָּלֵב בַּר יְפֻנֶּה. 
לד,כ וּלְמַטֵּה֙ בְּנֵ֣י שִׁמְע֔וֹן שְׁמוּאֵ֖ל בֶּן־עַמִּיהֽוּד׃ וּלְשִׁבְטָא דִּבְנֵי שִׁמְעוֹן, שְׁמוּאֵל בַּר עַמִּיהוּד. 
לד,כא לְמַטֵּ֣ה בִנְיָמִ֔ן אֱלִידָ֖ד בֶּן־כִּסְלֽוֹן׃ לְשִׁבְטָא דְּבִנְיָמִין, אֱלִידָד בַּר כִּסְלוֹן. 
לד,כב וּלְמַטֵּ֥ה בְנֵי־דָ֖ן נָשִׂ֑יא בֻּקִּ֖י בֶּן־יָגְלִֽי׃ וּלְשִׁבְטָא דִּבְנֵי דָּן, רַבָּא--בֻּקִּי, בַּר יָגְלִי. 
לד,כג לִבְנֵ֣י יוֹסֵ֔ף לְמַטֵּ֥ה בְנֵֽי־מְנַשֶּׁ֖ה נָשִׂ֑יא חַנִּיאֵ֖ל בֶּן־אֵפֹֽד׃ לִבְנֵי יוֹסֵף, לְשִׁבְטָא דִּבְנֵי מְנַשֶּׁה רַבָּא--חַנִּיאֵל, בַּר אֵיפוֹד. 
לד,כד וּלְמַטֵּ֥ה בְנֵֽי־אֶפְרַ֖יִם נָשִׂ֑יא קְמוּאֵ֖ל בֶּן־שִׁפְטָֽן׃ וּלְשִׁבְטָא דִּבְנֵי אֶפְרַיִם, רַבָּא--קְמוּאֵל, בַּר שִׁפְטָן. 
לד,כה וּלְמַטֵּ֥ה בְנֵֽי־זְבוּלֻ֖ן נָשִׂ֑יא אֱלִֽיצָפָ֖ן בֶּן־פַּרְנָֽךְ׃ וּלְשִׁבְטָא דִּבְנֵי זְבוּלוּן, רַבָּא--אֱלִיצָפָן, בַּר פַּרְנָךְ. 
לד,כו וּלְמַטֵּ֥ה בְנֵֽי־יִשָּׂשכָ֖ר נָשִׂ֑יא פַּלְטִיאֵ֖ל בֶּן־עַזָּֽן׃ וּלְשִׁבְטָא דִּבְנֵי יִשָּׂשכָר, רַבָּא--פַּלְטִיאֵל, בַּר עַזָּן. 
לד,כז וּלְמַטֵּ֥ה בְנֵֽי־אָשֵׁ֖ר נָשִׂ֑יא אֲחִיה֖וּד בֶּן־שְׁלֹמִֽי׃ וּלְשִׁבְטָא דִּבְנֵי אָשֵׁר, רַבָּא--אֲחִיהוּד, בַּר שְׁלוֹמִי. 
לד,כח וּלְמַטֵּ֥ה בְנֵֽי־נַפְתָּלִ֖י נָשִׂ֑יא פְּדַהְאֵ֖ל בֶּן־עַמִּיהֽוּד׃ וּלְשִׁבְטָא דִּבְנֵי נַפְתָּלִי, רַבָּא--פְּדַהְאֵל, בַּר עַמִּיהוּד. 
לד,כט אֵ֕לֶּה אֲשֶׁ֖ר צִוָּ֣ה יְהוָ֑ה לְנַחֵ֥ל אֶת־בְּנֵֽי־יִשְׂרָאֵ֖ל בְּאֶ֥רֶץ כְּנָֽעַן׃ {פ} אִלֵּין, דְּפַקֵּיד יְיָ, לְאַחְסָנָא יָת בְּנֵי יִשְׂרָאֵל, בְּאַרְעָא דִּכְנָעַן. {פ}
לה,א וַיְדַבֵּ֧ר יְהוָ֛ה אֶל־מֹשֶׁ֖ה בְּעַֽרְבֹ֣ת מוֹאָ֑ב עַל־יַרְדֵּ֥ן יְרֵח֖וֹ לֵאמֹֽר׃ וּמַלֵּיל יְיָ עִם מֹשֶׁה, בְּמֵישְׁרַיָּא דְּמוֹאָב, עַל יַרְדְּנָא דִּירִיחוֹ, לְמֵימַר. 
לה,ב צַו֮ אֶת־בְּנֵ֣י יִשְׂרָאֵל֒ וְנָֽתְנ֣וּ לַלְוִיִּ֗ם מִֽנַּחֲלַ֛ת אֲחֻזָּתָ֖ם עָרִ֣ים לָשָׁ֑בֶת וּמִגְרָ֗שׁ לֶֽעָרִים֙ סְבִיבֹ֣תֵיהֶ֔ם תִּתְּנ֖וּ לַלְוִיִּֽם׃ פַּקֵּיד, יָת בְּנֵי יִשְׂרָאֵל, וְיִתְּנוּן לְלֵיוָאֵי מֵאַחְסָנַת אֲחֻדַּתְהוֹן, קִרְוִין לְמִתַּב; וְרֶוַח, לְקִרְוַיָּא סַחְרָנֵיהוֹן, תִּתְּנוּן, לְלֵיוָאֵי. 
לה,ג וְהָי֧וּ הֶֽעָרִ֛ים לָהֶ֖ם לָשָׁ֑בֶת וּמִגְרְשֵׁיהֶ֗ם יִֽהְי֤וּ לִבְהֶמְתָּם֙ וְלִרְכֻשָׁ֔ם וּלְכֹ֖ל חַיָּתָֽם׃ וִיהוֹן קִרְוַיָּא לְהוֹן, לְמִתַּב; וְרַוְחֵיהוֹן, יְהוֹן לִבְעִירְהוֹן וּלְקִנְיָנְהוֹן, וּלְכֹל, חֵיוַתְהוֹן. 
לה,ד וּמִגְרְשֵׁי֙ הֶֽעָרִ֔ים אֲשֶׁ֥ר תִּתְּנ֖וּ לַלְוִיִּ֑ם מִקִּ֤יר הָעִיר֙ וָח֔וּצָה אֶ֥לֶף אַמָּ֖ה סָבִֽיב׃ וְרַוְחֵי, קִרְוַיָּא, דְּתִתְּנוֹן, לְלֵיוָאֵי--מִכֹּתֶל קַרְתָּא וּלְבָרָא, אֶלֶף אַמִּין סְחוֹר סְחוֹר. 
לה,ה וּמַדֹּתֶ֞ם מִח֣וּץ לָעִ֗יר אֶת־פְּאַת־קֵ֣דְמָה אַלְפַּ֪יִם בָּֽאַמָּ֟ה וְאֶת־פְּאַת־נֶגֶב֩ אַלְפַּ֨יִם בָּֽאַמָּ֜ה וְאֶת־פְּאַת־יָ֣ם ׀ אַלְפַּ֣יִם בָּֽאַמָּ֗ה וְאֵ֨ת פְּאַ֥ת צָפ֛וֹן אַלְפַּ֥יִם בָּֽאַמָּ֖ה וְהָעִ֣יר בַּתָּ֑וֶךְ זֶ֚ה יִֽהְיֶ֣ה לָהֶ֔ם מִגְרְשֵׁ֖י הֶֽעָרִֽים׃ וְתִמְשְׁחוּן מִבַּרָא לְקַרְתָּא, יָת רוּחַ קִדּוּמָא תְּרֵין אַלְפִין אַמִּין וְיָת רוּחַ דָּרוֹמָא תְּרֵין אַלְפִין אַמִּין וְיָת רוּחַ מַעְרְבָא תְּרֵין אַלְפִין אַמִּין וְיָת רוּחַ צִפּוּנָא תְּרֵין אַלְפִין אַמִּין--וְקַרְתָּא בִּמְצִיעַ; דֵּין יְהֵי לְהוֹן, רַוְחֵי קִרְוַיָּא. 
לה,ו וְאֵ֣ת הֶֽעָרִ֗ים אֲשֶׁ֤ר תִּתְּנוּ֙ לַלְוִיִּ֔ם אֵ֚ת שֵׁשׁ־עָרֵ֣י הַמִּקְלָ֔ט אֲשֶׁ֣ר תִּתְּנ֔וּ לָנֻ֥ס שָׁ֖מָּה הָֽרֹצֵ֑חַ וַֽעֲלֵיהֶ֣ם תִּתְּנ֔וּ אַרְבָּעִ֥ים וּשְׁתַּ֖יִם עִֽיר׃ וְיָת קִרְוַיָּא, דְּתִתְּנוּן לְלֵיוָאֵי--יָת שֵׁית קִרְוֵי שֵׁיזָבוּתָא, דְּתִתְּנוּן לְמִעְרַק לְתַמָּן קָטוֹלָא; וַעֲלֵיהוֹן תִּתְּנוּן, אַרְבְּעִין וְתַרְתֵּין קִרְוִין. 
לה,ז כָּל־הֶֽעָרִ֗ים אֲשֶׁ֤ר תִּתְּנוּ֙ לַלְוִיִּ֔ם אַרְבָּעִ֥ים וּשְׁמֹנֶ֖ה עִ֑יר אֶתְהֶ֖ן וְאֶת־מִגְרְשֵׁיהֶֽן׃ כָּל קִרְוַיָּא, דְּתִתְּנוּן לְלֵיוָאֵי--אַרְבְּעִין וְתַמְנֵי, קִרְוִין:  יָתְהוֹן, וְיָת רַוְחֵיהוֹן. 
לה,ח וְהֶֽעָרִ֗ים אֲשֶׁ֤ר תִּתְּנוּ֙ מֵֽאֲחֻזַּ֣ת בְּנֵֽי־יִשְׂרָאֵ֔ל מֵאֵ֤ת הָרַב֙ תַּרְבּ֔וּ וּמֵאֵ֥ת הַמְעַ֖ט תַּמְעִ֑יטוּ אִ֗ישׁ כְּפִ֤י נַֽחֲלָתוֹ֙ אֲשֶׁ֣ר יִנְחָ֔לוּ יִתֵּ֥ן מֵֽעָרָ֖יו לַלְוִיִּֽם׃ {פ} וְקִרְוַיָּא, דְּתִתְּנוּן מֵאֲחֻדַּת בְּנֵי יִשְׂרָאֵל--מִן סַגִּיאֵי תַּסְגּוֹן, וּמִן זְעֵירֵי תַּזְעֲרוּן:  גְּבַר, כְּפֹם אַחְסָנְתֵיהּ דְּיַחְסְנוּן, יִתֵּין מִקִּרְווֹהִי, לְלֵיוָאֵי. {פ}
לה,ט וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
לה,י דַּבֵּר֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָֽמַרְתָּ֖ אֲלֵהֶ֑ם כִּ֥י אַתֶּ֛ם עֹֽבְרִ֥ים אֶת־הַיַּרְדֵּ֖ן אַ֥רְצָה כְּנָֽעַן׃ מַלֵּיל עִם בְּנֵי יִשְׂרָאֵל, וְתֵימַר לְהוֹן:  אֲרֵי אַתּוּן עָבְרִין יָת יַרְדְּנָא, לְאַרְעָא דִּכְנָעַן. 
לה,יא וְהִקְרִיתֶ֤ם לָכֶם֙ עָרִ֔ים עָרֵ֥י מִקְלָ֖ט תִּֽהְיֶ֣ינָה לָכֶ֑ם וְנָ֥ס שָׁ֨מָּה֙ רֹצֵ֔חַ מַכֵּה־נֶ֖פֶשׁ בִּשְׁגָגָֽה׃ וּתְזָמְנוּן לְכוֹן קִרְוִין, קִרְוֵי שֵׁיזָבוּתָא יִהְוְיָן לְכוֹן; וְיִעְרוֹק לְתַמָּן קָטוֹלָא, דְּיִקְטוֹל נַפְשָׁא בְּשָׁלוּ. 
לה,יב וְהָי֨וּ לָכֶ֧ם הֶֽעָרִ֛ים לְמִקְלָ֖ט מִגֹּאֵ֑ל וְלֹ֤א יָמוּת֙ הָֽרֹצֵ֔חַ עַד־עָמְד֛וֹ לִפְנֵ֥י הָֽעֵדָ֖ה לַמִּשְׁפָּֽט׃ וְיִהְוְיָן לְכוֹן קִרְוַיָּא לְשֵׁיזָבָא, מִגָּאֵיל דְּמָא; וְלָא יְמוּת קָטוֹלָא, עַד דִּיקוּם קֳדָם כְּנִשְׁתָּא לְדִינָא. 
לה,יג וְהֶֽעָרִ֖ים אֲשֶׁ֣ר תִּתֵּ֑נוּ שֵׁשׁ־עָרֵ֥י מִקְלָ֖ט תִּֽהְיֶ֥ינָה לָכֶֽם׃ וְקִרְוַיָּא, דְּתִתְּנוּן--שֵׁית קִרְוֵי שֵׁיזָבוּתָא, יִהְוְיָן לְכוֹן. 
לה,יד אֵ֣ת ׀ שְׁלֹ֣שׁ הֶֽעָרִ֗ים תִּתְּנוּ֙ מֵעֵ֣בֶר לַיַּרְדֵּ֔ן וְאֵת֙ שְׁלֹ֣שׁ הֶֽעָרִ֔ים תִּתְּנ֖וּ בְּאֶ֣רֶץ כְּנָ֑עַן עָרֵ֥י מִקְלָ֖ט תִּֽהְיֶֽינָה׃ יָת תְּלָת קִרְוַיָּא, תִּתְּנוּן מֵעִבְרָא לְיַרְדְּנָא, וְיָת תְּלָת קִרְוַיָּא, תִּתְּנוּן בְּאַרְעָא דִּכְנָעַן:  קִרְוֵי שֵׁיזָבוּתָא, יִהְוְיָן. 
לה,טו לִבְנֵ֣י יִשְׂרָאֵ֗ל וְלַגֵּ֤ר וְלַתּוֹשָׁב֙ בְּתוֹכָ֔ם תִּֽהְיֶ֛ינָה שֵׁשׁ־הֶֽעָרִ֥ים הָאֵ֖לֶּה לְמִקְלָ֑ט לָנ֣וּס שָׁ֔מָּה כָּל־מַכֵּה־נֶ֖פֶשׁ בִּשְׁגָגָֽה׃ לִבְנֵי יִשְׂרָאֵל, וּלְגִיּוֹרַיָּא וּלְתוֹתָבַיָּא דְּבֵינֵיהוֹן, יִהְוְיָן שֵׁית קִרְוַיָּא הָאִלֵּין, לְשֵׁיזָבָא--לְמִעְרַק לְתַמָּן, כָּל דְּיִקְטוֹל נַפְשָׁא בְּשָׁלוּ. 
לה,טז וְאִם־בִּכְלִ֨י בַרְזֶ֧ל ׀ הִכָּ֛הוּ וַיָּמֹ֖ת רֹצֵ֣חַֽ ה֑וּא מ֥וֹת יוּמַ֖ת הָֽרֹצֵֽחַ׃ וְאִם בְּמָן דְּבַרְזֶל מְחָהִי וְקַטְלֵיהּ, קָטוֹלָא הוּא:  אִתְקְטָלָא יִתְקְטִיל, קָטוֹלָא. 
לה,יז וְאִ֡ם בְּאֶ֣בֶן יָד֩ אֲשֶׁר־יָמ֨וּת בָּ֥הּ הִכָּ֛הוּ וַיָּמֹ֖ת רֹצֵ֣חַֽ ה֑וּא מ֥וֹת יוּמַ֖ת הָֽרֹצֵֽחַ׃ וְאִם בְּאַבְנָא דְּמִתְנַסְבָּא בְּיַד דְּהִיא כְּמִסַּת דִּימוּת בַּהּ מְחָהִי, וְקַטְלֵיהּ--קָטוֹלָא הוּא:  אִתְקְטָלָא יִתְקְטִיל, קָטוֹלָא. 
לה,יח א֡וֹ בִּכְלִ֣י עֵֽץ־יָד֩ אֲשֶׁר־יָמ֨וּת בּ֥וֹ הִכָּ֛הוּ וַיָּמֹ֖ת רֹצֵ֣חַֽ ה֑וּא מ֥וֹת יוּמַ֖ת הָֽרֹצֵֽחַ׃ אוֹ בְּמָן דְּאָע דְּמִתְנְסֵיב בְּיַד דְּהוּא כְּמִסַּת דִּימוּת בֵּיהּ מְחָהִי, וְקַטְלֵיהּ--קָטוֹלָא הוּא:  אִתְקְטָלָא יִתְקְטִיל, קָטוֹלָא. 
לה,יט גֹּאֵ֣ל הַדָּ֔ם ה֥וּא יָמִ֖ית אֶת־הָֽרֹצֵ֑חַ בְּפִגְעוֹ־ב֖וֹ ה֥וּא יְמִתֶֽנּוּ׃ גָּאֵיל דְּמָא, הוּא יִקְטוֹל יָת קָטוֹלָא:  כַּד אִתְחַיַּב לֵיהּ מִן דִּינָא, הוּא יִקְטְלִנֵּיהּ. 
לה,כ וְאִם־בְּשִׂנְאָ֖ה יֶהְדֳּפֶ֑נּוּ אֽוֹ־הִשְׁלִ֥יךְ עָלָ֛יו בִּצְדִיָּ֖ה וַיָּמֹֽת׃ וְאִם בְּסִנְאָה, דְּחָהִי, אוֹ רְמָא עֲלוֹהִי בְּכָמְנָא, וְקַטְלֵיהּ. 
לה,כא א֣וֹ בְאֵיבָ֞ה הִכָּ֤הוּ בְיָדוֹ֙ וַיָּמֹ֔ת מֽוֹת־יוּמַ֥ת הַמַּכֶּ֖ה רֹצֵ֣חַֽ ה֑וּא גֹּאֵ֣ל הַדָּ֗ם יָמִ֛ית אֶת־הָֽרֹצֵ֖חַ בְּפִגְעוֹ־בֽוֹ׃ אוֹ בִּדְבָבוּ מְחָהִי בִּידֵיהּ, וְקַטְלֵיהּ--אִתְקְטָלָא יִתְקְטִיל מָחְיָא, קָטוֹלָא הוּא:  גָּאֵיל דְּמָא, יִקְטוֹל יָת קָטוֹלָא--כַּד אִתְחַיַּב לֵיהּ. 
לה,כב וְאִם־בְּפֶ֥תַע בְּלֹֽא־אֵיבָ֖ה הֲדָפ֑וֹ אוֹ־הִשְׁלִ֥יךְ עָלָ֛יו כָּל־כְּלִ֖י בְּלֹ֥א צְדִיָּֽה׃ וְאִם בִּתְכֵּיף בְּלָא דְּבָבוּ, דְּחָהִי, אוֹ רְמָא עֲלוֹהִי כָּל מָן, בְּלָא כְּמַן לֵיהּ. 
לה,כג א֣וֹ בְכָל־אֶ֜בֶן אֲשֶׁר־יָמ֥וּת בָּהּ֙ בְּלֹ֣א רְא֔וֹת וַיַּפֵּ֥ל עָלָ֖יו וַיָּמֹ֑ת וְהוּא֙ לֹֽא־אוֹיֵ֣ב ל֔וֹ וְלֹ֥א מְבַקֵּ֖שׁ רָֽעָתֽוֹ׃ אוֹ בְּכָל אַבְנָא דְּהִיא כְּמִסַּת דִּימוּת בַּהּ, בְּלָא חָזֵי, וּרְמָא עֲלוֹהִי, וּמִית--וְהוּא לָא סָנֵי לֵיהּ, וְלָא תָּבַע בִּשְׁתֵּיהּ. 
לה,כד וְשָֽׁפְטוּ֙ הָֽעֵדָ֔ה בֵּ֚ין הַמַּכֶּ֔ה וּבֵ֖ין גֹּאֵ֣ל הַדָּ֑ם עַ֥ל הַמִּשְׁפָּטִ֖ים הָאֵֽלֶּה׃ וִידִינוּן, כְּנִשְׁתָּא, בֵּין מָחְיָא, וּבֵין גָּאֵיל דְּמָא--עַל דִּינַיָּא, הָאִלֵּין. 
לה,כה וְהִצִּ֨ילוּ הָֽעֵדָ֜ה אֶת־הָֽרֹצֵ֗חַ מִיַּד֮ גֹּאֵ֣ל הַדָּם֒ וְהֵשִׁ֤יבוּ אֹתוֹ֙ הָֽעֵדָ֔ה אֶל־עִ֥יר מִקְלָט֖וֹ אֲשֶׁר־נָ֣ס שָׁ֑מָּה וְיָ֣שַׁב בָּ֗הּ עַד־מוֹת֙ הַכֹּהֵ֣ן הַגָּדֹ֔ל אֲשֶׁר־מָשַׁ֥ח אֹת֖וֹ בְּשֶׁ֥מֶן הַקֹּֽדֶשׁ׃ וִישֵׁיזְבוּן כְּנִשְׁתָּא יָת קָטוֹלָא, מִיַּד גָּאֵיל דְּמָא, וְיָתִיבוּן יָתֵיהּ כְּנִשְׁתָּא, לְקִרְיַת שֵׁיזָבוּתֵיהּ דַּעֲרַק לְתַמָּן; וְיִתֵּיב בַּהּ, עַד דִּימוּת כָּהֲנָא רַבָּא, דְּרַבִּי יָתֵיהּ, בִּמְשַׁח קֻדְשָׁא. 
לה,כו וְאִם־יָצֹ֥א יֵצֵ֖א הָֽרֹצֵ֑חַ אֶת־גְּבוּל֙ עִ֣יר מִקְלָט֔וֹ אֲשֶׁ֥ר יָנ֖וּס שָֽׁמָּה׃ וְאִם מִפָּק יִפּוֹק, קָטוֹלָא, יָת תְּחוּם קִרְיַת שֵׁיזָבוּתֵיהּ, דְּיִעְרוֹק לְתַמָּן. 
לה,כז וּמָצָ֤א אֹתוֹ֙ גֹּאֵ֣ל הַדָּ֔ם מִח֕וּץ לִגְב֖וּל עִ֣יר מִקְלָט֑וֹ וְרָצַ֞ח גֹּאֵ֤ל הַדָּם֙ אֶת־הָ֣רֹצֵ֔חַ אֵ֥ין ל֖וֹ דָּֽם׃ וְיַשְׁכַּח יָתֵיהּ, גָּאֵיל דְּמָא, מִבַּרָא, לִתְחוּם קִרְיַת שֵׁיזָבוּתֵיהּ:  וְיִקְטוֹל גָּאֵיל דְּמָא, יָת קָטוֹלָא--לֵית לֵיהּ, דַּם. 
לה,כח כִּ֣י בְעִ֤יר מִקְלָטוֹ֙ יֵשֵׁ֔ב עַד־מ֖וֹת הַכֹּהֵ֣ן הַגָּדֹ֑ל וְאַֽחֲרֵ֥י מוֹת֙ הַכֹּהֵ֣ן הַגָּדֹ֔ל יָשׁוּב֙ הָֽרֹצֵ֔חַ אֶל־אֶ֖רֶץ אֲחֻזָּתֽוֹ׃ אֲרֵי בְּקִרְיַת שֵׁיזָבוּתֵיהּ יִתֵּיב, עַד דִּימוּת כָּהֲנָא רַבָּא; וּבָתַר דִּימוּת, כָּהֲנָא רַבָּא--יְתוּב קָטוֹלָא, לַאֲרַע אַחְסָנְתֵיהּ. 
לה,כט וְהָי֨וּ אֵ֧לֶּה לָכֶ֛ם לְחֻקַּ֥ת מִשְׁפָּ֖ט לְדֹרֹֽתֵיכֶ֑ם בְּכֹ֖ל מוֹשְׁבֹֽתֵיכֶֽם׃ וִיהוֹן אִלֵּין לְכוֹן לִגְזֵירַת דִּין, לְדָרֵיכוֹן, בְּכֹל, מוֹתְבָנֵיכוֹן. 
לה,ל כָּ֨ל־מַכֵּה־נֶ֔פֶשׁ לְפִ֣י עֵדִ֔ים יִרְצַ֖ח אֶת־הָֽרֹצֵ֑חַ וְעֵ֣ד אֶחָ֔ד לֹֽא־יַעֲנֶ֥ה בְנֶ֖פֶשׁ לָמֽוּת׃ כָּל דְּיִקְטוֹל נַפְשָׁא--לְפֹם סָהֲדִין, יִקְטוֹל יָת קָטוֹלָא; וְסָהִיד חַד, לָא יַסְהֵיד בַּאֲנָשׁ לְמִקְטַל. 
לה,לא וְלֹֽא־תִקְח֥וּ כֹ֨פֶר֙ לְנֶ֣פֶשׁ רֹצֵ֔חַ אֲשֶׁר־ה֥וּא רָשָׁ֖ע לָמ֑וּת כִּי־מ֖וֹת יוּמָֽת׃ וְלָא תְּקַבְּלוּן מָמוֹן עַל אֲנָשׁ קָטוֹל, דְּהוּא חַיָּב לִמְמָת:  אֲרֵי אִתְקְטָלָא, יִתְקְטִיל. 
לה,לב וְלֹֽא־תִקְח֣וּ כֹ֔פֶר לָנ֖וּס אֶל־עִ֣יר מִקְלָט֑וֹ לָשׁוּב֙ לָשֶׁ֣בֶת בָּאָ֔רֶץ עַד־מ֖וֹת הַכֹּהֵֽן׃ וְלָא תְּקַבְּלוּן מָמוֹן, לְמִעְרַק לְקִרְיַת שֵׁיזָבוּתֵיהּ, לִמְתָּב לְמִתַּב בְּאַרְעָא, עַד דִּימוּת כָּהֲנָא. 
לה,לג וְלֹֽא־תַחֲנִ֣יפוּ אֶת־הָאָ֗רֶץ אֲשֶׁ֤ר אַתֶּם֙ בָּ֔הּ כִּ֣י הַדָּ֔ם ה֥וּא יַֽחֲנִ֖יף אֶת־הָאָ֑רֶץ וְלָאָ֣רֶץ לֹֽא־יְכֻפַּ֗ר לַדָּם֙ אֲשֶׁ֣ר שֻׁפַּךְ־בָּ֔הּ כִּי־אִ֖ם בְּדַ֥ם שֹֽׁפְכֽוֹ׃ וְלָא תְּחַיְּבוּן יָת אַרְעָא, דְּאַתּוּן בַּהּ, אֲרֵי דְּמָא, הוּא מְחַיֵּיב יָת אַרְעָא; וּלְאַרְעָא לָא מִתְכַּפַּר, עַל דַּם זַכַּאי דְּאִתְאֲשַׁד בַּהּ, אֱלָהֵין, בְּדַם אָשְׁדֵיהּ. 
לה,לד וְלֹ֧א תְטַמֵּ֣א אֶת־הָאָ֗רֶץ אֲשֶׁ֤ר אַתֶּם֙ יֹֽשְׁבִ֣ים בָּ֔הּ אֲשֶׁ֥ר אֲנִ֖י שֹׁכֵ֣ן בְּתוֹכָ֑הּ כִּ֚י אֲנִ֣י יְהוָ֔ה שֹׁכֵ֕ן בְּת֖וֹךְ בְּנֵ֥י יִשְׂרָאֵֽל׃ {פ} וְלָא תְּסָאֲבוּן יָת אַרְעָא, דְּאַתּוּן יָתְבִין בַּהּ--דִּשְׁכִינְתִי, שָׁרְיָא בְּגַוַּהּ:  אֲרֵי, אֲנָא יְיָ--שְׁכִינְתִי שָׁרְיָא, בְּגוֹ בְּנֵי יִשְׂרָאֵל. {פ}
לו,א וַֽיִּקְרְב֞וּ רָאשֵׁ֣י הָֽאָב֗וֹת לְמִשְׁפַּ֤חַת בְּנֵֽי־גִלְעָד֙ בֶּן־מָכִ֣יר בֶּן־מְנַשֶּׁ֔ה מִֽמִּשְׁפְּחֹ֖ת בְּנֵ֣י יוֹסֵ֑ף וַֽיְדַבְּר֞וּ לִפְנֵ֤י מֹשֶׁה֙ וְלִפְנֵ֣י הַנְּשִׂאִ֔ים רָאשֵׁ֥י אָב֖וֹת לִבְנֵ֥י יִשְׂרָאֵֽל׃ וּקְרִיבוּ רֵישֵׁי אֲבָהָתָא, לְזַרְעִית בְּנֵי גִּלְעָד בַּר מָכִיר בַּר מְנַשֶּׁה--מִזַּרְעֲיָת, בְּנֵי יוֹסֵף; וּמַלִּילוּ קֳדָם מֹשֶׁה, וּקְדָם רַבְרְבַיָּא--רֵישֵׁי אֲבָהָתָא, לִבְנֵי יִשְׂרָאֵל. 
לו,ב וַיֹּֽאמְר֗וּ אֶת־אֲדֹנִי֙ צִוָּ֣ה יְהוָ֔ה לָתֵ֨ת אֶת־הָאָ֧רֶץ בְּנַֽחֲלָ֛ה בְּגוֹרָ֖ל לִבְנֵ֣י יִשְׂרָאֵ֑ל וַֽאדֹנִי֙ צֻוָּ֣ה בַֽיהוָ֔ה לָתֵ֗ת אֶֽת־נַחֲלַ֛ת צְלָפְחָ֥ד אָחִ֖ינוּ לִבְנֹתָֽיו׃ וַאֲמַרוּ, יָת רִבּוֹנִי פַּקֵּיד יְיָ, לְמִתַּן יָת אַרְעָא בְּאַחְסָנָא בְּעַדְבָּא, לִבְנֵי יִשְׂרָאֵל; וְרִבּוֹנִי, אִתְפַּקַּד בְּמֵימְרָא דַּייָ, לְמִתַּן יָת אַחְסָנַת צְלָפְחָד אֲחוּנָא, לִבְנָתֵיהּ. 
לו,ג וְ֠הָיוּ לְאֶחָ֞ד מִבְּנֵ֨י שִׁבְטֵ֥י בְנֵֽי־יִשְׂרָאֵל֮ לְנָשִׁים֒ וְנִגְרְעָ֤ה נַֽחֲלָתָן֙ מִנַּֽחֲלַ֣ת אֲבֹתֵ֔ינוּ וְנוֹסַ֕ף עַ֚ל נַֽחֲלַ֣ת הַמַּטֶּ֔ה אֲשֶׁ֥ר תִּֽהְיֶ֖ינָה לָהֶ֑ם וּמִגֹּרַ֥ל נַֽחֲלָתֵ֖נוּ יִגָּרֵֽעַ׃ וְיִהְוְיָן לְחַד מִבְּנֵי שִׁבְטֵי בְּנֵי יִשְׂרָאֵל, לִנְשִׁין, וְתִתְמְנַע אַחְסָנַתְהוֹן מֵאַחְסָנַת אֲבָהָתַנָא, וְתִתּוֹסַף עַל אַחְסָנַת שִׁבְטָא דְּיִהְוְיָן לְהוֹן; וּמֵעֲדַב אַחְסָנְתַנָא, יִתְמְנַע. 
לו,ד וְאִם־יִֽהְיֶ֣ה הַיֹּבֵל֮ לִבְנֵ֣י יִשְׂרָאֵל֒ וְנֽוֹסְפָה֙ נַֽחֲלָתָ֔ן עַ֚ל נַֽחֲלַ֣ת הַמַּטֶּ֔ה אֲשֶׁ֥ר תִּֽהְיֶ֖ינָה לָהֶ֑ם וּמִֽנַּחֲלַת֙ מַטֵּ֣ה אֲבֹתֵ֔ינוּ יִגָּרַ֖ע נַֽחֲלָתָֽן׃ וְאִם יְהֵי יוֹבֵילָא, לִבְנֵי יִשְׂרָאֵל, וְתִתּוֹסַף אַחְסָנַתְהוֹן, עַל אַחְסָנַת שִׁבְטָא דְּיִהְוְיָן לְהוֹן; וּמֵאַחְסָנַת שִׁבְטָא דַּאֲבָהָתַנָא, תִּתְמְנַע אַחְסָנַתְהוֹן. 
לו,ה וַיְצַ֤ו מֹשֶׁה֙ אֶת־בְּנֵ֣י יִשְׂרָאֵ֔ל עַל־פִּ֥י יְהוָ֖ה לֵאמֹ֑ר כֵּ֛ן מַטֵּ֥ה בְנֵֽי־יוֹסֵ֖ף דֹּֽבְרִֽים׃ וּפַקֵּיד מֹשֶׁה יָת בְּנֵי יִשְׂרָאֵל, עַל מֵימְרָא דַּייָ לְמֵימַר:  יָאוּת שִׁבְטָא דִּבְנֵי יוֹסֵף, מְמַלְּלִין. 
לו,ו זֶ֣ה הַדָּבָ֞ר אֲשֶׁר־צִוָּ֣ה יְהוָ֗ה לִבְנ֤וֹת צְלָפְחָד֙ לֵאמֹ֔ר לַטּ֥וֹב בְּעֵֽינֵיהֶ֖ם תִּֽהְיֶ֣ינָה לְנָשִׁ֑ים אַ֗ךְ לְמִשְׁפַּ֛חַת מַטֵּ֥ה אֲבִיהֶ֖ם תִּֽהְיֶ֥ינָה לְנָשִֽׁים׃ דֵּין פִּתְגָמָא דְּפַקֵּיד יְיָ, לִבְנָת צְלָפְחָד לְמֵימַר, לִדְתָּקֵין בְּעֵינֵיהוֹן, יִהְוְיָן לִנְשִׁין:  בְּרַם, לְזַרְעִית שִׁבְטָא דַּאֲבוּהוֹן--יִהְוְיָן לִנְשִׁין. 
לו,ז וְלֹֽא־תִסֹּ֤ב נַֽחֲלָה֙ לִבְנֵ֣י יִשְׂרָאֵ֔ל מִמַּטֶּ֖ה אֶל־מַטֶּ֑ה כִּ֣י אִ֗ישׁ בְּנַֽחֲלַת֙ מַטֵּ֣ה אֲבֹתָ֔יו יִדְבְּק֖וּ בְּנֵ֥י יִשְׂרָאֵֽל׃ וְלָא תַּסְחַר אַחְסָנָא לִבְנֵי יִשְׂרָאֵל, מִשִּׁבְטָא לְשִׁבְטָא:  אֲרֵי גְּבַר, בְּאַחְסָנַת שִׁבְטָא דַּאֲבָהָתוֹהִי, יִדְבְּקוּן, בְּנֵי יִשְׂרָאֵל. 
לו,ח וְכָל־בַּ֞ת יֹרֶ֣שֶׁת נַֽחֲלָ֗ה מִמַּטּוֹת֮ בְּנֵ֣י יִשְׂרָאֵל֒ לְאֶחָ֗ד מִמִּשְׁפַּ֛חַת מַטֵּ֥ה אָבִ֖יהָ תִּֽהְיֶ֣ה לְאִשָּׁ֑ה לְמַ֗עַן יִֽירְשׁוּ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל אִ֖ישׁ נַֽחֲלַ֥ת אֲבֹתָֽיו׃ וְכָל בְּרַתָּא יָרְתַת אַחְסָנָא, מִשִּׁבְטַיָּא דִּבְנֵי יִשְׂרָאֵל--לְחַד מִזַּרְעִית שִׁבְטָא דַּאֲבוּהָא, תְּהֵי לְאִתּוּ:  בְּדִיל, דְּיֵירְתוּן בְּנֵי יִשְׂרָאֵל, גְּבַר, אַחְסָנַת אֲבָהָתוֹהִי. 
לו,ט וְלֹֽא־תִסֹּ֧ב נַֽחֲלָ֛ה מִמַּטֶּ֖ה לְמַטֶּ֣ה אַחֵ֑ר כִּי־אִישׁ֙ בְּנַ֣חֲלָת֔וֹ יִדְבְּק֕וּ מַטּ֖וֹת בְּנֵ֥י יִשְׂרָאֵֽל׃ וְלָא תַּסְחַר אַחְסָנָא מִשִּׁבְטָא, לְשִׁבְטָא אֻחְרָנָא:  אֲרֵי גְּבַר, בְּאַחְסָנְתֵיהּ, יִדְבְּקוּן, שִׁבְטַיָּא דִּבְנֵי יִשְׂרָאֵל. 
לו,י כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶׁ֑ה כֵּ֥ן עָשׂ֖וּ בְּנ֥וֹת צְלָפְחָֽד׃ כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה, כֵּין עֲבַדָא, בְּנָת צְלָפְחָד. 
לו,יא וַתִּֽהְיֶ֜ינָה מַחְלָ֣ה תִרְצָ֗ה וְחָגְלָ֧ה וּמִלְכָּ֛ה וְנֹעָ֖ה בְּנ֣וֹת צְלָפְחָ֑ד לִבְנֵ֥י דֹֽדֵיהֶ֖ן לְנָשִֽׁים׃ וַהֲוַאָה מַחְלָה תִּרְצָה, וְחָגְלָה וּמִלְכָּה וְנוֹעָה--בְּנָת צְלָפְחָד:  לִבְנֵי אֲחֵי אֲבוּהוֹן, לִנְשִׁין. 
לו,יב מִֽמִּשְׁפְּחֹ֛ת בְּנֵֽי־מְנַשֶּׁ֥ה בֶן־יוֹסֵ֖ף הָי֣וּ לְנָשִׁ֑ים וַתְּהִי֙ נַֽחֲלָתָ֔ן עַל־מַטֵּ֖ה מִשְׁפַּ֥חַת אֲבִיהֶֽן׃ מִזַּרְעֲיָת בְּנֵי מְנַשֶּׁה בַּר יוֹסֵף, הֲוַאָה לִנְשִׁין; וַהֲוָת, אַחְסָנַתְהוֹן, עַל שֵׁיבַט, זַרְעִית אֲבוּהוֹן. 
לו,יג אֵ֣לֶּה הַמִּצְו֞‍ֹת וְהַמִּשְׁפָּטִ֗ים אֲשֶׁ֨ר צִוָּ֧ה יְהוָ֛ה בְּיַד־מֹשֶׁ֖ה אֶל־בְּנֵ֣י יִשְׂרָאֵ֑ל בְּעַֽרְבֹ֣ת מוֹאָ֔ב עַ֖ל יַרְדֵּ֥ן יְרֵחֽוֹ׃ {ש} אִלֵּין פִּקּוֹדַיָּא וְדִינַיָּא, דְּפַקֵּיד יְיָ בִּידָא דְּמֹשֶׁה--לְוָת בְּנֵי יִשְׂרָאֵל:  בְּמֵישְׁרַיָּא דְּמוֹאָב, עַל יַרְדְּנָא דִּירִיחוֹ. {ש}
