איוב פרק לג

א וְֽאוּלָ֗ם שְׁמַֽע־נָ֣א אִיּ֣וֹב מִלָּ֑י    וְֽכָל־דְּבָרַ֥י הַֽאֲזִֽינָה׃
ב הִנֵּה־נָ֭א פָּתַ֣חְתִּי פִ֑י    דִּבְּרָ֖ה לְשׁוֹנִ֣י בְחִכִּֽי׃
ג יֹֽשֶׁר־לִבִּ֥י אֲמָרָ֑י    וְדַ֥עַת שְׂ֝פָתַ֗י בָּר֥וּר מִלֵּֽלוּ׃
ד רֽוּחַ־אֵ֥ל עָשָׂ֑תְנִי    וְנִשְׁמַ֖ת שַׁדַּ֣י תְּחַיֵּֽנִי׃
ה אִם־תּוּכַ֥ל הֲשִׁיבֵ֑נִי    עֶרְכָ֥ה לְ֝פָנַ֗י הִתְיַצָּֽבָה׃
ו הֵן־אֲנִ֣י כְפִ֣יךָ לָאֵ֑ל    מֵ֝חֹ֗מֶר קֹרַ֥צְתִּי גַם־אָֽנִי׃
ז הִנֵּ֣ה אֵֽ֭מָתִי לֹ֣א תְבַֽעֲתֶ֑ךָּ    וְ֝אַכְפִּ֗י עָלֶ֥יךָ לֹֽא־יִכְבָּֽד׃
ח אַ֭ךְ אָמַ֣רְתָּ בְאָזְנָ֑י    וְק֖וֹל מִלִּ֣ין אֶשְׁמָֽע׃
ט זַ֥ךְ אֲנִ֗י בְּֽלִ֫י פָ֥שַׁע    חַ֥ף אָֽנֹכִ֑י וְלֹ֖א עָו֣‍ֹן לִֽי׃
י הֵ֣ן תְּ֭נוּאוֹת עָלַ֣י יִמְצָ֑א    יַחְשְׁבֵ֖נִי לְאוֹיֵ֣ב לֽוֹ׃
יא יָשֵׂ֣ם בַּסַּ֣ד רַגְלָ֑י    יִ֝שְׁמֹ֗ר כָּל־אָרְחֹתָֽי׃
יב הֶן־זֹ֣את לֹֽא־צָדַ֣קְתָּ אֶֽעֱנֶ֑ךָּ    כִּֽי־יִרְבֶּ֥ה אֱ֝ל֗וֹהַּ מֵֽאֱנֽוֹשׁ׃
יג מַ֭דּוּעַ אֵלָ֣יו רִיב֑וֹתָ    כִּ֥י כָל־דְּ֝בָרָ֗יו לֹ֣א יַֽעֲנֶֽה׃
יד כִּֽי־בְאַחַ֥ת יְדַבֶּר־אֵ֑ל    וּ֝בִשְׁתַּ֗יִם לֹ֣א יְשׁוּרֶֽנָּה׃
טו בַּֽחֲל֤וֹם ׀ חֶזְי֬וֹן לַ֗יְלָה בִּנְפֹ֣ל תַּ֭רְדֵּמָה עַל־אֲנָשִׁ֑ים    בִּ֝תְנוּמ֗וֹת עֲלֵ֣י מִשְׁכָּֽב׃
טז אָ֣ז יִ֭גְלֶה אֹ֣זֶן אֲנָשִׁ֑ים    וּבְמֹ֖סָרָ֣ם יַחְתֹּֽם׃
יז לְ֭הָסִיר אָדָ֣ם מַֽעֲשֶׂ֑ה    וְגֵוָ֖ה מִגֶּ֣בֶר יְכַסֶּֽה׃
יח יַחְשֹׂ֣ךְ נַ֭פְשׁוֹ מִנִּי־שָׁ֑חַת    וְ֝חַיָּת֗וֹ מֵֽעֲבֹ֥ר בַּשָּֽׁלַח׃
יט וְהוּכַ֣ח בְּ֭מַכְאוֹב עַל־מִשְׁכָּב֑וֹ    וריב (וְר֖וֹב) עֲצָמָ֣יו אֵתָֽן׃
כ וְזִֽהֲמַ֣תּוּ חַיָּת֣וֹ לָ֑חֶם    וְ֝נַפְשׁ֗וֹ מַֽאֲכַ֥ל תַּֽאֲוָֽה׃
כא יִ֣כֶל בְּשָׂר֣וֹ מֵרֹ֑אִי    ושפי (וְשֻׁפּ֥וּ) עַ֝צְמֹתָ֗יו לֹ֣א רֻאּֽוּ׃
כב וַתִּקְרַ֣ב לַשַּׁ֣חַת נַפְשׁ֑וֹ    וְ֝חַיָּת֗וֹ לַֽמְמִתִֽים׃
כג אִם־יֵ֤שׁ עָלָ֨יו ׀ מַלְאָ֗ךְ מֵלִ֗יץ אֶחָ֥ד מִנִּי־אָ֑לֶף    לְהַגִּ֖יד לְאָדָ֣ם יָשְׁרֽוֹ׃
כד וַיְחֻנֶּ֗נּוּ וַיֹּ֗אמֶר פְּ֭דָעֵהוּ מֵרֶ֥דֶת שָׁ֗חַת    מָצָ֥אתִי כֹֽפֶר׃
כה רֻֽטְפַ֣שׁ בְּשָׂר֣וֹ מִנֹּ֑עַר    יָ֝שׁ֗וּב לִימֵ֥י עֲלוּמָֽיו׃
כו יֶעְתַּ֤ר אֶל־אֱל֨וֹהַּ ׀ וַיִּרְצֵ֗הוּ    וַיַּ֣רְא פָּ֭נָיו בִּתְרוּעָ֑ה
וַיָּ֥שֶׁב לֶֽ֝אֱנ֗וֹשׁ    צִדְקָתֽוֹ׃
כז יָשֹׁ֤ר ׀ עַל־אֲנָשִׁ֗ים וַיֹּ֗אמֶר חָ֭טָאתִי וְיָשָׁ֥ר הֶֽעֱוֵ֗יתִי    וְלֹא־שָׁ֥וָה לִֽי׃
כח פָּדָ֣ה נפשי (נַ֭פְשׁוֹ) מֵֽעֲבֹ֣ר בַּשָּׁ֑חַת    וחיתי (וְ֝חַיָּת֗וֹ) בָּא֥וֹר תִּרְאֶֽה׃
כט הֶן־כָּל־אֵ֭לֶּה יִפְעַל־אֵ֑ל    פַּֽעֲמַ֖יִם שָׁל֣וֹשׁ עִם־גָּֽבֶר׃
ל לְהָשִׁ֣יב נַ֭פְשׁוֹ מִנִּי־שָׁ֑חַת    לֵ֝א֗וֹר בְּא֣וֹר הַֽחַיִּֽים׃
לא הַקְשֵׁ֖ב אִיּ֥וֹב שְֽׁמַֽע־לִ֑י    הַֽ֝חֲרֵ֗שׁ וְאָֽנֹכִ֥י אֲדַבֵּֽר׃
לב אִם־יֵשׁ־מִלִּ֥ין הֲשִׁיבֵ֑נִי    דַּ֝בֵּ֗ר כִּֽי־חָפַ֥צְתִּי צַדְּקֶֽךָּ׃
לג אִם־אַ֭יִן אַתָּ֥ה שְֽׁמַֽע־לִ֑י    הַֽ֝חֲרֵ֗שׁ וַֽאֲאַלֶּפְךָ֥ חָכְמָֽה׃
