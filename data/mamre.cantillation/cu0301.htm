ויקרא

א,א וַיִּקְרָ֖א אֶל־מֹשֶׁ֑ה וַיְדַבֵּ֤ר יְהוָה֙ אֵלָ֔יו מֵאֹ֥הֶל מוֹעֵ֖ד לֵאמֹֽר׃ וּקְרָא, לְמֹשֶׁה; וּמַלֵּיל יְיָ עִמֵּיהּ, מִמַּשְׁכַּן זִמְנָא לְמֵימַר. 
א,ב דַּבֵּ֞ר אֶל־בְּנֵ֤י יִשְׂרָאֵל֙ וְאָֽמַרְתָּ֣ אֲלֵהֶ֔ם אָדָ֗ם כִּֽי־יַקְרִ֥יב מִכֶּ֛ם קָרְבָּ֖ן לַֽיהוָ֑ה מִן־הַבְּהֵמָ֗ה מִן־הַבָּקָר֙ וּמִן־הַצֹּ֔אן תַּקְרִ֖יבוּ אֶת־קָרְבַּנְכֶֽם׃ מַלֵּיל עִם בְּנֵי יִשְׂרָאֵל, וְתֵימַר לְהוֹן, אֲנָשׁ אֲרֵי יְקָרֵיב מִנְּכוֹן קֻרְבָּנָא, קֳדָם יְיָ--מִן בְּעִירָא, מִן תּוֹרֵי וּמִן עָנָא, תְּקָרְבוּן, יָת קֻרְבָּנְכוֹן. 
א,ג אִם־עֹלָ֤ה קָרְבָּנוֹ֙ מִן־הַבָּקָ֔ר זָכָ֥ר תָּמִ֖ים יַקְרִיבֶ֑נּוּ אֶל־פֶּ֜תַח אֹ֤הֶל מוֹעֵד֙ יַקְרִ֣יב אֹת֔וֹ לִרְצֹנ֖וֹ לִפְנֵ֥י יְהוָֽה׃ אִם עֲלָתָא קֻרְבָּנֵיהּ מִן תּוֹרֵי, דְּכַר שְׁלִים יְקָרְבִנֵּיהּ; לִתְרַע מַשְׁכַּן זִמְנָא, יְקָרֵיב יָתֵיהּ, לְרַעֲוָא לֵיהּ, קֳדָם יְיָ. 
א,ד וְסָמַ֣ךְ יָד֔וֹ עַ֖ל רֹ֣אשׁ הָֽעֹלָ֑ה וְנִרְצָ֥ה ל֖וֹ לְכַפֵּ֥ר עָלָֽיו׃ וְיִסְמוֹךְ יְדֵיהּ, עַל רֵישׁ עֲלָתָא; וְיִתְרְעֵי לֵיהּ, לְכַפָּרָא עֲלוֹהִי. 
א,ה וְשָׁחַ֛ט אֶת־בֶּ֥ן הַבָּקָ֖ר לִפְנֵ֣י יְהוָ֑ה וְ֠הִקְרִיבוּ בְּנֵ֨י אַֽהֲרֹ֤ן הַכֹּֽהֲנִים֙ אֶת־הַדָּ֔ם וְזָֽרְק֨וּ אֶת־הַדָּ֤ם עַל־הַמִּזְבֵּ֨חַ֙ סָבִ֔יב אֲשֶׁר־פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ וְיִכּוֹס יָת בַּר תּוֹרֵי, קֳדָם יְיָ; וִיקָרְבוּן בְּנֵי אַהֲרוֹן כָּהֲנַיָּא, יָת דְּמָא, וְיִזְרְקוּן יָת דְּמָא עַל מַדְבְּחָא סְחוֹר סְחוֹר, דְּבִתְרַע מַשְׁכַּן זִמְנָא. 
א,ו וְהִפְשִׁ֖יט אֶת־הָֽעֹלָ֑ה וְנִתַּ֥ח אֹתָ֖הּ לִנְתָחֶֽיהָ׃ וְיַשְׁלַח, יָת עֲלָתָא; וִיפַלֵּיג יָתַהּ, לְאֶבְרַהָא. 
א,ז וְ֠נָֽתְנוּ בְּנֵ֨י אַֽהֲרֹ֧ן הַכֹּהֵ֛ן אֵ֖שׁ עַל־הַמִּזְבֵּ֑חַ וְעָֽרְכ֥וּ עֵצִ֖ים עַל־הָאֵֽשׁ׃ וְיִתְּנוּן בְּנֵי אַהֲרוֹן כָּהֲנָא, אִישָׁתָא--עַל מַדְבְּחָא; וִיסַדְּרוּן אָעַיָּא, עַל אִישָׁתָא. 
א,ח וְעָֽרְכ֗וּ בְּנֵ֤י אַֽהֲרֹן֙ הַכֹּ֣הֲנִ֔ים אֵ֚ת הַנְּתָחִ֔ים אֶת־הָרֹ֖אשׁ וְאֶת־הַפָּ֑דֶר עַל־הָֽעֵצִים֙ אֲשֶׁ֣ר עַל־הָאֵ֔שׁ אֲשֶׁ֖ר עַל־הַמִּזְבֵּֽחַ׃ וִיסַדְּרוּן, בְּנֵי אַהֲרוֹן כָּהֲנַיָּא, יָת אֶבְרַיָּא, יָת רֵישָׁא וְיָת תַּרְבָּא--עַל אָעַיָּא דְּעַל אִישָׁתָא, דְּעַל מַדְבְּחָא. 
א,ט וְקִרְבּ֥וֹ וּכְרָעָ֖יו יִרְחַ֣ץ בַּמָּ֑יִם וְהִקְטִ֨יר הַכֹּהֵ֤ן אֶת־הַכֹּל֙ הַמִּזְבֵּ֔חָה עֹלָ֛ה אִשֵּׁ֥ה רֵֽיחַ־נִיח֖וֹחַ לַֽיהוָֽה׃ {ס} וְגַוֵּיהּ וּכְרָעוֹהִי, יְחַלֵּיל בְּמַיָּא; וְיַסֵּיק כָּהֲנָא יָת כּוֹלָא לְמַדְבְּחָא, עֲלָתָא קֻרְבַּן דְּמִתְקַבַּל בְּרַעֲוָא קֳדָם יְיָ. {ס}
א,י וְאִם־מִן־הַצֹּ֨אן קָרְבָּנ֧וֹ מִן־הַכְּשָׂבִ֛ים א֥וֹ מִן־הָֽעִזִּ֖ים לְעֹלָ֑ה זָכָ֥ר תָּמִ֖ים יַקְרִיבֶֽנּוּ׃ וְאִם מִן עָנָא קֻרְבָּנֵיהּ מִן אִמְּרַיָּא אוֹ מִן בְּנֵי עִזַּיָּא, לַעֲלָתָא--דְּכַר שְׁלִים, יְקָרְבִנֵּיהּ. 
א,יא וְשָׁחַ֨ט אֹת֜וֹ עַ֣ל יֶ֧רֶךְ הַמִּזְבֵּ֛חַ צָפֹ֖נָה לִפְנֵ֣י יְהוָ֑ה וְזָֽרְק֡וּ בְּנֵי֩ אַֽהֲרֹ֨ן הַכֹּֽהֲנִ֧ים אֶת־דָּמ֛וֹ עַל־הַמִּזְבֵּ֖חַ סָבִֽיב׃ וְיִכּוֹס יָתֵיהּ עַל שִׁדָּא דְּמַדְבְּחָא, צִפּוּנָא--קֳדָם יְיָ; וְיִזְרְקוּן בְּנֵי אַהֲרוֹן כָּהֲנַיָּא יָת דְּמֵיהּ, עַל מַדְבְּחָא--סְחוֹר סְחוֹר. 
א,יב וְנִתַּ֤ח אֹתוֹ֙ לִנְתָחָ֔יו וְאֶת־רֹאשׁ֖וֹ וְאֶת־פִּדְר֑וֹ וְעָרַ֤ךְ הַכֹּהֵן֙ אֹתָ֔ם עַל־הָֽעֵצִים֙ אֲשֶׁ֣ר עַל־הָאֵ֔שׁ אֲשֶׁ֖ר עַל־הַמִּזְבֵּֽחַ׃ וִיפַלֵּיג יָתֵיהּ לְאֶבְרוֹהִי, וְיָת רֵישֵׁיהּ וְיָת תַּרְבֵּיהּ; וְיַסְדַּר כָּהֲנָא, יָתְהוֹן, עַל אָעַיָּא דְּעַל אִישָׁתָא, דְּעַל מַדְבְּחָא. 
א,יג וְהַקֶּ֥רֶב וְהַכְּרָעַ֖יִם יִרְחַ֣ץ בַּמָּ֑יִם וְהִקְרִ֨יב הַכֹּהֵ֤ן אֶת־הַכֹּל֙ וְהִקְטִ֣יר הַמִּזְבֵּ֔חָה עֹלָ֣ה ה֗וּא אִשֵּׁ֛ה רֵ֥יחַ נִיחֹ֖חַ לַֽיהוָֽה׃ {פ} וְגַוָּא וּכְרָעַיָּא, יְחַלֵּיל בְּמַיָּא; וִיקָרֵיב כָּהֲנָא יָת כּוֹלָא, וְיַסֵּיק לְמַדְבְּחָא--עֲלָתָא הוּא קֻרְבַּן דְּמִתְקַבַּל בְּרַעֲוָא, קֳדָם יְיָ. {פ}
א,יד וְאִ֧ם מִן־הָע֛וֹף עֹלָ֥ה קָרְבָּנ֖וֹ לַֽיהוָ֑ה וְהִקְרִ֣יב מִן־הַתֹּרִ֗ים א֛וֹ מִן־בְּנֵ֥י הַיּוֹנָ֖ה אֶת־קָרְבָּנֽוֹ׃ וְאִם מִן עוֹפָא עֲלָתָא קֻרְבָּנֵיהּ, קֳדָם יְיָ:  וִיקָרֵיב מִן שַׁפְנִינַיָּא, אוֹ מִן בְּנֵי יוֹנָה--יָת קֻרְבָּנֵיהּ. 
א,טו וְהִקְרִיב֤וֹ הַכֹּהֵן֙ אֶל־הַמִּזְבֵּ֔חַ וּמָלַק֙ אֶת־רֹאשׁ֔וֹ וְהִקְטִ֖יר הַמִּזְבֵּ֑חָה וְנִמְצָ֣ה דָמ֔וֹ עַ֖ל קִ֥יר הַמִּזְבֵּֽחַ׃ וִיקָרְבִנֵּיהּ כָּהֲנָא, לְמַדְבְּחָא, וְיִמְלוֹק יָת רֵישֵׁיהּ, וְיַסֵּיק לְמַדְבְּחָא; וְיִתְמְצֵי דְּמֵיהּ, עַל כֹּתֶל מַדְבְּחָא. 
א,טז וְהֵסִ֥יר אֶת־מֻרְאָת֖וֹ בְּנֹֽצָתָ֑הּ וְהִשְׁלִ֨יךְ אֹתָ֜הּ אֵ֤צֶל הַמִּזְבֵּ֨חַ֙ קֵ֔דְמָה אֶל־מְק֖וֹם הַדָּֽשֶׁן׃ וְיַעְדֵּי יָת זְפָקֵיהּ, בְּאֻכְלֵיהּ; וְיִרְמֵי יָתַהּ בִּסְטַר מַדְבְּחָא, קִדּוּמָא--בַּאֲתַר, דְּמַקְרִין קִטְמָא. 
א,יז וְשִׁסַּ֨ע אֹת֣וֹ בִכְנָפָיו֮ לֹ֣א יַבְדִּיל֒ וְהִקְטִ֨יר אֹת֤וֹ הַכֹּהֵן֙ הַמִּזְבֵּ֔חָה עַל־הָֽעֵצִ֖ים אֲשֶׁ֣ר עַל־הָאֵ֑שׁ עֹלָ֣ה ה֗וּא אִשֵּׁ֛ה רֵ֥יחַ נִיחֹ֖חַ לַֽיהוָֽה׃ {ס} וִיפָרֵיק יָתֵיהּ בְּכַנְפוֹהִי, לָא יַפְרֵישׁ, וְיַסֵּיק יָתֵיהּ כָּהֲנָא לְמַדְבְּחָא, עַל אָעַיָּא דְּעַל אִישָׁתָא:  עֲלָתָא הוּא, קֻרְבַּן דְּמִתְקַבַּל בְּרַעֲוָא--קֳדָם יְיָ. {ס}
ב,א וְנֶ֗פֶשׁ כִּֽי־תַקְרִ֞יב קָרְבַּ֤ן מִנְחָה֙ לַֽיהוָ֔ה סֹ֖לֶת יִֽהְיֶ֣ה קָרְבָּנ֑וֹ וְיָצַ֤ק עָלֶ֨יהָ֙ שֶׁ֔מֶן וְנָתַ֥ן עָלֶ֖יהָ לְבֹנָֽה׃ וַאֲנָשׁ, אֲרֵי יְקָרֵיב קֻרְבַּן מִנְחָתָא קֳדָם יְיָ--סֻלְתָּא, יְהֵי קֻרְבָּנֵיהּ; וִירִיק עֲלַהּ מִשְׁחָא, וְיִתֵּין עֲלַהּ לְבוֹנְתָא. 
ב,ב וֶֽהֱבִיאָ֗הּ אֶל־בְּנֵ֣י אַֽהֲרֹן֮ הַכֹּֽהֲנִים֒ וְקָמַ֨ץ מִשָּׁ֜ם מְלֹ֣א קֻמְצ֗וֹ מִסָּלְתָּהּ֙ וּמִשַּׁמְנָ֔הּ עַ֖ל כָּל־לְבֹֽנָתָ֑הּ וְהִקְטִ֨יר הַכֹּהֵ֜ן אֶת־אַזְכָּֽרָתָהּ֙ הַמִּזְבֵּ֔חָה אִשֵּׁ֛ה רֵ֥יחַ נִיחֹ֖חַ לַֽיהוָֽה׃ וְיַיְתֵינַהּ, לְוָת בְּנֵי אַהֲרוֹן כָּהֲנַיָּא, וְיִקְמוֹץ מִתַּמָּן מְלֵי קֻמְצֵיהּ מִסֻּלְתַּהּ וּמִמִּשְׁחַהּ, עַל כָּל לְבוֹנְתַהּ; וְיַסֵּיק כָּהֲנָא יָת אַדְכָרְתַהּ, לְמַדְבְּחָא--קֻרְבַּן דְּמִתְקַבַּל בְּרַעֲוָא, קֳדָם יְיָ. 
ב,ג וְהַנּוֹתֶ֨רֶת֙ מִן־הַמִּנְחָ֔ה לְאַֽהֲרֹ֖ן וּלְבָנָ֑יו קֹ֥דֶשׁ קָֽדָשִׁ֖ים מֵֽאִשֵּׁ֥י יְהוָֽה׃ {ס} וּדְיִשְׁתְּאַר, מִן מִנְחָתָא--לְאַהֲרוֹן, וְלִבְנוֹהִי:  קֹדֶשׁ קֻדְשִׁין, מִקֻּרְבָּנַיָּא דַּייָ. {ס}
ב,ד וְכִ֥י תַקְרִ֛ב קָרְבַּ֥ן מִנְחָ֖ה מַֽאֲפֵ֣ה תַנּ֑וּר סֹ֣לֶת חַלּ֤וֹת מַצֹּת֙ בְּלוּלֹ֣ת בַּשֶּׁ֔מֶן וּרְקִיקֵ֥י מַצּ֖וֹת מְשֻׁחִ֥ים בַּשָּֽׁמֶן׃ {ס} וַאֲרֵי תְּקָרֵיב קֻרְבַּן מִנְחָתָא, מַאֲפֵה תַּנּוּר--סֹלֶת גְּרִיצָן פַּטִּירָן דְּפִילָן בִּמְשַׁח, וְאֶסְפּוֹגִין פַּטִּירִין דִּמְשִׁיחִין בִּמְשַׁח. {ס}
ב,ה וְאִם־מִנְחָ֥ה עַל־הַֽמַּחֲבַ֖ת קָרְבָּנֶ֑ךָ סֹ֛לֶת בְּלוּלָ֥ה בַשֶּׁ֖מֶן מַצָּ֥ה תִֽהְיֶֽה׃ וְאִם מִנְחָתָא עַל מַסְרֵיתָא, קֻרְבָּנָךְ--סֻלְתָּא דְּפִילָא בִּמְשַׁח, פַּטִּיר תְּהֵי. 
ב,ו פָּת֤וֹת אֹתָהּ֙ פִּתִּ֔ים וְיָֽצַקְתָּ֥ עָלֶ֖יהָ שָׁ֑מֶן מִנְחָ֖ה הִֽוא׃ {ס} בַּצַּע יָתַהּ בִּצּוּעִין, וּתְרִיק עֲלַהּ מִשְׁחָא; מִנְחָתָא, הִיא. {ס}
ב,ז וְאִם־מִנְחַ֥ת מַרְחֶ֖שֶׁת קָרְבָּנֶ֑ךָ סֹ֥לֶת בַּשֶּׁ֖מֶן תֵּֽעָשֶֽׂה׃ וְאִם מִנְחָתָא רָדְתָא, קֻרְבָּנָךְ--סֹלֶת בִּמְשַׁח, תִּתְעֲבֵיד. 
ב,ח וְהֵֽבֵאתָ֣ אֶת־הַמִּנְחָ֗ה אֲשֶׁ֧ר יֵֽעָשֶׂ֛ה מֵאֵ֖לֶּה לַֽיהוָ֑ה וְהִקְרִיבָהּ֙ אֶל־הַכֹּהֵ֔ן וְהִגִּישָׁ֖הּ אֶל־הַמִּזְבֵּֽחַ׃ וְתַיְתֵי יָת מִנְחָתָא, דְּתִתְעֲבֵיד מֵאִלֵּין--קֳדָם יְיָ; וִיקָרְבִנַּהּ, לְכָהֲנָא, וִיקָרְבִנַּהּ, לְמַדְבְּחָא. 
ב,ט וְהֵרִ֨ים הַכֹּהֵ֤ן מִן־הַמִּנְחָה֙ אֶת־אַזְכָּ֣רָתָ֔הּ וְהִקְטִ֖יר הַמִּזְבֵּ֑חָה אִשֵּׁ֛ה רֵ֥יחַ נִיחֹ֖חַ לַֽיהוָֽה׃ וְיַפְרֵישׁ כָּהֲנָא מִן מִנְחָתָא יָת אַדְכָרְתַהּ, וְיַסֵּיק לְמַדְבְּחָא--קֻרְבַּן דְּמִתְקַבַּל בְּרַעֲוָא, קֳדָם יְיָ. 
ב,י וְהַנּוֹתֶ֨רֶת֙ מִן־הַמִּנְחָ֔ה לְאַֽהֲרֹ֖ן וּלְבָנָ֑יו קֹ֥דֶשׁ קָֽדָשִׁ֖ים מֵֽאִשֵּׁ֥י יְהוָֽה׃ וּדְיִשְׁתְּאַר, מִן מִנְחָתָא--לְאַהֲרוֹן, וְלִבְנוֹהִי:  קֹדֶשׁ קֻדְשִׁין, מִקֻּרְבָּנַיָּא דַּייָ. 
ב,יא כָּל־הַמִּנְחָ֗ה אֲשֶׁ֤ר תַּקְרִ֨יבוּ֙ לַֽיהוָ֔ה לֹ֥א תֵֽעָשֶׂ֖ה חָמֵ֑ץ כִּ֤י כָל־שְׂאֹר֙ וְכָל־דְּבַ֔שׁ לֹֽא־תַקְטִ֧ירוּ מִמֶּ֛נּוּ אִשֶּׁ֖ה לַֽיהוָֽה׃ כָּל מִנְחָתָא, דִּתְקָרְבוּן קֳדָם יְיָ--לָא תִּתְעֲבֵיד, חֲמִיעַ:  אֲרֵי כָּל חֲמִיר וְכָל דְּבַשׁ, לָא תַּסְּקוּן מִנֵּיהּ קֻרְבָּנָא קֳדָם יְיָ. 
ב,יב קָרְבַּ֥ן רֵאשִׁ֛ית תַּקְרִ֥יבוּ אֹתָ֖ם לַֽיהוָ֑ה וְאֶל־הַמִּזְבֵּ֥חַ לֹֽא־יַעֲל֖וּ לְרֵ֥יחַ נִיחֹֽחַ׃ קֻרְבַּן קַדְמַאי תְּקָרְבוּן יָתְהוֹן, קֳדָם יְיָ; וּלְמַדְבְּחָא לָא יִתַּסְקוּן, לְאִתְקַבָּלָא בְּרַעֲוָא. 
ב,יג וְכָל־קָרְבַּ֣ן מִנְחָֽתְךָ֮ בַּמֶּ֣לַח תִּמְלָח֒ וְלֹ֣א תַשְׁבִּ֗ית מֶ֚לַח בְּרִ֣ית אֱלֹהֶ֔יךָ מֵעַ֖ל מִנְחָתֶ֑ךָ עַ֥ל כָּל־קָרְבָּֽנְךָ֖ תַּקְרִ֥יב מֶֽלַח׃ {ס} וְכָל קֻרְבַּן מִנְחָתָךְ, בְּמִלְחָא תִּמְלַח, וְלָא תְּבַטֵּיל מְלַח קְיָם אֱלָהָךְ, מֵעַל מִנְחָתָךְ; עַל כָּל קֻרְבָּנָךְ, תְּקָרֵיב מִלְחָא. {ס}
ב,יד וְאִם־תַּקְרִ֛יב מִנְחַ֥ת בִּכּוּרִ֖ים לַֽיהוָ֑ה אָבִ֞יב קָל֤וּי בָּאֵשׁ֙ גֶּ֣רֶשׂ כַּרְמֶ֔ל תַּקְרִ֕יב אֵ֖ת מִנְחַ֥ת בִּכּוּרֶֽיךָ׃ וְאִם תְּקָרֵיב מִנְחַת בִּכּוּרִין, קֳדָם יְיָ--אֲבִיב קְלֵי בְּנוּר, פֵּירוּכָן רַכִּיכָן, תְּקָרֵיב, יָת מִנְחַת בִּכּוּרָךְ. 
ב,טו וְנָֽתַתָּ֤ עָלֶ֨יהָ֙ שֶׁ֔מֶן וְשַׂמְתָּ֥ עָלֶ֖יהָ לְבֹנָ֑ה מִנְחָ֖ה הִֽוא׃ וְתִתֵּין עֲלַהּ מִשְׁחָא, וּתְשַׁוֵּי עֲלַהּ לְבוֹנְתָא; מִנְחָתָא, הִיא. 
ב,טז וְהִקְטִ֨יר הַכֹּהֵ֜ן אֶת־אַזְכָּֽרָתָ֗הּ מִגִּרְשָׂהּ֙ וּמִשַּׁמְנָ֔הּ עַ֖ל כָּל־לְבֹֽנָתָ֑הּ אִשֶּׁ֖ה לַֽיהוָֽה׃ {פ} וְיַסֵּיק כָּהֲנָא יָת אַדְכָרְתַהּ, מִגִּרְסַהּ וּמִמִּשְׁחַהּ, עַל, כָּל לְבוֹנְתַהּ--קֻרְבָּנָא, קֳדָם יְיָ. {פ}
ג,א וְאִם־זֶ֥בַח שְׁלָמִ֖ים קָרְבָּנ֑וֹ אִ֤ם מִן־הַבָּקָר֙ ה֣וּא מַקְרִ֔יב אִם־זָכָר֙ אִם־נְקֵבָ֔ה תָּמִ֥ים יַקְרִיבֶ֖נּוּ לִפְנֵ֥י יְהוָֽה׃ וְאִם נִכְסַת קֻדְשַׁיָּא, קֻרְבָּנֵיהּ--אִם מִן תּוֹרֵי, הוּא מְקָרֵיב, אִם דְּכַר אִם נֻקְבָּא, שְׁלִים יְקָרְבִנֵּיהּ קֳדָם יְיָ. 
ג,ב וְסָמַ֤ךְ יָדוֹ֙ עַל־רֹ֣אשׁ קָרְבָּנ֔וֹ וּשְׁחָט֕וֹ פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד וְזָֽרְק֡וּ בְּנֵי֩ אַֽהֲרֹ֨ן הַכֹּֽהֲנִ֧ים אֶת־הַדָּ֛ם עַל־הַמִּזְבֵּ֖חַ סָבִֽיב׃ וְיִסְמוֹךְ יְדֵיהּ, עַל רֵישׁ קֻרְבָּנֵיהּ, וְיִכְּסִנֵּיהּ, בִּתְרַע מַשְׁכַּן זִמְנָא; וְיִזְרְקוּן בְּנֵי אַהֲרוֹן כָּהֲנַיָּא יָת דְּמָא, עַל מַדְבְּחָא--סְחוֹר סְחוֹר. 
ג,ג וְהִקְרִיב֙ מִזֶּ֣בַח הַשְּׁלָמִ֔ים אִשֶּׁ֖ה לַֽיהוָ֑ה אֶת־הַחֵ֨לֶב֙ הַֽמְכַסֶּ֣ה אֶת־הַקֶּ֔רֶב וְאֵת֙ כָּל־הַחֵ֔לֶב אֲשֶׁ֖ר עַל־הַקֶּֽרֶב׃ וִיקָרֵיב מִנִּכְסַת קֻדְשַׁיָּא, קֻרְבָּנָא קֳדָם יְיָ--יָת תַּרְבָּא, דְּחָפֵי יָת גַּוָּא, וְיָת כָּל תַּרְבָּא, דְּעַל גַּוָּא. 
ג,ד וְאֵת֙ שְׁתֵּ֣י הַכְּלָיֹ֔ת וְאֶת־הַחֵ֨לֶב֙ אֲשֶׁ֣ר עֲלֵהֶ֔ן אֲשֶׁ֖ר עַל־הַכְּסָלִ֑ים וְאֶת־הַיֹּתֶ֨רֶת֙ עַל־הַכָּבֵ֔ד עַל־הַכְּלָי֖וֹת יְסִירֶֽנָּה׃ וְיָת, תַּרְתֵּין כּוֹלְיָן, וְיָת תַּרְבָּא דַּעֲלֵיהוֹן, דְּעַל גִּסְסַיָּא; וְיָת חַצְרָא, דְּעַל כַּבְדָּא, עַל כּוֹלְיָתָא, יַעְדֵּינַהּ. 
ג,ה וְהִקְטִ֨ירוּ אֹת֤וֹ בְנֵֽי־אַהֲרֹן֙ הַמִּזְבֵּ֔חָה עַל־הָ֣עֹלָ֔ה אֲשֶׁ֥ר עַל־הָֽעֵצִ֖ים אֲשֶׁ֣ר עַל־הָאֵ֑שׁ אִשֵּׁ֛ה רֵ֥יחַ נִיחֹ֖חַ לַֽיהוָֽה׃ {פ} וְיַסְּקוּן יָתֵיהּ בְּנֵי אַהֲרוֹן, לְמַדְבְּחָא, עַל עֲלָתָא, דְּעַל אָעַיָּא דְּעַל אִישָׁתָא--קֻרְבַּן דְּמִתְקַבַּל בְּרַעֲוָא, קֳדָם יְיָ. {פ}
ג,ו וְאִם־מִן־הַצֹּ֧אן קָרְבָּנ֛וֹ לְזֶ֥בַח שְׁלָמִ֖ים לַֽיהוָ֑ה זָכָר֙ א֣וֹ נְקֵבָ֔ה תָּמִ֖ים יַקְרִיבֶֽנּוּ׃ וְאִם מִן עָנָא קֻרְבָּנֵיהּ לְנִכְסַת קֻדְשַׁיָּא, קֳדָם יְיָ--דְּכַר אוֹ נֻקְבָּא, שְׁלִים יְקָרְבִנֵּיהּ. 
ג,ז אִם־כֶּ֥שֶׂב הֽוּא־מַקְרִ֖יב אֶת־קָרְבָּנ֑וֹ וְהִקְרִ֥יב אֹת֖וֹ לִפְנֵ֥י יְהוָֽה׃ אִם אִמַּר הוּא מְקָרֵיב, יָת קֻרְבָּנֵיהּ--וִיקָרֵיב יָתֵיהּ, קֳדָם יְיָ. 
ג,ח וְסָמַ֤ךְ אֶת־יָדוֹ֙ עַל־רֹ֣אשׁ קָרְבָּנ֔וֹ וְשָׁחַ֣ט אֹת֔וֹ לִפְנֵ֖י אֹ֣הֶל מוֹעֵ֑ד וְ֠זָֽרְקוּ בְּנֵ֨י אַֽהֲרֹ֧ן אֶת־דָּמ֛וֹ עַל־הַמִּזְבֵּ֖חַ סָבִֽיב׃ וְיִסְמוֹךְ יָת יְדֵיהּ, עַל רֵישׁ קֻרְבָּנֵיהּ, וְיִכּוֹס יָתֵיהּ, קֳדָם מַשְׁכַּן זִמְנָא; וְיִזְרְקוּן בְּנֵי אַהֲרוֹן יָת דְּמֵיהּ, עַל מַדְבְּחָא--סְחוֹר סְחוֹר. 
ג,ט וְהִקְרִ֨יב מִזֶּ֣בַח הַשְּׁלָמִים֮ אִשֶּׁ֣ה לַֽיהוָה֒ חֶלְבּוֹ֙ הָֽאַלְיָ֣ה תְמִימָ֔ה לְעֻמַּ֥ת הֶֽעָצֶ֖ה יְסִירֶ֑נָּה וְאֶת־הַחֵ֨לֶב֙ הַֽמְכַסֶּ֣ה אֶת־הַקֶּ֔רֶב וְאֵת֙ כָּל־הַחֵ֔לֶב אֲשֶׁ֖ר עַל־הַקֶּֽרֶב׃ וִיקָרֵיב מִנִּכְסַת קֻדְשַׁיָּא, קֻרְבָּנָא קֳדָם יְיָ--תַּרְבֵּיהּ אַלְיְתָא שַׁלְמְתָא, לָקֳבֵיל שַׁזַרְתָּא יַעְדֵּינַהּ; וְיָת תַּרְבָּא, דְּחָפֵי יָת גַּוָּא, וְיָת כָּל תַּרְבָּא, דְּעַל גַּוָּא. 
ג,י וְאֵת֙ שְׁתֵּ֣י הַכְּלָיֹ֔ת וְאֶת־הַחֵ֨לֶב֙ אֲשֶׁ֣ר עֲלֵהֶ֔ן אֲשֶׁ֖ר עַל־הַכְּסָלִ֑ים וְאֶת־הַיֹּתֶ֨רֶת֙ עַל־הַכָּבֵ֔ד עַל־הַכְּלָיֹ֖ת יְסִירֶֽנָּה׃ וְיָת, תַּרְתֵּין כּוֹלְיָן, וְיָת תַּרְבָּא דַּעֲלֵיהוֹן, דְּעַל גִּסְסַיָּא; וְיָת חַצְרָא, דְּעַל כַּבְדָּא, עַל כּוֹלְיָתָא, יַעְדֵּינַהּ. 
ג,יא וְהִקְטִיר֥וֹ הַכֹּהֵ֖ן הַמִּזְבֵּ֑חָה לֶ֥חֶם אִשֶּׁ֖ה לַֽיהוָֽה׃ {פ} וְיַסְּקִנֵּיהּ כָּהֲנָא, לְמַדְבְּחָא--לְחֵים קֻרְבָּנָא, קֳדָם יְיָ. {פ}
ג,יב וְאִ֥ם עֵ֖ז קָרְבָּנ֑וֹ וְהִקְרִיב֖וֹ לִפְנֵ֥י יְהוָֽה׃ וְאִם מִן בְּנֵי עִזַּיָּא, קֻרְבָּנֵיהּ--וִיקָרְבִנֵּיהּ, קֳדָם יְיָ. 
ג,יג וְסָמַ֤ךְ אֶת־יָדוֹ֙ עַל־רֹאשׁ֔וֹ וְשָׁחַ֣ט אֹת֔וֹ לִפְנֵ֖י אֹ֣הֶל מוֹעֵ֑ד וְ֠זָֽרְקוּ בְּנֵ֨י אַֽהֲרֹ֧ן אֶת־דָּמ֛וֹ עַל־הַמִּזְבֵּ֖חַ סָבִֽיב׃ וְיִסְמוֹךְ יָת יְדֵיהּ, עַל רֵישֵׁיהּ, וְיִכּוֹס יָתֵיהּ, קֳדָם מַשְׁכַּן זִמְנָא; וְיִזְרְקוּן בְּנֵי אַהֲרוֹן יָת דְּמֵיהּ, עַל מַדְבְּחָא--סְחוֹר סְחוֹר. 
ג,יד וְהִקְרִ֤יב מִמֶּ֨נּוּ֙ קָרְבָּנ֔וֹ אִשֶּׁ֖ה לַֽיהוָ֑ה אֶת־הַחֵ֨לֶב֙ הַֽמְכַסֶּ֣ה אֶת־הַקֶּ֔רֶב וְאֵת֙ כָּל־הַחֵ֔לֶב אֲשֶׁ֖ר עַל־הַקֶּֽרֶב׃ וִיקָרֵיב מִנֵּיהּ קֻרְבָּנֵיהּ, קֻרְבָּנָא קֳדָם יְיָ--יָת תַּרְבָּא, דְּחָפֵי יָת גַּוָּא, וְיָת כָּל תַּרְבָּא, דְּעַל גַּוָּא. 
ג,טו וְאֵת֙ שְׁתֵּ֣י הַכְּלָיֹ֔ת וְאֶת־הַחֵ֨לֶב֙ אֲשֶׁ֣ר עֲלֵהֶ֔ן אֲשֶׁ֖ר עַל־הַכְּסָלִ֑ים וְאֶת־הַיֹּתֶ֨רֶת֙ עַל־הַכָּבֵ֔ד עַל־הַכְּלָיֹ֖ת יְסִירֶֽנָּה׃ וְיָת, תַּרְתֵּין כּוֹלְיָן, וְיָת תַּרְבָּא דַּעֲלֵיהוֹן, דְּעַל גִּסְסַיָּא; וְיָת חַצְרָא, דְּעַל כַּבְדָּא, עַל כּוֹלְיָתָא, יַעְדֵּינַהּ. 
ג,טז וְהִקְטִירָ֥ם הַכֹּהֵ֖ן הַמִּזְבֵּ֑חָה לֶ֤חֶם אִשֶּׁה֙ לְרֵ֣יחַ נִיחֹ֔חַ כָּל־חֵ֖לֶב לַֽיהוָֽה׃ וְיַסֵּיקִנּוּן כָּהֲנָא, לְמַדְבְּחָא--לְחֵים קֻרְבָּנָא לְאִתְקַבָּלָא בְּרַעֲוָא, כָּל תַּרְבָּא קֳדָם יְיָ. 
ג,יז חֻקַּ֤ת עוֹלָם֙ לְדֹרֹ֣תֵיכֶ֔ם בְּכֹ֖ל מוֹשְׁבֹֽתֵיכֶ֑ם כָּל־חֵ֥לֶב וְכָל־דָּ֖ם לֹ֥א תֹאכֵֽלוּ׃ {פ} קְיָם עָלַם לְדָרֵיכוֹן, בְּכֹל מוֹתְבָנֵיכוֹן--כָּל תַּרְבָּא וְכָל דְּמָא, לָא תֵּיכְלוּן. {פ}
ד,א וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ד,ב דַּבֵּ֞ר אֶל־בְּנֵ֣י יִשְׂרָאֵל֮ לֵאמֹר֒ נֶ֗פֶשׁ כִּֽי־תֶחֱטָ֤א בִשְׁגָגָה֙ מִכֹּל֙ מִצְו֣‍ֹת יְהוָ֔ה אֲשֶׁ֖ר לֹ֣א תֵֽעָשֶׂ֑ינָה וְעָשָׂ֕ה מֵֽאַחַ֖ת מֵהֵֽנָּה׃ מַלֵּיל עִם בְּנֵי יִשְׂרָאֵל, לְמֵימַר--אֲנָשׁ אֲרֵי יְחוּב בְּשָׁלוּ מִכֹּל פִּקּוֹדַיָּא דַּייָ, דְּלָא כָּשְׁרִין לְאִתְעֲבָדָא; וְיַעֲבֵיד, מִן חַד מִנְּהוֹן. 
ד,ג אִ֣ם הַכֹּהֵ֧ן הַמָּשִׁ֛יחַ יֶֽחֱטָ֖א לְאַשְׁמַ֣ת הָעָ֑ם וְהִקְרִ֡יב עַ֣ל חַטָּאתוֹ֩ אֲשֶׁ֨ר חָטָ֜א פַּ֣ר בֶּן־בָּקָ֥ר תָּמִ֛ים לַֽיהוָ֖ה לְחַטָּֽאת׃ אִם כָּהֲנָא רַבָּא יְחוּב, לְחוֹבַת עַמָּא:  וִיקָרֵיב עַל חוֹבְתֵיהּ דְּחָב תּוֹר בַּר תּוֹרֵי שְׁלִים, קֳדָם יְיָ--לְחַטָּתָא. 
ד,ד וְהֵבִ֣יא אֶת־הַפָּ֗ר אֶל־פֶּ֛תַח אֹ֥הֶל מוֹעֵ֖ד לִפְנֵ֣י יְהוָ֑ה וְסָמַ֤ךְ אֶת־יָדוֹ֙ עַל־רֹ֣אשׁ הַפָּ֔ר וְשָׁחַ֥ט אֶת־הַפָּ֖ר לִפְנֵ֥י יְהוָֽה׃ וְיַיְתֵי יָת תּוֹרָא, לִתְרַע מַשְׁכַּן זִמְנָא--לִקְדָם יְיָ; וְיִסְמוֹךְ יָת יְדֵיהּ עַל רֵישׁ תּוֹרָא, וְיִכּוֹס יָת תּוֹרָא קֳדָם יְיָ. 
ד,ה וְלָקַ֛ח הַכֹּהֵ֥ן הַמָּשִׁ֖יחַ מִדַּ֣ם הַפָּ֑ר וְהֵבִ֥יא אֹת֖וֹ אֶל־אֹ֥הֶל מוֹעֵֽד׃ וְיִסַּב כָּהֲנָא רַבָּא, מִדְּמָא דְּתוֹרָא; וְיַעֵיל יָתֵיהּ, לְמַשְׁכַּן זִמְנָא. 
ד,ו וְטָבַ֧ל הַכֹּהֵ֛ן אֶת־אֶצְבָּע֖וֹ בַּדָּ֑ם וְהִזָּ֨ה מִן־הַדָּ֜ם שֶׁ֤בַע פְּעָמִים֙ לִפְנֵ֣י יְהוָ֔ה אֶת־פְּנֵ֖י פָּרֹ֥כֶת הַקֹּֽדֶשׁ׃ וְיִטְבּוֹל כָּהֲנָא יָת אֶצְבְּעֵיהּ, בִּדְמָא; וְיַדֵּי מִן דְּמָא שְׁבַע זִמְנִין, קֳדָם יְיָ, קֳדָם, פָּרֻכְתָּא דְּקֻדְשָׁא. 
ד,ז וְנָתַן֩ הַכֹּהֵ֨ן מִן־הַדָּ֜ם עַל־קַ֠רְנוֹת מִזְבַּ֨ח קְטֹ֤רֶת הַסַּמִּים֙ לִפְנֵ֣י יְהוָ֔ה אֲשֶׁ֖ר בְּאֹ֣הֶל מוֹעֵ֑ד וְאֵ֣ת ׀ כָּל־דַּ֣ם הַפָּ֗ר יִשְׁפֹּךְ֙ אֶל־יְסוֹד֙ מִזְבַּ֣ח הָֽעֹלָ֔ה אֲשֶׁר־פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ וְיִתֵּין כָּהֲנָא מִן דְּמָא עַל קַרְנָת מַדְבְּחָא דִּקְטֹרֶת בֻּסְמַיָּא, דִּקְדָם יְיָ--דִּבְמַשְׁכַּן זִמְנָא; וְיָת כָּל דְּמָא דְּתוֹרָא, יִשְׁפּוֹךְ לִיסוֹדָא דְּמַדְבְּחָא דַּעֲלָתָא, דְּבִתְרַע, מַשְׁכַּן זִמְנָא. 
ד,ח וְאֶת־כָּל־חֵ֛לֶב פַּ֥ר הַֽחַטָּ֖את יָרִ֣ים מִמֶּ֑נּוּ אֶת־הַחֵ֨לֶב֙ הַֽמְכַסֶּ֣ה עַל־הַקֶּ֔רֶב וְאֵת֙ כָּל־הַחֵ֔לֶב אֲשֶׁ֖ר עַל־הַקֶּֽרֶב׃ וְיָת כָּל תְּרַב תּוֹרָא דְּחַטָּתָא, יַפְרֵישׁ מִנֵּיהּ--יָת תַּרְבָּא, דְּחָפֵי עַל גַּוָּא, וְיָת כָּל תַּרְבָּא, דְּעַל גַּוָּא. 
ד,ט וְאֵת֙ שְׁתֵּ֣י הַכְּלָיֹ֔ת וְאֶת־הַחֵ֨לֶב֙ אֲשֶׁ֣ר עֲלֵיהֶ֔ן אֲשֶׁ֖ר עַל־הַכְּסָלִ֑ים וְאֶת־הַיֹּתֶ֨רֶת֙ עַל־הַכָּבֵ֔ד עַל־הַכְּלָי֖וֹת יְסִירֶֽנָּה׃ וְיָת, תַּרְתֵּין כּוֹלְיָן, וְיָת תַּרְבָּא דַּעֲלֵיהוֹן, דְּעַל גִּסְסַיָּא; וְיָת חַצְרָא, דְּעַל כַּבְדָּא, עַל כּוֹלְיָתָא, יַעְדֵּינַהּ. 
ד,י כַּֽאֲשֶׁ֣ר יוּרַ֔ם מִשּׁ֖וֹר זֶ֣בַח הַשְּׁלָמִ֑ים וְהִקְטִירָם֙ הַכֹּהֵ֔ן עַ֖ל מִזְבַּ֥ח הָֽעֹלָֽה׃ כְּמָא דְּמִתַּפְרַשׁ, מִתּוֹר נִכְסַת קֻדְשַׁיָּא; וְיַסֵּיקִנּוּן, כָּהֲנָא, עַל, מַדְבְּחָא דַּעֲלָתָא. 
ד,יא וְאֶת־ע֤וֹר הַפָּר֙ וְאֶת־כָּל־בְּשָׂר֔וֹ עַל־רֹאשׁ֖וֹ וְעַל־כְּרָעָ֑יו וְקִרְבּ֖וֹ וּפִרְשֽׁוֹ׃ וְיָת מְשַׁךְ תּוֹרָא וְיָת כָּל בִּסְרֵיהּ, עַל רֵישֵׁיהּ וְעַל כְּרָעוֹהִי; וְגַוֵּיהּ, וְאֻכְלֵיהּ. 
ד,יב וְהוֹצִ֣יא אֶת־כָּל־הַ֠פָּר אֶל־מִח֨וּץ לַֽמַּחֲנֶ֜ה אֶל־מָק֤וֹם טָהוֹר֙ אֶל־שֶׁ֣פֶךְ הַדֶּ֔שֶׁן וְשָׂרַ֥ף אֹת֛וֹ עַל־עֵצִ֖ים בָּאֵ֑שׁ עַל־שֶׁ֥פֶךְ הַדֶּ֖שֶׁן יִשָּׂרֵֽף׃ {פ} וְיַפֵּיק יָת כָּל תּוֹרָא לְמִבַּרָא לְמַשְׁרִיתָא לַאֲתַר דְּכֵי, לַאֲתַר בֵּית מֵישַׁד קִטְמָא, וְיוֹקֵיד יָתֵיהּ עַל אָעַיָּא, בְּאִישָׁתָא; עַל אֲתַר בֵּית מֵישַׁד קִטְמָא, יִתּוֹקַד. {פ}
ד,יג וְאִ֨ם כָּל־עֲדַ֤ת יִשְׂרָאֵל֙ יִשְׁגּ֔וּ וְנֶעְלַ֣ם דָּבָ֔ר מֵֽעֵינֵ֖י הַקָּהָ֑ל וְ֠עָשׂוּ אַחַ֨ת מִכָּל־מִצְו֧‍ֹת יְהוָ֛ה אֲשֶׁ֥ר לֹֽא־תֵעָשֶׂ֖ינָה וְאָשֵֽׁמוּ׃ וְאִם כָּל כְּנִשְׁתָּא דְּיִשְׂרָאֵל, יִשְׁתְּלוֹן, וִיהֵי מְכֻסַּא פִּתְגָמָא, מֵעֵינֵי קְהָלָא; וְיַעְבְּדוּן חַד מִכָּל פִּקּוֹדַיָּא דַּייָ, דְּלָא כָּשְׁרִין לְאִתְעֲבָדָא--וִיחוּבוּן. 
ד,יד וְנֽוֹדְעָה֙ הַֽחַטָּ֔את אֲשֶׁ֥ר חָֽטְא֖וּ עָלֶ֑יהָ וְהִקְרִ֨יבוּ הַקָּהָ֜ל פַּ֤ר בֶּן־בָּקָר֙ לְחַטָּ֔את וְהֵבִ֣יאוּ אֹת֔וֹ לִפְנֵ֖י אֹ֥הֶל מוֹעֵֽד׃ וְתִתְיְדַע, חוֹבְתָא, דְּחָבוּ, עֲלַהּ--וִיקָרְבוּן קְהָלָא תּוֹר בַּר תּוֹרֵי, לְחַטָּתָא, וְיַיְתוֹן יָתֵיהּ, לִקְדָם מַשְׁכַּן זִמְנָא. 
ד,טו וְ֠סָֽמְכוּ זִקְנֵ֨י הָֽעֵדָ֧ה אֶת־יְדֵיהֶ֛ם עַל־רֹ֥אשׁ הַפָּ֖ר לִפְנֵ֣י יְהוָ֑ה וְשָׁחַ֥ט אֶת־הַפָּ֖ר לִפְנֵ֥י יְהוָֽה׃ וְיִסְמְכוּן סָבֵי כְּנִשְׁתָּא יָת יְדֵיהוֹן, עַל רֵישׁ תּוֹרָא--קֳדָם יְיָ; וְיִכּוֹס יָת תּוֹרָא, קֳדָם יְיָ. 
ד,טז וְהֵבִ֛יא הַכֹּהֵ֥ן הַמָּשִׁ֖יחַ מִדַּ֣ם הַפָּ֑ר אֶל־אֹ֖הֶל מוֹעֵֽד׃ וְיַעֵיל כָּהֲנָא רַבָּא, מִדְּמָא דְּתוֹרָא, לְמַשְׁכַּן, זִמְנָא. 
ד,יז וְטָבַ֧ל הַכֹּהֵ֛ן אֶצְבָּע֖וֹ מִן־הַדָּ֑ם וְהִזָּ֞ה שֶׁ֤בַע פְּעָמִים֙ לִפְנֵ֣י יְהוָ֔ה אֵ֖ת פְּנֵ֥י הַפָּרֹֽכֶת׃ וְיִטְבּוֹל כָּהֲנָא אֶצְבְּעֵיהּ, מִן דְּמָא; וְיַדֵּי שְׁבַע זִמְנִין, קֳדָם יְיָ--קֳדָם פָּרֻכְתָּא. 
ד,יח וּמִן־הַדָּ֞ם יִתֵּ֣ן ׀ עַל־קַרְנֹ֣ת הַמִּזְבֵּ֗חַ אֲשֶׁר֙ לִפְנֵ֣י יְהוָ֔ה אֲשֶׁ֖ר בְּאֹ֣הֶל מוֹעֵ֑ד וְאֵ֣ת כָּל־הַדָּ֗ם יִשְׁפֹּךְ֙ אֶל־יְסוֹד֙ מִזְבַּ֣ח הָֽעֹלָ֔ה אֲשֶׁר־פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃ וּמִן דְּמָא יִתֵּין עַל קַרְנָת מַדְבְּחָא, דִּקְדָם יְיָ--דִּבְמַשְׁכַּן זִמְנָא; וְיָת כָּל דְּמָא, יִשְׁפּוֹךְ לִיסוֹדָא דְּמַדְבְּחָא דַּעֲלָתָא, דְּבִתְרַע, מַשְׁכַּן זִמְנָא. 
ד,יט וְאֵ֥ת כָּל־חֶלְבּ֖וֹ יָרִ֣ים מִמֶּ֑נּוּ וְהִקְטִ֖יר הַמִּזְבֵּֽחָה׃ וְיָת כָּל תַּרְבֵּיהּ, יַפְרֵישׁ מִנֵּיהּ; וְיַסֵּיק, לְמַדְבְּחָא. 
ד,כ וְעָשָׂ֣ה לַפָּ֔ר כַּֽאֲשֶׁ֤ר עָשָׂה֙ לְפַ֣ר הַֽחַטָּ֔את כֵּ֖ן יַֽעֲשֶׂה־לּ֑וֹ וְכִפֶּ֧ר עֲלֵהֶ֛ם הַכֹּהֵ֖ן וְנִסְלַ֥ח לָהֶֽם׃ וְיַעֲבֵיד לְתוֹרָא--כְּמָא דַּעֲבַד לְתוֹרָא דְּחַטָּתָא, כֵּין יַעֲבֵיד לֵיהּ; וִיכַפַּר עֲלֵיהוֹן כָּהֲנָא, וְיִשְׁתְּבֵיק לְהוֹן. 
ד,כא וְהוֹצִ֣יא אֶת־הַפָּ֗ר אֶל־מִחוּץ֙ לַֽמַּחֲנֶ֔ה וְשָׂרַ֣ף אֹת֔וֹ כַּֽאֲשֶׁ֣ר שָׂרַ֔ף אֵ֖ת הַפָּ֣ר הָֽרִאשׁ֑וֹן חַטַּ֥את הַקָּהָ֖ל הֽוּא׃ {פ} וְיַפֵּיק יָת תּוֹרָא, לְמִבַּרָא לְמַשְׁרִיתָא, וְיוֹקֵיד יָתֵיהּ, כְּמָא דְּאוֹקֵיד יָת תּוֹרָא קַדְמָאָה:  חַטַּת קְהָלָא, הוּא. {פ}
ד,כב אֲשֶׁ֥ר נָשִׂ֖יא יֶֽחֱטָ֑א וְעָשָׂ֡ה אַחַ֣ת מִכָּל־מִצְו‍ֹת֩ יְהוָ֨ה אֱלֹהָ֜יו אֲשֶׁ֧ר לֹֽא־תֵעָשֶׂ֛ינָה בִּשְׁגָגָ֖ה וְאָשֵֽׁם׃ אִם רַבָּא, יְחוּב; וְיַעֲבֵיד חַד מִכָּל פִּקּוֹדַיָּא דַּייָ אֱלָהֵיהּ דְּלָא כָּשְׁרִין לְאִתְעֲבָדָא, בְּשָׁלוּ--וִיחוּב. 
ד,כג אֽוֹ־הוֹדַ֤ע אֵלָיו֙ חַטָּאת֔וֹ אֲשֶׁ֥ר חָטָ֖א בָּ֑הּ וְהֵבִ֧יא אֶת־קָרְבָּנ֛וֹ שְׂעִ֥יר עִזִּ֖ים זָכָ֥ר תָּמִֽים׃ אוֹ אִתְיְדַע לֵיהּ חוֹבְתֵיהּ, דְּחָב בַּהּ--וְיַיְתֵי יָת קֻרְבָּנֵיהּ צְפִיר בַּר עִזִּין, דְּכַר שְׁלִים. 
ד,כד וְסָמַ֤ךְ יָדוֹ֙ עַל־רֹ֣אשׁ הַשָּׂעִ֔יר וְשָׁחַ֣ט אֹת֗וֹ בִּמְק֛וֹם אֲשֶׁר־יִשְׁחַ֥ט אֶת־הָֽעֹלָ֖ה לִפְנֵ֣י יְהוָ֑ה חַטָּ֖את הֽוּא׃ וְיִסְמוֹךְ יְדֵיהּ, עַל רֵישׁ צְפִירָא, וְיִכּוֹס יָתֵיהּ בְּאַתְרָא דְּיִכּוֹס יָת עֲלָתָא, קֳדָם יְיָ; חַטָּתָא, הוּא. 
ד,כה וְלָקַ֨ח הַכֹּהֵ֜ן מִדַּ֤ם הַֽחַטָּאת֙ בְּאֶצְבָּע֔וֹ וְנָתַ֕ן עַל־קַרְנֹ֖ת מִזְבַּ֣ח הָֽעֹלָ֑ה וְאֶת־דָּמ֣וֹ יִשְׁפֹּ֔ךְ אֶל־יְס֖וֹד מִזְבַּ֥ח הָֽעֹלָֽה׃ וְיִסַּב כָּהֲנָא מִדְּמָא דְּחַטָּתָא, בְּאֶצְבְּעֵיהּ, וְיִתֵּין, עַל קַרְנָת מַדְבְּחָא דַּעֲלָתָא; וְיָת דְּמֵיהּ יִשְׁפּוֹךְ, לִיסוֹדָא דְּמַדְבְּחָא דַּעֲלָתָא. 
ד,כו וְאֶת־כָּל־חֶלְבּוֹ֙ יַקְטִ֣יר הַמִּזְבֵּ֔חָה כְּחֵ֖לֶב זֶ֣בַח הַשְּׁלָמִ֑ים וְכִפֶּ֨ר עָלָ֧יו הַכֹּהֵ֛ן מֵֽחַטָּאת֖וֹ וְנִסְלַ֥ח לֽוֹ׃ {פ} וְיָת כָּל תַּרְבֵּיהּ יַסֵּיק לְמַדְבְּחָא, כִּתְרַב נִכְסַת קֻדְשַׁיָּא; וִיכַפַּר עֲלוֹהִי כָּהֲנָא מֵחוֹבְתֵיהּ, וְיִשְׁתְּבֵיק לֵיהּ. {פ}
ד,כז וְאִם־נֶ֧פֶשׁ אַחַ֛ת תֶּֽחֱטָ֥א בִשְׁגָגָ֖ה מֵעַ֣ם הָאָ֑רֶץ בַּֽ֠עֲשֹׂתָהּ אַחַ֨ת מִמִּצְו֧‍ֹת יְהוָ֛ה אֲשֶׁ֥ר לֹֽא־תֵעָשֶׂ֖ינָה וְאָשֵֽׁם׃ וְאִם אֲנָשׁ חַד יְחוּב בְּשָׁלוּ, מֵעַמָּא דְּאַרְעָא:  בְּמִעְבְּדֵיהּ חַד מִפִּקּוֹדַיָּא דַּייָ, דְּלָא כָּשְׁרִין לְאִתְעֲבָדָא--וִיחוּב. 
ד,כח א֚וֹ הוֹדַ֣ע אֵלָ֔יו חַטָּאת֖וֹ אֲשֶׁ֣ר חָטָ֑א וְהֵבִ֨יא קָרְבָּנ֜וֹ שְׂעִירַ֤ת עִזִּים֙ תְּמִימָ֣ה נְקֵבָ֔ה עַל־חַטָּאת֖וֹ אֲשֶׁ֥ר חָטָֽא׃ אוֹ אִתְיְדַע לֵיהּ, חוֹבְתֵיהּ דְּחָב--וְיַיְתֵי קֻרְבָּנֵיהּ צְפִירַת עִזֵּי, שַׁלְמָא נֻקְבָּא, עַל חוֹבְתֵיהּ, דְּחָב. 
ד,כט וְסָמַךְ֙ אֶת־יָד֔וֹ עַ֖ל רֹ֣אשׁ הַֽחַטָּ֑את וְשָׁחַט֙ אֶת־הַ֣חַטָּ֔את בִּמְק֖וֹם הָֽעֹלָֽה׃ וְיִסְמוֹךְ, יָת יְדֵיהּ, עַל, רֵישׁ חַטָּתָא; וְיִכּוֹס, יָת חַטָּתָא, בְּאַתְרָא, דַּעֲלָתָא. 
ד,ל וְלָקַ֨ח הַכֹּהֵ֤ן מִדָּמָהּ֙ בְּאֶצְבָּע֔וֹ וְנָתַ֕ן עַל־קַרְנֹ֖ת מִזְבַּ֣ח הָֽעֹלָ֑ה וְאֶת־כָּל־דָּמָ֣הּ יִשְׁפֹּ֔ךְ אֶל־יְס֖וֹד הַמִּזְבֵּֽחַ׃ וְיִסַּב כָּהֲנָא מִדְּמַהּ, בְּאֶצְבְּעֵיהּ, וְיִתֵּין, עַל קַרְנָת מַדְבְּחָא דַּעֲלָתָא; וְיָת כָּל דְּמַהּ יִשְׁפּוֹךְ, לִיסוֹדָא דְּמַדְבְּחָא. 
ד,לא וְאֶת־כָּל־חֶלְבָּ֣הּ יָסִ֗יר כַּֽאֲשֶׁ֨ר הוּסַ֣ר חֵלֶב֮ מֵעַ֣ל זֶ֣בַח הַשְּׁלָמִים֒ וְהִקְטִ֤יר הַכֹּהֵן֙ הַמִּזְבֵּ֔חָה לְרֵ֥יחַ נִיחֹ֖חַ לַֽיהוָ֑ה וְכִפֶּ֥ר עָלָ֛יו הַכֹּהֵ֖ן וְנִסְלַ֥ח לֽוֹ׃ {פ} וְיָת כָּל תַּרְבַּהּ יַעְדֵּי, כְּמָא דְּאִתַּעְדַּא תְּרַב מֵעַל נִכְסַת קֻדְשַׁיָּא, וְיַסֵּיק כָּהֲנָא לְמַדְבְּחָא, לְאִתְקַבָּלָא בְּרַעֲוָא קֳדָם יְיָ; וִיכַפַּר עֲלוֹהִי כָּהֲנָא, וְיִשְׁתְּבֵיק לֵיהּ. {פ}
ד,לב וְאִם־כֶּ֛בֶשׂ יָבִ֥יא קָרְבָּנ֖וֹ לְחַטָּ֑את נְקֵבָ֥ה תְמִימָ֖ה יְבִיאֶֽנָּה׃ וְאִם אִמַּר יַיְתֵי קֻרְבָּנֵיהּ, לְחַטָּתָא--נֻקְבָּא שַׁלְמָא, יַיְתֵינַהּ. 
ד,לג וְסָמַךְ֙ אֶת־יָד֔וֹ עַ֖ל רֹ֣אשׁ הַֽחַטָּ֑את וְשָׁחַ֤ט אֹתָהּ֙ לְחַטָּ֔את בִּמְק֕וֹם אֲשֶׁ֥ר יִשְׁחַ֖ט אֶת־הָֽעֹלָֽה׃ וְיִסְמוֹךְ, יָת יְדֵיהּ, עַל, רֵישׁ חַטָּתָא; וְיִכּוֹס יָתַהּ, לְחַטָּתָא, בְּאַתְרָא, דְּיִכּוֹס יָת עֲלָתָא. 
ד,לד וְלָקַ֨ח הַכֹּהֵ֜ן מִדַּ֤ם הַֽחַטָּאת֙ בְּאֶצְבָּע֔וֹ וְנָתַ֕ן עַל־קַרְנֹ֖ת מִזְבַּ֣ח הָֽעֹלָ֑ה וְאֶת־כָּל־דָּמָ֣הּ יִשְׁפֹּ֔ךְ אֶל־יְס֖וֹד הַמִּזְבֵּֽחַ׃ וְיִסַּב כָּהֲנָא מִדְּמָא דְּחַטָּתָא, בְּאֶצְבְּעֵיהּ, וְיִתֵּין, עַל קַרְנָת מַדְבְּחָא דַּעֲלָתָא; וְיָת כָּל דְּמַהּ יִשְׁפּוֹךְ, לִיסוֹדָא דְּמַדְבְּחָא. 
ד,לה וְאֶת־כָּל־חֶלְבָּ֣הּ יָסִ֗יר כַּֽאֲשֶׁ֨ר יוּסַ֥ר חֵֽלֶב־הַכֶּשֶׂב֮ מִזֶּ֣בַח הַשְּׁלָמִים֒ וְהִקְטִ֨יר הַכֹּהֵ֤ן אֹתָם֙ הַמִּזְבֵּ֔חָה עַ֖ל אִשֵּׁ֣י יְהוָ֑ה וְכִפֶּ֨ר עָלָ֧יו הַכֹּהֵ֛ן עַל־חַטָּאת֥וֹ אֲשֶׁר־חָטָ֖א וְנִסְלַ֥ח לֽוֹ׃ {פ} וְיָת כָּל תַּרְבַּהּ יַעְדֵּי, כְּמָא דְּמִתַּעְדַּא תְּרַב אִמַּר מִנִּכְסַת קֻדְשַׁיָּא, וְיַסֵּיק כָּהֲנָא יָתְהוֹן לְמַדְבְּחָא, עַל קֻרְבָּנַיָּא דַּייָ; וִיכַפַּר עֲלוֹהִי כָּהֲנָא עַל חוֹבְתֵיהּ דְּחָב, וְיִשְׁתְּבֵיק לֵיהּ. {פ}
ה,א וְנֶ֣פֶשׁ כִּֽי־תֶחֱטָ֗א וְשָֽׁמְעָה֙ ק֣וֹל אָלָ֔ה וְה֣וּא עֵ֔ד א֥וֹ רָאָ֖ה א֣וֹ יָדָ֑ע אִם־ל֥וֹא יַגִּ֖יד וְנָשָׂ֥א עֲו‍ֹנֽוֹ׃ וַאֲנָשׁ אֲרֵי יְחוּב, וְיִשְׁמַע קָל מוֹמֵי, וְהוּא סָהִיד, אוֹ חֲזָא אוֹ יְדַע; אִם לָא יְחַוֵּי, וִיקַבֵּיל חוֹבֵיהּ. 
ה,ב א֣וֹ נֶ֗פֶשׁ אֲשֶׁ֣ר תִּגַּע֮ בְּכָל־דָּבָ֣ר טָמֵא֒ אוֹ֩ בְנִבְלַ֨ת חַיָּ֜ה טְמֵאָ֗ה א֚וֹ בְּנִבְלַת֙ בְּהֵמָ֣ה טְמֵאָ֔ה א֕וֹ בְּנִבְלַ֖ת שֶׁ֣רֶץ טָמֵ֑א וְנֶעְלַ֣ם מִמֶּ֔נּוּ וְה֥וּא טָמֵ֖א וְאָשֵֽׁם׃ אוֹ אֲנָשׁ, דְּיִקְרַב בְּכָל מִדָּעַם מְסָאַב, אוֹ בְּנִבְלַת חַיְתָא מְסָאַבְתָּא אוֹ בְּנִבְלַת בְּעִירָא מְסָאֲבָא, אוֹ בְּנִבְלַת רְחֵישׁ מְסָאַב; וִיהֵי מְכֻסַּא מִנֵּיהּ, וְהוּא מְסָאַב וְחָב. 
ה,ג א֣וֹ כִ֤י יִגַּע֙ בְּטֻמְאַ֣ת אָדָ֔ם לְכֹל֙ טֻמְאָת֔וֹ אֲשֶׁ֥ר יִטְמָ֖א בָּ֑הּ וְנֶעְלַ֣ם מִמֶּ֔נּוּ וְה֥וּא יָדַ֖ע וְאָשֵֽׁם׃ אוֹ אֲרֵי יִקְרַב, בְּסוֹאֲבָת אֲנָשָׁא, לְכֹל סְאוֹבְתֵיהּ, דְּיִסְתָּאַב בַּהּ; וִיהֵי מְכֻסַּא מִנֵּיהּ, וְהוּא יְדַע וְחָב. 
ה,ד א֣וֹ נֶ֡פֶשׁ כִּ֣י תִשָּׁבַע֩ לְבַטֵּ֨א בִשְׂפָתַ֜יִם לְהָרַ֣ע ׀ א֣וֹ לְהֵיטִ֗יב לְ֠כֹל אֲשֶׁ֨ר יְבַטֵּ֧א הָֽאָדָ֛ם בִּשְׁבֻעָ֖ה וְנֶעְלַ֣ם מִמֶּ֑נּוּ וְהֽוּא־יָדַ֥ע וְאָשֵׁ֖ם לְאַחַ֥ת מֵאֵֽלֶּה׃ אוֹ אֲנָשׁ אֲרֵי יְקַיֵּים לְפָרָשָׁא בְּסִפְוָן לְאַבְאָשָׁא אוֹ לְאֵיטָבָא, לְכֹל דִּיפָרֵישׁ אֲנָשָׁא בְּקִיּוּם--וִיהֵי מְכֻסַּא מִנֵּיהּ; וְהוּא יְדַע וְחָב, לַחֲדָא מֵאִלֵּין. 
ה,ה וְהָיָ֥ה כִֽי־יֶאְשַׁ֖ם לְאַחַ֣ת מֵאֵ֑לֶּה וְהִ֨תְוַדָּ֔ה אֲשֶׁ֥ר חָטָ֖א עָלֶֽיהָ׃ וִיהֵי אֲרֵי יְחוּב, לַחֲדָא מֵאִלֵּין:  וִיוַדֵּי--דְּחָב, עֲלַהּ. 
ה,ו וְהֵבִ֣יא אֶת־אֲשָׁמ֣וֹ לַֽיהוָ֡ה עַ֣ל חַטָּאתוֹ֩ אֲשֶׁ֨ר חָטָ֜א נְקֵבָ֨ה מִן־הַצֹּ֥אן כִּשְׂבָּ֛ה אֽוֹ־שְׂעִירַ֥ת עִזִּ֖ים לְחַטָּ֑את וְכִפֶּ֥ר עָלָ֛יו הַכֹּהֵ֖ן מֵֽחַטָּאתֽוֹ׃ וְיַיְתֵי יָת אֲשָׁמֵיהּ לִקְדָם יְיָ עַל חוֹבְתֵיהּ דְּחָב נֻקְבָּא מִן עָנָא אִמַּרְתָּא, אוֹ צְפִירַת עִזֵּי--לְחַטָּתָא; וִיכַפַּר עֲלוֹהִי כָּהֲנָא, מֵחוֹבְתֵיהּ. 
ה,ז וְאִם־לֹ֨א תַגִּ֣יעַ יָדוֹ֮ דֵּ֣י שֶׂה֒ וְהֵבִ֨יא אֶת־אֲשָׁמ֜וֹ אֲשֶׁ֣ר חָטָ֗א שְׁתֵּ֥י תֹרִ֛ים אֽוֹ־שְׁנֵ֥י בְנֵֽי־יוֹנָ֖ה לַֽיהוָ֑ה אֶחָ֥ד לְחַטָּ֖את וְאֶחָ֥ד לְעֹלָֽה׃ וְאִם לָא תִּמְטֵי יְדֵיהּ, כְּמִסַּת אִמְּרָא--וְיַיְתֵי יָת חוֹבְתֵיהּ דְּחָב תְּרֵין שַׁפְנִינִין אוֹ תְּרֵין בְּנֵי יוֹנָה, לִקְדָם יְיָ:  חַד לְחַטָּתָא, וְחַד לַעֲלָתָא. 
ה,ח וְהֵבִ֤יא אֹתָם֙ אֶל־הַכֹּהֵ֔ן וְהִקְרִ֛יב אֶת־אֲשֶׁ֥ר לַֽחַטָּ֖את רִֽאשׁוֹנָ֑ה וּמָלַ֧ק אֶת־רֹאשׁ֛וֹ מִמּ֥וּל עָרְפּ֖וֹ וְלֹ֥א יַבְדִּֽיל׃ וְיַיְתֵי יָתְהוֹן לְוָת כָּהֲנָא, וִיקָרֵיב יָת דִּלְחַטָּתָא קַדְמוּתָא; וְיִמְלוֹק יָת רֵישֵׁיהּ מִקֳּבֵיל קְדָלֵיהּ, וְלָא יַפְרֵישׁ. 
ה,ט וְהִזָּ֞ה מִדַּ֤ם הַֽחַטָּאת֙ עַל־קִ֣יר הַמִּזְבֵּ֔חַ וְהַנִּשְׁאָ֣ר בַּדָּ֔ם יִמָּצֵ֖ה אֶל־יְס֣וֹד הַמִּזְבֵּ֑חַ חַטָּ֖את הֽוּא׃ וְיַדֵּי מִדְּמָא דְּחַטָּתָא, עַל כֹּתֶל מַדְבְּחָא, וּדְיִשְׁתְּאַר בִּדְמָא, יִתְמְצֵי לִיסוֹדָא דְּמַדְבְּחָא; חַטָּתָא, הוּא. 
ה,י וְאֶת־הַשֵּׁנִ֛י יַֽעֲשֶׂ֥ה עֹלָ֖ה כַּמִּשְׁפָּ֑ט וְכִפֶּ֨ר עָלָ֧יו הַכֹּהֵ֛ן מֵֽחַטָּאת֥וֹ אֲשֶׁר־חָטָ֖א וְנִסְלַ֥ח לֽוֹ׃ {ס} וְיָת תִּנְיָנָא יַעֲבֵיד עֲלָתָא, כְּדַחְזֵי; וִיכַפַּר עֲלוֹהִי כָּהֲנָא מֵחוֹבְתֵיהּ דְּחָב, וְיִשְׁתְּבֵיק לֵיהּ. {ס}
ה,יא וְאִם־לֹא֩ תַשִּׂ֨יג יָד֜וֹ לִשְׁתֵּ֣י תֹרִ֗ים אוֹ֮ לִשְׁנֵ֣י בְנֵֽי־יוֹנָה֒ וְהֵבִ֨יא אֶת־קָרְבָּנ֜וֹ אֲשֶׁ֣ר חָטָ֗א עֲשִׂירִ֧ת הָֽאֵפָ֛ה סֹ֖לֶת לְחַטָּ֑את לֹֽא־יָשִׂ֨ים עָלֶ֜יהָ שֶׁ֗מֶן וְלֹֽא־יִתֵּ֤ן עָלֶ֨יהָ֙ לְבֹנָ֔ה כִּ֥י חַטָּ֖את הִֽוא׃ וְאִם לָא תַּדְבֵּיק יְדֵיהּ לִתְרֵין שַׁפְנִינִין, אוֹ לִתְרֵין בְּנֵי יוֹנָה--וְיַיְתֵי יָת קֻרְבָּנֵיהּ דְּחָב חַד מִן עַסְרָא בִּתְלָת סְאִין סֻלְתָּא, לְחַטָּתָא; לָא יְשַׁוֵּי עֲלַהּ מִשְׁחָא, וְלָא יִתֵּין עֲלַהּ לְבוֹנְתָא--אֲרֵי חַטָּתָא, הִיא. 
ה,יב וֶֽהֱבִיאָהּ֮ אֶל־הַכֹּהֵן֒ וְקָמַ֣ץ הַכֹּהֵ֣ן ׀ מִ֠מֶּנָּה מְל֨וֹא קֻמְצ֜וֹ אֶת־אַזְכָּֽרָתָהּ֙ וְהִקְטִ֣יר הַמִּזְבֵּ֔חָה עַ֖ל אִשֵּׁ֣י יְהוָ֑ה חַטָּ֖את הִֽוא׃ וְיַיְתֵינַהּ, לְוָת כָּהֲנָא, וְיִקְמוֹץ כָּהֲנָא מִנַּהּ מְלֵי קֻמְצֵיהּ יָת אַדְכָרְתַהּ וְיַסֵּיק לְמַדְבְּחָא, עַל קֻרְבָּנַיָּא דַּייָ; חַטָּתָא, הִיא. 
ה,יג וְכִפֶּר֩ עָלָ֨יו הַכֹּהֵ֜ן עַל־חַטָּאת֧וֹ אֲשֶׁר־חָטָ֛א מֵֽאַחַ֥ת מֵאֵ֖לֶּה וְנִסְלַ֣ח ל֑וֹ וְהָֽיְתָ֥ה לַכֹּהֵ֖ן כַּמִּנְחָֽה׃ {ס} וִיכַפַּר עֲלוֹהִי כָּהֲנָא עַל חוֹבְתֵיהּ דְּחָב, מֵחֲדָא מֵאִלֵּין--וְיִשְׁתְּבֵיק לֵיהּ; וּתְהֵי לְכָהֲנָא, כְּמִנְחָתָא. {ס}
ה,יד וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ה,טו נֶ֚פֶשׁ כִּֽי־תִמְעֹ֣ל מַ֔עַל וְחָֽטְאָה֙ בִּשְׁגָגָ֔ה מִקָּדְשֵׁ֖י יְהוָ֑ה וְהֵבִיא֩ אֶת־אֲשָׁמ֨וֹ לַֽיהוָ֜ה אַ֧יִל תָּמִ֣ים מִן־הַצֹּ֗אן בְּעֶרְכְּךָ֛ כֶּֽסֶף־שְׁקָלִ֥ים בְּשֶֽׁקֶל־הַקֹּ֖דֶשׁ לְאָשָֽׁם׃ אֲנָשׁ, אֲרֵי יְשַׁקַּר שְׁקַר, וִיחוּב בְּשָׁלוּ, מִקֻּדְשַׁיָּא דַּייָ:  וְיַיְתֵי יָת אֲשָׁמֵיהּ לִקְדָם יְיָ דְּכַר שְׁלִים מִן עָנָא, בְּפֻרְסָנֵיהּ כְּסַף סִלְעִין בְּסִלְעֵי קֻדְשָׁא--לַאֲשָׁמָא. 
ה,טז וְאֵ֣ת אֲשֶׁר֩ חָטָ֨א מִן־הַקֹּ֜דֶשׁ יְשַׁלֵּ֗ם וְאֶת־חֲמִֽישִׁתוֹ֙ יוֹסֵ֣ף עָלָ֔יו וְנָתַ֥ן אֹת֖וֹ לַכֹּהֵ֑ן וְהַכֹּהֵ֗ן יְכַפֵּ֥ר עָלָ֛יו בְּאֵ֥יל הָֽאָשָׁ֖ם וְנִסְלַ֥ח לֽוֹ׃ {פ} וְיָת דְּחָב מִן קֻדְשָׁא יְשַׁלֵּים, וְיָת חֻמְשֵׁיהּ יוֹסֵיף עֲלוֹהִי, וְיִתֵּין יָתֵיהּ, לְכָהֲנָא; וְכָהֲנָא, יְכַפַּר עֲלוֹהִי בְּדִכְרָא דַּאֲשָׁמָא--וְיִשְׁתְּבֵיק לֵיהּ. {פ}
ה,יז וְאִם־נֶ֨פֶשׁ֙ כִּ֣י תֶֽחֱטָ֔א וְעָֽשְׂתָ֗ה אַחַת֙ מִכָּל־מִצְו֣‍ֹת יְהוָ֔ה אֲשֶׁ֖ר לֹ֣א תֵֽעָשֶׂ֑ינָה וְלֹֽא־יָדַ֥ע וְאָשֵׁ֖ם וְנָשָׂ֥א עֲו‍ֹנֽוֹ׃ וְאִם אֲנָשׁ, אֲרֵי יְחוּב, וְיַעֲבֵיד חַד מִכָּל פִּקּוֹדַיָּא דַּייָ, דְּלָא כָּשְׁרִין לְאִתְעֲבָדָא; וְלָא יְדַע וְחָב, וִיקַבֵּיל חוֹבֵיהּ. 
ה,יח וְ֠הֵבִיא אַ֣יִל תָּמִ֧ים מִן־הַצֹּ֛אן בְּעֶרְכְּךָ֥ לְאָשָׁ֖ם אֶל־הַכֹּהֵ֑ן וְכִפֶּר֩ עָלָ֨יו הַכֹּהֵ֜ן עַ֣ל שִׁגְגָת֧וֹ אֲשֶׁר־שָׁגָ֛ג וְה֥וּא לֹֽא־יָדַ֖ע וְנִסְלַ֥ח לֽוֹ׃ וְיַיְתֵי דְּכַר שְׁלִים מִן עָנָא, בְּפֻרְסָנֵיהּ לַאֲשָׁמָא--לְוָת כָּהֲנָא; וִיכַפַּר עֲלוֹהִי כָּהֲנָא עַל שָׁלוּתֵיהּ דְּאִשְׁתְּלִי, וְהוּא לָא יְדַע--וְיִשְׁתְּבֵיק לֵיהּ. 
ה,יט אָשָׁ֖ם ה֑וּא אָשֹׁ֥ם אָשַׁ֖ם לַֽיהוָֽה׃ {פ} אֲשָׁמָא, הוּא:  עַל חוֹבְתֵיהּ דְּחָב אֲשָׁמָא יְקָרֵיב, קֳדָם יְיָ. {פ}
ה,כ וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
ה,כא נֶ֚פֶשׁ כִּ֣י תֶֽחֱטָ֔א וּמָֽעֲלָ֥ה מַ֖עַל בַּֽיהוָ֑ה וְכִחֵ֨שׁ בַּֽעֲמִית֜וֹ בְּפִקָּד֗וֹן אֽוֹ־בִתְשׂ֤וּמֶת יָד֙ א֣וֹ בְגָזֵ֔ל א֖וֹ עָשַׁ֥ק אֶת־עֲמִיתֽוֹ׃ אֲנָשׁ אֲרֵי יְחוּב, וִישַׁקַּר שְׁקַר קֳדָם יְיָ; וִיכַדֵּיב בְּחַבְרֵיהּ בְּפִקְדוֹנָא, אוֹ בְּשֻׁתָּפוּת יְדָא אוֹ בְּגָזֵילָא, אוֹ, עֲשַׁק יָת חַבְרֵיהּ. 
ה,כב אֽוֹ־מָצָ֧א אֲבֵדָ֛ה וְכִ֥חֶשׁ בָּ֖הּ וְנִשְׁבַּ֣ע עַל־שָׁ֑קֶר עַל־אַחַ֗ת מִכֹּ֛ל אֲשֶׁר־יַֽעֲשֶׂ֥ה הָֽאָדָ֖ם לַֽחֲטֹ֥א בָהֵֽנָּה׃ אוֹ אַשְׁכַּח אֲבֵידְתָא וְכַדֵּיב בַּהּ, וְאִשְׁתְּבַע עַל שִׁקְרָא; עַל חֲדָא, מִכֹּל דְּיַעֲבֵיד אֲנָשָׁא--לִמְחָב בְּהוֹן. 
ה,כג וְהָיָה֮ כִּֽי־יֶחֱטָ֣א וְאָשֵׁם֒ וְהֵשִׁ֨יב אֶת־הַגְּזֵלָ֜ה אֲשֶׁ֣ר גָּזָ֗ל א֤וֹ אֶת־הָעֹ֨שֶׁק֙ אֲשֶׁ֣ר עָשָׁ֔ק א֚וֹ אֶת־הַפִּקָּד֔וֹן אֲשֶׁ֥ר הָפְקַ֖ד אִתּ֑וֹ א֥וֹ אֶת־הָֽאֲבֵדָ֖ה אֲשֶׁ֥ר מָצָֽא׃ וִיהֵי, אֲרֵי יִחְטֵי וִיחוּב--וְיָתִיב יָת גָּזֵילָא דִּגְזַל אוֹ יָת עֻשְׁקָא דַּעֲשַׁק, אוֹ יָת פִּקְדוֹנָא דְּאִתַּפְקַד לְוָתֵיהּ; אוֹ יָת אֲבֵידְתָא, דְּאַשְׁכַּח. 
ה,כד א֠וֹ מִכֹּ֞ל אֲשֶׁר־יִשָּׁבַ֣ע עָלָיו֮ לַשֶּׁקֶר֒ וְשִׁלַּ֤ם אֹתוֹ֙ בְּרֹאשׁ֔וֹ וַחֲמִֽשִׁתָ֖יו יֹסֵ֣ף עָלָ֑יו לַֽאֲשֶׁ֨ר ה֥וּא ל֛וֹ יִתְּנֶ֖נּוּ בְּי֥וֹם אַשְׁמָתֽוֹ׃ אוֹ מִכּוֹלָא דְּיִשְׁתְּבַע עֲלוֹהִי, לְשִׁקְרָא--וִישַׁלֵּים יָתֵיהּ בְּרֵישֵׁיהּ, וְחֻמְשׁוֹהִי יוֹסֵיף עֲלוֹהִי:  לִדְהוּא דִּילֵיהּ יִתְּנִנֵּיהּ, בְּיוֹמָא דְּחוֹבְתֵיהּ. 
ה,כה וְאֶת־אֲשָׁמ֥וֹ יָבִ֖יא לַֽיהוָ֑ה אַ֣יִל תָּמִ֧ים מִן־הַצֹּ֛אן בְּעֶרְכְּךָ֥ לְאָשָׁ֖ם אֶל־הַכֹּהֵֽן׃ וְיָת אֲשָׁמֵיהּ יַיְתֵי, לִקְדָם יְיָ, דְּכַר שְׁלִים מִן עָנָא בְּפֻרְסָנֵיהּ לַאֲשָׁמָא, לְוָת כָּהֲנָא. 
ה,כו וְכִפֶּ֨ר עָלָ֧יו הַכֹּהֵ֛ן לִפְנֵ֥י יְהוָ֖ה וְנִסְלַ֣ח ל֑וֹ עַל־אַחַ֛ת מִכֹּ֥ל אֲשֶֽׁר־יַעֲשֶׂ֖ה לְאַשְׁמָ֥ה בָֽהּ׃ {פ} וִיכַפַּר עֲלוֹהִי כָּהֲנָא קֳדָם יְיָ, וְיִשְׁתְּבֵיק לֵיהּ, עַל חֲדָא מִכֹּל דְּיַעֲבֵיד, לִמְחָב בַּהּ. {פ}
