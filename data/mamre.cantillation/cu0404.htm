שלח-לך

יג,א וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
יג,ב שְׁלַח־לְךָ֣ אֲנָשִׁ֗ים וְיָתֻ֨רוּ֙ אֶת־אֶ֣רֶץ כְּנַ֔עַן אֲשֶׁר־אֲנִ֥י נֹתֵ֖ן לִבְנֵ֣י יִשְׂרָאֵ֑ל אִ֣ישׁ אֶחָד֩ אִ֨ישׁ אֶחָ֜ד לְמַטֵּ֤ה אֲבֹתָיו֙ תִּשְׁלָ֔חוּ כֹּ֖ל נָשִׂ֥יא בָהֶֽם׃ שְׁלַח לָךְ גֻּבְרִין, וִיאַלְּלוּן יָת אַרְעָא דִּכְנַעַן, דַּאֲנָא יָהֵיב, לִבְנֵי יִשְׂרָאֵל:  גֻּבְרָא חַד גֻּבְרָא חַד לְשִׁבְטָא דַּאֲבָהָתוֹהִי, תִּשְׁלְחוּן--כֹּל, רַבָּא דִּבְהוֹן. 
יג,ג וַיִּשְׁלַ֨ח אֹתָ֥ם מֹשֶׁ֛ה מִמִּדְבַּ֥ר פָּארָ֖ן עַל־פִּ֣י יְהוָ֑ה כֻּלָּ֣ם אֲנָשִׁ֔ים רָאשֵׁ֥י בְנֵֽי־יִשְׂרָאֵ֖ל הֵֽמָּה׃ וּשְׁלַח יָתְהוֹן מֹשֶׁה מִמַּדְבְּרָא דְּפָארָן, עַל מֵימְרָא דַּייָ:  כֻּלְּהוֹן גֻּבְרִין, רֵישֵׁי בְּנֵי יִשְׂרָאֵל אִנּוּן. 
יג,ד וְאֵ֖לֶּה שְׁמוֹתָ֑ם לְמַטֵּ֣ה רְאוּבֵ֔ן שַׁמּ֖וּעַ בֶּן־זַכּֽוּר׃ וְאִלֵּין, שְׁמָהָתְהוֹן:  לְשִׁבְטָא דִּרְאוּבֵן, שַׁמּוּעַ בַּר זַכּוּר. 
יג,ה לְמַטֵּ֣ה שִׁמְע֔וֹן שָׁפָ֖ט בֶּן־חוֹרִֽי׃ לְשִׁבְטָא דְּשִׁמְעוֹן, שָׁפָט בַּר חוֹרִי. 
יג,ו לְמַטֵּ֣ה יְהוּדָ֔ה כָּלֵ֖ב בֶּן־יְפֻנֶּֽה׃ לְשִׁבְטָא דִּיהוּדָה, כָּלֵב בַּר יְפֻנֶּה. 
יג,ז לְמַטֵּ֣ה יִשָּׂשכָ֔ר יִגְאָ֖ל בֶּן־יוֹסֵֽף׃ לְשִׁבְטָא דְּיִשָּׂשכָר, יִגְאָל בַּר יוֹסֵף. 
יג,ח לְמַטֵּ֥ה אֶפְרָ֖יִם הוֹשֵׁ֥עַ בִּן־נֽוּן׃ לְשִׁבְטָא דְּאֶפְרַיִם, הוֹשֵׁעַ בַּר נוּן. 
יג,ט לְמַטֵּ֣ה בִנְיָמִ֔ן פַּלְטִ֖י בֶּן־רָפֽוּא׃ לְשִׁבְטָא דְּבִנְיָמִין, פַּלְטִי בַּר רָפוּא. 
יג,י לְמַטֵּ֣ה זְבוּלֻ֔ן גַּדִּיאֵ֖ל בֶּן־סוֹדִֽי׃ לְשִׁבְטָא דִּזְבוּלוּן, גַּדִּיאֵל בַּר סוֹדִי. 
יג,יא לְמַטֵּ֥ה יוֹסֵ֖ף לְמַטֵּ֣ה מְנַשֶּׁ֑ה גַּדִּ֖י בֶּן־סוּסִֽי׃ לְשִׁבְטָא דְּיוֹסֵף, לְשִׁבְטָא דִּמְנַשֶּׁה--גַּדִּי, בַּר סוּסִי. 
יג,יב לְמַטֵּ֣ה דָ֔ן עַמִּיאֵ֖ל בֶּן־גְּמַלִּֽי׃ לְשִׁבְטָא דְּדָן, עַמִּיאֵל בַּר גְּמַלִּי. 
יג,יג לְמַטֵּ֣ה אָשֵׁ֔ר סְת֖וּר בֶּן־מִֽיכָאֵֽל׃ לְשִׁבְטָא דְּאָשֵׁר, סְתוּר בַּר מִיכָאֵל. 
יג,יד לְמַטֵּ֣ה נַפְתָּלִ֔י נַחְבִּ֖י בֶּן־וָפְסִֽי׃ לְשִׁבְטָא דְּנַפְתָּלִי, נַחְבִּי בַּר וָפְסִי. 
יג,טו לְמַטֵּ֣ה גָ֔ד גְּאוּאֵ֖ל בֶּן־מָכִֽי׃ לְשִׁבְטָא דְּגָד, גְּאוּאֵל בַּר מָכִי. 
יג,טז אֵ֚לֶּה שְׁמ֣וֹת הָֽאֲנָשִׁ֔ים אֲשֶׁר־שָׁלַ֥ח מֹשֶׁ֖ה לָת֣וּר אֶת־הָאָ֑רֶץ וַיִּקְרָ֥א מֹשֶׁ֛ה לְהוֹשֵׁ֥עַ בִּן־נ֖וּן יְהוֹשֻֽׁעַ׃ אִלֵּין שְׁמָהָת גֻּבְרַיָּא, דִּשְׁלַח מֹשֶׁה לְאַלָּלָא יָת אַרְעָא; וּקְרָא מֹשֶׁה לְהוֹשֵׁעַ בַּר נוּן, יְהוֹשׁוּעַ. 
יג,יז וַיִּשְׁלַ֤ח אֹתָם֙ מֹשֶׁ֔ה לָת֖וּר אֶת־אֶ֣רֶץ כְּנָ֑עַן וַיֹּ֣אמֶר אֲלֵהֶ֗ם עֲל֥וּ זֶה֙ בַּנֶּ֔גֶב וַֽעֲלִיתֶ֖ם אֶת־הָהָֽר׃ וּשְׁלַח יָתְהוֹן מֹשֶׁה, לְאַלָּלָא יָת אַרְעָא דִּכְנָעַן; וַאֲמַר לְהוֹן, סַקוּ דָּא בְּדָרוֹמָא, וְתִסְּקוּן, לְטוּרָא. 
יג,יח וּרְאִיתֶ֥ם אֶת־הָאָ֖רֶץ מַה־הִ֑וא וְאֶת־הָעָם֙ הַיֹּשֵׁ֣ב עָלֶ֔יהָ הֶֽחָזָ֥ק הוּא֙ הֲרָפֶ֔ה הַמְעַ֥ט ה֖וּא אִם־רָֽב׃ וְתִחְזוֹן יָת אַרְעָא, מָא הִיא; וְיָת עַמָּא, דְּיָתֵיב עֲלַהּ--הֲתַקִּיף הוּא אִם חַלָּשׁ, הַזְעֵיר הוּא אִם סַגִּי. 
יג,יט וּמָ֣ה הָאָ֗רֶץ אֲשֶׁר־הוּא֙ יֹשֵׁ֣ב בָּ֔הּ הֲטוֹבָ֥ה הִ֖וא אִם־רָעָ֑ה וּמָ֣ה הֶֽעָרִ֗ים אֲשֶׁר־הוּא֙ יוֹשֵׁ֣ב בָּהֵ֔נָּה הַבְּמַֽחֲנִ֖ים אִ֥ם בְּמִבְצָרִֽים׃ וּמָא אַרְעָא, דְּהוּא יָתֵיב בַּהּ--הֲטָבָא הִיא, אִם בִּישָׁא; וּמָא קִרְוַיָּא, דְּהוּא יָתֵיב בְּהוֹן--הַבְּפַצְחִין, אִם בְּכַּרְכִין. 
יג,כ וּמָ֣ה הָ֠אָרֶץ הַשְּׁמֵנָ֨ה הִ֜וא אִם־רָזָ֗ה הֲיֵֽשׁ־בָּ֥הּ עֵץ֙ אִם־אַ֔יִן וְהִ֨תְחַזַּקְתֶּ֔ם וּלְקַחְתֶּ֖ם מִפְּרִ֣י הָאָ֑רֶץ וְהַ֨יָּמִ֔ים יְמֵ֖י בִּכּוּרֵ֥י עֲנָבִֽים׃ וּמָא אַרְעָא הֲעַתִּירָא הִיא אִם מִסְכֵּינָא, הֲאִית בַּהּ אִילָנִין אִם לָא, וְתִתַּקְּפוּן, וְתִסְּבוּן מֵאִבָּא דְּאַרְעָא; וְיוֹמַיָּא--יוֹמֵי, בִּכּוּרֵי עִנְבִּין. 
יג,כא וַֽיַּעֲל֖וּ וַיָּתֻ֣רוּ אֶת־הָאָ֑רֶץ מִמִּדְבַּר־צִ֥ן עַד־רְחֹ֖ב לְבֹ֥א חֲמָֽת׃ וּסְלִיקוּ, וְאַלִּילוּ יָת אַרְעָא, מִמַּדְבְּרָא דְּצִין עַד רְחוֹב, לְמָטֵי חֲמָת. 
יג,כב וַיַּֽעֲל֣וּ בַנֶּגֶב֮ וַיָּבֹ֣א עַד־חֶבְרוֹן֒ וְשָׁ֤ם אֲחִימַן֙ שֵׁשַׁ֣י וְתַלְמַ֔י יְלִידֵ֖י הָֽעֲנָ֑ק וְחֶבְר֗וֹן שֶׁ֤בַע שָׁנִים֙ נִבְנְתָ֔ה לִפְנֵ֖י צֹ֥עַן מִצְרָֽיִם׃ וּסְלִיקוּ בְּדָרוֹמָא, וַאֲתוֹ עַד חֶבְרוֹן, וְתַמָּן אֲחִימַן שֵׁשַׁי וְתַלְמַי, בְּנֵי גִּבָּרָא; וְחֶבְרוֹן, שְׁבַע שְׁנִין אִתְבְּנִיאַת, קֳדָם, טַאנֵיס דְּמִצְרָיִם. 
יג,כג וַיָּבֹ֜אוּ עַד־נַ֣חַל אֶשְׁכֹּ֗ל וַיִּכְרְת֨וּ מִשָּׁ֤ם זְמוֹרָה֙ וְאֶשְׁכּ֤וֹל עֲנָבִים֙ אֶחָ֔ד וַיִּשָּׂאֻ֥הוּ בַמּ֖וֹט בִּשְׁנָ֑יִם וּמִן־הָֽרִמֹּנִ֖ים וּמִן־הַתְּאֵנִֽים׃ וַאֲתוֹ עַד נַחְלָא דְּאֶתְכָּלָא, וְקָצוּ מִתַּמָּן עוֹבַרְתָּא וְאֶתְכָּל דְּעִנְבִּין חַד, וְנַטְלוּהִי בַּאֲרִיחָא, בִּתְרֵין; וּמִן רִמּוֹנַיָּא, וּמִן תֵּינַיָּא. 
יג,כד לַמָּק֣וֹם הַה֔וּא קָרָ֖א נַ֣חַל אֶשְׁכּ֑וֹל עַ֚ל אֹד֣וֹת הָֽאֶשְׁכּ֔וֹל אֲשֶׁר־כָּֽרְת֥וּ מִשָּׁ֖ם בְּנֵ֥י יִשְׂרָאֵֽל׃ לְאַתְרָא הַהוּא, קְרָא נַחְלָא דְּאֶתְכָּלָא, עַל עֵיסַק אֶתְכָּלָא, דְּקָצוּ מִתַּמָּן בְּנֵי יִשְׂרָאֵל. 
יג,כה וַיָּשֻׁ֖בוּ מִתּ֣וּר הָאָ֑רֶץ מִקֵּ֖ץ אַרְבָּעִ֥ים יֽוֹם׃ וְתָבוּ, מִלְּאַלָּלָא אַרְעָא, מִסּוֹף, אַרְבְּעִין יוֹמִין. 
יג,כו וַיֵּֽלְכ֡וּ וַיָּבֹאוּ֩ אֶל־מֹשֶׁ֨ה וְאֶֽל־אַהֲרֹ֜ן וְאֶל־כָּל־עֲדַ֧ת בְּנֵֽי־יִשְׂרָאֵ֛ל אֶל־מִדְבַּ֥ר פָּארָ֖ן קָדֵ֑שָׁה וַיָּשִׁ֨יבוּ אֹתָ֤ם דָּבָר֙ וְאֶת־כָּל־הָ֣עֵדָ֔ה וַיַּרְא֖וּם אֶת־פְּרִ֥י הָאָֽרֶץ׃ וַאֲזַלוּ וַאֲתוֹ לְוָת מֹשֶׁה וּלְוָת אַהֲרוֹן וּלְוָת כָּל כְּנִשְׁתָּא דִּבְנֵי יִשְׂרָאֵל, לְמַדְבְּרָא דְּפָארָן--לִרְקַם; וַאֲתִיבוּ יָתְהוֹן פִּתְגָמָא וְיָת כָּל כְּנִשְׁתָּא, וְאַחְזִיאוּנוּן יָת אִבָּא דְּאַרְעָא. 
יג,כז וַיְסַפְּרוּ־לוֹ֙ וַיֹּ֣אמְר֔וּ בָּ֕אנוּ אֶל־הָאָ֖רֶץ אֲשֶׁ֣ר שְׁלַחְתָּ֑נוּ וְ֠גַם זָבַ֨ת חָלָ֥ב וּדְבַ֛שׁ הִ֖וא וְזֶה־פִּרְיָֽהּ׃ וְאִשְׁתַּעִיאוּ לֵיהּ, וַאֲמַרוּ, אֲתֵינָא, לְאַרְעָא דִּשְׁלַחְתַּנָא; וְאַף עָבְדָא חֲלָב וּדְבַשׁ, הִיא--וְדֵין אִבַּהּ. 
יג,כח אֶ֚פֶס כִּֽי־עַ֣ז הָעָ֔ם הַיֹּשֵׁ֖ב בָּאָ֑רֶץ וְהֶֽעָרִ֗ים בְּצֻר֤וֹת גְּדֹלֹת֙ מְאֹ֔ד וְגַם־יְלִדֵ֥י הָֽעֲנָ֖ק רָאִ֥ינוּ שָֽׁם׃ לְחוֹד אֲרֵי תַּקִּיף עַמָּא, דְּיָתֵיב בְּאַרְעָא; וְקִרְוַיָּא, כְּרִיכָן רַבְרְבָן לַחְדָּא, וְאַף בְּנֵי גִּבָּרָא, חֲזֵינָא תַּמָּן. 
יג,כט עֲמָלֵ֥ק יוֹשֵׁ֖ב בְּאֶ֣רֶץ הַנֶּ֑גֶב וְ֠הַֽחִתִּי וְהַיְבוּסִ֤י וְהָֽאֱמֹרִי֙ יוֹשֵׁ֣ב בָּהָ֔ר וְהַֽכְּנַעֲנִי֙ יוֹשֵׁ֣ב עַל־הַיָּ֔ם וְעַ֖ל יַ֥ד הַיַּרְדֵּֽן׃ עֲמָלְקָאָה יָתֵיב, בַּאֲרַע דָּרוֹמָא; וְחִתָּאָה וִיבוּסָאָה וֶאֱמוֹרָאָה, יָתֵיב בְּטוּרָא, וּכְנַעֲנָאָה יָתֵיב עַל יַמָּא, וְעַל כֵּיף יַרְדְּנָא. 
יג,ל וַיַּ֧הַס כָּלֵ֛ב אֶת־הָעָ֖ם אֶל־מֹשֶׁ֑ה וַיֹּ֗אמֶר עָלֹ֤ה נַֽעֲלֶה֙ וְיָרַ֣שְׁנוּ אֹתָ֔הּ כִּֽי־יָכ֥וֹל נוּכַ֖ל לָֽהּ׃ וְאַצֵּית כָּלֵב יָת עַמָּא, לְמֹשֶׁה; וַאֲמַר, מִסָּק נִסַּק וְנֵירַת יָתַהּ--אֲרֵי מִכָּל נִכּוֹל, לַהּ. 
יג,לא וְהָ֨אֲנָשִׁ֜ים אֲשֶׁר־עָל֤וּ עִמּוֹ֙ אָֽמְר֔וּ לֹ֥א נוּכַ֖ל לַֽעֲל֣וֹת אֶל־הָעָ֑ם כִּֽי־חָזָ֥ק ה֖וּא מִמֶּֽנּוּ׃ וְגֻבְרַיָּא דִּסְלִיקוּ עִמֵּיהּ, אֲמַרוּ, לָא נִכּוֹל, לְמִסַּק לְוָת עַמָּא:  אֲרֵי תַּקִּיף הוּא, מִנַּנָא. 
יג,לב וַיֹּצִ֜יאוּ דִּבַּ֤ת הָאָ֨רֶץ֙ אֲשֶׁ֣ר תָּר֣וּ אֹתָ֔הּ אֶל־בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר הָאָ֡רֶץ אֲשֶׁר֩ עָבַ֨רְנוּ בָ֜הּ לָת֣וּר אֹתָ֗הּ אֶ֣רֶץ אֹכֶ֤לֶת יֽוֹשְׁבֶ֨יהָ֙ הִ֔וא וְכָל־הָעָ֛ם אֲשֶׁר־רָאִ֥ינוּ בְתוֹכָ֖הּ אַנְשֵׁ֥י מִדּֽוֹת׃ וְאַפִּיקוּ שׁוֹם בִּישׁ עַל אַרְעָא, דְּאַלִּילוּ יָתַהּ, לְוָת בְּנֵי יִשְׂרָאֵל, לְמֵימַר:  אַרְעָא דַּעֲבַרְנָא בַּהּ לְאַלָּלָא יָתַהּ, אֲרַע מְקַטְּלָא יָתְבַהָא הִיא, וְכָל עַמָּא דַּחֲזֵינָא בְּגַוַּהּ, אֲנָשִׁין דְּמִשְׁחָן. 
יג,לג וְשָׁ֣ם רָאִ֗ינוּ אֶת־הַנְּפִילִ֛ים בְּנֵ֥י עֲנָ֖ק מִן־הַנְּפִלִ֑ים וַנְּהִ֤י בְעֵינֵ֨ינוּ֙ כַּֽחֲגָבִ֔ים וְכֵ֥ן הָיִ֖ינוּ בְּעֵֽינֵיהֶֽם׃ וְתַמָּן חֲזֵינָא, יָת גִּבָּרַיָּא בְּנֵי עֲנָק--מִן גִּבָּרַיָּא; וַהֲוֵינָא בְּעֵינֵי נַפְשַׁנָא כְּקִמְצִין, וְכֵין הֲוֵינָא בְּעֵינֵיהוֹן. 
יד,א וַתִּשָּׂא֙ כָּל־הָ֣עֵדָ֔ה וַֽיִּתְּנ֖וּ אֶת־קוֹלָ֑ם וַיִּבְכּ֥וּ הָעָ֖ם בַּלַּ֥יְלָה הַהֽוּא׃ וַאֲרֵימַת, כָּל כְּנִשְׁתָּא, וִיהַבוּ, יָת קָלְהוֹן; וּבְכוֹ עַמָּא, בְּלֵילְיָא הַהוּא. 
יד,ב וַיִּלֹּ֨נוּ֙ עַל־מֹשֶׁ֣ה וְעַֽל־אַהֲרֹ֔ן כֹּ֖ל בְּנֵ֣י יִשְׂרָאֵ֑ל וַיֹּֽאמְר֨וּ אֲלֵהֶ֜ם כָּל־הָֽעֵדָ֗ה לוּ־מַ֨תְנוּ֙ בְּאֶ֣רֶץ מִצְרַ֔יִם א֛וֹ בַּמִּדְבָּ֥ר הַזֶּ֖ה לוּ־מָֽתְנוּ׃ וְאִתְרָעַמוּ עַל מֹשֶׁה וְעַל אַהֲרוֹן, כֹּל בְּנֵי יִשְׂרָאֵל; וַאֲמַרוּ לְהוֹן כָּל כְּנִשְׁתָּא, לְוֵי דְּמֵיתְנָא בְּאַרְעָא דְּמִצְרַיִם, אוֹ בְּמַדְבְּרָא הָדֵין, לְוֵי דְּמֵיתְנָא. 
יד,ג וְלָמָ֣ה יְ֠הוָה מֵבִ֨יא אֹתָ֜נוּ אֶל־הָאָ֤רֶץ הַזֹּאת֙ לִנְפֹּ֣ל בַּחֶ֔רֶב נָשֵׁ֥ינוּ וְטַפֵּ֖נוּ יִֽהְי֣וּ לָבַ֑ז הֲל֧וֹא ט֦וֹב לָ֖נוּ שׁ֥וּב מִצְרָֽיְמָה׃ וּלְמָא יְיָ מַעֵיל יָתַנָא לְאַרְעָא הָדָא, לְמִפַּל בְּחַרְבָּא--נְשַׁנָא וְטַפְלַנָא, יְהוֹן לְבִזָּא; הֲלָא טָב לַנָא, דִּנְתוּב לְמִצְרָיִם. 
יד,ד וַיֹּֽאמְר֖וּ אִ֣ישׁ אֶל־אָחִ֑יו נִתְּנָ֥ה רֹ֖אשׁ וְנָשׁ֥וּבָה מִצְרָֽיְמָה׃ וַאֲמַרוּ, גְּבַר לַאֲחוּהִי:  נְמַנֵּי רֵישָׁא, וּנְתוּב לְמִצְרָיִם. 
יד,ה וַיִּפֹּ֥ל מֹשֶׁ֛ה וְאַֽהֲרֹ֖ן עַל־פְּנֵיהֶ֑ם לִפְנֵ֕י כָּל־קְהַ֥ל עֲדַ֖ת בְּנֵ֥י יִשְׂרָאֵֽל׃ וּנְפַל מֹשֶׁה וְאַהֲרוֹן, עַל אַפֵּיהוֹן, קֳדָם, כָּל קְהָלָא כְּנִשְׁתָּא דִּבְנֵי יִשְׂרָאֵל. 
יד,ו וִֽיהוֹשֻׁ֣עַ בִּן־נ֗וּן וְכָלֵב֙ בֶּן־יְפֻנֶּ֔ה מִן־הַתָּרִ֖ים אֶת־הָאָ֑רֶץ קָֽרְע֖וּ בִּגְדֵיהֶֽם׃ וִיהוֹשׁוּעַ בַּר נוּן, וְכָלֵב בַּר יְפֻנֶּה, מִן מְאַלְּלֵי, אַרְעָא--בַּזַּעוּ, לְבוּשֵׁיהוֹן. 
יד,ז וַיֹּ֣אמְר֔וּ אֶל־כָּל־עֲדַ֥ת בְּנֵֽי־יִשְׂרָאֵ֖ל לֵאמֹ֑ר הָאָ֗רֶץ אֲשֶׁ֨ר עָבַ֤רְנוּ בָהּ֙ לָת֣וּר אֹתָ֔הּ טוֹבָ֥ה הָאָ֖רֶץ מְאֹ֥ד מְאֹֽד׃ וַאֲמַרוּ, לְכָל כְּנִשְׁתָּא דִּבְנֵי יִשְׂרָאֵל לְמֵימַר:  אַרְעָא, דַּעֲבַרְנָא בַּהּ לְאַלָּלָא יָתַהּ--טָבָא אַרְעָא, לַחְדָּא לַחְדָּא. 
יד,ח אִם־חָפֵ֥ץ בָּ֨נוּ֙ יְהוָ֔ה וְהֵבִ֤יא אֹתָ֨נוּ֙ אֶל־הָאָ֣רֶץ הַזֹּ֔את וּנְתָנָ֖הּ לָ֑נוּ אֶ֕רֶץ אֲשֶׁר־הִ֛וא זָבַ֥ת חָלָ֖ב וּדְבָֽשׁ׃ אִם רַעֲוָא בַּנָא, קֳדָם יְיָ--וְיַעֵיל יָתַנָא לְאַרְעָא הָדָא, וְיִתְּנַהּ לַנָא:  אַרְעָא, דְּהִיא עָבְדָא חֲלָב וּדְבָשׁ. 
יד,ט אַ֣ךְ בַּֽיהוָה֮ אַל־תִּמְרֹדוּ֒ וְאַתֶּ֗ם אַל־תִּֽירְאוּ֙ אֶת־עַ֣ם הָאָ֔רֶץ כִּ֥י לַחְמֵ֖נוּ הֵ֑ם סָ֣ר צִלָּ֧ם מֵֽעֲלֵיהֶ֛ם וַֽיהוָ֥ה אִתָּ֖נוּ אַל־תִּֽירָאֻֽם׃ בְּרַם בְּמֵימְרָא דַּייָ, לָא תִּמְרְדוּן, וְאַתּוּן לָא תִּדְחֲלוּן מִן עַמָּא דְּאַרְעָא, אֲרֵי בִּידַנָא מְסִירִין אִנּוּן; עֲדָא תֻּקְפְהוֹן מִנְּהוֹן וּמֵימְרָא דַּייָ בְּסַעֲדַנָא, לָא תִּדְחֲלוּן מִנְּהוֹן. 
יד,י וַיֹּֽאמְרוּ֙ כָּל־הָ֣עֵדָ֔ה לִרְגּ֥וֹם אֹתָ֖ם בָּֽאֲבָנִ֑ים וּכְב֣וֹד יְהוָ֗ה נִרְאָה֙ בְּאֹ֣הֶל מוֹעֵ֔ד אֶֽל־כָּל־בְּנֵ֖י יִשְׂרָאֵֽל׃ {פ} וַאֲמַרוּ, כָּל כְּנִשְׁתָּא, לְמִרְגַּם יָתְהוֹן, בְּאַבְנַיָּא; וִיקָרָא דַּייָ, אִתְגְּלִי בְּמַשְׁכַּן זִמְנָא, לְכָל בְּנֵי, יִשְׂרָאֵל. {פ}
יד,יא וַיֹּ֤אמֶר יְהוָה֙ אֶל־מֹשֶׁ֔ה עַד־אָ֥נָה יְנַֽאֲצֻ֖נִי הָעָ֣ם הַזֶּ֑ה וְעַד־אָ֨נָה֙ לֹֽא־יַאֲמִ֣ינוּ בִ֔י בְּכֹל֙ הָֽאֹת֔וֹת אֲשֶׁ֥ר עָשִׂ֖יתִי בְּקִרְבּֽוֹ׃ וַאֲמַר יְיָ לְמֹשֶׁה, עַד אִמַּתִּי יְהוֹן מַרְגְּזִין קֳדָמַי עַמָּא הָדֵין; וְעַד אִמַּתִּי, לָא יְהֵימְנוּן בְּמֵימְרִי, בְּכֹל אָתַיָּא, דַּעֲבַדִית בֵּינֵיהוֹן. 
יד,יב אַכֶּ֥נּוּ בַדֶּ֖בֶר וְאֽוֹרִשֶׁ֑נּוּ וְאֶֽעֱשֶׂה֙ אֹֽתְךָ֔ לְגוֹי־גָּד֥וֹל וְעָצ֖וּם מִמֶּֽנּוּ׃ אֶמְחֵינוּן בְּמוֹתָא, וַאֲשֵׁיצֵינוּן; וְאַעֲבֵיד, יָתָךְ, לְעַם רַב וְתַקִּיף, מִנְּהוֹן. 
יד,יג וַיֹּ֥אמֶר מֹשֶׁ֖ה אֶל־יְהוָ֑ה וְשָֽׁמְע֣וּ מִצְרַ֔יִם כִּֽי־הֶעֱלִ֧יתָ בְכֹֽחֲךָ֛ אֶת־הָעָ֥ם הַזֶּ֖ה מִקִּרְבּֽוֹ׃ וַאֲמַר מֹשֶׁה, קֳדָם יְיָ:  וְיִשְׁמְעוּן מִצְרָאֵי, דַּאַסֵּיקְתָּא בְּחֵילָךְ יָת עַמָּא הָדֵין מִבֵּינֵיהוֹן. 
יד,יד וְאָֽמְר֗וּ אֶל־יוֹשֵׁב֮ הָאָ֣רֶץ הַזֹּאת֒ שָֽׁמְעוּ֙ כִּֽי־אַתָּ֣ה יְהוָ֔ה בְּקֶ֖רֶב הָעָ֣ם הַזֶּ֑ה אֲשֶׁר־עַ֨יִן בְּעַ֜יִן נִרְאָ֣ה ׀ אַתָּ֣ה יְהוָ֗ה וַעֲנָֽנְךָ֙ עֹמֵ֣ד עֲלֵהֶ֔ם וּבְעַמֻּ֣ד עָנָ֗ן אַתָּ֨ה הֹלֵ֤ךְ לִפְנֵיהֶם֙ יוֹמָ֔ם וּבְעַמּ֥וּד אֵ֖שׁ לָֽיְלָה׃ וְיֵימְרוּן, לְיָתֵיב אַרְעָא הָדָא, דִּשְׁמַעוּ אֲרֵי אַתְּ יְיָ, שְׁכִינְתָךְ שָׁרְיָא בְּגוֹ עַמָּא הָדֵין:  דִּבְעֵינֵיהוֹן חֲזוֹ שְׁכִינַת יְקָרָךְ יְיָ, וַעֲנָנָךְ מַטַּל עִלָּוֵיהוֹן, וּבְעַמּוּדָא דַּעֲנָנָא אַתְּ מְדַבַּר קֳדָמֵיהוֹן בִּימָמָא, וּבְעַמּוּדָא דְּאִישָׁתָא בְּלֵילְיָא. 
יד,טו וְהֵֽמַתָּ֛ה אֶת־הָעָ֥ם הַזֶּ֖ה כְּאִ֣ישׁ אֶחָ֑ד וְאָֽמְרוּ֙ הַגּוֹיִ֔ם אֲשֶׁר־שָֽׁמְע֥וּ אֶֽת־שִׁמְעֲךָ֖ לֵאמֹֽר׃ וּתְקַטֵּיל יָת עַמָּא הָדֵין, כְּגֻבְרָא חַד; וְיֵימְרוּן, עַמְמַיָּא, דִּשְׁמַעוּ יָת שֵׁימַע גְּבוּרְתָךְ, לְמֵימַר. 
יד,טז מִבִּלְתִּ֞י יְכֹ֣לֶת יְהוָ֗ה לְהָבִיא֙ אֶת־הָעָ֣ם הַזֶּ֔ה אֶל־הָאָ֖רֶץ אֲשֶׁר־נִשְׁבַּ֣ע לָהֶ֑ם וַיִּשְׁחָטֵ֖ם בַּמִּדְבָּֽר׃ מִדְּלֵית יֻכְלָא קֳדָם יְיָ, לְאַעָלָא יָת עַמָּא הָדֵין, לְאַרְעָא, דְּקַיֵּים לְהוֹן; וְקַטֵּילִנּוּן, בְּמַדְבְּרָא. 
יד,יז וְעַתָּ֕ה יִגְדַּל־נָ֖א כֹּ֣חַ אֲדֹנָ֑י כַּֽאֲשֶׁ֥ר דִּבַּ֖רְתָּ לֵאמֹֽר׃ וּכְעַן, סַגִּי כְּעַן חֵילָא קֳדָמָךְ יְיָ, כְּמָא דְּמַלֵּילְתָּא, לְמֵימַר. 
יד,יח יְהוָ֗ה אֶ֤רֶךְ אַפַּ֨יִם֙ וְרַב־חֶ֔סֶד נֹשֵׂ֥א עָו֖‍ֹן וָפָ֑שַׁע וְנַקֵּה֙ לֹ֣א יְנַקֶּ֔ה פֹּקֵ֞ד עֲו֤‍ֹן אָבוֹת֙ עַל־בָּנִ֔ים עַל־שִׁלֵּשִׁ֖ים וְעַל־רִבֵּעִֽים׃ יְיָ, מַרְחֵיק רְגַז וּמַסְגֵּי לְמֶעֱבַד טָבְוָן, שָׁבֵיק לַעֲוָיָן, וְלִמְרוֹד; סָלַח לִדְתָּיְבִין לְאוֹרָיְתֵיהּ, וְלִדְלָא תָּיְבִין לָא מְזַכֵּי--מַסְעַר חוֹבֵי אֲבָהָן עַל בְּנִין מָרָדִין, עַל דָּר תְּלִיתַאי וְעַל דָּר רְבִיעָאי. 
יד,יט סְלַֽח־נָ֗א לַֽעֲו֛‍ֹן הָעָ֥ם הַזֶּ֖ה כְּגֹ֣דֶל חַסְדֶּ֑ךָ וְכַֽאֲשֶׁ֤ר נָשָׂ֨אתָה֙ לָעָ֣ם הַזֶּ֔ה מִמִּצְרַ֖יִם וְעַד־הֵֽנָּה׃ שְׁבוֹק כְּעַן, לְחוֹבֵי עַמָּא הָדֵין--כְּסַגִּיאוּת טָבְוָתָךְ; וּכְמָא דִּשְׁבַקְתָּא לְעַמָּא הָדֵין, מִמִּצְרַיִם וְעַד כְּעַן. 
יד,כ וַיֹּ֣אמֶר יְהוָ֔ה סָלַ֖חְתִּי כִּדְבָרֶֽךָ׃ וַאֲמַר יְיָ, שְׁבַקִית כְּפִתְגָמָךְ. 
יד,כא וְאוּלָ֖ם חַי־אָ֑נִי וְיִמָּלֵ֥א כְבוֹד־יְהוָ֖ה אֶת־כָּל־הָאָֽרֶץ׃ וּבְרַם, קַיָּם אֲנָא:  וּמָלְיָא יְקָרָא דַּייָ, יָת כָּל אַרְעָא. 
יד,כב כִּ֣י כָל־הָֽאֲנָשִׁ֗ים הָֽרֹאִ֤ים אֶת־כְּבֹדִי֙ וְאֶת־אֹ֣תֹתַ֔י אֲשֶׁר־עָשִׂ֥יתִי בְמִצְרַ֖יִם וּבַמִּדְבָּ֑ר וַיְנַסּ֣וּ אֹתִ֗י זֶ֚ה עֶ֣שֶׂר פְּעָמִ֔ים וְלֹ֥א שָֽׁמְע֖וּ בְּקוֹלִֽי׃ אֲרֵי כָּל גֻּבְרַיָּא, דַּחֲזוֹ יָת יְקָרִי וְיָת אָתְוָתַי, דַּעֲבַדִית בְּמִצְרַיִם, וּבְמַדְבְּרָא; וְנַסִּיאוּ קֳדָמַי, דְּנָן עֲסַר זִמְנִין, וְלָא קַבִּילוּ, לְמֵימְרִי. 
יד,כג אִם־יִרְאוּ֙ אֶת־הָאָ֔רֶץ אֲשֶׁ֥ר נִשְׁבַּ֖עְתִּי לַֽאֲבֹתָ֑ם וְכָל־מְנַֽאֲצַ֖י לֹ֥א יִרְאֽוּהָ׃ אִם יִחְזוֹן, יָת אַרְעָא, דְּקַיֵּימִית, לַאֲבָהָתְהוֹן; וְכָל דְּאַרְגִּיזוּ קֳדָמַי, לָא יִחְזוֹנַהּ. 
יד,כד וְעַבְדִּ֣י כָלֵ֗ב עֵ֣קֶב הָֽיְתָ֞ה ר֤וּחַ אַחֶ֨רֶת֙ עִמּ֔וֹ וַיְמַלֵּ֖א אַֽחֲרָ֑י וַהֲבִֽיאֹתִ֗יו אֶל־הָאָ֨רֶץ֙ אֲשֶׁר־בָּ֣א שָׁ֔מָּה וְזַרְע֖וֹ יֽוֹרִשֶֽׁנָּה׃ וְעַבְדִּי כָּלֵב, חֲלָף דַּהֲוָת רוּחַ אֻחְרִי עִמֵּיהּ, וְאַשְׁלֵים, בָּתַר דַּחְלְתִי--וְאַעֵילִנֵּיהּ, לְאַרְעָא דְּעָאל לְתַמָּן, וּבְנוֹהִי, יוֹרְתִנַּהּ. 
יד,כה וְהָֽעֲמָלֵקִ֥י וְהַֽכְּנַעֲנִ֖י יוֹשֵׁ֣ב בָּעֵ֑מֶק מָחָ֗ר פְּנ֨וּ וּסְע֥וּ לָכֶ֛ם הַמִּדְבָּ֖ר דֶּ֥רֶךְ יַם־סֽוּף׃ {פ} וַעֲמָלְקָאָה וּכְנַעֲנָאָה, יָתֵיב בְּמֵישְׁרָא; מְחַר, אִתְפְּנוֹ וְטוּלוּ לְכוֹן לְמַדְבְּרָא--אוֹרַח יַמָּא דְּסוּף. {פ}
יד,כו וַיְדַבֵּ֣ר יְהוָ֔ה אֶל־מֹשֶׁ֥ה וְאֶֽל־אַהֲרֹ֖ן לֵאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה וּלְאַהֲרוֹן לְמֵימַר. 
יד,כז עַד־מָתַ֗י לָֽעֵדָ֤ה הָֽרָעָה֙ הַזֹּ֔את אֲשֶׁ֛ר הֵ֥מָּה מַלִּינִ֖ים עָלָ֑י אֶת־תְּלֻנּ֞וֹת בְּנֵ֣י יִשְׂרָאֵ֗ל אֲשֶׁ֨ר הֵ֧מָּה מַלִּינִ֛ים עָלַ֖י שָׁמָֽעְתִּי׃ עַד אִמַּתִּי, לִכְנִשְׁתָּא בִּשְׁתָּא הָדָא, דְּאִנּוּן מִתְרָעֲמִין, עֲלָי; יָת תֻּרְעֲמָת בְּנֵי יִשְׂרָאֵל, דְּאִנּוּן מִתְרָעֲמִין עֲלַי--שְׁמִיעַ קֳדָמָי. 
יד,כח אֱמֹ֣ר אֲלֵהֶ֗ם חַי־אָ֨נִי֙ נְאֻם־יְהוָ֔ה אִם־לֹ֕א כַּֽאֲשֶׁ֥ר דִּבַּרְתֶּ֖ם בְּאָזְנָ֑י כֵּ֖ן אֶֽעֱשֶׂ֥ה לָכֶֽם׃ אֵימַר לְהוֹן, קַיָּם אֲנָא אֲמַר יְיָ, אִם לָא, כְּמָא דְּמַלֵּילְתּוּן קֳדָמָי:  כֵּין, אַעֲבֵיד לְכוֹן. 
יד,כט בַּמִּדְבָּ֣ר הַ֠זֶּה יִפְּל֨וּ פִגְרֵיכֶ֜ם וְכָל־פְּקֻֽדֵיכֶם֙ לְכָל־מִסְפַּרְכֶ֔ם מִבֶּ֛ן עֶשְׂרִ֥ים שָׁנָ֖ה וָמָ֑עְלָה אֲשֶׁ֥ר הֲלִֽינֹתֶ֖ם עָלָֽי׃ בְּמַדְבְּרָא הָדֵין יִפְּלוּן פִּגְרֵיכוֹן וְכָל מִנְיָנְכוֹן, לְכָל חֻשְׁבָּנְכוֹן, מִבַּר עַסְרִין שְׁנִין, וּלְעֵילָא:  דְּאִתְרָעֲמִתּוּן, עֲלָי. 
יד,ל אִם־אַתֶּם֙ תָּבֹ֣אוּ אֶל־הָאָ֔רֶץ אֲשֶׁ֤ר נָשָׂ֨אתִי֙ אֶת־יָדִ֔י לְשַׁכֵּ֥ן אֶתְכֶ֖ם בָּ֑הּ כִּ֚י אִם־כָּלֵ֣ב בֶּן־יְפֻנֶּ֔ה וִֽיהוֹשֻׁ֖עַ בִּן־נֽוּן׃ אִם אַתּוּן, תֵּיעֲלוּן לְאַרְעָא, דְּקַיֵּימִית בְּמֵימְרִי, לְאַשְׁרָאָה יָתְכוֹן בַּהּ--אֱלָהֵין כָּלֵב בַּר יְפֻנֶּה, וִיהוֹשׁוּעַ בַּר נוּן. 
יד,לא וְטַ֨פְּכֶ֔ם אֲשֶׁ֥ר אֲמַרְתֶּ֖ם לָבַ֣ז יִֽהְיֶ֑ה וְהֵֽבֵיאתִ֣י אֹתָ֔ם וְיָֽדְעוּ֙ אֶת־הָאָ֔רֶץ אֲשֶׁ֥ר מְאַסְתֶּ֖ם בָּֽהּ׃ וְטַפְלְכוֹן--דַּאֲמַרְתּוּן, לְבִזָּא יְהוֹן:  וְאַעֵיל יָתְהוֹן--וְיִדְּעוּן יָת אַרְעָא, דְּקַצְתּוּן בַּהּ. 
יד,לב וּפִגְרֵיכֶ֖ם אַתֶּ֑ם יִפְּל֖וּ בַּמִּדְבָּ֥ר הַזֶּֽה׃ וּפִגְרֵיכוֹן, דִּילְכוֹן--יִפְּלוּן, בְּמַדְבְּרָא הָדֵין. 
יד,לג וּ֠בְנֵיכֶם יִֽהְי֨וּ רֹעִ֤ים בַּמִּדְבָּר֙ אַרְבָּעִ֣ים שָׁנָ֔ה וְנָֽשְׂא֖וּ אֶת־זְנֽוּתֵיכֶ֑ם עַד־תֹּ֥ם פִּגְרֵיכֶ֖ם בַּמִּדְבָּֽר׃ וּבְנֵיכוֹן יְהוֹן מְאַחֲרִין בְּמַדְבְּרָא, אַרְבְּעִין שְׁנִין, וִיקַבְּלוּן, יָת חוֹבֵיכוֹן--עַד דִּיסוּפוּן פִּגְרֵיכוֹן, בְּמַדְבְּרָא. 
יד,לד בְּמִסְפַּ֨ר הַיָּמִ֜ים אֲשֶׁר־תַּרְתֶּ֣ם אֶת־הָאָרֶץ֮ אַרְבָּעִ֣ים יוֹם֒ י֣וֹם לַשָּׁנָ֞ה י֣וֹם לַשָּׁנָ֗ה תִּשְׂאוּ֙ אֶת־עֲו‍ֹנֹ֣תֵיכֶ֔ם אַרְבָּעִ֖ים שָׁנָ֑ה וִֽידַעְתֶּ֖ם אֶת־תְּנֽוּאָתִֽי׃ בְּמִנְיַן יוֹמַיָּא דְּאַלֵּילְתּוּן יָת אַרְעָא, אַרְבְּעִין יוֹמִין--יוֹמָא לְשַׁתָּא יוֹמָא לְשַׁתָּא תְּקַבְּלוּן יָת חוֹבֵיכוֹן, אַרְבְּעִין שְׁנִין; וְתִדְּעוּן, יָת דְּאִתְרָעֲמִתּוּן עֲלָי. 
יד,לה אֲנִ֣י יְהוָה֮ דִּבַּרְתִּי֒ אִם־לֹ֣א ׀ זֹ֣את אֶֽעֱשֶׂ֗ה לְכָל־הָֽעֵדָ֤ה הָֽרָעָה֙ הַזֹּ֔את הַנּֽוֹעָדִ֖ים עָלָ֑י בַּמִּדְבָּ֥ר הַזֶּ֛ה יִתַּ֖מּוּ וְשָׁ֥ם יָמֻֽתוּ׃ אֲנָא יְיָ, גְּזַרִית בְּמֵימְרִי, אִם לָא דָּא אַעֲבֵיד לְכָל כְּנִשְׁתָּא בִּשְׁתָּא הָדָא, דְּאִזְדָּמַנוּ עֲלָי; בְּמַדְבְּרָא הָדֵין יְסוּפוּן, וְתַמָּן יְמוּתוּן. 
יד,לו וְהָ֣אֲנָשִׁ֔ים אֲשֶׁר־שָׁלַ֥ח מֹשֶׁ֖ה לָת֣וּר אֶת־הָאָ֑רֶץ וַיָּשֻׁ֗בוּ וילונו (וַיַּלִּ֤ינוּ) עָלָיו֙ אֶת־כָּל־הָ֣עֵדָ֔ה לְהוֹצִ֥יא דִבָּ֖ה עַל־הָאָֽרֶץ׃ וְגֻבְרַיָּא, דִּשְׁלַח מֹשֶׁה לְאַלָּלָא יָת אַרְעָא; וְתָבוּ, וְאַרְעִימוּ עֲלוֹהִי יָת כָּל כְּנִשְׁתָּא, לְאַפָּקָא שׁוֹם בִּישׁ, עַל אַרְעָא. 
יד,לז וַיָּמֻ֨תוּ֙ הָֽאֲנָשִׁ֔ים מֽוֹצִאֵ֥י דִבַּת־הָאָ֖רֶץ רָעָ֑ה בַּמַּגֵּפָ֖ה לִפְנֵ֥י יְהוָֽה׃ וּמִיתוּ, גֻּבְרַיָּא, דְּאַפִּיקוּ שׁוֹם בִּישׁ, עַל אַרְעָא--בְּמוֹתָנָא, קֳדָם יְיָ. 
יד,לח וִֽיהוֹשֻׁ֣עַ בִּן־נ֔וּן וְכָלֵ֖ב בֶּן־יְפֻנֶּ֑ה חָיוּ֙ מִן־הָֽאֲנָשִׁ֣ים הָהֵ֔ם הַֽהֹלְכִ֖ים לָת֥וּר אֶת־הָאָֽרֶץ׃ וִיהוֹשׁוּעַ בַּר נוּן, וְכָלֵב בַּר יְפֻנֶּה, אִתְקַיַּמוּ מִן גֻּבְרַיָּא הָאִנּוּן, דַּאֲזַלוּ לְאַלָּלָא יָת אַרְעָא. 
יד,לט וַיְדַבֵּ֤ר מֹשֶׁה֙ אֶת־הַדְּבָרִ֣ים הָאֵ֔לֶּה אֶֽל־כָּל־בְּנֵ֖י יִשְׂרָאֵ֑ל וַיִּֽתְאַבְּל֥וּ הָעָ֖ם מְאֹֽד׃ וּמַלֵּיל מֹשֶׁה יָת פִּתְגָמַיָּא הָאִלֵּין, עִם כָּל בְּנֵי יִשְׂרָאֵל; וְאִתְאַבַּלוּ עַמָּא, לַחְדָּא. 
יד,מ וַיַּשְׁכִּ֣מוּ בַבֹּ֔קֶר וַיַּֽעֲל֥וּ אֶל־רֹאשׁ־הָהָ֖ר לֵאמֹ֑ר הִנֶּ֗נּוּ וְעָלִ֛ינוּ אֶל־הַמָּק֛וֹם אֲשֶׁר־אָמַ֥ר יְהוָ֖ה כִּ֥י חָטָֽאנוּ׃ וְאַקְדִּימוּ בְּצַפְרָא, וּסְלִיקוּ לְרֵישׁ טוּרָא לְמֵימַר:  הָא אֲנַחְנָא, סָלְקִין לְאַתְרָא דַּאֲמַר יְיָ--אֲרֵי חַבְנָא. 
יד,מא וַיֹּ֣אמֶר מֹשֶׁ֔ה לָ֥מָּה זֶּ֛ה אַתֶּ֥ם עֹֽבְרִ֖ים אֶת־פִּ֣י יְהוָ֑ה וְהִ֖וא לֹ֥א תִצְלָֽח׃ וַאֲמַר מֹשֶׁה, לְמָא דְּנָן אַתּוּן עָבְרִין עַל גְּזֵירַת מֵימְרָא דַּייָ; וְהִיא, לָא תַּצְלַח. 
יד,מב אַֽל־תַּעֲל֔וּ כִּ֛י אֵ֥ין יְהוָ֖ה בְּקִרְבְּכֶ֑ם וְלֹא֙ תִּנָּ֣גְפ֔וּ לִפְנֵ֖י אֹֽיְבֵיכֶֽם׃ לָא תִּסְּקוּן, אֲרֵי לֵית שְׁכִינְתָא דַּייָ בֵּינֵיכוֹן; וְלָא, תִּתַּבְרוּן, קֳדָם, בַּעֲלֵי דְּבָבֵיכוֹן. 
יד,מג כִּי֩ הָעֲמָֽלֵקִ֨י וְהַכְּנַֽעֲנִ֥י שָׁם֙ לִפְנֵיכֶ֔ם וּנְפַלְתֶּ֖ם בֶּחָ֑רֶב כִּֽי־עַל־כֵּ֤ן שַׁבְתֶּם֙ מֵאַֽחֲרֵ֣י יְהוָ֔ה וְלֹֽא־יִהְיֶ֥ה יְהוָ֖ה עִמָּכֶֽם׃ אֲרֵי עֲמָלְקָאָה וּכְנַעֲנָאָה תַּמָּן קֳדָמֵיכוֹן, וְתִפְּלוּן בְּחַרְבָּא:  אֲרֵי עַל כֵּין תַּבְתּוּן מִבָּתַר פֻּלְחָנָא דַּייָ, וְלָא יְהֵי מֵימְרָא דַּייָ בְּסַעְדְּכוֹן. 
יד,מד וַיַּעְפִּ֕לוּ לַֽעֲל֖וֹת אֶל־רֹ֣אשׁ הָהָ֑ר וַֽאֲר֤וֹן בְּרִית־יְהוָה֙ וּמֹשֶׁ֔ה לֹא־מָ֖שׁוּ מִקֶּ֥רֶב הַֽמַּחֲנֶֽה׃ וְאַרְשַׁעוּ, לְמִסַּק לְרֵישׁ טוּרָא; וַאֲרוֹן קְיָמָא דַּייָ וּמֹשֶׁה, לָא עֲדוֹ מִגּוֹ מַשְׁרִיתָא. 
יד,מה וַיֵּ֤רֶד הָעֲמָֽלֵקִי֙ וְהַֽכְּנַעֲנִ֔י הַיֹּשֵׁ֖ב בָּהָ֣ר הַה֑וּא וַיַּכּ֥וּם וַֽיַּכְּת֖וּם עַד־הַֽחָרְמָֽה׃ {פ} וּנְחַת עֲמָלְקָאָה וּכְנַעֲנָאָה, דְּיָתֵיב בְּטוּרָא הַהוּא; וּמְחוֹנוּן וּטְרַדוּנוּן, עַד חָרְמָה. {פ}
טו,א וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
טו,ב דַּבֵּר֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָֽמַרְתָּ֖ אֲלֵהֶ֑ם כִּ֣י תָבֹ֗אוּ אֶל־אֶ֨רֶץ֙ מוֹשְׁבֹ֣תֵיכֶ֔ם אֲשֶׁ֥ר אֲנִ֖י נֹתֵ֥ן לָכֶֽם׃ מַלֵּיל עִם בְּנֵי יִשְׂרָאֵל, וְתֵימַר לְהוֹן:  אֲרֵי תֵּיעֲלוּן, לַאֲרַע מוֹתְבָנֵיכוֹן, דַּאֲנָא, יָהֵיב לְכוֹן. 
טו,ג וַֽעֲשִׂיתֶ֨ם אִשֶּׁ֤ה לַֽיהוָה֙ עֹלָ֣ה אוֹ־זֶ֔בַח לְפַלֵּא־נֶ֨דֶר֙ א֣וֹ בִנְדָבָ֔ה א֖וֹ בְּמֹֽעֲדֵיכֶ֑ם לַֽעֲשׂ֞וֹת רֵ֤יחַ נִיחֹ֨חַ֙ לַֽיהוָ֔ה מִן־הַבָּקָ֖ר א֥וֹ מִן־הַצֹּֽאן׃ וְתַעְבְּדוּן קֻרְבָּנָא קֳדָם יְיָ, עֲלָתָא אוֹ נִכְסַת קֻדְשַׁיָּא, לְפָרָשָׁא נִדְרָא אוֹ בִּנְדַבְתָּא, אוֹ בְּמוֹעֲדֵיכוֹן--לְמֶעֱבַד לְאִתְקַבָּלָא בְּרַעֲוָא, קֳדָם יְיָ, מִן תּוֹרֵי, אוֹ מִן עָנָא. 
טו,ד וְהִקְרִ֛יב הַמַּקְרִ֥יב קָרְבָּנ֖וֹ לַֽיהוָ֑ה מִנְחָה֙ סֹ֣לֶת עִשָּׂר֔וֹן בָּל֕וּל בִּרְבִעִ֥ית הַהִ֖ין שָֽׁמֶן׃ וִיקָרֵיב דִּמְקָרֵיב קֻרְבְּנֵיהּ, קֳדָם יְיָ--מִנְחָתָא, סֻלְתָּא עֶסְרוֹנָא, דְּפִיל, בְּרַבְעוּת הִינָא מִשְׁחָא. 
טו,ה וְיַ֤יִן לַנֶּ֨סֶךְ֙ רְבִיעִ֣ית הַהִ֔ין תַּֽעֲשֶׂ֥ה עַל־הָֽעֹלָ֖ה א֣וֹ לַזָּ֑בַח לַכֶּ֖בֶשׂ הָֽאֶחָֽד׃ וְחַמְרָא לְנִסְכָּא רַבְעוּת הִינָא, תַּעֲבֵיד עַל עֲלָתָא אוֹ לְנִכְסַת קֻדְשַׁיָּא--לְאִמְּרָא, חַד. 
טו,ו א֤וֹ לָאַ֨יִל֙ תַּֽעֲשֶׂ֣ה מִנְחָ֔ה סֹ֖לֶת שְׁנֵ֣י עֶשְׂרֹנִ֑ים בְּלוּלָ֥ה בַשֶּׁ֖מֶן שְׁלִשִׁ֥ית הַהִֽין׃ אוֹ לְדִכְרָא תַּעֲבֵיד מִנְחָתָא, סֻלְתָּא תְּרֵין עֶסְרוֹנִין, דְּפִילָא בִּמְשַׁח, תַּלְתוּת הִינָא. 
טו,ז וְיַ֥יִן לַנֶּ֖סֶךְ שְׁלִשִׁ֣ית הַהִ֑ין תַּקְרִ֥יב רֵֽיחַ־נִיחֹ֖חַ לַֽיהוָֽה׃ וְחַמְרָא לְנִסְכָּא, תַּלְתוּת הִינָא--תְּקָרֵיב לְאִתְקַבָּלָא בְּרַעֲוָא, קֳדָם יְיָ. 
טו,ח וְכִֽי־תַעֲשֶׂ֥ה בֶן־בָּקָ֖ר עֹלָ֣ה אוֹ־זָ֑בַח לְפַלֵּא־נֶ֥דֶר אֽוֹ־שְׁלָמִ֖ים לַֽיהוָֽה׃ וַאֲרֵי תַּעֲבֵיד בַּר תּוֹרֵי, עֲלָתָא אוֹ נִכְסַת קֻדְשַׁיָּא, לְפָרָשָׁא נִדְרָא אוֹ נִכְסַת קֻדְשַׁיָּא, קֳדָם יְיָ. 
טו,ט וְהִקְרִ֤יב עַל־בֶּן־הַבָּקָר֙ מִנְחָ֔ה סֹ֖לֶת שְׁלֹשָׁ֣ה עֶשְׂרֹנִ֑ים בָּל֥וּל בַּשֶּׁ֖מֶן חֲצִ֥י הַהִֽין׃ וִיקָרֵיב עַל בַּר תּוֹרֵי מִנְחָתָא, סֻלְתָּא תְּלָתָא עֶסְרוֹנִין, דְּפִיל בִּמְשַׁח, פַּלְגוּת הִינָא. 
טו,י וְיַ֛יִן תַּקְרִ֥יב לַנֶּ֖סֶךְ חֲצִ֣י הַהִ֑ין אִשֵּׁ֥ה רֵֽיחַ־נִיחֹ֖חַ לַֽיהוָֽה׃ וְחַמְרָא תְּקָרֵיב לְנִסְכָּא, פַּלְגוּת הִינָא--קֻרְבַּן דְּמִתְקַבַּל בְּרַעֲוָא, קֳדָם יְיָ. 
טו,יא כָּ֣כָה יֵֽעָשֶׂ֗ה לַשּׁוֹר֙ הָֽאֶחָ֔ד א֖וֹ לָאַ֣יִל הָֽאֶחָ֑ד אֽוֹ־לַשֶּׂ֥ה בַכְּבָשִׂ֖ים א֥וֹ בָֽעִזִּֽים׃ כְּדֵין יִתְעֲבֵיד, לְתוֹרָא חַד, אוֹ, לְדִכְרָא חַד; אוֹ לְאִמַּר בְּאִמְּרַיָּא, אוֹ בְּעִזַּיָּא. 
טו,יב כַּמִּסְפָּ֖ר אֲשֶׁ֣ר תַּֽעֲשׂ֑וּ כָּ֛כָה תַּֽעֲשׂ֥וּ לָֽאֶחָ֖ד כְּמִסְפָּרָֽם׃ כְּמִנְיַן, דְּתַעְבְּדוּן--כְּדֵין תַּעְבְּדוּן לְחַד, כְּמִנְיָנְהוֹן. 
טו,יג כָּל־הָֽאֶזְרָ֥ח יַֽעֲשֶׂה־כָּ֖כָה אֶת־אֵ֑לֶּה לְהַקְרִ֛יב אִשֵּׁ֥ה רֵֽיחַ־נִיחֹ֖חַ לַֽיהוָֽה׃ כָּל יַצִּיבָא יַעֲבֵיד כְּדֵין, יָת אִלֵּין, לְקָרָבָא קֻרְבַּן דְּמִתְקַבַּל בְּרַעֲוָא, קֳדָם יְיָ. 
טו,יד וְכִֽי־יָגוּר֩ אִתְּכֶ֨ם גֵּ֜ר א֤וֹ אֲשֶׁר־בְּתֽוֹכְכֶם֙ לְדֹרֹ֣תֵיכֶ֔ם וְעָשָׂ֛ה אִשֵּׁ֥ה רֵֽיחַ־נִיחֹ֖חַ לַֽיהוָ֑ה כַּֽאֲשֶׁ֥ר תַּֽעֲשׂ֖וּ כֵּ֥ן יַֽעֲשֶֽׂה׃ וַאֲרֵי יִתְגַּיַּר עִמְּכוֹן גִּיּוֹרָא אוֹ דְּבֵינֵיכוֹן, לְדָרֵיכוֹן, וְיַעֲבֵיד קֻרְבַּן דְּמִתְקַבַּל בְּרַעֲוָא, קֳדָם יְיָ--כְּמָא דְּתַעְבְּדוּן, כֵּין יַעֲבֵיד. 
טו,טו הַקָּהָ֕ל חֻקָּ֥ה אַחַ֛ת לָכֶ֖ם וְלַגֵּ֣ר הַגָּ֑ר חֻקַּ֤ת עוֹלָם֙ לְדֹרֹ֣תֵיכֶ֔ם כָּכֶ֛ם כַּגֵּ֥ר יִֽהְיֶ֖ה לִפְנֵ֥י יְהוָֽה׃ קְהָלָא, קְיָמָא חַד לְכוֹן וּלְגִיּוֹרַיָּא דְּיִתְגַּיְּרוּן:  קְיָם עָלַם לְדָרֵיכוֹן, כְּוָתְכוֹן גִּיּוֹרָא יְהֵי קֳדָם יְיָ. 
טו,טז תּוֹרָ֥ה אַחַ֛ת וּמִשְׁפָּ֥ט אֶחָ֖ד יִֽהְיֶ֣ה לָכֶ֑ם וְלַגֵּ֖ר הַגָּ֥ר אִתְּכֶֽם׃ {פ} אוֹרָיְתָא חֲדָא וְדִינָא חַד, יְהֵי לְכוֹן, וּלְגִיּוֹרַיָּא, דְּיִתְגַּיְּרוּן עִמְּכוֹן. {פ}
טו,יז וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וּמַלֵּיל יְיָ, עִם מֹשֶׁה לְמֵימַר. 
טו,יח דַּבֵּר֙ אֶל־בְּנֵ֣י יִשְׂרָאֵ֔ל וְאָֽמַרְתָּ֖ אֲלֵהֶ֑ם בְּבֹֽאֲכֶם֙ אֶל־הָאָ֔רֶץ אֲשֶׁ֥ר אֲנִ֛י מֵבִ֥יא אֶתְכֶ֖ם שָֽׁמָּה׃ מַלֵּיל עִם בְּנֵי יִשְׂרָאֵל, וְתֵימַר לְהוֹן:  בְּמֵיעַלְכוֹן, לְאַרְעָא, דַּאֲנָא מַעֵיל יָתְכוֹן, לְתַמָּן. 
טו,יט וְהָיָ֕ה בַּֽאֲכָלְכֶ֖ם מִלֶּ֣חֶם הָאָ֑רֶץ תָּרִ֥ימוּ תְרוּמָ֖ה לַֽיהוָֽה׃ וִיהֵי, בְּמֵיכַלְכוֹן מִלַּחְמָא דְּאַרְעָא--תַּפְרְשׁוּן אַפְרָשׁוּתָא, קֳדָם יְיָ. 
טו,כ רֵאשִׁית֙ עֲרִסֹ֣תֵכֶ֔ם חַלָּ֖ה תָּרִ֣ימוּ תְרוּמָ֑ה כִּתְרוּמַ֣ת גֹּ֔רֶן כֵּ֖ן תָּרִ֥ימוּ אֹתָֽהּ׃ רֵישׁ, אָצְוָתְכוֹן--חַלְּתָא, תַּפְרְשׁוּן אַפְרָשׁוּתָא:  כְּמָא דְּמַפְרְשִׁין מִן אִדְּרָא, כֵּין תַּפְרְשׁוּן יָתַהּ. 
טו,כא מֵֽרֵאשִׁית֙ עֲרִסֹ֣תֵיכֶ֔ם תִּתְּנ֥וּ לַֽיהוָ֖ה תְּרוּמָ֑ה לְדֹרֹ֖תֵיכֶֽם׃ {ס} מֵרֵישׁ, אָצְוָתְכוֹן, תִּתְּנוּן קֳדָם יְיָ, אַפְרָשׁוּתָא--לְדָרֵיכוֹן. {ס}
טו,כב וְכִ֣י תִשְׁגּ֔וּ וְלֹ֣א תַֽעֲשׂ֔וּ אֵ֥ת כָּל־הַמִּצְו֖‍ֹת הָאֵ֑לֶּה אֲשֶׁר־דִּבֶּ֥ר יְהוָ֖ה אֶל־מֹשֶֽׁה׃ וַאֲרֵי תִּשְׁתְּלוּן--וְלָא תַּעְבְּדוּן, יָת כָּל פִּקּוֹדַיָּא הָאִלֵּין:  דְּמַלֵּיל יְיָ, עִם מֹשֶׁה. 
טו,כג אֵת֩ כָּל־אֲשֶׁ֨ר צִוָּ֧ה יְהוָ֛ה אֲלֵיכֶ֖ם בְּיַד־מֹשֶׁ֑ה מִן־הַיּ֞וֹם אֲשֶׁ֨ר צִוָּ֧ה יְהוָ֛ה וָהָ֖לְאָה לְדֹרֹֽתֵיכֶֽם׃ יָת כָּל דְּפַקֵּיד יְיָ לְכוֹן, בִּידָא דְּמֹשֶׁה:  מִן יוֹמָא דְּפַקֵּיד יְיָ, וּלְהַלְאָה--לְדָרֵיכוֹן. 
טו,כד וְהָיָ֗ה אִ֣ם מֵֽעֵינֵ֣י הָֽעֵדָה֮ נֶֽעֶשְׂתָ֣ה לִשְׁגָגָה֒ וְעָשׂ֣וּ כָל־הָֽעֵדָ֡ה פַּ֣ר בֶּן־בָּקָר֩ אֶחָ֨ד לְעֹלָ֜ה לְרֵ֤יחַ נִיחֹ֨חַ֙ לַֽיהוָ֔ה וּמִנְחָת֥וֹ וְנִסְכּ֖וֹ כַּמִּשְׁפָּ֑ט וּשְׂעִיר־עִזִּ֥ים אֶחָ֖ד לְחַטָּֽת׃ וִיהֵי, אִם מֵעֵינֵי כְּנִשְׁתָּא אִתְעֲבֵידַת לְשָׁלוּ, וְיַעְבְּדוּן כָּל כְּנִשְׁתָּא תּוֹר בַּר תּוֹרֵי חַד לַעֲלָתָא לְאִתְקַבָּלָא בְּרַעֲוָא קֳדָם יְיָ, וּמִנְחָתֵיהּ וְנִסְכֵּיהּ כְּדַחְזֵי; וּצְפִיר בַּר עִזִּין חַד, לְחַטָּתָא. 
טו,כה וְכִפֶּ֣ר הַכֹּהֵ֗ן עַֽל־כָּל־עֲדַ֛ת בְּנֵ֥י יִשְׂרָאֵ֖ל וְנִסְלַ֣ח לָהֶ֑ם כִּֽי־שְׁגָגָ֣ה הִ֔וא וְהֵם֩ הֵבִ֨יאוּ אֶת־קָרְבָּנָ֜ם אִשֶּׁ֣ה לַֽיהוָ֗ה וְחַטָּאתָ֛ם לִפְנֵ֥י יְהוָ֖ה עַל־שִׁגְגָתָֽם׃ וִיכַפַּר כָּהֲנָא, עַל כָּל כְּנִשְׁתָּא דִּבְנֵי יִשְׂרָאֵל--וְיִשְׁתְּבֵיק לְהוֹן:  אֲרֵי שָׁלוּתָא הִיא--וְאִנּוּן אֵיתִיאוּ יָת קֻרְבָּנְהוֹן קֻרְבָּנָא קֳדָם יְיָ וְחוֹבַתְהוֹן קֳדָם יְיָ, עַל שָׁלוּתְהוֹן. 
טו,כו וְנִסְלַ֗ח לְכָל־עֲדַת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל וְלַגֵּ֖ר הַגָּ֣ר בְּתוֹכָ֑ם כִּ֥י לְכָל־הָעָ֖ם בִּשְׁגָגָֽה׃ {ס} וְיִשְׁתְּבֵיק, לְכָל כְּנִשְׁתָּא דִּבְנֵי יִשְׂרָאֵל, וּלְגִיּוֹרַיָּא, דְּיִתְגַּיְּרוּן בֵּינֵיהוֹן:  אֲרֵי לְכָל עַמָּא, בְּשָׁלוּתָא. {ס}
טו,כז וְאִם־נֶ֥פֶשׁ אַחַ֖ת תֶּֽחֱטָ֣א בִשְׁגָגָ֑ה וְהִקְרִ֛יבָה עֵ֥ז בַּת־שְׁנָתָ֖הּ לְחַטָּֽאת׃ וְאִם אֲנָשׁ חַד, יְחוּב בְּשָׁלוּ--וִיקָרֵיב עִזָּא בַּת שַׁתַּהּ, לְחַטָּתָא. 
טו,כח וְכִפֶּ֣ר הַכֹּהֵ֗ן עַל־הַנֶּ֧פֶשׁ הַשֹּׁגֶ֛גֶת בְּחֶטְאָ֥ה בִשְׁגָגָ֖ה לִפְנֵ֣י יְהוָ֑ה לְכַפֵּ֥ר עָלָ֖יו וְנִסְלַ֥ח לֽוֹ׃ וִיכַפַּר כָּהֲנָא, עַל אֲנָשׁ דְּאִשְׁתְּלִי בְּמֵיחֲבֵיהּ בְּשָׁלוּ--קֳדָם יְיָ:  לְכַפָּרָא עֲלוֹהִי, וְיִשְׁתְּבֵיק לֵיהּ. 
טו,כט הָֽאֶזְרָח֙ בִּבְנֵ֣י יִשְׂרָאֵ֔ל וְלַגֵּ֖ר הַגָּ֣ר בְּתוֹכָ֑ם תּוֹרָ֤ה אַחַת֙ יִֽהְיֶ֣ה לָכֶ֔ם לָֽעֹשֶׂ֖ה בִּשְׁגָגָֽה׃ יַצִּיבָא בִּבְנֵי יִשְׂרָאֵל, וּלְגִיּוֹרַיָּא דְּיִתְגַּיְּרוּן בֵּינֵיהוֹן--אוֹרָיְתָא חֲדָא יְהֵי לְכוֹן, לִדְיַעֲבֵיד בְּשָׁלוּ. 
טו,ל וְהַנֶּ֜פֶשׁ אֲשֶֽׁר־תַּעֲשֶׂ֣ה ׀ בְּיָ֣ד רָמָ֗ה מִן־הָֽאֶזְרָח֙ וּמִן־הַגֵּ֔ר אֶת־יְהוָ֖ה ה֣וּא מְגַדֵּ֑ף וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מִקֶּ֥רֶב עַמָּֽהּ׃ וַאֲנָשׁ דְּיַעֲבֵיד בְּרֵישׁ גְּלֵי, מִן יַצִּיבַיָּא וּמִן גִּיּוֹרַיָּא--קֳדָם יְיָ, הוּא מַרְגֵּיז; וְיִשְׁתֵּיצֵי אֲנָשָׁא הַהוּא, מִגּוֹ עַמֵּיהּ. 
טו,לא כִּ֤י דְבַר־יְהוָה֙ בָּזָ֔ה וְאֶת־מִצְוָת֖וֹ הֵפַ֑ר הִכָּרֵ֧ת ׀ תִּכָּרֵ֛ת הַנֶּ֥פֶשׁ הַהִ֖וא עֲו‍ֹנָ֥ה בָֽהּ׃ {פ} אֲרֵי עַל פִּתְגָמָא דַּייָ בַּסַּר, וְיָת פִּקּוֹדוֹהִי אַשְׁנִי; אִשְׁתֵּיצָאָה יִשְׁתֵּיצֵי אֲנָשָׁא הַהוּא, חוֹבֵיהּ בֵּיהּ. {פ}
טו,לב וַיִּֽהְי֥וּ בְנֵֽי־יִשְׂרָאֵ֖ל בַּמִּדְבָּ֑ר וַֽיִּמְצְא֗וּ אִ֛ישׁ מְקֹשֵׁ֥שׁ עֵצִ֖ים בְּי֥וֹם הַשַּׁבָּֽת׃ וַהֲווֹ בְּנֵי יִשְׂרָאֵל, בְּמַדְבְּרָא; וְאַשְׁכַּחוּ, גְּבַר כַּד מְגָבֵיב אָעִין--בְּיוֹמָא דְּשַׁבְּתָא. 
טו,לג וַיַּקְרִ֣יבוּ אֹת֔וֹ הַמֹּֽצְאִ֥ים אֹת֖וֹ מְקֹשֵׁ֣שׁ עֵצִ֑ים אֶל־מֹשֶׁה֙ וְאֶֽל־אַהֲרֹ֔ן וְאֶ֖ל כָּל־הָֽעֵדָֽה׃ וְקָרִיבוּ יָתֵיהּ, דְּאַשְׁכַּחוּ יָתֵיהּ כַּד מְגָבֵיב אָעִין--לְוָת מֹשֶׁה, וּלְוָת אַהֲרוֹן, וּלְוָת, כָּל כְּנִשְׁתָּא. 
טו,לד וַיַּנִּ֥יחוּ אֹת֖וֹ בַּמִּשְׁמָ֑ר כִּ֚י לֹ֣א פֹרַ֔שׁ מַה־יֵּֽעָשֶׂ֖ה לֽוֹ׃ {ס} וַאֲסַרוּ יָתֵיהּ, בְּבֵית מַטְּרָא:  אֲרֵי לָא אִתְפָּרַשׁ לְהוֹן, מָא דְּיַעְבְּדוּן לֵיהּ. {ס}
טו,לה וַיֹּ֤אמֶר יְהוָה֙ אֶל־מֹשֶׁ֔ה מ֥וֹת יוּמַ֖ת הָאִ֑ישׁ רָג֨וֹם אֹת֤וֹ בָֽאֲבָנִים֙ כָּל־הָ֣עֵדָ֔ה מִח֖וּץ לַֽמַּחֲנֶֽה׃ וַאֲמַר יְיָ לְמֹשֶׁה, אִתְקְטָלָא יִתְקְטִיל גֻּבְרָא; רְגוּמוּ יָתֵיהּ בְּאַבְנַיָּא כָּל כְּנִשְׁתָּא, מִבַּרָא לְמַשְׁרִיתָא. 
טו,לו וַיֹּצִ֨יאוּ אֹת֜וֹ כָּל־הָֽעֵדָ֗ה אֶל־מִחוּץ֙ לַֽמַּחֲנֶ֔ה וַיִּרְגְּמ֥וּ אֹת֛וֹ בָּֽאֲבָנִ֖ים וַיָּמֹ֑ת כַּֽאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־מֹשֶֽׁה׃ {פ} וְאַפִּיקוּ יָתֵיהּ כָּל כְּנִשְׁתָּא, לְמִבַּרָא לְמַשְׁרִיתָא, וּרְגַמוּ יָתֵיהּ בְּאַבְנַיָּא, וּמִית:  כְּמָא דְּפַקֵּיד יְיָ, יָת מֹשֶׁה. {פ}
טו,לז וַיֹּ֥אמֶר יְהוָ֖ה אֶל־מֹשֶׁ֥ה לֵּאמֹֽר׃ וַאֲמַר יְיָ, לְמֹשֶׁה לְמֵימַר. 
טו,לח דַּבֵּ֞ר אֶל־בְּנֵ֤י יִשְׂרָאֵל֙ וְאָֽמַרְתָּ֣ אֲלֵהֶ֔ם וְעָשׂ֨וּ לָהֶ֥ם צִיצִ֛ת עַל־כַּנְפֵ֥י בִגְדֵיהֶ֖ם לְדֹֽרֹתָ֑ם וְנָ֥תְנ֛וּ עַל־צִיצִ֥ת הַכָּנָ֖ף פְּתִ֥יל תְּכֵֽלֶת׃ מַלֵּיל עִם בְּנֵי יִשְׂרָאֵל, וְתֵימַר לְהוֹן, וְיַעְבְּדוּן לְהוֹן כְּרֻסְפְּדִין עַל כַּנְפֵי כְּסוּתְהוֹן, לְדָרֵיהוֹן; וְיִתְּנוּן עַל כְּרֻסְפַּד כַּנְפָא, חוּטָא דִּתְכִילְתָא. 
טו,לט וְהָיָ֣ה לָכֶם֮ לְצִיצִת֒ וּרְאִיתֶ֣ם אֹת֗וֹ וּזְכַרְתֶּם֙ אֶת־כָּל־מִצְו֣‍ֹת יְהוָ֔ה וַֽעֲשִׂיתֶ֖ם אֹתָ֑ם וְלֹֽא־תָת֜וּרוּ אַֽחֲרֵ֤י לְבַבְכֶם֙ וְאַֽחֲרֵ֣י עֵֽינֵיכֶ֔ם אֲשֶׁר־אַתֶּ֥ם זֹנִ֖ים אַֽחֲרֵיהֶֽם׃ וִיהוֹן לְכוֹן, לִכְרֻסְפְּדִין, וְתִחְזוֹן יָתֵיהּ וְתִדְכְרוּן יָת כָּל פִּקּוֹדַיָּא דַּייָ, וְתַעְבְּדוּן יָתְהוֹן; וְלָא תִּטְעוֹן בָּתַר הַרְהוּר לִבְּכוֹן, וּבָתַר חֵיזוּ עֵינֵיכוֹן, דְּאַתּוּן טָעַן, בָּתְרֵיהוֹן. 
טו,מ לְמַ֣עַן תִּזְכְּר֔וּ וַֽעֲשִׂיתֶ֖ם אֶת־כָּל־מִצְו‍ֹתָ֑י וִֽהְיִיתֶ֥ם קְדֹשִׁ֖ים לֵֽאלֹהֵיכֶֽם׃ בְּדִיל דְּתִדְכְרוּן, וְתַעְבְּדוּן יָת כָּל פִּקּוֹדָי; וּתְהוֹן קַדִּישִׁין, קֳדָם אֱלָהֲכוֹן. 
טו,מא אֲנִ֞י יְהוָ֣ה אֱלֹֽהֵיכֶ֗ם אֲשֶׁ֨ר הוֹצֵ֤אתִי אֶתְכֶם֙ מֵאֶ֣רֶץ מִצְרַ֔יִם לִֽהְי֥וֹת לָכֶ֖ם לֵֽאלֹהִ֑ים אֲנִ֖י יְהוָ֥ה אֱלֹֽהֵיכֶֽם׃ {פ} אֲנָא יְיָ אֱלָהֲכוֹן, דְּאַפֵּיקִית יָתְכוֹן מֵאַרְעָא דְּמִצְרַיִם, לְמִהְוֵי לְכוֹן, לֶאֱלָהּ:  אֲנָא, יְיָ אֱלָהֲכוֹן. {פ}
